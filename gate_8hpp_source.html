156 template<typename LVT2_, typename MT2_>
157 void convert(const Gate<LVT2_, MT2_>& other)
158 {
159 if((void*)this == (void*)&other)
160 return;
161
162 this->_ranks.clear();
163 this->_mirrors.clear();
164
165 this->_comm = other._comm;
166 this->_ranks = other._ranks;
167
168 for(auto& other_mirrors_i : other._mirrors)
169 {
170 MirrorType mirrors_i;
171 mirrors_i.convert(other_mirrors_i);
172 this->_mirrors.push_back(std::move(mirrors_i));
173 }
174
175 this->_freqs.convert(other._freqs);
176 }

197 template<typename LVT2_>
198 void convert(const Gate<LVT2_, Mirror_>& other, LocalVector_&& vector, LAFEM::CloneMode mode = LAFEM::CloneMode::Shallow)
199 {
200 if((void*)this == (void*)&other)
201 return;
202
203 this->_ranks.clear();
204 this->_mirrors.clear();
205 this->_mirrors.resize(other._mirrors.size());
206 this->_freqs.clear(); // will be rebuild by compile function
207
208 this->_comm = other._comm;
209 this->_ranks = other._ranks;
210
211 // shallow-clone mirrors
212 for(std::size_t i(0); i < _mirrors.size(); ++i)
213 {
214 _mirrors.at(i).clone(other._mirrors.at(i), mode);
215 }
216
217 // compile this gate
218 compile(std::forward<LocalVector_>(vector));
219 }

266 void compile(LocalVector_&& vector)
267 {
268 // initialize frequency vector
269 _freqs = std::move(vector);
270 _freqs.format(DataType(1));
271
272 // loop over all mirrors
273 for(std::size_t i(0); i < _mirrors.size(); ++i)
274 {
275 // sum up number of ranks per frequency entry, listed in different mirrors
276 auto temp = _mirrors.at(i).create_buffer(_freqs);
277 temp.format(DataType(1));
278
279 // gather-axpy into frequency vector
280 _mirrors.at(i).scatter_axpy(_freqs, temp);
281 }
282
283 // invert frequencies
284 _freqs.component_invert(_freqs);
285 }

312 template<LAFEM::Perspective perspective_ = LAFEM::Perspective::native>
313 Index get_num_global_dofs() const
314 {
315 XASSERT(this->_comm != nullptr);
316 if(this->_comm->size() <= 1)
317 return this->template get_num_local_dofs<perspective_>();
318
319 // get my rank
320 const int my_rank = this->_comm->rank();
321
322 // get local number of DOFs
323 const Index loc_dofs = this->template get_num_local_dofs<perspective_>();
324
325 // create a local mask vector and format it to 1
326 std::vector<int> mask(std::size_t(loc_dofs), 1);
327
328 // set all DOFs, which are shared with a lower rank neighbor, to 0
329 for(std::size_t i(0); i < _mirrors.size(); ++i)
330 {
331 if(this->_ranks.at(i) < my_rank)
332 this->_mirrors.at(i).template mask_scatter<perspective_>(this->_freqs, mask, 0);
333 }
334
335 // count the number of DOFs that are still 1 and thus not owned by a lower rank neighbor
336 Index owned_dofs(0u);
337 for(const auto& k : mask)
338 owned_dofs += Index(k);
339
340 // now sum up the number of owned DOFs over all processes
341 Index global_dofs(0u);
342 this->_comm->allreduce(&owned_dofs, &global_dofs, std::size_t(1), Dist::op_sum);
343
344 // done!
345 return global_dofs;
346 }

476 DataType dot(const LocalVector_& x, const LocalVector_& y) const
477 {
478 // This is if there is only one process
479 if(_comm == nullptr || _comm->size() == 1)
480 {
481 return x.dot(y);
482 }
483 // Even if there are no neighbors, we still need to sum up globally
484 else if(_ranks.empty())
485 {
486 return sum(x.dot(y));
487 }
488 // If there are neighbors, we have to use the frequencies and sum up globally
489 else
490 {
491 return sum(_freqs.triple_dot(x, y));
