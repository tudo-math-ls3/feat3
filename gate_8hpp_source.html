192 template<typename LVT2_, typename MT2_>
193 void convert(const Gate<LVT2_, MT2_>& other)
194 {
195 if((void*)this == (void*)&other)
196 return;
197
198 this->_ranks.clear();
199 this->_mirrors.clear();
200
201 this->_comm = other._comm;
202 this->_ranks = other._ranks;
203
204 for(auto& other_mirrors_i : other._mirrors)
205 {
206 MirrorType mirrors_i;
207 mirrors_i.convert(other_mirrors_i);
208 this->_mirrors.push_back(std::move(mirrors_i));
209 }
210
211 this->_freqs.convert(other._freqs);
212 }

233 template<typename LVT2_>
234 void convert(const Gate<LVT2_, Mirror_>& other, LocalVector_&& vector, LAFEM::CloneMode mode = LAFEM::CloneMode::Shallow)
235 {
236 if((void*)this == (void*)&other)
237 return;
238
239 this->_ranks.clear();
240 this->_mirrors.clear();
241 this->_mirrors.resize(other._mirrors.size());
242 this->_freqs.clear(); // will be rebuild by compile function
243
244 this->_comm = other._comm;
245 this->_ranks = other._ranks;
246
247 // shallow-clone mirrors
248 for(std::size_t i(0); i < _mirrors.size(); ++i)
249 {
250 _mirrors.at(i).clone(other._mirrors.at(i), mode);
251 }
252
253 // compile this gate
254 compile(std::forward<LocalVector_>(vector));
255 }

302 void compile(LocalVector_&& vector)
303 {
304 // initialize frequency vector
305 _freqs = std::move(vector);
306 _freqs.format(DataType(1));
307
308 // loop over all mirrors
309 for(std::size_t i(0); i < _mirrors.size(); ++i)
310 {
311 // sum up number of ranks per frequency entry, listed in different mirrors
312 auto temp = _mirrors.at(i).create_buffer(_freqs);
313 temp.format(DataType(1));
314
315 // gather-axpy into frequency vector
316 _mirrors.at(i).scatter_axpy(_freqs, temp);
317 }
318
319 // invert frequencies
320 _freqs.component_invert(_freqs);
321 }

348 template<LAFEM::Perspective perspective_ = LAFEM::Perspective::native>
349 Index get_num_global_dofs() const
350 {
351 XASSERT(this->_comm != nullptr);
352 if(this->_comm->size() <= 1)
353 return this->template get_num_local_dofs<perspective_>();
354
355 // get my rank
356 const int my_rank = this->_comm->rank();
357
358 // get local number of DOFs
359 const Index loc_dofs = this->template get_num_local_dofs<perspective_>();
360
361 // create a local mask vector and format it to 1
362 std::vector<int> mask(std::size_t(loc_dofs), 1);
363
364 // set all DOFs, which are shared with a lower rank neighbor, to 0
365 for(std::size_t i(0); i < _mirrors.size(); ++i)
366 {
367 if(this->_ranks.at(i) < my_rank)
368 this->_mirrors.at(i).template mask_scatter<perspective_>(this->_freqs, mask, 0);
369 }
370
371 // count the number of DOFs that are still 1 and thus not owned by a lower rank neighbor
372 Index owned_dofs(0u);
373 for(const auto& k : mask)
374 owned_dofs += Index(k);
375
376 // now sum up the number of owned DOFs over all processes
377 Index global_dofs(0u);
378 this->_comm->allreduce(&owned_dofs, &global_dofs, std::size_t(1), Dist::op_sum);
379
380 // done!
381 return global_dofs;
382 }

512 DataType dot(const LocalVector_& x, const LocalVector_& y) const
513 {
514 // This is if there is only one process
515 if(_comm == nullptr || _comm->size() == 1)
516 {
517 return x.dot(y);
518 }
519 // Even if there are no neighbors, we still need to sum up globally
520 else if(_ranks.empty())
521 {
522 return sum(x.dot(y));
523 }
524 // If there are neighbors, we have to use the frequencies and sum up globally
525 else
526 {
527 return sum(_freqs.triple_dot(x, y));
528 }
529 }
