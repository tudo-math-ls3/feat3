79 template<typename Adjactor_>
80 static void write(std::ostream& os, const Adjactor_& adj)
81 {
82 typedef std::uint8_t u8;
83
84 // set up header
85 u8 header[18];
86 for(int i(0); i < 18; ++i)
87 header[i] = 0;
88
89 // get dimensions
90 const Index w = adj.get_num_nodes_image();
91 const Index h = adj.get_num_nodes_domain();
92
93 // make sure we do not exceed the 16-bit range
94 XASSERTM((w <= Index(32767)) && (h <= Index(32767)), "TGA dimensions are limited to 32767");
95
96 // set dimensions
97 header[12] = u8( w & 0xFF);
98 header[13] = u8((w >> 8) & 0x7F);
99 header[14] = u8( h & 0xFF);
100 header[15] = u8((h >> 8) & 0x7F);
101
102 // set basic stuff
103 header[ 2] = u8( 3); // datatype code
104 header[16] = u8( 8); // bits per pixel
105 header[17] = u8(32); // image descriptor
106
107 // write header to file
108 os.write(reinterpret_cast<char*>(header), 18);
109
110 // allocate a scan-line
111 std::vector<char> scan(std::size_t(w), -1);
112
113 // okay, let's loop over all domain nodes
114 for(Index i(0); i < h; ++i)
115 {
116 // get the image iterators
117 auto jt = adj.image_end(i);
118
119 // loop over all image nodes and set black pixels
120 for(auto it = adj.image_begin(i); it != jt; ++it)
121 {
122 // set pixel to black
123 scan[*it] = 0;
124 }
125
126 // write scan-line
127 os.write(scan.data(), std::streamsize(scan.size()));
128
129 // clear pixels
130 for(auto it = adj.image_begin(i); it != jt; ++it)
131 {
132 scan[*it] = -1;
133 }
134 }
135
136 // finally, write the TGA v2 footer
137 const char* footer = "\0\0\0\0\0\0\0\0TRUEVISION-XFILE.\0";
138 os.write(footer, 26);
139 }
