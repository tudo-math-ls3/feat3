93 template<typename Matrix_, typename Filter_>
94 class SuperLU :
95 public ADPSolverBase<Matrix_, Filter_>
96 {
97 public:
99 typedef ADPSolverBase<Matrix_, Filter_> BaseClass;
101 typedef typename Matrix_::VectorTypeL VectorType;
102
103 protected:
105 void* _core;
106
107 public:
111 explicit SuperLU(const Matrix_& matrix, const Filter_& filter) :
112 BaseClass(matrix, filter),
113 _core(nullptr)
114 {
115 }
116
117 virtual String name() const override
118 {
119 return "SuperLU";
120 }
121
122 virtual void init_symbolic() override
123 {
124 BaseClass::init_symbolic();
125 XASSERT(this->_core == nullptr);
126
127 // create our core wrapper object
128 this->_core = SuperLU_Aux::create_core(
129#ifdef FEAT_HAVE_MPI
130 &this->_get_comm()->mpi_comm(),
131#else
132 nullptr, // no communicator for non-MPI builds
133#endif
134 this->_get_num_global_dofs(),
135 this->_get_global_dof_offset(),
136 this->_get_num_owned_dofs(),
137 this->_get_mat_row_ptr(),
138 this->_get_mat_col_idx());
139 }
140
141 virtual void done_symbolic() override
142 {
143 XASSERT(this->_core != nullptr);
144
145 SuperLU_Aux::destroy_core(this->_core);
146 this->_core = nullptr;
147
148 BaseClass::done_symbolic();
149 }
150
151 virtual void init_numeric() override
152 {
153 XASSERT(this->_core != nullptr);
154 BaseClass::init_numeric();
155
156 // set the new matrix values
157 const int info = SuperLU_Aux::init_numeric(this->_core, this->_get_mat_vals());
158
159 // info = 0 means success
160 // info < 0 means illegal argument; this should never happen
161 XASSERTM(info >= 0, "SuperLU: illegal argument in init_numeric()");
162 if(0 < info)
163 {
164 // 0 < info < N: zero pivot = singular matrix
165 if(info <= int(this->_get_num_global_dofs()))
166 throw SingularMatrixException("SuperLU: singular matrix");
167 else // info > N: out of memory
168 throw SolverException("SuperLU: out of memory");
169 }
170 }
171
172 virtual Status apply(VectorType& vec_cor, const VectorType& vec_def) override
173 {
174 // upload defect vector into internal correction vector
175 this->_upload_vec_cor(vec_def);
176
177 // solve
178 int info = SuperLU_Aux::solve(this->_core, this->_get_vec_cor_vals(vec_cor));
179
180 // info != 0 means something went wrong, but this should never happen here...
181 XASSERTM(info == 0, "SuperLU: solve returned non-zero info value");
182
183 // download solution into correction vector
184 this->_download_vec_cor(vec_cor);
185
186 // okay
187 return Status::success;
188 }
189 }; // class SuperLU
