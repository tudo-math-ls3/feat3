352 void _build_inv_lin_trafo(const TrafoEvaluator& trafo_eval)
353 {
354 // create a domain point in the barycentre of the cell
355 DomainPointType dom_point(DataType(0));
356
357 // create the trafo data
358 InvLinTrafoData trafo_data;
359 trafo_eval(trafo_data, dom_point);
360
361 // store inverse trafo linearization
362 _inv_lin_mat = trafo_data.jac_inv;
363 _inv_lin_vec = trafo_data.img_point;
364 }
365
367 void _build_coeff_matrix(const TrafoEvaluator& trafo_eval)
368 {
369 // fetch the trafo
370 const TrafoType& trafo = trafo_eval.get_trafo();
371
372 // fetch the mesh
373 const MeshType& mesh = trafo.get_mesh();
374
375 // fetch the facet-index-set of the mesh
376 const FacetIndexSetType& facet_index_set = mesh.template get_index_set<2,1>();
377
378 // fetch the cell index of the currently active cell
379 const Index cell = trafo_eval.get_cell_index();
380
381 // create a nodal matrix
382 CoeffMatrixType _nodal_mat(DataType(0));
383
384 // create a facet trafo evaluator
385 FacetTrafoEvaluator facet_eval(trafo);
386
387 // create facet evaluation data
388 FacetTrafoData facet_data;
389
390 // define 2-point Gauss cubature point coordinate
391 const DataType g = Math::sqrt(DataType(1) / DataType(3));
392 const typename FacetEvalTraits::DomainPointType g1(-g), g2(+g);
393 DomainPointType q1, q2;
394
395 // loop over all 4 edges of the quad
396 for(int i(0); i < 4; ++i)
397 {
398 // initialize facet evaluator
399 facet_eval.prepare(Index(facet_index_set(cell, i)));
400
401 // map first cubature point
402 facet_eval(facet_data, g1);
403 DataType w1(facet_data.jac_det);
404 q1.set_mat_vec_mult(_inv_lin_mat, facet_data.img_point - _inv_lin_vec);
405
406 // map second cubature point
407 facet_eval(facet_data, g2);
408 DataType w2(facet_data.jac_det);
409 q2.set_mat_vec_mult(_inv_lin_mat, facet_data.img_point - _inv_lin_vec);
410
411 // release facet evaluator
412 facet_eval.finish();
413
414 // compute reciprocal edge length
415 DataType v = DataType(1) / (w1 + w2);
416
417 // evaluate node functionals
418 _nodal_mat(0,i) = DataType(1); // = v*(w1 + w2)
419 _nodal_mat(1,i) = v*(w1*q1[0] + w2*q2[0]);
420 _nodal_mat(2,i) = v*(w1*q1[1] + w2*q2[1]);
421 _nodal_mat(3,i) = v*(w1*((q1[0]+q1[1])*(q1[0]-q1[1])) + w2*((q2[0]+q2[1])*(q2[0]-q2[1])));
422 }
423
424 // invert nodal matrix to obtain coefficient matrix
425 _coeff_mat.set_inverse(_nodal_mat);
426 }

471 template<SpaceTags space_cfg_, TrafoTags trafo_cfg_>
472 void eval_values(
473 EvalData<SpaceEvalTraits, space_cfg_>& data,
474 const Trafo::EvalData<TrafoEvalTraits, trafo_cfg_>& trafo_data) const
475 {
476 // transform image point
477 DomainPointType pt;
478 pt.set_mat_vec_mult(_inv_lin_mat, trafo_data.img_point - _inv_lin_vec);
479
480 // "evaluate" monomials
481 DataType x(pt[0]);
482 DataType y(pt[1]);
483 DataType r((x+y)*(x-y));
484
485 // loop over all basis functions
486 for(int i(0); i < 4; ++i)
487 {
488 data.phi[i].value = _coeff_mat(i,0) + _coeff_mat(i,1)*x + _coeff_mat(i,2)*y + _coeff_mat(i,3)*r;
489 }
490 }

493 template<SpaceTags space_cfg_, TrafoTags trafo_cfg_>
494 void eval_gradients(
495 EvalData<SpaceEvalTraits, space_cfg_>& data,
496 const Trafo::EvalData<TrafoEvalTraits, trafo_cfg_>& trafo_data) const
497 {
498 // transform image point to local coordinate system
499 DomainPointType pt, loc_grad;
500 pt.set_mat_vec_mult(_inv_lin_mat, trafo_data.img_point - _inv_lin_vec);
501
502 // loop over all basis functions
503 for(int i(0); i < 4; ++i)
504 {
505 // calculate local gradients
506 loc_grad(0) = _coeff_mat(i,1) + DataType(2) * _coeff_mat(i,3) * pt[0];
507 loc_grad(1) = _coeff_mat(i,2) - DataType(2) * _coeff_mat(i,3) * pt[1];
508
509 // multiply by transpose inverse linearized trafo matrix for chain rule
510 data.phi[i].grad.set_vec_mat_mult(loc_grad, _inv_lin_mat);
511 }
512 }

632 void _build_coeff_matrix(const TrafoEvaluator& trafo_eval)
633 {
634 // fetch the trafo
635 const TrafoType& trafo = trafo_eval.get_trafo();
636
637 // fetch the mesh
638 const MeshType& mesh = trafo.get_mesh();
639
640 // fetch the facet-index-set of the mesh
641 const FacetIndexSetType& facet_index_set = mesh.template get_index_set<3,2>();
642
643 // fetch the cell index of the currently active cell
644 const Index cell = trafo_eval.get_cell_index();
645
646 // create a nodal matrix
647 CoeffMatrixType _nodal_mat(DataType(0));
648
649 // create a facet trafo evaluator
650 FacetTrafoEvaluator facet_eval(trafo);
651
652 // create facet evaluation data
653 FacetTrafoData facet_data;
654
655 // define 2-point Gauss cubature point coordinate
656 const DataType g = Math::sqrt(DataType(1) / DataType(3));
657 typename FacetEvalTraits::DomainPointType g1, g2, g3, g4;
658 DomainPointType q1, q2, q3, q4;
659
660 // set up cubature points
661 g1[0] = g2[0] = -g;
662 g3[0] = g4[0] = +g;
663 g1[1] = g3[1] = -g;
664 g2[1] = g4[1] = +g;
665
666 // loop over all 6 faces of the hexa
667 for(int i(0); i < 6; ++i)
668 {
669 // initialize facet evaluator
670 facet_eval.prepare(Index(facet_index_set(cell, i)));
671
672 // map first cubature point
673 facet_eval(facet_data, g1);
674 DataType w1(facet_data.jac_det);
675 q1.set_mat_vec_mult(_inv_lin_mat, facet_data.img_point - _inv_lin_vec);
676
677 // map second cubature point
678 facet_eval(facet_data, g2);
679 DataType w2(facet_data.jac_det);
680 q2.set_mat_vec_mult(_inv_lin_mat, facet_data.img_point - _inv_lin_vec);
681
682 // map third cubature point
683 facet_eval(facet_data, g3);
684 DataType w3(facet_data.jac_det);
685 q3.set_mat_vec_mult(_inv_lin_mat, facet_data.img_point - _inv_lin_vec);
686
687 // map fourth cubature point
688 facet_eval(facet_data, g4);
689 DataType w4(facet_data.jac_det);
690 q4.set_mat_vec_mult(_inv_lin_mat, facet_data.img_point - _inv_lin_vec);
691
692 // release facet evaluator
693 facet_eval.finish();
694
695 // compute reciprocal quad area
696 DataType v = DataType(1) / (w1 + w2 + w3 + w4);
697
698 // evaluate node functionals
699 _nodal_mat(0,i) = DataType(1); // = v*(w1 + w2 + w3 + w4)
700 _nodal_mat(1,i) = v*(w1*q1[0] + w2*q2[0] + w3*q3[0] + w4*q4[0]);
701 _nodal_mat(2,i) = v*(w1*q1[1] + w2*q2[1] + w3*q3[1] + w4*q4[1]);
702 _nodal_mat(3,i) = v*(w1*q1[2] + w2*q2[2] + w3*q3[2] + w4*q4[2]);
703 _nodal_mat(4,i) = v*( // x^2 - y^2
704 w1*(q1[0] + q1[1])*(q1[0] - q1[1]) + w2*(q2[0] + q2[1])*(q2[0] - q2[1]) +
705 w3*(q3[0] + q3[1])*(q3[0] - q3[1]) + w4*(q4[0] + q4[1])*(q4[0] - q4[1]));
706 _nodal_mat(5,i) = v*( // y^2 - z^2
707 w1*(q1[1] + q1[2])*(q1[1] - q1[2]) + w2*(q2[1] + q2[2])*(q2[1] - q2[2]) +
708 w3*(q3[1] + q3[2])*(q3[1] - q3[2]) + w4*(q4[1] + q4[2])*(q4[1] - q4[2]));
709 }
710
711 // invert nodal matrix to obtain coefficient matrix
712 _coeff_mat.set_inverse(_nodal_mat);
713 }

758 template<SpaceTags space_cfg_, TrafoTags trafo_cfg_>
759 void eval_values(
760 EvalData<SpaceEvalTraits, space_cfg_>& data,
761 const Trafo::EvalData<TrafoEvalTraits, trafo_cfg_>& trafo_data) const
762 {
763 // transform image point
764 DomainPointType pt;
765 pt.set_mat_vec_mult(_inv_lin_mat, trafo_data.img_point - _inv_lin_vec);
766
767 // "evaluate" monomials
768 DataType x(pt[0]);
769 DataType y(pt[1]);
770 DataType z(pt[2]);
771 DataType rxy((x+y)*(x-y)); // = x^2 - y^2
772 DataType ryz((y+z)*(y-z)); // = y^2 - z^2
773
774 // loop over all basis functions
775 for(int i(0); i < 6; ++i)
776 {
777 data.phi[i].value = _coeff_mat(i,0) + _coeff_mat(i,1)*x + _coeff_mat(i,2)*y + _coeff_mat(i,3)*z
778 + _coeff_mat(i,4)*rxy + _coeff_mat(i,5)*ryz;
779 }

783 template<SpaceTags space_cfg_, TrafoTags trafo_cfg_>
784 void eval_gradients(
785 EvalData<SpaceEvalTraits, space_cfg_>& data,
786 const Trafo::EvalData<TrafoEvalTraits, trafo_cfg_>& trafo_data) const
787 {
788 // transform image point to local coordinate system
789 DomainPointType pt, loc_grad;
790 pt.set_mat_vec_mult(_inv_lin_mat, trafo_data.img_point - _inv_lin_vec);
791
792 // loop over all basis functions
793 for(int i(0); i < 6; ++i)
794 {
795 // calculate local gradients
796 loc_grad[0] = _coeff_mat(i,1) + DataType(2) * pt[0] * _coeff_mat(i,4);
797 loc_grad[1] = _coeff_mat(i,2) + DataType(2) * pt[1] * (_coeff_mat(i,5) - _coeff_mat(i,4));
798 loc_grad[2] = _coeff_mat(i,3) - DataType(2) * pt[2] * _coeff_mat(i,5);
799
800 // multiply by transpose inverse linearized trafo matrix for chain rule
801 data.phi[i].grad.set_vec_mat_mult(loc_grad, _inv_lin_mat);
802 }
