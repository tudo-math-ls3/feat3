337 void read_from(FileMode mode, std::istream& file)
338 {
339 this->clear();
340
341 switch(mode)
342 {
343 case FileMode::fm_mtx:
344 {
345
346 Index trows, tcols;
347 String line;
348 std::getline(file, line); // !!? Test on overflow error... could be an enormous matrix... !??
349 //for now, just array real general (aka dense) matrices
350 const bool array_format((line.find("%%MatrixMarket matrix array real general") != String::npos) ? true : false);
351 if (array_format == false)
352 {
353 XABORTM("Input-file is not a compatible array real mtx-file");
354 }
355
356 while(!file.eof())
357 {
358 std::getline(file,line);
359 if (file.eof())
360 XABORTM("Input-file is empty");
361
362 String::size_type begin(line.find_first_not_of(" "));
363 if (line.at(begin) != '%')
364 break;
365 }
366 //Read in number of rows and columns
367 {
368 String::size_type begin(line.find_first_not_of(" "));
369 line.erase(0, begin);
370 String::size_type end(line.find_first_of(" "));
371 String srow(line, 0, end);
372 trows = Index(atol(srow.c_str()));
373 line.erase(0, end);
374
375 begin = line.find_first_not_of(" ");
376 line.erase(0, begin);
377 end = line.find_first_of(" ");
378 String scol(line, 0, end);
379 tcols = Index(atol(scol.c_str()));
380 line.erase(0, end);
381 }
382
383 DenseMatrix<DT_, IT_> result(Index(trows), tcols);
384 Index i(0);
385
386 //Read in value of lines:
387 while(!file.eof())
388 {
389 std::getline(file, line);
390 if(file.eof())
391 break;
392
393 String::size_type begin = line.find_first_not_of(" ");
394 line.erase(0, begin);
395 String::size_type end = line.find_first_of(" ");
396 String sval(line, 0, end);
397 DT_ tval((DT_)atof(sval.c_str()));
398
399 Index row(i / tcols);
400 Index col(i % tcols);
401 result(row, col, tval);
402
403 ++i;
404 }
405 XASSERTM(i == trows * tcols, "Dense MTX file did not contain enough entries!");
406
407 this->assign(result);
408 break;
409
410 }
411 case FileMode::fm_dm:
412 case FileMode::fm_binary:
413 this->template _deserialize<double, std::uint64_t>(FileMode::fm_dm, file);
414 break;
415 default:
416 XABORTM("Filemode not supported!");
417 }
418 }

540 {
541 XASSERTM(x.rows() == this->rows(), "Row count does not match!");
542 XASSERTM(x.columns() == this->columns(), "Column count does not match!");
543 XASSERTM(x.used_elements() == this->used_elements(), "Nonzero count does not match!");

609 void apply(DenseVector<DT_, IT_> & r, const DenseVector<DT_, IT_> & x) const
610 {
611 XASSERTM(r.size() == this->rows(), "Vector size of r does not match!");
612 XASSERTM(x.size() == this->columns(), "Vector size of x does not match!");
613
614 XASSERTM(r.template elements<Perspective::pod>() != x.template elements<Perspective::pod>(), "Vector x and r must not share the same memory!");
615
616 TimeStamp ts_start;
617
618 Statistics::add_flops(this->used_elements() * 2);
619 Arch::Apply::dense(r.elements(), DT_(1), DT_(0), r.elements(), this->elements(),
620 x.elements(), this->rows(), this->columns());
621
622 TimeStamp ts_stop;
623 Statistics::add_time_blas2(ts_stop.elapsed(ts_start));
624 }

634 void apply(
635 DenseVector<DT_, IT_> & r,
636 const DenseVector<DT_, IT_> & x,
637 const DenseVector<DT_, IT_> & y,
638 const DT_ alpha = DT_(1)) const
639 {
640 XASSERTM(r.size() == this->rows(), "Vector size of r does not match!");
641 XASSERTM(x.size() == this->columns(), "Vector size of x does not match!");
642 XASSERTM(y.size() == this->rows(), "Vector size of y does not match!");
643
644 XASSERTM(r.template elements<Perspective::pod>() != x.template elements<Perspective::pod>(), "Vector x and r must not share the same memory!");
645
646 TimeStamp ts_start;
647
648 if(Math::abs(alpha) < Math::eps<DT_>())
649 {
650 r.copy(y);
651 return;
652 }
653
654 Statistics::add_flops( (this->used_elements() + this->rows()) * 2 );
655 Arch::Apply::dense(r.elements(), alpha, DT_(1), y.elements(), this->elements(),
656 x.elements(), this->rows(), this->columns());
657
658 TimeStamp ts_stop;
659 Statistics::add_time_blas2(ts_stop.elapsed(ts_start));
660 }

761 void invert()
762 {
763 XASSERTM(this->rows() == this->columns(), "matrix must be square!");
764
765 TimeStamp ts_start;
766 Statistics::add_flops(this->used_elements() * this->columns()*2);
767
768 IT_ * temp = new IT_[this->rows()];
769 Math::invert_matrix((IT_)this->rows(), (IT_)this->rows(), this->elements(), temp);
770 delete[] temp;
771
772 TimeStamp ts_stop;
773 Statistics::add_time_blas3(ts_stop.elapsed(ts_start));

866 friend bool operator== (const DenseMatrix & a, const DenseMatrix<DT_, IT_> & b)
867 {
868 if (a.size() != b.size())
869 return false;
870 if (a.get_elements().size() != b.get_elements().size())
871 return false;
872 if (a.get_indices().size() != b.get_indices().size())
873 return false;
874 if (a.rows() != b.rows())
875 return false;
876 if (a.columns() != b.columns())
877 return false;
878
879 if(a.size() == 0 && b.size() == 0 && a.get_elements().size() == 0 && b.get_elements().size() == 0)
880 return true;
881
882 bool ret(true);
883
884 DT_ * ta;
885 DT_ * tb;
886
887 ta = const_cast<DT_*>(a.elements());
888 tb = const_cast<DT_*>(b.elements());
889
890 for (Index i(0) ; i < a.size() ; ++i)
891 {
892 if (ta[i] != tb[i])
893 {
894 ret = false;
895 break;
896 }
897 }
898
899 return ret;
900 }
