337 void read_from(FileMode mode, std::istream& file)
338 {
339 this->clear();
340
341 switch(mode)
342 {
343 case FileMode::fm_mtx:
344 {
345
346 Index trows, tcols;
347 String line;
348 std::getline(file, line); // !!? Test on overflow error... could be an enormous matrix... !??
349 //for now, just array real general (aka dense) matrices
350 const bool array_format((line.find("%%MatrixMarket matrix array real general") != String::npos) ? true : false);
351 if (array_format == false)
352 {
353 XABORTM("Input-file is not a compatible array real mtx-file");
354 }
355
356 while(!file.eof())
357 {
358 std::getline(file,line);
359 if (file.eof())
360 XABORTM("Input-file is empty");
361
362 String::size_type begin(line.find_first_not_of(" "));
363 if (line.at(begin) != '%')
364 break;
365 }
366 //Read in number of rows and columns
367 {
368 String::size_type begin(line.find_first_not_of(" "));
369 line.erase(0, begin);
370 String::size_type end(line.find_first_of(" "));
371 String srow(line, 0, end);
372 trows = Index(atol(srow.c_str()));
373 line.erase(0, end);
374
375 begin = line.find_first_not_of(" ");
376 line.erase(0, begin);
377 end = line.find_first_of(" ");
378 String scol(line, 0, end);
379 tcols = Index(atol(scol.c_str()));
380 line.erase(0, end);
381 }
382
383 DenseMatrix<DT_, IT_> result(Index(trows), tcols);
384 Index i(0);
385
386 //Read in value of lines:
387 while(!file.eof())
388 {
389 std::getline(file, line);
390 if(file.eof())
391 break;
392
393 String::size_type begin = line.find_first_not_of(" ");
394 line.erase(0, begin);
395 String::size_type end = line.find_first_of(" ");
396 String sval(line, 0, end);
397 DT_ tval((DT_)atof(sval.c_str()));
398
399 Index row(i / tcols);
400 Index col(i % tcols);
401 result(row, col, tval);
402
403 ++i;
404 }
405 XASSERTM(i == trows * tcols, "Dense MTX file did not contain enough entries!");
406
407 this->move(std::move(result));
408 break;
409
410 }
411 case FileMode::fm_dm:
412 case FileMode::fm_binary:
413 this->template _deserialize<double, std::uint64_t>(FileMode::fm_dm, file);
414 break;
415 default:
416 XABORTM("Filemode not supported!");
417 }
418 }

656 void apply(
657 DenseVector<DT_, IT_> & r,
658 const DenseVector<DT_, IT_> & x,
659 const DenseVector<DT_, IT_> & y,
660 const DT_ alpha = DT_(1)) const
661 {
662 XASSERTM(r.size() == this->rows(), "Vector size of r does not match!");
663 XASSERTM(x.size() == this->columns(), "Vector size of x does not match!");
664 XASSERTM(y.size() == this->rows(), "Vector size of y does not match!");
665
666 XASSERTM(r.template elements<Perspective::pod>() != x.template elements<Perspective::pod>(), "Vector x and r must not share the same memory!");
667
668 TimeStamp ts_start;
669
670 if(Math::abs(alpha) < Math::eps<DT_>())
671 {
672 r.copy(y);
673 return;
674 }
675
676 Statistics::add_flops( (this->used_elements() + this->rows()) * 2 );
677 Arch::Apply::dense(r.elements(), alpha, DT_(1), y.elements(), this->elements(),
678 x.elements(), this->rows(), this->columns());
679
680 TimeStamp ts_stop;
681 Statistics::add_time_blas2(ts_stop.elapsed(ts_start));
682 }

924 friend bool operator== (const DenseMatrix & a, const DenseMatrix<DT_, IT_> & b)
925 {
926 if (a.size() != b.size())
927 return false;
928 if (a.get_elements().size() != b.get_elements().size())
929 return false;
930 if (a.get_indices().size() != b.get_indices().size())
931 return false;
932 if (a.rows() != b.rows())
933 return false;
934 if (a.columns() != b.columns())
935 return false;
936
937 if(a.size() == 0 && b.size() == 0 && a.get_elements().size() == 0 && b.get_elements().size() == 0)
938 return true;
939
940 bool ret(true);
941
942 DT_ * ta;
943 DT_ * tb;
944
945 ta = const_cast<DT_*>(a.elements());
946 tb = const_cast<DT_*>(b.elements());
947
948 for (Index i(0) ; i < a.size() ; ++i)
949 {
950 if (ta[i] != tb[i])
951 {
952 ret = false;
953 break;
954 }
955 }
956
957 return ret;
958 }
