168 protected:
169 virtual Status _apply_intern(VectorType& vec_sol)
170 {
171 Statistics::add_solver_expression(std::make_shared<ExpressionStartSolve>(this->name()));
172 const MatrixType& matrix(this->_system_matrix);
173 const FilterType& filter(this->_system_filter);
174 VectorType& vec_p(this->_vec_p);
175 VectorType& vec_q(this->_vec_q);
176 VectorType& vec_r(this->_vec_r);
177 VectorType& vec_s(this->_vec_s);
178 // Note: y and z are temporary vectors whose usage does
179 // not overlap, so we use the same vector for them
180 VectorType& vec_y(this->_vec_t);
181 VectorType& vec_z(this->_vec_t);
182
183 // Note:
184 // In the algorithm below, the following relations hold:
185 // q[k] = A * p[k]
186 // s[k] = M^{-1} * r[k]
187 // y[k] = A * s[k]
188 // z[k] = M^{-1} * q[k]
189
190 // set initial defect:
191 // r[0] := b - A*x[0]
192 Status status = this->_set_initial_defect(vec_r, vec_sol);
193 if(status != Status::progress)
194 {
195 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), status, this->get_num_iter()));
196 return status;
197 }
198
199 // apply preconditioner to defect vector:
200 // s[0] := M^{-1} * r[0]
201 if(!this->_apply_precond(vec_s, vec_r, filter))
202 {
203 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
204 return Status::aborted;
205 }
206
207 // compute initial p and q
208 // p[0] := s[0]
209 vec_p.copy(vec_s);
210
211 // q[0] := A*p[0]
212 matrix.apply(vec_q, vec_p);
213 filter.filter_def(vec_q);
214
215 // compute initial gamma:
216 // gamma[0] := < p[0], q[0] >
217 DataType gamma = vec_p.dot(vec_q);
218
219 // start iterating
220 while(status == Status::progress)
221 {
222 IterationStats stat(*this);
223
224 // z[k] := M^{-1} * q[k]
225 if(!this->_apply_precond(vec_z, vec_q, filter))
226 {
227 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
228 return Status::aborted;
229 }
230
231 // compute alpha
232 // alpha[k] := gamma[k] / < z[k], q[k] >
233 DataType alpha = gamma / vec_z.dot(vec_q);
234
235 // update solution vector
236 // x[k+1] := x[k] + alpha[k] * p[k]
237 vec_sol.axpy(vec_p, vec_sol, alpha);
238
239 // update real residual vector
240 // r[k+1] := r[k] - alpha[k] * q[k]
241 vec_r.axpy(vec_q, vec_r, -alpha);
242
243 // compute defect norm and check for convergence
244 status = this->_set_new_defect(vec_r, vec_sol);
245 if(status != Status::progress)
246 {
247 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), status, this->get_num_iter()));
248 return status;
249 }
250
251 // update preconditioned residual vector
252 // s[k+1] := s[k] - alpha[k] * z[k]
253 vec_s.axpy(vec_z, vec_s, -alpha);
254
255 // y[k+1] := A*s[k+1]
256 matrix.apply(vec_y, vec_s);
257 filter.filter_def(vec_y);
258
259 // compute new gamma:
260 // gamma[k+1] := < s[k+1], y[k+1] >
261 DataType gamma2 = gamma;
262 gamma = vec_s.dot(vec_y);
263
264 // compute beta:
265 // beta[k] := gamma[k+1] / gamma[k]
266 DataType beta = gamma / gamma2;
267
268 // update direction vectors:
269 // p[k+1] := s[k+1] + beta[k] * p[k]
270 vec_p.axpy(vec_p, vec_s, beta);
271
272 // q[k+1] := y[k+1] + beta[k] * q[k]
273 vec_q.axpy(vec_q, vec_y, beta);
274 }
275
276 // we should never reach this point...
277 {
278 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::undefined, this->get_num_iter()));
279 return Status::undefined;
280 }
281 }
