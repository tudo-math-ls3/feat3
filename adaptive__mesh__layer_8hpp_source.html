103 Index get_num_vertices() const
104 {
105 return _mesh->get_num_entities(_layer, 0);
106 }
107
115 VertexType& operator[](Index idx)
116 {
117 if(_perm)
118 {
119 return _mesh->vertex(_layer, _perm.value().map(idx));

126 {
127 if(_perm)

133
134 // NOTE: Unsupported on purpose. Transform underlying mesh if required.
135 //void transform(const VertexType& origin, const VertexType& angles, const VertexType& offset)
136
149 void permute(const Adjacency::Permutation& perm, bool invert = false)
150 {
151 if(!_perm)
152 {
153 _perm = std::optional{invert ? perm.inverse() : perm.clone()};
154 }
155 else
156 {
157 if(invert)
158 {
159 _perm.value().concat(perm.inverse());
160 }
161 else
162 {
163 _perm.value().concat(perm);
164 }
165 }
166 }

214 AdaptiveIndexTuple(std::shared_ptr<AdaptiveMeshType> mesh, Layer layer, Index entity_idx) :

469 void permute(const Adjacency::Permutation& perm, const Adjacency::Permutation& inv_perm_face)
470 {
471 // Permute index tuples

473 {

483 {
484 _inverse_face_permutation.value().concat(inv_perm_face);
485 }
486 else
487 {
488 _inverse_face_permutation = inv_perm_face.clone();
489 }

1314 bool validate_element_coloring() const
1315 {
1316 // no coloring?
1317 const std::vector<Index>& coloring = this->get_mesh_permutation().get_element_coloring();
1318 if(coloring.empty())
1319 return true;
1320
1321 // get vertices-at-element index set
1322 const auto& verts_at_elem = this->template get_index_set<shape_dim, 0>();
1323
1324 // render transpose
1325 Adjacency::Graph elems_at_vert(Adjacency::RenderType::transpose, verts_at_elem);
1326
1327 // loop over all color blocks
1328 for(std::size_t icol(0); icol+1u < coloring.size(); ++icol)
1329 {
1330 // get the bounds of our current color block
1331 const Index iel_beg = coloring[icol];
1332 const Index iel_end = coloring[icol+1u];
1333
1334 // loop over all elements in the current color block
1335 for(Index iel(iel_beg); iel < iel_end; ++iel)
1336 {
1337 // loop over all vertices adjacent to this element
1338 for(int ivt(0); ivt < verts_at_elem.num_indices; ++ivt)
1339 {
1340 // loop over all elements adjacent to this vertex
1341 const Index ivtx = verts_at_elem(iel, ivt);
1342 for(auto it = elems_at_vert.image_begin(ivtx); it != elems_at_vert.image_end(ivtx); ++it)
1343 {
1344 // two adjacent element must not be in the same color block
1345 if((iel_beg <= *it) && (*it < iel_end) && (*it != iel))
1346 return false; // invalid coloring
1347 }
1348 }
1349 }
1350 } // next color block
1351
1352 // ok, coloring is valid
1353 return true;
1354 }

1364 bool validate_element_layering() const
1365 {
1366 // no layering?
1367 const std::vector<Index>& layering = this->get_mesh_permutation().get_element_layering();
1368 if(layering.empty())
1369 return true;
1370
1371 // get vertices-at-element index set
1372 const auto& verts_at_elem = this->template get_index_set<shape_dim, 0>();
1373
1374 // render transpose
1375 Adjacency::Graph elems_at_vert(Adjacency::RenderType::transpose, verts_at_elem);
1376
1377 // loop over all layers
1378 for(std::size_t ilay(0); ilay+1u < layering.size(); ++ilay)
1379 {
1380 // get the bounds of our current layer
1381 const Index iel_beg = layering[ilay];
1382 const Index iel_end = layering[ilay+1u];
1383
1384 // get the lower bound for valid neighbors of our current layer = beginning of previous layer
1385 const Index iel_lower = layering[Math::max(ilay, std::size_t(1)) - 1u];
1386
1387 // get the upper bound for valid neighbors of our current layer = end of next layer
1388 const Index iel_upper = layering[Math::min(ilay+2u, layering.size()-1u)];
1389
1390 // loop over all elements in the current layer
1391 for(Index iel(iel_beg); iel < iel_end; ++iel)
1392 {
1393 // loop over all vertices adjacent to this element
1394 for(int ivt(0); ivt < verts_at_elem.num_indices; ++ivt)
1395 {
1396 // loop over all elements adjacent to this vertex
1397 const Index ivtx = verts_at_elem(iel, ivt);
1398 for(auto it = elems_at_vert.image_begin(ivtx); it != elems_at_vert.image_end(ivtx); ++it)
1399 {
1400 // adjacent element outside of adjacent layers?
1401 if(!((iel_lower <= *it) && (*it < iel_upper)))
1402 return false; // invalid layer
1403 }
1404 }
1405 }
1406 } // next color block
1407
1408 // ok, layering is valid
