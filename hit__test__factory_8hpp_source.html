118 template<
119 typename HitFunc_,
120 typename Mesh_,
121 typename Shape_>
122 struct HitTestCompute
123 {
124 typedef std::vector<std::vector<Index>> TargetData;
125 static void wrap(TargetData& target_data, const Mesh_& mesh, const HitFunc_& hit_test)
126 {
127 typedef typename Shape::FaceTraits<Shape_, Shape_::dimension-1>::ShapeType FacetType;
128 HitTestCompute<HitFunc_, Mesh_, FacetType>::wrap(target_data, mesh, hit_test);
129 apply(target_data, mesh, hit_test);
130 }
131
132 static void apply(TargetData& trg, const Mesh_& mesh, const HitFunc_& hit_test)
133 {
134 static constexpr int shape_dim(Shape_::dimension);
135 const Index num_cells(mesh.get_num_entities(shape_dim));
136 const auto& index_set(mesh.template get_index_set<shape_dim,0>());
137 trg[shape_dim].reserve(num_cells);
138 for(Index i(0); i < num_cells; ++i)
139 {
140 if (hit_test(get_midpoint(mesh.get_vertex_set(), index_set[i])))
141 {
142 trg[shape_dim].push_back(i);
143 }
144 }
145 }
146
147 template<
148 typename VertexSet_,
149 typename IndexTuple_>
150 static Tiny::Vector<
151 typename VertexSet_::CoordType,
152 VertexSet_::num_coords>
153 get_midpoint(
154 const VertexSet_& vertex_set,
155 const IndexTuple_& index_tuple)
156 {
157 typedef typename VertexSet_::CoordType DataType;
158 int num_vertex(Shape::FaceTraits<Shape_,0>::count);
159 Tiny::Vector<DataType, VertexSet_::num_coords> mid_point;
160 mid_point.format();
161 for(int i(0); i < num_vertex; ++i)
162 {
163 for(int j(0); j < VertexSet_::num_coords; ++j)
164 {
165 mid_point[j] += vertex_set[index_tuple[i]][j];
166 }
167 }
168 return mid_point * (DataType(1) / DataType(num_vertex));
169 }
170 };

196 template<typename VertexSet_>
197 static Tiny::Vector<
198 typename VertexSet_::CoordType,
199 VertexSet_::num_coords>
200 get_midpoint(const VertexSet_& vertex_set, const Index vertex)
201 {
202 typedef typename VertexSet_::CoordType DataType;
203 Tiny::Vector<DataType, VertexSet_::num_coords> mid_point;
204 for(int j(0); j < VertexSet_::num_coords; ++j)
205 {
206 mid_point[j] = vertex_set[vertex][j];
207 }
208 return mid_point;
209 }

268 template<typename DataType_, int dim_>

271 public:
273 typedef Tiny::Vector<DataType_, dim_> PointType;
274
275 private:
277 PointType _midpoint;
279 DataType_ _radius;

281 public:
288 explicit SphereHitTestFunction(DataType_ radius) :

293 }
294
304 SphereHitTestFunction(PointType midpoint, DataType_ radius) :
305 _midpoint(midpoint),
306 _radius(radius)
307 {
308 XASSERTM(radius > DataType_(0), "sphere radius must be > 0");
309 }
310
