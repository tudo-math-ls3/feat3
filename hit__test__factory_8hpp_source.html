154 template<
155 typename HitFunc_,
156 typename Mesh_,
157 typename Shape_>
158 struct HitTestCompute
159 {
160 typedef std::vector<std::vector<Index>> TargetData;
161 static void wrap(TargetData& target_data, const Mesh_& mesh, const HitFunc_& hit_test)
162 {
163 typedef typename Shape::FaceTraits<Shape_, Shape_::dimension-1>::ShapeType FacetType;
164 HitTestCompute<HitFunc_, Mesh_, FacetType>::wrap(target_data, mesh, hit_test);
165 apply(target_data, mesh, hit_test);
166 }
167
168 static void apply(TargetData& trg, const Mesh_& mesh, const HitFunc_& hit_test)
169 {
170 static constexpr int shape_dim(Shape_::dimension);
171 const Index num_cells(mesh.get_num_entities(shape_dim));
172 const auto& index_set(mesh.template get_index_set<shape_dim,0>());
173 trg[shape_dim].reserve(num_cells);
174 for(Index i(0); i < num_cells; ++i)
175 {
176 if (hit_test(get_midpoint(mesh.get_vertex_set(), index_set[i])))
177 {
178 trg[shape_dim].push_back(i);
179 }
180 }
181 }
182
183 template<
184 typename VertexSet_,
185 typename IndexTuple_>
186 static Tiny::Vector<
187 typename VertexSet_::CoordType,
188 VertexSet_::num_coords>
189 get_midpoint(
190 const VertexSet_& vertex_set,
191 const IndexTuple_& index_tuple)
192 {
193 typedef typename VertexSet_::CoordType DataType;
194 int num_vertex(Shape::FaceTraits<Shape_,0>::count);
195 Tiny::Vector<DataType, VertexSet_::num_coords> mid_point;
196 mid_point.format();
197 for(int i(0); i < num_vertex; ++i)
198 {
199 for(int j(0); j < VertexSet_::num_coords; ++j)
200 {
201 mid_point[j] += vertex_set[index_tuple[i]][j];
202 }
203 }
204 return mid_point * (DataType(1) / DataType(num_vertex));
205 }
206 };

219 static void apply(TargetData& trg, const Mesh_& mesh, const HitFunc_& hit_test)

221 const Index num_cells(mesh.get_num_entities(0));
222 trg[0].reserve(num_cells);
223 for(Index i(0); i < num_cells; ++i)
224 {
225 if (hit_test(get_midpoint(mesh.get_vertex_set(), i)))
226 {
227 trg[0].push_back(i);
228 }
229 }

232 template<typename VertexSet_>
233 static Tiny::Vector<
234 typename VertexSet_::CoordType,
235 VertexSet_::num_coords>
236 get_midpoint(const VertexSet_& vertex_set, const Index vertex)
237 {
238 typedef typename VertexSet_::CoordType DataType;
239 Tiny::Vector<DataType, VertexSet_::num_coords> mid_point;
240 for(int j(0); j < VertexSet_::num_coords; ++j)
241 {
242 mid_point[j] = vertex_set[vertex][j];
243 }
244 return mid_point;
245 }

304 template<typename DataType_, int dim_>
305 class SphereHitTestFunction
306 {
307 public:
309 typedef Tiny::Vector<DataType_, dim_> PointType;
310
311 private:
313 PointType _midpoint;
315 DataType_ _radius;
316
317 public:
324 explicit SphereHitTestFunction(DataType_ radius) :
325 _midpoint(DataType_(0)),
326 _radius(radius)
327 {
328 XASSERTM(radius > DataType_(0), "sphere radius must be > 0");
329 }
330
340 SphereHitTestFunction(PointType midpoint, DataType_ radius) :
341 _midpoint(midpoint),
342 _radius(radius)
343 {
344 XASSERTM(radius > DataType_(0), "sphere radius must be > 0");
345 }
346
356 bool operator()(PointType point) const
357 {
358 return (point - _midpoint).norm_euclid() <= _radius;
359 }
