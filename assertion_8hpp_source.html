133
134 // write basic information
135 fprintf(stderr, "Expression: %s\n", expr_str);
136 fprintf(stderr, "Function..: %s\n", func);
137 fprintf(stderr, "File......: %s\n", file);
138 fprintf(stderr, "Line......: %i\n", line);
139
140 // flush stderr
141 fflush(stderr);
142
143 // abort execution;
144 // this may also write a call-stack dump if possible
145 Runtime::abort();
146 #else
147 // cuda does only know printf
148 if(msg != nullptr)
149 printf("\n>>> FATAL ERROR: ASSERTION FAILED: %s\n", msg);

155 printf("Function..: %s\n", func);
156 printf("File......: %s\n", file);

158
