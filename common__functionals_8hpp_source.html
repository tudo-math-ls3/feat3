202 template<typename T_, int m_, int n_, int sm_, int sn_>
203 void calc_trace(const Tiny::Matrix<T_, m_, n_, sm_, sn_>& hess)

208 template<typename T_, int l_, int m_, int n_, int sl_, int sm_, int sn_>
209 void calc_trace(const Tiny::Tensor3<T_, l_, m_, n_, sl_, sm_, sn_>& hess)
210 {
211 _func_laplace.format();
212 for(int i(0); i < hess.l; ++i)
213 for(int j(0); j < hess.m; ++j)
214 _func_laplace[i] -= hess[i][j][j];
215 }
