111 explicit Circle(CoordType mid_x, CoordType mid_y, CoordType radius,
112 CoordType param_l, CoordType param_r) :
113 _radius(radius),
114 _have_domain(true),
115 _trafo_a(-param_l),
116 _trafo_b((CoordType(2) * Math::pi<CoordType>()) / (param_r - param_l))
117 {
118 XASSERTM(radius > CoordType(0), "invalid circle radius");

127 virtual bool can_explicit() const override
128 {
129 return _have_domain;
130 }

133 virtual void transform(const WorldPoint& origin, const WorldPoint& angles, const WorldPoint& offset) override
134 {
135 // create rotation matrix
136 Tiny::Matrix<CoordType, 2, 2> rot;
137 rot.set_rotation_2d(angles(0));
138
139 // transform midpoint
140 WorldPoint tmp;
141 tmp = _midpoint - origin;
142 _midpoint.set_mat_vec_mult(rot, tmp) += offset;
143
144 // transform domain
145 if(_have_domain)
146 _trafo_a += angles(0);
147 }

156 void project_point(WorldPoint& point) const
157 {
158 WorldPoint grad_dist(point - _midpoint);
159 CoordType distance(grad_dist.norm_euclid());
160
161 if(distance < Math::eps<CoordType>())
162 {
163 grad_dist(0) = _radius;
164 point += grad_dist;
165 }
166 else
167 {
168 point = _midpoint + (_radius / distance)*grad_dist;
169 }
170 }
171
173 CoordType compute_dist(const WorldPoint& point) const
174 {
175 return Math::abs(compute_signed_dist(point));
176 }
177
179 CoordType compute_dist(const WorldPoint& point, WorldPoint& grad_dist) const
180 {
181 WorldPoint projected(point);
182 project_point(projected);
183
184 grad_dist = (projected - point);
185 CoordType my_dist(grad_dist.norm_euclid());
186 grad_dist.normalize();
187
188 return my_dist;
189 }
190
192 CoordType compute_signed_dist(const WorldPoint& point) const
193 {
194 return _radius - (point - _midpoint).norm_euclid();
195 }
196
198 CoordType compute_signed_dist(const WorldPoint& point, WorldPoint& grad_dist) const
199 {
200 WorldPoint projected(point);
201 project_point(projected);
202
203 CoordType my_dist(_radius - (point - _midpoint).norm_euclid());
204
205 if(my_dist < Math::eps<CoordType>())
206 {
207 grad_dist = (_midpoint - point);
208 }
209 else
210 {
211 grad_dist = (point - projected)*Math::signum(my_dist);
212 }
213
214 grad_dist.normalize();
215
216 return my_dist;
217 }

220 void project_meshpart(Mesh_& mesh, const MeshPart<Mesh_>& meshpart) const
221 {
222 auto& vtx = mesh.get_vertex_set();
223 const auto& target_vtx = meshpart.template get_target_set<0>();
224
225 for(Index i(0); i < meshpart.get_num_entities(0); ++i)
226 {
227 project_point(reinterpret_cast<WorldPoint&>(vtx[target_vtx[i]]));
228 }
229 }
230
241 void map_param(WorldPoint& point, const ParamPoint& param) const
242 {
243 // transform parameter to interval [0, 2*pi)
244 CoordType x = (param[0] + _trafo_a) * _trafo_b;
245 point[0] = this->_midpoint[0] + this->_radius * Math::cos(x);
246 point[1] = this->_midpoint[1] + this->_radius * Math::sin(x);
247 }

256 virtual void write(std::ostream& os, const String& sindent) const override
257 {
258 os << sindent << "<Circle";
259 os << " radius=\"" << this->_radius << "\"";
260 os << " midpoint=\"" << this->_midpoint[0] << " " << this->_midpoint[1] << "\"";
261 // reconstruct domain from trafo
262 if(_have_domain)
263 {
264 CoordType param_l(-_trafo_a);
265 CoordType param_r(param_l + CoordType(2) * Math::pi<CoordType>() / _trafo_b);
266 os << " domain=\"" << param_l << " " << param_r << "\"";
267 }
268 os << " />" << std::endl;
269 }

287 virtual bool attribs(std::map<String,bool>& attrs) const override

289 attrs.emplace("radius", true);

294
295 virtual void create(
296 int iline,
297 const String& sline,
298 const String&,
299 const std::map<String, String>& attrs,
300 bool) override
301 {
302 CoordType radius = CoordType(0);
303 CoordType mid_x = CoordType(0);
304 CoordType mid_y = CoordType(0);
305 CoordType dom_0 = CoordType(0);
306 CoordType dom_1 = CoordType(1);
307 bool have_domain(false);
308
309 // try to parse the radius
310 if(!attrs.find("radius")->second.parse(radius))
311 throw Xml::GrammarError(iline, sline, "Failed to parse circle radius");
312 if(radius < CoordType(1E-5))
313 throw Xml::GrammarError(iline, sline, "Invalid circle radius");
314
315 // try to parse midpoint
316 std::deque<String> mids = attrs.find("midpoint")->second.split_by_whitespaces();
317 if(mids.size() != std::size_t(2))
318 throw Xml::GrammarError(iline, sline, "Invalid circle midpoint string");
319 if(!mids.front().parse(mid_x) || !mids.back().parse(mid_y))
320 throw Xml::GrammarError(iline, sline, "'Failed to parse circle midpoint");
321
322 // do we have a domain?
323 auto it = attrs.find("domain");
324 if(it != attrs.end())
325 {
326 have_domain = true;
327 std::deque<String> doms = it->second.split_by_whitespaces();
328 if(doms.size() != std::size_t(2))
329 throw Xml::GrammarError(iline, sline, "Invalid circle domain string");
330 if(!doms.front().parse(dom_0) || !doms.back().parse(dom_1))
331 throw Xml::GrammarError(iline, sline, "'Failed to parse circle domain");
332 }
333
334 // everything seems fine, let's create the chart then
335 if(have_domain)
336 _chart.reset(new ChartType(mid_x, mid_y, radius, dom_0, dom_1));
337 else
338 _chart.reset(new ChartType(mid_x, mid_y, radius));
339 }
