153 explicit DuDvFunctional(
154 Geometry::RootMeshNode<MeshType>* rmn_,
155 TrafoType& trafo,
156 const std::deque<String>& dirichlet_list, const std::deque<String>& slip_list):
157 BaseClass(rmn_),
158 matrix_sys(),
159 _trafo(trafo),
160 _lambda(rmn_->get_mesh()->get_num_entities(MeshType::shape_dim)),
161 _dirichlet_asm(),
162 _slip_asm(),
163 _cubature_factory("auto-degree:"+stringify(int(_local_degree))),
164 trafo_space(trafo)
165 {
166 // The symbolic assembly has to happen in the constructor because the matrix is shallow copied immediately
167 // after construction in the Control::QuadraticSystemLevel
168 Assembly::SymbolicAssembler::assemble_matrix_std1(matrix_sys, trafo_space);
169
170 // For every MeshPart specified in the list of Dirichlet boundaries, create a UnitFilterAssembler and
171 // insert it into the std::map
172 for(auto& it : dirichlet_list)
173 {
174 // Create empty assembler
175 auto new_asm = std::make_shared<Assembly::UnitFilterAssembler<MeshType>>();
176
177 // Add the MeshPart to the assembler if it is there. There are legimate reasons for it NOT to be
178 // there, i.e. we are in parallel and our patch is not adjacent to that MeshPart
179 auto* mpp = rmn_->find_mesh_part(it);
180 if(mpp != nullptr)
181 {
182 new_asm->add_mesh_part(*mpp);
183 }
184
185 // Insert into the map
186 String identifier(it);
187 _dirichlet_asm.emplace(identifier, new_asm);
188 }
189
190 // For every MeshPart specified in the list of slip boundaries, create a SlipFilterAssembler and
191 // insert it into the std::map
192 for(auto& it : slip_list)
193 {
194 // Create empty assembler
195 auto new_asm = std::make_shared<Assembly::SlipFilterAssembler<TrafoType>>(this->_trafo);
196
197 // Add the MeshPart to the assembler if it is there. There are legimate reasons for it NOT to be
198 // there, i.e. we are in parallel and our patch is not adjacent to that MeshPart
199 auto* mpp = rmn_->find_mesh_part(it);
200 if(mpp != nullptr)
201 {
202 new_asm->add_mesh_part(*mpp);
203 }
204
205 // Insert into the map
206 String identifier(it);
207 _slip_asm.emplace(identifier, new_asm);
208 }

280 virtual void prepare(VectorTypeR& vec_state, FilterType& filter)
281 {
282 for(Index cell(0); cell < this->get_mesh()->get_num_entities(MeshType::shape_dim); ++cell)
283 {
284 _lambda(cell, DataType(_trafo.template compute_vol<typename MeshType::ShapeType>(cell)));
285 }
286
287 auto& dirichlet_filters = filter.template at<1>();
288
289 for(auto& it : dirichlet_filters)
290 {
291 const auto& assembler = _dirichlet_asm.find(it.first);
292 if(assembler == _dirichlet_asm.end())
293 {
294 XABORTM("Could not find dirichlet assembler for filter with key "+it.first);
295 }
296
297 assembler->second->assemble(it.second, trafo_space, vec_state);
298 }
299
300 // The slip filter contains the outer unit normal, so reassemble it
301 auto& slip_filters = filter.template at<0>();
302
303 for(auto& it : slip_filters)
304 {
305 const auto& assembler = _slip_asm.find(it.first);
306 if(assembler == _slip_asm.end())
307 {
308 XABORTM("Could not find slip filter assembler for filter with key "+it.first);
309 }
310
311 assembler->second->assemble(it.second, trafo_space);
312 }
313
314 for(const auto& it:slip_filters)
315 {
316 this->_mesh_node->adapt_by_name(it.first);
317 }

357 virtual CoordType compute_cell_size_defect(CoordType& lambda_min, CoordType& lambda_max, CoordType& vol_min, CoordType& vol_max, CoordType& vol) const override
358 {
359
360 compute_cell_size_defect_pre_sync(vol_min, vol_max, vol);
361 return compute_cell_size_defect_post_sync(lambda_min, lambda_max, vol_min, vol_max, vol);

364
386 void compute_cell_size_defect_pre_sync(CoordType& vol_min, CoordType& vol_max, CoordType& vol) const
387 {
388 vol = CoordType(0);
389
390 vol_min = Math::huge<CoordType>();
391 vol_max = CoordType(0);
392
393 for(Index cell(0); cell < this->get_mesh()->get_num_entities(ShapeType::dimension); ++cell)
394 {
395 CoordType my_vol = this->_trafo.template compute_vol<ShapeType, CoordType>(cell);
396 vol_min = Math::min(vol_min, my_vol);
397 vol_max = Math::max(vol_min, my_vol);
398 vol += my_vol;
399 }
400 }
401
429 virtual CoordType compute_cell_size_defect_post_sync(CoordType& lambda_min, CoordType& lambda_max, CoordType& vol_min, CoordType& vol_max, const CoordType& vol) const
430 {
431 CoordType size_defect(0);
432 lambda_min = Math::huge<CoordType>();
433 lambda_max = CoordType(0);
434
435 for(Index cell(0); cell < this->get_mesh()->get_num_entities(ShapeType::dimension); ++cell)
436 {
437 size_defect += Math::abs(this->_trafo.template compute_vol<ShapeType, CoordType>(cell)/vol - this->_lambda(cell));
438 lambda_min = Math::min(lambda_min, CoordType(this->_lambda(cell)));
439 lambda_max = Math::max(lambda_max, CoordType(this->_lambda(cell)));
440 }
441
442 vol_min /= vol;
443 vol_max/= vol;
444
445 return size_defect;
446 }
