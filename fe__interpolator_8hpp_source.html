86 template<typename Vector_>
87 static void recurse(Vector_& to_coeffs, const Vector_& from_coeffs, const ToSpace& to_space)
88 {
89 typedef typename Vector_::DataType DataType;
90
91 const auto& mesh = to_space.get_trafo().get_mesh();
92
93 // We use this to identify the vertex DoF at our shape
94 const auto& from_vertex_at_shape_idx = mesh.template get_index_set<shape_dim_, 0>();
95
96 // This is for mapping to the correct Lagrange2 DoF
97 typename ToSpace::template DofAssignment<shape_dim_, DataType>::Type to_dof_assignment(to_space);
98
99 for(Index k(0); k < mesh.get_num_entities(shape_dim_); ++k)
100 {
101 to_dof_assignment.prepare(k);
102
103 // There is only one Lagrange2 DoF at each entity, but this is easier to understand for future reference
104 for(int j(0); j < to_dof_assignment.get_num_assigned_dofs(); ++j)
105 {
106
107 typename Vector_::ValueType tmp(0);
108
109 for(int i(0); i < from_vertex_at_shape_idx.get_num_indices(); ++i)
110 {
111 tmp += from_coeffs(from_vertex_at_shape_idx(k,i));
112 }
113
114 // Linear interpolation
115 tmp *= ( DataType(1)/DataType(from_vertex_at_shape_idx.get_num_indices()) );
116
117 // Emplacement operator
118 to_coeffs(to_dof_assignment.get_index(j), tmp);
119 }
120 }
121
122 // recurse down
123 Lagrange1To2DofAtEntity<Trafo_, shape_dim_-1>::recurse( to_coeffs, from_coeffs, to_space);
124 }
125 };
