222 //template<typename MatrixA_, typename MatrixB_, typename MatrixD_>
223 //static bool deduct_macro_dofs(const LAFEM::SaddlePointMatrix<MatrixA_, MatrixB_, MatrixD_>& matrix,
224 template<typename DT_, typename IT_, int bs_>
225 static bool deduct_macro_dofs(
226 const LAFEM::SaddlePointMatrix<
227 LAFEM::SparseMatrixBCSR<DT_, IT_, bs_, bs_>,
228 LAFEM::SparseMatrixBCSR<DT_, IT_, bs_, 1>,
229 LAFEM::SparseMatrixBCSR<DT_, IT_, 1, bs_>>& matrix,
230 std::vector<Adjacency::Graph>& macro_dofs)
231 {
232 typedef IT_ IndexType;
233
234 // fetch matrix dimensions
235 const Index num_dofs_p = Index(matrix.block_d().rows());
236
237 // fetch matrix array
238 const IndexType* row_ptr_b = matrix.block_b().row_ptr();
239 const IndexType* col_idx_b = matrix.block_b().col_ind();
240 const IndexType* row_ptr_d = matrix.block_d().row_ptr();
241 const IndexType* col_idx_d = matrix.block_d().col_ind();
242
243 // PHASE 1: determine pressure macros
244 std::map<IndexType, int> map_s;
245 std::vector<int> mask(std::size_t(num_dofs_p), 0);
246 std::vector<Index> p_ptr, p_idx;
247 p_ptr.push_back(Index(0));
248
249 // loop over all pressure nodes
250 for(Index i(0); i < num_dofs_p; ++i)
251 {
252 // is this node already processed?
253 if(mask[i] != 0)
254 continue;
255
256 // clear the local map
257 map_s.clear();
258
259 // okay, loop over all entries of D
260 for(IndexType kd = row_ptr_d[i]; kd < row_ptr_d[i+1]; ++kd)
261 {
262 // fetch the column index of D
263 const IndexType col_d = col_idx_d[kd];
264
265 // loop over the row of B
266 for(IndexType kb = row_ptr_b[col_d]; kb < row_ptr_b[col_d+1]; ++kb)
267 {
268 // fetch the column index of B
269 const IndexType col_b = col_idx_b[kb];
270
271 // insert into map
272 auto ib = map_s.emplace(col_b, 1);
273 if(!ib.second)
274 {
275 // node already exists, increment counter then
276 ++(ib.first->second);
277 }
278 }
279 }
280
281 // compute the map degree
282 int ideg = 0;
283 for(auto it = map_s.begin(); it != map_s.end(); ++it)
284 ideg = Math::max(ideg, it->second);
285
286 // loop over all map entries with maximum degree
287 for(auto it = map_s.begin(); it != map_s.end(); ++it)
288 {
289 if(ideg == it->second)
290 {
291 // insert into map
292 p_idx.push_back(Index(it->first));
293 mask[it->first] = 1;
294 }
295 }
296
297 // push row
298 p_ptr.push_back(Index(p_idx.size()));
299 }
300
301 // PHASE 2: determine velocity macros
302
303 // fetch number of pressure macros
304 const Index num_macros = Index(p_ptr.size()-1u);
305
306 // clear block arrays
307 std::set<IndexType> set_v;
308 std::vector<Index> v_ptr, v_idx;
309 v_ptr.reserve(num_macros+1u);
310 v_ptr.push_back(Index(0));
311
312 // loop over all pressure blocks
313 for(IndexType i(0); i < num_macros; ++i)
314 {
315 // loop over all pressure dofs in the current block
316 for(Index j = p_ptr[i]; j < p_ptr[i+1]; ++j)
317 {
318 // get the pressure dof index
319 const Index pix = p_idx[j];
320
321 // now loop over the corresponding row of D
322 for(IndexType k = row_ptr_d[pix]; k < row_ptr_d[pix+1]; ++k)
323 {
324 // insert velocity dof into block set
325 set_v.insert(col_idx_d[k]);
326 }
327 }
328
329 // push velocity block
330 for(auto it = set_v.begin(); it != set_v.end(); ++it)
331 {
332 v_idx.push_back(Index(*it));
333 }
334
335 // update pointer
336 v_ptr.push_back(Index(v_idx.size()));
337
338 // clear local set
339 set_v.clear();
340 }
341
342 // create graphs
343 macro_dofs.resize(std::size_t(2));
344 macro_dofs.front() = Adjacency::Graph(num_macros, matrix.block_b().rows(), Index(v_idx.size()), v_ptr.data(), v_idx.data());
345 macro_dofs.back() = Adjacency::Graph(num_macros, matrix.block_d().rows(), Index(p_idx.size()), p_ptr.data(), p_idx.data());
346 return true;
347 }

398 template<typename DT_, typename IT_, int bh_, int bw_>
399 static std::pair<Index,Index> gather(const LAFEM::NullMatrix<DT_, IT_, bh_, bw_>&,
400 DT_*, const Index, const Index macro, const std::vector<Adjacency::Graph>& macro_dofs,
401 const Index, const Index row_block, const Index, const Index col_block)
402 {
403 // get graph domain pointer arrays
404 const Index* row_dom_ptr = macro_dofs.at(row_block).get_domain_ptr();
405 const Index* col_dom_ptr = macro_dofs.at(col_block).get_domain_ptr();
406
407 // get the number of rows/cols to gather
408 const Index num_rows = row_dom_ptr[macro+1] - row_dom_ptr[macro];
409 const Index num_cols = col_dom_ptr[macro+1] - col_dom_ptr[macro];
410
411 // return the number of gathered rows
412 return std::make_pair(num_rows * Index(bh_), num_cols * Index(bw_));
413 }

416 template<typename DT_, typename IT_>
417 static std::pair<Index,Index> gather(const LAFEM::SparseMatrixCSR<DT_, IT_>& matrix,
418 DT_* local, const Index stride, const Index macro, const std::vector<Adjacency::Graph>& macro_dofs,
419 const Index row_off, const Index row_block, const Index col_off, const Index col_block)
420 {
421 // get matrix arrays
422 const DT_* vals = matrix.val();
423 const IT_* row_ptr = matrix.row_ptr();
424 const IT_* col_idx = matrix.col_ind();
425
426 // get the dofs for our row/col blocks
427 const Adjacency::Graph& row_dofs = macro_dofs.at(row_block);
428 const Adjacency::Graph& col_dofs = macro_dofs.at(col_block);
429
430 // get graph domain pointer arrays
431 const Index* row_dom_ptr = row_dofs.get_domain_ptr();
432 const Index* col_dom_ptr = col_dofs.get_domain_ptr();
433
434 // get the number of rows/cols to gather
435 const Index num_rows = row_dom_ptr[macro+1] - row_dom_ptr[macro];
436 const Index num_cols = col_dom_ptr[macro+1] - col_dom_ptr[macro];
437
438 // get graph image index arrays for our macro
439 const Index* row_img_idx = &(row_dofs.get_image_idx()[row_dom_ptr[macro]]);
440 const Index* col_img_idx = &(col_dofs.get_image_idx()[col_dom_ptr[macro]]);
441
442 // loop over all rows
443 for(Index i(0); i < num_rows; ++i)
444 {
445 // get the index of this row in the big matrix
446 const Index irow = row_img_idx[i];
447
448 // initialize loop variable for local columns
449 Index j = Index(0);
450
451 // row over the row
452 for(IT_ ai = row_ptr[irow]; ai < row_ptr[irow + Index(1)]; ++ai)
453 {
454 // get the column index
455 const Index icol = Index(col_idx[ai]);
456
457 // now search its position in our local matrix
458 while((j < num_cols) && (col_img_idx[j] < icol))
459 {
460 ++j;
461 }
462
463 // did we find our local entry?
464 if((j < num_cols) && (col_img_idx[j] == icol))
465 {
466 // copy macro row
467 local[(row_off + i) * stride + col_off + j] = vals[ai];
468 }
469 }
470 }
471
472 // return the number of gathered rows
473 return std::make_pair(num_rows, num_cols);
474 }

477 template<typename DT_, typename IT_, int bh_, int bw_>
478 static std::pair<Index,Index> gather(const LAFEM::SparseMatrixBCSR<DT_, IT_, bh_, bw_>& matrix,
479 DT_* local, const Index stride, const Index macro, const std::vector<Adjacency::Graph>& macro_dofs,
480 const Index row_off, const Index row_block, const Index col_off, const Index col_block)
481 {
482 // get matrix arrays
483 const Tiny::Matrix<DT_, bh_, bw_>* vals = matrix.val();
484 const IT_* row_ptr = matrix.row_ptr();
485 const IT_* col_idx = matrix.col_ind();
486
487 // get the dofs for our row/col blocks
488 const Adjacency::Graph& row_dofs = macro_dofs.at(row_block);
489 const Adjacency::Graph& col_dofs = macro_dofs.at(col_block);
490
491 // get graph domain pointer arrays
492 const Index* row_dom_ptr = row_dofs.get_domain_ptr();
493 const Index* col_dom_ptr = col_dofs.get_domain_ptr();
494
495 // get the number of rows/cols to gather
496 const Index num_rows = row_dom_ptr[macro+1] - row_dom_ptr[macro];
497 const Index num_cols = col_dom_ptr[macro+1] - col_dom_ptr[macro];
498
499 // get graph image index arrays for our macro
500 const Index* row_img_idx = &(row_dofs.get_image_idx()[row_dom_ptr[macro]]);
501 const Index* col_img_idx = &(col_dofs.get_image_idx()[col_dom_ptr[macro]]);
502
503 // loop over all rows
504 for(Index i(0); i < num_rows; ++i)
505 {
506 // get the index of this row in the big matrix
507 const Index irow = row_img_idx[i];
508
509 // initialize loop variable for local columns
510 Index j = Index(0);
511
512 // row over the row
513 for(IT_ ai = row_ptr[irow]; ai < row_ptr[irow + Index(1)]; ++ai)
514 {
515 // get the column index
516 const Index icol = Index(col_idx[ai]);
517
518 // now search its position in our local matrix
519 while((j < num_cols) && (col_img_idx[j] < icol))
520 {
521 ++j;
522 }
523
524 // did we find our local entry?
525 if((j < num_cols) && (col_img_idx[j] == icol))
526 {
527 // found it, so store it in our local matrix
528 const Tiny::Matrix<DT_, bh_, bw_>& mv = vals[ai];
529
530 // copy matrix macro
531 for(int ii(0); ii < bh_; ++ii)
532 {
533 // compute offset within local matrix
534 const Index lro = (row_off + i * Index(bh_) + Index(ii)) * stride + col_off + j * Index(bw_);
535
536 // copy macro row
537 for(int jj(0); jj < bw_; ++jj)
538 {
539 local[lro + IT_(jj)] = mv[ii][jj];
540 }
541 }
542 }
543 }
544 }
545
546 // return the number of gathered rows
547 return std::make_pair(num_rows * Index(bh_), num_cols * Index(bw_));
548 }

552 static std::pair<Index,Index> gather(const LAFEM::TupleMatrixRow<First_, Second_, Rest_...>& matrix,

606 static std::pair<Index,Index> gather(const LAFEM::SaddlePointMatrix<MatrixA_, MatrixB_, MatrixD_>& matrix,
607 DT_* local, const Index stride, const Index macro, const std::vector<Adjacency::Graph>& macro_dofs,

620 macro, macro_dofs, row_off + nrc_a.first, row_block + Index(1), col_off, col_block);
621
622 XASSERTM(nrc_a.first == nrc_b.first, "block row count mismatch");
623 XASSERTM(nrc_a.second == nrc_d.second, "block column count mismatch");
624
625 return std::make_pair(nrc_a.first + nrc_d.first, nrc_a.second + nrc_b.second);

776 template<typename DT_, typename IT_>
777 static std::pair<Index,Index> scatter_add(LAFEM::SparseMatrixCSR<DT_, IT_>& matrix,
778 const DT_* local, const Index stride, const Index macro, const std::vector<Adjacency::Graph>& macro_dofs,
779 const Index row_off, const Index row_block, const Index col_off, const Index col_block)
780 {
781 // get matrix arrays
782 DT_* vals = matrix.val();
783 const IT_* row_ptr = matrix.row_ptr();
784 const IT_* col_idx = matrix.col_ind();
785
786 // get the dofs for our row/col blocks
787 const Adjacency::Graph& row_dofs = macro_dofs.at(row_block);
788 const Adjacency::Graph& col_dofs = macro_dofs.at(col_block);
789
790 // get graph domain pointer arrays
791 const Index* row_dom_ptr = row_dofs.get_domain_ptr();
792 const Index* col_dom_ptr = col_dofs.get_domain_ptr();
793
794 // get the number of rows/cols to gather
795 const Index num_rows = row_dom_ptr[macro+1] - row_dom_ptr[macro];
796 const Index num_cols = col_dom_ptr[macro+1] - col_dom_ptr[macro];
797
798 // get graph image index arrays for our macro
799 const Index* row_img_idx = &(row_dofs.get_image_idx()[row_dom_ptr[macro]]);
800 const Index* col_img_idx = &(col_dofs.get_image_idx()[col_dom_ptr[macro]]);
801
802 // loop over all rows
803 for(Index i(0); i < num_rows; ++i)
804 {
805 // get the index of this row in the big matrix
806 const Index irow = row_img_idx[i];
807
808 // initialize loop variable for local columns
809 Index j = Index(0);
810
811 // row over the row
812 for(IT_ ai = row_ptr[irow]; ai < row_ptr[irow + Index(1)]; ++ai)
813 {
814 // get the column index
815 const Index icol = Index(col_idx[ai]);
816
817 // now search its position in our local matrix
818 while((j < num_cols) && (col_img_idx[j] < icol))
819 {
820 ++j;
821 }
822
823 // did we find our local entry?
824 if((j < num_cols) && (col_img_idx[j] == icol))
825 {
826 vals[ai] += local[(row_off + i) * stride + col_off + j];
827 }
828 }
829 }
830
831 // return the number of scattered rows
832 return std::make_pair(num_rows, num_cols);
833 }

993 template<typename DT_, typename IT_>
994 static void scale_rows(LAFEM::SparseMatrixCSR<DT_, IT_>& matrix, const DT_ omega,
995 const std::vector<Adjacency::Graph>& dof_macros, const std::vector<int>& macro_mask, const Index row_block)
996 {
997 // get matrix arrays
998 DT_* vals = matrix.val();
999 const IT_* row_ptr = matrix.row_ptr();
1000
1001 // get the dofs for our row blocks
1002 const Adjacency::Graph& row_blocks = dof_macros.at(row_block);
1003 const Index num_rows = row_blocks.get_num_nodes_domain();
1004 XASSERT(matrix.rows() == num_rows);
1005
1006 // get graph domain pointer arrays
1007 const Index* row_dom_ptr = row_blocks.get_domain_ptr();
1008 const Index* row_img_idx = row_blocks.get_image_idx();
1009
1010 // loop over all rows/dofs
1011 for(Index i(0); i < num_rows; ++i)
1012 {
1013 XASSERT(row_dom_ptr[i] < row_dom_ptr[i+1]);
1014
1015 // get number of active macros for this DOF
1016 Index n(0);
1017 if(macro_mask.empty())
1018 n = row_dom_ptr[i+1] - row_dom_ptr[i];
1019 else
1020 {
1021 for(Index j(row_dom_ptr[i]); j < row_dom_ptr[i+1]; ++j)
1022 n += Index(macro_mask[row_img_idx[j]]);
1023 }
1024
1025 const DT_ sc = omega / DT_(Math::max(n,Index(1)));
1026 for(IT_ j(row_ptr[i]); j < row_ptr[i+1]; ++j)
1027 vals[j] *= sc;
1028 }
1029 }

1118 template<typename Matrix_>
1119 static Index calc_stride(const Matrix_& matrix, const std::vector<Adjacency::Graph>& macro_dofs)
1120 {
1121 const std::size_t num_graphs = macro_dofs.size();
1122 const Index num_macros = macro_dofs.front().get_num_nodes_domain();
1123
1124 // compute macro sizes
1125 std::vector<Index> block_sizes;
1126 AmaVankaCore::block_sizes(matrix, block_sizes);
1127
1128 XASSERT(num_graphs == block_sizes.size());
1129
1130 // loop over all macros
1131 Index rtn = Index(0);
1132 for(Index imacro(0); imacro < num_macros; ++imacro)
1133 {
1134 Index count = Index(0);
1135
1136 // loop over all graphs and summarize degrees
1137 for(std::size_t igraph(0); igraph < num_graphs; ++igraph)
1138 {
1139 count += macro_dofs.at(igraph).degree(imacro) * block_sizes.at(igraph);
1140 }
1141
1142 rtn = Math::max(rtn, count);
1143 }
1144
1145 return rtn;
1146 }

1465 virtual void init_numeric() override
1466 {
1467 const DataType eps = Math::eps<DataType>();
1468
1469 watch_init_numeric.start();
1470 BaseClass::init_numeric();
1471
1472 // get maximum macro size
1473 const Index num_macros = Index(this->_macro_dofs.front().get_num_nodes_domain());
1474 const Index stride = Intern::AmaVankaCore::calc_stride(this->_vanka, this->_macro_dofs);
1475
1476 // allocate arrays for local matrix
1477 std::vector<DataType> vec_local(stride*stride, DataType(0)), vec_local_t(stride*stride, DataType(0));
1478 std::vector<Index> vec_pivot(stride);
1479 DataType* local = vec_local.data();
1480 DataType* local_t = vec_local_t.data();
1481 Index* pivot = vec_pivot.data();
1482
1483 this->_vanka.format();
1484
1485 // loop over all macros
1486 for(Index imacro(0); imacro < num_macros; ++imacro)
1487 {
1488 // gather local matrix
1489 const std::pair<Index,Index> nrc = Intern::AmaVankaCore::gather(this->_matrix,
1490 local, stride, imacro, this->_macro_dofs, Index(0), Index(0), Index(0), Index(0));
1491
1492 // make sure we have gathered a square matrix
1493 XASSERTM(nrc.first == nrc.second, "local matrix is not square");
1494
1495 // do we check for singular macros?
1496 if(this->_skip_singular)
1497 {
1498 // the approach used for checking the regularity of the local matrix is to check whether
1499 //
1500 // || I - A*A^{-1} ||_F^2 < eps
1501 //
1502 // we could try to analyse the pivots returned by invert_matrix function instead, but
1503 // unfortunately this approach sometimes leads to false positives
1504
1505 // make a backup if checking for singularity
1506 for(Index i(0); i < nrc.first; ++i)
1507 for(Index j(0); j < nrc.second; ++j)
1508 local_t[i*stride+j] = local[i*stride+j];
1509
1510 // invert local matrix
1511 Math::invert_matrix(nrc.first, stride, local, pivot);
1512
1513 // compute (squared) Frobenius norm of (I - A*A^{-1})
1514 DataType norm = DataType(0);
1515 for(Index i(0); i < nrc.first; ++i)
1516 {
1517 for(Index j(0); j < nrc.first; ++j)
1518 {
1519 DataType xij = DataType(i == j ? 1 : 0);
1520 for(Index k(0); k < nrc.first; ++k)
1521 xij -= local_t[i*stride+k] * local[k*stride+j]; // A_ik * (A^{-1})_kj
1522 norm += xij * xij;
1523 }
1524 }
1525
1526 // is the matrix block singular?
1527 // Note: we check for !(norm < eps) instead of (norm >= eps),
1528 // because the latter one evaluates to false if norm is NaN,
1529 // which would result in a false negative
1530 const bool singular = !(norm < eps);
1531
1532 // set macro regularity mask
1533 this->_macro_mask[imacro] = (singular ? 0 : 1);
1534
1535 // scatter local matrix
1536 if(!singular)
1537 {
1538 Intern::AmaVankaCore::scatter_add(this->_vanka, local, stride, imacro, this->_macro_dofs,
1539 Index(0), Index(0), Index(0), Index(0));
1540 }
1541 }
1542 else // no singularity check
1543 {
1544 // invert local matrix
1545 Math::invert_matrix(nrc.first, stride, local, pivot);
1546
1547 // scatter local matrix
1548 Intern::AmaVankaCore::scatter_add(this->_vanka, local, stride, imacro, this->_macro_dofs,
1549 Index(0), Index(0), Index(0), Index(0));
1550 }
1551
1552 // reformat local matrix
1553 for(Index i(0); i < nrc.first; ++i)
1554 for(Index j(0); j < nrc.second; ++j)
1555 local[i*stride+j] = DataType(0);
1556 }
1557
1558 // scale rows of Vanka matrix
1559 Solver::Intern::AmaVankaCore::scale_rows(this->_vanka, this->_omega, this->_dof_macros, this->_macro_mask, Index(0));
1560
1561 watch_init_numeric.stop();
1562 }
1563
1565 virtual Status apply(VectorType& vec_x, const VectorType& vec_b) override
1566 {
1567 watch_apply.start();
1568
1569 // first step
1570 this->_vanka.apply(vec_x, vec_b);
1571 this->_filter.filter_cor(vec_x);
1572
1573 // steps 2, ..., n (if any)
1574 for(Index step(1); step < _num_steps; ++step)
1575 {
1576 // compute defect
1577 this->_matrix.apply(this->_vec_d, vec_x, vec_b, -DataType(1));
1578 // filter defect
1579 this->_filter.filter_def(this->_vec_d);
1580 // apply Vanka matrix
1581 this->_vanka.apply(this->_vec_c, this->_vec_d);
1582 // filter correct
1583 this->_filter.filter_cor(this->_vec_c);
1584 // update solution
1585 vec_x.axpy(this->_vec_c, vec_x);
1586 }
1587
1588 watch_apply.stop();
1589
1590 return Status::success;
1591 }
