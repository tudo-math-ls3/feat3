2929 template<int n_>
2930 struct DetHelper<n_,n_>
2931 {
2932 template<typename T_, int sm_, int sn_>
2933 static T_ compute(const T_ (&a)[sm_][sn_])
2934 {
2935 // create temporary copy of a and a pivot array
2936 T_ b[n_*n_];
2937 int p[n_];
2938
2939 // copy matrix a to b
2940 for (int i(0); i < n_; ++i)
2941 {
2942 for (int j(0); j < n_; ++j)
2943 {
2944 b[i*n_+j] = a[i][j];
2945 }
2946 }
2947
2948 // perform matrix inversion which returns the determinant
2949 const T_ det = Math::invert_matrix(n_, n_, b, p);
2950
2951 // if the returned value is not normal, we can assume that the matrix is singular
2952 return Math::isnormal(det) ? det : T_(0);
2953 }
2954 };
