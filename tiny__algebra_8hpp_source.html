510 {
511 #ifndef __CUDACC__
512 const DataType norm2(this->norm_euclid());
513 ASSERTM(norm2 > Math::eps<DataType>(), "Trying to normalize a null vector!");
514 return ((*this) *= (DataType(1)/norm2));
515 #else
516 const DataType norm2_sqr(this->norm_euclid_sqr());
517 ASSERTM(norm2_sqr > CudaMath::cuda_get_eps<DataType>(), "Trying to normalize a null vector!");
518 return ((*this) *= CudaMath::cuda_rsqrt(norm2_sqr));

3812 struct InverseHelper<n_, n_>
3813 {
3814 template<typename T_, int smb_, int snb_, int sma_, int sna_>
3815 CUDA_HOST_DEVICE static T_ compute(Tiny::Matrix<T_, n_, n_, smb_, snb_>& b, const Tiny::Matrix<T_, n_, n_, sma_, sna_>& a)
3816 {
3817 // copy matrix a to b
3818 for (int i(0); i < n_; ++i)
3819 {
3820 for (int j(0); j < n_; ++j)
3821 {
3822 b(i,j) = a(i,j);
3823 }
3824 }
3825
3826 // create pivot array
3827 int p[n_];
3828
3829 // initialize identity permutation
3830 for(int i(0); i < n_; ++i)
3831 {
3832 p[i] = i;
3833 }
3834
3835 // initialize determinant to 1
3836 T_ det = T_(1);
3837
3838 // primary column elimination loop
3839 for(int k(0); k < n_; ++k)
3840 {
3841 // step 1: find a pivot for the elimination of column k
3842 {
3843 // for this, we only check the rows p[j] with j >= k, as all
3844 // rows p[j] with j < k have already been eliminated and are
3845 // therefore not candidates for pivoting
3846 #ifdef __CUDACC__
3847 T_ pivot = CudaMath::cuda_abs(b(p[k], p[k]));
3848 #else
3849 T_ pivot = Math::abs(b(p[k], p[k]));
3850 #endif
3851 int i = k;
3852
3853 // loop over all unprocessed rows
3854 for(int j(k+1); j < n_; ++j)
3855 {
3856 // get our matrix value and check whether it can be a pivot
3857 #ifdef __CUDACC__
3858 T_ abs_val = CudaMath::cuda_abs(b(p[j], p[j]));
3859 #else
3860 T_ abs_val = Math::abs(b(p[j], p[j]));
3861 #endif
3862 if(abs_val > pivot)
3863 {
3864 pivot = abs_val;
3865 i = j;
3866 }
3867 }
3868
3869 // do we have to swap rows i and k?
3870 if(i > k)
3871 {
3872 // swap rows "virtually" by exchanging their permutation positions
3873 int t = p[k];
3874 p[k] = p[i];
3875 p[i] = t;
3876 }
3877 }
3878
3879 // step 2: process pivot row
3880 {
3881 // update determinant by multiplying with the pivot element
3882 det *= b(p[k], p[k]);
3883
3884 // get our inverted pivot element
3885 const T_ pivot = T_(1) / b(p[k], p[k]);
3886
3887 // replace column entry by unit column entry
3888 b(p[k], p[k]) = T_(1);
3889
3890 // divide the whole row by the inverted pivot
3891 for(int j(0); j < n_; ++j)
3892 {
3893 b(p[k], j) *= pivot;
3894 }
3895 }
3896
3897 // step 3: eliminate pivot column
3898
3899 // loop over all rows of the matrix
3900 for(int i(0); i < n_; ++i)
3901 {
3902 // skip the pivot row
3903 if(i == p[k])
3904 continue;
3905
3906 // fetch elimination factor
3907 const T_ factor = b(i, p[k]);
3908
3909 // replace by unit column entry
3910 b(i, p[k]) = T_(0);
3911
3912 // process the row
3913 for(int j(0); j < n_; ++j)
3914 {
3915 b(i, j) -= b(p[k], j) * factor;
3916 }
3917 }
3918 }
3919
3920 // return determinant
3921 return det;
3922 }
