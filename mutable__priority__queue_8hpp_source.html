38/*
39 A priority queue in which the priorities of entries may be
40 updated in lg(n) time.
41
42 Simply two maps glued together, one mapping keys (priorities)
43 to values and the other mapping values to keys. Both keys
44 and values must meet the requirements for storage in a
45 std::map, i.e. be less than comparable. Typically not a
46 problem for priorities, but can be a pain if Values are stored
47 by value in the structures. Not a problem if pointers to
48 values are stored.
49
50 insert, erase and update operations are all logarithmic, but
51 approximately 2x slower than std::map since two std::maps
52 are updated in each case.
53*/
54 template< typename _Key, typename _Value, typename _KeyComp=std::greater<_Key>, typename _ValueComp=std::less<_Value> >
55 class mutable_priority_queue {
56 public:
57 // iterator type for the key2val map, allows in-order traversal of the queue
58 typedef typename std::multimap<_Key,_Value,_KeyComp>::const_iterator iterator;
59 typedef typename std::multimap<_Key,_Value,_KeyComp>::const_iterator const_iterator;
60
61 // default constructor, uses std::greater and std::less for key and value comparisons
62 mutable_priority_queue() : key2Val(_KeyComp()), val2Key(_ValueComp()) { }
63
64 // constructor with key comparison predicate
65 mutable_priority_queue( const _KeyComp &keyComp ) : key2Val(keyComp), val2Key(_ValueComp()){ }
66
67 // constructor with value comparison predicate
68 mutable_priority_queue( const _ValueComp &valComp ) : key2Val(_KeyComp()), val2Key(valComp) { }
69
70 // constructor with both comparison predicates
71 mutable_priority_queue( const _KeyComp &keyComp, const _ValueComp &valComp ) : key2Val(keyComp), val2Key(valComp) { }
72
73 // copy constructor
74 mutable_priority_queue( const mutable_priority_queue<_Key,_Value,_KeyComp,_ValueComp> &x ) : key2Val(x.key2Val), val2Key(x.val2Key) {}
75
76 // destructor, clears both maps
77 ~mutable_priority_queue(){
78 key2Val.clear();
79 val2Key.clear();
80 }
81
82 // assignment operator
83 inline void operator=( mutable_priority_queue<_Key,_Value,_KeyComp,_ValueComp> x ){
84 key2Val = x.key2Val;
85 val2Key = x.val2Key;
86 }
87
88 // returns an iterator for the front of the queue
89 inline iterator begin(){
90 return key2Val.begin();
91 }
92
93 // returns an iterator for the end of the queue
94 inline iterator end(){
95 return key2Val.end();
96 }
97
98 // empties the queue
99 inline void clear(){
100 key2Val.clear();
101 val2Key.clear();
102 }
103
104 // removes from both maps the value val
105 inline void erase( _Value val ){
106 typename std::map<_Value,_Key,_ValueComp>::iterator iter=val2Key.find( val );
107 if( iter != val2Key.end() ){
108 key2Val.erase( _find(iter->second,iter->first) );
109 val2Key.erase( iter );
110 }
111 }
112
113 // returns true if the queue is empty
114 inline bool empty(){
115 return key2Val.empty();
116 }
117
118 // returns the value at the front of the queue
119 inline _Value front_value(){
120 return (*key2Val.begin()).second;
121 }
122
123 // returns the key at the front of the queue
124 inline _Value front_key(){
125 return (*key2Val.begin()).first;
126 }
127
128 // removes the front entry in the queue
129 inline void pop(){
130 auto iter = key2Val.begin();
131 _erase( iter );
132 }
133
134 // push an entry onto the queue
135 inline void push( _Value val, _Key key ){
136 insert( val, key );
137 }
138
139 // adds a value key pair to the queue.
140 // an attempt first has to be made to erase
141 // the item, since it could exist with another
142 // key and the queue does not allow multiple
143 // keys per value
144 inline void insert( _Value val, _Key key ){
145 erase( val );
146 key2Val.insert( std::pair<_Key,_Value>( key, val ) );
147 val2Key.insert( std::pair<_Value,_Key>( val, key ) );
148 }
149
150 // update the key associated with a value,
151 // simply a convenience function that calls
152 // insert()
153 inline void update( _Value val, _Key key ){
154 insert( val, key );
155 }
156
157 // returns the number of elements in the queue
158 inline size_t size(){
159 return key2Val.size();
160 }
161
162 inline size_t count(_Value val)
163 {
164 return val2Key.count(val);
165 }
166
167 private:
168 std::multimap< _Key, _Value, _KeyComp > key2Val;
169 std::map< _Value, _Key, _ValueComp > val2Key;
170
171 inline void _erase( typename std::multimap<_Key,_Value,_KeyComp>::iterator it ){
172 val2Key.erase( it->second );
173 key2Val.erase( it );
174 }
175
176 inline typename std::multimap<_Key,_Value,_KeyComp>::iterator _find( _Key k, _Value v ){
177 typename std::multimap<_Key,_Value,_KeyComp>::iterator it = key2Val.lower_bound(k);
178 typename std::multimap<_Key,_Value,_KeyComp>::iterator hi = key2Val.upper_bound(k);
179 while( it != hi ){
180 if( it->second == v )
181 return it;
182 it++;
183 }
184 return key2Val.end();
185 }
186 };
187}
