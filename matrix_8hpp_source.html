256 template<LAFEM::Perspective perspective_ = LAFEM::Perspective::pod>
257 Index used_elements() const
258 {
259 Index my_used_elements(_matrix.template used_elements<perspective_>());
260 const Dist::Comm* comm = (_row_gate != nullptr ? _row_gate->get_comm() : (_col_gate != nullptr ? _col_gate->get_comm() : nullptr));
261 if((comm != nullptr) && (comm->size() > 1))
262 comm->allreduce(&my_used_elements, &my_used_elements, std::size_t(1), Dist::op_sum);
263 return my_used_elements;

365 void apply(VectorTypeL& r, const VectorTypeR& x, const VectorTypeL& y, const DataType alpha = DataType(1)) const
366 {
367 // copy y to r
368 r.copy(y);
369
370 // convert from type-1 to type-0
371 r.from_1_to_0();
372
373 // r <- r + alpha*A*x
374 _matrix.apply(r.local(), x.local(), r.local(), alpha);
375
376 // synchronize r
377 r.sync_0();
378 }

497 LocalMatrix_ convert_to_1() const
498 {
499 LocalMatrix_ locmat = _matrix.clone(LAFEM::CloneMode::Weak);
500 if((_row_gate != nullptr) && (_col_gate != nullptr))
501 {
502 SynchMatrix<LocalMatrix_, RowMirror_> synch(*_row_gate->_comm, _row_gate->_ranks, _row_gate->_mirrors, _col_gate->_mirrors);
503 synch.init(locmat);
