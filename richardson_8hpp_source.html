174 virtual Status correct(VectorType& vec_sol, const VectorType& vec_rhs) override
175 {
176 // compute defect
177 this->_system_matrix.apply(this->_vec_def, vec_sol, vec_rhs, -DataType(1));
178 this->_system_filter.filter_def(this->_vec_def);
179
180 // apply
181 this->_status = _apply_intern(vec_sol, vec_rhs);
182 this->plot_summary();

186 protected:
187 virtual Status _apply_intern(VectorType& vec_sol, const VectorType& vec_rhs)
188 {
189 IterationStats pre_iter(*this);
190 Statistics::add_solver_expression(std::make_shared<ExpressionStartSolve>(this->name()));
191
192 VectorType& vec_def(this->_vec_def);
193 VectorType& vec_cor(this->_vec_cor);
194 const MatrixType& matrix(this->_system_matrix);
195 const FilterType& filter(this->_system_filter);
196
197 // compute initial defect
198 Status status = this->_set_initial_defect(vec_def, vec_sol);
199
200 pre_iter.destroy();
201
202 // start iterating
203 while(status == Status::progress)
204 {
205 IterationStats stat(*this);
206
207 // apply preconditioner
208 if(!this->_apply_precond(vec_cor, vec_def, filter))
209 {
210 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
211 return Status::aborted;
212 }
213 //filter.filter_cor(vec_cor);
214
215 // update solution vector
216 vec_sol.axpy(vec_cor, this->_omega);
217
218 // compute new defect vector
219 matrix.apply(vec_def, vec_sol, vec_rhs, -DataType(1));
220 filter.filter_def(vec_def);
221
222 // compute new defect norm
223 status = this->_set_new_defect(vec_def, vec_sol);
224 }
225
226 // return our status
227 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), status, this->get_num_iter()));
228 return status;
229 }
