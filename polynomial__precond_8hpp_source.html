195 virtual Status apply(VectorType& vec_cor, const VectorType& vec_def) override
196 {
197 /*
198 * preconditioner is given by
199 * \f$ M^-1 = \left(I + (I - \tilde M^{-1}A) + ... + (I - \tilde M^{-1 }A)^m\right) \tilde M^{-1} \f$
200 *
201 * the preconditioner only works, if
202 * ||I - \tilde M^{-1} A||_2 < 1.
203 */
204
205 vec_cor.component_product(_inv_diag, vec_def);
206 _aux3.copy(vec_cor);
207
208 for (Index i = 1; i <= _m; ++i)
209 {
210 _matrix.apply(_aux1, vec_cor);
211 _filter.filter_def(_aux1);
212 _aux2.component_product(_inv_diag, _aux1);
213 vec_cor.axpy(_aux3);
214 vec_cor.axpy(_aux2, DataType(-1.0));
215 }
216
217 this->_filter.filter_cor(vec_cor);
218
219 return Status::success;
220 }
