217 void filter_rhs(VectorType& vector) const
218 {
219 // subtract to dual integral mean zero
220 if(!_vec_prim.empty())
221 {
222 ValueType tmp(vector.dot_blocked(_vec_prim));
223 for(int i(0); i<ValueType::n; ++i)
224 {
225 tmp(i) /= -_volume(i);
226 }
227 vector.axpy_blocked(_vec_dual, vector, tmp);
228 }
229 }

237 void filter_sol(VectorType& vector) const
238 {
239 // given: v=vec_prim, w=vec_dual, vol=v*w, input: x
240 // note that int(v) = v*w and int(x) = x*w
241 // we want:
242 // 1/vol * int (x + alpha*v) = sol_mean
243 // <==> int(x) + alpha*int(v) = sol_mean*vol
244 // <==> x*w + alpha*v*w = sol_mean*v*w
245 // <==> alpha*v*w = sol_mean*v*w - x*w
246 // <==> alpha = sol_mean - (x*w/v*w)
247 if(!_vec_prim.empty())
248 {
249 ValueType tmp(vector.dot_blocked(_vec_dual));
250 for(int i(0); i<ValueType::n; ++i)
251 {
252 tmp(i)= _sol_mean(i) - tmp(i) * DT_(DT_(1)/_volume(i));
253 }
254 vector.axpy_blocked(_vec_prim, vector, tmp);
255 }
256 }
