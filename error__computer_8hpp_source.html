41 template<typename AsmTraits_, typename AnaTraits_>
42 struct SecHelperH0<true, AsmTraits_, AnaTraits_>
43 {
44 typedef typename AsmTraits_::DataType DataType;
45
46 template<typename FuncEval_, typename LocalVec_>
47 static DataType eval(
48 FuncEval_& func_eval,
49 const typename AsmTraits_::TrafoEvalData& trafo_data,
50 const typename AsmTraits_::SpaceEvalData& space_data,
51 const LocalVec_& lvad,
52 const int num_loc_dofs
53 )
54 {
55 // evaluate function value
56 typename AnaTraits_::ValueType value = func_eval.value(trafo_data.img_point);
57
58 // subtract FE function value
59 for(int i(0); i < num_loc_dofs; ++i)
60 value -= lvad[i] * space_data.phi[i].value;
61
62 // return result
63 return Math::abs(value);
64 }

86 template<typename AsmTraits_, typename AnaTraits_>
87 struct SecHelperH1<true, AsmTraits_, AnaTraits_, false>

112 // sub-dimensional variant for H1
113 template<typename AsmTraits_, typename AnaTraits_>
114 struct SecHelperH1<true, AsmTraits_, AnaTraits_, true>
115 {
116 typedef typename AsmTraits_::DataType DataType;
117
118 template<typename FuncEval_, typename LocalVec_>
119 static DataType eval(
120 FuncEval_& func_eval,
121 const typename AsmTraits_::TrafoEvalData& trafo_data,
122 const typename AsmTraits_::SpaceEvalData& space_data,
123 const LocalVec_& lvad,
124 const int num_loc_dofs
125 )
126 {
127 // get the dimensions
128 static constexpr int dom_dim = AsmTraits_::domain_dim;
129
130 // evaluate function gradient
131 typename AnaTraits_::GradientType grad = func_eval.gradient(trafo_data.img_point);
132
133 // transform back onto reference element by applying chain rule
134 Tiny::Vector<DataType, dom_dim> ref_grad;
135 ref_grad.set_vec_mat_mult(grad, trafo_data.jac_mat);
136
137 // subtract FE function reference gradient
138 for(int i(0); i < num_loc_dofs; ++i)
139 ref_grad.axpy(-lvad[i], space_data.phi[i].ref_grad);
140
141 // compute gram matrix of transformation
142 Tiny::Matrix<DataType, dom_dim, dom_dim> gram_mat, gram_inv;
143 gram_mat.set_gram(trafo_data.jac_mat);
144 gram_inv.set_inverse(gram_mat);
145
146 // return result
147 return gram_inv.scalar_product(ref_grad, ref_grad);
148 }
149 };

285 void synchronize(const Dist::Comm& comm)
286 {
287 DataType_ verr[4] =
288 {
289 have_h0 ? Math::sqr(norm_h0) : DataType_(0),
290 have_h1 ? Math::sqr(norm_h1) : DataType_(0),
291 have_h2 ? Math::sqr(norm_h2) : DataType_(0),
292 have_l1 ? norm_l1 : DataType_(0)
293 };
294
295 comm.allreduce(verr, verr, std::size_t(4), Dist::op_sum);
296
297 if(have_h0) norm_h0 = Math::sqrt(verr[0]);
298 if(have_h1) norm_h1 = Math::sqrt(verr[1]);
299 if(have_h2) norm_h2 = Math::sqrt(verr[2]);
300 if(have_l1) norm_l1 = verr[3];
301
302 if(have_lmax)
303 {
304 comm.allreduce(&norm_lmax, &norm_lmax, std::size_t(1), Dist::op_max);
305 }
306 }

376 template<int max_norm_ = 0, bool sub_dimensional_ = false>
377 class ScalarErrorComputer
378 {
379 static_assert(max_norm_ >= 0, "invalid max_norm_ parameter");
380
381 private:
383
386 static constexpr TrafoTags trafo_config = TrafoTags::img_point | TrafoTags::jac_mat | TrafoTags::jac_det;
387
391 static constexpr SpaceTags space_config =
392 ((max_norm_ >= 0) ? SpaceTags::value : SpaceTags::none) |
393 ((max_norm_ >= 1) ? (sub_dimensional_ ? SpaceTags::ref_grad : SpaceTags::grad) : SpaceTags::none) |
394 ((max_norm_ >= 2) ? SpaceTags::hess : SpaceTags::none);
396
397 public:
421 template<
422 typename Vector_,
423 typename Function_,
424 typename Space_,
425 typename CubatureFactory_>
426 static ScalarErrorInfo<typename Vector_::DataType> compute(
427 const Vector_& vector,
428 const Function_& function,
429 const Space_& space,
430 const CubatureFactory_& cubature_factory)
431 {
432 // make sure the function has everything we need
433 static_assert(Function_::can_value, "function values are required for H0-Error");
434 static_assert(Function_::can_grad || (max_norm_ < 1), "function gradients are required for H1-Error");
435 static_assert(Function_::can_hess || (max_norm_ < 2), "function hessians are required for H2-Error");
436
437 // validate vector dimensions
438 XASSERTM(vector.size() == space.get_num_dofs(), "invalid vector size");
439
440 // vector type
441 typedef Vector_ VectorType;
442 // analytic function type
443 typedef Function_ FunctionType;
444 // space type
445 typedef Space_ SpaceType;
446 // assembly traits
447 typedef AsmTraits1<typename Vector_::DataType, SpaceType, trafo_config, space_config> AsmTraits;
448 // data type
449 typedef typename AsmTraits::DataType DataType;
450
451 // create the cubature rule
452 typename AsmTraits::CubatureRuleType cubature_rule(Cubature::ctor_factory, cubature_factory);
453
454 // fetch the trafo
455 const typename AsmTraits::TrafoType& trafo = space.get_trafo();
456
457 // define our analytic evaluation traits
458 typedef Analytic::EvalTraits<DataType, Function_> AnalyticEvalTraits;
459
460 // create a function evaluator
461 typename FunctionType::template Evaluator<AnalyticEvalTraits> func_eval(function);
462
463 // create a trafo evaluator
464 typename AsmTraits::TrafoEvaluator trafo_eval(trafo);
465
466 // create a space evaluator and evaluation data
467 typename AsmTraits::SpaceEvaluator space_eval(space);
468
469 // create a dof-mapping
470 typename AsmTraits::DofMapping dof_mapping(space);
471
472 // create trafo evaluation data
473 typename AsmTraits::TrafoEvalData trafo_data;
474
475 // create space evaluation data
476 typename AsmTraits::SpaceEvalData space_data;
477
478 // create local vector data
479 typename AsmTraits::template TLocalVector<DataType> lvad;
480
481 // create matrix scatter-axpy
482 typename VectorType::GatherAxpy gather_axpy(vector);
483
484 // initialize result
485 ScalarErrorInfo<DataType> result;
486 result.have_h0 = (max_norm_ >= 0);
487 result.have_h1 = (max_norm_ >= 1);
488 result.have_h2 = (max_norm_ >= 2);
489 result.have_l1 = (max_norm_ >= 0);
490 result.have_lmax = (max_norm_ >= 0);
491
492 // H0/H1/H2 helpers
493 typedef Intern::SecHelperH0<(max_norm_ >= 0), AsmTraits, AnalyticEvalTraits> SecH0;
494 typedef Intern::SecHelperH1<(max_norm_ >= 1), AsmTraits, AnalyticEvalTraits, sub_dimensional_> SecH1;
495 typedef Intern::SecHelperH2<(max_norm_ >= 2), AsmTraits, AnalyticEvalTraits> SecH2;
496
497 // loop over all cells of the mesh
498 for(typename AsmTraits::CellIterator cell(trafo_eval.begin()); cell != trafo_eval.end(); ++cell)
499 {
500 // format local vector
501 lvad.format();
502
503 // initialize dof-mapping
504 dof_mapping.prepare(cell);
505
506 // incorporate local matrix
507 gather_axpy(lvad, dof_mapping);
508
509 // finish dof-mapping
510 dof_mapping.finish();
511
512 // prepare trafo evaluator
513 trafo_eval.prepare(cell);
514
515 // prepare space evaluator
516 space_eval.prepare(trafo_eval);
517
518 // fetch number of local dofs
519 int num_loc_dofs = space_eval.get_num_local_dofs();
520
521 // loop over all quadrature points and integrate
522 for(int k(0); k < cubature_rule.get_num_points(); ++k)
523 {
524 // compute trafo data
525 trafo_eval(trafo_data, cubature_rule.get_point(k));
526
527 // compute basis function data
528 space_eval(space_data, trafo_data);
529
530 // compute integration weight
531 const DataType omega = trafo_data.jac_det * cubature_rule.get_weight(k);
532
533 // get absolute point error
534 const DataType abs_err = SecH0::eval(func_eval, trafo_data, space_data, lvad, num_loc_dofs);
535
536 // update results
537 result.norm_lmax = Math::max(result.norm_lmax, abs_err);
538 result.norm_l1 += omega * abs_err;
539 result.norm_h0 += omega * abs_err * abs_err;
540 result.norm_h1 += omega * SecH1::eval(func_eval, trafo_data, space_data, lvad, num_loc_dofs);
541 result.norm_h2 += omega * SecH2::eval(func_eval, trafo_data, space_data, lvad, num_loc_dofs);
542
543 // continue with next cubature point
544 }
545
546 // finish evaluators
547 space_eval.finish();
548 trafo_eval.finish();
549
550 // continue with next cell
551 }
552
553 // take square-roots
554 result.norm_h0 = Math::sqrt(result.norm_h0);
555 result.norm_h1 = Math::sqrt(result.norm_h1);
556 result.norm_h2 = Math::sqrt(result.norm_h2);
557
558 // okay, that's it
559 return result;
560 }
561 }; // class ScalarErrorComputer

861 void synchronize(const Dist::Comm& comm)
862 {
863 DataType_ verr[4+4*dim_] =
864 {
865 have_h0 ? Math::sqr(norm_h0) : DataType_(0),
866 have_h1 ? Math::sqr(norm_h1) : DataType_(0),
867 have_h2 ? Math::sqr(norm_h2) : DataType_(0),
868 have_l1 ? norm_l1 : DataType_(0)
869 };
870 for(int i(0); i < dim_; ++i)
871 {
872 verr[4+0*dim_+i] = (have_h0 ? Math::sqr(norm_h0_comp[i]) : DataType_(0));
873 verr[4+1*dim_+i] = (have_h0 ? Math::sqr(norm_h1_comp[i]) : DataType_(0));
874 verr[4+2*dim_+i] = (have_h0 ? Math::sqr(norm_h2_comp[i]) : DataType_(0));
875 verr[4+3*dim_+i] = (have_l1 ? norm_l1_comp[i] : DataType_(0));
876 }
877
878 comm.allreduce(verr, verr, std::size_t(4*dim+4), Dist::op_sum);
879
880 if(have_h0)
881 {
882 norm_h0 = Math::sqrt(verr[0]);
883 for(int i(0); i < dim_; ++i)
884 norm_h0_comp[i] = Math::sqrt(verr[4+i]);
885 }
886 if(have_h1)
887 {
888 norm_h1 = Math::sqrt(verr[1]);
889 for(int i(0); i < dim_; ++i)
890 norm_h1_comp[i] = Math::sqrt(verr[4+dim_+i]);
891 }
892 if(have_h2)
893 {
894 norm_h2 = Math::sqrt(verr[2]);
895 for(int i(0); i < dim_; ++i)
896 norm_h2_comp[i] = Math::sqrt(verr[4+2*dim_+i]);
897 }
898 if(have_l1)
899 {
900 norm_l1 = verr[3];
901 for(int i(0); i < dim_; ++i)
902 norm_l1_comp[i] = verr[4+3*dim_+i];
903 }
904 if(have_lmax)
905 {
906 DataType_ vmax[1+dim_] = { norm_lmax };
907 for(int i(0); i < dim_; ++i)
908 vmax[i+1] = norm_lmax_comp[i];
909 comm.allreduce(vmax, vmax, std::size_t(dim+1), Dist::op_max);
910 norm_lmax = vmax[0];
911 for(int i(0); i < dim_; ++i)
912 norm_lmax_comp[i] = vmax[i+1];
913 }

1008 class VectorErrorComputer
1009 {
1010 static_assert(max_norm_ >= 0, "invalid max_norm_ parameter");
1011
1012 private:
1014
1017 static constexpr TrafoTags trafo_config = TrafoTags::img_point | TrafoTags::jac_det;
1018
1022 static constexpr SpaceTags space_config =
1023 ((max_norm_ >= 0) ? SpaceTags::value : SpaceTags::none) |
1024 ((max_norm_ >= 1) ? SpaceTags::grad : SpaceTags::none) |
1025 ((max_norm_ >= 2) ? SpaceTags::hess : SpaceTags::none);

1052 template<
1053 typename Vector_,
1054 typename Function_,
1055 typename Space_,
1056 typename CubatureFactory_>
1057 static VectorErrorInfo<typename Vector_::DataType, Function_::ImageType::scalar_components> compute(
1058 const Vector_& vector,
1059 const Function_& function,
1060 const Space_& space,
1061 const CubatureFactory_& cubature_factory)
1062 {
1063 // make sure the function has everything we need
1064 static_assert(Function_::can_value, "function values are required for H0-Error");
1065 static_assert(Function_::can_grad || (max_norm_ < 1), "function gradients are required for H1-Error");
1066 static_assert(Function_::can_hess || (max_norm_ < 2), "function hessians are required for H2-Error");
1067
1068 // validate vector dimensions
1069 XASSERTM(vector.size() == space.get_num_dofs(), "invalid vector size");
1070
1071 // vector type
1072 typedef Vector_ VectorType;
1073 // analytic function type
1074 typedef Function_ FunctionType;
1075 // space type
1076 typedef Space_ SpaceType;
1078 static constexpr int dim = Function_::ImageType::scalar_components;
1080 typedef AsmTraits1
1081 <
1082 typename Vector_::DataType,
1083 SpaceType,
1084 trafo_config,
1085 space_config
1086 > AsmTraits;
1087 // data type
1088 typedef typename AsmTraits::DataType DataType;
1089
1090 // create the cubature rule
1091 typename AsmTraits::CubatureRuleType cubature_rule(Cubature::ctor_factory, cubature_factory);
1092
1093 // fetch the trafo
1094 const typename AsmTraits::TrafoType& trafo = space.get_trafo();
1095
1096 // define our analytic evaluation traits
1097 typedef Analytic::EvalTraits<DataType, Function_> AnalyticEvalTraits;
1098
1099 // create a function evaluator
1100 typename FunctionType::template Evaluator<AnalyticEvalTraits> func_eval(function);
1101
1102 // create a trafo evaluator
1103 typename AsmTraits::TrafoEvaluator trafo_eval(trafo);
1104
1105 // create a space evaluator and evaluation data
1106 typename AsmTraits::SpaceEvaluator space_eval(space);
1107
1108 // create a dof-mapping
1109 typename AsmTraits::DofMapping dof_mapping(space);
1110
1111 // create trafo evaluation data
1112 typename AsmTraits::TrafoEvalData trafo_data;
1113
1114 // create space evaluation data
1115 typename AsmTraits::SpaceEvalData space_data;
1116
1117 // create local vector data
1118 typedef Tiny::Vector<DataType, dim> VectorValue;
1119
1120 // create local vector data
1121 typename AsmTraits::template TLocalVector<VectorValue> lvad;
1122
1123 // create matrix scatter-axpy
1124 typename VectorType::GatherAxpy gather_axpy(vector);
1125
1126 // initialize result
1127 VectorErrorInfo<DataType, dim> info;
1128 info.have_h0 = (max_norm_ >= 0);
1129 info.have_h1 = (max_norm_ >= 1);
1130 info.have_h2 = (max_norm_ >= 2);
1131 info.have_l1 = (max_norm_ >= 0);
1132 info.have_lmax = (max_norm_ >= 0);
1133
1134 // H0/H1/H2 helpers
1135 typedef Intern::VecHelperH0<(max_norm_ >= 0), AsmTraits, AnalyticEvalTraits> VecH0;
1136 typedef Intern::VecHelperH1<(max_norm_ >= 1), AsmTraits, AnalyticEvalTraits> VecH1;
1137 typedef Intern::VecHelperH2<(max_norm_ >= 2), AsmTraits, AnalyticEvalTraits> VecH2;
1138
1139 // loop over all cells of the mesh
1140 for(typename AsmTraits::CellIterator cell(trafo_eval.begin()); cell != trafo_eval.end(); ++cell)
1141 {
1142 // format local vector
1143 lvad.format();
1144
1145 // initialize dof-mapping
1146 dof_mapping.prepare(cell);
1147
1148 // incorporate local matrix
1149 gather_axpy(lvad, dof_mapping);
1150
1151 // finish dof-mapping
1152 dof_mapping.finish();
1153
1154 // prepare trafo evaluator
1155 trafo_eval.prepare(cell);
1156
1157 // prepare space evaluator
1158 space_eval.prepare(trafo_eval);
1159
1160 // fetch number of local dofs
1161 int num_loc_dofs = space_eval.get_num_local_dofs();
1162
1163 // loop over all quadrature points and integrate
1164 for(int k(0); k < cubature_rule.get_num_points(); ++k)
1165 {
1166 // compute trafo data
1167 trafo_eval(trafo_data, cubature_rule.get_point(k));
1168
1169 // compute basis function data
1170 space_eval(space_data, trafo_data);
1171
1172 // compute integration weight
1173 const DataType omega = trafo_data.jac_det * cubature_rule.get_weight(k);
1174
1175 // get absolute point error
1176 auto abs_err = VecH0::eval(func_eval, trafo_data, space_data, lvad, num_loc_dofs);
1177
1178 // update H0/L1/Lmax results
1179 for(int i(0); i < dim; ++i)
1180 {
1181 info.norm_lmax_comp[i] = Math::max(info.norm_lmax_comp[i], abs_err[i]);
1182 info.norm_l1_comp[i] += omega * abs_err[i];
1183 info.norm_h0_comp[i] += omega * Math::sqr(abs_err[i]);
1184 }
1185
1186 // update H1/H2 results
1187 info.norm_h1_comp.axpy(omega, VecH1::eval(func_eval, trafo_data, space_data, lvad, num_loc_dofs));
1188 info.norm_h2_comp.axpy(omega, VecH2::eval(func_eval, trafo_data, space_data, lvad, num_loc_dofs));
1189
1190 // continue with next cubature point
1191 }
1192
1193 // finish evaluators
1194 space_eval.finish();
1195 trafo_eval.finish();
1196
1197 // continue with next cell
1198 }
1199
1200 // compute the rest
1201 for(int i(0); i < dim; ++i)
1202 {
1203 info.norm_h0 += info.norm_h0_comp[i];
1204 info.norm_h1 += info.norm_h1_comp[i];
1205 info.norm_h2 += info.norm_h2_comp[i];
1206 info.norm_l1 += info.norm_l1_comp[i];
1207 info.norm_lmax = Math::max(info.norm_lmax, info.norm_lmax_comp[i]);
1208 info.norm_h0_comp[i] = Math::sqrt(info.norm_h0_comp[i]);
1209 info.norm_h1_comp[i] = Math::sqrt(info.norm_h1_comp[i]);
1210 info.norm_h2_comp[i] = Math::sqrt(info.norm_h2_comp[i]);
1211 }
1212
1213 // take square-roots
1214 info.norm_h0 = Math::sqrt(info.norm_h0);
1215 info.norm_h1 = Math::sqrt(info.norm_h1);
1216 info.norm_h2 = Math::sqrt(info.norm_h2);
1217
1218 // okay, that's it
1219 return info;
1220 }
