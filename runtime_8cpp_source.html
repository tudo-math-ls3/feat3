37void Runtime::initialize(int& argc, char**& argv)
38{
43#if defined(_WIN32) && defined(FEAT_TESTING_VC)
44 Windows::disable_error_prompts();
45 Windows::install_seh_filter();
46#endif
47
48 if (_initialized)
49 {
50 std::cerr << "ERROR: Runtime::initialize called twice!" << std::endl;
51 std::cerr.flush();
52 Runtime::abort();
53 }
54 if (_finalized)
55 {
56 std::cerr << "ERROR: Runtime::initialize called after Runtime::finalize!" << std::endl;
57 std::cerr.flush();
58 Runtime::abort();
59 }
60
61 // initialize Dist operations
62 if(!Dist::initialize(argc, argv))
63 {
64 std::cerr << "ERROR: Failed to initialize Dist operations!" << std::endl;
65 std::cerr.flush();
66 Runtime::abort();
67 }
68
69 // initialize memory pool for main memory
70 MemoryPool::initialize();
71
72#ifdef FEAT_HAVE_CUDA
73 // initialize memory pool for CUDA memory
74 int rank = Dist::Comm::world().rank();
75 Util::cuda_initialize(rank, 1, 1, 1);
76 Util::cuda_set_blocksize(256, 256, 256, 256);
77#endif
78
79 _initialized = true;
80}

82void Runtime::abort(bool dump_call_stack)
83{
84 if(dump_call_stack)
85 {
86#if defined(__linux) || defined(__unix__)
87#if defined(FEAT_HAVE_DEATH_HANDLER) and not defined(FEAT_HAVE_MPI)
88 Debug::DeathHandler death_handler;
89#else
90 // https://www.gnu.org/software/libc/manual/html_node/Backtraces.html
91 void* buffer[1024];
92 auto bt_size = backtrace(buffer, 1024);
93 char** bt_symb = backtrace_symbols(buffer, bt_size);
94 if((bt_size > 0) && (bt_symb != nullptr))
95 {
96 fprintf(stderr, "\nCall-Stack Back-Trace:\n");
97 fprintf(stderr, "----------------------\n");
98 for(decltype(bt_size) i(0); i < bt_size; ++i)
99 fprintf(stderr, "%s\n", bt_symb[i]);
100 fflush(stderr);
101 }
102#endif
103#elif defined(_WIN32)
104 Windows::dump_call_stack_to_stderr();
105#endif
106#ifdef FEAT_HAVE_MPI
107 ::MPI_Abort(MPI_COMM_WORLD, 1);
108#endif
109
110 std::abort();
111 }
112 else
113 {
114#ifdef FEAT_HAVE_MPI
115 ::MPI_Abort(MPI_COMM_WORLD, 1);
116#endif
117 std::abort();
118 }
119}

121int Runtime::finalize()
122{
123 if (!_initialized)
124 {
125 std::cerr << "ERROR: Runtime::finalize called before Runtime::initialize!" << std::endl;
126 std::cerr.flush();
127 Runtime::abort();
128 }
129 if (_finalized)
130 {
131 std::cerr << "ERROR: Runtime::finalize called twice!" << std::endl;
132 std::cerr.flush();
133 Runtime::abort();
134 }
135
136 MemoryPool::finalize();
137#ifdef FEAT_HAVE_CUDA
138 Util::cuda_finalize();
139#endif
140
141 // finalize Dist operations
142 Dist::finalize();
143
144 _finalized = true;
145
146 // return successful exit code
147 return EXIT_SUCCESS;
148}
