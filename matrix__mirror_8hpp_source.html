148 void gather(
149 LAFEM::MatrixMirrorBuffer<DT_, IT_>& buffer,
150 const LAFEM::SparseMatrixCSR<DT_, IT_>& matrix) const
151 {
152 XASSERT(buffer.entries_per_nonzero() == Index(1));
153 XASSERT(buffer.rows() == this->_row_mirror.num_indices());
154 XASSERT(buffer.columns() == this->_col_mirror.num_indices());
155 XASSERT(matrix.rows() == this->_row_mirror.size());
156 XASSERT(matrix.columns() == this->_col_mirror.size());
157
158 // fetch system matrix arrays
159 const IT_* row_ptr_a(matrix.row_ptr());
160 const IT_* col_idx_a(matrix.col_ind());
161 const DT_* val_a(matrix.val());
162
163 // fetch buffer arrays
164 const IT_* row_ptr_b(buffer.row_ptr());
165 const IT_* col_idx_b(buffer.col_ind());
166 DT_* val_b(buffer.val());
167
168 // fetch row/column mirror indices
169 const IT_* mir_idx_r(this->_row_mirror.indices());
170 const IT_* mir_idx_c(this->_col_mirror.indices());
171
172 // loop over all buffer matrix rows
173 for(IT_ i(0); i < IT_(buffer.rows()); ++i)
174 {
175 // get the row-index
176 const IT_ ridx = mir_idx_r[i];
177
178 // loop over all non-zeros in current row
179 for(IT_ j(row_ptr_b[i]); j < row_ptr_b[i+1]; ++j)
180 {
181 // format buffer entry
182 val_b[j] = DT_(0);
183
184 // get the column index
185 const IT_ cidx = mir_idx_c[col_idx_b[j]];
186
187 // try to find this entry in the input matrix
188 for(IT_ k(row_ptr_a[ridx]); k < row_ptr_a[ridx+1]; ++k)
189 {
190 if(col_idx_a[k] == cidx)
191 {
192 val_b[j] = val_a[k];
193 break;
194 }
195 }
196 }
197 }

618 protected:
619 template<typename MT_>
620 Adjacency::Graph _create_buffer_graph(const MT_& tmpl_mat) const
621 {
622 // compute the sparsity pattern of
623 // X := A * Y * B^T,
624 // where:
625 // X is the buffer matrix
626 // Y is the system matrix
627 // A is the row-mirror gather matrix
628 // B is the col-mirror gather matrix
629
630 // render the matrix structure to a graph to obtain
631 // the celebrated ptr/idx array pair
632 Adjacency::Graph mat_graph(Adjacency::RenderType::injectify, tmpl_mat);
633 const Index* dom_ptr = mat_graph.get_domain_ptr();
634 const Index* img_idx = mat_graph.get_image_idx();
635
636 // get row/column mirror indices
637 const Index nrows = _row_mirror.num_indices();
638 const Index ncols = _col_mirror.num_indices();
639 const auto* cidx = _col_mirror.indices();
640 const auto* ridx = _row_mirror.indices();
641
642 // build map of column mirror indices
643 std::map<Index,Index> col_map;
644 for(Index i(0); i < ncols; ++i)
645 col_map.emplace(cidx[i], i);
646
647 // count number of non-zeros in matrix buffer
648 Index count(0);
649 for(Index i(0); i < nrows; ++i)
650 {
651 const Index irow = ridx[i];
652 for(Index j(dom_ptr[irow]); j < dom_ptr[irow+1]; ++j)
653 {
654 if(col_map.find(img_idx[j]) != col_map.end())
655 ++count;
656 }
657 }
658
659 // allocate output graph
660 Adjacency::Graph graph(nrows, ncols, count);
661 Index* row_ptr = graph.get_domain_ptr();
662 Index* col_idx = graph.get_image_idx();
663
664 // compute output graph
665 row_ptr[0] = Index(0);
666 for(Index i(0); i < nrows; ++i)
667 {
668 const Index irow = ridx[i];
669 Index k = row_ptr[i];
670 for(Index j(dom_ptr[irow]); j < dom_ptr[irow+1]; ++j)
671 {
672 auto it = col_map.find(img_idx[j]);
673 if(it != col_map.end())
674 {
675 col_idx[k] = it->second;
676 ++k;
677 }
678 }
679 row_ptr[i+1] = k;
680 }
681 XASSERT(row_ptr[nrows] == count);
682
683 // sort column indices
684 graph.sort_indices();
685
686 return graph;
687 }
