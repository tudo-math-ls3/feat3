339 static MeshPartType meshpart_difference(const MeshPartType& left, const MeshPartType& right)
340 {
341 // NOTE(mmuegge): Differences between meshparts with topology are technically supported.
342 // They currently behave exactly as defined, which means they do not include any boundary entities
343 // along the common border of the left and right mesh parts.
344 // The implementation puts ~Index(0) values into the meshpart topology for these entities.
345 // That is the proper operation, but not very useful in the context of meshparts
346 // and quite surprising.
347 // Hence the assertion below.
348 // We could include these boundary entities in the difference.
349 // To do so we would need to either determine how many additional entities
350 // we need to add, so that we can size the meshpart topology appropriately,
351 // or create a dynamic data structure for topologies.
352 // Both of these solution are quite a lot of work, so we leave out differences between
353 // meshparts with topology for now.
354 XASSERTM(
355 !left.has_topology() && !right.has_topology(),
356 "Difference between meshparts with topology are not supported.");
357
358 const auto merge = [](AttributeDataType& a, AttributeDataType& /*b*/) { return a; };
359 const auto join_pred = [](const JoinEntry& e) { return e.left.has_value() && !e.right.has_value(); };
360 return meshpart_op(left, right, merge, join_pred);
361 }

394 static MeshPartType meshpart_symmetric_difference(const MeshPartType& left, const MeshPartType& right)
395 {
396 // Identity merge, because no common entities end up in result
397 const auto merge = [](AttributeDataType& a, AttributeDataType& /*b*/) { return a; };
398 MeshPartType a = meshpart_difference(left, right, merge);
399 MeshPartType b = meshpart_difference(right, left, merge); // NOLINT
400 return meshpart_union(a, b, merge);
401 }

421 template<typename AttributeFn_, typename PredFn_>
422 static MeshPartType meshpart_op(
423 const MeshPartType& left,
424 const MeshPartType& right,
425 const AttributeFn_& attribute_merge,
426 const PredFn_& join_pred)
427 {
428 // Determine join of mesh parts
429 // The join tracks which indices belong to the same entity on the left, right, and result meshparts.
430 Join join;
431 join_elements(join, left.get_target_set_holder(), right.get_target_set_holder());
432
433 // The join predicate determines which elements get added to the target set holder
434 TargetSetHolderType tsh;
435 build_target_set_holder(tsh, join, join_pred, left.get_target_set_holder(), right.get_target_set_holder());
436
437 // Determine result size from target set
438 std::array<Index, ShapeType::dimension + 1> size{};
439 for(int i(0); i < ShapeType::dimension + 1; i++)
440 {
441 size[i] = tsh.get_num_entities(i);
442 }
443
444 // The result meshpart has a topology as long as any of the inputs have a topology
445 const bool create_topology = left.has_topology() || right.has_topology();
446
447 // Create result mesh part
448 MeshPartType result(size.data(), create_topology);
449 result.get_target_set_holder() = std::move(tsh);
450
451 if(create_topology)
452 {
453 // Empty dummy IndexSetHolder. Allows us to handle all configurations of has/has-no topology the same
454 IndexSetHolderType empty;
455 const IndexSetHolderType* left_ish = left.has_topology() ? left.get_topology() : &empty;
456 const IndexSetHolderType* right_ish = right.has_topology() ? right.get_topology() : &empty;
457
458 build_topology(
459 *result.get_topology(),
460 join,
461 *left_ish,
462 left.get_target_set_holder(),
463 *right_ish,
464 right.get_target_set_holder());
465 }
466
467 // Handle attributes
468 copy_attributes(attribute_merge, join, result, left, right);
469
470 // Return intersection mesh part
471 return std::move(result);
472 }

512 template<typename AttributeFn_>
513 static void copy_attributes(
514 const AttributeFn_& fn,
515 const Join& join,
516 MeshPartType& result,
517 const MeshPartType& left,
518 const MeshPartType& right)
519 {
520 // Determine attributes that exist on both the left and right meshpart
521 std::set<String> attributes = common_attributes(left, right);
522
523 // Copy common attributes to result meshpart
524 for(const String& name : attributes)
525 {
526 const AttributeSetType* left_attribute = left.find_attribute(name);
527 const AttributeSetType* right_attribute = right.find_attribute(name);
528
529 const int dim = left_attribute->get_dimension();
530
531 auto new_attribute = std::make_unique<AttributeSetType>(result.get_num_entities(0), dim);
532
533 const JoinVec& join_vec = join.template get<0>();
534
535 for(const JoinEntry& t : join_vec)
536 {
537 if(!t.index.has_value())
538 {
539 // Entity is not in result meshpart. Skip it.
540 continue;
541 }
542
543 if(t.left.has_value() && t.right.has_value())
544 {
545 // Entity exists on both meshpart. Merge the attribute values
546 for(int i(0); i < dim; i++)
547 {
548 AttributeDataType l = left_attribute->operator()(t.left.value(), i);
549 AttributeDataType r = right_attribute->operator()(t.right.value(), i);
550 new_attribute->operator()(t.index.value(), i) = fn(l, r);
551 }
552 }
553 else if(t.left.has_value())
554 {
555 // Entity exists only on left meshpart. Copy the attribute value
556 for(int i(0); i < dim; i++)
557 {
558 AttributeDataType l = left_attribute->operator()(t.left.value(), i);
559 new_attribute->operator()(t.index.value(), i) = l;
560 }
561 }
562 else if(t.right.has_value())
563 {
564 // Entity exists only on right meshpart. Copy the attribute value
565 for(int i(0); i < dim; i++)
566 {
567 AttributeDataType r = right_attribute->operator()(t.right.value(), i);
568 new_attribute->operator()(t.index.value(), i) = r;
569 }
570 }
571 else
572 {
573 XABORTM("Entity in meshpart operation result without source index");
574 }
575 }
576 result.add_attribute(std::move(new_attribute), name);
577 }

596 static void build_topology(
597 IndexSetHolderType& output_ish,
598 Join& join,

632 template<int dim_, int codim_>
633 static void build_topology_inner(
634 IndexSetHolderType& output_ish,
635 Join& join,
636 const IndexSetHolderType& left_ish,
637 const TargetSetHolderType& left_tsh,
638 const IndexSetHolderType& right_ish,
639 const TargetSetHolderType& right_tsh)
640 {
641 if constexpr(codim_ < 0)
642 {
643 // There are no facets with dimension lower than 0. No more work to be done.
644 return;
645 }
646 else
647 {
648 // Recursive call to sub entities
649 build_topology_inner<dim_, codim_ - 1>(output_ish, join, left_ish, left_tsh, right_ish, right_tsh);
650
651 using DimShape = typename Shape::FaceTraits<ShapeType, dim_>::ShapeType;
652 constexpr int num_indices = Shape::FaceTraits<DimShape, codim_>::count;
653 using IndexSetType = IndexSet<num_indices>;
654
655 JoinVec& dim_entities = join.template get<dim_>();
656 JoinVec& codim_entities = join.template get<codim_>();
657
658 const IndexSetType& right_is = right_ish.template get_index_set<dim_, codim_>();
659 const IndexSetType& left_is = left_ish.template get_index_set<dim_, codim_>();
660
661 IndexSetType& out = output_ish.template get_index_set<dim_, codim_>();
662
663 // NOTE(mmuegge): It is tempting to turn the JoinVecs into hashmaps,
664 // but we first search for right-indices and then left-indicies.
665 // This means we need to duplicate information two maps with different keys.
666 // So this might actually be more elegant.
667
669 // Right Search
671
672 // In the following, we need to search for entities in the codim_entities vector.
673 // To do so efficiently we sort it beforehand, to allow for binary searches later
674 std::sort(
675 codim_entities.begin(),
676 codim_entities.end(),
677 [](const JoinEntry& a, const JoinEntry& b) { return a.right < b.right; });
678
679 // We first search for topology in the entities of the right meshpart.
680 // If, for any entitiy in the result meshpart, we do not find the topology there,
681 // we need additionally search in the left meshpart.
682 // Because we need to re-sort the codim vector, we do these steps one after the other
683 // and use this bool to indicate whether we need to do the left-search at all.
684 bool left_entities = false;
685
686 // Determine topology for all entities that are part of the result
687 // mesh part and stem from the right mesh part
688 for(const JoinEntry& t : dim_entities)
689 {
690 if(!t.index.has_value())
691 {
692 // This entry is not part of the result mesh part. Skip it.
693 continue;
694 }
695
696 if(!t.right.has_value())
697 {
698 // This entry does not stem from the right mesh part. Skip it.
699
700 // This also means we need to handle entities stemming from the left
701 // mesh part after this
702 left_entities = true;
703 continue;
704 }
705
706 for(int j(0); j < num_indices; j++)
707 {
708 // Search for join entry of j-th sub-entity.
709
710 // NOTE(mmuegge): We are searching among the "std::optional<Index> right"
711 // members of the join entries. We thus wrap our target value itself into
712 // an optional before starting the search
713
714 const std::optional<Index> target(right_is[t.right.value()][j]);
715 std::optional<Index> codim_index = search(
716 codim_entities.begin(),
717 codim_entities.end(),
718 target,
719 [](const JoinEntry& a, const std::optional<Index>& i) { return a.right < i; });
720 XASSERT(codim_index.has_value());
721
722 const JoinEntry& codim_entry = codim_entities[codim_index.value()];
723
724 // NOTE(mmuegge): codim_entry.index is only empty when trying to produce differences on meshparts.
725 // In that case we could be searching for a topology element that has been removed as part of the
726 // shared boundary between left and right.
727 // That case is forbidden via an assertion above, but is safely handled here anyway.
728 out(t.index.value(), j) = codim_entry.index.value_or(~Index(0));
729 }
730 }
731
732 if(!left_entities)
733 {
734 return;
735 }
736
738 // Left Search
740
741 // Prepare join for searching entities of right mesh part
742 std::sort(
743 codim_entities.begin(),
744 codim_entities.end(),
745 [](const JoinEntry& a, const JoinEntry& b) { return a.left < b.left; });
746
747 for(const JoinEntry& t : dim_entities)
748 {
749 if(!t.index.has_value() || t.right.has_value() || !t.left.has_value())
750 {
751 // We only handle entries that are part of the result,
752 // part of the left meshpart,
753 // and have not been handled by the right meshpart case above
754 continue;
755 }
756
757 for(int j(0); j < num_indices; j++)
758 {
759 const std::optional<Index> target(left_is[t.left.value()][j]);
760 std::optional<Index> codim_index = search(
761 codim_entities.begin(),
762 codim_entities.end(),
763 target,
764 [](const JoinEntry& a, const std::optional<Index>& i) { return a.left < i; });
765 XASSERT(codim_index.has_value());
766
767 const JoinEntry& codim_entry = codim_entities[codim_index.value()];
768 out(t.index.value(), j) = codim_entry.index.value_or(~Index(0));
769 }
770 }
771 }

790 template<int dim_ = ShapeType::dimension>
791 static void build_target_set_holder(
792 TargetSetHolderType& output_tsh,
793 Join& join,
794 const std::function<bool(const JoinEntry&)>& pred,
795 const TargetSetHolderType& left_tsh,
796 const TargetSetHolderType& right_tsh)
797 {
798 if constexpr(dim_ < 0)
799 {
800 // No more work to be done
801 return;
802 }
803 else
804 {
805 // Recurse to lower dimension
806 build_target_set_holder<dim_ - 1>(output_tsh, join, pred, left_tsh, right_tsh);
807
808 JoinVec& entities = join.template get<dim_>();
809 const TargetSet& right_ts = right_tsh.template get_target_set<dim_>();
810 const TargetSet& left_ts = left_tsh.template get_target_set<dim_>();
811
812 std::vector<Index> target_set_vec;
813 for(JoinEntry& t : entities)
814 {
815 if(pred(t))
816 {
817 // Entity should be added; assign it an index
818 t.index = target_set_vec.size();
819 if(t.left)
820 {
821 // Pull target set information from left target set
822 target_set_vec.push_back(left_ts[t.left.value()]);
823 }
824 else if(t.right)
825 {
826 // Pull target set information from right target set
827 target_set_vec.push_back(right_ts[t.right.value()]);
828 }
829 else
830 {
831 // NOTE(mmuegge): This seems at first glance like the starting place
832 // for a complement operation. But I think we need to special case that,
833 // because we would need to grab any entitity not in one of the meshparts
834 // here and keep track of which entities we have already added.
835 XABORTM("No source index for entity!");
836 }
837 }
838 }
839
840 // Copy to actual target set, now that we know the final size.
841 TargetSet result(target_set_vec.size());
842 for(Index i(0); i < target_set_vec.size(); i++)
843 {
844 result[i] = target_set_vec[i];
845 }
846
847 // Move into output target set holder
848 output_tsh.template get_target_set<dim_>() = std::move(result);
849 }
850 }

867 template<int dim_ = ShapeType::dimension>
868 static void join_elements(Join& join, const TargetSetHolderType& left_tsh, const TargetSetHolderType& right_tsh)
869 {
870 if constexpr(dim_ < 0)
871 {
872 return;
873 }
874 else
875 {
876 // Recurse to lower dimension
877 join_elements<dim_ - 1>(join, left_tsh, right_tsh);
878
879 JoinVec& join_vec = join.template get<dim_>();
880
881 const TargetSet& left = left_tsh.template get_target_set<dim_>();
882 const TargetSet& right = right_tsh.template get_target_set<dim_>();
883
885 // Prepare lookup vector for left mesh part
887
888 // Create a copy of the target set we can safely modify
889 TargetSet left_lookup = left.clone();
890
891 Index* lbegin = left_lookup.get_indices();
892 Index* lend = left_lookup.get_indices() + left_lookup.get_num_entities();
893
894 // The target set will be sorted to enable binary searches.
895 // We need a way to retrieve the original indices later.
896 // We thus create this vector of indices and sort it the same way
897 // we sort the target set.
898 std::vector<Index> left_indices(left_lookup.get_num_entities());
899 for(Index i(0); i < left_lookup.get_num_entities(); i++)
900 {
901 left_indices[i] = i;
902 }
903
904 std::sort(
905 left_indices.begin(),
906 left_indices.end(),
907 [&](Index a, Index b) { return left_lookup[a] < left_lookup[b]; });
908 std::sort(lbegin, lend);
909
911 // Find entities exclusive to right mesh part and entities in intersection
913
914 for(Index i(0); i < right.get_num_entities(); i++)
915 {
916 JoinEntry entry;
917 entry.right = i;
918
919 const std::optional<Index> sorted_left = search(lbegin, lend, right[i]);
920 if(sorted_left)
921 {
922 entry.left = left_indices[sorted_left.value()];
923 }
924
925 join_vec.push_back(entry);
926 }
927
928 // OPTIMIZATION(mmuegge): At this point we have sufficient information to
929 // determine the intersection of two mesh parts. We calculate the full join
930 // only for other mesh part operations. If mesh part intersection ever
931 // becomes a performance bottleneck, we can calculate these joins more fine grained.
932
934 // Prepare lookup vector for left mesh part
936
937 // Create a copy of the target set we can safely modify
938 TargetSet right_lookup = right.clone();
939
940 Index* rbegin = right_lookup.get_indices();
941 Index* rend = right_lookup.get_indices() + right_lookup.get_num_entities();
942
943 // The target set will be sorted to enable binary searches.
944 // We need a way to retrieve the original indices later.
945 // We thus create this vector of indices and sort it the same way
946 // we sort the target set.
947 std::vector<Index> right_indices(right_lookup.get_num_entities());
948 for(Index i(0); i < right_lookup.get_num_entities(); i++)
949 {
950 right_indices[i] = i;
951 }
952
953 std::sort(
954 right_indices.begin(),
955 right_indices.end(),
956 [&](Index a, Index b) { return right_lookup[a] < right_lookup[b]; });
957 std::sort(rbegin, rend);
958
960 // Find entities exclusive to left mesh part
962
963 for(Index i(0); i < left.get_num_entities(); i++)
964 {
965 const std::optional<Index> sorted_right = search(rbegin, rend, left[i]);
966 if(!sorted_right.has_value())
967 {
968 JoinEntry entry;
969 entry.left = i;
970 join_vec.push_back(entry);
971 }
972 }
973 }
974 }

988 template<typename Iter_, typename T_>
989 static std::optional<Index> search(Iter_ begin, Iter_ end, const T_& value)
990 {
991 Iter_ it = std::lower_bound(begin, end, value);
992 if(it == end || *it != value)
993 {
994 return std::nullopt;
995 }
996 else
997 {
998 return std::distance(begin, it);
999 }
1000 }

1016 template<typename Iter_, typename T_, typename CompFn_>
1017 static std::optional<Index> search(Iter_ begin, Iter_ end, const T_& value, const CompFn_& fn)
1018 {
1019 Iter_ it = std::lower_bound(begin, end, value, fn);
1020 if(it == end || fn(*it, value))
1021 {
1022 return std::nullopt;
1023 }
1024 else
1025 {
1026 return std::distance(begin, it);
