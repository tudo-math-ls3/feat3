419 template<typename TrafoData_, typename Evaluator_>
420 static void calc_hess_inv(TrafoData_& trafo_data, const Evaluator_&)
421 {
422 if(!*(Evaluator_::eval_caps & TrafoTags::hess_inv))
423 XABORTM("trafo evaluator can't compute inverse hessian tensors");
424
425 typedef typename TrafoData_::EvalTraits EvalTraits;
426 typedef typename EvalTraits::DataType DataType;
427 typename EvalTraits::HessianTensorType hess_jac;
428
429 // compute inverse
430 hess_jac.format();
431 hess_jac.add_double_mat_mult(trafo_data.hess_ten, trafo_data.jac_inv, trafo_data.jac_inv);
432 trafo_data.hess_inv.format();
433 trafo_data.hess_inv.add_mat_tensor_mult(trafo_data.jac_inv, hess_jac, -DataType(1));
434 }
