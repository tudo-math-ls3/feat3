309 explicit HyperelasticityFunctional(
310 Geometry::RootMeshNode<MeshType>* rmn_,
311 TrafoType& trafo,
312 const std::deque<String>& dirichlet_list,
313 const std::deque<String>& slip_list,
314 std::shared_ptr<CellFunctionalType_> cell_functional_)
315 : BaseClass(rmn_),
316 _cell_functional(cell_functional_),
317 _trafo(trafo),
318 _mu(rmn_->get_mesh()->get_num_entities(ShapeType::dimension), DataType(1)),
319 _lambda(rmn_->get_mesh()->get_num_entities(ShapeType::dimension)),
320 _h(rmn_->get_mesh()->get_num_entities(ShapeType::dimension)),
321 _dirichlet_asm(),
322 _slip_asm(),
323 _mesh_conc(nullptr),
324 trafo_space(trafo),

326 sync_vecs(),
327 _columns(trafo_space.get_num_dofs()),
328 _rows(trafo_space.get_num_dofs()),
329 _scale_computation(ScaleComputation::once_uniform),
330 _penalty_param(0),
331 _alignment_constraint(0),
332 _sum_det(0),
333 _sum_lambda(0),
334 _sum_mu(0)
335 {
336
337 XASSERTM(cell_functional_ != nullptr, "cell functional must not be nullptr");
338
339 // For every MeshPart specified in the list of Dirichlet boundaries, create a UnitFilterAssembler and
340 // insert it into the std::map
341 for(auto& it : dirichlet_list)
342 {
343 // Create empty assembler
344 auto new_asm = std::make_shared<Assembly::UnitFilterAssembler<MeshType>>();
345
346 // Add the MeshPart to the assembler if it is there. There are legitimate reasons for it NOT to be
347 // there, i.e. we are in parallel and our patch is not adjacent to that MeshPart
348 auto* mpp = rmn_->find_mesh_part(it);
349 if(mpp != nullptr)
350 {
351 new_asm->add_mesh_part(*mpp);
352 }
353
354 // Insert into the map
355 String identifier(it);
356 _dirichlet_asm.emplace(identifier, new_asm);
357 }
358
359 // For every MeshPart specified in the list of slip boundaries, create a SlipFilterAssembler and
360 // insert it into the std::map
361 for(auto& it : slip_list)
362 {
363 // Create empty assembler
364 auto new_asm = std::make_shared<Assembly::SlipFilterAssembler<TrafoType>>(this->_trafo);
365
366 // Add the MeshPart to the assembler if it is there. There are legitimate reasons for it NOT to be
367 // there, i.e. we are in parallel and our patch is not adjacent to that MeshPart
368 auto* mpp = rmn_->find_mesh_part(it);
369 if(mpp != nullptr)
370 {
371 new_asm->add_mesh_part(*mpp);
372 }
373
374 // Insert into the map
375 String identifier(it);
376 _slip_asm.emplace(identifier, new_asm);
377 }
378
379 for(Index i(0); i < _mu.size(); ++i)
380 {
381 _sum_mu += _mu(i);
382 }
383
384 sync_scalars.emplace("_sum_mu",&_sum_mu);
385
386 // Compute desired element size distribution
387 _compute_scales_once();
388 }

606 virtual void add_to_vtk_exporter(Geometry::ExportVTK<typename Trafo_::MeshType>& exporter) const override
607 {
608 exporter.add_cell_scalar("h", this->_h.elements());
609 exporter.add_cell_scalar("lambda", this->_lambda.elements());
610 exporter.add_cell_scalar("mu", this->_mu.elements());
611
612 DataType* fval_norm(new DataType[this->get_mesh()->get_num_entities(MeshType::shape_dim)]);
613 DataType* fval_det(new DataType[this->get_mesh()->get_num_entities(MeshType::shape_dim)]);
614 DataType* fval_rec_det(new DataType[this->get_mesh()->get_num_entities(MeshType::shape_dim)]);
615
616 DataType fval(0);
617 eval_fval_cellwise(fval, fval_norm, fval_det, fval_rec_det);
618 exporter.add_cell_vector("fval", fval_norm, fval_det, fval_rec_det);
619
620 if(_mesh_conc != nullptr)
621 {
622 exporter.add_vertex_scalar("dist", _mesh_conc->get_dist().elements());
623 exporter.add_vertex_vector("grad_dist", _mesh_conc->get_grad_dist());
624
625 if(this->_penalty_param > DataType(0))
626 {
627 DataType* constraint_at_cell = new DataType[this->get_mesh()->get_num_entities(MeshType::shape_dim)];
628
629 this->_mesh_conc->compute_constraint(constraint_at_cell);
630 exporter.add_cell_scalar("alignment_constraint", constraint_at_cell);
631
632 delete[] constraint_at_cell;
633 }
634 }
635
636 delete [] fval_norm;
637 delete [] fval_det;
638 delete [] fval_rec_det;
639

711 virtual void init_pre_sync()
712 {
713 // Write any potential changes to the mesh
714 this->buffer_to_mesh();
715
716 // Compute distances if necessary
717 if((this->_scale_computation == ScaleComputation::current_concentration) ||
718 (this->_scale_computation == ScaleComputation::iter_concentration) ||
719 (_penalty_param > DataType(0)) )
720 {
721 if(_mesh_conc == nullptr)
722 XABORTM("Scale computation set to "+stringify(_scale_computation)+"and alignment penalty factor to "+
723 stringify_fp_sci(_penalty_param)+", but no concentration function was given");
724
725 _mesh_conc->compute_dist();
726 }
727
728 // Compute desired element size distribution
729 this->_compute_scales_init();
730 }
731
737 virtual void init_post_sync()
738 {
739 // Scale mu so that || mu ||_1 = 1
740 // For this, _sum_mu needs to have been summed up over all patches in the synchronization phase.
741 if(_sum_mu != DataType(1))
742 {
743 _mu.scale(_mu, DataType(1)/_sum_mu);
744 _sum_mu = DataType(1);
745 }
746
747 // Scale lambda so that || lambda ||_1 = 1
748 // For this, _sum_lambda needs to have been summed up over all patches in the synchronization phase.
749 if(_sum_lambda != DataType(1))
750 {
751 _lambda.scale(_lambda, DataType(1)/_sum_lambda);
752 _sum_lambda = DataType(1);
753 }
754
755 // Compute the new optimal scales. For this, lambda needs to be scaled correctly and _sum_det needs to be
756 // summed up over all patches in the synchronization phase.
757 RefCellTrafo_::compute_h(_h, _lambda, _sum_det);
758 }

790 virtual void prepare_pre_sync(const VectorTypeR& vec_state, FilterType& filter)
791 {
792 // Download state if necessary
793 //CoordsBufferType vec_buf;
794 //vec_buf.convert(vec_state);
795
796 // Copy to buffer and mesh
797 this->_coords_buffer.copy(vec_state);
798 this->buffer_to_mesh();
799
800 auto& dirichlet_filters = filter.template at<1>();
801
802 for(auto& it : dirichlet_filters)
803 {
804 const auto& assembler = _dirichlet_asm.find(it.first);
805
806 if(assembler == _dirichlet_asm.end())
807 XABORTM("Could not find unit filter assembler for filter with key "+it.first);
808
809 assembler->second->assemble(it.second, trafo_space, vec_state);
810 }
811
812 // The slip filter contains the outer unit normal, so reassemble it
813 auto& slip_filters = filter.template at<0>();
814 for(const auto& it:slip_filters)
815 this->_mesh_node->adapt_by_name(it.first);
816
817 // Copy back to the buffer as adaption might have changed the mesh
818 this->mesh_to_buffer();
819
820 for(auto& it : slip_filters)
821 {
822 const auto& assembler = _slip_asm.find(it.first);
823
824 if(assembler == _slip_asm.end())
825 {
826 XABORTM("Could not find slip filter assembler for filter with key "+it.first);
827 }
828
829 assembler->second->assemble(it.second, trafo_space);
830 }
831
832 if( (this->_scale_computation == ScaleComputation::iter_concentration) ||
833 (_penalty_param > DataType(0)))
834 {
835 if(_mesh_conc == nullptr)
836 {
837 XABORTM("Scale computation set to "+stringify(_scale_computation)+"and alignment penalty factor to "+
838 stringify_fp_sci(_penalty_param)+", but no concentration function was given");
839 }
840
841 _mesh_conc->compute_dist();
842 }
843
844 _compute_scales_iter();
845
846 if(_penalty_param > DataType(0))
847 {
848 _alignment_constraint = this->_mesh_conc->compute_constraint();
849 this->sync_scalars.emplace("_alignment_constraint",&_alignment_constraint);
850 }
851
852 }

948 void compute_cell_size_defect_pre_sync(CoordType& vol_min, CoordType& vol_max, CoordType& vol) const
949 {
950 vol = CoordType(0);
951
952 vol_min = Math::huge<CoordType>();
953 vol_max = CoordType(0);
954
955 for(Index cell(0); cell < this->get_mesh()->get_num_entities(ShapeType::dimension); ++cell)
956 {
957 CoordType my_vol = this->_trafo.template compute_vol<ShapeType, CoordType>(cell);
958 vol_min = Math::min(vol_min, my_vol);
959 vol_max = Math::max(vol_min, my_vol);
960 vol += my_vol;
961 }
962 }

991 virtual CoordType compute_cell_size_defect_post_sync(CoordType& lambda_min, CoordType& lambda_max,
992 CoordType& vol_min, CoordType& vol_max, const CoordType& vol) const
993 {
994 CoordType size_defect(0);
995 lambda_min = Math::huge<CoordType>();
996 lambda_max = CoordType(0);
997
998 for(Index cell(0); cell < this->get_mesh()->get_num_entities(ShapeType::dimension); ++cell)
999 {
1000 size_defect += Math::abs(this->_trafo.template
1001 compute_vol<ShapeType, CoordType>(cell)/vol - this->_lambda(cell));
1002
1003 lambda_min = Math::min(lambda_min, this->_lambda(cell));
1004 lambda_max = Math::max(lambda_max, this->_lambda(cell));
1005 }
1006
1007 vol_min /= vol;
1008 vol_max/= vol;
1009
1010 return size_defect;
1011 }

1030 virtual void eval_fval_grad(CoordType& fval, VectorTypeL& grad, const bool& add_penalty_fval = true)
1031 {
1032 typedef typename TrafoType::template Evaluator<ShapeType, DataType>::Type TrafoEvaluator;
1033 typedef typename SpaceType::template Evaluator<TrafoEvaluator>::Type SpaceEvaluator;
1034
1035 // Increase number of functional evaluations
1036 this->_num_func_evals++;
1037 this->_num_grad_evals++;
1038
1039 // Total number of cells in the mesh
1040 Index ncells(this->get_mesh()->get_num_entities(ShapeType::dimension));
1041
1042 // Index set for local/global numbering
1043 auto& idx = this->get_mesh()->template get_index_set<ShapeType::dimension,0>();
1044
1045 // This will hold the coordinates for one element for passing to other routines
1046 FEAT::Tiny::Matrix <CoordType, Shape::FaceTraits<ShapeType,0>::count, MeshType::world_dim> x;
1047 // This will hold the local gradient for one element for passing to other routines
1048 FEAT::Tiny::Matrix<CoordType, Shape::FaceTraits<ShapeType,0>::count, MeshType::world_dim> grad_loc;
1049
1050 // Clear gradient vector
1051 grad.format();
1052 fval = DataType(0);
1053
1054 TrafoEvaluator trafo_eval(this->_trafo);
1055 SpaceEvaluator space_eval(this->trafo_space);
1056
1057 // Compute the functional value for each cell
1058 for(Index cell(0); cell < ncells; ++cell)
1059 {
1060 DataType fval_loc(0);
1061 trafo_eval.prepare(cell);
1062 space_eval.prepare(trafo_eval);
1063
1064 // Get local coordinates
1065 for(int j(0); j < Shape::FaceTraits<ShapeType,0>::count; ++j)
1066 {
1067 x[j] = this->_coords_buffer(idx(cell,j));
1068 }
1069
1070 auto mat_tensor = RefCellTrafo_::compute_mat_tensor(x, this->_h(cell));
1071
1072 this->_cell_functional->eval_fval_grad(
1073 fval_loc, grad_loc, mat_tensor, trafo_eval, space_eval, x, this->_h(cell));
1074
1075 // Add the contribution from the dependence of h on the vertex coordinates
1076 if(this->_mesh_conc != nullptr && this->_mesh_conc->use_derivative())
1077 {
1078 const auto& grad_h = this->_mesh_conc->get_grad_h();
1079 this->_cell_functional->add_grad_h_part(
1080 grad_loc, mat_tensor, trafo_eval, space_eval, x, this->_h(cell), grad_h(cell));
1081 }
1082
1083 fval += this->_mu(cell)*fval_loc;
1084 // Add local contributions to global gradient vector
1085 for(int j(0); j < Shape::FaceTraits<ShapeType,0>::count; ++j)
1086 {
1087 Index i(idx(cell,j));
1088 Tiny::Vector<CoordType, MeshType::world_dim, MeshType::world_dim> tmp(grad(i));
1089 tmp += this->_mu(cell)*grad_loc[j];
1090
1091 grad(i,tmp);
1092 }
1093 }
1094
1095 if(this->_penalty_param > DataType(0))
1096 {
1097 XASSERTM(this->_mesh_conc != nullptr,
1098 "You need a mesh concentration function for imposing a mesh alignment constraint!");
1099
1100 if(add_penalty_fval)
1101 {
1102 // Add the quadratic penalty term to the functional value
1103 fval += this->_penalty_param*DataType(0.5)*Math::sqr(this->_alignment_constraint);
1104 }
1105
1106 // Add the gradient of the penalty term
1107 this->_mesh_conc->add_constraint_grad(grad, this->_alignment_constraint, this->_penalty_param);
1108
1109 }
