91 static std::shared_ptr <Control::Meshopt::MeshoptControlBase<DomCtrl_>>
92 create_dudv_control(DomCtrl_& dom_ctrl, const String& section_key, PropertyMap* meshopt_config, PropertyMap* solver_config)
93 {
94 std::shared_ptr<Control::Meshopt::MeshoptControlBase<DomCtrl_>> result(nullptr);
95
96 // -1 causes the DuDvFunctionalControl to use max level of the underlying domain control
97 int meshopt_lvl(-1);
98
99 // Get Meshopt configuration section
100 auto meshopt_section = meshopt_config->query_section(section_key);
101 XASSERTM(meshopt_section != nullptr, "Application config is missing the mandatory MeshOptimizer section!");
102
103 // Verify the type
104 auto type_p = meshopt_section->query("type");
105 XASSERTM(type_p.second, "MeshOptimizer section is missing the mandatory type!");
106 XASSERTM(type_p.first == "DuDv", "Invalid type string!");
107
108 // Get list of boundary conditions
109 auto dirichlet_list_p = meshopt_section->query("dirichlet_boundaries");
110 std::deque<String> dirichlet_list = dirichlet_list_p.first.split_by_whitespaces();
111
112 // Get list of boundary conditions
113 auto slip_list_p = meshopt_section->query("slip_boundaries");
114 std::deque<String> slip_list = slip_list_p.first.split_by_whitespaces();
115
116 // Get meshopt level (if any)
117 auto meshopt_lvl_p = meshopt_section->query("meshopt_lvl");
118 if(meshopt_lvl_p.second)
119 {
120 meshopt_lvl = std::stoi(meshopt_lvl_p.first);
121 }
122
123 auto config_section_p = meshopt_section->query("config_section");
124 XASSERTM(config_section_p.second, "MeshOptimizer config section is missing config_section entry!");
125
126 auto dudv_config_section = meshopt_config->query_section(config_section_p.first);
127 if(dudv_config_section == nullptr)
128 {
129 XABORTM("config_section "+config_section_p.first+" not found!");
130 }
131
132 auto solver_p = dudv_config_section->query("solver_config");
133 XASSERTM(solver_p.second, "DuDv config section is missing solver_config entry!");
134
135 bool fixed_reference_domain(false);
136 auto fixed_reference_domain_p = dudv_config_section->query("fixed_reference_domain");
137 if(fixed_reference_domain_p.second)
138 {
139 fixed_reference_domain = (std::stoi(fixed_reference_domain_p.first) == 1);
140 }
141
142 typedef Control::Meshopt::DuDvFunctionalControl<DT_, IT_, DomCtrl_> DuDvCtrl;
143 result = std::make_shared<DuDvCtrl>(
144 dom_ctrl, meshopt_lvl,
145 dirichlet_list, slip_list, solver_p.first, *solver_config, fixed_reference_domain);
146
147 return result;
148 } // create_dudv_control

344 template<typename DomCtrl_, typename CellFunctional_>
345 static std::shared_ptr <Control::Meshopt::MeshoptControlBase<DomCtrl_>>
346 create_hyperelasticity_control_with_cell_functional( DomCtrl_& dom_ctrl,
347 const int meshopt_lvl,
348 PropertyMap* hyperelasticity_config_section,
349 PropertyMap* meshopt_config,
350 PropertyMap* solver_config,
351 std::shared_ptr<CellFunctional_> my_functional,
352 const std::deque<String>& dirichlet_list, const std::deque<String>& slip_list)
353 {
354 typedef typename DomCtrl_::LevelType::TrafoType TrafoType;
355
356 std::shared_ptr<Control::Meshopt::MeshoptControlBase<DomCtrl_>> result(nullptr);
357
358 // Get the global mesh quality functional
359 auto global_functional_p = hyperelasticity_config_section->query("global_functional");
360 if(!global_functional_p.second)
361 {
362 XABORTM("Hyperelasticity config section is missing global_functional entry!");
363 }
364
365 // Get scale computation type, default is once_uniform
366 FEAT::Meshopt::ScaleComputation scale_computation(FEAT::Meshopt::ScaleComputation::once_uniform);
367
368 auto scale_computation_p = hyperelasticity_config_section->query("scale_computation");
369 if(scale_computation_p.second)
370 {
371 scale_computation << scale_computation_p.first;
372 }
373
374 int align_mesh(0);
375 auto align_mesh_p = hyperelasticity_config_section->query("align_mesh");
376 if(align_mesh_p.second)
377 {
378 align_mesh = std::stoi(align_mesh_p.first);
379 }
380
381 // Get the solver config section
382 auto solver_p = hyperelasticity_config_section->query("solver_config");
383 if(!solver_p.second)
384 {
385 XABORTM("Hyperelasticity config section is missing solver entry!");
386 }
387
388 if(global_functional_p.first == "HyperelasticityFunctional")
389 {
390 typedef typename FEAT::Meshopt::
391 HyperelasticityFunctional<DT_, IT_, TrafoType, CellFunctional_>::RefCellTrafo RefCellTrafo;
392
393 std::shared_ptr<FEAT::Meshopt::MeshConcentrationFunctionBase<TrafoType, RefCellTrafo>>
394 mesh_conc_func(nullptr);
395
396 if( scale_computation == FEAT::Meshopt::ScaleComputation::current_concentration ||
397 scale_computation == FEAT::Meshopt::ScaleComputation::iter_concentration ||
398 align_mesh == 1)
399 {
400 auto conc_func_section_p = hyperelasticity_config_section->query("conc_function");
401 XASSERTM(conc_func_section_p.second, "conc_function missing!");
402 mesh_conc_func = FEAT::Meshopt::MeshConcentrationFunctionFactory<TrafoType, RefCellTrafo>::
403 create(conc_func_section_p.first, meshopt_config);
404 }
405
406 result = std::make_shared<Control::Meshopt::HyperelasticityFunctionalControl
407 <DT_, IT_, DomCtrl_,
408 SetCellFunctional<FEAT::Meshopt::HyperelasticityFunctional, CellFunctional_>::template Functional>>
409 (dom_ctrl, meshopt_lvl, dirichlet_list, slip_list, solver_p.first,
410 *solver_config, my_functional, scale_computation, mesh_conc_func, DT_(align_mesh));
411 }
412 else
413 {
414 XABORTM("Unknown global_functional "+global_functional_p.first);
415 }
416
417 return result;
418 } // create_hyperelasticity_control_with_functional

440 template<typename DomCtrl_>
441 static std::shared_ptr<Control::Meshopt::MeshoptControlBase<DomCtrl_>>
442 create_meshopt_control(DomCtrl_& dom_ctrl, const String& section_key,
443 PropertyMap* meshopt_config, PropertyMap* solver_config)
444 {
445 std::shared_ptr<Control::Meshopt::MeshoptControlBase<DomCtrl_>> result(nullptr);
446
447 String type("");
448
449 auto meshopt_section = meshopt_config->query_section(section_key);
450 if(meshopt_section == nullptr)
451 {
452 XABORTM("Meshopt config is missing the ["+section_key+"] section!");
453 }
454 else
455 {
456 // Get mandatory quality functional entry
457 auto type_p = meshopt_section->query("type");
458 if(!type_p.second)
459 {
460 XABORTM("MeshOptimizer section is missing the mandatory type!");
461 }
462 else
463 {
464 type = type_p.first;
465 }
466 }
467
468 if(type == "DuDv")
469 {
470 result = create_dudv_control(dom_ctrl, section_key, meshopt_config, solver_config);
471 }
472 if(type == "Hyperelasticity")
473 {
474 result = create_hyperelasticity_control(dom_ctrl, section_key, meshopt_config, solver_config);
475 }
476
477 if(result == nullptr)
478 {
479 XABORTM("MeshOptimizer section has unhandled type "+type);
480 }
481
482 return result;
483 }
