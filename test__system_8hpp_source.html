347#define TEST_CHECK_THROWS(a, b) \
348 do { \
349 try { \
350 try { \

352 this->check(__func__, __FILE__, __LINE__, false, \
353 this->_id + "\n" + "Expected exception of type '" #b "' not thrown"); \
354 } catch (b &) { \
355 TEST_CHECK(true); \
356 } \
357 } catch (const FEAT::TestSystem::TestFailedException &) { \
358 throw; \
359 } catch (const std::exception & test_e) { \
360 throw FEAT::TestSystem::TestFailedException(__func__, __FILE__, __LINE__, \
361 "Test threw unexpected exception "+ FEAT::stringify(test_e.what()) + \
362 " inside a TEST_CHECK_THROWS block"); \
363 } catch (...) { \
364 throw FEAT::TestSystem::TestFailedException(__func__, __FILE__, __LINE__, \
365 "Test threw unexpected unknown exception inside a TEST_CHECK_THROWS block"); \
366 } \
367 } while (false)
