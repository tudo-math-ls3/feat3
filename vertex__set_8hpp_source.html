200 void transform(const VertexType& origin, const VertexType& angles, const VertexType& offset)
201 {
202 // create rotation matrix
203 Tiny::Matrix<CoordType, num_coords_, num_coords_> rot;
204 _aux_rot_mat(rot, angles);
205
206 // transform all vertices
207 for(auto& v : _vertices)
208 {
209 v.set_mat_vec_mult(rot, v - origin) += offset;
210 }
211 }
