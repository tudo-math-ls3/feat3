58 template<typename EvalTraits_>
59 class Evaluator :
60 public Analytic::Function::Evaluator<EvalTraits_>
61 {
62 public:
64 typedef typename EvalTraits_::DataType DataType;
66 typedef typename EvalTraits_::PointType PointType;
68 typedef typename EvalTraits_::ValueType ValueType;
70 typedef typename EvalTraits_::GradientType GradientType;
72 typedef typename EvalTraits_::HessianType HessianType;
73
74 private:
76 const PointType _origin;
77
78 public:
80 explicit Evaluator(const DistanceFunction& function) :
81 _origin(function._origin)
82 {
83 }
84
85 ValueType value(const PointType& point) const
86 {
87 return (point - _origin).norm_euclid();
88 }
89
90 GradientType gradient(const PointType& point) const
91 {
92 DataType norm (this->value(point));
93
94 if(norm < Math::eps<DataType>())
95 return GradientType::null();
96
97 return (DataType(1) / norm) * (point - _origin);
98 }
99
100 HessianType hessian(const PointType& point) const
101 {
102 DataType norm (this->value(point));
103 if(norm < Math::eps<DataType>())
104 return HessianType::null();
105
106 HessianType hess (DataType(0));
107 norm = DataType(1)/norm;
108 DataType denom = Math::sqr(norm)*norm;
109
110 for(int i(0); i < dim_; ++i)
111 {
112 hess[i][i] = norm;
113 for(int j(0); j < dim_; ++j)
114 {
115 hess[i][j] -= ((point[i] - _origin[i]) * (point[j] - _origin[j]))*denom;
116 }
117 }
118 return hess;
119 }

124 PointType _origin;

129 _origin(origin)
