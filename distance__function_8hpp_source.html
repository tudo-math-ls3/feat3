59 template<typename EvalTraits_>
60 class Evaluator :
61 public Analytic::Function::Evaluator<EvalTraits_>
62 {
63 public:
65 typedef typename EvalTraits_::DataType DataType;
67 typedef typename EvalTraits_::PointType PointType;
69 typedef typename EvalTraits_::ValueType ValueType;
71 typedef typename EvalTraits_::GradientType GradientType;
73 typedef typename EvalTraits_::HessianType HessianType;
74
75 private:
77 const PointType _origin;
78
79 public:
81 explicit Evaluator(const DistanceFunction& function) :
82 _origin(function._origin)
83 {
84 }
85
86 ValueType value(const PointType& point) const
87 {
88 return (point - _origin).norm_euclid();
89 }
90
91 GradientType gradient(const PointType& point) const
92 {
93 DataType norm (this->value(point));
94
95 if(norm < Math::eps<DataType>())
96 return GradientType::null();
97
98 return (DataType(1) / norm) * (point - _origin);
99 }
100
101 HessianType hessian(const PointType& point) const
102 {
103 DataType norm (this->value(point));
104 if(norm < Math::eps<DataType>())
105 return HessianType::null();
106
107 HessianType hess (DataType(0));
108 norm = DataType(1)/norm;
109 DataType denom = Math::sqr(norm)*norm;
110
111 for(int i(0); i < dim_; ++i)
112 {
113 hess[i][i] = norm;
114 for(int j(0); j < dim_; ++j)
115 {
116 hess[i][j] -= ((point[i] - _origin[i]) * (point[j] - _origin[j]))*denom;
117 }
118 }
119 return hess;
120 }
121 }; // class DistanceFunction::Evaluator<...>

125 PointType _origin;

130 _origin(origin)

132 }

138 }
