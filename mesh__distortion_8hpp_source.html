99 void distort_uniform(CoordType rad)
100 {
101 _calc_boundary_vertices();
102
103 auto& vtx = _mesh.get_vertex_set();
104
105 if(world_dim == 2)
106 {
107 for (Index i(0); i < vtx.get_num_vertices(); ++i)
108 {
109 if(_boundary_vertices[i] == 0)
110 {
111 CoordType t(0);
112 _rand >> t;
113 t *= CoordType(2) * _pi_val;
114 vtx[i][0] += rad * Math::cos(t);
115 vtx[i][1] += rad * Math::sin(t);
116 }
117 }
118 }
119 else if (world_dim == 3)
120 {
121 for (Index i(0); i < vtx.get_num_vertices(); ++i)
122 {
123 if(_boundary_vertices[i] == 0)
124 {
125 CoordType theta(0), phi(0);
126 _rand >> theta;
127 _rand >> phi;
128 theta *= _pi_val;
129 phi *= CoordType(2) * _pi_val;
130 vtx[i][0] += rad * Math::sin(theta) * Math::cos(phi);
131 vtx[i][1] += rad * Math::sin(theta) * Math::sin(phi);
132 vtx[i][2] += rad * Math::cos(theta);
133 }
134 }
135 }
136
137 // synchronize over all processes, if necessary
138 _synchronize();
139 }

155 void distort_shortest_edge_uniform(CoordType scale = CoordType(0.1))
156 {
157 _calc_boundary_vertices();
158 _calc_shortest_edge();
159 CoordType min = *std::min_element(_shortest_edge.begin(), _shortest_edge.end());
160 distort_uniform(min * scale);
161 }

176 void distort_shortest_edge_local(CoordType scale = CoordType(0.1))
177 {
178 _calc_boundary_vertices();
179 _calc_shortest_edge();
180
181 CoordType rad;
182
183 auto& vtx = _mesh.get_vertex_set();
184
185 if(world_dim == 2)
186 {
187 for (Index i(0); i < vtx.get_num_vertices(); ++i)
188 {
189 if(_boundary_vertices[i] == 0)
190 {
191 rad = scale * _shortest_edge[i];
192 CoordType t(0);
193 _rand >> t;
194 t *= CoordType(2) * _pi_val;
195 vtx[i][0] += rad * Math::cos(t);
196 vtx[i][1] += rad * Math::sin(t);
197 }
198 }
199 }
200 else if (world_dim == 3)
201 {
202 for (Index i(0); i < vtx.get_num_vertices(); ++i)
203 {
204 if(_boundary_vertices[i] == 0)
205 {
206 rad = scale * _shortest_edge[i];
207 CoordType theta(0), phi(0);
208 _rand >> theta;
209 _rand >> phi;
210 theta *= _pi_val;
211 phi *= CoordType(2) * _pi_val;
212 vtx[i][0] += rad * Math::sin(theta) * Math::cos(phi);
213 vtx[i][1] += rad * Math::sin(theta) * Math::sin(phi);
214 vtx[i][2] += rad * Math::cos(theta);
215 }

223 protected:
225 virtual void _calc_boundary_facets()
226 {
227 const auto& vtx_neighbors = _mesh.get_neighbors();
228
229 const auto& faces_at_elem = _mesh.template get_index_set<shape_dim, shape_dim - 1>();
230
231 int num_faces = faces_at_elem.num_indices;
232
233 Index num_elements(_mesh.get_num_elements());
234
235 // loop over all elements
236 for (Index i = 0; i < num_elements; ++i)
237 {
238 // loop over faces of each element
239 for (int j = 0; j < num_faces; ++j)
240 {
241 // check whether there is a neighbor at that edge
242 if (vtx_neighbors[i][j] > num_elements)
243 {
244 // if not: facet is at the boundary
245 _boundary_facets[faces_at_elem[i][j]] = 1;
246 }
247 }
248 }
249 }
250
252 virtual void _calc_boundary_vertices()
253 {
254 _calc_boundary_facets();
255 const auto& verts_at_face = _mesh.template get_index_set<shape_dim - 1, 0>();
256
257 int vert_per_face = verts_at_face.num_indices;
258
259 // loop over all facets
260 for (Index i = 0; i < _mesh.get_num_entities(shape_dim - 1); ++i)
261 {
262 // check if facet is at boundary
263 if(_boundary_facets[i])
264 {
265 // loop over adjacent vertices
266 for (int j = 0; j < vert_per_face; ++j)
267 {
268 _boundary_vertices[verts_at_face[i][j]] = 1;
269 }
270 }
271 }
272 }
273
275 virtual void _calc_shortest_edge()
276 {
277 auto& vtx = _mesh.get_vertex_set();
278
279 const auto& verts_at_edg = _mesh.template get_index_set<1, 0>();
280
281 // loop over all edges
282 for (Index i = 0; i < verts_at_edg.get_num_entities(); ++i)
283 {
284 Index ind_vert1 = verts_at_edg[i][0];
285 Index ind_vert2 = verts_at_edg[i][1];
286 CoordType edge_length = (vtx[ind_vert1] - vtx[ind_vert2]).norm_euclid();
287
288 // update the two nodes belonging to the edge
289 _shortest_edge[ind_vert1] = Math::min(_shortest_edge[ind_vert1], edge_length);
290 _shortest_edge[ind_vert2] = Math::min(_shortest_edge[ind_vert2], edge_length);
291 }
292 }
293
295 virtual void _synchronize()
296 {
297 // nothing to do here
298 }

318 template<typename MeshType_>
319 class DistributedMeshDistortion :
320 public MeshDistortion<MeshType_>
321 {
322 public:
323 typedef MeshDistortion<MeshType_> BaseClass;
324 typedef MeshType_ MeshType;
325 typedef MeshPart<MeshType> MeshPartType;
326 typedef RootMeshNode<MeshType> RootMeshNodeType;
327
328 static constexpr int world_dim = MeshType::world_dim;
329 static constexpr int facet_dim = MeshType::shape_dim - 1;
330
331 typedef typename BaseClass::CoordType CoordType;
332
333 protected:
335 const Dist::Comm& _comm;
337 RootMeshNodeType& _root_mesh_node;
339 std::vector<int> _vertex_owners;
340
341 public:
354 explicit DistributedMeshDistortion(const Dist::Comm& comm, RootMeshNodeType& root_mesh_node,
355 Random::SeedType seed = Random::def_seed) :
356 BaseClass(*root_mesh_node.get_mesh(), seed),
357 _comm(comm),
358 _root_mesh_node(root_mesh_node),
359 _vertex_owners()
360 {
361 }
362
364 virtual ~DistributedMeshDistortion()
365 {
366 }

368 protected:
370 virtual void _calc_boundary_facets() override
371 {
372 // call base class to determine all boundary facets
373 BaseClass::_calc_boundary_facets();
374
375 // Each process has already calculated its local boundary facets, so now we need to loop
376 // over all halos and remove all halo facets from the set of boundary facets, because
377 // halo facets are by definition not a part of the actual domain boundary.
378
379 // loop over all halos and kick out all halo facets
380 const auto& halo_map = _root_mesh_node.get_halo_map();
381 for(const auto& v : halo_map)
382 {
383 // loop over all facets of the halo and remove them from the boundary set
384 const TargetSet& halo_facets = v.second->template get_target_set<facet_dim>();
385 for(Index i(0); i < halo_facets.get_num_entities(); ++i)
386 this->_boundary_facets.at(halo_facets[i]) = 0;
387 }
388 }
389
391 virtual void _calc_shortest_edge() override
392 {
393 // call base class to determine the shortest edges for this patch
394 BaseClass::_calc_shortest_edge();
395
396 // Each process has already calculated the shortest edge for each of its vertices, so now
397 // we need to loop over all halos and exchange the shortest edge length for each halo
398 // vertex with the corresponding neighbor process and determine for each vertex whether
399 // that neighbor has found a shorter edge for it.
400
401 // get our halo map
402 const auto& halo_map = _root_mesh_node.get_halo_map();
403 const std::size_t num_halos = halo_map.size();
404
405 // nothing to do here?
406 if(halo_map.empty())
407 return;
408
409 // create buffers
410 std::vector<int> ranks;
411 std::vector<std::vector<CoordType>> send_bufs, recv_bufs;
412 Dist::RequestVector send_reqs, recv_reqs;
413 ranks.reserve(num_halos);
414 send_bufs.resize(num_halos);
415 recv_bufs.resize(num_halos);
416 send_reqs.reserve(num_halos);
417 recv_reqs.reserve(num_halos);
418
419 // loop over all our halos/neighbors
420 for(auto it = halo_map.begin(); it != halo_map.end(); ++it)
421 {
422 // get halo and the number of halo vertices
423 const int halo_rank = it->first;
424 const TargetSet& halo_vtx = it->second->template get_target_set<0>();
425 const Index halo_size = halo_vtx.get_num_entities();
426
427 // allocate buffers
428 auto& sbuf = send_bufs.at(ranks.size());
429 auto& rbuf = recv_bufs.at(ranks.size());
430 sbuf.resize(halo_size);
431 rbuf.resize(halo_size);
432 ranks.push_back(halo_rank);
433
434 // post receive
435 recv_reqs.push_back(_comm.irecv(rbuf.data(), rbuf.size(), halo_rank));
436
437 // compute source buffer = shortest edge for each halo vertex
438 for(Index k(0); k < halo_size; ++k)
439 sbuf.at(k) = this->_shortest_edge.at(halo_vtx[k]);
440
441 // post send
442 send_reqs.push_back(_comm.isend(sbuf.data(), sbuf.size(), halo_rank));
443 }
444
445 // process all pending receives
446 for(std::size_t idx(0u); recv_reqs.wait_any(idx); )
447 {
448 // get the halo rank
449 const int halo_rank = ranks.at(idx);
450
451 // find the halo mesh part
452 auto it = halo_map.find(halo_rank);
453 XASSERT(it != halo_map.end());
454
455 // get the halo vertex target set
456 const TargetSet& halo_vtx = it->second->template get_target_set<0>();
457 const Index halo_size = halo_vtx.get_num_entities();
458 auto& rbuf = recv_bufs.at(idx);
459
460 // update the shortest edge length for each halo vertex
461 for(Index k(0); k < halo_size; ++k)
462 this->_shortest_edge.at(halo_vtx[k]) = Math::min(this->_shortest_edge.at(halo_vtx[k]), rbuf.at(k));
463 }
464
465 // wait for all sends to finish
466 send_reqs.wait_all();
467 }
468
470 virtual void _calc_vertex_owners()
471 {
472 if(!_vertex_owners.empty())
473 return;
474
475 // initially assume that we own all our vertices
476 _vertex_owners.resize(_root_mesh_node.get_mesh()->get_num_vertices(), _comm.rank());
477
478 // loop over all our halos/neighbors
479 const auto& halo_map = _root_mesh_node.get_halo_map();
480 for(auto it = halo_map.begin(); it != halo_map.end(); ++it)
481 {
482 const int halo_rank = it->first;
483 const TargetSet& halo_vtx = it->second->template get_target_set<0>();
484
485 // update vertex owner ranks
486 for(Index i(0); i < halo_vtx.get_num_entities(); ++i)
487 {
488 _vertex_owners[halo_vtx[i]] = Math::min(_vertex_owners[halo_vtx[i]], halo_rank);
489 }
490 }
491 }
492
493
494 // synchronize over all processes, if necessary
495 virtual void _synchronize() override
496 {
497 // Each process has already disturbed its local mesh, so now we need to ensure that the
498 // coordinates of all vertices, which are shared by more than one process, have consistent
499 // (= identical) coordinates on all sharing process. We do this by employing the convention
500 // that the coordinates of a shared vertex are defined by the process with the smallest rank
501 // that shares this particular vertex and we call that process the 'owner' of that vertex.
502 // With this definition, our approach is as follows: Loop over all halos/neighbors and
503 // determine whether we have small rank than that neighbor and if...
504 // yes: then we collect the vertex coordinates for this halo and send these coordinates
505 // to our neighbor.
506 // no: then we post a receive request for the vertex coordinates and once that receive
507 // request is fulfilled, we loop over all vertices in that halo, check for each
508 // have vertex if that neighbor is the owner and if yes, then we overwrite our
509 // vertex coordinates with the ones we receives from our (owner) neighbor.
510
511 // get our halo map
512 const auto& halo_map = _root_mesh_node.get_halo_map();
513 const std::size_t num_halos = halo_map.size();
514
515 // nothing to do here?
516 if(halo_map.empty())
517 return;
518
519 // compute the vertex owner ranks
520 _calc_vertex_owners();
521
522 auto& vertex_set = _root_mesh_node.get_mesh()->get_vertex_set();
523
524 const int my_rank = _comm.rank();
525
526 // create buffers
527 std::vector<int> ranks;
528 std::vector<std::vector<CoordType>> buffers;
529 Dist::RequestVector requests;
530 ranks.reserve(num_halos);
531 buffers.resize(num_halos);
532 requests.reserve(num_halos);
533
534 // loop over all our halos/neighbors
535 for(auto it = halo_map.begin(); it != halo_map.end(); ++it)
536 {
537 // get halo and buffer size
538 const int halo_rank = it->first;
539 const TargetSet& halo_vtx = it->second->template get_target_set<0>();
540 const Index halo_size = halo_vtx.get_num_entities();
541 const std::size_t buf_size = halo_size * Index(world_dim);
542
543 // allocate buffers
544 auto& buf = buffers.at(ranks.size());
545 buf.resize(buf_size);
546 ranks.push_back(halo_rank);
547
548 // should we send or receive the vertex coordinates?
549 if(my_rank < halo_rank)
550 {
551 // copy vertex coordinates into buffer
552 for(Index i(0), j(0); i < halo_size; ++i)
553 {
554 const auto& vtx = vertex_set[halo_vtx[i]];
555 for(int k(0); k < world_dim; ++j, ++k)
556 buf[j] = vtx[k];
557 }
558
559 // post send
560 requests.push_back(_comm.isend(buf.data(), buf.size(), halo_rank));
561 }
562 else
563 {
564 // post receive
565 requests.push_back(_comm.irecv(buf.data(), buf.size(), halo_rank));
566 }
567 }
568
569 // process all pending receives
570 for(std::size_t idx(0u); requests.wait_any(idx); )
571 {
572 // get the halo rank
573 const int halo_rank = ranks.at(idx);
574
575 // was this a send-request?
576 if(my_rank < halo_rank)
577 continue; // nothing to do after a send
578
579 // find the halo mesh part
580 auto it = halo_map.find(halo_rank);
581 XASSERT(it != halo_map.end());
582
583 // get the halo vertex target set
584 const TargetSet& halo_vtx = it->second->template get_target_set<0>();
585 const Index halo_size = halo_vtx.get_num_entities();
586 auto& buf = buffers.at(idx);
587
588 // loop over all halo vertices
589 for(Index i(0), j(0); i < halo_size; ++i, j += world_dim)
590 {
591 // is the sender process the actual owner of the vertex?
592 if(_vertex_owners[halo_vtx[i]] == halo_rank)
593 {
594 // yup, so accept the owner's vertex coordinates
595 auto& vtx = vertex_set[halo_vtx[i]];
596 for(int k(0); k < world_dim; ++k)
597 vtx[k] = buf[j+Index(k)];
598 }
599 }
600 }
601 }
