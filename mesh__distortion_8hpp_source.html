97 void distort_uniform(CoordType rad)
98 {
99 _calc_boundary_vertices();
100
101 auto& vtx = _mesh.get_vertex_set();
102
103 if(world_dim == 2)
104 {
105 for (Index i(0); i < vtx.get_num_vertices(); ++i)
106 {
107 if(_boundary_vertices[i] == 0)
108 {
109 CoordType t(0);
110 _rand >> t;
111 t *= CoordType(2) * _pi_val;
112 vtx[i][0] += rad * Math::cos(t);
113 vtx[i][1] += rad * Math::sin(t);
114 }
115 }
116 }
117 else if (world_dim == 3)
118 {
119 for (Index i(0); i < vtx.get_num_vertices(); ++i)
120 {
121 if(_boundary_vertices[i] == 0)
122 {
123 CoordType theta(0), phi(0);
124 _rand >> theta;
125 _rand >> phi;
126 theta *= _pi_val;
127 phi *= CoordType(2) * _pi_val;
128 vtx[i][0] += rad * Math::sin(theta) * Math::cos(phi);
129 vtx[i][1] += rad * Math::sin(theta) * Math::sin(phi);
130 vtx[i][2] += rad * Math::cos(theta);
131 }
132 }
133 }
134
135 // synchronize over all processes, if necessary
136 _synchronize();
137 }

153 void distort_shortest_edge_uniform(CoordType scale = CoordType(0.1))
154 {
155 _calc_boundary_vertices();
156 _calc_shortest_edge();
157 CoordType min = *std::min_element(_shortest_edge.begin(), _shortest_edge.end());
158 distort_uniform(min * scale);

174 void distort_shortest_edge_local(CoordType scale = CoordType(0.1))
175 {
176 _calc_boundary_vertices();
177 _calc_shortest_edge();
178
179 CoordType rad;
180
181 auto& vtx = _mesh.get_vertex_set();
182
183 if(world_dim == 2)
184 {
185 for (Index i(0); i < vtx.get_num_vertices(); ++i)
186 {
187 if(_boundary_vertices[i] == 0)
188 {
189 rad = scale * _shortest_edge[i];
190 CoordType t(0);
191 _rand >> t;
192 t *= CoordType(2) * _pi_val;
193 vtx[i][0] += rad * Math::cos(t);
194 vtx[i][1] += rad * Math::sin(t);
195 }
196 }
197 }
198 else if (world_dim == 3)
199 {
200 for (Index i(0); i < vtx.get_num_vertices(); ++i)
201 {
202 if(_boundary_vertices[i] == 0)
203 {
204 rad = scale * _shortest_edge[i];
205 CoordType theta(0), phi(0);
206 _rand >> theta;
207 _rand >> phi;
208 theta *= _pi_val;
209 phi *= CoordType(2) * _pi_val;
210 vtx[i][0] += rad * Math::sin(theta) * Math::cos(phi);
211 vtx[i][1] += rad * Math::sin(theta) * Math::sin(phi);
212 vtx[i][2] += rad * Math::cos(theta);
213 }

223 virtual void _calc_boundary_facets()
224 {
225 const auto& vtx_neighbors = _mesh.get_neighbors();
226
227 const auto& faces_at_elem = _mesh.template get_index_set<shape_dim, shape_dim - 1>();
228
229 int num_faces = faces_at_elem.num_indices;
230
231 Index num_elements(_mesh.get_num_elements());
232
233 // loop over all elements
234 for (Index i = 0; i < num_elements; ++i)
235 {
236 // loop over faces of each element
237 for (int j = 0; j < num_faces; ++j)
238 {
239 // check whether there is a neighbor at that edge
240 if (vtx_neighbors[i][j] > num_elements)
241 {
242 // if not: facet is at the boundary
243 _boundary_facets[faces_at_elem[i][j]] = 1;
244 }
245 }
246 }
247 }
248
250 virtual void _calc_boundary_vertices()
251 {
252 _calc_boundary_facets();
253 const auto& verts_at_face = _mesh.template get_index_set<shape_dim - 1, 0>();
254
255 int vert_per_face = verts_at_face.num_indices;
256
257 // loop over all facets
258 for (Index i = 0; i < _mesh.get_num_entities(shape_dim - 1); ++i)
259 {
260 // check if facet is at boundary
261 if(_boundary_facets[i])
262 {
263 // loop over adjacent vertices
264 for (int j = 0; j < vert_per_face; ++j)
265 {
266 _boundary_vertices[verts_at_face[i][j]] = 1;
267 }
268 }
269 }
270 }
271
273 virtual void _calc_shortest_edge()
274 {
275 auto& vtx = _mesh.get_vertex_set();
276
277 const auto& verts_at_edg = _mesh.template get_index_set<1, 0>();
278
279 // loop over all edges
280 for (Index i = 0; i < verts_at_edg.get_num_entities(); ++i)
281 {
282 Index ind_vert1 = verts_at_edg[i][0];
283 Index ind_vert2 = verts_at_edg[i][1];
284 CoordType edge_length = (vtx[ind_vert1] - vtx[ind_vert2]).norm_euclid();
285
286 // update the two nodes belonging to the edge
287 _shortest_edge[ind_vert1] = Math::min(_shortest_edge[ind_vert1], edge_length);
288 _shortest_edge[ind_vert2] = Math::min(_shortest_edge[ind_vert2], edge_length);
289 }
290 }

293 virtual void _synchronize()
294 {
295 // nothing to do here
296 }

317 class DistributedMeshDistortion :
318 public MeshDistortion<MeshType_>
319 {
320 public:
321 typedef MeshDistortion<MeshType_> BaseClass;
322 typedef MeshType_ MeshType;
323 typedef MeshPart<MeshType> MeshPartType;
324 typedef RootMeshNode<MeshType> RootMeshNodeType;

327 static constexpr int facet_dim = MeshType::shape_dim - 1;

352 explicit DistributedMeshDistortion(const Dist::Comm& comm, RootMeshNodeType& root_mesh_node,

355 _comm(comm),

366 protected:
368 virtual void _calc_boundary_facets() override
369 {
370 // call base class to determine all boundary facets
371 BaseClass::_calc_boundary_facets();
372
373 // Each process has already calculated its local boundary facets, so now we need to loop
374 // over all halos and remove all halo facets from the set of boundary facets, because
375 // halo facets are by definition not a part of the actual domain boundary.
376
377 // loop over all halos and kick out all halo facets
378 const auto& halo_map = _root_mesh_node.get_halo_map();
379 for(const auto& v : halo_map)
380 {
381 // loop over all facets of the halo and remove them from the boundary set
382 const TargetSet& halo_facets = v.second->template get_target_set<facet_dim>();
383 for(Index i(0); i < halo_facets.get_num_entities(); ++i)
384 this->_boundary_facets.at(halo_facets[i]) = 0;
385 }
386 }
387
389 virtual void _calc_shortest_edge() override
390 {
391 // call base class to determine the shortest edges for this patch
392 BaseClass::_calc_shortest_edge();
393
394 // Each process has already calculated the shortest edge for each of its vertices, so now
395 // we need to loop over all halos and exchange the shortest edge length for each halo
396 // vertex with the corresponding neighbor process and determine for each vertex whether
397 // that neighbor has found a shorter edge for it.
398
399 // get our halo map
400 const auto& halo_map = _root_mesh_node.get_halo_map();
401 const std::size_t num_halos = halo_map.size();
402
403 // nothing to do here?
404 if(halo_map.empty())
405 return;
406
407 // create buffers
408 std::vector<int> ranks;
409 std::vector<std::vector<CoordType>> send_bufs, recv_bufs;
410 Dist::RequestVector send_reqs, recv_reqs;
411 ranks.reserve(num_halos);
412 send_bufs.resize(num_halos);
413 recv_bufs.resize(num_halos);
414 send_reqs.reserve(num_halos);
415 recv_reqs.reserve(num_halos);
416
417 // loop over all our halos/neighbors
418 for(auto it = halo_map.begin(); it != halo_map.end(); ++it)
419 {
420 // get halo and the number of halo vertices
421 const int halo_rank = it->first;
422 const TargetSet& halo_vtx = it->second->template get_target_set<0>();
423 const Index halo_size = halo_vtx.get_num_entities();
424
425 // allocate buffers
426 auto& sbuf = send_bufs.at(ranks.size());
427 auto& rbuf = recv_bufs.at(ranks.size());
428 sbuf.resize(halo_size);
429 rbuf.resize(halo_size);
430 ranks.push_back(halo_rank);
431
432 // post receive
433 recv_reqs.push_back(_comm.irecv(rbuf.data(), rbuf.size(), halo_rank));
434
435 // compute source buffer = shortest edge for each halo vertex
436 for(Index k(0); k < halo_size; ++k)
437 sbuf.at(k) = this->_shortest_edge.at(halo_vtx[k]);
438
439 // post send
440 send_reqs.push_back(_comm.isend(sbuf.data(), sbuf.size(), halo_rank));
441 }
442
443 // process all pending receives
444 for(std::size_t idx(0u); recv_reqs.wait_any(idx); )
445 {
446 // get the halo rank
447 const int halo_rank = ranks.at(idx);
448
449 // find the halo mesh part
450 auto it = halo_map.find(halo_rank);
451 XASSERT(it != halo_map.end());
452
453 // get the halo vertex target set
454 const TargetSet& halo_vtx = it->second->template get_target_set<0>();
455 const Index halo_size = halo_vtx.get_num_entities();
456 auto& rbuf = recv_bufs.at(idx);
457
458 // update the shortest edge length for each halo vertex
459 for(Index k(0); k < halo_size; ++k)
460 this->_shortest_edge.at(halo_vtx[k]) = Math::min(this->_shortest_edge.at(halo_vtx[k]), rbuf.at(k));
461 }
462
463 // wait for all sends to finish
464 send_reqs.wait_all();
465 }

468 virtual void _calc_vertex_owners()
469 {
470 if(!_vertex_owners.empty())
471 return;
472
473 // initially assume that we own all our vertices
474 _vertex_owners.resize(_root_mesh_node.get_mesh()->get_num_vertices(), _comm.rank());
475
476 // loop over all our halos/neighbors
477 const auto& halo_map = _root_mesh_node.get_halo_map();
478 for(auto it = halo_map.begin(); it != halo_map.end(); ++it)
479 {
480 const int halo_rank = it->first;
481 const TargetSet& halo_vtx = it->second->template get_target_set<0>();
482
483 // update vertex owner ranks
484 for(Index i(0); i < halo_vtx.get_num_entities(); ++i)
485 {
486 _vertex_owners[halo_vtx[i]] = Math::min(_vertex_owners[halo_vtx[i]], halo_rank);
487 }
488 }
489 }

492 // synchronize over all processes, if necessary
493 virtual void _synchronize() override
494 {
495 // Each process has already disturbed its local mesh, so now we need to ensure that the
496 // coordinates of all vertices, which are shared by more than one process, have consistent
497 // (= identical) coordinates on all sharing process. We do this by employing the convention
498 // that the coordinates of a shared vertex are defined by the process with the smallest rank
499 // that shares this particular vertex and we call that process the 'owner' of that vertex.
500 // With this definition, our approach is as follows: Loop over all halos/neighbors and
501 // determine whether we have small rank than that neighbor and if...
502 // yes: then we collect the vertex coordinates for this halo and send these coordinates
503 // to our neighbor.
504 // no: then we post a receive request for the vertex coordinates and once that receive
505 // request is fulfilled, we loop over all vertices in that halo, check for each
506 // have vertex if that neighbor is the owner and if yes, then we overwrite our
507 // vertex coordinates with the ones we receives from our (owner) neighbor.
508
509 // get our halo map
510 const auto& halo_map = _root_mesh_node.get_halo_map();
511 const std::size_t num_halos = halo_map.size();
512
513 // nothing to do here?
514 if(halo_map.empty())
515 return;
516
517 // compute the vertex owner ranks
518 _calc_vertex_owners();
519
520 auto& vertex_set = _root_mesh_node.get_mesh()->get_vertex_set();
521
522 const int my_rank = _comm.rank();
523
524 // create buffers
525 std::vector<int> ranks;
526 std::vector<std::vector<CoordType>> buffers;
527 Dist::RequestVector requests;
528 ranks.reserve(num_halos);
529 buffers.resize(num_halos);
530 requests.reserve(num_halos);
531
532 // loop over all our halos/neighbors
533 for(auto it = halo_map.begin(); it != halo_map.end(); ++it)
534 {
535 // get halo and buffer size
536 const int halo_rank = it->first;
537 const TargetSet& halo_vtx = it->second->template get_target_set<0>();
538 const Index halo_size = halo_vtx.get_num_entities();
539 const std::size_t buf_size = halo_size * Index(world_dim);
540
541 // allocate buffers
542 auto& buf = buffers.at(ranks.size());
543 buf.resize(buf_size);
544 ranks.push_back(halo_rank);
545
546 // should we send or receive the vertex coordinates?
547 if(my_rank < halo_rank)
548 {
549 // copy vertex coordinates into buffer
550 for(Index i(0), j(0); i < halo_size; ++i)
551 {
552 const auto& vtx = vertex_set[halo_vtx[i]];
553 for(int k(0); k < world_dim; ++j, ++k)
554 buf[j] = vtx[k];
555 }
556
557 // post send
558 requests.push_back(_comm.isend(buf.data(), buf.size(), halo_rank));
559 }
560 else
561 {
562 // post receive
563 requests.push_back(_comm.irecv(buf.data(), buf.size(), halo_rank));
564 }
565 }
566
567 // process all pending receives
568 for(std::size_t idx(0u); requests.wait_any(idx); )
569 {
570 // get the halo rank
571 const int halo_rank = ranks.at(idx);
572
573 // was this a send-request?
574 if(my_rank < halo_rank)
575 continue; // nothing to do after a send
576
577 // find the halo mesh part
578 auto it = halo_map.find(halo_rank);
579 XASSERT(it != halo_map.end());
580
581 // get the halo vertex target set
582 const TargetSet& halo_vtx = it->second->template get_target_set<0>();
583 const Index halo_size = halo_vtx.get_num_entities();
584 auto& buf = buffers.at(idx);
585
586 // loop over all halo vertices
587 for(Index i(0), j(0); i < halo_size; ++i, j += world_dim)
588 {
589 // is the sender process the actual owner of the vertex?
590 if(_vertex_owners[halo_vtx[i]] == halo_rank)
591 {
592 // yup, so accept the owner's vertex coordinates
593 auto& vtx = vertex_set[halo_vtx[i]];
594 for(int k(0); k < world_dim; ++k)
595 vtx[k] = buf[j+Index(k)];
596 }
597 }
598 }
599 }
