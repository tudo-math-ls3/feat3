196 DT_ wait()
197 {
198 XASSERTM(!_finished, "ticket was already completed by a wait call");
199
200#ifdef FEAT_HAVE_MPI
201#ifdef FEAT_MPI_THREAD_MULTIPLE
202 _thread.join();
203 Statistics::add_time_mpi_execute_reduction(_mpi_exec);
204 Statistics::add_time_mpi_wait_reduction(_mpi_wait);
205#else // no FEAT_MPI_THREAD_MULTIPLE
206 TimeStamp ts_start;
207 _req.wait();
208 Statistics::add_time_mpi_wait_reduction(ts_start.elapsed_now());
209#endif // FEAT_MPI_THREAD_MULTIPLE
210#endif // FEAT_HAVE_MPI
211 _finished = true;
212 return (_sqrt ? Math::sqrt(_r) : _r);
213 }
