200 DT_ wait()
201 {
202 XASSERTM(!_finished, "ticket was already completed by a wait call");
203
204#ifdef FEAT_HAVE_MPI
205#ifdef FEAT_MPI_THREAD_MULTIPLE
206 _thread.join();
207 Statistics::add_time_mpi_execute_reduction(_mpi_exec);
208 Statistics::add_time_mpi_wait_reduction(_mpi_wait);
209#else // no FEAT_MPI_THREAD_MULTIPLE

211 _req.wait();
212 Statistics::add_time_mpi_wait_reduction(ts_start.elapsed_now());
213#endif // FEAT_MPI_THREAD_MULTIPLE
214#endif // FEAT_HAVE_MPI
215 _finished = true;
216 return (_sqrt ? Math::sqrt(_r) : _r);
217 }
