215 void project_point(WorldPoint& point, CoordType& signed_distance, WorldPoint& grad_distance) const
216 {
217 // There is nothing to do if the surface mesh does not have any cells
218 if(_surface_mesh->get_num_entities(SurfaceMeshType::shape_dim) == Index(0))
219 return;
220
221 // Vertex at cell information
222 const auto& idx(_surface_mesh->template get_index_set<SurfaceMeshType::shape_dim, 0>());
223 // The mesh's vertex set so we can get at the coordinates
224 const auto& vtx(_surface_mesh->get_vertex_set());
225
226 // When a point is found, this will hold the coefficients of its image under the projection wrt. the
227 // standard P1/Q1 transformation
228 Tiny::Vector<CoordType, SurfaceMeshType::shape_dim+1> coeffs(CoordType(0));
229 // This will hold the coefficients of the transformation mapping
230 Tiny::Vector<CoordType, SurfaceMeshType::shape_dim+1> bary(CoordType(0));
231
232 // Since we want to project the point an arbitrary distance, we set this to something huge
233 CoordType acceptable_distance(Math::huge<CoordType>());
234 // Find the facet in the SurfaceMesh we project to and compute the coefficients
235 Index best_facet =
236 find_cell(coeffs, point, Index(0), acceptable_distance, *_surface_mesh, true);
237
238 // Clip the coefficients
239 for(int j(0); j < SurfaceMeshType::shape_dim; ++j)
240 {
241 coeffs[j] = Math::max(coeffs[j], CoordType(0));
242 coeffs[j] = Math::min(coeffs[j], CoordType(1));
243 }
244 // Now that we have the surface mesh cell with the lowest distance, we can compute the projected point
245 WorldPoint projected_point(CoordType(0));
246
247 // Evaluate the surface mesh trafo for the computed coefficients
248 CoordType coeff0(CoordType(1));
249 for(int j(0); j < SurfaceMeshType::shape_dim; ++j)
250 {
251 // Index of the local vertex i in the SurfaceMesh
252 Index i(idx(best_facet, j+1));
253 projected_point += coeffs[j]*vtx[i];
254 coeff0 -= coeffs[j];
255 }
256
257 projected_point += coeff0*vtx[idx(best_facet, Index(0))];
258
259 grad_distance = (projected_point - point);
260 signed_distance = grad_distance.norm_euclid();
261
262 // If the distance is too small, we set the gradient vector to zero
263 if(signed_distance < Math::eps<CoordType>())
264 grad_distance.format(CoordType(0));
265 else
266 {
267 grad_distance.normalize();
268 WorldPoint nu(get_normal_on_tria(best_facet, coeffs));
269 signed_distance *= Math::signum(Tiny::dot(nu, grad_distance));
270 }
271
272 point = projected_point;
273
274 }

288 WorldPoint get_normal_on_tria(
289 const Index facet, const Tiny::Vector<CoordType, SurfaceMeshType::shape_dim+1> DOXY(coeffs)) const
290 {
291 // Vertex at cell information
292 const auto& idx(_surface_mesh->template get_index_set<SurfaceMeshType::shape_dim, 0>());
293 // The mesh's vertex set so we can get at the coordinates
294 const auto& vtx(_surface_mesh->get_vertex_set());
295
296 Tiny::Matrix<CoordType, 2, SurfaceMeshType::world_dim> coords(CoordType(0));
297 Tiny::Matrix<CoordType, SurfaceMeshType::world_dim, 2> coords_transpose(CoordType(0));
298
299 Index i0(idx(facet, 0));
300 Index i1(idx(facet, 1));
301 Index i2(idx(facet, 2));
302
303 coords[0] = vtx[i1] - vtx[i0];
304 coords[1] = vtx[i2] - vtx[i0];
305
306 for(int i(0); i < coords.m; ++i)
307 {
308 for(int j(0); j < coords.n; ++j)
309 coords_transpose(j,i) = coords(i,j);
310 }
311
312 WorldPoint nu(Tiny::orthogonal(coords_transpose));
313 nu.normalize();
314
315 return nu;
316 }

332 void project_meshpart(Mesh_& mesh, const MeshPart<Mesh_>& meshpart) const
333 {
334 // We need the topology of the meshpart because searching for points is cell-wise and we need to know
335 // about the vertices in each cell
336 XASSERTM(meshpart.get_topology() != nullptr, "The meshpart needs a topology for SurfaceMesh::project()!");
337
338 Index num_facets(meshpart.get_num_entities(SurfaceMeshType::shape_dim));
339
340 // There is nothing to do if the meshpart does not have any cells (which refer to facets of the original
341 // mesh)
342 if(num_facets == Index(0))
343 return;
344
345 // The number of vertices that need to be projected
346 const Index num_verts(meshpart.get_num_entities(0));
347 // Mapping of vertices from the meshpart to the real mesh
348 const auto& ts_verts(meshpart.template get_target_set<0>());
349
350 // Vertex at surface mesh cell information
351 const auto& idx_sm(_surface_mesh->template get_index_set<SurfaceMeshType::shape_dim, 0>());
352 // Vertex set of the surface mesh
353 const auto& vtx_sm(_surface_mesh->get_vertex_set());
354
355 // Vertex at facet information for the meshpart
356 const auto& idx_mp(meshpart.template get_index_set<SurfaceMeshType::shape_dim, 0>());
357 // Subfacet at facet information for the meshpart. i.e. if Mesh is a Hypercube<3> mesh, the boundary is a
358 // Hypercube<2> mesh, the facets are quads and THEIR subfacets are edges, so this is edge@quad
359 // Needed for computing neighbors.
360 const auto& facet_idx_mp(meshpart.template get_index_set
361 <SurfaceMeshType::shape_dim, SurfaceMeshType::shape_dim-1>());
362
363 // Neighbor information for all facets in the MeshPart. Needed for adding facets to the search stack
364 // so we can exploit local information
365 typename MeshPart<Mesh_>::template IndexSet<SurfaceMeshType::shape_dim, SurfaceMeshType::shape_dim-1>::Type
366 neigh_mp(meshpart.get_num_entities(SurfaceMeshType::shape_dim));
367
368 Geometry::Intern::FacetNeighbors::compute(neigh_mp, facet_idx_mp);
369
370 // The mesh's vertex set, since we modify the coordinates by projection
371 auto& vtx(mesh.get_vertex_set());
372
373 // When a point is found, this will hold the coefficients of its image under the projection wrt. the
374 // standard P1/Q1 transformation
375 Tiny::Vector<CoordType, SurfaceMeshType::shape_dim+1> coeffs(CoordType(0));
376
377 // For every vertex, we need to know if it was already projected
378 bool* vert_todo(new bool[num_verts]);
379 for(Index i(0); i < num_verts; ++i)
380 vert_todo[i] = true;
381
382 // For every facet, we need to know if it was already added to the search stack
383 bool* facet_todo(new bool[num_facets]);
384 for(Index i(0); i < num_facets; ++i)
385 facet_todo[i] = true;
386
387 // For every facet we keep a guess, a cell number in the surface mesh where we start looking for vertices
388 // from our facet.
389 Index* guesses(new Index[num_facets]);
390 for(Index i(0); i < num_facets; ++i)
391 guesses[i] = Index(0);
392
393 // The search stack for all facets. We add neighbors to it to keep the search as local as possible
394 std::deque<Index> facet_stack;
395 facet_stack.push_front(0);
396
397 // Treat all facets
398 while(!facet_stack.empty())
399 {
400 // This is the facet from the mesh part that's being worked on
401 Index current_facet(facet_stack.front());
402 // This will be the facet from the surface mesh were the vertex was found, set it to the best guess
403 // initially
404 Index facet_sm(guesses[current_facet]);
405
406 facet_stack.pop_front();
407 facet_todo[current_facet] = false;
408
409 // We want to orthogonally project a point, so we need to figure out an acceptable distance for this
410 // projection. Otherwise, it might happen that we project to the far side of the domain
411 CoordType acceptable_distance(CoordType(1));
412
413 // Compute acceptable distance. Because we do not have the transformation available, we cannot compute
414 // i.e. the volume of the facet to deduce an acceptable distance from that. So instead, we pick a
415 // random vertex of the facet, compute the distances to all other points, and take the geometric mean.
416 for(int j(0); j < idx_mp.num_indices-1; ++j)
417 {
418 Index i0(idx_mp(current_facet, 0));
419 Index i_mp(idx_mp(current_facet, j+1));
420 acceptable_distance *= ((vtx[i_mp] - vtx[i0]).norm_euclid());
421 }
422 acceptable_distance = Math::pow(
423 acceptable_distance, CoordType(1)/CoordType(idx_mp.num_indices) - CoordType(1));
424
425 // Find all vertices in the current facet
426 for(int j(0); j < idx_mp.num_indices; ++j)
427 {
428 // The index of the local vertex j in the Meshpart
429 Index i_mp(idx_mp(current_facet, j));
430
431 if(vert_todo[i_mp])
432 {
433 vert_todo[i_mp] = false;
434 const auto& x(vtx[ts_verts[i_mp]]);
435
436 // Find the facet in the SurfaceMesh we project to and compute the coefficients. We break at the
437 // first facet fulfilling the criteria
438 facet_sm = find_cell(coeffs, x, guesses[current_facet], acceptable_distance, * _surface_mesh, false);
439 // Update the guess for this facet because there might be more points to project
440 guesses[current_facet] = facet_sm;
441
442 // This is the index of the local vertex j in the real mesh
443 Index i_parent(ts_verts[i_mp]);
444 // Clear the old vertex coordinates so we can just add onto that
445 vtx[i_parent].format(CoordType(0));
446 // Evaluate the surface mesh trafo for the computed coefficients
447 CoordType coeff0(CoordType(1));
448 for(int i(0); i < SurfaceMeshType::shape_dim; ++i)
449 {
450 // Index of the local vertex i in the SurfaceMesh
451 Index i_sm(idx_sm(facet_sm, i+1));
452 vtx[i_parent] += coeffs[i]*vtx_sm[i_sm];
453 coeff0 -= coeffs[i];
454 }
455 vtx[i_parent] += coeff0*vtx_sm[idx_sm(facet_sm, 0)];
456
457 } // vert_todo[i_mp]
458 } // vertices in current facet
459
460 // Now add all neighbors that are still on the todo list to the search stack
461 for(int l(0); l < neigh_mp.num_indices; ++l)
462 {
463 Index k(neigh_mp(current_facet, l));
464 if(k != ~Index(0) && facet_todo[k])
465 {
466 facet_todo[k] = false;
467 facet_stack.push_back(k);
468 // The last vertex was found in facet_sm, and since k is a neighbor of current, it cannot be too
469 // far off
470 guesses[k] = facet_sm;
471 }
472 } // adding neighbors
473
474 } // search stack
475
476 // Clean up
477 delete[] facet_todo;
478 delete[] vert_todo;
479 delete[] guesses;
480 }

553 template<int world_dim, int sc_, int sx_>
554 Index find_cell(
555 Tiny::Vector<CoordType, SurfaceMeshType::shape_dim+1, sc_>& coeffs,
556 const Tiny::Vector<CoordType, world_dim, sx_>& x, Index guess, CoordType acceptable_distance,
557 const SurfaceMeshType& mesh, bool find_best_approx) const
558 {
559 // For every subfacet, we need to know if it was already traversed by adding the neighbor across said
560 // facet to the search stack. If the surface is concave, it might happen that facet A says "go to my
561 // neighbor, B", and then B says "go to my neighbor, A". It's then clear that the wanted point lies on
562 // the facet between A and B.
563
564 // Clear traversed information
565 for(Index facet(0); facet < mesh.get_num_entities(SurfaceMeshType::shape_dim-1); ++facet)
566 _edge_marker[facet] = false;
567
568 Index num_cells(mesh.get_num_entities(SurfaceMeshType::shape_dim));
569 // Clear cell todo information
570 for(Index cell(0); cell < num_cells; ++cell)
571 _cell_marker[cell] = false;
572
573 // Type of the vertex at cell index set
574 typedef typename SurfaceMeshType::template IndexSet<SurfaceMeshType::shape_dim, 0>::Type VertAtCellIdxType;
575 // Number of vertices per cell
576 static constexpr int num_vert_loc = VertAtCellIdxType::num_indices;
577 // Vertex at cell information
578 const VertAtCellIdxType& idx(mesh.template get_index_set<SurfaceMeshType::shape_dim, 0>());
579 // Facet at cell information
580 const auto& facet_idx(mesh.template
581 get_index_set<SurfaceMeshType::shape_dim, SurfaceMeshType::shape_dim-1>());
582 // The mesh's vertex set so we can get at the coordinates
583 const auto& vtx(mesh.get_vertex_set());
584
585 const auto& neigh(mesh.get_neighbors());
586
587 // This will contain all vertices making up a cell in the mesh. Since the mesh is of shape
588 // Simplex<shape_dim>, there are shape_dim+1 vertices.
589 Tiny::Matrix<CoordType, num_vert_loc, world_dim> coords(CoordType(0));
590
591 // This is the search stack
592 std::deque<Index> cell_stack;
593 // We start with our guess
594 cell_stack.push_front(guess);
595
596
597 Index best_approx_cell(~Index(0));
598
599 CoordType best_approx_dist(Math::huge<CoordType>());
600
601 Tiny::Vector<CoordType, SurfaceMeshType::shape_dim+1, sc_> current_coeffs(CoordType(0));
602
603 // Search all cells
604 while(!cell_stack.empty())
605 {
606 // Get new cell
607 Index current(cell_stack.front());
608 cell_stack.pop_front();
609 _cell_marker[current] = true;
610
611 // Collect all vertices that make up the cell in the coords matrix
612 for(int j(0); j < SurfaceMeshType::shape_dim+1; ++j)
613 {
614 Index i(idx(current, j));
615 coords[j] = vtx[i];
616 }
617
618 // Compute the coefficients of the inverse coordinate transformation. The last coefficient is the
619 // signed distance of x to the facet made up by coords
620 compute_inverse_coeffs(current_coeffs, x, coords);
621
622 // From the coefficients, we can compute the barycentric coordinates (as the surface mesh is of Simplex
623 // shape) so we get an idea about in which direction we need to search first
624 Tiny::Vector<CoordType, SurfaceMeshType::shape_dim+1> bary(CoordType(1));
625 CoordType ortho_dist = compute_bary_and_dist(bary, current_coeffs);
626
627 // From all this, we can now determine if there is an orthogonal projection of x onto this facet. For
628 // this, we need to know the traversed information as well and the neighbor information of the current
629 // facet
630 bool success = determine_success(ortho_dist, acceptable_distance, bary, facet_idx[current], _edge_marker);
631
632 // Ok, maybe we are done
633 if(success && !find_best_approx)
634 {
635 best_approx_cell = current;
636 coeffs = current_coeffs;
637 break;
638 }
639 // Check if we need to update the best approx data
640 else if(success && find_best_approx)
641 {
642 if(ortho_dist < best_approx_dist)
643 {
644 best_approx_dist = ortho_dist;
645 best_approx_cell = current;
646 coeffs = current_coeffs;
647
648 }
649 }
650
651 // If we are not done, add all neighbors that have not been searched to the search stack
652 for(int l(0); l < SurfaceMeshType::shape_dim+1; ++l)
653 {
654 // The neighbor over the l-th facet
655 Index n(neigh(current, l));
656 // Check if the neighbor exists and is still on the to do list
657 if(n != ~ Index(0) && !_cell_marker[n])
658 {
659 Index facet(facet_idx(current,l));
660 // If the neighbor is still on the to do list and the facet it lies at is in the traversed list,
661 // this means the barycentric coordinate for the corresponding direction was negative, meaning
662 // that x lies in that direction. So we add that neighbor to the front.
663 if(_edge_marker[facet])
664 cell_stack.push_front(n);
665 else
666 cell_stack.push_back(n);
667
668 // Since we added the neighbor to the stack, prevent it from being added again later.
669 _cell_marker[n] = true;
670 }
671 } // adding neighbors
672
673 } // search loop
674
675 // If we get here without success being true, the point could not be found
676 XASSERTM(best_approx_cell == ~Index(0), "Could not find point "+stringify(x)+" in SurfaceMesh");
677
678 // Otherwise, it was found in current and coeffs is still set from that evaluation
679 return best_approx_cell;
680 } // find_cell()

732 template<typename DT_, int n_, int sn_, int sb_>
733 static DT_ compute_bary_and_dist(Tiny::Vector<DT_, n_, sb_>& bary, const Tiny::Vector<DT_, n_, sn_>& coeffs)
734 {
735 bary(0) = CoordType(1);
736 for(int j(0); j < n_-1; ++j)
737 {
738 bary(0) -= coeffs(j);
739 bary(j+1) = coeffs(j);
740 }
741
742 return Math::abs(coeffs(n_-1));
743 }

809 template<typename DT_, int n_, int sn_, typename IdxType_>
810 static bool determine_success(
811 DT_ ortho_dist,
812 DT_ acceptable_distance,
813 Tiny::Vector<DT_, n_, sn_>& bary,
814 const IdxType_& facet_idx,
815 bool* traversed)
816 {
817 static const DT_ tol = Math::sqrt(Math::eps<DT_>());
818 bool success(true);
819
820 // Determine success
821 if(ortho_dist > acceptable_distance)
822 success = false;
823
824 int num_negative(0);
825
826 for(int l(0); l < n_; ++l)
827 {
828 if(bary(l) < - tol)
829 {
830 Index facet(facet_idx[l]);
831 ++num_negative;
832 if(traversed[facet])
833 {
834 bary(l) = CoordType(0);
835 }
836 else
837 {
838 success = false;
839 traversed[facet] = true;
840 }
841 }
842 }
843
844 if(num_negative > 1)
845 success = false;
846
847 return success;
848 }

856 template<typename DT_, int sb_, int sp_, int smx_, int snx_>
857 void compute_inverse_coeffs(
858 Tiny::Vector<DT_, 3, sb_>& coeffs,
859 const Tiny::Vector<DT_, 3, sp_>& point,
860 const Tiny::Matrix<DT_, 3, 3, smx_, snx_>& x) const
861 {
862 coeffs.format(DT_(0));
863
864 // A will contain the transformation matrix
865 Tiny::Matrix<DT_, 3, 3> A(DT_(0));
866 // Fill the all rows in the first shape_dim = world_dim-1 columns
867 for(int i(0); i < 3; ++i)
868 {
869 for(int j(0); j < 2; ++j)
870 A(i,j) = x(j+1,i) - x(0,i);
871 }
872
873 // The last column is the additional direction for our augmented simplex and it is orthogonal to the rest
874 Tiny::Vector<DT_, 3> ortho = Tiny::orthogonal(A.template size_cast<3, 2>());
875 // Normalize this so the last coefficient is the signed distance
876 ortho.normalize();
877
878 // Set the last column in A
879 for(int i(0); i < 3; ++i)
880 A(i,2) = ortho(i);
881
882 // This will be the inverse of the transformation matrix
883 Tiny::Matrix<DT_, 3 , 3> Ainv(DT_(0));
884 Ainv.set_inverse(A);
885
886 // This is the solution of A u = point - x[0]
887 coeffs = Ainv*(point - x[0]);
888 }

895 template<typename DT_, int world_dim, int sc_, int sp_, int smx_, int snx_>
896 void compute_inverse_coeffs(
897 Tiny::Vector<DT_, 2, sc_>& coeffs,
898 const Tiny::Vector<DT_, world_dim, sp_>& point,
899 const Tiny::Matrix<DT_, 2, world_dim, smx_, snx_>& x) const
900 {
901 static_assert( (world_dim == 2 || world_dim == 3),
902 "world dim has to be 2 or 3 for complementary barycentric coordinates");
903
904 auto tmp = x[1]-x[0];
905 DT_ sp(Tiny::dot(point - x[0],tmp));
906 DT_ nsqr(Math::sqr(tmp.norm_euclid()));
907 coeffs[0] = sp/nsqr;
908 tmp = point - (x[0] + coeffs[0]*(x[1]-x[0]));
909 coeffs[1] = tmp.norm_euclid();
910 }
