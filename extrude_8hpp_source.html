118 void transform_2d_to_3d(WorldPoint& p, const SubWorldPoint& q, const CoordType z = CoordType(0)) const
119 {
120 // apply our extrude transformation:
121 // p = w + R * (q - v)
122 WorldPoint x;
123 x[0] = q[0] - _origin[0];
124 x[1] = q[1] - _origin[1];
125 x[2] = z;
126 p.set_mat_vec_mult(_rotation, x);
127 p += _offset;
128 }
129
130 void transform_3d_to_2d(const WorldPoint& p, SubWorldPoint& q, CoordType& z) const
131 {
132 // apply our inverse extrude transformation:
133 // q = v + R^{-1} * (p - w)
134 WorldPoint x = p - _offset;
135 WorldPoint y;
136 // rotation matrices are orthogonal, i.e. R^{-1} = R^T
137 y.set_vec_mat_mult(x, _rotation);
138 q[0] = y[0] - _origin[0];
139 q[1] = y[1] - _origin[1];
140 z = y[2];
141 }
142
143 void transform_3d_to_2d(const WorldPoint& p, SubWorldPoint& q) const
144 {
145 CoordType z = CoordType(0);
146 transform_3d_to_2d(p, q, z);
147 }
148
149 virtual void transform(const WorldPoint& origin, const WorldPoint& angles, const WorldPoint& offset) override
150 {
151 // the rotation matrix
152 Tiny::Matrix<CoordType, 3, 3> rotation;
153 rotation.set_rotation_3d(angles[0], angles[1], angles[2]);
154
155 // we now have 2 transformations:
156 // 1. from the extrude chart itself and
157 // 2. from the parameters given to this function
158 //
159 // The new extrusion chart transformation is given by the concatenation of the
160 // two above transformations:
161 //
162 // v2 + R2 * ( (v1 + R1*(x - w1)) - w2)
163 // = v2 + R2*(v1 - w2) + R2*R1*(x - w1)
164 // \---------------/ \---/ \/
165 // v3 R3 w3
166 //
167 // where
168 // vi = offset
169 // Ri = rotation
170 // wi = origin
171
172 WorldPoint tmp1, tmp2;
173
174 // let us first update the offset vector v3:
175 // compute offset vector v3 := v2 + R1*(v1 - w2)
176 tmp1 = _offset - origin;
177 tmp2.set_mat_vec_mult(rotation, tmp1);
178 _offset = offset + tmp2;
179
180 // compute rotation matrix R3 := R2*R1
181 Tiny::Matrix<CoordType, 3, 3> old_rot(_rotation);
182 _rotation.set_mat_mat_mult(rotation, old_rot);
183
184 // our origin vector w3 remains the same (= w1)
185 }

187 void project_point(WorldPoint& point) const
188 {
189 // transform to 2d
190 SubWorldPoint sub_point;
191 CoordType z = CoordType(0);
192 transform_3d_to_2d(point, sub_point, z);
193 // project in 2d
194 _sub_chart->project_point(sub_point);
195 // transform to 3d
196 transform_2d_to_3d(point, sub_point, z);
197 }

199 void project_meshpart(Mesh_& mesh, const MeshPart<Mesh_>& meshpart) const
200 {
201 auto& vtx = mesh.get_vertex_set();
202 const auto& target_vtx = meshpart.template get_target_set<0>();
203 for(Index i(0); i < meshpart.get_num_entities(0); ++i)
204 {
205 project_point(reinterpret_cast<WorldPoint&>(vtx[target_vtx[i]]));
206 }
207 }

281 virtual void write(std::ostream& os, const String& sindent) const override
282 {
283 String sind(sindent);
284 if(!sind.empty())
285 sind.append(" ");
286
287 const CoordType tol = Math::pow(Math::eps<CoordType>(), CoordType(0.7));
288
289 os << sindent << "<Extrude";
290 if(_origin.norm_euclid_sqr() > tol)
291 os << " origin=\"" << _origin[0] << " " << _origin[1] << "\"";
292 if(_offset.norm_euclid_sqr() > tol)
293 os << " offset=\"" << _offset[0] << " " << _offset[1] << " " << _offset[2] << "\"";
294
295 // check whether our rotation matrix is the identity matrix
296 if(Math::sqr(_rotation.norm_sub_id_frobenius()) > tol)
297 {
298 // Now that's tricky: we have to reconstruct the yaw-pitch-roll angles
299 // from the rotation matrix: see \cite Craig04, page 43, eq. (2.66) -- (2.68)
300 const CoordType pi = Math::pi<CoordType>();
301 const CoordType pi_2 = CoordType(0.5) * pi;
302 const CoordType r11 = _rotation(0,0);
303 const CoordType r12 = _rotation(0,1);
304 const CoordType r21 = _rotation(1,0);
305 const CoordType r22 = _rotation(1,1);
306 const CoordType r31 = _rotation(2,0);
307 const CoordType r32 = _rotation(2,1);
308 const CoordType r33 = _rotation(2,2);
309 CoordType ay = CoordType(0); // yaw (in book: alpha)
310 CoordType ap = CoordType(0); // pitch (in book: beta)
311 CoordType ar = CoordType(0); // roll (in book: gamma)
312
313 // determine the pitch first
314 ap = Math::atan2(-r31, Math::sqrt(r11*r11 + r21*r21));
315
316 // check pitch for "gimbal lock" angles +pi/2 and -pi/2
317 if(ap >= pi_2 - tol)
318 {
319 ap = pi_2;
320 ar = Math::atan2(r12, r22);
321 }
322 else if(ap <= -pi_2 + tol)
323 {
324 ap = -pi_2;
325 ar = -Math::atan2(r12, r22);
326 }
327 else // no gimbal lock
328 {
329 // Note: we can safely drop the "c\beta" denominators from the book,
330 // as these always cancel out by definition of atan2
331 ay = Math::atan2(r21, r11);
332 ar = Math::atan2(r32, r33);
333 }
334
335 // convert angles from radians to revolutions and write out
336 const CoordType mult = CoordType(0.5) / pi;
337 os << " angles=\"" << (mult*ay) << " " << (mult*ap) << " " << (mult*ar) << "\"";
338 }
339 os << ">" << std::endl;
340
341 _sub_chart->write(os, sind);
342 os << sindent << "</Extrude>" << std::endl;
343 }

391 virtual void create(
392 int iline,
393 const String& sline,
394 const String&,
395 const std::map<String, String>& attrs,
396 bool) override
397 {
398 // parse 2D origin (if given)
399 {
400 auto it = attrs.find("origin");
401 if(it != attrs.end())
402 {
403 std::deque<String> sori = it->second.split_by_whitespaces();
404 if(sori.size() != std::size_t(2))
405 throw Xml::GrammarError(iline, sline, "Invalid Extrude chart origin attribute");
406 if(!sori.front().parse(_ori_x) || !sori.back().parse(_ori_y))
407 throw Xml::GrammarError(iline, sline, "Failed to parse extrude origin attribute");
408
409 }
410 }
411
412 // parse 3D offset (if given)
413 {
414 auto it = attrs.find("offset");
415 if(it != attrs.end())
416 {
417 std::deque<String> soff = it->second.split_by_whitespaces();
418 if(soff.size() != std::size_t(3))
419 throw Xml::GrammarError(iline, sline, "Invalid Extrude chart offset attribute");
420 if(!soff.at(0).parse(_off_x) || !soff.at(1).parse(_off_y) || !soff.at(2).parse(_off_z))
421 throw Xml::GrammarError(iline, sline, "Failed to parse extrude offset attribute");
422
423 }
424 }
425
426 // parse angles (if given)
427 {
428 auto it = attrs.find("angles");
429 if(it != attrs.end())
430 {
431 std::deque<String> sang = it->second.split_by_whitespaces();
432 if(sang.size() != std::size_t(3))
433 throw Xml::GrammarError(iline, sline, "Invalid Extrude chart angles attribute");
434 if(!sang.at(0).parse(_yaw) || !sang.at(1).parse(_pitch) || !sang.at(2).parse(_roll))
435 throw Xml::GrammarError(iline, sline, "Failed to parse extrude angles attribute");
436
437 // Note: the angles are specifies in revolutions, but we need radians, so multiply by 2*pi
438 const CoordType mult = CoordType(2) * Math::pi<CoordType>();
439 _yaw *= mult;
440 _pitch *= mult;
441 _roll *= mult;
442 }
443 }
444 }

466 std::unique_ptr<Extrude<Mesh_, Circle<SubMeshType>>> ext(new Extrude<Mesh_, Circle<SubMeshType>>());
467 ext->set_origin(_ori_x, _ori_y);
468 ext->set_offset(_off_x, _off_y, _off_z);
469 ext->set_angles(_yaw, _pitch, _roll);
470 std::unique_ptr<Circle<SubMeshType>>& sub_chart = ext->_sub_chart;

473 }
474 if(name == "Bezier")

481 _chart = std::move(ext);
482 return std::make_shared<Atlas::BezierChartParser<SubMeshType, Bezier<SubMeshType>>>(sub_chart);
483 }

485 return nullptr;
