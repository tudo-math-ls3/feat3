110 explicit ExtVtkWriter(const TrafoType& trafo, Index num_refines = 1) :
111 _trafo(trafo),
112 _ref_mesh(nullptr)
113 {
114 // create refined reference cell mesh
115 typedef Geometry::ReferenceCellFactory<ShapeType> RefCellFactory;
116 typedef Geometry::StandardRefinery<RefMeshType> RefCellRefinery;
117 RefCellFactory ref_factory;
118 RefMeshType* ref_mesh = new RefMeshType(ref_factory);
119 for(Index i(0); i < num_refines; ++i)
120 {
121 RefMeshType* old_mesh = ref_mesh;
122 {
123 RefCellRefinery refinery(*old_mesh);
124 ref_mesh = new RefMeshType(refinery);
125 }
126 delete old_mesh;
127 }
128 _ref_mesh = ref_mesh;
129 }

142 bool open(String filename)
143 {
144 _ofs.open(filename.c_str());
145 if(!_ofs.is_open() || !_ofs.good())
146 return false;
147
148 // write VTK header
149 _ofs << "# vtk DataFile Version 2.0" << std::endl;
150 _ofs << "Generated by FEAT v" << version_major << "." << version_minor << "." << version_patch << std::endl;
151 _ofs << "ASCII" << std::endl;
152
153 // write mesh type
154 _ofs << "DATASET UNSTRUCTURED_GRID" << std::endl;
155
156 // write vertices
157 write_vertices();
158
159 // write indices
160 write_indices();
161
162 // write point data header
163 _ofs << "POINT_DATA " << (_trafo.get_mesh().get_num_entities(ShapeType::dimension)
164 * _ref_mesh->get_num_entities(0)) << std::endl;
165
166 // okay
167 return true;
168 }

190 template<typename Space_, typename VectorType_>
191 void write_values_blocked(String name, const Space_& space, const VectorType_& v)
192 {
193 typedef Space_ SpaceType;
194 typedef typename SpaceType::template Evaluator<TrafoEval>::Type SpaceEval;
195 //static_assert(SpaceEval::can_value != 0, "space cannot evaluate basis function values!");
196 typedef typename SpaceType::DofMappingType DofMappingType;
197 typedef typename SpaceEval::template ConfigTraits<SpaceTags::value> SpaceConfigTraits;
198 typedef typename SpaceConfigTraits::EvalDataType SpaceData;
199 static constexpr TrafoTags trafo_value_config = SpaceConfigTraits::trafo_config;
200 typedef typename TrafoEval::template ConfigTraits<trafo_value_config>::EvalDataType TrafoData;
201 typedef typename VectorType_::ValueType ValueType;
202 Tiny::Vector<ValueType, SpaceEval::max_local_dofs> loc_vec;
203
204 DofMappingType dof_map(space);
205 TrafoEval trafo_eval(_trafo);
206 SpaceEval space_eval(space);
207 TrafoData trafo_data;
208 SpaceData space_data;
209
210 // write basic info
211 _ofs << "VECTORS " << name << " double" << std::endl;
212
213 // loop over all cells
214 for(Index cell(trafo_eval.begin()); cell != trafo_eval.end(); ++cell)
215 {
216 // gather local vector
217 dof_map.prepare(cell);
218 for(int i(0); i < dof_map.get_num_local_dofs(); ++i)
219 {
220 loc_vec[i] = v(dof_map.get_index(i));
221 }
222 dof_map.finish();
223
224 // prepare evaluators
225 trafo_eval.prepare(cell);
226 space_eval.prepare(trafo_eval);
227
228 // loop over all points
229 for(Index pt(0); pt < _ref_mesh->get_num_entities(0); ++pt)
230 {
231 // get domain point
232 typename TrafoEval::DomainPointType dom_point;
233 get_dom_point(dom_point, pt);
234
235 // evaluate trafo and space
236 trafo_eval(trafo_data, dom_point);
237 space_eval(space_data, trafo_data);
238
239 // compute value
240 ValueType value(0);
241 for(int i(0); i < space_eval.get_num_local_dofs(); ++i)
242 value += space_data.phi[i].value * loc_vec[i];
243
244 // write value
245 for(int d(0); d < ValueType::n; ++d)
246 _ofs << " " << value(d);
247
248 for(int d(ValueType::n); d < 3; ++d)
249 _ofs << " 0.0";
250 _ofs << std::endl;
251 }
252
253 // finish evaluators
254 space_eval.finish();
255 trafo_eval.finish();
256 }

519 void write_vertices()
520 {
521 // fetch index set and vertex set
522 const TrafoType& trafo = _trafo;
523 const MeshType& mesh = trafo.get_mesh();
524
525 // write vertex set
526 const Index num_cells = mesh.get_num_entities(ShapeType::dimension);
527 const Index num_ref_verts = num_cells * _ref_mesh->get_num_entities(0);
528 _ofs << "POINTS " << num_ref_verts << " double" << std::endl;
529
530 // create trafo evaluator and data
531 typedef typename TrafoEval::DomainPointType DomainPointType;
532 typedef typename TrafoEval::template ConfigTraits<TrafoTags::img_point>::EvalDataType TrafoData;
533 DomainPointType dom_point;
534 TrafoEval trafo_eval(trafo);
535 TrafoData trafo_data;
536
537 // loop over all cells
538 for(Index cell(trafo_eval.begin()); cell != trafo_eval.end(); ++cell)
539 {
540 // prepare trafo
541 trafo_eval.prepare(cell);
542
543 // loop over all reference vertices
544 for(Index i(0); i < _ref_mesh->get_num_entities(0); ++i)
545 {
546 // get reference vertex
547 get_dom_point(dom_point, i);
548
549 // apply trafo
550 trafo_eval(trafo_data, dom_point);
551
552 // write image coords
553 _ofs << trafo_data.img_point[0];
554 for(int j(1); j < ShapeType::dimension; ++j)
555 _ofs << " " << trafo_data.img_point[j];
556 for(int j(ShapeType::dimension); j < 3; ++j)
557 _ofs << " 0.0";
558 _ofs << std::endl;
559 }
560
561 // finish trafo
562 trafo_eval.finish();

566 void write_indices()
567 {
568 // fetch index set and vertex set
569 const TrafoType& trafo = _trafo;
570 const MeshType& mesh = trafo.get_mesh();
571
572 // write index set
573 const Index num_cells = mesh.get_num_entities(ShapeType::dimension);
574 const Index num_ref_verts = _ref_mesh->get_num_entities(0);
575 const Index num_ref_cells = _ref_mesh->get_num_entities(ShapeType::dimension);
576 const typename RefMeshType::template IndexSet<ShapeType::dimension,0>::Type& idx =
577 _ref_mesh->template get_index_set<ShapeType::dimension, 0>();
578 int num_idx = idx.get_num_indices();
579
580 typedef Geometry::Intern::VTKShape<ShapeType> VTKHelperType;
581
582 _ofs << "CELLS " << (num_cells*num_ref_cells) << " " << (Index(num_idx+1)*num_cells*num_ref_cells) << std::endl;
583
584 // loop over all mesh cells
585 for(Index cell(0); cell < num_cells; ++cell)
586 {
587 // compute vertex offset
588 Index voff = num_ref_verts * cell;
589 for(Index i(0); i < num_ref_cells; ++i)
590 {
591 _ofs << num_idx;
592 for(int j(0); j < num_idx; ++j)
593 {
594 _ofs << " " << voff + idx(i,VTKHelperType::map(j));
595 }
596 _ofs << std::endl;
597 }
598 }
599
600 // write cell types
601 _ofs << "CELL_TYPES " << (num_cells*num_ref_cells) << std::endl;
602
603 // loop over all mesh cells
604 for(Index cell(0); cell < num_cells*num_ref_cells; ++cell)
605 {
606 _ofs << VTKHelperType::type << std::endl;
607 }
608 }
