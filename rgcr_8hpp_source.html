176 protected:
177 virtual Status _apply_intern(VectorType& vec_sol)
178 {
179 Statistics::add_solver_expression(std::make_shared<ExpressionStartSolve>(this->name()));
180
181 const MatrixType& matrix(this->_system_matrix);
182 const FilterType& filter(this->_system_filter);
183 VectorType& vec_r(this->_vec_r);
184 VectorType& vec_p_hat(this->_vec_p_hat);
185 VectorType& vec_q_hat(this->_vec_q_hat);
186
187 // set initial defect:
188 // r[0] := b - A*x[0]
189 Status status = this->_set_initial_defect(vec_r, vec_sol);
190 if(status != Status::progress)
191 {
192 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), status, this->get_num_iter()));
193 return status;
194 }
195
196 // start iterating
197 while(status == Status::progress)
198 {
199 IterationStats stat(*this);
200
201 // no stored descent vectors left, calculate new one
202 if (this->_num_iter >= p_list.size())
203 {
204 // apply preconditioner to defect vector
205 if(!this->_apply_precond(vec_p_hat, vec_r, filter))
206 {
207 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
208 return Status::aborted;
209 }
210
211 //vec_p_hat.copy(vec_r);
212 matrix.apply(vec_q_hat, vec_p_hat);
213 filter.filter_def(vec_q_hat);
214 for (Index j(0) ; j < this->_num_iter ; ++j)
215 {
216 auto beta = vec_q_hat.dot(q_list.at(j));
217 vec_q_hat.axpy(q_list.at(j), -beta);
218 vec_p_hat.axpy(p_list.at(j), -beta);
219 }
220 const auto gamma = DataType(1) / vec_q_hat.norm2();
221 vec_q_hat.scale(vec_q_hat, gamma);
222 q_list.push_back(vec_q_hat.clone());
223 vec_p_hat.scale(vec_p_hat, gamma);
224 p_list.push_back(vec_p_hat.clone());
225 }
226
227 auto alpha = vec_r.dot(q_list.at(this->_num_iter));
228 vec_sol.axpy(p_list.at(this->_num_iter), alpha);
229 vec_r.axpy(q_list.at(this->_num_iter), -alpha);
230 // compute defect norm
231 status = this->_set_new_defect(vec_r, vec_sol);
232 if(status != Status::progress)
233 {
234 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), status, this->get_num_iter()));
235 return status;
236 }
237 }
238
239 // we should never reach this point...
240 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::undefined, this->get_num_iter()));
241 return Status::undefined;
242 }
