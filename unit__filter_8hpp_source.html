246 void filter_mat(SparseMatrixCSR<DT_, IT_> & matrix) const
247 {
248 if(_sv.used_elements() == Index(0))
249 return;
250
251 XASSERTM(_sv.size() == matrix.rows(), "Matrix size does not match!");
252
253 const IndexType* row_ptr(matrix.row_ptr());
254 const IndexType* col_idx(matrix.col_ind());
255 DT_* v(matrix.val());
256
257 for(Index i(0); i < _sv.used_elements(); ++i)
258 {
259 Index ix(_sv.indices()[i]);
260 // replace by unit row
261 for(IndexType j(row_ptr[ix]); j < row_ptr[ix + 1]; ++j)
262 {
263 v[j] = (col_idx[j] == ix) ? DT_(1) : DT_(0);
264 }
265 }
266 }

338 void filter_weak_matrix_rows(SparseMatrixCSR<DT_, IT_> & matrix_a, const SparseMatrixCSR<DT_, IT_>& matrix_m) const
339 {
340 if(_sv.used_elements() == Index(0))
341 return;
342
343 XASSERTM(_sv.size() == matrix_a.rows(), "Matrix size does not match!");
344 XASSERTM(_sv.size() == matrix_m.rows(), "Matrix size does not match!");
345
346 const IndexType* row_ptr(matrix_a.row_ptr());
347 const IndexType* col_idx(matrix_m.col_ind());
348 XASSERTM(row_ptr == matrix_m.row_ptr(), "matrix A and M must share their layout");
349 XASSERTM(col_idx == matrix_m.col_ind(), "matrix A and M must share their layout");
350
351 DT_* val_a(matrix_a.val());
352 const DT_* val_m(matrix_m.val());
353
354 const IT_* idx = get_indices();
355 const DT_* val = get_values();
356
357 // loop over all filter entries
358 for(Index i(0); i < _sv.used_elements(); ++i)
359 {
360 // replace row of A by scaled row of M
361 Index row(idx[i]);
362 for(IndexType j(row_ptr[row]); j < row_ptr[row + 1]; ++j)
363 {
364 val_a[j] = val[i] * val_m[j];
365 }
366 }
367 }
