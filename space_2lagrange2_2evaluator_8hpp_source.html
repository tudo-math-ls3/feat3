30 // p0, p1 and p2 are the 1D basis functions on the reference interval [-1,+1].
31 // These are used for the tensor-product approach in the Hypercube evaluators.
32
33 // basis function for left vertex
34 template<typename T_>
35 inline T_ p0(T_ x)
36 {
37 return T_(0.5) * x * (x - T_(1));
38 }
39
40 // basis function for right vertex
41 template<typename T_>
42 inline T_ p1(T_ x)
43 {
44 return T_(0.5) * x * (x + T_(1));
45 }
46
47 // basis function for edge midpoint
48 template<typename T_>
49 inline T_ p2(T_ x)
50 {
51 return (T_(1) - x) * (T_(1) + x);
52 }
53
54 // first order derivatives
55
56 template<typename T_>
57 inline T_ d1p0(T_ x)
58 {
59 return x - T_(0.5);
60 }
61
62 template<typename T_>
63 inline T_ d1p1(T_ x)
64 {
65 return x + T_(0.5);
66 }
67
68 template<typename T_>
69 inline T_ d1p2(T_ x)
70 {
71 return -T_(2) * x;
72 }
73
74 // second order derivatives
75
76 template<typename T_>
77 inline T_ d2p0(T_)
78 {
79 return T_(1);
80 }
81
82 template<typename T_>
83 inline T_ d2p1(T_)
84 {
85 return T_(1);
86 }
87
88 template<typename T_>
89 inline T_ d2p2(T_)
90 {
91 return -T_(2);
92 }
