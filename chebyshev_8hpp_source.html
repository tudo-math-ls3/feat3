178 virtual void init_numeric() override
179 {
180 BaseClass::init_numeric();

182 //PowerMethod for maximum eigenvalue
183 //http://blogs.sas.com/content/iml/2012/05/09/the-power-method.html
184 //https://en.wikipedia.org/wiki/Power_iteration
185 VectorType& v(this->_vec_cor);
186 VectorType& z(this->_vec_def);
187 const MatrixType& matrix(this->_system_matrix);
188 v.format(DataType(3));
189 DataType tolerance(DataType(1e-4));
190 Index max_iters(40);
191 v.scale(v, DataType(1) / v.norm2());
192 DataType lambda_old(0), lambda(0);
193 for (Index i(0) ; i < max_iters ; ++i)
194 {
195 matrix.apply(z, v);
196 v.scale(z, DataType(1) / z.norm2());
197 lambda = v.dot(z);
198 if (Math::abs((lambda - lambda_old) / lambda) < tolerance)
199 break;
200 lambda_old = lambda;
201 }
202
203 _min_ev = lambda * _fraction_min_ev;
204 _max_ev = lambda * _fraction_max_ev;
205 }

221 virtual Status correct(VectorType& vec_sol, const VectorType& vec_rhs) override

223 // compute defect

225 this->_system_filter.filter_def(this->_vec_def);

233 protected:
234 virtual Status _apply_intern(VectorType& vec_sol, const VectorType& vec_rhs)
235 {
236 Statistics::add_solver_expression(std::make_shared<ExpressionStartSolve>(this->name()));
237
238 VectorType& vec_def(this->_vec_def);
239 VectorType& vec_cor(this->_vec_cor);
240 const MatrixType& matrix(this->_system_matrix);
241 const FilterType& filter(this->_system_filter);
242
243 // compute initial defect
244 Status status = this->_set_initial_defect(vec_def, vec_sol);
245
246 DataType d = (_max_ev + _min_ev) / DataType(2);
247 DataType c = (_max_ev - _min_ev) / DataType(2);
248 DataType alpha(0);
249 DataType beta(0);
250 vec_cor.scale(vec_def, (DataType(1) / d));
251
252 // start iterating
253 while(status == Status::progress)
254 {
255 switch (this->get_num_iter())
256 {
257 case 0:
258 alpha = DataType(1) / d;
259 break;
260 case 1:
261 alpha = DataType(2) * d * (DataType(1) / ((DataType(2) * d * d) - (c * c)));
262 break;
263 default:
264 alpha = DataType(1) / (d - ((alpha * c * c) / DataType(4)));
265 }
266
267 beta = (alpha * d) - DataType(1);
268 vec_cor.scale(vec_cor, beta);
269 vec_cor.axpy(vec_def, vec_cor, alpha);
270
271 // update solution vector
272 vec_sol.axpy(vec_cor, vec_sol);
273
274 // compute new defect vector
275 matrix.apply(vec_def, vec_sol, vec_rhs, -DataType(1));
276 filter.filter_def(vec_def);
277
278 // compute new defect norm
279 status = this->_set_new_defect(vec_def, vec_sol);
280 }
281
282 // return our status
283 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), status, this->get_num_iter()));
284 return status;
285 }
