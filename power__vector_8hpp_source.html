255 void restore_from_checkpoint_data(std::vector<char> & data)
256 {
257 std::uint64_t isize = *(std::uint64_t*) data.data(); //get size of checkpointed _first
258 std::vector<char>::iterator start = std::begin(data) + sizeof(std::uint64_t);
259 std::vector<char>::iterator last_of_first = std::begin(data) + sizeof(std::uint64_t) + (int) isize;
260 std::vector<char> buffer_first(start, last_of_first);
261 _first.restore_from_checkpoint_data(buffer_first);
262
263 data.erase(std::begin(data), last_of_first);
264 _rest.restore_from_checkpoint_data(data);
265 }
266
268 std::uint64_t set_checkpoint_data(std::vector<char>& data, SerialConfig& config)
269 {
270 std::size_t old_size = data.size();
271 data.insert(std::end(data), sizeof(std::uint64_t), 0); //add placeholder
272 std::uint64_t ireal_size = _first.set_checkpoint_data(data, config); //add data of _first to the overall checkpoint and save its size
273 char* csize = reinterpret_cast<char*>(&ireal_size);
274 for(std::size_t i(0) ; i < sizeof(std::uint64_t) ; ++i) //overwrite the guessed datalength
275 {
276 data[old_size +i] = csize[i];
277 }
278
279 return sizeof(std::uint64_t) + ireal_size + _rest.set_checkpoint_data(data, config); //generate and add checkpoint data for the _rest
280 }

602 void read_from(FileMode mode, std::istream& file)
603 {
604 switch(mode)
605 {
606 case FileMode::fm_binary:
607 {
608 std::uint64_t magic; // magic_number
609 file.read((char *)&magic, (long)(sizeof(std::uint64_t)));
610 if (magic != 100)
611 XABORTM("Given file or file component is no PowerVector!");
612 std::uint64_t count; // subvector count
613 file.read((char *)&count, (long)(sizeof(std::uint64_t)));
614 if (count != count_)
615 XABORTM("PowerVector file read in component count mismatch: class has " + stringify(count_) + "- " + stringify(count) + " read in!");
616
617 _read_from_binary(file);
618 break;
619 }
620 default:
621 XABORTM("Filemode not supported!");
622 }
623 }
