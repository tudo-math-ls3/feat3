148 template<typename IndexVectorType_>
149 std::pair<bool,Index> find(const IndexVectorType_& index_vector) const
150 {
151 // calculate representative
152 IndexVector representative;
153 Intern::IndexRepresentative<Shape_>::compute(representative, index_vector);
154
155 // get the corresponding representative set
156 const RepSet& rep_set = _rep_set_vec[representative[0]];
157
158 // try to find the representative
159 typename RepSet::const_iterator iter = rep_set.find(representative);
160 if(iter == rep_set.end())
161 return std::make_pair(false, Index(0));
162 else
163 return std::make_pair(true, Index((*iter)[0]));
164 }

196 template<typename IndexSet_>
197 void parse(const IndexSet_& index_set)
198 {
199 static_assert(int(IndexSet_::num_indices) == int(num_indices), "index count mismatch");
200
201 // fetch number of entities
202 const Index num_entities = index_set.get_num_entities();
203
204 // loop over all entities
205 for(Index i(0); i < num_entities; ++i)
206 {
207 // insert the index vector
208 insert(index_set[i], i);
209 }
210 }
211
222 Index enumerate()
223 {
224 Index cur_id = 0;
225
226 // loop over all index vector sets
227 Index n = Index(_rep_set_vec.size());
228 for(Index i(0); i < n; ++i)
229 {
230 typename RepSet::iterator it(_rep_set_vec[i].begin());
231 typename RepSet::iterator jt(_rep_set_vec[i].end());
232 for(; it != jt; ++it, ++cur_id)
233 {
234 // The RepSet is an std::set and its elements cannot be modified as this might screw up the ordering.
235 // As we specifically abuse the 0th entry in each set element for saving an index and exclude that
236 // from the comparison operator, it is ok to cast away the const qualifier and modify the 0th entry.
237 const_cast<IndexVector&>(*it)[0] = cur_id;
238 }
239 }
240
241 // return total number of index vectors
242 return cur_id;
243 }

259 template<
260 typename Shape_,
261 int face_dim_>
262 class IndexCalculator
263 {
264 public:
266 typedef typename Shape::FaceTraits<Shape_, face_dim_>::ShapeType FaceType;
268 typedef IndexTree<FaceType> IndexTreeType;
269
270 public:
296 template<
297 typename IndexSetIn_,
298 typename IndexSetOut_>
299 static bool compute(
300 const IndexTreeType& index_tree,
301 const IndexSetIn_& index_set_in,
302 IndexSetOut_& index_set_out)
303 {
304 // index vector reference
305 typedef typename IndexSetIn_::IndexTupleType IndexTupleTypeIn;
306 typedef Intern::FaceIndexMapping<Shape_, face_dim_, 0> FimType;
307
308 // fetch number of shapes
309 const Index num_entities = index_set_in.get_num_entities();
310
311 typename IndexTreeType::IndexVector current_face_indices;
312
313 // loop over all shapes
314 for(Index i(0); i < num_entities; ++i)
315 {
316 // get vertex-index-vector of shape i
317 const IndexTupleTypeIn& current_cell_in = index_set_in[i];
318
319 // loop over all cells of shape i
320 for(int j(0); j < IndexSetOut_::num_indices; ++j)
321 {
322 // get the vertex-index-vector of cell j:
323 // loop over all indices of the cell-vertex-vector
324 for(int k(0); k < IndexTreeType::num_indices; ++k)
325 {
326 current_face_indices[k] = current_cell_in[FimType::map(j, k)];
327 }
328
329 // try to find the index of the vector within the tree
330 std::pair<bool, Index> bi = index_tree.find(current_face_indices);
331 if(!bi.first)
332 return false;
333 index_set_out[i][j] = bi.second;
334 }
335 }
336
337 // okay, all index vectors found
338 return true;
339 }

344 template<typename IndexSetIn_, typename IndexSetOut_>
345 static void compute_vertex_subshape( const IndexSetIn_& index_set_in, IndexSetOut_& index_set_out)
346 {
347 // Type for shape to vertex@subshape mapping
348 typedef Intern::FaceIndexMapping<Shape_, face_dim_, 0> FimType;
349 // Number of shapes
350 const Index num_shapes(index_set_in.get_num_entities());
351 // Number of verticex
352 const Index num_verts(index_set_in.get_index_bound());
353 // Number of vertices per subshape
354 const int num_verts_subshape(Shape::FaceTraits<FaceType,0>::count);
355 // Number of subshapes per shape, i.e. a Simplex<3> has four Simplex<2> as subshapes
356 const int num_subshapes_shape(Shape::FaceTraits<Shape_,face_dim_>::count);
357
358 // For every vertex, the IndexTree saves which subshapes it is contained in
359 IndexTreeType my_index_tree(num_verts);
360 // This saves the global vertex numbers in the current subshape
361 typename IndexTreeType::IndexVector current_face_indices;
362
363 // Generate the IndexTree of subshapes
364 for(Index k(0); k < num_shapes; ++k)
365 {
366 for(int l(0); l < num_subshapes_shape; ++l)
367 {
368 // Get the ith index in the current subshape from the subshape to index mapping for shape k
369 for(int i(0); i < num_verts_subshape; ++i)
370 current_face_indices[i] = index_set_in[k][FimType::map(l,i)];
371
372 // Insert the current subshape into the IndexTree. We need to give an id as 2nd argument, this does not
373 // get used in any way
374 my_index_tree.insert(current_face_indices, num_verts+1);
375 }
376 }
377
378 // Enumerate the subshape entities. my_index_tree[i] is the set of all subshapes having i as first vertex.
379 // Each set k of those consists of IndexVectors v, where v[0] is the index of the subshape in the subshape
380 // numbering and the subsequent entries are the vertex numbers.
381 const Index num_subshapes(my_index_tree.enumerate());
382
383 // The output IndexSet has num_subshapes entities and the maximum index is the number of vertices
384 IndexSetOut_ my_index_set(num_subshapes, num_verts);
385 index_set_out = std::move(my_index_set);
386
387 for(Index i(0); i < num_verts; ++i)
388 {
389 // Size of the ith set in the IndexTree
390 Index n = my_index_tree.get_set_size(i);
391
392 // Iterate over the set
393 for(Index j(0); j < n; ++j)
394 {
395 // This is the index of the subshape
396 Index my_index = my_index_tree.get_index(i, j, 0);
397 index_set_out[my_index][0] = i;
398
399 // Iterate over the IndexVector that the set element j represents
400 // Skip index 0 as this is contains the index of the subshape in the subshape numbering
401 for(int k(1); k < IndexTreeType::num_indices; ++k)
402 index_set_out[my_index][k] = my_index_tree.get_index(i, j, k);
403 }
404
405 }
406 return;
407 }

461 template<typename Shape_, int face_dim_ = Shape_::dimension - 1>
462 struct RisbWrapper
463 {
464 typedef typename Shape::FaceTraits<Shape_, face_dim_>::ShapeType FaceType;
465 static constexpr int num_verts = Shape::FaceTraits<FaceType, 0>::count;
466
477 static void wrap(IndexSetHolder<Shape_>& ish)
478 {
479 auto& vert_at_subshape_index_set(ish.template get_index_set<face_dim_,0>());
480
481 // Check if vertex@subshape information is present (needed later on) and compute it if necessary
482 if(vert_at_subshape_index_set.get_num_entities() == 0)
483 {
484 IndexCalculator<Shape_, face_dim_>::compute_vertex_subshape(
485 ish.template get_index_set<Shape_::dimension, 0>(), vert_at_subshape_index_set);
486
487 // Update dimensions of other IndexSets in the IndexSetHolder
488 IndexSetHolderDimensionUpdater<Shape_::dimension, Shape_::dimension-face_dim_>::update(ish);
489 }
490
491 // recurse down
492 RisbWrapper<Shape_, face_dim_ - 1>::wrap(ish);
493
494 // get vertices-at-face index set
495 IndexSet<num_verts>& vert_adj(vert_at_subshape_index_set);
496
497 // build an index-tree from it
498 IndexTree<FaceType> idx_tree(vert_adj.get_index_bound());
499 idx_tree.parse(vert_adj);
500
501 // Call the helper. This in turn calls IndexCalculater::compute, which needs vertex\@shape and
502 // vertex@subshape information for all subshapes.
503 RisbHelper<Shape_, face_dim_>::compute(ish, idx_tree);
504 }
505 }; // RisbWrapper<Shape_, face_dim_>
