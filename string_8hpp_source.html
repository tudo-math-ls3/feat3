469 std::deque<String> split_by_charset(const String & charset) const
470 {
471 std::deque<String> words;
472 if(empty() || charset.empty())
473 {
474 return words;
475 }
476
477 // find first occurrence of split substring
478 size_type off1(find_first_not_of(charset));
479 if(off1 == npos)
480 {
481 // only delimiter characters; nothing to be extracted
482 return words;
483 }
484
485 // go splitting
486 while(off1 != npos)
487 {
488 // find next occurrence of delimiter string
489 size_type off2(find_first_of(charset, off1));
490
491 // add next split substring to vector
492 if(off2 == npos)
493 {
494 // extract last substring
495 words.push_back(substr(off1));
496 return words;
497 }
498 else
499 {
500 // extract next substring
501 words.push_back(substr(off1, (off2 == npos ? npos : off2 - off1)));
502 }
503
504 // find next occurrence of split substring
505 off1 = find_first_not_of(charset, off2);
506 }
507
508 // okay
509 return words;
510 }

541 std::deque<String> split_by_string(const String & delimiter) const
542 {
543 std::deque<String> words;
544 if(empty() || delimiter.empty())
545 return words;
546
547 // find first occurrence of delimiter substring
548 size_type off1(find(delimiter));
549 words.push_back(substr(0, off1));
550 if(off1 == npos)
551 return words;
552
553 // go splitting
554 const size_type del_len(delimiter.size());
555 while(off1 != npos)
556 {
557 // increase leading offset by delimiter length
558 off1 += del_len;
559
560 // find next substring occurrence
561 size_type off2 = find(delimiter, off1);
562 if(off2 == npos)
563 {
564 // extract last substring
565 words.push_back(substr(off1));
566 return words;
567 }
568 else
569 {
570 // extract next substring
571 words.push_back(substr(off1, off2 - off1));
572 }
573
574 // update offset
575 off1 = off2;
576 }
577
578 // okay
579 return words;
580 }

600 size_type replace_all(const String & find_string, const String & replace_string)
601 {
602 size_type flen(find_string.size());
603 size_type rlen(replace_string.size());
604 if(flen <= 0)
605 return size_type(0);
606
607 // find first occurrence of find string
608 size_type pos(find(find_string));
609 size_type counter(size_type(0));
610 while(pos != npos)
611 {
612 // replace substring
613 replace(pos, flen, replace_string);
614
615 // increment counter
616 ++counter;
617
618 // find next occurrence
619 pos = find(find_string, pos + rlen);
620 }
621
622 // return replacement count
623 return counter;
624 }

681 int compare_no_case(const String& other) const
682 {
683#ifdef FEAT_COMPILER_MICROSOFT
684 // The MS C library offers a function for this task.
685 return _stricmp(c_str(), other.c_str());
686#else
687 // reference implementation
688 std::locale loc;
689 size_type n1 = size();
690 size_type n2 = other.size();
691 size_type n = std::min(n1, n2);
692
693 // loop over all characters and compare them
694 for(size_type i = 0; i < n; ++i)
695 {
696 int k = int(std::tolower<char>((*this)[i], loc)) - int(std::tolower<char>(other[i], loc));
697 if(k < 0)
698 {
699 return -1;
700 }
701 else if(k > 0)
702 {
703 return 1;
704 }
705 }
706
707 // If we come out here, then both strings are identical for the first n characters.
708 // Now let's check whether their length is equal, too.
709 if(n1 < n2)
710 {
711 return -1;
712 }
713 else if(n1 > n2)
714 {
715 return 1;
716 }
717 else
718 {
719 return 0;
720 }
721#endif
722 }

813 bool is_one_of(const String& set, const String& sep = " ", bool case_insensitive = false) const
814 {
815 std::deque<String> sset = set.split_by_string(sep);
816 for(const auto& s : sset)
817 {
818 if(case_insensitive)
819 {
820 if(this->compare_no_case(s) == 0)
821 return true;
822 }
823 else if(this->compare(s) == 0)
824 return true;
825 }
826 return false;
827 }

1263 inline String stringify_bytes(std::uint64_t bytes, int precision = 3, int width = 7)
1264 {
1265 // the bounds are chosen such that the output will never have more than 3 leading digits
1266 if(bytes <= 999ull)
1267 return stringify(bytes).pad_front(std::size_t(width)) + " Bytes";
1268 else if(bytes <= 1'022'976ull) // = 999*1024
1269 return stringify_fp_fix(double(bytes) / (1024.), precision, width) + " KiB";
1270 else if(bytes <= 1'047'527'424ull) // = 999*1024^2
1271 return stringify_fp_fix(double(bytes) / (1048576.), precision, width) + " MiB";
1272 else if(bytes <= 1'072'668'082'176ull) // = 999*1024^3
1273 return stringify_fp_fix(double(bytes) / (1073741824.), precision, width) + " GiB";
1274 else if(bytes <= 1'098'412'116'148'224ull) // = 999*1024^4
1275 return stringify_fp_fix(double(bytes) / (1099511627776.), precision, width) + " TiB";
1276 else
1277 return stringify_fp_fix(double(bytes) / (1125899906842624.), precision, width) + " PiB";
1278 }
