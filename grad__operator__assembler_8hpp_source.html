68 template<typename DT_, typename IT_, int dim_,
69 typename SpaceTest_, typename SpaceTrial_,
70 typename CubatureFactory_>
71 static void assemble(
72 LAFEM::SparseMatrixBCSR<DT_, IT_, dim_, 1>& matrix_g,
73 const SpaceTest_& test_space,
74 const SpaceTrial_& trial_space,
75 const CubatureFactory_& cubature_factory,
76 const DT_ scale = DT_(1)
77 )
78 {
79 typedef DT_ DataType;
80 typedef IT_ IndexType;
81 static constexpr int dim = dim_;
82
83 // ensure that the matrices have the correct dimensions
84 static_assert(SpaceTest_::shape_dim == dim_, "invalid matrix block sizes");
85
86 // make sure that trial and test spaces have the same trafo
87 XASSERTM((&test_space.get_trafo()) == (&trial_space.get_trafo()),
88 "Trial and test spaces must be defined on the same trafo!");

90 typedef LAFEM::SparseMatrixBCSR<DataType, IndexType, dim, 1> MatrixG;
91
92 // assembly traits
93 typedef AsmTraits2<
94 DataType,
95 SpaceTest_,
96 SpaceTrial_,
97 TrafoTags::jac_det,
98 SpaceTags::value,
99 SpaceTags::grad> AsmTraits;
100
101 // fetch the trafo
102 const typename AsmTraits::TrafoType& trafo = test_space.get_trafo();
103
104 // create a trafo evaluator
105 typename AsmTraits::TrafoEvaluator trafo_eval(trafo);
106
107 // create space evaluators
108 typename AsmTraits::TestEvaluator test_eval(test_space);
109 typename AsmTraits::TrialEvaluator trial_eval(trial_space);
110
111 // create dof-mappings
112 typename AsmTraits::TestDofMapping test_dof_mapping(test_space);
113 typename AsmTraits::TrialDofMapping trial_dof_mapping(trial_space);
114
115 // create trafo evaluation data
116 typename AsmTraits::TrafoEvalData trafo_data;
117
118 // create space evaluation data
119 typename AsmTraits::TestEvalData test_data;
120 typename AsmTraits::TrialEvalData trial_data;
121
122 // maximum number of dofs
123 static constexpr int max_test_dofs = AsmTraits::max_local_test_dofs;
124 static constexpr int max_trial_dofs = AsmTraits::max_local_trial_dofs;
125
126 // create local matrix data
127 //typename AsmTraits::LocalMatrixType lmd;
128 typedef Tiny::Matrix<DataType, dim, 1> GEntryType;
129 Tiny::Matrix<GEntryType, max_test_dofs, max_trial_dofs> local_matrix;
130
131 // create cubature rule
132 typename AsmTraits::CubatureRuleType cubature_rule(Cubature::ctor_factory, cubature_factory);
133
134 // first of all, check whether we need to assemble the matrix structures
135 if(matrix_g.empty())
136 {
137 Adjacency::Graph graph_g(SymbolicAssembler::assemble_graph_std2(test_space, trial_space));
138 matrix_g = MatrixG(graph_g);
139 }
140 matrix_g.format();
141
142 // create matrix scatter-axpy (if needed)
143 typename MatrixG::ScatterAxpy scatter_matrix(matrix_g);
144
145 // loop over all cells of the mesh
146 for(typename AsmTraits::CellIterator cell(trafo_eval.begin()); cell != trafo_eval.end(); ++cell)
147 {
148 // prepare trafo evaluator
149 trafo_eval.prepare(cell);
150
151 // prepare space evaluators
152 test_eval.prepare(trafo_eval);
153 trial_eval.prepare(trafo_eval);
154
155 // fetch number of local dofs
156 const int num_loc_test_dofs = test_eval.get_num_local_dofs();
157 const int num_loc_trial_dofs = trial_eval.get_num_local_dofs();
158
159 // format local matrix and vector
160 local_matrix.format();
161
162 // loop over all quadrature points and integrate
163 for(int pt(0); pt < cubature_rule.get_num_points(); ++pt)
164 {
165 // compute trafo data
166 trafo_eval(trafo_data, cubature_rule.get_point(pt));
167
168 // compute basis function data
169 test_eval(test_data, trafo_data);
170 trial_eval(trial_data, trafo_data);
171
172 // trial function loop
173 for(int i(0); i < num_loc_test_dofs; ++i)
174 {
175 // test function loop
176 for(int j(0); j < num_loc_trial_dofs; ++j)
177 {
178 const DataType pv = trafo_data.jac_det * cubature_rule.get_weight(pt) * test_data.phi[i].value;
179
180 // dimension loop
181 for(int k(0); k < dim; ++k)
182 {
183 local_matrix[i][j][k][0] += trial_data.phi[j].grad[k] * pv;
184 }
185
186 // continue with next trial function
187 }
188 // continue with next test function
189 }
190 // continue with next cubature point
191 }
192
193 // finish evaluators
194 trial_eval.finish();
195 test_eval.finish();
196 trafo_eval.finish();
197
198 // initialize dof-mappings
199 test_dof_mapping.prepare(cell);
200 trial_dof_mapping.prepare(cell);
201
202 // scatter into matrix
203 scatter_matrix(local_matrix, test_dof_mapping, trial_dof_mapping, scale);
204
205 // finish dof mapping
206 trial_dof_mapping.finish();
207 test_dof_mapping.finish();
208
209 // continue with next cell
210 }
211 }

252 template<typename DT_, typename IT_, int dim_,
253 typename SpaceTest_, typename SpaceTrial_,
254 typename CubatureFactory_>
255 static void assemble(
256 LAFEM::DenseVectorBlocked<DT_, IT_, dim_>& vec_asm,
257 const LAFEM::DenseVector<DT_, IT_>& vec_in,
258 const SpaceTest_& test_space,
259 const SpaceTrial_& trial_space,
260 const CubatureFactory_& cubature_factory,

275 typedef LAFEM::DenseVectorBlocked<DataType, IndexType, dim> VectorAsm;
276 typedef LAFEM::DenseVector<DataType, IndexType> VectorIn;

278 // assembly traits

281 SpaceTest_,
282 SpaceTrial_,
283 TrafoTags::jac_det,
284 SpaceTags::value,

290 // create a trafo evaluator
291 typename AsmTraits::TrafoEvaluator trafo_eval(trafo);
292
293 // create space evaluators
294 typename AsmTraits::TestEvaluator test_eval(test_space);
295 typename AsmTraits::TrialEvaluator trial_eval(trial_space);
296
297 // create dof-mappings
298 typename AsmTraits::TestDofMapping test_dof_mapping(test_space);
299 typename AsmTraits::TrialDofMapping trial_dof_mapping(trial_space);
300
301 // create trafo evaluation data
302 typename AsmTraits::TrafoEvalData trafo_data;
303
304 // create space evaluation data
305 typename AsmTraits::TestEvalData test_data;
306 typename AsmTraits::TrialEvalData trial_data;
307
308 // maximum number of dofs
309 static constexpr int max_test_dofs = AsmTraits::max_local_test_dofs;
310 static constexpr int max_trial_dofs = AsmTraits::max_local_trial_dofs;
311
312 // create local vector data
313 typedef Tiny::Vector<DataType, dim> VectorValue;
314 typedef Tiny::Vector<VectorValue, max_test_dofs> LocalVectorType;
315 LocalVectorType local_vector;
316
317 typedef Tiny::Vector<DataType, max_trial_dofs> LocalVecInType;
318 LocalVecInType local_vec_in_dofs;
319
320 // our local trial function gradient
321 Tiny::Vector<DataType, dim> loc_grad_trial;
322
323 // create cubature rule
324 typename AsmTraits::CubatureRuleType cubature_rule(Cubature::ctor_factory, cubature_factory);
325
326 // Create vector-scatter-axpy for the vector to be assembled
327 typename VectorAsm::ScatterAxpy scatter_vector(vec_asm);
328 // Create vector gather axpy for the input vector
329 typename VectorIn::GatherAxpy gather_vec_in(vec_in);
330
331 // loop over all cells of the mesh
332 for(typename AsmTraits::CellIterator cell(trafo_eval.begin()); cell != trafo_eval.end(); ++cell)
333 {
334 // prepare trafo evaluator
335 trafo_eval.prepare(cell);
336
337 // prepare space evaluators
338 test_eval.prepare(trafo_eval);
339 trial_eval.prepare(trafo_eval);
340
341 // Initialize trial DoF mapping
342 trial_dof_mapping.prepare(cell);
343
344 // Gather our local input DoF
345 local_vec_in_dofs.format();
346 gather_vec_in(local_vec_in_dofs, trial_dof_mapping);
347
348 // fetch number of local dofs
349 const int num_loc_test_dofs = test_eval.get_num_local_dofs();
350 const int num_loc_trial_dofs = trial_eval.get_num_local_dofs();
351
352 // format local matrix and vector
353 local_vector.format();
354
355 // loop over all quadrature points and integrate
356 for(int pt(0); pt < cubature_rule.get_num_points(); ++pt)
357 {
358 // compute trafo data
359 trafo_eval(trafo_data, cubature_rule.get_point(pt));
360
361 // compute basis function data
362 test_eval(test_data, trafo_data);
363 trial_eval(trial_data, trafo_data);
364
365 // evaluate trial function gradient
366 loc_grad_trial.format();
367 for(int i(0); i < num_loc_trial_dofs; ++i)
368 {
369 for(int k(0); k < dim; ++k)
370 {
371 loc_grad_trial[k] += local_vec_in_dofs[i] * trial_data.phi[i].grad[k];
372 }
373 }
374
375 // test function loop
376 for(int i(0); i < num_loc_test_dofs; ++i)
377 {
378 const DataType v(trafo_data.jac_det * cubature_rule.get_weight(pt) * test_data.phi[i].value);
379
380 // dimension loop
381 for(int k(0); k < dim; ++k)
382 {
383 //local_matrix[i][j][k][0] += test_data.phi[i].grad[k] * pv;
384 // update vector
385 local_vector[i][k] += v * loc_grad_trial[k];
386 }
387 } // continue with next test function
388
389 }// continue with next cubature point
390
391 // finish evaluators
392 trial_eval.finish();
393 test_eval.finish();
394 trafo_eval.finish();
395
396 // Initialize test DoF mapping
397 test_dof_mapping.prepare(cell);
398
399 // scatter into vector
400 scatter_vector(local_vector, test_dof_mapping, scale);
401
402 // Finish all DoF mappings
403 trial_dof_mapping.finish();
404 test_dof_mapping.finish();
405
406 } // continue with next cell
407 }
