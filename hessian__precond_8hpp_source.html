85 public:
96 explicit HessianPrecond(OperatorType& op_, const FilterType& filter_) :
97 _op(op_),
98 _filter(filter_),
99 _hessian(DT_(0)),
100 _inv_hessian(DT_(0))
101 {
102 }

107 virtual ~HessianPrecond()
108 {
109 }
110
112 virtual String name() const override
113 {
114 return "HessianPrecond";
115 }
116
118 virtual void prepare(const VectorType& DOXY(vec_state), FilterType& DOXY(filter)) override
119 {
120 _op.compute_hess(_hessian);
121 _inv_hessian.set_inverse(_hessian);
122 }
123
125 virtual Status apply(VectorType& vec_cor, const VectorType& vec_def) override
126 {
127 Statistics::add_solver_expression(std::make_shared<ExpressionStartSolve>(this->name()));
128 Statistics::add_solver_expression(std::make_shared<ExpressionCallPrecond>(this->name(), _op.name()));
129
130 vec_cor(0, _inv_hessian*vec_def(0));
131 this->_filter.filter_cor(vec_cor);
132
133 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::success, 1));
134
135 return Status::success;
136 }
