243 template<typename Space_, typename Cubature_>
244 void assemble_laplace_matrix(const Space_& space, const Cubature_& cubature, const DataType nu = DataType(1))
245 {
246 // get local matrix
247 auto& loc_matrix = this->matrix_sys.local();
248
249 // assemble structure?
250 if(loc_matrix.empty())
251 {
252 Assembly::SymbolicAssembler::assemble_matrix_std1(loc_matrix, space);
253 }
254
255 // format and assemble Laplace
256 loc_matrix.format();
257 Assembly::Common::LaplaceOperator laplace_op;
258 Assembly::BilinearOperatorAssembler::assemble_matrix1(loc_matrix, laplace_op, space, cubature, nu);
259 }
