143 {
144 this->_norm_dir = vec_dir.norm2();

165 virtual Status correct(VectorType& vec_sol, const VectorType& vec_dir) override
166 {
167 this->_norm_dir = vec_dir.norm2();
168 vec_sol.axpy(vec_dir, vec_sol, _step_length);
169
170 this->_functional.prepare(vec_sol, this->_filter);
171 this->_functional.eval_fval_grad(this->_fval_min, this->_vec_grad);
172 this->_filter.filter_def(this->_vec_grad);
173
174 return Status::success;
