237 void convert(const ScalarMixedSystemLevel<dim_, D_, I_, SMA_, SMB_, SMD_, SM_, TV_, TP_> & other)

240 gate_pres.convert(other.gate_pres);
241 gate_sys.convert(other.gate_sys);
242
243 coarse_muxer_velo.convert(other.coarse_muxer_velo);
244 coarse_muxer_pres.convert(other.coarse_muxer_pres);
245 coarse_muxer_sys.convert(other.coarse_muxer_sys);
246
247 matrix_a.convert(&gate_velo, &gate_velo, other.matrix_a);
248 matrix_b.convert(&gate_velo, &gate_pres, other.matrix_b);
249 matrix_d.convert(&gate_pres, &gate_velo, other.matrix_d);

254 this->compile_system_matrix();
255 this->compile_system_transfer();
256
257 filter_velo.convert(other.filter_velo);
258 filter_pres.convert(other.filter_pres);
259
260 this->compile_system_filter();

263 template<typename DomainLevel_>
264 void assemble_gates(const Domain::VirtualLevel<DomainLevel_>& virt_dom_lvl)
265 {
266 const auto& dom_level = virt_dom_lvl.level();
267 const auto& dom_layer = virt_dom_lvl.layer();
268 const auto& space_velo = dom_level.space_velo;
269 const auto& space_pres = dom_level.space_pres;
270
271 // set the gate comm
272 this->gate_velo.set_comm(dom_layer.comm_ptr());
273 this->gate_pres.set_comm(dom_layer.comm_ptr());
274 this->gate_sys.set_comm(dom_layer.comm_ptr());
275
276 // loop over all ranks
277 for(Index i(0); i < dom_layer.neighbor_count(); ++i)
278 {
279 int rank = dom_layer.neighbor_rank(i);
280
281 // try to find our halo
282 auto* halo = dom_level.find_halo_part(rank);
283 XASSERT(halo != nullptr);
284
285 // create (empty) velocity mirror
286 VeloMirror mirror_velo;
287 Assembly::MirrorAssembler::assemble_mirror(mirror_velo, space_velo, *halo);
288
289 // create (empty) pressure mirror
290 PresMirror mirror_pres;
291 Assembly::MirrorAssembler::assemble_mirror(mirror_pres, space_pres, *halo);
292
293 // create a system mirror
294 SystemMirror mirror_sys(mirror_velo.clone(), mirror_pres.clone());
295
296 // push mirrors into gates
297 if(!mirror_velo.empty())
298 gate_velo.push(rank, std::move(mirror_velo));
299 if(!mirror_pres.empty())
300 gate_pres.push(rank, std::move(mirror_pres));
301 if(!mirror_sys.empty())
302 gate_sys.push(rank, std::move(mirror_sys));
303 }
304
305 // create local template vectors
306 LocalVeloVector tmpl_v(space_velo.get_num_dofs());
307 LocalPresVector tmpl_p(space_pres.get_num_dofs());
308 LocalSystemVector tmpl_s(tmpl_v.clone(), tmpl_p.clone());
309
310 // compile gates
311 this->gate_velo.compile(std::move(tmpl_v));
312 this->gate_pres.compile(std::move(tmpl_p));
313 this->gate_sys.compile(std::move(tmpl_s));
314 }

316 template<typename DomainLevel_>
317 void assemble_coarse_muxers(const Domain::VirtualLevel<DomainLevel_>& virt_lvl_coarse)
318 {
319 // assemble muxer parent
320 if(virt_lvl_coarse.is_parent())
321 {
322 XASSERT(virt_lvl_coarse.is_child());
323
324 const auto& layer_c = virt_lvl_coarse.layer_c();
325 const DomainLevel_& level_p = virt_lvl_coarse.level_p();
326
327 // loop over all children
328 for(Index i(0); i < layer_c.child_count(); ++i)
329 {
330 const auto* child = level_p.find_patch_part(int(i));
331 XASSERT(child != nullptr);
332 SystemMirror child_mirror_sys;
333 VeloMirror& child_mirror_v = child_mirror_sys.template at<0>();
334 PresMirror& child_mirror_p = child_mirror_sys.template at<1>();
335 Assembly::MirrorAssembler::assemble_mirror(child_mirror_v, level_p.space_velo, *child);
336 Assembly::MirrorAssembler::assemble_mirror(child_mirror_p, level_p.space_pres, *child);
337 this->coarse_muxer_velo.push_child(child_mirror_v.clone(LAFEM::CloneMode::Shallow));
338 this->coarse_muxer_pres.push_child(child_mirror_p.clone(LAFEM::CloneMode::Shallow));
339 this->coarse_muxer_sys.push_child(std::move(child_mirror_sys));
340 }
341 }
342
343 // assemble muxer child
344 if(virt_lvl_coarse.is_child())
345 {
346 const auto& layer_c = virt_lvl_coarse.layer_c();
347 const DomainLevel_& level_c = virt_lvl_coarse.level_c();
348
349 SystemMirror parent_mirror_sys;
350 VeloMirror& parent_mirror_v = parent_mirror_sys.template at<0>();
351 PresMirror& parent_mirror_p = parent_mirror_sys.template at<1>();
352
353 // manually set up an identity gather/scatter matrix
354 {
355 Index n = level_c.space_velo.get_num_dofs();
356 parent_mirror_v = ScalarMirror(n, n);
357 auto* idx = parent_mirror_v.indices();
358 for(Index i(0); i < n; ++i)
359 idx[i] = i;
360 }
361 {
362 Index n = level_c.space_pres.get_num_dofs();
363 parent_mirror_p = ScalarMirror(n, n);
364 auto* idx = parent_mirror_p.indices();
365 for(Index i(0); i < n; ++i)
366 idx[i] = i;
367 }
368
369 // set parent and sibling comms
370 this->coarse_muxer_velo.set_parent(
371 layer_c.sibling_comm_ptr(),
372 layer_c.get_parent_rank(),
373 parent_mirror_v.clone(LAFEM::CloneMode::Shallow)
374 );
375 this->coarse_muxer_pres.set_parent(
376 layer_c.sibling_comm_ptr(),
377 layer_c.get_parent_rank(),
378 parent_mirror_p.clone(LAFEM::CloneMode::Shallow)
379 );
380 this->coarse_muxer_sys.set_parent(
381 layer_c.sibling_comm_ptr(),
382 layer_c.get_parent_rank(),
383 std::move(parent_mirror_sys)
384 );
385
386 // compile muxer
387 LocalVeloVector tmpl_v(level_c.space_velo.get_num_dofs());
388 LocalPresVector tmpl_p(level_c.space_pres.get_num_dofs());
389 LocalSystemVector tmpl_s(tmpl_v.clone(), tmpl_p.clone());
390 this->coarse_muxer_velo.compile(tmpl_v);
391 this->coarse_muxer_pres.compile(tmpl_p);
392 this->coarse_muxer_sys.compile(tmpl_s);
393 }

396 template<typename DomainLevel_, typename Cubature_>
397 void assemble_velocity_transfer(
398 const Domain::VirtualLevel<DomainLevel_>& virt_lvl_fine,
399 const Domain::VirtualLevel<DomainLevel_>& virt_lvl_coarse,
400 const Cubature_& cubature)
401 {
402 // get fine and coarse domain levels
403 const DomainLevel_& level_f = *virt_lvl_fine;
404 const DomainLevel_& level_c = virt_lvl_coarse.is_child() ? virt_lvl_coarse.level_c() : *virt_lvl_coarse;
405
406 const auto& space_f = level_f.space_velo;
407 const auto& space_c = level_c.space_velo;
408
409 // get local transfer operator
410 LocalVeloTransfer& loc_trans = this->transfer_velo.local();
411
412 // get local transfer matrices
413 LocalVeloTransferMatrix& loc_prol_wrapped = loc_trans.get_mat_prol();
414 LocalVeloTransferMatrix& loc_rest_wrapped = loc_trans.get_mat_rest();
415
416 // get the unwrapped types
417 typename LocalVeloTransferMatrix::BaseClass& loc_prol = loc_prol_wrapped;
418 typename LocalVeloTransferMatrix::BaseClass& loc_rest = loc_rest_wrapped;
419
420 // assemble structure?
421 if (loc_prol.empty())
422 {
423 Assembly::SymbolicAssembler::assemble_matrix_2lvl(loc_prol, space_f, space_c);
424 }
425
426 // create a local weight vector
427 LocalVeloVector loc_vec_weight = loc_prol_wrapped.create_vector_l();
428
429 // create a scalar weight vector for the assembly
430 auto loc_scal_vec_weight = loc_prol.create_vector_l();
431
432 // get the data arrays of the weight vectors
433 auto* v_wb = loc_vec_weight.elements();
434 auto* v_ws = loc_scal_vec_weight.elements();
435
436 // assemble prolongation matrix
437 {
438 loc_prol.format();
439 loc_scal_vec_weight.format();
440
441 // assemble prolongation matrix
442 Assembly::GridTransfer::assemble_prolongation(loc_prol, loc_scal_vec_weight,
443 space_f, space_c, cubature);
444
445 // copy weights from scalar to blocked
446 for(Index i(0); i < loc_prol.rows(); ++i)
447 v_wb[i] = v_ws[i];
448
449 // synchronize blocked weight vector
450 this->gate_velo.sync_0(loc_vec_weight);
451
452 // copy weights from blocked to scalar
453 for(Index i(0); i < loc_prol.rows(); ++i)
454 v_ws[i] = v_wb[i][0];
455
456 // invert weight components
457 loc_scal_vec_weight.component_invert(loc_scal_vec_weight);
458
459 // scale prolongation matrix
460 loc_prol.scale_rows(loc_prol, loc_scal_vec_weight);
461
462 // copy and transpose
463 loc_rest = loc_prol.transpose();
464 }
465 }

551 void assemble_global_filters()
552 {
553 // Sync the filter vector in the SlipFilter
554 const VeloGate& my_col_gate(this->gate_velo);
555
556 // For all slip filters...
557 for(auto& it : filter_sys.local().template at<0>())
558 {
559
560 // Get the filter vector
561 auto& slip_filter_vector = it.second.get_filter_vector();
562
563 // If the filter is not empty (meaning the MeshPart is on our patch):
564 if(slip_filter_vector.used_elements() > 0)
565 {
566 // Temporary DenseVector for syncing
567 LocalVeloVector tmp(slip_filter_vector.size(), DataType_(0));
568
569 auto* tmp_elements = tmp.template elements<LAFEM::Perspective::native>();
570 auto* sfv_elements = slip_filter_vector.template elements<LAFEM::Perspective::native>();
571
572 // Copy sparse filter vector contents to DenseVector
573 for(Index isparse(0); isparse < slip_filter_vector.used_elements(); ++isparse)
574 {
575 Index idense(slip_filter_vector.indices()[isparse]);
576 tmp_elements[idense] = sfv_elements[isparse];
577 }
578
579 // Synchronize the temporary DenseVector
580 my_col_gate.sync_0(tmp);
581
582 // Copy sparse filter vector contents to DenseVector
583 for(Index isparse(0); isparse < slip_filter_vector.used_elements(); ++isparse)
584 {
585 Index idense(slip_filter_vector.indices()[isparse]);
586 tmp_elements[idense].normalize();
587 sfv_elements[isparse] = tmp_elements[idense];
588
589 }
590 }
591 // This happens if the non-empty MeshPart belonging to this filter is not present on this patch
592 else
593 {
594 // Temporary DenseVector for syncing
595 LocalVeloVector tmp(slip_filter_vector.size(), DataType_(0));
596 my_col_gate.sync_0(tmp);
597 }
598 }
