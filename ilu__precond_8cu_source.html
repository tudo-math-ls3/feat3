36 int pBufferSize_U;

51 void * cuda_ilu_init_symbolic(int m, int nnz, double * csrVal, int * csrRowPtr, int * csrColInd)
52 {
53 CudaIluSolveInfo * info = new CudaIluSolveInfo;
54 info->m = m;
55 info->nnz = nnz;
56
57 cudaMalloc((void**)&(info->z), m * sizeof(double));
58
59
60 cusparseStatus_t status;
61
62 cusparseCreateMatDescr(&(info->descr_M));
63 cusparseSetMatIndexBase(info->descr_M, CUSPARSE_INDEX_BASE_ZERO);
64 cusparseSetMatType(info->descr_M, CUSPARSE_MATRIX_TYPE_GENERAL);
65
66 cusparseCreateMatDescr(&(info->descr_L));
67 cusparseSetMatIndexBase(info->descr_L, CUSPARSE_INDEX_BASE_ZERO);
68 cusparseSetMatType(info->descr_L, CUSPARSE_MATRIX_TYPE_GENERAL);
69 cusparseSetMatFillMode(info->descr_L, CUSPARSE_FILL_MODE_LOWER);
70 cusparseSetMatDiagType(info->descr_L, CUSPARSE_DIAG_TYPE_UNIT);
71
72 cusparseCreateMatDescr(&(info->descr_U));
73 cusparseSetMatIndexBase(info->descr_U, CUSPARSE_INDEX_BASE_ZERO);
74 cusparseSetMatType(info->descr_U, CUSPARSE_MATRIX_TYPE_GENERAL);
75 cusparseSetMatFillMode(info->descr_U, CUSPARSE_FILL_MODE_UPPER);
76 cusparseSetMatDiagType(info->descr_U, CUSPARSE_DIAG_TYPE_NON_UNIT);
77
78 cusparseCreateCsrilu02Info(&(info->info_M));
79 cusparseCreateCsrsv2Info(&(info->info_L));
80 cusparseCreateCsrsv2Info(&(info->info_U));
81
82 status = cusparseDcsrilu02_bufferSize(Util::Intern::cusparse_handle, m, nnz,
83 info->descr_M, csrVal, csrRowPtr, csrColInd, info->info_M, &(info->pBufferSize_M));
84 if (status != CUSPARSE_STATUS_SUCCESS)
85 throw InternalError(__func__, __FILE__, __LINE__, "cusparsecsrilu02_bufferSize failed with status code: " + stringify(status));
86
87 status = cusparseDcsrsv2_bufferSize(Util::Intern::cusparse_handle, info->trans_L, m, nnz,
88 info->descr_L, csrVal, csrRowPtr, csrColInd, info->info_L, &(info->pBufferSize_L));
89 if (status != CUSPARSE_STATUS_SUCCESS)
90 throw InternalError(__func__, __FILE__, __LINE__, "cusparseDcsrsv2_bufferSize failed with status code: " + stringify(status));
91
92 status = cusparseDcsrsv2_bufferSize(Util::Intern::cusparse_handle, info->trans_U, m, nnz,
93 info->descr_U, csrVal, csrRowPtr, csrColInd, info->info_L, &(info->pBufferSize_U));
94 if (status != CUSPARSE_STATUS_SUCCESS)
95 throw InternalError(__func__, __FILE__, __LINE__, "cusparseDcsrsv2_bufferSize failed with status code: " + stringify(status));
96
97 info->pBufferSize = max(info->pBufferSize_M, max(info->pBufferSize_L, info->pBufferSize_U));
98 cudaMalloc((void**)&(info->pBuffer), info->pBufferSize_M);
99
100 status = cusparseDcsrilu02_analysis(Util::Intern::cusparse_handle, m, nnz, info->descr_M,
101 csrVal, csrRowPtr, csrColInd, info->info_M,
102 info->policy_M, info->pBuffer);
103 if (status != CUSPARSE_STATUS_SUCCESS)
104 throw InternalError(__func__, __FILE__, __LINE__, "cusparsecsrilu02_analysis failed with status code: " + stringify(status));
105 status = cusparseXcsrilu02_zeroPivot(Util::Intern::cusparse_handle, info->info_M, &(info->structural_zero));
106 if (CUSPARSE_STATUS_ZERO_PIVOT == status)
107 {
108 throw InternalError(__func__, __FILE__, __LINE__, "CUSPARSE ZERO PIVOT ERROR!");
109 }
110
111 status = cusparseDcsrsv2_analysis(Util::Intern::cusparse_handle, info->trans_L, m, nnz, info->descr_L,
112 csrVal, csrRowPtr, csrColInd, info->info_L, info->policy_L, info->pBuffer);
113 if (status != CUSPARSE_STATUS_SUCCESS)
114 throw InternalError(__func__, __FILE__, __LINE__, "cusparse_csrv_analysis failed with status code: " + stringify(status));
115
116 status = cusparseDcsrsv2_analysis(Util::Intern::cusparse_handle, info->trans_U, m, nnz, info->descr_U,
117 csrVal, csrRowPtr, csrColInd, info->info_U, info->policy_U, info->pBuffer);
118 if (status != CUSPARSE_STATUS_SUCCESS)
119 throw InternalError(__func__, __FILE__, __LINE__, "cusparse_csrv_analysis failed with status code: " + stringify(status));
120
121 cudaDeviceSynchronize();
122#ifdef FEAT_DEBUG_MODE
123 cudaError_t last_error(cudaGetLastError());
124 if (cudaSuccess != last_error)
125 throw InternalError(__func__, __FILE__, __LINE__, "CUDA error occurred in execution!\n" + stringify(cudaGetErrorString(last_error)));
126#endif
127
128 return (void*)info;
129 }

146 int cuda_ilu_apply(double * y, const double * x, double * csrVal, int * csrRowPtr, int * csrColInd, void * vinfo)
147 {
148 CudaIluSolveInfo * info = (CudaIluSolveInfo *) vinfo;
149 const double alpha = 1.;
150
151 cusparseStatus_t status = cusparseDcsrsv2_solve(Util::Intern::cusparse_handle, info->trans_L, info->m, info->nnz, &alpha, info->descr_L,
152 csrVal, csrRowPtr, csrColInd, info->info_L,
153 x, info->z, info->policy_L, info->pBuffer);
154 if (status != CUSPARSE_STATUS_SUCCESS)
155 throw InternalError(__func__, __FILE__, __LINE__, "sucparseDcsrsv2_solve failed with status code: " + stringify(status));
156
157 status = cusparseDcsrsv2_solve(Util::Intern::cusparse_handle, info->trans_U, info->m, info->nnz, &alpha, info->descr_U,
158 csrVal, csrRowPtr, csrColInd, info->info_U,
159 info->z, y, info->policy_U, info->pBuffer);
160 if (status != CUSPARSE_STATUS_SUCCESS)
161 throw InternalError(__func__, __FILE__, __LINE__, "cusparsecsr2_solve failed with status code: " + stringify(status));
162
163 cudaDeviceSynchronize();
164#ifdef FEAT_DEBUG_MODE
165 cudaError_t last_error(cudaGetLastError());
166 if (cudaSuccess != last_error)
167 throw InternalError(__func__, __FILE__, __LINE__, "CUDA error occurred in execution!\n" + stringify(cudaGetErrorString(last_error)));
168#endif
169
170 return 0;
171 }

211 void * cuda_ilub_init_symbolic(int m, int nnz, double * csrVal, int * csrRowPtr, int * csrColInd, const int blocksize)
212 {
213 double * z;
214 cudaMalloc((void**)&z, m * blocksize * sizeof(double));
215
216 cusparseMatDescr_t descr_M = 0;
217 cusparseMatDescr_t descr_L = 0;
218 cusparseMatDescr_t descr_U = 0;
219 bsrilu02Info_t info_M = 0;
220 bsrsv2Info_t info_L = 0;
221 bsrsv2Info_t info_U = 0;
222 int pBufferSize_M;
223 int pBufferSize_L;
224 int pBufferSize_U;
225 int pBufferSize;
226 void *pBuffer = 0;
227 int structural_zero;
228 const cusparseSolvePolicy_t policy_M = CUSPARSE_SOLVE_POLICY_USE_LEVEL;
229 const cusparseSolvePolicy_t policy_L = CUSPARSE_SOLVE_POLICY_USE_LEVEL;
230 const cusparseSolvePolicy_t policy_U = CUSPARSE_SOLVE_POLICY_USE_LEVEL;
231 const cusparseOperation_t trans_L = CUSPARSE_OPERATION_NON_TRANSPOSE;
232 const cusparseOperation_t trans_U = CUSPARSE_OPERATION_NON_TRANSPOSE;
233 const cusparseDirection_t dir = CUSPARSE_DIRECTION_ROW;
234
235 cusparseStatus_t status;
236
237 cusparseCreateMatDescr(&descr_M);
238 cusparseSetMatIndexBase(descr_M, CUSPARSE_INDEX_BASE_ZERO);
239 cusparseSetMatType(descr_M, CUSPARSE_MATRIX_TYPE_GENERAL);
240
241 cusparseCreateMatDescr(&descr_L);
242 cusparseSetMatIndexBase(descr_L, CUSPARSE_INDEX_BASE_ZERO);
243 cusparseSetMatType(descr_L, CUSPARSE_MATRIX_TYPE_GENERAL);
244 cusparseSetMatFillMode(descr_L, CUSPARSE_FILL_MODE_LOWER);
245 cusparseSetMatDiagType(descr_L, CUSPARSE_DIAG_TYPE_UNIT);
246
247 cusparseCreateMatDescr(&descr_U);
248 cusparseSetMatIndexBase(descr_U, CUSPARSE_INDEX_BASE_ZERO);
249 cusparseSetMatType(descr_U, CUSPARSE_MATRIX_TYPE_GENERAL);
250 cusparseSetMatFillMode(descr_U, CUSPARSE_FILL_MODE_UPPER);
251 cusparseSetMatDiagType(descr_U, CUSPARSE_DIAG_TYPE_NON_UNIT);
252
253 cusparseCreateBsrilu02Info(&info_M);
254 cusparseCreateBsrsv2Info(&info_L);
255 cusparseCreateBsrsv2Info(&info_U);
256
257 cusparseDbsrilu02_bufferSize(Util::Intern::cusparse_handle, dir, m, nnz,
258 descr_M, csrVal, csrRowPtr, csrColInd, blocksize, info_M, &pBufferSize_M);
259 cusparseDbsrsv2_bufferSize(Util::Intern::cusparse_handle, dir, trans_L, m, nnz,
260 descr_L, csrVal, csrRowPtr, csrColInd, blocksize, info_L, &pBufferSize_L);
261 cusparseDbsrsv2_bufferSize(Util::Intern::cusparse_handle, dir, trans_U, m, nnz,
262 descr_U, csrVal, csrRowPtr, csrColInd, blocksize, info_U, &pBufferSize_U);
263
264 pBufferSize = max(pBufferSize_M, max(pBufferSize_L, pBufferSize_U));
265
266 cudaMalloc((void**)&pBuffer, pBufferSize);
267
268 status = cusparseDbsrilu02_analysis(Util::Intern::cusparse_handle, dir, m, nnz, descr_M,
269 csrVal, csrRowPtr, csrColInd, blocksize, info_M,
270 policy_M, pBuffer);
271 if (status != CUSPARSE_STATUS_SUCCESS)
272 throw InternalError(__func__, __FILE__, __LINE__, "cusparsebsrilu02 failed with status code: " + stringify(status));
273 status = cusparseXbsrilu02_zeroPivot(Util::Intern::cusparse_handle, info_M, &structural_zero);
274 if (CUSPARSE_STATUS_ZERO_PIVOT == status)
275 {
276 throw InternalError(__func__, __FILE__, __LINE__, "CUSPARSE ZERO PIVOT ERROR!");
277 }
278
279 status = cusparseDbsrsv2_analysis(Util::Intern::cusparse_handle, dir, trans_L, m, nnz, descr_L,
280 csrVal, csrRowPtr, csrColInd, blocksize, info_L, policy_L, pBuffer);
281 if (status != CUSPARSE_STATUS_SUCCESS)
282 throw InternalError(__func__, __FILE__, __LINE__, "cusparsebsrv2_analysis failed with status code: " + stringify(status));
283
284 status = cusparseDbsrsv2_analysis(Util::Intern::cusparse_handle, dir, trans_U, m, nnz, descr_U,
285 csrVal, csrRowPtr, csrColInd, blocksize, info_U, policy_U, pBuffer);
286 if (status != CUSPARSE_STATUS_SUCCESS)
287 throw InternalError(__func__, __FILE__, __LINE__, "cusparsebsrv2_analysis failed with status code: " + stringify(status));
288
289 cudaDeviceSynchronize();
290#ifdef FEAT_DEBUG_MODE
291 cudaError_t last_error(cudaGetLastError());
292 if (cudaSuccess != last_error)
293 throw InternalError(__func__, __FILE__, __LINE__, "CUDA error occurred in execution!\n" + stringify(cudaGetErrorString(last_error)));
294#endif
295
296 CudaIluBSolveInfo * info = new CudaIluBSolveInfo;
297 info->descr_M = descr_M;
298 info->descr_L = descr_L;
299 info->descr_U = descr_U;
300 info->info_M = info_M;
301 info->info_L = info_L;
302 info->info_U = info_U;
303 info->trans_L = trans_L;
304 info->trans_U = trans_U;
305 info->dir = dir;
306 info->policy_M = policy_M;
307 info->policy_L = policy_L;
308 info->policy_U = policy_U;
309 info->pBuffer = pBuffer;
310 info->z = z;
311 info->m = m;
312 info->nnz = nnz;
313 info->blocksize = blocksize;
314
315 return (void*)info;
316 }
