200 protected:
201 void _apply_intern(const LAFEM::SparseMatrixCSR<DataType, IndexType>& matrix, VectorType& vec_cor, const VectorType& vec_def)
202 {
203 // create pointers
204 DataType* pout(vec_cor.elements());
205 const DataType* pin(vec_def.elements());
206 const DataType* pval(matrix.val());
207 const IndexType* pcol_ind(matrix.col_ind());
208 const IndexType* prow_ptr(matrix.row_ptr());
209 const IndexType n((IndexType(matrix.rows())));
210
211 // __forward-insertion__
212 // iteration over all rows
213 for (IndexType i(0); i < n; ++i)
214 {
215 IndexType col;
216 DataType d(0);
217 // iteration over all elements on the left side of the main-diagonal
218 for (col = prow_ptr[i]; pcol_ind[col] < i; ++col)
219 {
220 d += pval[col] * pout[pcol_ind[col]];
221 }
222 pout[i] = _omega * (pin[i] - d) / pval[col];
223 }
224 }

244 void _apply_intern(const MatrixType& matrix, VectorType& vec_cor, const VectorType& vec_def)
245 {
246 // create pointers
247 auto* pout(vec_cor.elements());
248 const auto* pin(vec_def.elements());
249 const auto* pval(matrix.val());
250 const IndexType* pcol_ind(matrix.col_ind());
251 const IndexType* prow_ptr(matrix.row_ptr());
252 const IndexType n((IndexType(matrix.rows())));
253 typename MatrixType::ValueType inverse;
254
255 // __forward-insertion__
256 // iteration over all rows
257
258 for (IndexType i(0); i < n; ++i)
259 {
260 IndexType col;
261 typename VectorType::ValueType d(0);
262
263 // iteration over all elements on the left side of the main-diagonal
264 for (col = prow_ptr[i]; pcol_ind[col] < i; ++col)
265 {
266 d += pval[col] * pout[pcol_ind[col]];
267 }
268 inverse.set_inverse(pval[col]);
269 pout[i] = _omega * inverse * (pin[i] - d);
270 }
271 }
