245 void clone(const MeshPart& other)
246 {
247 this->_num_entities = other._num_entities;
248
249 if(other._index_set_holder)
250 {
251 if(this->_index_set_holder)
252 this->_index_set_holder->clone(*other._index_set_holder);
253 else
254 this->_index_set_holder.reset(new IndexSetHolderType(other._index_set_holder->clone()));
255 }
256 else if(this->_index_set_holder)
257 {
258 this->_index_set_holder.reset();
259 }
260
261 this->_target_set_holder.clone(other._target_set_holder);
262
263 this->_mesh_attributes.clear();
264
265 for(auto it = other._mesh_attributes.begin(); it != other._mesh_attributes.end(); ++it)
266 this->add_attribute(std::unique_ptr<AttributeSetType>(new AttributeSetType(it->second->clone())), it->first);

270 MeshPart clone() const
271 {

273 if(this->_index_set_holder)

706 explicit StandardRefinery(const MeshType& coarse_meshpart, const ParentMesh_& parent_mesh) :
707 _coarse_meshpart(coarse_meshpart),
708 _parent_mesh(&parent_mesh),
709 _parent_meshpart(nullptr)
710 {
711 // get number of entities in coarse mesh
712 for(int i(0); i <= shape_dim; ++i)
713 {
714 _num_entities_fine[i] = _num_entities_coarse[i] = coarse_meshpart.get_num_entities(i);

718 Intern::EntityCountWrapper<Intern::StandardRefinementTraits, ShapeType>::query(_num_entities_fine);

762 virtual Index get_num_entities(int dim) override
763 {
764 return _num_entities_fine[dim];
765 }
766
773 virtual void fill_attribute_sets(AttributeSetContainer& attribute_container) override
774 {
775 // Attributes of shape dimension 0 only make sense if we have a mesh topology
776 if(!_coarse_meshpart.has_topology())
777 return;
778
779 // Iterate over the attributes in the coarse mesh
780 typename MeshType::AttributeSetConstIterator it(_coarse_meshpart.get_mesh_attributes().begin());
781 typename MeshType::AttributeSetConstIterator jt(_coarse_meshpart.get_mesh_attributes().end());
782
783 for(; it != jt; ++it)
784 {
785 // Create a new empty attribute of the desired size
786 std::unique_ptr<AttributeType> refined_attribute(new AttributeType(
787 get_num_entities(0), it->second->get_dimension()));
788
789 // Refine the attribute in the coarse mesh and write the result to the new attribute
790 Intern::StandardAttribRefineWrapper<ShapeType, AttributeType>
791 ::refine(*refined_attribute, *(it->second), *_coarse_meshpart.get_topology());
792
793 // Add the attribute to the corresponding set
794 XASSERTM(attribute_container.emplace(it->first, std::move(refined_attribute)).second, "Error refining attribute " + it->first);
795 }
796 }
797
804 virtual void fill_index_sets(std::unique_ptr<IndexSetHolderType>& index_set_holder) override
805 {
806 XASSERT(index_set_holder.get() == nullptr);
807
808 // Only create the topology for the refined mesh if the coarse mesh has a topology
809 if(!_coarse_meshpart.has_topology())
810 return;
811
812 index_set_holder.reset(new IndexSetHolderType(_num_entities_fine));
813
814 // refine indices
815 Intern::IndexRefineWrapper<ShapeType>::refine(
816 *index_set_holder, _num_entities_coarse, *_coarse_meshpart.get_topology());
817 }
818
825 virtual void fill_target_sets(TargetSetHolderType& target_set_holder) override
826 {
827 // We have to use a helper class here, because the target set refinement
828 // depends heavily on the underlying mesh type. For this, we first need
829 // to check whether our parent is the root mesh or another mesh-part.
830 if(_parent_mesh != nullptr)
831 {
832 // the parent is the root mesh
833 Intern::TargetSetRefineParentWrapper<ParentMesh_>::fill_target_sets(target_set_holder,
834 _coarse_meshpart.get_target_set_holder(), _coarse_meshpart.get_topology(), *_parent_mesh);
835 }
836 else if(_parent_meshpart != nullptr)
837 {
838 // the parent is a mesh-part
839 Intern::TargetSetRefineParentWrapper<ParentMesh_>::fill_target_sets(target_set_holder,
840 _coarse_meshpart.get_target_set_holder(), _coarse_meshpart.get_topology(), *_parent_meshpart);
841 }
842 else
843 XABORTM("no parent present");
844 }

854 template<typename Shape_, int num_coords_, typename Coord_>
855 struct TargetSetRefineParentWrapper<ConformalMesh<Shape_, num_coords_, Coord_>>
856 {
857 static constexpr int shape_dim = Shape_::dimension;
858
859 // Note: This functions works for both mesh parents and mesh-part parents.
860 template<typename TargetSetHolderType_, typename IndexSetHolder_, typename Parent_>
861 static void fill_target_sets(
862 TargetSetHolderType_& target_set_holder,
863 const TargetSetHolderType_& coarse_target_set_holder,
864 const IndexSetHolder_* coarse_ish,
865 const Parent_& parent)
866 {
867 // get number of parent entities
868 Index num_entities_parent[shape_dim+1];
869 for(int i(0); i <= shape_dim; ++i)
870 num_entities_parent[i] = parent.get_num_entities(i);
871
872 if(coarse_ish != nullptr)
873 {
874 // The coarse mesh-part has a topology, so refine the target set to match.
875 // The parent must have a topology, too, which is non-trivial if the
876 // parent is a mesh-part itself.
877 const auto* parent_topo = parent.get_topology();
878 XASSERTM(parent_topo != nullptr, "mesh-part has topology, but parent doesn't");
879 Intern::TargetRefineWrapper<Shape_>::refine(target_set_holder,
880 num_entities_parent, coarse_target_set_holder, *coarse_ish, *parent_topo);
881 }
882 else
883 {
884 // The coarse mesh-part does not have a topology, stick to simple refinement
885 Intern::SimpleTargetRefineWrapper<Shape_>::refine(
886 target_set_holder, num_entities_parent, coarse_target_set_holder);
887 }
888 }
889 }; // // struct TargetSetRefineParentWrapper<ConformalMesh<...>>
