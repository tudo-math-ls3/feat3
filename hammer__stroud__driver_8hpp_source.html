56 template<
57 typename Weight_,
58 typename Coord_,
59 typename Point_>
60 static void fill(Rule<Shape::Simplex<dim_>, Weight_, Coord_, Point_>& rule)
61 {
62 // auxiliary variables
63 Coord_ r = (Coord_(dim_ + 2) - Math::sqrt(Coord_(dim_ + 2)))/
64 (Coord_(dim_ + 2) * Coord_(dim_ + 1));
65 Coord_ s = (Coord_(dim_ + 2) + Coord_(dim_) * Math::sqrt(Coord_(dim_ + 2)))/
66 (Coord_(dim_ + 2) * Coord_(dim_ + 1));
67
68 for(int i(0); i <= dim_; ++i)
69 {
70 // set weight
71 rule.get_weight(i) = Weight_(1) / Weight_(MetaMath::Factorial<dim_ + 1>::value);
72
73 // set point coords
74 for(int j(0); j < dim_; ++j)
75 {
76 rule.get_coord(i,j) = (i == j) ? s : r;
77 }
78 }
79 }

119 template<
120 typename Weight_,
121 typename Coord_,
122 typename Point_>
123 static void fill(Rule<Shape::Simplex<dim_>, Weight_, Coord_, Point_>& rule)
124 {
125 // auxiliary variables
126 Weight_ V = Weight_(1) / Weight_(MetaMath::Factorial<dim_>::value);
127 Weight_ B = - Weight_((dim_ + 1)*(dim_ + 1))/Weight_(4*dim_ + 8) * V;
128 Weight_ C = Weight_((dim_ + 3)*(dim_ + 3))/Weight_(4*(dim_ + 1)*(dim_ + 2)) * V;
129
130 // B-point
131 rule.get_weight(0) = B;
132 for(int j(0); j < dim_; ++j)
133 {
134 rule.get_coord(0,j) = Coord_(1) / Coord_(dim_ + 1);
135 }
136
137 // C-points
138 int count = 0;
139
140 for(int i(0); i <= dim_; ++i)
141 {
142 ++count;
143
144 // set weight
145 rule.get_weight(count) = C;
146
147 // set point coords
148 for(int j(0); j < dim_; ++j)
149 {
150 rule.get_coord(count,j) = Coord_((i == j) ? 3 : 1) / Coord_(dim_ + 3);
151 }
152 }
153
154 }

173 template<>
174 class HammerStroudD5Driver<Shape::Simplex<3> > :
175 public DriverBase<Shape::Simplex<3> >
176 {
177 public:
179 static constexpr bool variadic = false;
180 static constexpr int dim = 3;
181 static constexpr int num_points = 15;
182
184 static String name()
185 {
186 return "hammer-stroud-degree-5";
187 }
188
195 template<
196 typename Weight_,
197 typename Coord_,
198 typename Point_>
199 static void fill(Rule<Shape::Simplex<3>, Weight_, Coord_, Point_>& rule)
200 {
201 // auxiliary variables
202 Weight_ V = Weight_(1) / Weight_(MetaMath::Factorial<dim>::value);
203 Weight_ A = Weight_(16)/Weight_(135) * V;
204 Weight_ B1 = (Weight_(2665) + Weight_(14)*Math::sqrt(Weight_(15)))/Weight_(37800)*V;
205 Weight_ B2 = (Weight_(2665) - Weight_(14)*Math::sqrt(Weight_(15)))/Weight_(37800)*V;
206 Weight_ C = Weight_(20)/Weight_(378)*V;
207
208 Coord_ s1 = (Coord_(7) - Math::sqrt(Coord_(15)))/Coord_(34);
209 Coord_ s2 = (Coord_(7) + Math::sqrt(Coord_(15)))/Coord_(34);
210 Coord_ t1 = (Coord_(13) + Coord_(3)*Math::sqrt(Coord_(15)))/Coord_(34);
211 Coord_ t2 = (Coord_(13) - Coord_(3)*Math::sqrt(Coord_(15)))/Coord_(34);
212 Coord_ u = (Coord_(10) - Coord_(2)*Math::sqrt(Coord_(15)))/Coord_(40);
213 Coord_ v = (Coord_(10) + Coord_(2)*Math::sqrt(Coord_(15)))/Coord_(40);
214
215 // counter
216 int count = 0;
217
218 // A-point
219 rule.get_weight(count) = A;
220 for(int j(0); j < dim; ++j)
221 {
222 rule.get_coord(count,j) = Coord_(1)/Coord_(4);
223 }
224 ++count;
225
226 // B1-points
227 for(int i(0); i <= dim; ++i)
228 {
229 // set weight
230 rule.get_weight(count) = B1;
231
232 // set point coords
233 for(int j(0); j < dim; ++j)
234 {
235 rule.get_coord(count,j) = (i == j) ? t1 : s1;
236 }
237 ++count;
238 }
239
240 // B2-points
241 for(int i(0); i <= dim; ++i)
242 {
243
244 // set weight
245 rule.get_weight(count) = B2;
246
247 // set point coords
248 for(int j(0); j < dim; ++j)
249 {
250 rule.get_coord(count,j) = (i == j) ? t2 : s2;
251 }
252 ++count;
253 }
254
255 // C-points
256 for(int i(0); i <= dim; ++i)
257 {
258 for(int j(0); j < i; ++j)
259 {
260 if(i != j)
261 {
262 // set weight
263 rule.get_weight(count) = C;
264
265 // set point coords
266 for(int k(0); k < dim; ++k)
267 {
268 rule.get_coord(count,k) = (k == i) || (k == j) ? u : v;
269 }
270 ++count;
271 }
272 }
273 }
274 }
275 }; // class HammerStroudD5Driver<Simplex<3>,...>
