426 static DataType compute_constraint(DataType* constraint_vec, const Mesh_& mesh, const Dist_& dist)
427 {
428 XASSERT(constraint_vec != nullptr);
429 typedef Geometry::Intern::FaceIndexMapping<ShapeType, 1, 0> VertAtEdge;
430 typedef Geometry::Intern::FaceIndexMapping<ShapeType, 2, 0> VertAtFace;
431
432 DataType constraint(0);
433 const auto& idx = mesh.template get_index_set<ShapeType::dimension,0>();
434 for(Index cell(0); cell < mesh.get_num_entities(ShapeType::dimension); ++cell)
435 {
436 constraint_vec[cell] = DataType(0);
437
438 for(int edge(0); edge < Shape::FaceTraits<ShapeType,1>::count; ++edge)
439 {
440 //These are the indices of the vertices on edge
441 int i(VertAtEdge::map(edge,0));
442 int j(VertAtEdge::map(edge,1));
443 DataType my_constraint(FEAT::Analytic::Common::template HeavisideRegStatic<DataType>::
444 eval(- DataType(1)*dist(idx(cell, Index(i))) * dist(idx(cell, Index(j)))) );
445
446 constraint_vec[cell] += my_constraint;
447
448 constraint += my_constraint;
449 }
450
451 for(int face(0); face < Shape::FaceTraits<ShapeType,2>::count; ++face)
452 {
453 // These are the indices of the vertices on edge
454 int i0(VertAtFace::map(face,0));
455 int i1(VertAtFace::map(face,1));
456 int i2(VertAtFace::map(face,2));
457 int i3(VertAtFace::map(face,3));
458
459 DataType my_constraint(0);
460 my_constraint += FEAT::Analytic::Common::template HeavisideRegStatic<DataType>::
461 eval(- DataType(1)*dist(idx(cell, Index(i0))) * dist(idx(cell, Index(i3))));
462 my_constraint += FEAT::Analytic::Common::template HeavisideRegStatic<DataType>::
463 eval(- DataType(1)*dist(idx(cell, Index(i1))) * dist(idx(cell, Index(i2))));
464
465 constraint_vec[cell] += my_constraint;
466
467 constraint += my_constraint;
468 }
469
470 DataType my_constraint(0);
471 my_constraint += FEAT::Analytic::Common::template HeavisideRegStatic<DataType>::
472 eval( - dist(idx(cell,0)) * dist(idx(cell,7)));
473 my_constraint += FEAT::Analytic::Common::template HeavisideRegStatic<DataType>::
474 eval( - dist(idx(cell,1)) * dist(idx(cell,6)));
475 my_constraint += FEAT::Analytic::Common::template HeavisideRegStatic<DataType>::
476 eval( - dist(idx(cell,2)) * dist(idx(cell,5)));

479
480 constraint_vec[cell] += my_constraint;
481
482 constraint += my_constraint;
483
484 }

489 template<typename Vector_, typename Mesh_, typename Dist_, typename GradDist_, typename EdgeFreqs_>
490 static void add_constraint_grad(
491 Vector_& grad, const DataType alignment_fval, const DataType fac, const Mesh_& mesh, const Dist_& dist,
492 const GradDist_& grad_dist, const EdgeFreqs_& edge_freqs)
493 {
494 const auto& edge_idx = mesh.template get_index_set<1,0>();
496 typedef Tiny::Vector<DataType, Mesh_::world_dim> WorldPoint;
497
498 WorldPoint grad_loc(DataType(0));
499
500 for(Index edge(0); edge < mesh.get_num_entities(1); ++edge)
501 {
502 Index i(edge_idx(edge,0));
503 Index j(edge_idx(edge,1));
504
505 auto dist_prod = dist(i) * dist(j);
506 // Derivative of the heaviside function
507 auto heaviside_der = FEAT::Analytic::Common::template HeavisideRegStatic<DataType>::der_x(-dist_prod);
508
509 grad_loc = (-fac * alignment_fval * heaviside_der * dist(j)) * grad_dist(i);
510 grad(i, grad(i) + edge_freqs(edge)*grad_loc);
511
512 grad_loc = (-fac * alignment_fval * heaviside_der * dist(i)) * grad_dist(j);
513 grad(j, grad(j) + edge_freqs(edge)*grad_loc);
514 }
515
516 const auto& face_idx = mesh.template get_index_set<2,0>();
517 for(Index face(0); face < mesh.get_num_entities(2); ++face)
518 {
519 Index i(face_idx(face,Index(0)));
520 Index j(face_idx(face,Index(3)));
521
522 auto dist_prod = dist(i) * dist(j);
523 // Derivative of the heaviside function
524 auto heaviside_der = FEAT::Analytic::Common::template HeavisideRegStatic<DataType>::der_x(-dist_prod);
525
526 grad_loc = (-fac * alignment_fval * heaviside_der * dist(j)) * grad_dist(i);
527 grad(i, grad(i)+grad_loc);
528
529 grad_loc = (-fac * alignment_fval * heaviside_der * dist(i)) * grad_dist(j);
530 grad(j, grad(j)+grad_loc);
531
532 i = face_idx(face,Index(1));
533 j = face_idx(face,Index(2));
534
535 dist_prod = dist(i) * dist(j);
536 // Derivative of the heaviside function
537 heaviside_der = FEAT::Analytic::Common::template HeavisideRegStatic<DataType>::der_x(-dist_prod);
538
539 grad_loc = (-fac * alignment_fval * heaviside_der * dist(j)) * grad_dist(i);
540 grad(i, grad(i)+grad_loc);
541
542 grad_loc = (-fac * alignment_fval * heaviside_der * dist(i)) * grad_dist(j);
543 grad(j, grad(j)+grad_loc);
544 }
545
546 const auto& cell_idx = mesh.template get_index_set<3,0>();
547 for(Index cell(0); cell < mesh.get_num_entities(2); ++cell)
548 {
549 Index i[4]{cell_idx(cell,Index(0)), cell_idx(cell,Index(1)),cell_idx(cell,Index(2)),cell_idx(cell,Index(3))};
550 Index j[4]{cell_idx(cell,Index(7)), cell_idx(cell,Index(6)),cell_idx(cell,Index(5)),cell_idx(cell,Index(2))};
551
552 for(int k(0); k < 4; ++k)
553 {
554
555 auto dist_prod = dist(i[k]) * dist(j[k]);
556 // Derivative of the heaviside function

561
562 grad_loc = (-fac * alignment_fval * heaviside_der * dist(i[k])) * grad_dist(j[k]);
563 grad(j[k], grad(j[k]) + grad_loc);
564 }
565
566 }

1181 {

1183 _grad_conc.format();
1184
1185 // Index set for local/global numbering
1186 auto& idx = _mesh_node->get_mesh()->template get_index_set<ShapeType::dimension,0>();
1187
1188 // This will hold the coordinates for one element for passing to other routines
1189 FEAT::Tiny::Matrix <CoordType, Shape::FaceTraits<ShapeType,0>::count, MeshType::world_dim> x;
1190 // Local cell dimensions for passing to other routines
1191 FEAT::Tiny::Vector <CoordType,MeshType::world_dim> h;
1192 // This will hold the local gradient for one element for passing to other routines
1193 FEAT::Tiny::Matrix <CoordType, Shape::FaceTraits<ShapeType,0>::count, MeshType::world_dim> grad_loc;

1197 FEAT::Tiny::Matrix <CoordType, Shape::FaceTraits<ShapeType,0>::count, MeshType::world_dim> dist_loc;

1199 // Compute the functional value for each cell
1200 for(Index cell(0); cell < _mesh_node->get_mesh()->get_num_entities(ShapeType::dimension); ++cell)
1201 {
1202 // Collect levelset and levelset grad values
1203 for(int j(0); j < Shape::FaceTraits<ShapeType,0>::count; ++j)

1205 // Global vertex/dof index
1206 Index i(idx(cell, j));
1207 // Get levelset

1209 // Get levelset gradient

1212

1216 // Add local contributions to global gradient vector
1217 for(int j(0); j < Shape::FaceTraits<ShapeType,0>::count; ++j)
1218 {
1219 Index i(idx(cell, j));
1220 this->_grad_conc(i, _grad_conc(i) + grad_loc[j]);
