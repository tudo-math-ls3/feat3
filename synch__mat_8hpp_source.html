77#if defined(FEAT_HAVE_MPI) || defined(DOXYGEN)
78 SynchMatrix(const Dist::Comm& comm, const std::vector<int>& ranks,
79 const std::vector<VMT_>& mirrors_row, const std::vector<VMT_>& mirrors_col) :
80 _initialized(false),
81 _comm(comm),
82 _ranks(ranks),
83 _send_reqs(ranks.size()),
84 _recv_reqs(ranks.size()),
85 _send_bufs(ranks.size()),
86 _recv_bufs(ranks.size())
87 {
88 const std::size_t n = ranks.size();
89
90 XASSERTM(mirrors_row.size() == n, "invalid row vector mirror count");
91 XASSERTM(mirrors_col.size() == n, "invalid column vector mirror count");
92
93 _mirrors.reserve(n);
94
95 // create matrix mirrors and buffers
96 for(std::size_t i(0); i < n; ++i)
97 {
98 const VMT_& mir_r = mirrors_row.at(i);
99 const VMT_& mir_c = mirrors_col.at(i);
100
101 // create matrix mirror
102 _mirrors.push_back(MatrixMirrorType(mir_r, mir_c));
103 }
104 }

125#if defined(FEAT_HAVE_MPI) || defined(DOXYGEN)
126 void init(const MT_& matrix)
127 {
128 XASSERTM(!_initialized, "SynchMatrix object is already initialized");
129
130 const std::size_t n = _ranks.size();
131
132 // create matrix mirror buffers
133 for(std::size_t i(0); i < n; ++i)
134 {
135 _send_bufs.at(i) = _mirrors.at(i).create_buffer(matrix);
136 }
137
138 // receive buffer dimensions vector
139 std::vector<std::array<Index,4>> recv_dims(n), send_dims(n);
140
141 // post send-buffer dimension receives
142 for(std::size_t i(0); i < n; ++i)
143 {
144 _recv_reqs[i] = _comm.irecv(recv_dims.at(i).data(), std::size_t(4), _ranks.at(i));
145 }
146
147 // send send-buffer dimensions
148 for(std::size_t i(0); i < n; ++i)
149 {
150 const BufferMatrixType& sbuf = _send_bufs.at(i);
151 send_dims.at(i)[0] = sbuf.rows();
152 send_dims.at(i)[1] = sbuf.columns();
153 send_dims.at(i)[2] = sbuf.entries_per_nonzero();
154 send_dims.at(i)[3] = sbuf.used_elements();
155 _send_reqs[i] = _comm.isend(send_dims.at(i).data(), std::size_t(4), _ranks.at(i));
156 }
157
158 // wait for all receives to finish
159 _recv_reqs.wait_all();
160
161 // create receive buffers and post receives
162 for(std::size_t i(0); i < n; ++i)
163 {
164 // get the receive buffer dimensions
165 Index nrows = recv_dims.at(i)[0];
166 Index ncols = recv_dims.at(i)[1];
167 Index nepnz = recv_dims.at(i)[2];
168 Index nnze = recv_dims.at(i)[3];
169
170 // allocate receive buffer
171 _recv_bufs.at(i) = BufferMatrixType(nrows, ncols, nnze, nepnz);
172 }
173
174 // post buffer row-pointer array receives
175 for(std::size_t i(0); i < n; ++i)
176 {
177 _recv_reqs[i] = _comm.irecv(_recv_bufs.at(i).row_ptr(), _recv_bufs.at(i).rows()+std::size_t(1), _ranks.at(i));
178 }
179
180 // wait for all previous sends to finish
181 _send_reqs.wait_all();
182
183 // post buffer row-pointer array sends
184 for(std::size_t i(0); i < n; ++i)
185 {
186 _send_reqs[i] = _comm.isend(_send_bufs.at(i).row_ptr(), _send_bufs.at(i).rows()+std::size_t(1), _ranks.at(i));
187 }
188
189 // wait for all previous receives to finish
190 _recv_reqs.wait_all();
191
192 // post buffer column-index array receives
193 for(std::size_t i(0); i < n; ++i)
194 {
195 _recv_reqs[i] = _comm.irecv(_recv_bufs.at(i).col_ind(), _recv_bufs.at(i).used_elements(), _ranks.at(i));
196 }
197
198 // wait for all previous sends to finish
199 _send_reqs.wait_all();
200
201 // post buffer column-index array sends
202 for(std::size_t i(0); i < n; ++i)
203 {
204 _send_reqs[i] = _comm.isend(_send_bufs.at(i).col_ind(), _send_bufs.at(i).used_elements(), _ranks.at(i));
205 }
206
207 // wait for all receives and sends to finish
208 _recv_reqs.wait_all();
209 _send_reqs.wait_all();
210
211 _initialized = true;
212 }
213#else // non-MPI version
214 void init(const MT_&)
215 {
216 XASSERTM(!_initialized, "SynchMatrix object is already initialized");
217 _initialized = true;
218 }
219#endif // FEAT_HAVE_MPI

227#if defined(FEAT_HAVE_MPI) || defined(DOXYGEN)
228 void exec(MT_& matrix)
229 {
230 XASSERTM(_initialized, "SynchMatrix object has not been initialized");
231
232 const std::size_t n = _ranks.size();
233
234 // create receive buffers and post receives
235 for(std::size_t i(0); i < n; ++i)
236 {
237 BufferMatrixType& buf = _recv_bufs.at(i);
238
239 // post receive
240 _recv_reqs.get_request(i) = _comm.irecv(buf.val(), buf.val_size(), _ranks.at(i));
241 }
242
243 // post sends
244 for(std::size_t i(0); i < n; ++i)
245 {
246 BufferMatrixType& buf = _send_bufs.at(i);
247
248 // gather from mirror
249 _mirrors.at(i).gather(buf, matrix);
250
251 // post send
252 _send_reqs.get_request(i) = _comm.isend(buf.val(), buf.val_size(), _ranks.at(i));
253 }
254
255 // process all pending receives
256 for(std::size_t idx(0u); _recv_reqs.wait_any(idx); )
257 {
258 // scatter the receive buffer
259 _mirrors.at(idx).scatter_axpy(matrix, _recv_bufs.at(idx));
260 }
261
262 // wait for all sends to finish
263 _send_reqs.wait_all();
264 }
265#else // non-MPI version
266 void exec(MT_&)
267 {
268 XASSERTM(_initialized, "SynchMatrix object has not been initialized");
269 }
270#endif // FEAT_HAVE_MPI
271 }; // class SynchMatrix

288 SynchMatrix(const Dist::Comm& comm, const std::vector<int>& ranks,
289 const std::vector<SVMT_>& mirrors_row, const std::vector<SVMT_>& mirrors_col) :

292 mirrors_row_split(blocks_),
293 mirrors_col_split(blocks_)

296 {
297 for (Index i(0) ; i < mirrors_row.size() ; ++i)
298 {

300 }
301 for (Index i(0) ; i < mirrors_col.size() ; ++i)
302 {
303 mirrors_col_split.at((size_t)block).push_back(mirrors_col.at(i).get(block).clone(LAFEM::CloneMode::Shallow));
304 }
305
306 synch_matrix_list.at((size_t)block) = std::make_shared<SynchMatrix<MT_, VMT_>>(comm, ranks, mirrors_row_split.at((size_t)block), mirrors_col_split.at((size_t)block));

308 }
