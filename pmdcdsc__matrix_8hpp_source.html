425 void init_symbolic()
426 {
427 watch_init_symbolic.start();
428
429 // get the velocity and pressure gates
430 const GateTypeV* gate_v = this->_matrix_b.get_row_gate();
431 const GateTypeP* gate_p = this->_matrix_d.get_row_gate();
432 XASSERT(gate_v != nullptr);
433 XASSERT(gate_p != nullptr);
434
435 // the pressure gate must be empty, otherwise the pressure space is not discontinuous
436 XASSERTM(gate_p->_ranks.empty(), "pressure space is not discontinuous");
437
438 // compute the local matrix structure of S by D * B
439 {
440 watch_init_sym_matrix_loc.start();
441
442 // compose structures of D and B
443 Adjacency::Graph graph_s(Adjacency::RenderType::injectify_sorted, _matrix_d.local(), _matrix_b.local());
444 // create the matrix layout of S
445 _matrix_s = LocalMatrixTypeS(graph_s);
446
447 watch_init_sym_matrix_loc.stop();
448 }
449
450 // get our communicator
451 const Dist::Comm& comm = *gate_v->get_comm();
452
453 // get neighbor ranks
454 this->_ranks = gate_v->_ranks;
455
456 // get the number of our neighbors
457 const std::size_t num_neighs = this->_ranks.size();
458 if(num_neighs <= std::size_t(0))
459 {
460 watch_init_symbolic.stop();
461 return; // no neighbors, no problems :)
462 }
463
464 // copy the layout of B into (D*A)^T
465 this->_matrix_da = this->_matrix_b.local().clone(LAFEM::CloneMode::Layout);
466
467 // resize our member arrays
468 this->_pres_mirrors.resize(num_neighs);
469 this->_data_mirrors.resize(num_neighs);
470 this->_neighbor_graphs.resize(num_neighs);
471 this->_neighbor_matrices.resize(num_neighs);
472
473 // allocate a vector of graphs for B
474 std::vector<Adjacency::Graph> my_graphs(num_neighs);
475
476 // loop over all neighbor processes
477 for(std::size_t i(0); i < num_neighs; ++i)
478 {
479 // get the velocity mirror
480 const MirrorTypeV& mirror_v = gate_v->_mirrors.at(i);
481
482 // assemble the pressure mirror for this neighbor
483 watch_init_sym_pres_mirror.start();
484 MirrorTypeP& mirror_p = this->_pres_mirrors.at(i);
485 this->_asm_pres_mirror(mirror_p, mirror_v, this->_matrix_b.local());
486 watch_init_sym_pres_mirror.stop();
487
488 // assemble reduced B-matrix graph
489 watch_init_sym_reduced_b.start();
490 my_graphs.at(i) = this->_asm_reduced_b( mirror_p, mirror_v, this->_matrix_b.local());
491 watch_init_sym_reduced_b.stop();
492
493 // assemble B' data mirror
494 watch_init_sym_data_mirror.start();
495 this->_data_mirrors.at(i) = this->_asm_data_mirror(mirror_p, mirror_v, this->_matrix_b.local(), my_graphs.at(i));
496 watch_init_sym_data_mirror.stop();
497 }
498
499 // dimension send/receive buffers and requests
500 std::vector<std::array<Index,3>> recv_dims(num_neighs), send_dims(num_neighs);
501 Dist::RequestVector recv_reqs(num_neighs), send_reqs(num_neighs);
502
503 // post receive requests for dimensions
504 for(std::size_t i(0); i < num_neighs; ++i)
505 {
506 recv_reqs[i] = comm.irecv(recv_dims.at(i).data(), std::size_t(3), this->_ranks.at(i));
507 }
508
509 // send dimensions
510 for(std::size_t i(0); i < num_neighs; ++i)
511 {
512 const Adjacency::Graph& g = my_graphs.at(i);
513 auto& sdim = send_dims.at(i);
514 sdim[0] = g.get_num_nodes_domain(); // corresponds to velocity mirror index size
515 sdim[1] = g.get_num_nodes_image();
516 sdim[2] = g.get_num_indices();
517 send_reqs[i] = comm.isend(sdim.data(), std::size_t(3), this->_ranks.at(i));
518 }
519
520 // process all pending receives
521 for(std::size_t i(0u); recv_reqs.wait_any(i); )
522 {
523 // get received dimension
524 auto& rdim = recv_dims.at(i);
525
526 // the first dimension must match our velocity mirror index set size
527 XASSERT(rdim[0] == gate_v->_mirrors.at(i).num_indices());
528
529 // allocate graph of corresponding dimensions
530 this->_neighbor_graphs.at(i) = Adjacency::Graph(rdim[0], rdim[1], rdim[2]);
531 }
532
533 // post domain-pointer array receives
534 for(std::size_t i(0); i < num_neighs; ++i)
535 {
536 recv_reqs[i] = comm.irecv(this->_neighbor_graphs.at(i).get_domain_ptr(),
537 recv_dims.at(i)[0] + std::size_t(1), this->_ranks.at(i));
538 }
539
540 // wait for all previous sends to finish
541 send_reqs.wait_all();
542
543 // post domain-pointer array sends
544 for(std::size_t i(0); i < num_neighs; ++i)
545 {
546 send_reqs[i] = comm.isend(my_graphs.at(i).get_domain_ptr(),
547 send_dims.at(i)[0] + std::size_t(1), this->_ranks.at(i));
548 }
549
550 // wait for all pending receives to finish
551 recv_reqs.wait_all();
552
553 // post image-index array receives
554 for(std::size_t i(0); i < num_neighs; ++i)
555 {
556 recv_reqs[i] = comm.irecv(this->_neighbor_graphs.at(i).get_image_idx(),
557 recv_dims.at(i)[2], this->_ranks.at(i));
558 }
559
560 // wait for all previous sends to finish
561 send_reqs.wait_all();
562
563 // post image-index array sends
564 for(std::size_t i(0); i < num_neighs; ++i)
565 {
566 send_reqs[i] = comm.isend(my_graphs.at(i).get_image_idx(), send_dims.at(i)[2], this->_ranks.at(i));
567 }
568
569 // wait for all pending receives to finish
570 recv_reqs.wait_all();
571
572 // wait for all previous sends to finish
573 send_reqs.wait_all();
574
575 // compute Schur-matrix structures for neighbors
576 for(std::size_t i(0); i < num_neighs; ++i)
577 {
578 watch_init_sym_neighbor_s.start();
579
580 // D*M^T = (M*B)^T
581 Adjacency::Graph graph_dm(Adjacency::RenderType::injectify_transpose, gate_v->_mirrors.at(i), this->_matrix_b.local());
582
583 // S = (D*M^T) * B'
584 Adjacency::Graph graph_s(Adjacency::RenderType::injectify_sorted, graph_dm, this->_neighbor_graphs.at(i));
585
586 // allocate Schur-matrix
587 this->_neighbor_matrices.at(i).convert(NeighMatrixTypeS(graph_s));
588
589 watch_init_sym_neighbor_s.stop();
590 }
591
592 // that's it
593 watch_init_symbolic.stop();
594 }

607 void init_numeric()
608 {
609 watch_init_numeric.start();
610
611 // pre-multiply local matrix product
612 watch_init_num_matrix_loc.start();
613 this->_matrix_s.format();
614 _asm_local_schur_matrix(this->_matrix_s, this->_matrix_d.local(), this->_diagonal_a.local(), this->_matrix_b.local());
615 watch_init_num_matrix_loc.stop();
616
617 // get the number of our neighbors
618 const std::size_t num_neighs = this->_ranks.size();
619 if(num_neighs <= std::size_t(0))
620 {
621 watch_init_numeric.stop();
622 return; // no neighbors, no problems :)
623 }
624
625 // get our communicator
626 const Dist::Comm& comm = *this->_matrix_b.get_comm();
627
628 // send/receive buffers and requests
629 std::vector<BufferVectorType> recv_bufs(num_neighs), send_bufs(num_neighs);
630 Dist::RequestVector recv_reqs(num_neighs), send_reqs(num_neighs);
631
632 // allocate receive buffer matrices B' and post receives
633 for(std::size_t i(0); i < num_neighs; ++i)
634 {
635 recv_bufs.at(i) = BufferVectorType(Index(dim) * this->_neighbor_graphs.at(i).get_num_indices());
636 recv_reqs[i] = comm.irecv(recv_bufs.at(i).elements(), recv_bufs.at(i).size(), this->_ranks.at(i));
637 }
638
639 // extract reduced matrix data and post send
640 for(std::size_t i(0); i < num_neighs; ++i)
641 {
642 watch_init_num_gather_b.start();
643 send_bufs.at(i) = _gather_b(this->_data_mirrors.at(i), this->_matrix_b.local());
644 watch_init_num_gather_b.stop();
645 send_reqs[i] = comm.isend(send_bufs.at(i).elements(), send_bufs.at(i).size(), this->_ranks.at(i));
646 }
647
648 // pre-multiply D*A and store in transposed form, i.e. CSC rather than CSR
649 watch_init_num_premult_da.start();
650 _premult_da(this->_matrix_da, this->_matrix_d.local(), this->_diagonal_a.local());
651 watch_init_num_premult_da.stop();
652
653 // process receives and compute neighbor schur matrices
654 for(std::size_t i(0u); recv_reqs.wait_any(i); )
655 {
656 watch_init_num_neighbor_s.start();
657 this->_neighbor_matrices.at(i).format();
658 _asm_neighbor_schur_matrix(this->_neighbor_matrices.at(i), this->_matrix_da,
659 this->_matrix_b.get_row_gate()->_mirrors.at(i), this->_neighbor_graphs.at(i), recv_bufs.at(i));
660 watch_init_num_neighbor_s.stop();
661 }
662
663 // wait for all previous sends to finish
664 send_reqs.wait_all();
665 watch_init_numeric.stop();
666 }

714 LocalMatrixTypeS asm_adp_symbolic(Index& glob_dof_offset, Index& glob_dof_count) const
715 {
716 // no neighbors?
717 if(_ranks.empty())
718 {
719 glob_dof_offset = Index(0);
720 glob_dof_count = _matrix_s.rows();
721 return _matrix_s.clone(LAFEM::CloneMode::Layout);
722 }
723
724 // get our communicator
725 const Dist::Comm& comm = *this->get_comm();
726
727 const std::size_t num_neighs = this->_ranks.size();
728
729 // get number of local DOFs
730 const Index num_loc_dofs = _matrix_s.rows();
731
732 // compute our global DOF offset and count
733 comm.exscan(&num_loc_dofs, &glob_dof_offset, std::size_t(1), Dist::op_sum);
734 comm.allreduce(&num_loc_dofs, &glob_dof_count, std::size_t(1), Dist::op_sum);
735
736 // The columns of our neighbor matrices correspond to the entries in the pressure mirror.
737 // However, for the desired ADP matrix, we have to translate these into global DOF indices.
738 // For this, each process has to map the DOF in its pressure mirrors to global DOFs and then
739 // send these DOF indices to the corresponding neighbor, so that it can map the column
740 // indices of its neighbor matrix to global DOF indices.
741
742 // send/receive mirrors and requests
743 std::vector<std::vector<IndexType>> recv_dofs(num_neighs), send_dofs(num_neighs);
744 Dist::RequestVector recv_reqs(num_neighs), send_reqs(num_neighs);
745
746 // allocate receive vectors and post receives
747 for(std::size_t i(0); i < num_neighs; ++i)
748 {
749 recv_dofs.at(i).resize(_neighbor_graphs.at(i).get_num_nodes_image());
750 recv_reqs[i] = comm.irecv(recv_dofs.at(i).data(), recv_dofs.at(i).size(), this->_ranks.at(i));
751 }
752
753 // translate our pressure mirrors to global DOF vectors and post sends
754 for(std::size_t i(0); i < num_neighs; ++i)
755 {
756 const Index num_idx = _pres_mirrors.at(i).num_indices();
757 const IndexType* pidx = _pres_mirrors.at(i).indices();
758 send_dofs.at(i).resize(num_idx);
759 IndexType* sidx = send_dofs.at(i).data();
760 for(Index k(0); k < num_idx; ++k)
761 sidx[k] = glob_dof_offset + pidx[k];
762 send_reqs[i] = comm.isend(send_dofs.at(i).data(), send_dofs.at(i).size(), this->_ranks.at(i));
763 }
764
765 // get local matrix stuff
766 const Index num_rows = num_loc_dofs;
767 const IndexType* row_ptr_s = _matrix_s.row_ptr();
768 const IndexType* col_idx_s = _matrix_s.col_ind();
769
770 // compute number of non-zeros per row and total
771 Index num_nzes = _matrix_s.used_elements();
772 std::vector<IndexType> row_aux(num_rows, IndexType(0));
773 for(Index i(0); i < num_rows; ++i)
774 row_aux[i] = (row_ptr_s[i+1] - row_ptr_s[i]);
775
776 for(const auto& x : _neighbor_matrices)
777 {
778 num_nzes += x.used_elements();
779 const Index used_rows = x.used_rows();
780 const IndexType* row_ptr_x = x.row_ptr();
781 const IndexType* row_idx_x = x.row_numbers();
782 for(Index i(0); i < used_rows; ++i)
783 row_aux[row_idx_x[i]] += (row_ptr_x[i+1] - row_ptr_x[i]);
784 }
785
786 // allocate our matrix
787 LocalMatrixTypeS matrix_g(num_rows, glob_dof_count, num_nzes);
788
789 // get our index arrays
790 IndexType* row_ptr_g = matrix_g.row_ptr();
791 IndexType* col_idx_g = matrix_g.col_ind();
792
793 // compute row pointer array and store backup in aux
794 row_ptr_g[0] = IndexType(0);
795 for(Index i(0); i < num_rows; ++i)
796 {
797 row_ptr_g[i+1] = row_ptr_g[i] + row_aux[i];
798 row_aux[i] = row_ptr_g[i];
799 }
800
801 // Note: For the sake of compatibility with picky third-party libraries, we want to
802 // ensure that the column indices of the output matrix are in ascending order.
803 // For this, we have to combine the matrix layout from our own local matrix and
804 // the matrices of our neighbors in rank-ascending order. So we first create two
805 // rank maps for our neighbors with lower and higher ranks, so that we can easily
806 // loop over all matrices in rank-order.
807
808 // create two neighbors maps: one of all lower ranks and one of all higher ranks
809 std::map<int, std::size_t> neigh_map_l, neigh_map_h;
810 for(std::size_t ineigh(0); ineigh < num_neighs; ++ineigh)
811 {
812 if(_ranks.at(ineigh) < comm.rank())
813 neigh_map_l.emplace(_ranks.at(ineigh), ineigh);
814 else
815 neigh_map_h.emplace(_ranks.at(ineigh), ineigh);
816 }
817
818 // wait for all receive requests to finish
819 recv_reqs.wait_all();
820
821 // first, insert all neighbor matrices with a lower rank in rank-ascending order
822 for(auto it = neigh_map_l.begin(); it != neigh_map_l.end(); ++it)
823 {
824 std::size_t ineigh = it->second;
825 const Index used_rows = _neighbor_matrices.at(ineigh).used_rows();
826 const IndexType* row_ptr_x = _neighbor_matrices.at(ineigh).row_ptr();
827 const IndexType* row_idx_x = _neighbor_matrices.at(ineigh).row_numbers();
828 const IndexType* col_idx_x = _neighbor_matrices.at(ineigh).col_ind();
829 const IndexType* dof_idx_x = recv_dofs.at(ineigh).data();
830 for(Index i(0); i < used_rows; ++i)
831 {
832 IndexType& k = row_aux[row_idx_x[i]];
833 for(IndexType j(row_ptr_x[i]); j < row_ptr_x[i + 1]; ++j, ++k)
834 col_idx_g[k] = dof_idx_x[col_idx_x[j]];
835 }
836 }
837
838 // now insert our local matrix S
839 for(Index i(0); i < num_rows; ++i)
840 {
841 IndexType k = row_aux[i];
842 for(IndexType j(row_ptr_s[i]); j < row_ptr_s[i + 1]; ++j, ++k)
843 col_idx_g[k] = glob_dof_offset + col_idx_s[j];
844 row_aux[i] = k;
845 }
846
847 // finally, insert all neighbor matrices with a higher rank in rank-ascending order
848 for(auto it = neigh_map_h.begin(); it != neigh_map_h.end(); ++it)
849 {
850 std::size_t ineigh = it->second;
851 const Index used_rows = _neighbor_matrices.at(ineigh).used_rows();
852 const IndexType* row_ptr_x = _neighbor_matrices.at(ineigh).row_ptr();
853 const IndexType* row_idx_x = _neighbor_matrices.at(ineigh).row_numbers();
854 const IndexType* col_idx_x = _neighbor_matrices.at(ineigh).col_ind();
855 const IndexType* dof_idx_x = recv_dofs.at(ineigh).data();
856 for(Index i(0); i < used_rows; ++i)
857 {
858 IndexType& k = row_aux[row_idx_x[i]];
859 for(IndexType j(row_ptr_x[i]); j < row_ptr_x[i + 1]; ++j, ++k)
860 col_idx_g[k] = dof_idx_x[col_idx_x[j]];
861 }
862 }
863
864 // wait for all send requests to finish
865 send_reqs.wait_all();
866
867#ifdef DEBUG
868 // sanity check: ensure that the column indices are sorted correctly
869 for(Index i = 0; i < num_rows; ++i)
870 {
871 for(IndexType j(row_ptr_g[i]); j + 1 < row_ptr_g[i + 1]; ++j)
872 {
873 ASSERT(col_idx_g[j] < col_idx_g[j+1]);
874 }
875 }
876#endif // DEBUG
877
878 // that's it
879 return matrix_g;
880 }

1057 static Adjacency::Graph _asm_reduced_b(const MirrorTypeP& mirror_p, const MirrorTypeV& mirror_v,
1058 const LocalMatrixTypeB& matrix_b)
1059 {
1060 const Index num_dof_mir_v = mirror_v.num_indices();
1061 const Index num_dof_mir_p = mirror_p.num_indices();
1062 const IndexType* velo_idx = mirror_v.indices();
1063 const IndexType* pres_idx = mirror_p.indices();
1064 const IndexType* row_ptr = matrix_b.row_ptr();
1065 const IndexType* col_idx = matrix_b.col_ind();
1066
1067 // count number of non-zeros in indexed rows of B = non-zeros in B'
1068 Index num_red_nzes = Index(0);
1069 for(Index i(0); i < num_dof_mir_v; ++i)
1070 {
1071 const Index irow = velo_idx[i];
1072 num_red_nzes += Index(row_ptr[irow+1] - row_ptr[irow]);
1073 }
1074
1075 // allocate matrix graph for reduced part B' of B
1076 Adjacency::Graph graph(num_dof_mir_v, num_dof_mir_p, num_red_nzes);
1077 Index* dom_ptr = graph.get_domain_ptr();
1078 Index* img_idx = graph.get_image_idx();
1079
1080 // loop over all rows of reduced part B'
1081 dom_ptr[0] = Index(0);
1082 for(Index i(0); i < num_dof_mir_v; ++i)
1083 {
1084 // get the B-row index of the i-th row of B'
1085 const IndexType irow = velo_idx[i];
1086 // loop over all non-zeroes in the row of B / B'
1087 for(IndexType j(row_ptr[irow]), k(dom_ptr[i]); j < row_ptr[irow+1]; ++j, ++k)
1088 {
1089 // initialize invalid index for assertion below
1090 img_idx[k] = ~IndexType(0);
1091
1092 // try to find this column index (=pressure DOF) in our pressure DOF mirror
1093 for(Index l(0); l < num_dof_mir_p; ++l)
1094 {
1095 if(col_idx[j] == pres_idx[l])
1096 {
1097 // that's our pressure DOF, so store its index as column index of B'
1098 img_idx[k] = l;
1099 break;
1100 }
1101 }
1102 ASSERT(img_idx[k] != ~IndexType(0));
1103 }
1104 // set next row pointer of B'
1105 dom_ptr[i+1] = dom_ptr[i] + (row_ptr[irow+1] - row_ptr[irow]);
1106 }
1107
1108 // sort indices of B' and return graph
1109 graph.sort_indices();
1110 return graph;
1111 }

1292 static void _asm_local_schur_matrix(LocalMatrixTypeS& s, const LocalMatrixTypeD& d,
1293 const LocalVectorTypeV& a, const LocalMatrixTypeB& b)
1294 {
1295 // Note: this is a modified version of SparseMatrixCSR::add_double_mat_product for BCSR multiplicands
1296
1297 // validate matrix dimensions
1298 XASSERT(s.rows() == d.rows());
1299 XASSERT(d.columns() == a.size());
1300 XASSERT(a.size() == b.rows());
1301 XASSERT(b.columns() == s.columns());
1302
1303 // fetch matrix arrays:
1304 DataType* data_s = s.val();
1305 const ValueTypeD* data_d = d.val();
1306 const ValueTypeA* data_a = a.elements();
1307 const ValueTypeB* data_b = b.val();
1308 const IndexType* row_ptr_s = s.row_ptr();
1309 const IndexType* col_idx_s = s.col_ind();
1310 const IndexType* row_ptr_d = d.row_ptr();
1311 const IndexType* col_idx_d = d.col_ind();
1312 const IndexType* row_ptr_b = b.row_ptr();
1313 const IndexType* col_idx_b = b.col_ind();
1314
1315 // loop over all rows of D and S, resp.
1316 for(IndexType i(0); i < IndexType(s.rows()); ++i)
1317 {
1318 // loop over all non-zeros D_ik in row i of D
1319 for(IndexType ik(row_ptr_d[i]); ik < row_ptr_d[i+1]; ++ik)
1320 {
1321 // get column index k
1322 const IndexType k = col_idx_d[ik];
1323
1324 // pre-compute (D_ik * A_kk)
1325 const ValueTypeD val_da = _mat_mult_d_a(data_d[ik], data_a[k]);
1326
1327 // S_i. += (D_ik * A_kk) * B_k.
1328 for(IndexType ij(row_ptr_s[i]), kj(row_ptr_b[k]); kj < row_ptr_b[k+1]; ++ij)
1329 {
1330 ASSERT(ij < row_ptr_s[i+1]);
1331 ASSERT(col_idx_s[ij] <= col_idx_b[kj]);
1332 if(col_idx_s[ij] == col_idx_b[kj])
1333 {
1334 data_s[ij] += _mat_mult_da_b(val_da, data_b[kj]);
1335 ++kj;
1336 }
1337 }
1338 }
1339 }

1360 static void _asm_neighbor_schur_matrix(NeighMatrixTypeS& s, const LocalMatrixTypeB& da,
1361 const MirrorTypeV& mirror_v, const Adjacency::Graph& graph_b, const BufferVectorType& buffer_b)
1362 {
1363 // validate matrix dimensions
1364 XASSERT(s.rows() == da.columns());
1365 XASSERT(da.rows() == mirror_v.size());
1366 XASSERT(graph_b.get_num_nodes_domain() == mirror_v.num_indices());
1367 XASSERT(graph_b.get_num_nodes_image() == s.columns());
1368 XASSERT(mirror_v.size() == da.rows());
1369
1370 // fetch matrix arrays:
1371 DataType* data_s = s.val();
1372 const IndexType* row_ptr_s = s.row_ptr();
1373 const IndexType* row_idx_s = s.row_numbers();
1374 const IndexType* col_idx_s = s.col_ind();
1375 const Index used_rows_s = s.used_rows();
1376
1377 // we use CSR for (D*A)^T here, which is effectively a CSC storage of (D*A),
1378 // thus rows and columns swap their meaning here
1379 const ValueTypeB* data_da = da.val();
1380 const IndexType* col_ptr_da = da.row_ptr();
1381 const IndexType* row_idx_da = da.col_ind();
1382
1383 const Index num_mir_idx = mirror_v.num_indices();
1384 const IndexType* mir_idx = mirror_v.indices();
1385
1386 const ValueTypeB* data_b = reinterpret_cast<const ValueTypeB*>(buffer_b.elements());
1387 const Index* dom_ptr_b = graph_b.get_domain_ptr();
1388 const Index* img_idx_b = graph_b.get_image_idx();
1389
1390 // loop over all velocity mirror entries, which correspond to the rows of B and the columns of D
1391 for(Index l(0); l < num_mir_idx; ++l)
1392 {
1393 // get velocity DOF index k = column index of D = row index of B
1394 const Index k = mir_idx[l];
1395
1396 // loop over all columns k of D
1397 for(Index ik(col_ptr_da[k]), si(0); (ik < col_ptr_da[k + 1]) && (si < used_rows_s); )
1398 {
1399 // check if we have found a common row of S and the k-th column of D
1400 if(row_idx_da[ik] < row_idx_s[si])
1401 {
1402 // row i exists in column k of D, but not in matrix S
1403 ++ik;
1404 continue;
1405 }
1406 if(row_idx_s[si] < row_idx_da[ik])
1407 {
1408 // row i exists in matrix S, but not in column k of D
1409 ++si;
1410 continue;
1411 }
1412
1413 // S_i. += (D_ik * A_kk) * B_l.
1414 for(IndexType ij(row_ptr_s[si]), kj(dom_ptr_b[l]); kj < dom_ptr_b[l+1]; ++ij)
1415 {
1416 ASSERT(ij < row_ptr_s[si+1]);
1417 ASSERT(col_idx_s[ij] <= img_idx_b[kj]);
1418 if(col_idx_s[ij] == img_idx_b[kj])
1419 {
1420 data_s[ij] += _mat_mult_da_b(data_da[ik], data_b[kj]);
1421 ++kj;
1422 }
1423 }
1424
1425 // okay, continue with next row of S and D
1426 ++ik;
1427 ++si;
1428 }
1429 }

1450 void _apply(LocalVectorTypeP& r, const LocalVectorTypeP& x, const LocalVectorTypeP& y, const DataType alpha, bool only_Ax) const
1451 {
1452 // get the number of our neighbors
1453 const std::size_t num_neighs = this->_ranks.size();
1454
1455 // no neighbors?
1456 if(num_neighs <= std::size_t(0))
1457 {
1458 // multiply by local schur matrix and return
1459 watch_apply_matrix_loc.start();
1460 if(only_Ax)
1461 this->_matrix_s.apply(r, x);
1462 else
1463 this->_matrix_s.apply(r, x, y, alpha);
1464 watch_apply_matrix_loc.stop();
1465 return;
1466 }
1467
1468 // get our communicator
1469 const Dist::Comm& comm = *this->_matrix_b.get_comm();
1470
1471 // send/receive buffers and requests
1472 std::vector<BufferVectorType> recv_bufs(num_neighs), send_bufs(num_neighs);
1473 Dist::RequestVector recv_reqs(num_neighs), send_reqs(num_neighs);
1474
1475 // allocate receive buffer vectors and post receives
1476 for(std::size_t i(0); i < num_neighs; ++i)
1477 {
1478 recv_bufs.at(i) = BufferVectorType(this->_neighbor_matrices.at(i).columns());
1479 recv_reqs[i] = comm.irecv(recv_bufs.at(i).elements(), recv_bufs.at(i).size(), this->_ranks.at(i));
1480 }
1481
