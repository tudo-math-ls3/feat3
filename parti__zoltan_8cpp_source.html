137 bool PartiZoltan::execute(const Adjacency::Graph& faces_at_elem, const Index num_parts)
138 {
139 // set number of desired partitions and total number of elements
140 this->_num_parts = num_parts;
141 this->_num_elems = faces_at_elem.get_num_nodes_domain();
142
143 // first of all, determine how many processes we want to use for Zoltan
144 int num_procs = Math::max(1, int(faces_at_elem.get_num_nodes_domain() / min_elems));
145 Math::mini(num_procs, max_procs);
146 Math::mini(num_procs, _comm.size());
147
148 // do we need to create a sub-communicator for Zoltan?
149 if(num_procs < _comm.size())
150 _zoltan_comm = _comm.comm_create_range_incl(num_procs);
151 else
152 _zoltan_comm = _comm.comm_dup();
153
154 // get my rank and size
155 //const int z_rank = _zoltan_comm.rank();
156 const int z_size = _zoltan_comm.size();
157
158 // make sure that the number of desired parts is not smaller than the size of the communicator
159 XASSERTM(Index(z_size) <= num_parts, "thou shall not create less partitions than thou hast ranks");
160
161 // if the size of the Zoltan communicator is zero, then this process does not participate
162 // in the actual partitioning process and it only needs to participate in the broadcast
163 bool is_ok = true;
164 if(z_size > 0)
165 {
166 // create our sub-hypergraph
167 this->_create_hypergraph(faces_at_elem);
168
169 // apply Zoltan partitioner
170 is_ok = this->_apply_zoltan();
171 }
172
173 // broadcast coloring from root rank and return
174 return this->_broadcast_coloring(is_ok);
175 }

177 void PartiZoltan::_create_hypergraph(const Adjacency::Graph& faces_at_elem)
178 {
179 // get number of elements and faces
180 const Index num_elems = faces_at_elem.get_num_nodes_domain();
181 const Index num_faces = faces_at_elem.get_num_nodes_image();
182
183 // get my rank and size
184 const int z_rank = _zoltan_comm.rank();
185 const int z_size = _zoltan_comm.size();
186
187 // this should not happen
188 XASSERT(z_size > 0);
189
190 // determine the first and the last element of this rank
191 Index first_elem = (Index(z_rank) * num_elems) / Index(z_size);
192 Index end_elem = (Index(z_rank+1) * num_elems) / Index(z_size);
193 Index my_num_elems = end_elem - first_elem;
194
195 const Index* dom_ptr = faces_at_elem.get_domain_ptr();
196 const Index* img_idx = faces_at_elem.get_image_idx();
197
198 // create a sub-graph for this rank
199 const Index first_ptr = dom_ptr[first_elem];
200 const Index my_num_idx = dom_ptr[end_elem] - first_ptr;
201 Adjacency::Graph faces_at_my_elem(my_num_elems, num_faces, my_num_idx);
202
203 Index* my_dom_ptr = faces_at_my_elem.get_domain_ptr();
204 Index* my_img_idx = faces_at_my_elem.get_image_idx();
205 for(Index i(0); i <= my_num_elems; ++i)
206 my_dom_ptr[i] = dom_ptr[first_elem + i] - first_ptr;
207 for(Index i(0); i < my_num_idx; ++i)
208 my_img_idx[i] = img_idx[first_ptr + i];
209
210 // transpose graph for this rank
211 Adjacency::Graph elems_at_faces(Adjacency::RenderType::transpose, faces_at_elem);
212
213 // combine graph into our sub-hypergraph
214 this->_hypergraph.first_elem = first_elem;
215 this->_hypergraph.sub_graph = Adjacency::Graph(Adjacency::RenderType::injectify_sorted, faces_at_my_elem, elems_at_faces);
216 }

218 bool PartiZoltan::_apply_zoltan()
219 {
220 // create Zoltan structure
221 Zoltan_Struct* zz = Zoltan_Create(_zoltan_comm.mpi_comm());
222 this->_zz = zz;
223
224 // define Zoltan parameters
225 Zoltan_Set_Param(zz, "DEBUG_LEVEL", "0");
226 Zoltan_Set_Param(zz, "LB_METHOD", "HYPERGRAPH");
227 Zoltan_Set_Param(zz, "HYPERGRAPH_PACKAGE", "PHG");
228 Zoltan_Set_Param(zz, "LB_APPROACH", "PARTITION");
229 Zoltan_Set_Param(zz, "NUM_GID_ENTRIES", "1");
230 Zoltan_Set_Param(zz, "NUM_LID_ENTRIES", "0");
231 Zoltan_Set_Param(zz, "RETURN_LISTS", "PARTS");
232 Zoltan_Set_Param(zz, "NUM_GLOBAL_PARTS", stringify(this->_num_parts).c_str());
233 Zoltan_Set_Param(zz, "OBJ_WEIGHT_DIM", "0");
234 Zoltan_Set_Param(zz, "EDGE_WEIGHT_DIM", "0");
235 //Zoltan_Set_Param(zz, "CHECK_GRAPH", "2");
236 Zoltan_Set_Param(zz, "PHG_EDGE_SIZE_THRESHOLD", "0.5");
237
238 // set hypergraph interface callbacks
239 Zoltan_Set_Num_Obj_Fn(zz, feat_zoltan_num_elems, &this->_hypergraph);
240 Zoltan_Set_Obj_List_Fn(zz, feat_zoltan_elem_ids, &this->_hypergraph);
241 Zoltan_Set_HG_Size_CS_Fn(zz, feat_zoltan_hypergraph_size, &this->_hypergraph);
242 Zoltan_Set_HG_CS_Fn(zz, feat_zoltan_hypergraph_data, &this->_hypergraph);
243
244 int changes(0), num_gid_entries(0), num_lid_entries(0), num_import(0), num_export(0);
245 ZOLTAN_ID_PTR import_global_ids(nullptr);
246 ZOLTAN_ID_PTR import_local_ids(nullptr);
247 int* import_procs(nullptr);
248 int* import_parts(nullptr);
249 ZOLTAN_ID_PTR export_global_ids(nullptr);
250 ZOLTAN_ID_PTR export_local_ids(nullptr);
251 int* export_procs(nullptr);
252 int* export_parts(nullptr);
253
254 int rtn = Zoltan_LB_Partition(zz, &changes, &num_gid_entries, &num_lid_entries,
255 &num_import, &import_global_ids, &import_local_ids, &import_procs, &import_parts,
256 &num_export, &export_global_ids, &export_local_ids, &export_procs, &export_parts);
257
258 // create coloring from Zoltan partitioning
259 bool gather_ok = true;
260 if(rtn == ZOLTAN_OK)
261 gather_ok = this->_gather_coloring(num_export, export_parts);
262
263 Zoltan_LB_Free_Part(&import_global_ids, &import_local_ids, &import_procs, &import_parts);
264 Zoltan_LB_Free_Part(&export_global_ids, &export_local_ids, &export_procs, &export_parts);
265
266 Zoltan_Destroy(&zz);
267 this->_zz = nullptr;
268
269 return gather_ok && (rtn == ZOLTAN_OK);
270 }

272 bool PartiZoltan::_gather_coloring(const int num_export, const int* export_parts)
273 {
274 // get my rank and size
275 const int z_rank = _zoltan_comm.rank();
276 const int z_size = _zoltan_comm.size();
277
278 // determine maximum export size
279 int max_export(0);
280 _zoltan_comm.allreduce(&num_export, &max_export, 1u, Dist::op_max);
281 max_export += 1; // for actual size
282
283 // allocate send buffer
284 std::vector<int> send_buf(max_export);
285 send_buf[0] = num_export;
286 for(int i(0); i < num_export; ++i)
287 send_buf[i+1] = export_parts[i];
288
289 // allocate receive buffer and gather coloring at rank 0
290 std::vector<int> recv_buf(z_rank == 0 ? max_export * z_size : std::size_t(0));
291 _zoltan_comm.gather(send_buf.data(), max_export, recv_buf.data(), max_export, 0);
292
293 // all ranks > 0 are done here
294 if(z_rank > 0)
295 return true;
296
297 // compute total element count
298 int num_elems(0);
299 for(int i(0); i < z_size; ++i)
300 num_elems += recv_buf[i*max_export];
301
302 // allocate ranks vector
303 this->_coloring = Adjacency::Coloring(num_elems, this->_num_parts);
304 Index* col = this->_coloring.get_coloring();
305
306 // build ranks vector
307 for(int i(0), k(0); i < z_size; ++i)
308 {
309 int off = i*max_export;
310 int n = recv_buf[off];
311 for(int j(1); j <= n; ++j, ++k)
312 col[k] = recv_buf[off+j];
313 }
314
315 // build color counts
316 std::vector<int> color_counts(this->_num_parts, 0);
317 for(Index i(0); i < this->_num_elems; ++i)
318 ++color_counts[col[i]];
319
320 // make sure we have at least 1 element per color/partition
321 for(Index i(0); i < this->_num_parts; ++i)
322 {
323 if(color_counts[i] == 0)
324 return false;
325 }
326
327 // okay
328 return true;
329 }

331 bool PartiZoltan::_broadcast_coloring(bool zoltan_ok)
332 {
333 // first of all, let's compare our Zoltan return codes;
334 // all processes which did not participate in the partitioning have a 'true' status
335 int was_ok(zoltan_ok ? 0 : 1), was_all_ok(0);
336 this->_comm.allreduce(&was_ok, &was_all_ok, std::size_t(1), Dist::op_sum);
337 if(was_all_ok > 0)
338 return false; // at least 1 process failed
339
340 // allocate coloring if not on root
341 if(this->_comm.rank() > 0)
342 this->_coloring = Adjacency::Coloring(this->_num_elems, this->_num_parts);
343
344 // broadcast coloring
345 this->_comm.bcast(this->_coloring.get_coloring(), this->_num_elems, 0);
346
347 // okay
348 return true;
349 }
