15 template<typename Space_, typename DT_, typename IT_, FEAT::Intern::MatrixGatherScatterPolicy pol_ = FEAT::Intern::MatrixGatherScatterPolicy::useLocalOps>
16 void full_burgers_assembler_matrix1_bcsr_host(DT_* matrix_data, const DT_* conv_data,
17 const IT_* matrix_row_ptr, const IT_* matrix_col_idx, Index matrix_num_rows, Index matrix_num_cols,

25 //define types
26 typedef Space_ SpaceType;
27 typedef DT_ DataType;
28 typedef IT_ IndexType;
29
30 const DataType& beta{burgers_params.beta};
31 const DataType& sd_delta{burgers_params.sd_delta};
32 const DataType& sd_v_norm{burgers_params.sd_v_norm};
33
34 const DataType tol_eps = Math::sqrt(Math::eps<DataType>());
35 // const DataType tol_eps = CudaMath::cuda_sqrt(DBL_EPSILON);
36 // const DataType tol_eps = DBL_EPSILON;
37 const bool need_streamline = (Math::abs(sd_delta) > DataType(0)) && (sd_v_norm > tol_eps);
38 const bool need_convection = Math::abs(beta) > DataType(0);
39
40 typedef VoxelAssembly::SpaceHelper<SpaceType, DT_, IT_> SpaceHelp;
41
42 constexpr int dim = SpaceHelp::dim;
43
44 //define local sizes
45 constexpr int num_loc_dofs = SpaceType::DofMappingType::dof_count;
46 //get number of nodes per element
47 constexpr int num_loc_verts = SpaceType::MeshType::template IndexSet<dim, 0>::Type::num_indices;
48
49 //our local datatypes
50 typedef Tiny::Vector<DataType, dim> VecValueType;
51 typedef Tiny::Matrix<DataType, dim, dim> MatValueType;
52 typedef Tiny::Vector<VecValueType, num_loc_dofs> LocalVectorType;
53 typedef Tiny::Matrix<MatValueType, num_loc_dofs, num_loc_dofs> LocalMatrixType;
54
55 #pragma omp parallel for
56 for(Index idx = 0; idx < coloring_size; ++idx)
57 {
58 // define local coefficients
59 DataType local_coeffs[dim][num_loc_verts];
60 // and local matrix
61 LocalMatrixType loc_mat;
62 LocalVectorType local_conv_dofs(DataType(0));
63 // now do work for this cell
64 Index cell = Index(coloring_map[idx]);
65 // std::cout << "Starting with cell " << cell << std::endl;
66 const IndexType* local_dofs = cell_to_dof + cell*num_loc_dofs;
67 const IndexType* local_dof_sorter = cell_to_dof_sorter + cell*num_loc_dofs;
68
69 SpaceHelp::set_coefficients(local_coeffs, local_dofs, nodes);
70 //if we need to, gather local convection vector
71 if(need_convection || need_streamline) //need stream diff or convection?
72 {
73 LAFEM::template VectorGatherScatterHelper<SpaceType, DataType, IndexType>::gather_vector_dense(local_conv_dofs,
74 (const VecValueType*)conv_data, IndexType(matrix_num_rows), local_dofs,DataType(1));
75 }
76
77 VoxelAssembly::Kernel::burgers_mat_assembly_kernel<SpaceHelp>(loc_mat, local_conv_dofs, local_coeffs, cub_pt, cub_wg, num_cubs, burgers_params,
78 need_streamline, need_convection, tol_eps);
79
80 //scatter
81 LAFEM::template MatrixGatherScatterHelper<SpaceType, DataType, IndexType, pol_>::scatter_matrix_csr(loc_mat, (MatValueType*)matrix_data, local_dofs, local_dofs, IndexType(matrix_num_rows), IndexType(matrix_num_cols), matrix_row_ptr, matrix_col_idx, alpha, local_dof_sorter);

166 template<typename DT_, int dim_>
167 void set_sd_v_norm_host(const Tiny::Vector<DT_, dim_>* convect, DT_* result, Index vec_size)
168 {
169 DT_ max_val(DT_(0));
170 // simply reduce over max_val
171 #pragma omp parallel for reduction(max : max_val)
172 for(int i = 0; i < int(vec_size); ++i)
173 {
174 //synchronize all threads in block, to guarentee that all values are written
175 max_val = Math::max(convect[i].norm_euclid(), max_val);
176 }
177 //and overwrite
178 *result = max_val;
179
180 }
