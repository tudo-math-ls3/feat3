69 template<typename DT_, int world_dim, int sc_, int sp_, int smx_, int snx_>
70 void inverse_mapping(
71 Tiny::Vector<DT_, world_dim, sc_>& coeffs,
72 const Tiny::Vector<DT_, world_dim, sp_>& point,
73 const Tiny::Matrix<DT_, world_dim+1, world_dim, smx_, snx_>& x)
74 {
75 // A will contain the transformation matrix
76 Tiny::Matrix<DT_, world_dim, world_dim> A(DT_(0));
77 for(int i(0); i < world_dim; ++i)
78 {
79 for(int j(0); j < world_dim; ++j)
80 {
81 A(i,j) = x(j+1,i) - x(0,i);
82 }
83 }
84
85 // This will be the inverse of the transformation matrix
86 Tiny::Matrix<DT_, world_dim, world_dim> Ainv(DT_(0));
87 Ainv.set_inverse(A);
88
89 // This is the solution of A u = point - x[0]
90 coeffs = Ainv*(point - x[0]);
91
92 }

142 template<typename DT_, int world_dim, int sc_, int sp_, int smx_, int snx_>
143 void inverse_mapping(
144 Tiny::Vector<DT_, 2, sc_>& coeffs,
145 const Tiny::Vector<DT_, world_dim, sp_>& point,
146 const Tiny::Matrix<DT_, 2, world_dim, smx_, snx_>& x)
147 {
148 static_assert( (world_dim == 2 || world_dim == 3),
149 "world dim has to be 2 or 3 for complementary barycentric coordinates");
150
151 auto tmp = x[1]-x[0];
152 DT_ sp(Tiny::dot(point - x[0],tmp));
153 DT_ nsqr(Math::sqr(tmp.norm_euclid()));
154 // This is omega
155 coeffs[0] = sp/nsqr;
156 tmp = point - (x[0] + coeffs[0]*(x[1]-x[0]));
157 // This is the distance of point to the straight line defined by x[0], x[1]
158 coeffs[1] = tmp.norm_euclid();
159 }

206 template<typename DT_, int sc_, int sp_, int smx_, int snx_>
207 void inverse_mapping(
208 Tiny::Vector<DT_, 3, sc_>& coeffs,
209 const Tiny::Vector<DT_, 3, sp_>& point,
210 const Tiny::Matrix<DT_, 3, 3, smx_, snx_>& x)
211 {
212 static constexpr int world_dim = 3;
213 static constexpr int shape_dim = world_dim-1;
214
215 // A will contain the transformation matrix
216 Tiny::Matrix<DT_, world_dim, world_dim> A(DT_(0));
217 // Fill the all rows in the first shape_dim = world_dim-1 columns
218 for(int i(0); i < world_dim; ++i)
219 {
220 for(int j(0); j < shape_dim; ++j)
221 A(i,j) = x(j+1,i) - x(0,i);
222 }
223
224 // The last column is the additional direction for our augmented simplex and it is orthogonal to the rest
225 Tiny::Vector<DT_, world_dim> ortho = Tiny::orthogonal(A.template size_cast<world_dim, shape_dim>());
226
227 // In 3d, the 2-norm of ortho is the 2d volume of the parallelogram defined by A[0], A[1]. That makes this
228 // axis badly scaled if the parallelogram is either very small or very large. So we rescale ortho to unity
229 // so the resulting coefficient is just the distance
230 ortho.normalize();
231
232 // Set the last column in A
233 for(int i(0); i < world_dim; ++i)
234 A(i,world_dim-1) = ortho(i);
235
236 // This will be the inverse of the transformation matrix
237 Tiny::Matrix<DT_, world_dim, world_dim> Ainv(DT_(0));
238 Ainv.set_inverse(A);
239
240 // This is the solution of A u = point - x[0]
241 coeffs = Ainv*(point - x[0]);
242 }
