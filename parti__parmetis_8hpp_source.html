126 template<int nv_, typename VertexSet_>
127 void _compute_midpoints(const IndexSet<nv_>& verts_at_elem, const VertexSet_& vertices)
128 {
129 typename VertexSet_::VertexType vx;
130 const Real sc = Real(1) / Real(verts_at_elem.num_indices);
131 static constexpr int nc = VertexSet_::num_coords;
132 _midpoints.reserve(_num_local_elems * Index(nc));
133 for(Index i(0); i < _num_local_elems; ++i)
134 {
135 vx.format();
136 for(int j(0); j < verts_at_elem.num_indices; ++j)
137 vx += vertices[verts_at_elem(i,j)];
138 for(int k(0); k < nc; ++k)
139 _midpoints.push_back(Real(vx[k]) * sc);
140 }
