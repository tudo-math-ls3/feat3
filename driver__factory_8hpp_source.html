189 static bool create(Rule<Shape_, Weight_, Coord_, Point_>& rule, int num_points)

216 template<typename Weight_, typename Coord_, typename Point_>
217 static bool create(Rule<Shape_, Weight_, Coord_, Point_>& rule, const String& name)
218 {
219 // map alias names
220 Intern::DriverFactoryAliasMapper<DriverType> mapper(name);
221 DriverType::alias(mapper);
222 String mapped_name(mapper.name());
223
224 // try to find a colon within the string
225 String::size_type k = mapped_name.find_first_of(':');
226 if(k == mapped_name.npos)
227 return false;
228
229 // extract substrings until the colon
230 String head(mapped_name.substr(0, k));
231 String tail(mapped_name.substr(k + 1));
232
233 // check head - this is the name of the formula
234 if(head.trim().compare_no_case(DriverType::name()) != 0)
235 return false;
236
237 // check substring
238 int num_points(0);
239 if(!tail.trim().parse(num_points))
240 return false;
241
242 // try to create the rule
243 return create(rule, num_points);
244 }
