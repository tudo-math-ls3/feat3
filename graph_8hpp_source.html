157 template<typename Adjactor_>
158 explicit Graph(RenderType render_type, const Adjactor_& adjactor) :
159 _num_nodes_image(0),
160 _domain_ptr(),
161 _image_idx()
162 {
163 switch(render_type)
164 {
165 case RenderType::as_is:
166 case RenderType::as_is_sorted:
167 _render_as_is(adjactor);
168 if(render_type == RenderType::as_is_sorted)
169 this->sort_indices();
170 break;
171
172 case RenderType::injectify:
173 case RenderType::injectify_sorted:
174 _render_injectify(adjactor);
175 if(render_type == RenderType::injectify_sorted)
176 this->sort_indices();
177 break;
178
179 case RenderType::transpose:
180 case RenderType::transpose_sorted:
181 _render_transpose(adjactor);
182 // transpose is automatically sorted
183 break;
184
185 case RenderType::injectify_transpose:
186 case RenderType::injectify_transpose_sorted:
187 _render_injectify_transpose(adjactor);
188 // transpose is automatically sorted
189 break;
190
191 default:
192 XABORTM("Invalid render_type parameter!");
193 }

403 template<typename Adjactor_>
404 void _render_as_is(const Adjactor_& adj)
405 {
406 typedef typename Adjactor_::ImageIterator AImIt;
407
408 // get counts
409 //_num_nodes_domain = adj.get_num_nodes_domain();
410 _num_nodes_image = adj.get_num_nodes_image();
411 Index num_indices_image = 0;
412
413 // allocate pointer vector
414 _domain_ptr = IndexVector(adj.get_num_nodes_domain() +1);
415
416 // count number of adjacencies and build pointer vector
417 for(Index i(0); i < adj.get_num_nodes_domain(); ++i)
418 {
419 _domain_ptr[i] = num_indices_image;
420 AImIt cur(adj.image_begin(i));
421 AImIt end(adj.image_end(i));
422 for(; cur != end; ++cur)
423 {
424 ++num_indices_image;
425 }
426 }
427 _domain_ptr[adj.get_num_nodes_domain()] = num_indices_image;
428
429 // allocate and build index vector
430 _image_idx = IndexVector(num_indices_image);
431 for(Index i(0); i < adj.get_num_nodes_domain(); ++i)
432 {
433 Index* idx = &_image_idx[_domain_ptr[i]];
434 AImIt cur(adj.image_begin(i));
435 AImIt end(adj.image_end(i));
436 for(; cur != end; ++cur, ++idx)
437 {
438 *idx = *cur;
439 }
440 }

458 // count number of adjacencies and build pointer vector
459 for(Index i(0); i < adj.get_num_nodes_domain(); ++i)

614 {

616 XASSERTM(adj1.get_num_nodes_image() == adj2.get_num_nodes_domain(), "Adjactor dimension mismatch!");
617
618 typedef typename Adjactor1_::ImageIterator AImIt1;
619 typedef typename Adjactor2_::ImageIterator AImIt2;
620
621 // get counts
622 _num_nodes_image = adj2.get_num_nodes_image();
623 Index num_indices_image = 0;
624
625 // allocate pointer vector
626 _domain_ptr = IndexVector(adj1.get_num_nodes_domain() + 1);
627
628 // count number of adjacencies and build pointer vector
629 for(Index i(0); i < adj1.get_num_nodes_domain(); ++i)
630 {
631 _domain_ptr[i] = num_indices_image;
632 AImIt1 cur1(adj1.image_begin(i));
633 AImIt1 end1(adj1.image_end(i));
634 for(; cur1 != end1; ++cur1)
635 {
636 AImIt2 cur2(adj2.image_begin(*cur1));
637 AImIt2 end2(adj2.image_end(*cur1));
638 for(; cur2 != end2; ++cur2)
639 {
640 ++num_indices_image;
641 }
642 }
643 }
644 _domain_ptr[adj1.get_num_nodes_domain()] = num_indices_image;
645
646 // allocate and build index vector
647 _image_idx =IndexVector(num_indices_image);
648 for(Index i(0); i < adj1.get_num_nodes_domain(); ++i)
649 {
650 Index* idx = &_image_idx[_domain_ptr[i]];
651 AImIt1 cur1(adj1.image_begin(i));
652 AImIt1 end1(adj1.image_end(i));
653 for(; cur1 != end1; ++cur1)
654 {
655 AImIt2 cur2(adj2.image_begin(*cur1));
656 AImIt2 end2(adj2.image_end(*cur1));
657 for(; cur2 != end2; ++cur2, ++idx)
