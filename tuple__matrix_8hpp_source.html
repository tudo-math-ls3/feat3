343
345 void restore_from_checkpoint_data(std::vector<char> & data)
346 {
347 std::uint64_t isize = *(std::uint64_t*) data.data(); //get size of checkpointed _first
348 std::vector<char>::iterator start = std::begin(data) + sizeof(std::uint64_t);
349 std::vector<char>::iterator last_of_first = std::begin(data) + sizeof(std::uint64_t) + (int) isize;
350 std::vector<char> buffer_first(start, last_of_first);
351 _first.restore_from_checkpoint_data(buffer_first);
352
353 data.erase(std::begin(data), last_of_first);
354 _rest.restore_from_checkpoint_data(data);
355 }
356
358 std::uint64_t set_checkpoint_data(std::vector<char>& data, SerialConfig& config)
359 {
360 std::size_t old_size = data.size();
361 data.insert(std::end(data), sizeof(std::uint64_t), 0); //add placeholder
362 std::uint64_t ireal_size = _first.set_checkpoint_data(data, config); //add data of _first to the overall checkpoint and save its size
363 char* csize = reinterpret_cast<char*>(&ireal_size);
364 for(std::size_t i(0) ; i < sizeof(std::uint64_t) ; ++i) //overwrite the guessed datalength
365 {
366 data[old_size + i] = csize[i];
367 }
368
369 return sizeof(std::uint64_t) + ireal_size + _rest.set_checkpoint_data(data, config); //generate and add checkpoint data for the _rest
370 }
