429 struct FileHeader
430 {
431 u64 magic_number; //< "VoxelMap" = 0x70614D6C65786F56ll
432 u64 header_size; //< 128 bytes
433 i64 min_x; //< minimum X-coordinate in nanometer
434 i64 max_x; //< maximum X-coordinate in nanometer
435 u64 num_x; //< number of points in X dimension
436 u64 stride_line; //< number of bytes for a single uncompressed X-line; a multiple of 16; usually stride_line = (((num_x + 7ull)/8ull + 15ull) & ~15ull)
437 i64 min_y; //< minimum Y-coordinate in nanometer
438 i64 max_y; //< maximum Y-coordinate in nanometer
439 u64 num_y; //< number of points in Y dimension
440 u64 stride_plane; //< number of bytes for a single uncompressed XY-plane; a multiple of 16; usually stride_plane = stride_line * num_y
441 i64 min_z; //< minimum Z-coordinate in nanometer
442 i64 max_z; //< maximum Z-coordinate in nanometer
443 u64 num_z; //< number of points in Z dimension
444 u64 stride_volume; //< number of bytes for a single uncompressed XYZ-volume; a multiple of 16; usually stride_volume = stride_plane * num_z
445 u64 planes_per_block; //< number of planes per compression block (0 if uncompressed)
446 u64 num_blocks; //< total number of compression blocks (0 if uncompressed)
447 }; // struct FileHeader: 128 bytes
448
450 static_assert(sizeof(FileHeader) == header_size, "invalid FileHeader size");
451
453 static u64 calc_line_stride(u64 num_x)
454 {
455 // compute number of bytes by rounding to next multiple of 8 then dividing by 8,
456 // afterwards round up to next multiple of 16
457 return (((num_x + 7ull)/8ull + 15ull) & ~15ull);
458 }

701 void set_num_slices(Index num_x, Index num_y, Index num_z = 0u)
702 {
703 XASSERT(num_x > 0ull);
704 XASSERT(num_y > 0ull);
705 XASSERT((num_z > 0ull) || (_bbox_min[2u] == _bbox_max[2u]));
706 _num_slices[0] = num_x;
707 _num_slices[1] = num_y;
708 _num_slices[2] = (num_z > 0u ? num_z : u64(1));
709
710 // compute other quantities
711 _stride_line = calc_line_stride(_num_slices[0]);
712 _stride_plane = _stride_line * _num_slices[1];
713 _num_planes = _num_slices[2];
714 _num_lines = _num_slices[1] * _num_planes;
715
716 // allocate voxel map
717 _voxel_map.resize(_num_planes * _stride_plane, 0u);
718 }

1085 WriteResult write(std::ostream& os, const u64 compress_block_size = 128u, const int compress_level = 9) const
1086 {
1087 std::size_t written = 0u;
1088
1089 // set up file header
1090 FileHeader header;
1091 memset(&header, 0, sizeof(header));
1092 header.magic_number = magic_number;
1093 header.header_size = sizeof(header);
1094 header.min_x = _bbox_min[0];
1095 header.max_x = _bbox_max[0];
1096 header.num_x = _num_slices[0];
1097 header.stride_line = _stride_line;
1098 header.min_y = _bbox_min[1];
1099 header.max_y = _bbox_max[1];
1100 header.num_y = _num_slices[1];
1101 header.stride_plane = _stride_plane;
1102 header.min_z = _bbox_min[2];
1103 header.max_z = _bbox_max[2];
1104 header.num_z = _num_slices[2];
1105 header.stride_volume = _stride_plane * _num_slices[2];
1106
1107 XASSERT(_voxel_map.size() == header.stride_volume);
1108
1109 // do we have to compress the map?
1110#ifdef FEAT_HAVE_ZLIB
1111 if(compress_block_size > 0u)
1112 {
1113 XASSERTM(compress_block_size < 1024u, "maximum compression block size is 1024 MB");
1114
1115 // compute number of planes per compression block
1116 header.planes_per_block = (compress_block_size*1024ull*1024ull) / _stride_plane;
1117 XASSERTM(header.planes_per_block > 0ull, "compression block size is to small to hold a single plane!");
1118
1119 header.num_blocks = (_num_planes + header.planes_per_block - 1u) / header.planes_per_block;
1120 XASSERTM(header.num_blocks > 0ull, "invalid number of compression blocks!");
1121 }
1122#endif // FEAT_HAVE_ZLIB
1123
1124 // write header
1125 os.write(reinterpret_cast<char*>(&header), sizeof(header));
1126 written += sizeof(header);
1127
1128 // write uncompressed voxel map data
1129 if(header.num_blocks == 0ull)
1130 {
1131 // write uncompressed voxel map data
1132 os.write(reinterpret_cast<const char*>(_voxel_map.data()), std::streamsize(header.stride_volume));
1133 return WriteResult(written + header.stride_volume);
1134 }
1135
1136#ifdef FEAT_HAVE_ZLIB
1137 // ensure that a compression block is not larger than 1 GB
1138 XASSERTM(header.planes_per_block * _stride_plane < 1073741824ull, "voxel map plane compression block size is too big");
1139
1140 // compression buffers
1141 std::vector<std::vector<char>> compress_buffers(header.num_blocks);
1142
1143 // compression block entries
1144 std::vector<u64> compress_blocks(header.num_blocks, 0u);
1145
1146 // get compression level
1147 const int cmp_lvl = Math::max(Math::min(compress_level, 9), 0);
1148
1149 // accumulated ZLIB results
1150 int failures = 0;
1151
1152 // loop over all compression blocks
1153 #pragma omp parallel for schedule(dynamic,1) reduction(+:failures)
1154 for(u64 iblock = 0; iblock < header.num_blocks; ++iblock)
1155 {
1156 // compute first and last planes
1157 u64 first_plane = iblock * header.planes_per_block;
1158 u64 block_bytes = Math::min(header.planes_per_block, _num_planes - first_plane) * _stride_plane;
1159
1160 // estimate compression buffer size
1161 u64 buffer_size = u64(::compressBound(uLong(block_bytes)));
1162
1163 // allocate compression buffer
1164 compress_buffers[iblock].resize(buffer_size);
1165
1166 // compression buffer size
1167 uLongf dest_size = uLongf(buffer_size);
1168
1169 // compress via ZLIB
1170 int rtn = ::compress2(reinterpret_cast<Bytef*>(compress_buffers[iblock].data()), &dest_size,
1171 reinterpret_cast<const Bytef*>(&_voxel_map[first_plane * _stride_plane]), uLongf(block_bytes), cmp_lvl);
1172
1173 // failure?
1174 if(rtn != Z_OK)
1175 {
1176 ++failures;
1177 continue;
1178 }
1179
1180 // save compression buffer size
1181 compress_blocks[iblock] = u64(dest_size);
1182 }
1183
1184 // did any compression fail?
1185 XASSERTM(failures == 0, "ZLIB failed to compress at least one plane block!");
1186
1187 // write the compression block headers
1188 os.write(reinterpret_cast<const char*>(compress_blocks.data()), std::streamsize(compress_blocks.size() * sizeof(u64)));
1189 written += compress_blocks.size() * sizeof(u64);
1190
1191 // okay, loop over all compression blocks and write them out
1192 u64 compress_size = 0u;
1193 for(std::size_t i(0); i < compress_buffers.size(); ++i)
1194 {
1195 os.write(reinterpret_cast<const char*>(compress_buffers[i].data()), std::streamsize(compress_blocks[i]));
1196 written += compress_blocks[i];
1197 compress_size += compress_blocks[i];
1198 }
1199
1200 // return result
1201 return WriteResult(written, compress_block_size, u64(compress_level), compress_buffers.size(), compress_size);
1202#else
1203 // we should never arrive in this #else case even if compiling without zlib
1204 (void)compress_block_size;
1205 (void)compress_level;
1206 XABORTM("INTERNAL ERROR");
1207#endif // FEAT_HAVE_ZLIB
1208 }

1899 template<typename CoordType, int dim_>
1900 void _compute_voxel_map(const Dist::Comm& comm, VoxelMasker<CoordType, dim_>& masker, bool gather_to_all)
1901 {
1902 // single process or MPI parallel?
1903 if(comm.size() <= 1)
1904 {
1905 // compute all voxel map lines
1906 this->_compute_voxel_map_lines(masker, 0u, this->_num_lines, 0u);
1907 }
1908 else if(u64(comm.size()) <= this->_num_lines)
1909 {
1910 // there are at least as many lines in the voxel map as there are MPI processes; this is the usual case
1911 // we can use the entire communicator (usually MPI_COMM_WORLD) for the voxel map computation
1912 Index comm_rank = Index(comm.rank());
1913 Index comm_size = Index(comm.size());
1914
1915 // compute line count for a single process and round up; this may be bigger than _num_lines,
1916 // so the last process may have less lines to chew through than all other processes
1917 Index line_count = (this->_num_lines + comm_size - 1u) / comm_size;
1918 std::size_t block_len = this->_stride_line * line_count;
1919
1920 // allocate voxel map and format to zero
1921 std::size_t vmap_size = block_len;
1922 // do we need the entire map on this process?
1923 if(gather_to_all || (comm_rank == 0))
1924 vmap_size *= Index(comm_size);
1925 this->_voxel_map.resize(vmap_size, 0u);

1957 // so the last process may have less lines to chew through than all other processes
1958 Index line_count = (this->_num_lines + comm_size - 1u) / comm_size;
1959 std::size_t block_len = this->_stride_line * line_count;
1960
1961 // allocate voxel map and format to zero
1962 std::size_t vmap_size = block_len;
1963 // do we need the entire map on this process?

1967

1971

1977

1979 sub_comm.gather(this->_voxel_map.data(), block_len, this->_voxel_map.data(), block_len, 0);
1980 }
1981 else
1982 {

1984 if(gather_to_all)
1985 this->_voxel_map.resize(this->_stride_line * this->_num_lines, 0u);
1986 }
