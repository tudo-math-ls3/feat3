230 typedef typename EvalPolicy::HessianTensorType HessianTensorType;

233

237 static constexpr int image_dim = EvalPolicy::image_dim;

241
242 protected:
244 DataType _coeff[image_dim];
245
246 public:
253 explicit Evaluator(const TrafoType& trafo) :
254 BaseClass(trafo)

264 void prepare(Index cell_index)
265 {
266 // prepare base-class
267 BaseClass::prepare(cell_index);
268
269 // fetch the mesh from the trafo
270 const MeshType& mesh = this->_trafo.get_mesh();
271
272 // fetch the vertex set from the mesh
273 typedef typename MeshType::VertexSetType VertexSetType;
274 const VertexSetType& vertex_set = mesh.get_vertex_set();
275
276 // fetch the vertex
277 typedef typename VertexSetType::VertexType VertexType;
278 const VertexType& vtx = vertex_set[cell_index];
279
280 // calculate transformation coefficients
281 for(int i(0); i < image_dim; ++i)
282 {
283 _coeff[i] = DataType(vtx[i]);
284 }
285 }
286
296 void map_point(ImagePointType& img_point, const DomainPointType& DOXY(dom_point)) const
297 {
298 for(int i(0); i < image_dim; ++i)
299 {
300 img_point[i] = _coeff[i];
301 }
302 }
