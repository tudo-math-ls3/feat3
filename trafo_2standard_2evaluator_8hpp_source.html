516 DataType width_directed(const ImagePointType& ray) const
517 {
518 // This one is a little tricky:
519 // We follow a similar approach as on quadrilaterals here, i.e. we first map the ray
520 // vector onto the reference cells by multiplying it to the inverse jacobian matrix.
521 // However, our reference cell is the "right triangle" spanned by the three points
522 // (0,0) (1,0) (0,1) and therefore its edges have different lengths, which would
523 // result in a "skew" width. To circumvent this, we will afterwards map the ray
524 // vector onto a "equilateral unit triangle" (i.e. the triangle where all edges
525 // have length = 1) and we compute the norm of that mapped vector then.
526 JacobianMatrixType jac_mat;
527 JacobianInverseType jac_inv;
528 DomainPointType ref_ray;
529
530 // compute jacobian matrix
531 for(int i(0); i < image_dim; ++i)
532 {
533 jac_mat(i,0) = _coeff[i][1];
534 jac_mat(i,1) = _coeff[i][2];
535 }
536
537 // invert jacobian matrix and multiply by ray vector
538 jac_inv.set_inverse(jac_mat);
539 ref_ray.set_mat_vec_mult(jac_inv, ray);
540
541 // We have mapped the ray onto our reference triangle. Now let's map that one
542 // onto a equilateral unit triangle, which can be performed by multiplying
543 // the following matrix by the reference ray vector:
544 //
545 // eut_ray := [ -1 +1 ] * ref_ray
546 // [ sqrt(3) sqrt(3) ]
547 //
548 // Finally, we have to compute the norm of that vector, which is explicitly
549 // written down in the denominator of the following expression:
550 return DataType(2) / Math::sqrt(Math::sqr(ref_ray[1]-ref_ray[0]) + DataType(3) * Math::sqr(ref_ray[0]+ref_ray[1]));
551 }
552 }; // class Evaluator<Simplex<2>,...>

749 // First, map the ray onto the reference element and then
750 // map it to a regular tetrahedron to compute its norm.
751 JacobianMatrixType jac_mat;
752 JacobianInverseType jac_inv;
753 DomainPointType ref_ray;
754
755 // compute jacobian matrix
756 for(int i(0); i < image_dim; ++i)
757 {
758 jac_mat(i,0) = _coeff[i][1];
759 jac_mat(i,1) = _coeff[i][2];
760 jac_mat(i,2) = _coeff[i][3];
761 }
762
763 // invert jacobian matrix and multiply by ray vector
764 jac_inv.set_inverse(jac_mat);

766
767 // The transformation from the reference tetrahedron to the regular unit tetrahedron
768 // can be performed by applying the following matrix-vector product:
769 //
770 // [ 2 1 1 ]
771 // eut_ray := [ 0 1 -1 ] * ref_ray
772 // [ 0 sqrt(2) sqrt(2) ]
773 //
774 // Finally, we have to compute the norm of that vector, which is explicitly
775 // written down in the denominator of the following expression:
776 return DataType(2) / Math::sqrt(Math::sqr(DataType(2) * ref_ray[0] + ref_ray[1] + ref_ray[2]) +
777 Math::sqr(ref_ray[1] - ref_ray[2]) + DataType(2) * Math::sqr(ref_ray[1] + ref_ray[2]));

1136 DataType volume() const
1137 {
1138 // According to Varignon's theorem, the area/volume of a quadrilateral is
1139 // equal to twice the area of the dual parallelogram of the quadrilateral,
1140 // which is spanned by the four edge midpoints of the quadrilateral.
1141 // The Jacobian matrix of this transformation evaluated at the barycentre
1142 // of the reference element spans a parallelogram, which intersects with
1143 // our original quadrilateral in the edge midpoints and therefore (again
1144 // using Varignon's theorem) has the same area as the original quadrilateral.
1145 // Now the area of the "Jacobian parallelogram" is equal to four times
1146 // the determinant of its Jacobian determinant, which finally gives us a
1147 // formula for our quadrilateral area: 4*det(Jacobian(0,0)).
1148 // Note that this is not a lousy approximation, but a true identity :)
1149
1150 // compute jacobian matrix at barycentre
1151 JacobianMatrixType jac_mat;
1152 for(int i(0); i < image_dim; ++i)
1153 {
1154 jac_mat(i,0) = _coeff[i][1];
1155 jac_mat(i,1) = _coeff[i][2];
1156 }
1157
1158 // return scaled volume
1159 return DataType(4) * jac_mat.vol();
1160 }
