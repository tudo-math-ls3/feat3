301 void prepare(const TrafoEvaluator& trafo_eval)
302 {
303 // create a domain point in the barycentre of the cell
304 DomainPointType dom_point(DataType(0));
305
306 // create the trafo data
307 InvLinTrafoData trafo_data;
308 trafo_eval(trafo_data, dom_point);
309
310 // store inverse trafo linearization
311 _inv_lin_mat = trafo_data.jac_inv;
312 _inv_lin_vec = trafo_data.img_point;
313 }

316 template<SpaceTags space_cfg_, TrafoTags trafo_cfg_>
317 void eval_values(
318 EvalData<SpaceEvalTraits, space_cfg_>& data,
319 const Trafo::EvalData<TrafoEvalTraits, trafo_cfg_>& trafo_data) const
320 {
321 // transform image point
322 DomainPointType pt;
323 pt.set_mat_vec_mult(_inv_lin_mat, trafo_data.img_point - _inv_lin_vec);
324
325 // evaluate basis functions
326 data.phi[0].value = DataType(1);
327 for(int i(0); i < shape_dim_; ++i)
328 data.phi[i+1].value = pt[i];
329 }
330
332 template<SpaceTags space_cfg_, TrafoTags trafo_cfg_>
333 void eval_gradients(
334 EvalData<SpaceEvalTraits, space_cfg_>& data,
335 const Trafo::EvalData<TrafoEvalTraits, trafo_cfg_>& trafo_data) const
336 {
337 // transform image point to local coordinate system
338 DomainPointType pt, loc_grad;
339 pt.set_mat_vec_mult(_inv_lin_mat, trafo_data.img_point - _inv_lin_vec);
340
341 // loop over all basis functions
342 for(int i(0); i < (shape_dim_+1); ++i)
343 {
344 // compute local gradient
345 for(int j(0); j < shape_dim_; ++j)
346 loc_grad[j] = DataType(i == (j+1) ? 1 : 0);
347
348 // multiply by transpose inverse linearized trafo matrix for chain rule
349 data.phi[i].grad.set_vec_mat_mult(loc_grad, _inv_lin_mat);
350 }
351 }
