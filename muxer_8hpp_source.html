336 void compile(const LocalVector_& vec_tmp_)
337 {
338 if(!is_child())
339 return; // nothing to
340
341 XASSERT(_sibling_comm != nullptr);
342 XASSERT(_parent_rank >= 0);
343 XASSERT(_parent_rank < _sibling_comm->size());
344
345 Index bufsize(0);
346
347 // Is this the parent?
348 if(_sibling_comm->rank() == _parent_rank)
349 {
350 // loop over all child buffers and compute the largest buffer size
351 for(const auto& cm : _child_mirrors)
352 {
353 bufsize = Math::max(bufsize, cm.buffer_size(vec_tmp_));
354 }
355 }
356
357 // broadcast buffer size to all siblings
358 _sibling_comm->bcast(&bufsize, std::size_t(1), _parent_rank);
359
360 // verify against length of parent buffer
361 XASSERT(bufsize >= _parent_mirror.buffer_size(vec_tmp_));
362
363 // store buffer size
364 this->_buffer_size = bufsize;
365 }

377 void join_send(const LocalVector_& vec_src) const
378 {
379 XASSERT(_sibling_comm != nullptr);
380 XASSERT(_sibling_comm->size() > 1);
381 XASSERT(_sibling_comm->rank() != _parent_rank); // parent must call join() instead
382
383 // create parent buffer in device memory
384 BufferType parent_buffer(_buffer_size);
385
386 // gather source to parent buffer
387 _parent_mirror.gather(parent_buffer, vec_src);
388
389 // gather to parent sibling
390 DataType* dummy = nullptr;
391 _sibling_comm->gather(parent_buffer.elements(), _buffer_size, dummy, std::size_t(0), _parent_rank);
392 }

403 void join(const LocalVector_& vec_src, LocalVector_& vec_trg) const
404 {
405 // if this muxer is not a child, then this operation is a simple copy
406 if((_sibling_comm == nullptr) || (_sibling_comm->size() <= 1))
407 {
408 vec_trg.copy(vec_src);
409 return;
410 }
411
412 XASSERT(_sibling_comm != nullptr);
413 XASSERT(_sibling_comm->size() > 0);
414 XASSERT(_sibling_comm->rank() == _parent_rank);
415
416 const Index num_children = Index(_child_mirrors.size());
417
418 // allocate parent buffer
419 BufferType parent_buffer(_buffer_size);
420
421 // gather source to parent buffer
422 _parent_mirror.gather(parent_buffer, vec_src);
423
424 // allocate child buffers
425 BufferType child_buffers(_buffer_size * num_children);
426
427 // gather from siblings
428 _sibling_comm->gather(parent_buffer.elements(), _buffer_size, child_buffers.elements(), _buffer_size, _parent_rank);
429
430 // format target vector
431 vec_trg.format();
432
433 // scatter child buffers into target vector
434 for(Index i(0); i < num_children; ++i)
435 {
436 // scatter buffers
437 _child_mirrors.at(i).scatter_axpy(vec_trg, child_buffers, DataType(1), i*_buffer_size);
438 }
439 }

478 void split(LocalVector_& vec_trg, const LocalVector_& vec_src) const
479 {
480 // if this muxer is not a child, then this operation is a simple copy
481 if((_sibling_comm == nullptr) || (_sibling_comm->size() <= 1))
482 {
483 vec_trg.copy(vec_src);
484 return;
485 }
486
487 XASSERT(_sibling_comm != nullptr);
488 XASSERT(_sibling_comm->size() > 0);
489 XASSERT(_sibling_comm->rank() == _parent_rank);
490
491 TimeStamp ts_execute;
492
493 const Index num_children = Index(_child_mirrors.size());
494
495 // allocate child buffers
496 BufferType child_buffers(_buffer_size * num_children);
497
498 // gather child buffers from target vector
499 for(Index i(0); i < num_children; ++i)
500 {
501 _child_mirrors.at(i).gather(child_buffers, vec_src, i*_buffer_size);
502 }
503
504 // create parent buffer
505 BufferType parent_buffer(_buffer_size);
506
507 // scatter to siblings
508 _sibling_comm->scatter(child_buffers.elements(), _buffer_size, parent_buffer.elements(), _buffer_size, _parent_rank);
509
510 // scatter into target vector
511 vec_trg.format();
512 _parent_mirror.scatter_axpy(vec_trg, parent_buffer);
513 }
