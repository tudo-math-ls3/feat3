186 template<typename LVT2_>

188 {
189 if((void*)this == (void*)&other)
190 return;
191
192 this->_sibling_comm = other._sibling_comm;
193 this->_parent_rank = other._parent_rank;
194 this->_buffer_size = Index(0); // will be rebuild by compile function
195 this->_parent_mirror.clone(other._parent_mirror, mode);

200 for(std::size_t i(0); i < other._child_mirrors.size(); ++i)
201 {
202 this->_child_mirrors.at(i).clone(other._child_mirrors.at(i), mode);
203 }
204
205 // compile this muxer
206 compile(std::forward<LocalVector_>(vector));
207 }

303 void compile(const LocalVector_& vec_tmp_)
304 {
305 if(!is_child())
306 return; // nothing to
307
308 XASSERT(_sibling_comm != nullptr);
309 XASSERT(_parent_rank >= 0);
310 XASSERT(_parent_rank < _sibling_comm->size());
311
312 Index bufsize(0);
313
314 // Is this the parent?
315 if(_sibling_comm->rank() == _parent_rank)
316 {
317 // loop over all child buffers and compute the largest buffer size
318 for(const auto& cm : _child_mirrors)
319 {
320 bufsize = Math::max(bufsize, cm.buffer_size(vec_tmp_));
321 }
322 }
323
324 // broadcast buffer size to all siblings
325 _sibling_comm->bcast(&bufsize, std::size_t(1), _parent_rank);
326
327 // verify against length of parent buffer
328 XASSERT(bufsize >= _parent_mirror.buffer_size(vec_tmp_));
329
330 // store buffer size
331 this->_buffer_size = bufsize;
332 }

344 void join_send(const LocalVector_& vec_src) const
345 {
346 XASSERT(_sibling_comm != nullptr);
347 XASSERT(_sibling_comm->size() > 1);
348 XASSERT(_sibling_comm->rank() != _parent_rank); // parent must call join() instead
349
350 TimeStamp ts_execute;
351
352 // create parent buffer in device memory
353 BufferType parent_buffer(_buffer_size);
354
355 // gather source to parent buffer
356 _parent_mirror.gather(parent_buffer, vec_src);
357
358 // convert buffer to main memory
359 BufferMain parent_buffer_main;
360 parent_buffer_main.convert(parent_buffer);
361
362 Statistics::add_time_mpi_execute_collective(ts_execute.elapsed_now());
363
364 // gather to parent sibling
365 TimeStamp ts_collective;
366 DataType* dummy = nullptr;
367 _sibling_comm->gather(parent_buffer_main.elements(), _buffer_size, dummy, std::size_t(0), _parent_rank);
368 Statistics::add_time_mpi_wait_collective(ts_collective.elapsed_now());
369 }

380 void join(const LocalVector_& vec_src, LocalVector_& vec_trg) const
381 {
382 // if this muxer is not a child, then this operation is a simple copy
383 if((_sibling_comm == nullptr) || (_sibling_comm->size() <= 1))
384 {
385 vec_trg.copy(vec_src);
386 return;
387 }
388
389 XASSERT(_sibling_comm != nullptr);
390 XASSERT(_sibling_comm->size() > 0);
391 XASSERT(_sibling_comm->rank() == _parent_rank);
392
393 TimeStamp ts_execute;
394
395 const Index num_children = Index(_child_mirrors.size());
396
397 // create parent buffer in device memory
398 BufferType parent_buffer(_buffer_size);
399
400 // gather source to parent buffer
401 _parent_mirror.gather(parent_buffer, vec_src);
402
403 // convert buffer to main memory
404 BufferMain parent_buffer_main;
405 parent_buffer_main.convert(parent_buffer);
406
407 // allocate child buffers
408 BufferMain child_buffers_main(_buffer_size * num_children);
409
410 Statistics::add_time_mpi_execute_collective(ts_execute.elapsed_now());
411
412 // gather from siblings
413 TimeStamp ts_collective;
414 _sibling_comm->gather(parent_buffer_main.elements(), _buffer_size, child_buffers_main.elements(), _buffer_size, _parent_rank);
415 Statistics::add_time_mpi_wait_collective(ts_collective.elapsed_now());
416
417 ts_execute.stamp();
418
419 // download child buffers to device memory
420 BufferType child_buffers;
421 child_buffers.convert(child_buffers_main);
422
423 // format target vector
424 vec_trg.format();
425
426 // scatter child buffers into target vector
427 for(Index i(0); i < num_children; ++i)
428 {
429 // scatter buffers
430 _child_mirrors.at(i).scatter_axpy(vec_trg, child_buffers, DataType(1), i*_buffer_size);
431 }
432 Statistics::add_time_mpi_execute_collective(ts_execute.elapsed_now());
433 }

481 void split(LocalVector_& vec_trg, const LocalVector_& vec_src) const
482 {
483 // if this muxer is not a child, then this operation is a simple copy
484 if((_sibling_comm == nullptr) || (_sibling_comm->size() <= 1))
485 {
486 vec_trg.copy(vec_src);
487 return;
488 }
489
490 XASSERT(_sibling_comm != nullptr);
491 XASSERT(_sibling_comm->size() > 0);
492 XASSERT(_sibling_comm->rank() == _parent_rank);
493
494 TimeStamp ts_execute;
495
496 const Index num_children = Index(_child_mirrors.size());
497
498 // allocate child buffers
499 BufferType child_buffers(_buffer_size * num_children);
500
501 // gather child buffers from target vector
502 for(Index i(0); i < num_children; ++i)
503 {
504 _child_mirrors.at(i).gather(child_buffers, vec_src, i*_buffer_size);
505 }
506
507 // convert child buffers to main memory
508 BufferMain child_buffers_main;
509 child_buffers_main.convert(child_buffers);
510
511 // create parent buffer
512 BufferMain parent_buffer_main(_buffer_size);
513
514 Statistics::add_time_mpi_execute_collective(ts_execute.elapsed_now());
515
516 // scatter to siblings
517 TimeStamp ts_collective;
518 _sibling_comm->scatter(child_buffers_main.elements(), _buffer_size, parent_buffer_main.elements(), _buffer_size, _parent_rank);
519 Statistics::add_time_mpi_wait_collective(ts_collective.elapsed_now());
520
521 ts_execute.stamp();
522
523 // convert to device memory
524 BufferType parent_buffer;
525 parent_buffer.convert(parent_buffer_main);
526
527 // scatter into target vector
528 vec_trg.format();
529 _parent_mirror.scatter_axpy(vec_trg, parent_buffer);
530
531 Statistics::add_time_mpi_execute_collective(ts_execute.elapsed_now());
532 }
