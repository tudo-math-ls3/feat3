206 MeshNodeType& create_base_mesh_2d(Index num_x, Index num_y, CoordType x_min, CoordType x_max, CoordType y_min, CoordType y_max)
207 {
208 XASSERTM(shape_dim == 2, "invalid dimension: this function can only be called for 2D domains!");
209 XASSERTM(_create_stage == 0, "base mesh already created!");
210
211 XASSERT(num_x > Index(0));
212 XASSERT(num_y > Index(0));
213 XASSERT(x_min < x_max);
214 XASSERT(y_min < y_max);
215
216 _watch_create_base_mesh.start();
217
218 _base_slices[0] = num_x;
219 _base_slices[1] = num_y;
220 _bbox_min[0] = x_min;
221 _bbox_max[0] = x_max;
222 _bbox_min[1] = y_min;
223 _bbox_max[1] = y_max;
224
225 // create a structured mesh from factory
226 _base_mesh_node = MeshNodeType::make_unique(
227 Geometry::StructUnitCubeFactory<MeshType>::make_unique_from(num_x, num_y), &this->_atlas);
228
229 // get the node's vertex set
230 auto& vtx = _base_mesh_node->get_mesh()->get_vertex_set();
231 Index n = vtx.get_num_vertices();
232 for(Index i(0); i < n; ++i)
233 {
234 vtx[i][0] = x_min + (x_max - x_min) * vtx[i][0];
235 vtx[i][1] = y_min + (y_max - y_min) * vtx[i][1];
236 }
237
238 _watch_create_base_mesh.stop();
239
240 // update creation stage
241 this->_create_stage = 1;
242
243 // return the base-mesh node
244 return *_base_mesh_node.get();
245 }

582 void create_hierarchy()
583 {
584 XASSERTM(this->_create_stage >= 2, "invalid creation stage; create the base-mesh and voxel map first");
585 XASSERTM(this->_create_stage <= 2, "invalid creation stage: domain hierarchy already created");
586
587 this->_watch_create_hierarchy.start();
588
589 // create the domain control
590#ifdef FEAT_HAVE_MPI
591 if(this->_comm.size() == 1)
592 {
593 // We've got just one process, so it's a simple choice:
594 this->_create_single_process(std::move(this->_base_mesh_node));
595 }
596 else if(this->_support_multi_layered && (this->_desired_levels.size() > std::size_t(2)))
597 {
598 // The application supports multi-layered domain controls and
599 // the user wants that, so create a multi-layered one:
600 this->_create_multi_layered(std::move(this->_base_mesh_node));
601 }
602 else
603 {
604 // Create a single-layered domain control:
605 this->_create_single_layered(std::move(this->_base_mesh_node));
606 }
607#else // not FEAT_HAVE_MPI
608 {
609 // In the non-MPI case, we always have only one process:
610 this->_create_single_process(std::move(this->_base_mesh_node));
611 }
612#endif // FEAT_HAVE_MPI
613
614 // compile virtual levels
615 this->compile_virtual_levels();
616
617 this->_watch_create_hierarchy.stop();
618
619 // collect statistics
620 FEAT::Statistics::toe_partition = this->_watch_create_base_mesh.elapsed() +
621 this->_watch_create_voxel_map.elapsed() + this->_watch_create_hierarchy.elapsed();
622
623 // update creation stage
624 this->_create_stage = 3;
625 this->_was_created = true;
626
627 // delete the voxel map unless we're meant to keep it
628 if(!this->_keep_voxel_map)
629 this->_voxel_map.reset();
630 }

728 void _precreate_voxel_map(Real resolution)
729 {
730 // set voxel map bounding box based on domain's bounding box
731 if constexpr(shape_dim == 2)
732 this->_voxel_map->set_bounding_box_2d(this->_bbox_min[0], this->_bbox_max[0], this->_bbox_min[1], this->_bbox_max[1]);
733 else
734 this->_voxel_map->set_bounding_box_3d(this->_bbox_min[0], this->_bbox_max[0],
735 this->_bbox_min[1], this->_bbox_max[1], this->_bbox_min[2], this->_bbox_max[2]);
736
737 // is the resolution given?
738 if(resolution > Real(0))
739 {
740 this->_voxel_map->set_resolution(resolution);
741 return;
742 }
743
744 // compute the mesh width on the base mesh level
745 Real base_res = (this->_bbox_max[0] - this->_bbox_min[0]) / Real(this->_base_slices[0]);
746 for(int i(1); i < shape_dim; ++i)
747 base_res = Math::min(base_res, this->_bbox_max[i] - this->_bbox_min[i]) / Real(this->_base_slices[std::size_t(i)]);
748
749 // set resolution for voxel map to base resolution divided by 2^{desired-level-max+1)
750 this->_voxel_map->set_resolution(base_res / Real(2ull << this->get_desired_level_max()));
751 }

767 std::vector<Index> _gather_masked_elements(const MeshType& mesh) const
768 {
769 const Index num_elems = mesh.get_num_elements();
770
771 // reserve the vector
772 std::vector<Index> masked_elems;
773 masked_elems.reserve(num_elems);
774
775 // get the vertex set and the vertices-at-element index set
776 const auto& vtx = mesh.get_vertex_set();
777 const auto& verts_at_elem = mesh.template get_index_set<shape_dim, 0>();
778
779 // element bounding box coordinates
780 Tiny::Vector<CoordType, shape_dim> elbox_min, elbox_max;
781
782 // loop over all elements
783 for(Index ielem(0); ielem < num_elems; ++ielem)
784 {
785 // compute bounding box of element
786 // note: actually, we should always have
787 // elbox_min = vtx[verts_at_elem(ielem, 0)]
788 // elbox_max = vtx[verts_at_elem(ielem, 1 << shape_dim - 1)]
789 // because of the element orientation of the base mesh, but we'll compute the bounding
790 // box for each element manually anyways, just to be sure in case that a derived class
791 // might give us a mesh object that is not directly refined from the base mesh using
792 // the standard 2-level refinement...
793 elbox_min = elbox_max = vtx[verts_at_elem(ielem, 0)];
794 for(int j(1); j < verts_at_elem.num_indices; ++j)
795 {
796 const auto& v = vtx[verts_at_elem(ielem, j)];
797 for(int k(0); k < shape_dim; ++k)
798 Math::minimax(v[k], elbox_min[k], elbox_max[k]);
799 }
800
801 // check the bounding box against the voxel map
802 if(this->_voxel_map->check_box(elbox_min, elbox_max))
803 masked_elems.push_back(ielem);
804 }
805
806 // return the list of all elements that we found
807 return masked_elems;
808 }

927 virtual void _deslag_patch_halos(
928 MeshNodeType& patch_mesh_node,
929 const MeshNodeType& base_mesh_node,
930 const Ancestor& ancestor,
931 bool is_child)
932 {
933 // get the map of the base-mesh halos
934 const std::map<int, std::unique_ptr<MeshPartType>>& base_halo_map = base_mesh_node.get_halo_map();
935
936 // if the base mesh has no halos, then we can jump out of here
937 if(base_halo_map.empty())
938 return;
939
940 // get number of halos
941 const std::size_t num_halos = base_halo_map.size();
942
943 // create a halo splitter
944 Geometry::PatchHaloSplitter<MeshType> halo_splitter(*base_mesh_node.get_mesh(), *base_mesh_node.get_patch(-1));
945
946 // add each base-mesh halo to our halo splitter and store the resulting split data size
947 std::vector<int> halo_ranks;
948 std::vector<std::size_t> halo_send_sizes;
949 for(auto it = base_halo_map.begin(); it != base_halo_map.end(); ++it)
950 {
951 // store halo rank
952 halo_ranks.push_back(it->first);
953
954 // add halo and compute send buffer size
955 halo_send_sizes.push_back(halo_splitter.add_halo(it->first, *it->second));
956 }
957
958 // This vector will receive the split halo data from all our potential neighbor processes
959 std::vector<std::size_t> halo_recv_sizes(num_halos);
960 std::vector<std::vector<Index>> halo_send_data(num_halos), halo_recv_data(num_halos);
961 Dist::RequestVector halo_recv_reqs(num_halos), halo_send_reqs(num_halos);
962
963 // get the layer index of our comm layer; the ancestor layer unless we got the child ancestry during
964 // multi-layered partitioning; this may be equal to -1 in either case if this process does not participate
965 // in that corresponding layer
966 const int layer_idx = is_child ? ancestor.layer_p : ancestor.layer;
967
968 // split and serialize halos and store sizes
969 for(std::size_t i(0); i < num_halos; ++i)
970 {
971 // serialize the split halo into the send data buffer
972 if(halo_send_sizes.at(i) > std::size_t(0))
973 halo_send_data.at(i) = halo_splitter.serialize_split_halo(halo_ranks[i], -1);
974 }
975
976 // exchange halo send data sizes over the corresponding layer communicator
977 if(layer_idx >= 0)
978 {
979 // get the corresponding layer communicator
980 const Dist::Comm& layer_comm = this->_layers.at(std::size_t(layer_idx))->comm();
981
982 // exchange halos sizes
983 for(std::size_t i(0); i < num_halos; ++i)
984 {
985 // post receive requests
986 halo_recv_reqs[i] = layer_comm.irecv(&halo_recv_sizes[i], std::size_t(1), halo_ranks[i]);
987
988 // post send requests
989 halo_send_reqs[i] = layer_comm.isend(&halo_send_sizes[i], std::size_t(1), halo_ranks[i]);
990 }
991
992 // wait for sends and receives to finish
993 halo_recv_reqs.wait_all();
994 halo_send_reqs.wait_all();
995
996 // exchange halo data
997 for(std::size_t i(0); i < num_halos; ++i)
998 {
999 // do we receive any data from our neighbor?
1000 if(halo_recv_sizes[i] > Index(0))
1001 {
1002 // resize buffer and post receive
1003 halo_recv_data.at(i).resize(halo_recv_sizes[i]);
1004 halo_recv_reqs[i] = layer_comm.irecv(halo_recv_data.at(i).data(), halo_recv_sizes.at(i), halo_ranks[i]);
1005 }
1006
1007 // do we send any data to our neighbor?
1008 if(halo_send_sizes.at(i) > Index(0))
1009 {
1010 // post send of actual halo buffer
1011 halo_send_reqs[i] = layer_comm.isend(halo_send_data.at(i).data(), halo_send_sizes.at(i), halo_ranks[i]);
1012 }
1013 }
1014
1015 // wait for sends and receives to finish
1016 halo_recv_reqs.wait_all();
1017 halo_send_reqs.wait_all();
1018 }
1019
1020 // broadcast halo receive data over progeny comm in case of multi-layered partitioning
1021 if(is_child)
1022 {
1023 // broadcast sizes
1024 ancestor.progeny_comm.bcast(halo_recv_sizes.data(), num_halos, 0);
1025
1026 // allocate buffers
1027 if(ancestor.progeny_comm.rank() != 0)
1028 {
1029 for(std::size_t i(0); i < num_halos; ++i)
1030 halo_recv_data.at(i).resize(halo_recv_sizes[i]);
1031 }
1032
1033 // broadcast all halos
1034 for(std::size_t i(0); i < num_halos; ++i)
1035 {
1036 if(halo_recv_sizes[i] > std::size_t(0))
1037 ancestor.progeny_comm.bcast(halo_recv_data[i].data(), halo_recv_sizes[i], 0);
1038 }
1039 }
1040
1041 /*{
1042 String s;
1043 for(std::size_t i(0); i < num_halos; ++i)
1044 {
1045 s += stringify(halo_ranks[i]);
1046 s += " >";
1047 for(Index j(halo_recv_data[i]); j < halo_recv_data[i+1]; ++j)
1048 (s += " ") += stringify(halo_recv_data[j]);
1049 s += "\n";
1050 }
1051 this->_comm.allprint(s);
1052 }*/
1053
1054 // create a vector of split halo sizes
1055 std::vector<std::array<Index, shape_dim+1>> halo_send_intsec_sizes(num_halos), halo_recv_intsec_sizes(num_halos);
1056
1057 // process all halos
1058 for(std::size_t i(0); i < num_halos; ++i)
1059 {
1060 // did we receive any data from our neighbor?
1061 if(halo_recv_sizes.at(i) == Index(0))
1062 continue;
1063
1064 // intersect with our other halo
1065 if(!halo_splitter.intersect_split_halo(halo_ranks[i], halo_recv_data.at(i), 0u))
1066 continue; // no intersection between halos
1067
1068 // create the new halo
1069 std::unique_ptr<MeshPartType> split_halo = halo_splitter.make_unique();
1070
1071 // store the entity counts of the halo in our split sizes vector
1072 for(int j(0); j <= shape_dim; ++j)
1073 halo_send_intsec_sizes[i][Index(j)] = split_halo->get_num_entities(j);
1074
1075 // create new halo mesh-part
1076 patch_mesh_node.add_halo(halo_ranks[i], std::move(split_halo));
1077 }
1078
1079 // exchange halo intersected halo sizes over progeny comm; we do this just to ensure consistency
1080 if((ancestor.layer_p >= 0) || (!is_child && (ancestor.layer >= 0)))
1081 {
1082 // get the corresponding layer communicator
1083 const LayerType& layer = *this->_layers.at(std::size_t(is_child ? ancestor.layer_p : ancestor.layer));
1084
1085 // exchange intersected halo sizes
1086 for(std::size_t i(0); i < num_halos; ++i)
1087 {
1088 halo_recv_reqs[i] = layer.comm().irecv(halo_recv_intsec_sizes.at(i).data(), std::size_t(shape_dim+1), halo_ranks[i]);
1089 halo_send_reqs[i] = layer.comm().isend(halo_send_intsec_sizes.at(i).data(), std::size_t(shape_dim+1), halo_ranks[i]);
1090 }
1091
1092 // wait for sends and receives to finish
1093 halo_recv_reqs.wait_all();
1094 halo_send_reqs.wait_all();
1095
1096 // ensure that the halo sizes are consistent between neighboring processes
1097 for(std::size_t i(0); i < num_halos; ++i)
1098 {
1099 for(int j(0); j <= shape_dim; ++j)
1100 {
1101 if(halo_recv_intsec_sizes[i][Index(j)] != halo_send_intsec_sizes[i][Index(j)])

1125 Adjacency::Coloring _compute_base_mesh_coloring(const MeshNodeType& slag_node) const
1126 {
1127 // get element target set of slag patch
1128 const Geometry::MeshPart<MeshType>* slag_part = slag_node.get_patch(-1);
1129 XASSERT(slag_part != nullptr);
1130 const Geometry::TargetSet& slag_target = slag_part->template get_target_set<MeshType::shape_dim>();
1131
1132 // number of colors = 2^shape_dim
1133 const Index num_colors = Index(1) << MeshType::shape_dim;
1134 const Index num_elems = slag_target.get_num_entities();
1135
1136 // allocate coloring
1137 Adjacency::Coloring coloring(num_elems, num_colors);
1138 Index* colors = coloring.get_coloring();
1139
1140 // unrefined base mesh: structured numbering
1141 if constexpr(MeshType::shape_dim == 1)
1142 {
1143 for(Index i(0); i < num_elems; ++i)
1144 colors[i] = slag_target[i] & 1;
1145 }
1146 else if constexpr(MeshType::shape_dim == 2)
1147 {
1148 for(Index i(0); i < num_elems; ++i)
1149 {
1150 Index iel = slag_target[i];
1151 Index ix = iel % this->_base_slices[0];
1152 Index iy = iel / this->_base_slices[0];
1153 colors[i] = (ix & 1) | ((iy & 1) << 1);
1154 }
1155 }
1156 else if constexpr(MeshType::shape_dim == 3)
1157 {
1158 for(Index i(0); i < num_elems; ++i)
1159 {
1160 Index iel = slag_target[i];
1161 Index ix = iel % this->_base_slices[0];
1162 Index iy = (iel / this->_base_slices[0]) % this->_base_slices[1];
1163 Index iz = iel / (this->_base_slices[0] * this->_base_slices[1]);
1164 colors[i] = (ix & 1) | ((iy & 1) << 1) | ((iz & 1) << (1 << 1));
1165 }
1166 }
1167
1168 return coloring;
1169 }

1248 Adjacency::Coloring _compute_base_mesh_layering(const MeshNodeType& slag_node) const
1249 {
1250 // get element target set of slag patch
1251 const Geometry::MeshPart<MeshType>* slag_part = slag_node.get_patch(-1);
1252 XASSERT(slag_part != nullptr);
1253 const Geometry::TargetSet& slag_target = slag_part->template get_target_set<MeshType::shape_dim>();
1254
1255 // maximum number of layers = number of slices in highest dimensions; this may be less than
1256 // base_slices[shape_dim], since entire layers might have been removed during the deslagging process
1257 const Index max_layers = this->_base_slices[shape_dim-1];
1258 const Index num_elems = slag_target.get_num_entities();
1259
1260 // compute denominator for layer index computation
1261 Index denom = 1u;
1262 for(int j(0); j+1 < shape_dim; ++j)
1263 denom *= this->_base_slices[Index(j)];
1264
1265 // compute number of elements for each potential layer
1266 std::vector<Index> aux(max_layers, 0u);
1267 for(Index i(0); i < num_elems; ++i)
1268 ++aux[slag_target[i] / denom];
1269
1270 // count number of actually used layers
1271 Index num_layers = 0u;
1272 for(Index i(0); i < max_layers; ++i)
1273 {
1274 Index k = num_layers;
1275 if(aux[i] > 0u)
1276 ++num_layers;
1277 aux[i] = k;
1278 }
1279
1280 // compute layering
1281 Adjacency::Coloring layering(num_elems, num_layers);
1282 Index* layers = layering.get_coloring();
1283 for(Index i(0); i < num_elems; ++i)
1284 layers[i] = aux[slag_target[i] / denom];
1285
1286 return layering;
1287 }

1298 Adjacency::Coloring _compute_refined_mesh_layering(const MeshNodeType& slag_node, const Adjacency::Coloring& coarse_layering) const
1299 {
1300 // get element target set of slag patch
1301 const Geometry::MeshPart<MeshType>* slag_part = slag_node.get_patch(-1);
1302 XASSERT(slag_part != nullptr);
1303 const Geometry::TargetSet& slag_target = slag_part->template get_target_set<MeshType::shape_dim>();
1304
1305 // maximum number of layers = twice the number of layers in coarse mesh; this may actually be less,
1306 // since entire layers might have been removed during the deslagging process
1307 const Index max_layers = 2 * coarse_layering.get_num_colors();
1308 const Index num_elems = slag_target.get_num_entities();
1309
1310 // compute number of elements for each potential layer
1311 std::vector<Index> aux(max_layers, 0u);
1312 for(Index i(0); i < num_elems; ++i)
1313 {
1314 Index qux = slag_target[i] >> (shape_dim - 1);
1315 ++aux[(coarse_layering[qux >> 1] << 1) | (qux & 1)]; // all hail to bit-shift magic!
1316 }
1317
1318 // count number of actually used layers
1319 Index num_layers = 0u;
1320 for(Index i(0); i < max_layers; ++i)
1321 {
1322 Index k = num_layers;
1323 if(aux[i] > 0u)
1324 ++num_layers;
1325 aux[i] = k;
1326 }
1327
1328 // compute layering
1329 Adjacency::Coloring layering(num_elems, num_layers);
1330 Index* layers = layering.get_coloring();
1331 for(Index i(0); i < num_elems; ++i)
1332 {
1333 Index qux = slag_target[i] >> (shape_dim - 1);
1334 layers[i] = aux[(coarse_layering[qux >> 1] << 1) | (qux & 1)];
1335 }
1336
1337 return layering;
1338 }

1402 virtual void _create_single_process(std::unique_ptr<MeshNodeType> base_mesh_node)
1403 {
1404 // create and push single layer
1405 this->push_layer(std::make_shared<LayerType>(this->_comm.comm_dup(), 0));

1408 this->_create_ancestry_single();

1411 // use the whole base mesh here

1413

1416

1418 base_mesh_node = this->_deslag_mesh_node(*base_slag_node);
1419

1421 Adjacency::Coloring base_layering = this->_compute_base_mesh_layering(*base_slag_node);
1422
1423 // refine and deslag up to desired minimum level
1424 int lvl = 0;
1425 for(; lvl < ancestor.desired_level_min; ++lvl)
1426 {
1427 base_slag_node = base_mesh_node->refine_unique(this->_adapt_mode);
1428 base_mesh_node = this->_deslag_mesh_node(*base_slag_node);

1434 if(lvl == 0)
1435 base_coloring = this->_compute_base_mesh_coloring(*base_slag_node);
1436 else
1437 base_coloring = this->_compute_refined_mesh_coloring(*base_slag_node);
1438

1440 if(lvl < ancestor.desired_level_max)
1441 this->_chosen_levels.push_front(std::make_pair(lvl, 0));
1442

1599 virtual void _create_multi_layered(std::unique_ptr<MeshNodeType> base_mesh_node)
1600 {
