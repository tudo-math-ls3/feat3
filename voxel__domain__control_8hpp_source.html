388 MeshNodeType& create_base_mesh_2d(Index num_x, Index num_y, CoordType x_min, CoordType x_max, CoordType y_min, CoordType y_max)
389 {
390 XASSERTM(shape_dim == 2, "invalid dimension: this function can only be called for 2D domains!");
391 XASSERTM(_create_stage == 0, "base mesh already created!");
392
393 XASSERT(num_x > Index(0));
394 XASSERT(num_y > Index(0));
395 XASSERT(x_min < x_max);
396 XASSERT(y_min < y_max);
397
398 _watch_create_base_mesh.start();
399
400 _base_slices[0] = num_x;
401 _base_slices[1] = num_y;
402 _bbox_min[0] = x_min;
403 _bbox_max[0] = x_max;
404 _bbox_min[1] = y_min;
405 _bbox_max[1] = y_max;
406
407 // create a structured mesh from factory
408 _base_mesh_node = MeshNodeType::make_unique(
409 Geometry::StructUnitCubeFactory<MeshType>::make_unique_from(num_x, num_y), &this->_atlas);
410
411 // get the node's vertex set
412 auto& vtx = _base_mesh_node->get_mesh()->get_vertex_set();
413 Index n = vtx.get_num_vertices();
414 for(Index i(0); i < n; ++i)
415 {
416 vtx[i][0] = x_min + (x_max - x_min) * vtx[i][0];
417 vtx[i][1] = y_min + (y_max - y_min) * vtx[i][1];
418 }
419
420 _watch_create_base_mesh.stop();
421
422 // update creation stage
423 this->_create_stage = 1;
424
425 // return the base-mesh node
426 return *_base_mesh_node.get();
427 }

657 void create_hierarchy()
658 {
659 XASSERTM(this->_create_stage >= 2, "invalid creation stage; create the base-mesh and slag mask first");
660 XASSERTM(this->_create_stage <= 2, "invalid creation stage: domain hierarchy already created");
661
662 this->_watch_create_hierarchy.start();
663
664 // create the domain control
665#ifdef FEAT_HAVE_MPI
666 if(this->_comm.size() == 1)
667 {
668 // We've got just one process, so it's a simple choice:
669 this->_create_single_process(std::move(this->_base_mesh_node));
670 }
671 else if(this->_support_multi_layered && (this->_desired_levels.size() > std::size_t(2)))
672 {
673 // The application supports multi-layered domain controls and
674 // the user wants that, so create a multi-layered one:
675 this->_create_multi_layered(std::move(this->_base_mesh_node));
676 }
677 else
678 {
679 // Create a single-layered domain control:
680 this->_create_single_layered(std::move(this->_base_mesh_node));
681 }
682#else // not FEAT_HAVE_MPI
683 {
684 // In the non-MPI case, we always have only one process:
685 this->_create_single_process(std::move(this->_base_mesh_node));
686 }
687#endif // FEAT_HAVE_MPI
688
689 // compile virtual levels
690 this->compile_virtual_levels();
691
692 this->_watch_create_hierarchy.stop();
693
694 // collect statistics
695 FEAT::Statistics::toe_partition = this->_watch_create_base_mesh.elapsed() +
696 this->_watch_create_slag_mask.elapsed() + this->_watch_create_hierarchy.elapsed();
697
698 // update creation stage
699 this->_create_stage = 3;
700 this->_was_created = true;
701 }

805 void _compress_slag_mask_row(const std::vector<int>& mask, const Index row)
806 {
807 ASSERTM(row < this->_slag_mask_row_count, "invalid drop mask row");
808 const Index n = Index(mask.size());
809 const Index off = row * this->_slag_mask_row_length;
810 for(Index i(0); i < n; ++i)
811 {
812 _slag_mask[off + (i >> 6)] |= std::uint64_t(mask[i] != 0) << (i & 0x3F);
813 }
814 }

825 Index _map_slag_idx(CoordType xyz, int dim) const
826 {
827 CoordType t = (xyz - _bbox_min[dim]) / (_bbox_max[dim] - _bbox_min[dim]);
828 int idx = int(std::nearbyint(CoordType(_mask_slices[std::size_t(dim)]) * t));
829 ASSERTM(idx >= 0, "tested vertex coordinate is not inside coarse bounding box");
830 ASSERTM(idx <= int(this->_mask_slices[Index(dim)]), "tested vertex coordinate is not inside coarse bounding box");
831 return Index(idx);
832 }

844 bool _check_slag_mask(const Tiny::Vector<CoordType, 3>& vtx) const
845 {

901 void _compute_slag_mask_rows(VoxelSlagMasker<CoordType, shape_dim>& masker,
902 const std::array<Index,3>& DOXY(slices), Index beg, Index end)
903 {
904 std::vector<int> row_mask(this->_mask_slices[0] + 1u, 0);
905 const CoordType sy = (_bbox_max[1] - _bbox_min[1]) / CoordType(this->_mask_slices[1]);
906 const CoordType sz = (_bbox_max[2] - _bbox_min[2]) / CoordType(this->_mask_slices[2]);
907 Tiny::Vector<CoordType, 3> coords;
908
909 for(Index i(beg); i < end; ++i)
910 {
911 // row = iz * (this->_mask_slices[1] + 1u) + iy
912 coords[1] = _bbox_min[1] + CoordType(i % (this->_mask_slices[1] + 1u)) * sy;
913 coords[2] = _bbox_min[2] + CoordType(i / (this->_mask_slices[1] + 1u)) * sz;
914 masker.mask_row(row_mask, _bbox_min[0], _bbox_max[0], coords);
915 this->_compress_slag_mask_row(row_mask, i);
916 }
917 }

980 WeightType _gather_slag_mask_weight(const std::array<Index, 3>& box_min, const std::array<Index, 3>& box_max) const
981 {
982 Index count = 0u;
983 for(Index zidx(box_min[2]); zidx <= box_max[2]; ++zidx)
984 {
985 for(Index yidx(box_min[1]); yidx <= box_max[1]; ++yidx)
986 {
987 Index row = zidx * (this->_mask_slices[1] + 1u) + yidx;
988 for(Index xidx(box_min[0]); xidx <= box_max[0]; ++xidx)
989 count += (this->_slag_mask[row * this->_slag_mask_row_length + (xidx >> 6)] >> (xidx & 0x3F)) & 0x1;
990 }
991 }
992 return WeightType(count) / WeightType((box_max[2] - box_min[2] + 1u) * (box_max[1] - box_min[1] + 1u) * (box_max[0] - box_min[0] + 1u));

1043 void _compute_slag_mask(VoxelSlagMasker<CoordType, shape_dim>& masker)
1044 {
1045 const int level_max = this->_desired_levels.front().first;
1046
1047 // compute number of fine mesh bounding box slices in each dimension and total number of vertices
1048 for(Index i(0); i < Index(shape_dim); ++i)
1049 this->_mask_slices[i] = this->_base_slices[i] << level_max;
1050
1051 // compute number of fine mesh bounding box vertices
1052 this->_slag_mask_row_length = this->_mask_slices[0] + 1u;
1053 this->_slag_mask_row_count = 1u;
1054 for(Index i(1); i < Index(shape_dim); ++i)
1055 this->_slag_mask_row_count *= (this->_mask_slices[i] + 1u);
1056
1057 // translate row length from bits to 64-bit integers, i.e. divide by 64 and round up if necessary
1058 this->_slag_mask_row_length = (this->_slag_mask_row_length / 64u) + (this->_slag_mask_row_length % 64u != 0);
1059
1060 // single process or MPI parallel?
1061 if(this->_comm.size() <= 1)
1062 {
1063 // allocate slag mask and format to zero
1064 this->_slag_mask.resize(this->_slag_mask_row_length * this->_slag_mask_row_count, 0u);
1065
1066 // compute all slag mask rows
1067 this->_compute_slag_mask_rows(masker, this->_mask_slices, 0u, this->_slag_mask_row_count);
1068 }
1069 else if(Index(this->_comm.size()) <= this->_slag_mask_row_count)
1070 {
1071 // there are at least as many rows in the slag mask as there are MPI processes; this is the usual case
1072 // we can use the entire communicator (usually MPI_COMM_WORLD) for the slag mask computation
1073 Index comm_rank = Index(this->_comm.rank());
1074 Index comm_size = Index(this->_comm.size());
1075
1076 // compute row count for a single process and round up; this may be bigger than _slag_mask_row_count,
1077 // so the last process may have less rows to chew through than all other processes
1078 Index row_count = (this->_slag_mask_row_count + comm_size - 1u) / comm_size;
1079 std::size_t block_len = this->_slag_mask_row_length * row_count;
1080
1081 // allocate slag mask and format to zero
1082 this->_slag_mask.resize(block_len * Index(comm_size), 0u);
1083
1084 // compute first and last+1 row of this process; make sure the last process doesn't go beyond the total count
1085 Index row_beg = comm_rank * row_count;
1086 Index row_end = Math::min((comm_rank + 1u) * row_count, this->_slag_mask_row_count);
1087
1088 // debug output
1089 //this->_comm.allprint(String(">>>") + stringify(row_beg).pad_front(6) + ":" + stringify(row_count) + ">" + stringify(row_end).pad_front(6));
1090
1091 // compute rows for this process
1092 this->_compute_slag_mask_rows(masker, this->_mask_slices, row_beg, row_end);
1093
1094 // perform an in-place allgather to synchronize the slag mask over all processes
1095 this->_comm.allgather(this->_slag_mask.data(), block_len, this->_slag_mask.data(), block_len);
1096 }
1097 else if(this->_slag_mask_row_count < Index(this->_comm.size()))
1098 {
1099 // more processes than slag rows; this is a borderline scenario, but we have to support it anyways
1100 // create a sub-communicator and then compute mask on that sub-communicator
1101 Dist::Comm sub_comm = this->_comm.comm_create_range_incl(int(this->_slag_mask_row_count));
1102
1103 // does this process participate in the sub-communicator?
1104 if(!sub_comm.is_null())
1105 {
1106 // get sub-communicator rank and size
1107 Index comm_rank = Index(sub_comm.rank());
1108 Index comm_size = Index(sub_comm.size());
1109
1110 // compute row count for a single process and round up; this may be bigger than _slag_mask_row_count,
1111 // so the last process may have less rows to chew through than all other processes
1112 Index row_count = (this->_slag_mask_row_count + comm_size - 1u) / comm_size;
1113 std::size_t block_len = this->_slag_mask_row_length * row_count;
1114
1115 // allocate slag mask and format to zero
1116 this->_slag_mask.resize(block_len * Index(comm_size), 0u);
1117
1118 // compute first and last+1 row of this process; make sure the last process doesn't go beyond the total count
1119 Index row_beg = comm_rank * row_count;
1120 Index row_end = Math::min((comm_rank + 1u) * row_count, this->_slag_mask_row_count);
1121
1122 // debug output
1123 //this->_comm.allprint(String(">>>") + stringify(row_beg).pad_front(6) + ":" + stringify(row_count) + ">" + stringify(row_end).pad_front(6));
1124
1125 // compute rows for this process
1126 this->_compute_slag_mask_rows(masker, this->_mask_slices, row_beg, row_end);
1127
1128 // perform an in-place gather to synchronize the slag mask on rank 0
1129 sub_comm.gather(this->_slag_mask.data(), block_len, this->_slag_mask.data(), block_len, 0);
1130 }
1131 else
1132 {
1133 // just allocate the slag mask vector for the following broadcast
1134 this->_slag_mask.resize(this->_slag_mask_row_length * this->_slag_mask_row_count, 0u);
1135 }
1136
1137 // broadcast from rank 0 to all processes our entire domain communicator
1138 this->_comm.bcast(this->_slag_mask.data(), this->_slag_mask_row_length * this->_slag_mask_row_count, 0);
1139 }
1140
1141 // that's it
1142 }

1158 std::vector<Index> _gather_masked_elements(const MeshType& mesh) const
1159 {
1160 const Index num_elems = mesh.get_num_elements();
1161
1162 // reserve the vector
1163 std::vector<Index> masked_elems;
1164 masked_elems.reserve(num_elems);
1165
1166 // get the vertex set and the vertices-at-element index set
1167 const auto& vtx = mesh.get_vertex_set();
1168 const auto& verts_at_elem = mesh.template get_index_set<shape_dim, 0>();
1169
1170 // element bounding box coordinates
1171 Tiny::Vector<CoordType, shape_dim> elbox_min, elbox_max;
1172
1173 // mapped element bounding box slag mask indices
1174 std::array<Index, shape_dim> slag_idx_min, slag_idx_max;
1175
1176 // loop over all elements
1177 for(Index ielem(0); ielem < num_elems; ++ielem)
1178 {
1179 // compute bounding box of element
1180 // note: actually, we should always have
1181 // elbox_min = vtx[verts_at_elem(ielem, 0)]
1182 // elbox_max = vtx[verts_at_elem(ielem, 1 << shape_dim - 1)]
1183 // because of the element orientation of the base mesh, but we'll compute the bounding
1184 // box for each element manually anyways, just to be sure in case that a derived class
1185 // might give us a mesh object that is not directly refined from the base mesh using
1186 // the standard 2-level refinement...
1187 elbox_min = elbox_max = vtx[verts_at_elem(ielem, 0)];
1188 for(int j(1); j < verts_at_elem.num_indices; ++j)
1189 {
1190 const auto& v = vtx[verts_at_elem(ielem, j)];
1191 for(int k(0); k < shape_dim; ++k)
1192 Math::minimax(v[k], elbox_min[k], elbox_max[k]);
1193 }
1194
1195 // map slag mask indices of bounding box in drop row indices
1196 for(int k(0); k < shape_dim; ++k)
1197 {
1198 slag_idx_min[Index(k)] = this->_map_slag_idx(elbox_min[k], k);
1199 slag_idx_max[Index(k)] = this->_map_slag_idx(elbox_max[k], k);
1200 }
1201
1202 // gather element weight based on the bounding box; any non-negative weight means that
1203 // there was at least one masked domain point in the slag mask covered by the element
1204 if(this->_gather_slag_mask_weight(slag_idx_min, slag_idx_max) > 1E-12)
1205 masked_elems.push_back(ielem);
1206 }
1207
1208 // return the list of all elements that we found
1209 return masked_elems;
1210 }

1347

1357

1359 std::vector<int> halo_ranks;

1361 for(auto it = base_halo_map.begin(); it != base_halo_map.end(); ++it)

1368 }
1369
1370 // This vector will receive the split halo data from all our potential neighbor processes
1371 std::vector<std::size_t> halo_recv_sizes(num_halos);
1372 std::vector<std::vector<Index>> halo_send_data(num_halos), halo_recv_data(num_halos);
1373 Dist::RequestVector halo_recv_reqs(num_halos), halo_send_reqs(num_halos);
1374
1375 // get the layer index of our comm layer; the ancestor layer unless we got the child ancestry during
1376 // multi-layered partitioning; this may be equal to -1 in either case if this process does not participate
1377 // in that corresponding layer
1378 const int layer_idx = is_child ? ancestor.layer_p : ancestor.layer;

1395 for(std::size_t i(0); i < num_halos; ++i)
1396 {
1397 // post receive requests
1398 halo_recv_reqs[i] = layer_comm.irecv(&halo_recv_sizes[i], std::size_t(1), halo_ranks[i]);
1399
1400 // post send requests
1401 halo_send_reqs[i] = layer_comm.isend(&halo_send_sizes[i], std::size_t(1), halo_ranks[i]);
1402 }
1403

1405 halo_recv_reqs.wait_all();
1406 halo_send_reqs.wait_all();
1407

1409 for(std::size_t i(0); i < num_halos; ++i)
1410 {

1414 // resize buffer and post receive
1415 halo_recv_data.at(i).resize(halo_recv_sizes[i]);
1416 halo_recv_reqs[i] = layer_comm.irecv(halo_recv_data.at(i).data(), halo_recv_sizes.at(i), halo_ranks[i]);
1417 }
1418
1419 // do we send any data to our neighbor?
1420 if(halo_send_sizes.at(i) > Index(0))
1421 {
1422 // post send of actual halo buffer
1423 halo_send_reqs[i] = layer_comm.isend(halo_send_data.at(i).data(), halo_send_sizes.at(i), halo_ranks[i]);
1424 }
1425 }
1426
1427 // wait for sends and receives to finish

1439 if(ancestor.progeny_comm.rank() != 0)
1440 {
1441 for(std::size_t i(0); i < num_halos; ++i)
1442 halo_recv_data.at(i).resize(halo_recv_sizes[i]);
1443 }
1444
1445 // broadcast all halos
1446 for(std::size_t i(0); i < num_halos; ++i)
1447 {
1448 if(halo_recv_sizes[i] > std::size_t(0))
1449 ancestor.progeny_comm.bcast(halo_recv_data[i].data(), halo_recv_sizes[i], 0);
1450 }
1451 }
1452

1455 for(std::size_t i(0); i < num_halos; ++i)
1456 {
1457 s += stringify(halo_ranks[i]);
1458 s += " >";

1468

1510 {
1511 for(int j(0); j <= shape_dim; ++j)
1512 {

1547

1575 Index iz = iel / (this->_base_slices[0] * this->_base_slices[1]);

1660 Adjacency::Coloring _compute_base_mesh_layering(const MeshNodeType& slag_node) const
1661 {
1662 // get element target set of slag patch
1663 const Geometry::MeshPart<MeshType>* slag_part = slag_node.get_patch(-1);
1664 XASSERT(slag_part != nullptr);
1665 const Geometry::TargetSet& slag_target = slag_part->template get_target_set<MeshType::shape_dim>();
1666
1667 // maximum number of layers = number of slices in highest dimensions; this may be less than
1668 // base_slices[shape_dim], since entire layers might have been removed during the deslagging process
1669 const Index max_layers = this->_base_slices[shape_dim-1];
1670 const Index num_elems = slag_target.get_num_entities();
1671
1672 // compute denominator for layer index computation
1673 Index denom = 1u;
1674 for(int j(0); j+1 < shape_dim; ++j)
1675 denom *= this->_base_slices[Index(j)];
1676
1677 // compute number of elements for each potential layer
1678 std::vector<Index> aux(max_layers, 0u);
1679 for(Index i(0); i < num_elems; ++i)
1680 ++aux[slag_target[i] / denom];
1681
1682 // count number of actually used layers
1683 Index num_layers = 0u;
1684 for(Index i(0); i < max_layers; ++i)
1685 {
1686 Index k = num_layers;
1687 if(aux[i] > 0u)
1688 ++num_layers;
1689 aux[i] = k;
1690 }
1691
1692 // compute layering
1693 Adjacency::Coloring layering(num_elems, num_layers);
1694 Index* layers = layering.get_coloring();
1695 for(Index i(0); i < num_elems; ++i)
1696 layers[i] = aux[slag_target[i] / denom];
1697
1698 return layering;

1710 Adjacency::Coloring _compute_refined_mesh_layering(const MeshNodeType& slag_node, const Adjacency::Coloring& coarse_layering) const
1711 {
1712 // get element target set of slag patch
1713 const Geometry::MeshPart<MeshType>* slag_part = slag_node.get_patch(-1);
1714 XASSERT(slag_part != nullptr);
1715 const Geometry::TargetSet& slag_target = slag_part->template get_target_set<MeshType::shape_dim>();
1716
1717 // maximum number of layers = twice the number of layers in coarse mesh; this may actually be less,
1718 // since entire layers might have been removed during the deslagging process
1719 const Index max_layers = 2 * coarse_layering.get_num_colors();
1720 const Index num_elems = slag_target.get_num_entities();

1722 // compute number of elements for each potential layer
1723 std::vector<Index> aux(max_layers, 0u);
1724 for(Index i(0); i < num_elems; ++i)
1725 {
1726 Index qux = slag_target[i] >> (shape_dim - 1);
1727 ++aux[(coarse_layering[qux >> 1] << 1) | (qux & 1)]; // all hail to bit-shift magic!
1728 }
1729
1730 // count number of actually used layers
1731 Index num_layers = 0u;
1732 for(Index i(0); i < max_layers; ++i)
1733 {
1734 Index k = num_layers;
1735 if(aux[i] > 0u)
1736 ++num_layers;
1737 aux[i] = k;
1738 }
1739

1741 Adjacency::Coloring layering(num_elems, num_layers);
1742 Index* layers = layering.get_coloring();
1743 for(Index i(0); i < num_elems; ++i)
1744 {
1745 Index qux = slag_target[i] >> (shape_dim - 1);
1746 layers[i] = aux[(coarse_layering[qux >> 1] << 1) | (qux & 1)];
1747 }
1748
1749 return layering;
