169 {
170 Statistics::add_solver_expression(std::make_shared<ExpressionStartSolve>(this->name()));
171 Statistics::add_solver_expression(std::make_shared<ExpressionCallPrecond>(this->name(), _op.name()));
172
173 Solver::Status st(_op.apply(vec_cor, vec_def));
174
175 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), st, 1));
176
