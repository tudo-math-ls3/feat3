34 template<int n_>
35 struct Basis
36 {
37 // interpolation points
38 template<typename T_>
39 static inline void points(T_ v[])
40 {
41 static const T_ s = T_(2) / T_(n_);
42 for(int i(0); i <= n_; ++i)
43 v[i] *= (T_(i)*s - T_(1));
44 }
45
46 // basis function values for generic P_n
47 template<typename T_>
48 static inline void val(T_ v[], const T_ x)
49 {
50 static const T_ s = T_(2) / T_(n_);
51
52 for(int j(0); j <= n_; ++j)
53 {
54 v[j] = T_(1);
55 for(int i(0); i <= n_; ++i)
56 v[j] *= (i != j ? (T_(1) + x - T_(i)*s) / (T_(j)*s - T_(i)*s) : T_(1));
57 }
58 }
59
60 // first order derivatives
61 template<typename T_>
62 static inline void d1(T_ v[], const T_ x)
63 {
64 static const T_ s = T_(2) / T_(n_);
65
66 for(int j(0); j <= n_; ++j)
67 {
68 v[j] = T_(0);
69 for(int k(0); k <= n_; ++k)
70 {
71 if(k == j)
72 continue;
73 T_ t = T_(1);
74 for(int i(0); i <= n_; ++i)
75 t *= ((i != k) && (i != j) ? (T_(1) + x - T_(i)*s) / (T_(j)*s - T_(i)*s) : T_(1));
76 v[j] += t / (T_(j)*s - T_(k)*s);
77 }
78 }
79 }
80
81 // second order derivatives
82 template<typename T_>
83 static inline void d2(T_ v[], const T_ x)
84 {
85 static const T_ s = T_(2) / T_(n_);
86
87 for(int j(0); j <= n_; ++j)
88 {
89 v[j] = T_(0);
90 for(int k(0); k <= n_; ++k)
91 {
92 if(k == j)
93 continue;
94 T_ t = T_(0);
95 for(int i(0); i <= n_; ++i)
96 {
97 if((i == k) || (i == j))
98 continue;
99 T_ r = T_(1);
100 for(int l(0); l <= n_; ++l)
101 r *= ((l != k) && (l != j) && (l != i) ? (T_(1) + x - T_(l)*s) / (T_(j)*s - T_(l)*s) : T_(1));
102 t += r / (T_(j)*s - T_(i)*s);
103 }
104 v[j] += t / (T_(j)*s - T_(k)*s);
105 }
106 }
107 }
108 }; // struct Basis<n_>

597 DataType volume() const
598 {
599 // for higher order, compute edge length by 3-point Gauss
600 if(degree_ > 2)
601 return volume_g3();
602
603 // first, compute the squared distance of the two edge end-points
604 const DataType a2 = (_iso_coeff[degree_] - _iso_coeff[0]).norm_euclid_sqr();
605 const DataType a = Math::sqrt(a2);
606
607 // degree = 1?
608 if(degree_ == 1)
609 return a;
610
611 // we're in the second degree case; so compute the squared distance between the edge midpoint
612 // and the midpoint between the two edge end-points
613 const DataType b2 = (_iso_coeff[1] - DataType(0.5) * (_iso_coeff[2] + _iso_coeff[0])).norm_euclid_sqr();
614
615 // if b2 = 0, then we have a straight edge, so its length is equal to a
616 if(b2 <= DataType(1E-5)*a2)
617 return a;
618
619 // now we have to compute the arc length of the polynomial (b/a^2)*x^2 over the interval [-a/2,+a/2],
620 // which can be computed by using the following magic formula:
621 const DataType b = DataType(4) * Math::sqrt(b2);
622 const DataType q = Math::sqrt(DataType(1) + DataType(16)*b2/a2);
623 return DataType(0.5) * a * (a * Math::log(q + b/a) / b + q);
624 }
625
635 DataType volume_g3() const
636 {
637 // 3-point Gauss-Legendre coordinate
638 static const DataType G = DataType(FEAT_F128C(0.77459666924148337703585307995647992216658434105));
639
640 DataType v = DataType(0);
641 JacobianMatrixType jac_mat;
642 DomainPointType dom_point;
643
644 // use 2x2 Gauss-Legendre rule to integrate the area/volume
645 dom_point[0] = DataType(0);
646 this->calc_jac_mat(jac_mat, dom_point);
647 v += DataType(8) * jac_mat.vol();
648
649 dom_point[0] = -G;
650 this->calc_jac_mat(jac_mat, dom_point);
651 v += DataType(5) * jac_mat.vol();
652
653 dom_point[0] = +G;
654 this->calc_jac_mat(jac_mat, dom_point);
655 v += DataType(5) * jac_mat.vol();
656
657 return v / DataType(9);
658 }

764 void prepare(Index cell_index)
765 {

768

771
772 // fetch the vertex set from the mesh

775
776 // fetch the vertex-index set

799 //DataType v[degree_+1];
800 //Intern::Basis<degree_>::points(v);
801 for(int i(1); i < degree_; ++i)
802 {
803 const DataType alpha = DataType(i) / DataType(degree_);
804 //const DataType alpha = (v[i] + DataType(1)) * DataType(0.5);

806 // bottom edge
807 _iso_coeff[0][i].set_convex(alpha, _iso_coeff[0][0], _iso_coeff[0][n]);

809 _iso_coeff[n][i].set_convex(alpha, _iso_coeff[n][0], _iso_coeff[n][n]);

817 const int ox[4] = {1, 1, 0, n};
818 const int oy[4] = {0, n, 1, 1};
819 const int ix[4] = {1, 1, 0, 0};
820 const int iy[4] = {0, 0, 1, 1};
821
822 // loop over all four edges of the element
823 for(int e = 0; e < 4; ++e)
824 {
825 // get global edge index
826 const Index edge_idx = edges_at_quad(cell_index, e);
827
828 // get chart for that edge
829 const ChartType* chart = _charts_1.at(edge_idx);
830
831 // no chart?
832 if(chart == nullptr)
833 continue;
