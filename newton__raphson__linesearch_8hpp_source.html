157 virtual Status _apply_intern(VectorType& vec_sol, const VectorType& vec_dir)
158 {
159 Statistics::add_solver_expression(std::make_shared<ExpressionStartSolve>(this->name()));
160
161 // The step length wrt. to the NORMALISED search direction
162 DataType alpha(0);
163 // The functional value
164 DataType fval(0);
166 DataType df(0);
167
168 // Perform initializations and checks
169 Status st = this->_startup(alpha, fval, df, vec_sol, vec_dir);
170 // Because of the Newton-Raphson update x[k+1] = x[k] + alpha[k+1] d and alpha[k+1] = alpha[k] - < g[k], d > / < d, H[k] d >,
171 // || d ||_2 does not matter and the relative update is invariant
172
173 // Compute new _alpha <- _alpha - grad.dot(vec_dir) / vec_dir.dot(Hess*vec_dir)
174 this->_functional.apply_hess(this->_vec_tmp, this->_vec_pn);
175 this->_filter.filter_def(this->_vec_tmp);
176
177 alpha = -this->_vec_grad.dot(this->_vec_pn)/this->_vec_pn.dot(this->_vec_tmp);
178
179 // start iterating
180 while(st == Status::progress)
181 {
182 IterationStats stat(*this);
183
184 // Increase iteration count
185 ++this->_num_iter;
186
187 this->_functional.prepare(vec_sol, this->_filter);
188 this->_functional.eval_fval_grad(fval, this->_vec_grad);
189 // DO NOT call trim() here, as this simple linesearch accepts bad steps too, which would then be trimmed
190 // and used.
191 //this->trim_func_grad(fval);
192 this->_filter.filter_def(this->_vec_grad);
193
194 df = this->_vec_pn.dot(this->_vec_grad);
195
196 if(fval <= this->_fval_min)
197 {
198 this->_fval_min = fval;
199 this->_alpha_min = alpha;
200 }
201
202 st = this->_check_convergence(fval, df, alpha);
203
204 if(st != Status::progress)
205 {
206 break;
207 }
208
209 // Compute new _alpha <- _alpha - grad.dot(this->_vec_pn) / this->_vec_pn.dot(Hess*this->_vec_pn)
210 this->_functional.apply_hess(this->_vec_tmp, this->_vec_pn);
211 this->_filter.filter_def(this->_vec_tmp);
212
213 alpha -= this->_vec_grad.dot(this->_vec_pn)/this->_vec_pn.dot(this->_vec_tmp);
214
215 // Update solution
216 vec_sol.axpy(this->_vec_pn, this->_vec_initial_sol, alpha);
217
218 }
219
220 // If we are not successful, we update the best step length and need to re-evaluate everything for that
221 // step
222 if(st != Status::success)
223 {
224 vec_sol.axpy(this->_vec_pn, this->_vec_initial_sol, this->_alpha_min);
225
226 // Prepare and evaluate
227 this->_functional.prepare(vec_sol, this->_filter);
228 this->_functional.eval_fval_grad(fval, this->_vec_grad);
229 // DO NOT call trim() here, as this simple linesearch accepts bad steps too, which would then be trimmed
230 // and used.
231 this->trim_func_grad(fval);
232 this->_filter.filter_def(this->_vec_grad);
233 }
234
235 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::undefined, this->get_num_iter()));
236 return st;
237 }
