285 LocalVector_ join(const Global::Vector<LocalVector_, Mirror_>& vector) const
286 {
287 // are there no other processes involved here?
288 if(is_single())
289 return vector.local().clone();
290
291 // make sure the root rank has a base-vector template
292 if(is_root())
293 {
294 XASSERTM(_base_vector_tmpl.size() > Index(0), "root process must have a base vector template");
295 }
296
297 // create an base-mesh vector (may be empty on all processes except for parent)
298 LocalVector_ vec_base(this->_base_vector_tmpl.clone(LAFEM::CloneMode::Layout));
299
300 // actual join
301 join(vec_base, vector);
302
303 // return local vector
304 return vec_base;
305 }

322 void join(LocalVector_& vec_base, const Global::Vector<LocalVector_, Mirror_>& vector) const
323 {
324 // are there no other processes involved here?
325 if(is_single())
326 {
327 vec_base.copy(vector.local());
328 return;
329 }
330
331 // Note: the 'join' of the muxer class works with type-0 vectors, so we need to create a
332 // temporary type-0 conversion of our input vector here, which is assumed to be a type-1
333 // vector. In theory, we also should apply a 'sync_0' on the returned vector, but since
334 // its a 'local' vector on the whole base-mesh domain, it is redundant and it also does
335 // not make sense due to the lack of a gate on the base-mesh, i.e. on the base-mesh
336 // type-0 and type-1 are the same thing.
337
338 // convert input vector to type 0
339 GlobalVectorType vector_0 = vector.clone(LAFEM::CloneMode::Deep);
340 vector_0.from_1_to_0();
341
342 // is this the muxer parent or only a child process?
343 if(_muxer.is_parent())
344 {
345 _muxer.join(vector_0.local(), vec_base);
346 }
347 else
348 {
349 XASSERT(_muxer.is_child());
350 _muxer.join_send(vector_0.local());
351 }
352 }

380 void split(LocalVector_& vector, const LocalVector_& vec_base) const
381 {
382 // are there no other processes involved here?
383 if(is_single())
384 {
385 // simply copy the local vector contents
386 vector.copy(vec_base);
387 return;
388 }
389
390 // do we have a base muxer?
391 if(_muxer.is_parent())
392 {
393 _muxer.split(vector, vec_base);
394 }
395 else
396 {
397 XASSERT(_muxer.is_child());
398 _muxer.split_recv(vector);
399 }
