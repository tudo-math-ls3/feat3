264 LocalVector_ join(const Global::Vector<LocalVector_, Mirror_>& vector) const
265 {
266 // are there no other processes involved here?
267 if(is_single())
268 return vector.local().clone();
269
270 // make sure the root rank has a base-vector template
271 if(is_root())
272 {
273 XASSERTM(_base_vector_tmpl.size() > Index(0), "root process must have a base vector template");
274 }
275
276 // create an base-mesh vector (may be empty on all processes except for parent)
277 LocalVector_ vec_base(this->_base_vector_tmpl.clone(LAFEM::CloneMode::Layout));
278
279 // actual join
280 join(vec_base, vector);
281
282 // return local vector
283 return vec_base;
284 }

301 void join(LocalVector_& vec_base, const Global::Vector<LocalVector_, Mirror_>& vector) const
302 {
303 // are there no other processes involved here?
304 if(is_single())
305 {
306 vec_base.copy(vector.local());
307 return;
308 }
309
310 // Note: the 'join' of the muxer class works with type-0 vectors, so we need to create a
311 // temporary type-0 conversion of our input vector here, which is assumed to be a type-1
312 // vector. In theory, we also should apply a 'sync_0' on the returned vector, but since
313 // its a 'local' vector on the whole base-mesh domain, it is redundant and it also does
314 // not make sense due to the lack of a gate on the base-mesh, i.e. on the base-mesh
315 // type-0 and type-1 are the same thing.
316
317 // convert input vector to type 0
318 GlobalVectorType vector_0 = vector.clone(LAFEM::CloneMode::Deep);
319 vector_0.from_1_to_0();
320
321 // is this the muxer parent or only a child process?
322 if(_muxer.is_parent())
323 {
324 _muxer.join(vector_0.local(), vec_base);
325 }
326 else
327 {
328 XASSERT(_muxer.is_child());
329 _muxer.join_send(vector_0.local());
330 }
331 }

359 void split(LocalVector_& vector, const LocalVector_& vec_base) const
360 {
361 // are there no other processes involved here?
362 if(is_single())
363 {
364 // simply copy the local vector contents
365 vector.copy(vec_base);
366 return;
367 }
368
369 // do we have a base muxer?
370 if(_muxer.is_parent())
371 {
372 _muxer.split(vector, vec_base);
373 }
374 else
375 {
376 XASSERT(_muxer.is_child());
377 _muxer.split_recv(vector);
378 }
379 }
