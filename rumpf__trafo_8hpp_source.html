173 template<typename DataType_>
174 struct RumpfTrafo<Trafo::Standard::Mapping<Geometry::ConformalMesh<Shape::Hypercube<2>,2,DataType_>>,DataType_>
175 {
176 public:
178 typedef Trafo::Standard::Mapping<Geometry::ConformalMesh<Shape::Hypercube<2>,2,DataType_>> TrafoType;
180 typedef DataType_ DataType;
182 typedef typename TrafoType::ShapeType ShapeType;
184 typedef typename TrafoType::MeshType MeshType;
186 typedef Tiny::Matrix<DataType_, Shape::FaceTraits<ShapeType,0>::count, MeshType::world_dim> Tx;
188 typedef Tiny::Matrix<DataType_, MeshType::world_dim, MeshType::world_dim> MatTensorType;
190 typedef LAFEM::DenseVector<DataType, Index> ScalarVectorType;
192 typedef LAFEM::DenseVectorBlocked<DataType, Index, MeshType::world_dim> VectorType;
193
194 private:
198 static DataType compute_det(Tx& x)
199 {
200 return (x(0,0) * x(1,1) - x(0,0) * x(2,1) - x(0,1) * x(1,0) + x(0,1) * x(2,0)
201 + x(1,0) * x(3,1) - x(1,1) * x(3,0) - x(2,0) * x(3,1) + x(2,1) * x(3,0)) / DataType(2);
202 }
203
207 static void compute_grad_det(Tx& grad, const Tx& x)
208 {
209 grad(0,0) = ( x(1,1) - x(2,1)) / DataType(2);
210 grad(0,1) = (-x(1,0) + x(2,0)) / DataType(2);
211 grad(1,0) = (-x(0,1) + x(3,1)) / DataType(2);
212 grad(1,1) = ( x(0,0) - x(3,0)) / DataType(2);
213 grad(2,0) = ( x(0,1) - x(3,1)) / DataType(2);
214 grad(2,1) = (-x(0,0) + x(3,0)) / DataType(2);
215 grad(3,0) = (-x(1,1) + x(2,1)) / DataType(2);
216 grad(3,1) = ( x(1,0) - x(2,0)) / DataType(2);
217 }
218
219 public:
223 static void compute_h(ScalarVectorType& h_, const ScalarVectorType& lambda_, const DataType& sum_det_)
224 {
225 DataType_ exponent = DataType_(1)/DataType_(MeshType::world_dim);
226
227 for(Index cell(0); cell < h_.size(); ++cell)
228 {
229 // For hypercubes, h is half the refence cell's edge length
230 h_(cell, DataType(0.5)*Math::pow(lambda_(cell)*sum_det_,exponent));
231 }
232 }

237 static DataType_ compute_sum_det(const VectorType& coords_, const MeshType& mesh_)
238 {
239 // This will hold the coordinates for one element for passing to other routines
240 Tiny::Matrix <DataType_, Shape::FaceTraits<ShapeType,0>::count, MeshType::world_dim> x;
241
242 // Index set for local/global numbering
243 const auto& idx = mesh_.template get_index_set<ShapeType::dimension,0>();
244
245 DataType sum_det(0);
246 for(Index cell(0); cell < mesh_.get_num_entities(ShapeType::dimension); ++cell)
247 {
248 // Get local coordinates
249 for(int j(0); j < Shape::FaceTraits<ShapeType,0>::count; ++j)
250 {
251 x[j] = coords_(idx(cell,j));
252 }
253
254 sum_det += compute_det(x);
255 }
256
257 return sum_det;
258 }
259
263 static void compute_grad_sum_det(VectorType& grad_, const VectorType& coords_, const MeshType& mesh_)
264 {
265 // This will hold the coordinates for one element for passing to other routines
266 Tx x(DataType_(0));
267
268 // Index set for local/global numbering
269 const auto& idx = mesh_.template get_index_set<ShapeType::dimension,0>();
270
271 Tx local_grad(DataType_(0));
272
273 grad_.format();
274
275 for(Index cell(0); cell < mesh_.get_num_entities(ShapeType::dimension); ++cell)
276 {
277 // Get local coordinates
278 for(int j(0); j < Shape::FaceTraits<ShapeType,0>::count; ++j)
279 {
280 x[j] = coords_(idx(cell,j));
281 }
282
283 compute_grad_det(local_grad, x);
284
285 for(int j(0); j < Shape::FaceTraits<ShapeType,0>::count; ++j)
286 {
287 Index i(idx(cell, j));
288 grad_(i, grad_(i) + local_grad[j]);
289 }
290
291 }
292 } // compute_grad_sum_det

297 static MatTensorType compute_mat_tensor(const Tx& DOXY(x), const DataType_& h)
298 {
299 MatTensorType mat_tensor(DataType_(0));
300 DataType scal(DataType(1)/h);
301
302 for(int d(0); d < MeshType::world_dim; ++d)
303 {
304 mat_tensor(d,d) = scal;
305 }
306
307 return mat_tensor;
308 }
