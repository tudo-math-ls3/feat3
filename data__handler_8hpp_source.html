154 template<typename ITX_>
155 bool _contains_common_element(const ITX_* a, const ITX_* ae, const ITX_* b, const ITX_* be) const
156 {
157 std::vector<ITX_> intersection;
158
159 std::set<ITX_> aa { a, ae };
160 std::set<ITX_> bb { b, be };
161
162 std::set_intersection(aa.begin(), aa.end(),
163 bb.begin(), bb.end(),
164 std::inserter(intersection, intersection.end()));
165
166 return !intersection.empty();
167 }
