166 bool _contains_common_element(const ITX_* a, const ITX_* ae, const ITX_* b, const ITX_* be) const
167 {
168 std::vector<ITX_> intersection;
169
170 std::set<ITX_> aa { a, ae };
171 std::set<ITX_> bb { b, be };
172
173 std::set_intersection(aa.begin(), aa.end(),
174 bb.begin(), bb.end(),
175 std::inserter(intersection, intersection.end()));
176
177 return !intersection.empty();
178 }
