173 virtual Status _apply_intern(VectorType& vec_sol, const VectorType& DOXY(vec_rhs))
174 {
175 IterationStats pre_iter(*this);
176 Statistics::add_solver_expression(std::make_shared<ExpressionStartSolve>(this->name()));
177
178 const MatrixType& matrix(this->_system_matrix);
179 const FilterType& filter(this->_system_filter);
180 VectorType& vec_r(this->_vec_r);
181 VectorType& vec_z(this->_vec_z);
182 VectorType& vec_s(this->_vec_s);
183 VectorType& vec_p(this->_vec_p);
184 VectorType& vec_S(this->_vec_S);
185 VectorType& vec_Z(this->_vec_Z);
186
187 DataType gamma, gamma_new, beta, alpha, t;
188 gamma = DataType(0);
189 gamma_new = DataType(0);
190 beta = DataType(0);
191 alpha = DataType(0);
192 t = DataType(0);
193
194 Status status = this->_set_initial_defect(vec_r, vec_sol);
195 if(status != Status::progress)
196 {
197 pre_iter.destroy();
198 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), status, this->get_num_iter()));
199 return status;
200 }
201
202 if(!this->_apply_precond(vec_z, vec_r, filter))
203 {
204 pre_iter.destroy();
205 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
206 return Status::aborted;
207 }
208
209 vec_p.copy(vec_z);
210
211 gamma = vec_z.dot(vec_r);
212
213 matrix.apply(vec_s, vec_p);
214 filter.filter_def(vec_s);
215
216 pre_iter.destroy();
217
218 // start iterating
219 while(status == Status::progress)
220 {
221 IterationStats stat(*this);
222
223 auto dot_t = vec_s.dot_async(vec_p);
224 if(!this->_apply_precond(vec_S, vec_s, filter))
225 {
226 stat.destroy();
227 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
228 return Status::aborted;
229 }
230 t = dot_t.wait();
231
232 alpha = gamma / t;
233
234 vec_r.axpy(vec_s, -alpha);
235 auto norm_def_cur = vec_r.norm2_async();
236
237 vec_sol.axpy(vec_p, alpha);
238
239 vec_z.axpy(vec_S, -alpha);
240
241 auto dot_gamma_new = vec_r.dot_async(vec_z);
242 matrix.apply(vec_Z, vec_z);
243 filter.filter_def(vec_Z);
244
245 gamma_new = dot_gamma_new.wait();
246 beta = gamma_new / gamma;
247 gamma = gamma_new;
248
249 vec_p.scale(vec_p, beta);
250 vec_p.axpy(vec_z);
251 vec_s.scale(vec_s, beta);
252 vec_s.axpy(vec_Z);
253
254 status = this->_update_defect(norm_def_cur.wait());
255 if(status != Status::progress)
256 {
257 stat.destroy();
258 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), status, this->get_num_iter()));
259 return status;
260 }
261 }
262
263 // we should never reach this point...
264 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::undefined, this->get_num_iter()));
265 return Status::undefined;
266 }
