44 template<int max_>
45 class CommonDofMap
46 {
47 public:
48 Index glob_dofs[max_];
49 int loc1_dofs[max_];
50 int loc2_dofs[max_];
51 int num_dofs;
52
53 explicit CommonDofMap() : num_dofs(0) {}
54
55 void clear()
56 {
57 num_dofs = 0;
58 }
59
60 int push_1(Index dof, int loc)
61 {
62 for(int i(0); i < num_dofs; ++i)
63 {
64 if(dof == glob_dofs[i])
65 {
66 loc1_dofs[i] = loc;
67 return i;
68 }
69 }
70 ASSERT(num_dofs < max_);
71 glob_dofs[num_dofs] = dof;
72 loc1_dofs[num_dofs] = loc;
73 loc2_dofs[num_dofs] = -1;
74 return num_dofs++;
75 }
76
77 int push_2(Index dof, int loc)
78 {
79 for(int i(0); i < num_dofs; ++i)
80 {
81 if(dof == glob_dofs[i])
82 {
83 loc2_dofs[i] = loc;
84 return i;
85 }
86 }
87 ASSERT(num_dofs < max_);
88 glob_dofs[num_dofs] = dof;
89 loc1_dofs[num_dofs] = -1;
90 loc2_dofs[num_dofs] = loc;
91 return num_dofs++;
92 }
93
94 int loc_1(int k) const
95 {
96 return loc1_dofs[k];
97 }
98
99 int loc_2(int k) const
100 {
101 return loc2_dofs[k];
102 }
103
104 int get_num_local_dofs() const
105 {
106 return num_dofs;
107 }
108
109 Index get_index(int i) const
110 {
111 return glob_dofs[i];
112 }
113 }; // class CommonDofMap<...>

137 template<typename Trafo_>
138 class TraceAssembler
139 {
140 public:
141 typedef Trafo_ TrafoType;
142 typedef typename TrafoType::MeshType MeshType;
143 typedef typename TrafoType::ShapeType ShapeType;
144
145 static constexpr int shape_dim = ShapeType::dimension;
146 static constexpr int facet_dim = shape_dim-1;
147
148 protected:
150 const TrafoType& _trafo;
152 std::vector<int> _facet_mask, _cell_facet, _facet_ori;
154 std::vector<Index> _cells, _facets;
155
156 public:
163 explicit TraceAssembler(const TrafoType& trafo) :
164 _trafo(trafo),
165 _facet_mask(trafo.get_mesh().get_num_entities(facet_dim), 0)
166 {
167 }
168
172 void clear()
173 {
174 _cell_facet.clear();
175 _facet_ori.clear();
176 _cells.clear();
177 _facets.clear();
178 for(auto& f : _facets)
179 f = Index(0);
180 }
181
188 void add_facet(Index ifacet)
189 {
190 ASSERTM(ifacet < Index(_facet_mask.size()), "invalid facet index");
191 _facet_mask.at(ifacet) = 1;
192 }

200 void add_mesh_part(const Geometry::MeshPart<MeshType>& mesh_part)
201 {
202 const auto& trg = mesh_part.template get_target_set<facet_dim>();
203 for(Index i(0); i < trg.get_num_entities(); ++i)
204 _facet_mask.at(trg[i]) = 1;
205 }

217 void compile()
218 {
219 _cell_facet.clear();
220 _facet_ori.clear();
221 _cells.clear();
222 _facets.clear();
223
224 // build elements-at-facet graph
225 Adjacency::Graph elem_at_facet(Adjacency::RenderType::injectify_transpose,
226 _trafo.get_mesh().template get_index_set<shape_dim, facet_dim>());
227
228 // loop over all facets
229 for(Index iface(0); iface < Index(_facet_mask.size()); ++iface)
230 {
231 // use this facet?
232 if(_facet_mask[iface] == 0)
233 continue;
234
235 // ensure that this is a boundary facet if required
236 //if(only_boundary && (elem_at_facet.degree(iface) != Index(1)))
237 //XABORTM("facet is adjacent to more than 1 element");
238
239 // add all elements
240 for(auto it = elem_at_facet.image_begin(iface); it != elem_at_facet.image_end(iface); ++it)
241 {
242 Index icell = *it;
243
244 // try to compute local facet index and orientation
245 int loc_face(0), face_ori(0);
246 if(!_find_local_facet(iface, icell, loc_face, face_ori))
247 XABORTM("failed to find local facet");
248
249 // alright, add this facet to our list
250 _facets.push_back(iface);
251 _cells.push_back(icell);
252 _cell_facet.push_back(loc_face);
253 _facet_ori.push_back(face_ori);
254 }
255 }
256 }

268 void compile_all_facets(bool inner, bool outer)
269 {
270 _cell_facet.clear();
271 _facet_ori.clear();
272 _cells.clear();
273 _facets.clear();
274
275 // build elements-at-facet graph
276 Adjacency::Graph elem_at_facet(Adjacency::RenderType::injectify_transpose,
277 _trafo.get_mesh().template get_index_set<shape_dim, facet_dim>());
278
279 // loop over all facets
280 for(Index iface(0); iface < Index(_facet_mask.size()); ++iface)
281 {
282 // how many elements are adjacent to this facet?
283 const int degree = int(elem_at_facet.degree(iface));
284 XASSERT(degree > 0);
285 XASSERT(degree < 3);
286
287 // outer facet with only 1 adjacent element?
288 if((degree == 1) && !outer)
289 continue;
290
291 // inner facet with 2 adjacent elements?
292 if((degree == 2) && !inner)
293 continue;
294
295 // add all elements
296 for(auto it = elem_at_facet.image_begin(iface); it != elem_at_facet.image_end(iface); ++it)
297 {
298 Index icell = *it;
299
300 // try to compute local facet index and orientation
301 int loc_face(0), face_ori(0);
302 if(!_find_local_facet(iface, icell, loc_face, face_ori))
303 XABORTM("failed to find local facet");
304
305 // alright, add this facet to our list
306 _facets.push_back(iface);
307 _cells.push_back(icell);
308 _cell_facet.push_back(loc_face);
309 _facet_ori.push_back(face_ori);
310 }
311 }

469 template<
470 typename Matrix_,
471 typename Operator_,
472 typename TestSpace_,
473 typename TrialSpace_,
474 typename CubatureFactory_>
475 void assemble_operator_matrix2(
476 Matrix_& matrix,
477 Operator_& operat,
478 const TestSpace_& test_space,
479 const TrialSpace_& trial_space,
480 const CubatureFactory_& cubature_factory,
481 typename Matrix_::DataType alpha = typename Matrix_::DataType(1)) const
482 {
483 // validate matrix dimensions
484 XASSERTM(matrix.rows() == test_space.get_num_dofs(), "invalid matrix dimensions");
485 XASSERTM(matrix.columns() == trial_space.get_num_dofs(), "invalid matrix dimensions");
486
487 // matrix type
488 typedef Matrix_ MatrixType;
489 // operator type
490 typedef Operator_ OperatorType;
491 // test-space type
492 typedef TestSpace_ TestSpaceType;
493 // trial-space type
494 typedef TrialSpace_ TrialSpaceType;
495
496 // assembly traits
497 typedef AsmTraits2<
498 typename MatrixType::DataType,
499 TestSpaceType,
500 TrialSpaceType,
501 OperatorType::trafo_config,
502 OperatorType::test_config,
503 OperatorType::trial_config> AsmTraits;
504
505 typedef typename AsmTraits::DataType DataType;
506
507 // shape types
508 typedef typename Shape::FaceTraits<ShapeType, ShapeType::dimension-1>::ShapeType FacetType;
509
510 // fetch the trafo
511 const TrafoType& trafo = test_space.get_trafo();
512
513 // create a trafo evaluator
514 typename AsmTraits::TrafoEvaluator trafo_eval(trafo);
515
516 // create a trafo facet evaluator
517 typedef typename TrafoType::template Evaluator<FacetType, DataType>::Type TrafoFacetEvaluator;
518 TrafoFacetEvaluator trafo_facet_eval(trafo);
519
520 static constexpr TrafoTags trafo_facet_tags = TrafoTags::jac_det | TrafoTags::jac_mat;
521 typedef typename TrafoFacetEvaluator::template ConfigTraits<trafo_facet_tags>::EvalDataType TrafoFacetEvalData;
522
523 // create space evaluators
524 typename AsmTraits::TestEvaluator test_eval(test_space);
525 typename AsmTraits::TrialEvaluator trial_eval(trial_space);
526
527 // create dof-mappings
528 typename AsmTraits::TestDofMapping test_dof_mapping(test_space);
529 typename AsmTraits::TrialDofMapping trial_dof_mapping(trial_space);
530
531 // create a operator evaluator
532 typename OperatorType::template Evaluator<AsmTraits> oper_eval(operat);
533
534 // create trafo evaluation data
535 typename AsmTraits::TrafoEvalData trafo_data;
536 TrafoFacetEvalData trafo_facet_data;
537
538 // create space evaluation data
539 typename AsmTraits::TestEvalData test_data;
540 typename AsmTraits::TrialEvalData trial_data;
541
542 // the value type of the operator
543 typedef typename OperatorType::template Evaluator<AsmTraits>::ValueType ValueType;
544
545 // ensure that the operator and matrix value types are compatible
546 static_assert(std::is_same<ValueType, typename MatrixType::ValueType>::value,
547 "matrix and bilinear operator have different value types!");
548
549 // create local matrix data
550 typename AsmTraits::template TLocalMatrix<ValueType> loc_mat;
551
552 // create cubature rule
553 typename Intern::CubatureTraits<TrafoFacetEvaluator>::RuleType cubature_rule(Cubature::ctor_factory, cubature_factory);
554
555 // create matrix scatter-axpy
556 typename MatrixType::ScatterAxpy scatter_axpy(matrix);
557
558 // trafo matrices and vectors
559 Tiny::Matrix<DataType, shape_dim, facet_dim> face_mat;
560 Tiny::Matrix<DataType, facet_dim, facet_dim> ori_mat;
561 Tiny::Vector<DataType, shape_dim> face_vec;
562 Tiny::Vector<DataType, facet_dim> ori_vec;
563
564 face_mat.format();
565 ori_mat.format();
566 face_vec.format();
567 ori_vec.format();
568
569 // loop over all cells of the mesh
570 for(Index f(0); f < Index(_facets.size()); ++f)
571 {
572 // get facet index
573 const Index face = _facets[f];
574 const Index cell = _cells[f];
575
576 // compute facet trafos
577 Geometry::Intern::FaceRefTrafo<ShapeType, facet_dim>::compute(face_mat, face_vec, _cell_facet[f]);
578
579 // compute orientation trafos
580 Geometry::Intern::CongruencyTrafo<FacetType>::compute(ori_mat, ori_vec, _facet_ori[f]);
581
582 // compute orientation of actual cell facet
583 const int cell_facet_ori = Geometry::Intern::CongruencySampler<FacetType>::orientation(_facet_ori[f])
584 * Shape::ReferenceCell<ShapeType>::facet_orientation(_cell_facet[f]);
585
586 // prepare trafo evaluators
587 trafo_facet_eval.prepare(face);
588 trafo_eval.prepare(cell);
589
590 // prepare space evaluators
591 test_eval.prepare(trafo_eval);
592 trial_eval.prepare(trafo_eval);
593
594 // prepare operator evaluator
595 oper_eval.prepare(trafo_eval);
596
597 // fetch number of local dofs
598 int num_loc_test_dofs = test_eval.get_num_local_dofs();
599 int num_loc_trial_dofs = trial_eval.get_num_local_dofs();
600
601 // format local matrix
602 loc_mat.format();
603
604 // loop over all quadrature points and integrate
605 for(int k(0); k < cubature_rule.get_num_points(); ++k)
606 {
607 // get cubature point
608 auto cub_pt = cubature_rule.get_point(k);
609
610 // transform to local facet
611 auto cub_cf = (face_mat * ((ori_mat * cub_pt) + ori_vec)) + face_vec;
612
613 // compute trafo data
614 trafo_facet_eval(trafo_facet_data, cub_pt);
615 trafo_eval(trafo_data, cub_cf);
616
617 // compute normal vector
618 trafo_data.normal = Tiny::orthogonal(trafo_facet_data.jac_mat).normalize();
619 if(cell_facet_ori < 0)
620 trafo_data.normal.negate();
621
622 // compute basis function data
623 test_eval(test_data, trafo_data);
624 trial_eval(trial_data, trafo_data);
625
626 // prepare bilinear operator
627 oper_eval.set_point(trafo_data);
628
629 // test function loop
630 for(int i(0); i < num_loc_test_dofs; ++i)
631 {
632 // trial function loop
633 for(int j(0); j < num_loc_trial_dofs; ++j)
634 {
635 // evaluate operator and integrate
636 Tiny::axpy(loc_mat(i,j), oper_eval.eval(trial_data.phi[j], test_data.phi[i]),
637 trafo_facet_data.jac_det * cubature_rule.get_weight(k));
638 // continue with next trial function
639 }
640 // continue with next test function
641 }
642 // continue with next cubature point
643 }
644
645 // finish operator evaluator
646 oper_eval.finish();
647
648 // finish evaluators
649 trial_eval.finish();
650 test_eval.finish();
651 trafo_eval.finish();
652 trafo_facet_eval.finish();
653
654 // initialize dof-mappings
655 test_dof_mapping.prepare(cell);
656 trial_dof_mapping.prepare(cell);
657
658 // incorporate local matrix
659 scatter_axpy(loc_mat, test_dof_mapping, trial_dof_mapping, alpha);
660
661 // finish dof mapping
662 trial_dof_mapping.finish();
663 test_dof_mapping.finish();
664
665 // continue with next cell
666 }
667
668 // okay, that's it
669 }

905
