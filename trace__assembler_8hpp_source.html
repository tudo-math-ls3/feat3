46 template<int max_>
47 class CommonDofMap
48 {
49 public:
50 Index glob_dofs[max_];
51 int loc1_dofs[max_];
52 int loc2_dofs[max_];
53 int num_dofs;
54
55 explicit CommonDofMap() : num_dofs(0) {}
56
57 void clear()
58 {
59 num_dofs = 0;
60 }
61
62 int push_1(Index dof, int loc)
63 {
64 for(int i(0); i < num_dofs; ++i)
65 {
66 if(dof == glob_dofs[i])
67 {
68 loc1_dofs[i] = loc;
69 return i;
70 }
71 }
72 ASSERT(num_dofs < max_);
73 glob_dofs[num_dofs] = dof;
74 loc1_dofs[num_dofs] = loc;
75 loc2_dofs[num_dofs] = -1;
76 return num_dofs++;
77 }
78
79 int push_2(Index dof, int loc)
80 {
81 for(int i(0); i < num_dofs; ++i)
82 {
83 if(dof == glob_dofs[i])
84 {
85 loc2_dofs[i] = loc;
86 return i;
87 }
88 }
89 ASSERT(num_dofs < max_);
90 glob_dofs[num_dofs] = dof;
91 loc1_dofs[num_dofs] = -1;
92 loc2_dofs[num_dofs] = loc;
93 return num_dofs++;
94 }
95
96 int loc_1(int k) const
97 {
98 return loc1_dofs[k];
99 }
100
101 int loc_2(int k) const
102 {
103 return loc2_dofs[k];
104 }
105
106 int get_num_local_dofs() const
107 {
108 return num_dofs;
109 }
110
111 Index get_index(int i) const
112 {
113 return glob_dofs[i];
114 }
115 }; // class CommonDofMap<...>
116 } // namespace Intern

139 template<typename Trafo_>
140 class TraceAssembler
141 {
142 public:
143 typedef Trafo_ TrafoType;
144 typedef typename TrafoType::MeshType MeshType;
145 typedef typename TrafoType::ShapeType ShapeType;
146
147 static constexpr int shape_dim = ShapeType::dimension;
148 static constexpr int facet_dim = shape_dim-1;
149
150 protected:
152 const TrafoType& _trafo;
154 std::vector<int> _facet_mask, _cell_facet, _facet_ori;
156 std::vector<Index> _cells, _facets;
157
158 public:
165 explicit TraceAssembler(const TrafoType& trafo) :
166 _trafo(trafo),
167 _facet_mask(trafo.get_mesh().get_num_entities(facet_dim), 0)
168 {
169 }
170
174 void clear()
175 {
176 _cell_facet.clear();
177 _facet_ori.clear();
178 _cells.clear();
179 _facets.clear();
180 for(auto& f : _facets)
181 f = Index(0);
182 }
183
190 void add_facet(Index ifacet)
191 {
192 ASSERTM(ifacet < Index(_facet_mask.size()), "invalid facet index");
193 _facet_mask.at(ifacet) = 1;
194 }
195
202 void add_mesh_part(const Geometry::MeshPart<MeshType>& mesh_part)
203 {
204 const auto& trg = mesh_part.template get_target_set<facet_dim>();
205 for(Index i(0); i < trg.get_num_entities(); ++i)
206 _facet_mask.at(trg[i]) = 1;
207 }
208
219 void compile()
220 {
221 _cell_facet.clear();
222 _facet_ori.clear();
223 _cells.clear();
224 _facets.clear();
225
226 // build elements-at-facet graph
227 Adjacency::Graph elem_at_facet(Adjacency::RenderType::injectify_transpose,
228 _trafo.get_mesh().template get_index_set<shape_dim, facet_dim>());
229
230 // loop over all facets
231 for(Index iface(0); iface < Index(_facet_mask.size()); ++iface)
232 {
233 // use this facet?
234 if(_facet_mask[iface] == 0)
235 continue;
236
237 // ensure that this is a boundary facet if required
238 //if(only_boundary && (elem_at_facet.degree(iface) != Index(1)))
239 //XABORTM("facet is adjacent to more than 1 element");
240
241 // add all elements
242 for(auto it = elem_at_facet.image_begin(iface); it != elem_at_facet.image_end(iface); ++it)
243 {
244 Index icell = *it;
245
246 // try to compute local facet index and orientation
247 int loc_face(0), face_ori(0);
248 if(!_find_local_facet(iface, icell, loc_face, face_ori))
249 XABORTM("failed to find local facet");
250
251 // alright, add this facet to our list
252 _facets.push_back(iface);
253 _cells.push_back(icell);
254 _cell_facet.push_back(loc_face);
255 _facet_ori.push_back(face_ori);
256 }
257 }
258 }

270 void compile_all_facets(bool inner, bool outer)
271 {
272 _cell_facet.clear();
273 _facet_ori.clear();
274 _cells.clear();
275 _facets.clear();
276
277 // build elements-at-facet graph
278 Adjacency::Graph elem_at_facet(Adjacency::RenderType::injectify_transpose,
279 _trafo.get_mesh().template get_index_set<shape_dim, facet_dim>());
280
281 // loop over all facets
282 for(Index iface(0); iface < Index(_facet_mask.size()); ++iface)
283 {
284 // how many elements are adjacent to this facet?
285 const int degree = int(elem_at_facet.degree(iface));
286 XASSERT(degree > 0);
287 XASSERT(degree < 3);
288
289 // outer facet with only 1 adjacent element?
290 if((degree == 1) && !outer)
291 continue;
292
293 // inner facet with 2 adjacent elements?
294 if((degree == 2) && !inner)
295 continue;
296
297 // add all elements
298 for(auto it = elem_at_facet.image_begin(iface); it != elem_at_facet.image_end(iface); ++it)
299 {
300 Index icell = *it;
301
302 // try to compute local facet index and orientation
303 int loc_face(0), face_ori(0);
304 if(!_find_local_facet(iface, icell, loc_face, face_ori))
305 XABORTM("failed to find local facet");
306
307 // alright, add this facet to our list
308 _facets.push_back(iface);
309 _cells.push_back(icell);
310 _cell_facet.push_back(loc_face);
311 _facet_ori.push_back(face_ori);
312 }
313 }

385 template<
386 typename Matrix_,
387 typename Operator_,
388 typename TestSpace_,
389 typename TrialSpace_,
390 typename CubatureFactory_>
391 void assemble_operator_matrix2(
392 Matrix_& matrix,
393 Operator_& operat,
394 const TestSpace_& test_space,
395 const TrialSpace_& trial_space,
396 const CubatureFactory_& cubature_factory,
397 typename Matrix_::DataType alpha = typename Matrix_::DataType(1)) const
398 {
399 // validate matrix dimensions
400 XASSERTM(matrix.rows() == test_space.get_num_dofs(), "invalid matrix dimensions");
401 XASSERTM(matrix.columns() == trial_space.get_num_dofs(), "invalid matrix dimensions");
402
403 // matrix type
404 typedef Matrix_ MatrixType;
405 // operator type
406 typedef Operator_ OperatorType;
407 // test-space type
408 typedef TestSpace_ TestSpaceType;
409 // trial-space type
410 typedef TrialSpace_ TrialSpaceType;
411
412 // assembly traits
413 typedef AsmTraits2<
414 typename MatrixType::DataType,
415 TestSpaceType,
416 TrialSpaceType,
417 OperatorType::trafo_config,
418 OperatorType::test_config,
419 OperatorType::trial_config> AsmTraits;
420
421 typedef typename AsmTraits::DataType DataType;
422
423 // shape types
424 typedef typename Shape::FaceTraits<ShapeType, ShapeType::dimension-1>::ShapeType FacetType;
425
426 // fetch the trafo
427 const TrafoType& trafo = test_space.get_trafo();
428
429 // create a trafo evaluator
430 typename AsmTraits::TrafoEvaluator trafo_eval(trafo);
431
432 // create a trafo facet evaluator
433 typedef typename TrafoType::template Evaluator<FacetType, DataType>::Type TrafoFacetEvaluator;
434 TrafoFacetEvaluator trafo_facet_eval(trafo);
435
436 static constexpr TrafoTags trafo_facet_tags = TrafoTags::jac_det | TrafoTags::jac_mat;
437 typedef typename TrafoFacetEvaluator::template ConfigTraits<trafo_facet_tags>::EvalDataType TrafoFacetEvalData;
438
439 // create space evaluators
440 typename AsmTraits::TestEvaluator test_eval(test_space);
441 typename AsmTraits::TrialEvaluator trial_eval(trial_space);
442
443 // create dof-mappings
444 typename AsmTraits::TestDofMapping test_dof_mapping(test_space);
445 typename AsmTraits::TrialDofMapping trial_dof_mapping(trial_space);
446
447 // create a operator evaluator
448 typename OperatorType::template Evaluator<AsmTraits> oper_eval(operat);
449
450 // create trafo evaluation data
451 typename AsmTraits::TrafoEvalData trafo_data;
452 TrafoFacetEvalData trafo_facet_data;
453
454 // create space evaluation data
455 typename AsmTraits::TestEvalData test_data;
456 typename AsmTraits::TrialEvalData trial_data;
457
458 // the value type of the operator
459 typedef typename OperatorType::template Evaluator<AsmTraits>::ValueType ValueType;
460
461 // ensure that the operator and matrix value types are compatible
462 static_assert(std::is_same<ValueType, typename MatrixType::ValueType>::value,
463 "matrix and bilinear operator have different value types!");
464
465 // create local matrix data
466 typename AsmTraits::template TLocalMatrix<ValueType> loc_mat;
467
468 // create cubature rule
469 typename Intern::CubatureTraits<TrafoFacetEvaluator>::RuleType cubature_rule(Cubature::ctor_factory, cubature_factory);
470
471 // create matrix scatter-axpy
472 typename MatrixType::ScatterAxpy scatter_axpy(matrix);
473
474 // trafo matrices and vectors
475 Tiny::Matrix<DataType, shape_dim, facet_dim> face_mat;
476 Tiny::Matrix<DataType, facet_dim, facet_dim> ori_mat;
477 Tiny::Vector<DataType, shape_dim> face_vec;
478 Tiny::Vector<DataType, facet_dim> ori_vec;
479
480 face_mat.format();
481 ori_mat.format();
482 face_vec.format();
483 ori_vec.format();
484
485 // loop over all cells of the mesh
486 for(Index f(0); f < Index(_facets.size()); ++f)
487 {
488 // get facet index
489 const Index face = _facets[f];
490 const Index cell = _cells[f];
491
492 // compute facet trafos
493 Geometry::Intern::FaceRefTrafo<ShapeType, facet_dim>::compute(face_mat, face_vec, _cell_facet[f]);
494
495 // compute orientation trafos
496 Geometry::Intern::CongruencyTrafo<FacetType>::compute(ori_mat, ori_vec, _facet_ori[f]);
497
498 // compute orientation of actual cell facet
499 const int cell_facet_ori = Geometry::Intern::CongruencySampler<FacetType>::orientation(_facet_ori[f])
500 * Shape::ReferenceCell<ShapeType>::facet_orientation(_cell_facet[f]);
501
502 // prepare trafo evaluators
503 trafo_facet_eval.prepare(face);
504 trafo_eval.prepare(cell);
505
506 // prepare space evaluators
507 test_eval.prepare(trafo_eval);
508 trial_eval.prepare(trafo_eval);
509
510 // prepare operator evaluator
511 oper_eval.prepare(trafo_eval);
512
513 // fetch number of local dofs
514 int num_loc_test_dofs = test_eval.get_num_local_dofs();
515 int num_loc_trial_dofs = trial_eval.get_num_local_dofs();
516
517 // format local matrix
518 loc_mat.format();
519
520 // loop over all quadrature points and integrate
521 for(int k(0); k < cubature_rule.get_num_points(); ++k)
522 {
523 // get cubature point
524 auto cub_pt = cubature_rule.get_point(k);
525
526 // transform to local facet
527 auto cub_cf = (face_mat * ((ori_mat * cub_pt) + ori_vec)) + face_vec;
528
529 // compute trafo data
530 trafo_facet_eval(trafo_facet_data, cub_pt);
531 trafo_eval(trafo_data, cub_cf);
532
533 // compute normal vector
534 trafo_data.normal = Tiny::orthogonal(trafo_facet_data.jac_mat).normalize();
535 if(cell_facet_ori < 0)
536 trafo_data.normal.negate();
537
538 // compute basis function data
539 test_eval(test_data, trafo_data);
540 trial_eval(trial_data, trafo_data);
541
542 // prepare bilinear operator
543 oper_eval.set_point(trafo_data);
544
545 // test function loop
546 for(int i(0); i < num_loc_test_dofs; ++i)
547 {
548 // trial function loop
549 for(int j(0); j < num_loc_trial_dofs; ++j)
550 {
551 // evaluate operator and integrate
552 Tiny::axpy(loc_mat(i,j), oper_eval.eval(trial_data.phi[j], test_data.phi[i]),
553 trafo_facet_data.jac_det * cubature_rule.get_weight(k));
554 // continue with next trial function
555 }
556 // continue with next test function
557 }
558 // continue with next cubature point
559 }
560
561 // finish operator evaluator
562 oper_eval.finish();
563
564 // finish evaluators
565 trial_eval.finish();
566 test_eval.finish();
567 trafo_eval.finish();
568 trafo_facet_eval.finish();
569
570 // initialize dof-mappings
571 test_dof_mapping.prepare(cell);
572 trial_dof_mapping.prepare(cell);
573
574 // incorporate local matrix
575 scatter_axpy(loc_mat, test_dof_mapping, trial_dof_mapping, alpha);
576
577 // finish dof mapping
578 trial_dof_mapping.finish();
579 test_dof_mapping.finish();
580
581 // continue with next cell
582 }
583
584 // okay, that's it
585 }

610 template<
611 typename Vector_,
612 typename Functional_,
613 typename CubatureFactory_,
614 typename Space_>
615 void assemble_functional_vector(
616 Vector_& vector,
617 const Functional_& functional,
618 const Space_& space,
619 const CubatureFactory_& cubature_factory,
620 typename Vector_::DataType alpha = typename Vector_::DataType(1)) const
621 {

634 typename VectorType::DataType,
635 SpaceType,
636 FunctionalType::trafo_config,
637 FunctionalType::test_config> AsmTraits;
638
639 typedef typename AsmTraits::DataType DataType;
640
641 // shape types
642 typedef typename Shape::FaceTraits<ShapeType, ShapeType::dimension-1>::ShapeType FacetType;
643
644 // fetch the trafo
645 const TrafoType& trafo = space.get_trafo();
646
647 // create a trafo evaluator
648 typename AsmTraits::TrafoEvaluator trafo_eval(trafo);
649
650 // create a trafo facet evaluator
651 typedef typename TrafoType::template Evaluator<FacetType, DataType>::Type TrafoFacetEvaluator;
652 TrafoFacetEvaluator trafo_facet_eval(trafo);
653
654 typedef typename TrafoFacetEvaluator::template ConfigTraits<TrafoTags::jac_det>::EvalDataType TrafoFacetEvalData;
655
656 // create a space evaluator and evaluation data
657 typename AsmTraits::TestEvaluator test_eval(space);
658
659 // create a dof-mapping
660 typename AsmTraits::DofMapping dof_mapping(space);
661
662 // create a functional evaluator
663 typename FunctionalType::template Evaluator<AsmTraits> func_eval(functional);
664
665 // create trafo evaluation data
666 typename AsmTraits::TrafoEvalData trafo_data;
667 TrafoFacetEvalData trafo_facet_data;
668
669 // create space evaluation data
670 typename AsmTraits::TestEvalData test_data;
671
672 // the value type of the functional
673 typedef typename FunctionalType::template Evaluator<AsmTraits>::ValueType ValueType;
674
675 // ensure that the functional and vector value types are compatible
676 static_assert(std::is_same<ValueType, typename VectorType::ValueType>::value,
677 "vector and linear functional have different value types!");
678
679 // create local vector data
680 typename AsmTraits::template TLocalVector<ValueType> loc_vec;
681
682 // create cubature rule
683 typename Intern::CubatureTraits<TrafoFacetEvaluator>::RuleType cubature_rule(Cubature::ctor_factory, cubature_factory);
684
685 // create matrix scatter-axpy
686 typename VectorType::ScatterAxpy scatter_axpy(vector);
687
688 // trafo matrices and vectors
689 Tiny::Matrix<DataType, shape_dim, facet_dim> face_mat;
690 Tiny::Matrix<DataType, facet_dim, facet_dim> ori_mat;
691 Tiny::Vector<DataType, shape_dim> face_vec;
692 Tiny::Vector<DataType, facet_dim> ori_vec;
693
694 face_mat.format();
695 ori_mat.format();
696 face_vec.format();
697 ori_vec.format();
698
699 // loop over all cells of the mesh
700 for(Index f(0); f < Index(_facets.size()); ++f)
701 {
702 // get facet index
703 const Index face = _facets[f];
704 const Index cell = _cells[f];
705
706 // compute facet trafos
707 Geometry::Intern::FaceRefTrafo<ShapeType, facet_dim>::compute(face_mat, face_vec, _cell_facet[f]);
708
709 // compute orientation trafos
710 Geometry::Intern::CongruencyTrafo<FacetType>::compute(ori_mat, ori_vec, _facet_ori[f]);
711
712 // compute orientation of actual cell facet
713 const int cell_facet_ori = Geometry::Intern::CongruencySampler<FacetType>::orientation(_facet_ori[f])
714 * Shape::ReferenceCell<ShapeType>::facet_orientation(_cell_facet[f]);
715
716 // prepare trafo evaluators
717 trafo_facet_eval.prepare(face);
718 trafo_eval.prepare(cell);
719
720 // prepare test-space evaluator
721 test_eval.prepare(trafo_eval);
722
723 // prepare functional evaluator
724 func_eval.prepare(trafo_eval);
725
726 // fetch number of local dofs
727 int num_loc_dofs = test_eval.get_num_local_dofs();
728
729 // format local matrix
730 loc_vec.format();
731
732 // loop over all quadrature points and integrate
733 for(int k(0); k < cubature_rule.get_num_points(); ++k)
734 {
735 // get cubature point
736 auto cub_pt = cubature_rule.get_point(k);
737
738 // transform to local facet
739 auto cub_cf = (face_mat * ((ori_mat * cub_pt) + ori_vec)) + face_vec;
740
741 // compute trafo data
742 trafo_facet_eval(trafo_facet_data, cub_pt);
743 trafo_eval(trafo_data, cub_cf);
744
745 // compute normal vector
746 trafo_data.normal = Tiny::orthogonal(trafo_facet_data.jac_mat).normalize();
747 if(cell_facet_ori < 0)
748 trafo_data.normal.negate();
749
750 // compute test basis function data
751 test_eval(test_data, trafo_data);
752
753 // prepare functional
754 func_eval.set_point(trafo_data);
755
756 // test function loop
757 for(int i(0); i < num_loc_dofs; ++i)
758 {
759 // evaluate functional and integrate
760 Tiny::axpy(loc_vec(i), func_eval.eval(test_data.phi[i]),
761 trafo_facet_data.jac_det * cubature_rule.get_weight(k));
762 // continue with next trial function
763 }
764 // continue with next test function
765 }
766
767 // finish functional evaluator
768 func_eval.finish();
769
770 // finish evaluators
771 test_eval.finish();
772 trafo_eval.finish();
773 trafo_facet_eval.finish();
774
775 // initialize dof-mapping
776 dof_mapping.prepare(cell);
777
778 // incorporate local matrix
779 scatter_axpy(loc_vec, dof_mapping, alpha);
780
781 // finish dof-mapping
782 dof_mapping.finish();
783
784 // continue with next cell
785 }
786 }
