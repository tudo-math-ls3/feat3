424 // JacobiHelper specialization for PowerFullMatrix
425 template<int n_>
426 struct ExtDiagPFM
427 {
428 static_assert(n_ > 1, "invalid block size");
429 template<typename SMT_, int m_>
430 static void extract_diag(
431 const PowerFullMatrix<SMT_, m_, m_>& matrix,
432 PowerVector<typename SMT_::VectorTypeL, m_>& diag)
433 {
434 ExtDiagPFM<n_-1>::extract_diag(matrix, diag);
435 matrix.template at<n_-1,n_-1>().extract_diag(diag.template at<n_-1>());
436 }
437 };
438
439 template<>
440 struct ExtDiagPFM<1>
441 {
442 template<typename SMT_, int m_>
443 static void extract_diag(
444 const PowerFullMatrix<SMT_, m_, m_>& matrix,
445 PowerVector<typename SMT_::VectorTypeL, m_>& diag)
446 {
447 matrix.template at<0,0>().extract_diag(diag.template at<0>());
448 }
449 };
