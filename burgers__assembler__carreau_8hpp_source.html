141 template<typename Space_, typename CubatureFactory_>
142 void assemble_matrix(
143 LAFEM::SparseMatrixBCSR<DataType_, IndexType_, dim_, dim_>& matrix,
144 const LAFEM::DenseVectorBlocked<DataType_, IndexType_, dim_>& convect,
145 const Space_& space,
146 const CubatureFactory_& cubature_factory,
147 const DataType_ scale = DataType_(1)
148 ) const
149 {
150 // validate matrix and vector dimensions
151 XASSERTM(matrix.rows() == space.get_num_dofs(), "invalid matrix dimensions");
152 XASSERTM(matrix.columns() == space.get_num_dofs(), "invalid matrix dimensions");
153 XASSERTM(convect.size() == space.get_num_dofs(), "invalid vector size");
154
155 typedef LAFEM::DenseVectorBlocked<DataType_, IndexType_, dim_> VectorType;
156 typedef LAFEM::SparseMatrixBCSR<DataType_, IndexType_, dim_, dim_> MatrixType;
157
158 // tolerance: sqrt(eps)
159 const DataType tol_eps = Math::sqrt(Math::eps<DataType>());
160
161 // first of all, let's see what we have to assemble
162
163 const bool need_conv = (Math::abs(beta) > DataType(0));
164 const bool need_conv_frechet = (Math::abs(frechet_beta) > DataType(0));
165 const bool need_reac = (Math::abs(theta) > DataType(0));
166 const bool need_streamdiff = (Math::abs(sd_delta) > DataType(0)) && (sd_v_norm > tol_eps);
167
168 // define our assembly traits
169 typedef AsmTraits1<DataType_, Space_, TrafoTags::jac_det, SpaceTags::value|SpaceTags::grad> AsmTraits;
170
171 // fetch our trafo
172 const typename AsmTraits::TrafoType& trafo = space.get_trafo();
173
174 // create a trafo evaluator
175 typename AsmTraits::TrafoEvaluator trafo_eval(trafo);
176
177 // create a space evaluator and evaluation data
178 typename AsmTraits::SpaceEvaluator space_eval(space);
179
180 // create a dof-mapping
181 typename AsmTraits::DofMapping dof_mapping(space);
182
183 // create trafo evaluation data
184 typename AsmTraits::TrafoEvalData trafo_data;
185
186 // create space evaluation data
187 typename AsmTraits::SpaceEvalData space_data;
188
189 // create cubature rule
190 typename AsmTraits::CubatureRuleType cubature_rule(Cubature::ctor_factory, cubature_factory);
191
192 // create matrix scatter-axpy
193 typename MatrixType::ScatterAxpy scatter_matrix(matrix);
194
195 // create convection gather-axpy
196 typename VectorType::GatherAxpy gather_conv(convect);
197
198 // get maximum number of local dofs
199 static constexpr int max_local_dofs = AsmTraits::max_local_test_dofs;
200
201 // create local matrix data
202 typedef Tiny::Matrix<DataType, dim_, dim_> MatrixValue;
203 typedef Tiny::Matrix<MatrixValue, max_local_dofs, max_local_dofs> LocalMatrixType;
204 LocalMatrixType local_matrix;
205
206 // create local vector data
207 typedef Tiny::Vector<DataType, dim_> VectorValue;
208 typedef Tiny::Vector<VectorValue, max_local_dofs> LocalVectorType;
209
210 // local convection field dofs
211 LocalVectorType local_conv_dofs;
212
213 // our local velocity value
214 VectorValue loc_v, mean_v;
215
216 // our local velocity gradient
217 MatrixValue loc_grad_v;
218 MatrixValue strain_rate_tensor_2;
219
220 loc_v.format();
221 mean_v.format();
222 loc_grad_v.format();
223
224 // compute reference element barycentre
225 Tiny::Vector<DataType, dim_> barycentre;
226 for(int i(0); i < dim_; ++i)
227 barycentre[i] = Shape::ReferenceCell<typename Space_::ShapeType>::template centre<DataType>(i);
228
229 // our local streamline diffusion coefficients
230 Tiny::Vector<DataType, max_local_dofs> streamdiff_coeffs;
231 streamdiff_coeffs.format();
232
233 // our local delta for streamline diffusion
234 DataType local_delta = DataType(0);
235
236 // loop over all cells of the mesh
237 for(typename AsmTraits::CellIterator cell(trafo_eval.begin()); cell != trafo_eval.end(); ++cell)
238 {
239 // prepare trafo evaluator
240 trafo_eval.prepare(cell);
241
242 // prepare space evaluator
243 space_eval.prepare(trafo_eval);
244
245 // initialize dof-mapping
246 dof_mapping.prepare(cell);
247
248 // fetch number of local dofs
249 const int num_loc_dofs = space_eval.get_num_local_dofs();
250
251 // gather our local convection dofs
252 local_conv_dofs.format();
253 gather_conv(local_conv_dofs, dof_mapping);
254
255 // compute mesh width if necessary
256 if(need_streamdiff)
257 {
258 // reset local delta
259 local_delta = DataType(0);
260
261 // evaluate trafo and space at barycentre
262 trafo_eval(trafo_data, barycentre);
263 space_eval(space_data, trafo_data);
264
265 // compute velocity at barycentre
266 mean_v.format();
267 for(int i(0); i < num_loc_dofs; ++i)
268 mean_v.axpy(space_data.phi[i].value, local_conv_dofs[i]);
269
270 // compute norm of mean velocity
271 const DataType local_norm_v = mean_v.norm_euclid();
272
273 // do we have a non-zero velocity?
274 if(local_norm_v > tol_eps)
275 {
276 // compute local mesh width w.r.t. mean velocity
277 const DataType local_h = trafo_eval.width_directed(mean_v) * local_norm_v;
278
279 // compute local Re_T
280 const DataType local_re = (local_norm_v * local_h) / this->sd_nu;
281
282 // compute local delta
283 local_delta = this->sd_delta * (local_h / this->sd_v_norm) * (DataType(2)*local_re) / (DataType(1) + local_re);
284 }
285 }
286
287 // format our local matrix and vector
288 local_matrix.format();
289
290 // loop over all quadrature points and integrate
291 for(int point(0); point < cubature_rule.get_num_points(); ++point)
292 {
293 // compute trafo data
294 trafo_eval(trafo_data, cubature_rule.get_point(point));
295
296 // compute basis function data
297 space_eval(space_data, trafo_data);
298
299 // pre-compute cubature weight
300 const DataType weight = trafo_data.jac_det * cubature_rule.get_weight(point);
301
302 // evaluate convection function and its gradient (if required)
303 if(need_conv || need_streamdiff)
304 {
305 loc_v.format();
306 for(int i(0); i < num_loc_dofs; ++i)
307 {
308 // update velocity value
309 loc_v.axpy(space_data.phi[i].value, local_conv_dofs[i]);
310 }
311 }
312 // always needed for material model
313 loc_grad_v.format();
314 for(int i(0); i < num_loc_dofs; ++i)
315 {
316 // update velocity gradient
317 loc_grad_v.add_outer_product(local_conv_dofs[i], space_data.phi[i].grad);
318 }
319
320
321 // evaluate streamline diffusion coefficients
322 if(need_streamdiff)
323 {
324 for(int i(0); i < num_loc_dofs; ++i)
325 {
326 streamdiff_coeffs[i] = Tiny::dot(loc_v, space_data.phi[i].grad);
327 }
328 }
329
330 //calculate nu_loc
331
332 strain_rate_tensor_2.set_transpose(loc_grad_v);
333 strain_rate_tensor_2.axpy(DataType(1.0), loc_grad_v);
334 DataType_ gamma_dot = Math::sqrt(0.5)*strain_rate_tensor_2.norm_frobenius();
335 DataType_ nu_loc = mu_inf + (mu_0 - mu_inf) * Math::pow((DataType_(1.0) + Math::pow(lambda*gamma_dot, a)), (n-DataType(1.0)) / a);
336
337
338 // assemble diffusion matrix?
339 if(diff && !deformation)
340 {
341 // assemble gradient-tensor diffusion
342
343 // test function loop
344 for(int i(0); i < num_loc_dofs; ++i)
345 {
346 // trial function loop
347 for(int j(0); j < num_loc_dofs; ++j)
348 {
349 // compute scalar value
350 const DataType value = nu_loc * weight * Tiny::dot(space_data.phi[i].grad, space_data.phi[j].grad);
351
352 // update local matrix
353 local_matrix[i][j].add_scalar_main_diag(value);
354 }
355 }
356 }
357 else if(diff && deformation)
358 {
359
360 // assemble deformation-tensor diffusion
361
362 // test function loop
363 for(int i(0); i < num_loc_dofs; ++i)
364 {
365 // trial function loop
366 for(int j(0); j < num_loc_dofs; ++j)
367 {
368 // compute inner product of grad(phi) and grad(psi)
369 const DataType value = nu_loc * weight * Tiny::dot(space_data.phi[j].grad, space_data.phi[i].grad);
370
371 // update local matrix
372 local_matrix[i][j].add_scalar_main_diag(value);
373
374 // add outer product of grad(phi) and grad(psi)
375 local_matrix[i][j].add_outer_product(space_data.phi[j].grad, space_data.phi[i].grad, nu_loc * weight);
376 }
377 }
378 }
379
380 if(frechet_diff)
381 {
382 XASSERT(deformation);
383 const DataType_ mu_prime = (mu_0 - mu_inf) * ((n-DataType_(1.0)) / a) *
384 Math::pow( DataType(1.0) + Math::pow(lambda*gamma_dot, a), ((n-DataType_(1.0) -a) / a)) *
385 a * Math::pow(lambda*gamma_dot, a-DataType(1.0));
386 const DataType fac = mu_prime / (gamma_dot + reg_eps);
387 //std::cout << fac ;
388 // test function loop
389 for(int i(0); i < num_loc_dofs; ++i)
390 {
391 VectorValue du_grad_phi;
392 du_grad_phi.set_mat_vec_mult(strain_rate_tensor_2, space_data.phi[i].grad);
393
394 // trial function loop
395 for(int j(0); j < num_loc_dofs; ++j)
396 {
397 VectorValue du_grad_psi;
398 du_grad_psi.set_mat_vec_mult(strain_rate_tensor_2, space_data.phi[j].grad);
399 // add outer product of grad(phi) and grad(psi)
400 local_matrix[i][j].add_outer_product(du_grad_phi, du_grad_psi, fac * weight);
401 }
402 }
403 }
404
405
406
407
408 // assemble convection?
409 if(need_conv)
410 {
411 // test function loop
412 for(int i(0); i < num_loc_dofs; ++i)
413 {
414 // trial function loop
415 for(int j(0); j < num_loc_dofs; ++j)
416 {
417 // compute scalar value
418 const DataType value = beta * weight * space_data.phi[i].value * Tiny::dot(loc_v, space_data.phi[j].grad);
419
420 // update local matrix
421 local_matrix[i][j].add_scalar_main_diag(value);
422 }
423 }
424 }
425
426 // assemble convection Frechet?
427 if(need_conv_frechet)
428 {
429
430 // test function loop
431 for(int i(0); i < num_loc_dofs; ++i)
432 {
433 // trial function loop
434 for(int j(0); j < num_loc_dofs; ++j)
435 {
436 // compute scalar value
437 const DataType value = frechet_beta * weight * space_data.phi[i].value * space_data.phi[j].value;
438
439 // update local matrix
440 local_matrix[i][j].axpy(value, loc_grad_v);
441 }
442 }
443 }
444
445 // assemble reaction?
446 if(need_reac)
447 {
448 // test function loop
449 for(int i(0); i < num_loc_dofs; ++i)
450 {
451 // trial function loop
452 for(int j(0); j < num_loc_dofs; ++j)
453 {
454 // compute scalar value
455 const DataType value = theta * weight * space_data.phi[i].value * space_data.phi[j].value;
456
457 // update local matrix
458 local_matrix[i][j].add_scalar_main_diag(value);
459 }
460 }
461 }
462
463 // assemble streamline diffusion?
464 if(need_streamdiff && (local_delta > tol_eps))
465 {
466 // test function loop
467 for(int i(0); i < num_loc_dofs; ++i)
468 {
469 // trial function loop
470 for(int j(0); j < num_loc_dofs; ++j)
471 {
472 // compute scalar value
473 const DataType value = local_delta * weight * streamdiff_coeffs[i] * streamdiff_coeffs[j];
474
475 // update local matrix
476 local_matrix[i][j].add_scalar_main_diag(value);
477 }
478 }
479 }
480
481 // continue with next cubature point
482 }
483
484 // scatter into matrix
485 scatter_matrix(local_matrix, dof_mapping, dof_mapping, scale);
486
487 // finish dof mapping
488 dof_mapping.finish();
489
490 // finish evaluators
491 space_eval.finish();
492 trafo_eval.finish();
493 }
494 }

517 template<typename Space_, typename CubatureFactory_>
518 void assemble_vector(

532
533
534
535 // first of all, let's see what we have to assemble

537 //const bool need_conv_frechet = (frechet_beta > DataType(0));
538 const bool need_reac = (Math::abs(theta) > DataType(0));
539

557

559 typename AsmTraits::SpaceEvalData space_data;
560
561 // create cubature rule
562 typename AsmTraits::CubatureRuleType cubature_rule(Cubature::ctor_factory, cubature_factory);
563
564 // create vector-scatter-axpy (if needed)
565 typename VectorType::ScatterAxpy scatter_vector(vector);
566
567 // create convection gather-axpy
568 typename VectorType::GatherAxpy gather_conv(convect);

572

574 static constexpr int max_local_dofs = AsmTraits::max_local_test_dofs;
575

579 LocalVectorType local_vector;
580
581 // local convection field dofs
582 LocalVectorType local_conv_dofs;
583
584 // local primal vector dofs
585 LocalVectorType local_prim_dofs;
586
587 // our local velocity value
588 Tiny::Vector<DataType, dim_> loc_v;
589
590 // our local velocity gradient
591 Tiny::Matrix<DataType, dim_, dim_> loc_grad_v;
592
593 Tiny::Matrix<DataType, dim_, dim_> strain_rate_tensor_2;
594
595 loc_v.format();
596 loc_grad_v.format();
597
598 // loop over all cells of the mesh
599 for(typename AsmTraits::CellIterator cell(trafo_eval.begin()); cell != trafo_eval.end(); ++cell)
600 {
601 // prepare trafo evaluator
602 trafo_eval.prepare(cell);
603
604 // prepare space evaluator
605 space_eval.prepare(trafo_eval);
606
607 // initialize dof-mapping
608 dof_mapping.prepare(cell);
609
610 // fetch number of local dofs
611 const int num_loc_dofs = space_eval.get_num_local_dofs();

613 // gather our local convection dofs
614 local_conv_dofs.format();
615 gather_conv(local_conv_dofs, dof_mapping);
616
617 // gather our local primal dofs
618 local_prim_dofs.format();
619 gather_prim(local_prim_dofs, dof_mapping);
620
621 // format our local vector
622 local_vector.format();
623
624 // loop over all quadrature points and integrate
625 for(int point(0); point < cubature_rule.get_num_points(); ++point)
626 {
627 // compute trafo data
628 trafo_eval(trafo_data, cubature_rule.get_point(point));
629
630 // compute basis function data
631 space_eval(space_data, trafo_data);
632
633 // pre-compute cubature weight
634 const DataType weight = trafo_data.jac_det * cubature_rule.get_weight(point);
635
636 // evaluate convection function and its gradient (if required)
637 if(need_conv)
638 {
639 loc_v.format();
640 for(int i(0); i < num_loc_dofs; ++i)
641 {
642 // update velocity value
643 loc_v.axpy(space_data.phi[i].value, local_conv_dofs[i]);
644 }
645 }
646 if(diff || frechet_diff)
647 {
648
649 loc_grad_v.format();
650 for(int i(0); i < num_loc_dofs; ++i)
651 {
652 // update velocity gradient
653 loc_grad_v.add_outer_product(local_conv_dofs[i], space_data.phi[i].grad);
654 }
655
656 }
657
658 loc_grad_v.format();
659 for(int i(0); i < num_loc_dofs; ++i)
660 {
661 // update velocity gradient
662 loc_grad_v.add_outer_product(local_conv_dofs[i], space_data.phi[i].grad);
663 }
664
665 //calculate nu_loc
666 strain_rate_tensor_2.set_transpose(loc_grad_v);
667 strain_rate_tensor_2.axpy(DataType(1.0), loc_grad_v);
668 DataType_ gamma_dot = Math::sqrt(0.5)*strain_rate_tensor_2.norm_frobenius();
669 DataType_ nu_loc = mu_inf + (mu_0 - mu_inf) * Math::pow((DataType_(1.0) + Math::pow(lambda*gamma_dot, a)), (n-DataType(1.0)) / a);

671
672 // assemble diffusion matrix?
673 if(diff && !deformation)
674 {

678 for(int i(0); i < num_loc_dofs; ++i)
679 {
680 // trial function loop
681 for(int j(0); j < num_loc_dofs; ++j)
682 {
683 // compute scalar value
684 const DataType value = nu_loc * weight * Tiny::dot(space_data.phi[i].grad, space_data.phi[j].grad);
685
686 // update local vector
687 local_vector[i].axpy(value, local_prim_dofs[j]);
688 }
689 }
690 }
691 else if(diff && deformation)
692 {
693 // assemble deformation-tensor diffusion
694
695 // test function loop
696 for(int i(0); i < num_loc_dofs; ++i)
697 {
698 // trial function loop
699 for(int j(0); j < num_loc_dofs; ++j)
700 {
701 // compute inner product of grad(phi) and grad(psi)
702 const DataType value1 = nu_loc * weight * Tiny::dot(space_data.phi[i].grad, space_data.phi[j].grad);
703
704 // compute outer product of grad(phi) and grad(psi)
705 const DataType value2 = nu_loc * weight * Tiny::dot(local_prim_dofs[j], space_data.phi[i].grad);
706
707 // update local vector
708 local_vector[i].axpy(value1, local_prim_dofs[j]);
709 local_vector[i].axpy(value2, space_data.phi[j].grad);
710 }
711 }
712 }
713
714 // assemble convection?
715 if(need_conv)
716 {
717 // test function loop
718 for(int i(0); i < num_loc_dofs; ++i)
719 {
720 // trial function loop
721 for(int j(0); j < num_loc_dofs; ++j)
722 {
723 // compute scalar value
724 const DataType value = beta * weight * space_data.phi[i].value * Tiny::dot(loc_v, space_data.phi[j].grad);
725
726 // update local vector
727 local_vector[i].axpy(value, local_prim_dofs[j]);
728 }
729 }
730 }
731
732 // assemble reaction?
733 if(need_reac)
734 {
735 // test function loop
736 for(int i(0); i < num_loc_dofs; ++i)
737 {
738 // trial function loop
739 for(int j(0); j < num_loc_dofs; ++j)
740 {
741 // compute scalar value
742 const DataType value = theta * weight * space_data.phi[i].value * space_data.phi[j].value;
743
744 // update local vector
745 local_vector[i].axpy(value, local_prim_dofs[j]);
746 }
747 }
748 }
749
750 // continue with next cubature point
751 }
752
753 // scatter into vector
754 scatter_vector(local_vector, dof_mapping, scale);
755
756 // finish dof mapping
757 dof_mapping.finish();
758
759 // finish evaluators
760 space_eval.finish();
761 trafo_eval.finish();
762 }

771 void set_sd_v_norm(const LAFEM::DenseVectorBlocked<DataType_, IndexType_, dim_>& convect)
772 {
773 const auto* vals = convect.elements();
774 DataType_ r = DataType(0);
775 for(Index i(0); i < convect.size(); ++i)
776 r = Math::max(r, vals[i].norm_euclid());
777 this->sd_v_norm = r;
778 }

789 template<typename LocalVector_, typename Mirror_>
790 void set_sd_v_norm(const Global::Vector<LocalVector_, Mirror_>& convect)

793 const auto* gate = convect.get_gate();
794 if(gate != nullptr)
795 this->sd_v_norm = gate->max(this->sd_v_norm);
