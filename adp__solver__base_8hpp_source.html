185
186 // assemble the algebraic dof partitioning

188
189 // upload the matrix structure
190 _adp_matrix = ADPMatrixType(&this->_alg_dof_parti);
191 _adp_matrix.upload_symbolic(this->_system_matrix);
192
193 // create two vectors

195 _adp_vec_cor = ADPVectorType(&this->_alg_dof_parti);

200 _adp_vec_cor.owned().format();
201 }

215

218 }

222 _adp_vec_cor.clear();
223 _adp_vec_def.clear();
224 _adp_matrix.clear();

227 BaseClass::done_symbolic();
