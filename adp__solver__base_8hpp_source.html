673 template<typename DTV_, typename RPT_, typename CIT_>
674 void _filter_mat(DTV_* val, const RPT_* row_ptr, const CIT_* col_idx,
675 const LAFEM::UnitFilter<DataType, IndexType>& filter)
676 {
677 const IndexType n = filter.used_elements();
678 const IndexType* fil_idx = filter.get_indices();
679 for(IndexType i = 0; i < n; ++i)
680 {
681 const Index row = fil_idx[i];
682 for(RPT_ j = row_ptr[row]; j < row_ptr[row+1]; ++j)
683 val[j] = DTV_(Index(col_idx[j]) == row ? 1 : 0);
684 }
685 }
