163 virtual Status apply(GlobalVectorType& vec_cor, const GlobalVectorType& vec_def) override
164 {
165 Statistics::add_solver_expression(std::make_shared<ExpressionStartSolve>(this->name()));
166
167 // apply local solver
168 Statistics::add_solver_expression(std::make_shared<ExpressionCallPrecond>(this->name(), this->_local_solver->name()));
169 Status status = _local_solver->apply(vec_cor.local(), vec_def.local());
170
171 // ignore or synchronize status codes?
172 if(this->_ignore_status)
173 status = Status::success;
174 else
175 {
176 // synchronize local status over communicator to obtain
177 // a consistent status code on all processes
178 const Dist::Comm* comm = vec_cor.get_comm();
179 if(comm != nullptr)
180 {
181 // local status: 0: success, 1: failure
182 int lstat = (status_success(status) ? 0 : 1);
183 int gstat = -1;
184
185 // sum up over all processes:
186 comm->allreduce(&lstat, &gstat, std::size_t(1), Dist::op_sum);
187 XASSERT(gstat >= 0);
188
189 // if all processes succeeded, the sum will also be 0
190 // if the sum is > 0, then at least one process failed,
191 // so the global status will also be failure
192 status = (gstat == 0 ? Status::success : Status::aborted);
193 }
194 }
195
196 // did all processes succeed?
197 if(status == Status::success)
198 {
199 // synchronize correction vector
200 vec_cor.sync_1();
201
202 // apply filter
203 _filter.filter_cor(vec_cor);
204 }
205
206 // okay
207 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), status, 0));
208 return status;
209 }
