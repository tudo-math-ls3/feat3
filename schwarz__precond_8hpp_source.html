125 virtual Status apply(GlobalVectorType& vec_cor, const GlobalVectorType& vec_def) override
126 {
127 Statistics::add_solver_expression(std::make_shared<ExpressionStartSolve>(this->name()));
128
129 // apply local solver
130 Statistics::add_solver_expression(std::make_shared<ExpressionCallPrecond>(this->name(), this->_local_solver->name()));
131 Status status = _local_solver->apply(vec_cor.local(), vec_def.local());
132
133 // synchronize local status over communicator to obtain
134 // a consistent status code on all processes
135 const Dist::Comm* comm = vec_cor.get_comm();
136 if(comm != nullptr)
137 {
138 // local status: 0: success, 1: failure
139 int lstat = (status_success(status) ? 0 : 1);
140 int gstat = -1;
141
142 // sum up over all processes:
143 comm->allreduce(&lstat, &gstat, std::size_t(1), Dist::op_sum);
144 XASSERT(gstat >= 0);
145
146 // if all processes succeeded, the sum will also be 0
147 // if the sum is > 0, then at least one process failed,
148 // so the global status will also be failure
149 status = (gstat == 0 ? Status::success : Status::aborted);
150 }
151
152 // did all processes succeed?
153 if(status == Status::success)
154 {
155 // synchronize correction vector
156 vec_cor.sync_1();
157
158 // apply filter
159 _filter.filter_cor(vec_cor);
160 }
161
162 // okay
163 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), status, 0));
164 return status;
165 }
