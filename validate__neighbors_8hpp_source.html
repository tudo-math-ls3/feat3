35 template<typename MeshType>
36 void validate_neighbors(const MeshType& mesh)
37 {
38 static constexpr int facet_dim = MeshType::shape_dim-1;
39 const auto& neigh = mesh.get_neighbors();
40 const auto& facet_idx = mesh.template get_index_set<MeshType::shape_dim, facet_dim>();
41
42 Geometry::BoundaryFactory<MeshType> boundary_factory(mesh);
43 Geometry::MeshPart<MeshType> boundary(boundary_factory);
44
45 Index num_facets(mesh.get_num_entities(MeshType::shape_dim-1));
46
47 bool* at_boundary(new bool[num_facets]);
48
49 for(Index l(0); l < num_facets; ++l)
50 at_boundary[l] = false;
51
52 // Mark all boundary facets
53 const auto& facet_trg = boundary.template get_target_set<MeshType::shape_dim-1>();
54 for(Index l(0); l < facet_trg.get_num_entities(); ++l)
55 {
56 at_boundary[facet_trg[l]] = true;
57 }
58
59 // Check the mesh
60 for(Index k(0); k < mesh.get_num_entities(MeshType::shape_dim); ++k)
61 {
62 for(int j(0); j < facet_idx.num_indices; ++j)
63 {
64 Index other_cell(neigh(k,j));
65 // If we have a neighbor...
66 if(other_cell != ~Index(0))
67 {
68 bool ok(false);
69 // ... then this should NOT be the boundary
70 if(at_boundary[facet_idx(k,j)])
71 throw String("Facet "+stringify(facet_idx(k,j))+
72 " is at the boundary, but the _neighborinformation says it is not!");
73
74 // Check for vice versa neighbor relationship
75 for(int i(0); i < facet_idx.num_indices; ++i)
76 {
77 if(neigh(other_cell,i) == k)
78 {
79 ok = true;
80 break;
81 }
82 }
83 if(!ok)
84 throw String("Cell "+stringify(k)+" has neighbor "+stringify(other_cell)+" but not vice versa!");
85 }
86 else
87 {
89 if(!at_boundary[facet_idx(k,j)])
90 throw String("Facet "+stringify(facet_idx(k,j))+
91 " is not at the boundary, but the _neighbor information says it is!");
92 }
93 }
94
95 }
96
97 // Clean up
98 delete[] at_boundary;
99
100 }
