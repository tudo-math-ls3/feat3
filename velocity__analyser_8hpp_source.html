109 void synchronize(const Dist::Comm& comm)
110 {
111 DataType_ verr[2*dim_+4] =
112 {
113 Math::sqr(norm_h0),
114 Math::sqr(norm_h1),
115 Math::sqr(divergence),
116 Math::sqr(vorticity),
117 };
118 for(int i(0); i < dim_; ++i)
119 {
120 verr[4+ i] = Math::sqr(norm_h0_comp[i]);
121 verr[4+dim_+i] = Math::sqr(norm_h1_comp[i]);
122 }
123
124 comm.allreduce(verr, verr, std::size_t(2*dim_+4), Dist::op_sum);
125
126 norm_h0 = Math::sqrt(verr[0]);
127 norm_h1 = Math::sqrt(verr[1]);
128 divergence = Math::sqrt(verr[2]);
129 vorticity = Math::sqrt(verr[3]);
130 for(int i(0); i < dim_; ++i)
131 {
132 norm_h0_comp[i] = Math::sqrt(verr[4+ i]);
133 norm_h1_comp[i] = Math::sqrt(verr[4+dim_+i]);
134 }
135 }

203 template<typename DT_, int sm_, int sn_>
204 static DT_ calc_vorticity(const Tiny::Matrix<DT_, 2, 2, sm_, sn_>& jac)
205 {
206 // 2D: (dx u2 - dy u1)^2
207 // J_ij = d/d_j f_i ==> (J_01 - J_10)^2
208 return Math::sqr(jac[1][0] - jac[0][1]);
209 }
210
211 template<typename DT_, int sm_, int sn_>
212 static DT_ calc_vorticity(const Tiny::Matrix<DT_, 3, 3, sm_, sn_>& jac)
213 {
214 // 3D: (dy u3 - dz u2)^2 + (dz u1 - dx u3)^2 + (dx u2 - dy u1)
215 return
216 Math::sqr(jac[2][1] - jac[1][2]) + // dy u3 - dz u2 = J_21 - J_12
217 Math::sqr(jac[0][2] - jac[2][0]) + // dz u1 - dx u3 = J_02 - J_20
218 Math::sqr(jac[1][0] - jac[0][1]); // dx u2 - dy u1 = J_10 - J_01
219 }

262 template<typename DataType_, typename IndexType_, int dim_, typename Space_>
263 static VelocityInfo<DataType_, dim_> compute(
264 const LAFEM::PowerVector<LAFEM::DenseVector<DataType_, IndexType_>, dim_>& vector,
265 const Space_& space, const Cubature::DynamicFactory& cubature_factory)
266 {
267 // first of all, verify the dimensions
268 static_assert(Space_::shape_dim == dim_, "invalid velocity field dimension");
269
271 typedef Space_ SpaceType;
272
273 typedef LAFEM::PowerVector<LAFEM::DenseVector<DataType_, IndexType_>, dim_> VectorType;
274
276 typedef AsmTraits1<DataType_, SpaceType, TrafoTags::jac_det, SpaceTags::value|SpaceTags::grad> AsmTraits;
277
279 typedef typename AsmTraits::DataType DataType;
280
281 // create the cubature rule
282 typename AsmTraits::CubatureRuleType cubature_rule(Cubature::ctor_factory, cubature_factory);
283
284 // fetch the trafo
285 const typename AsmTraits::TrafoType& trafo = space.get_trafo();
286
287 // create a trafo evaluator
288 typename AsmTraits::TrafoEvaluator trafo_eval(trafo);
289
290 // create a space evaluator and evaluation data
291 typename AsmTraits::SpaceEvaluator space_eval(space);
292 typedef typename AsmTraits::SpaceEvaluator SpaceEvaluator;
293
294 // create a dof-mapping
295 typename AsmTraits::DofMapping dof_mapping(space);
296
297 // create trafo evaluation data
298 typename AsmTraits::TrafoEvalData trafo_data;
299
300 // create space evaluation data
301 typename AsmTraits::SpaceEvalData space_data;
302
303 // create matrix scatter-axpy
304 Intern::MultiGather<VectorType> multi_gather(vector);
305
306 // initialize result
307 VelocityInfo<DataType_, dim_> info;
308
309 // local velocity values
310 Tiny::Matrix<DataType_, dim_, SpaceEvaluator::max_local_dofs> basis_val;
311
312 // vector field values
313 Tiny::Vector<DataType_, dim_> val;
314
315 // vector field derivatives
316 Tiny::Matrix<DataType_, dim_, dim_> jac;
317
318 // loop over all cells of the mesh
319 for(typename AsmTraits::CellIterator cell(trafo_eval.begin()); cell != trafo_eval.end(); ++cell)
320 {
321 // initialize dof-mapping
322 dof_mapping.prepare(cell);
323
324 // fetch local basis values
325 basis_val.format();
326 multi_gather.gather(basis_val, dof_mapping);
327
328 // finish dof-mapping
329 dof_mapping.finish();
330
331 // prepare trafo evaluator
332 trafo_eval.prepare(cell);
333
334 // prepare space evaluator
335 space_eval.prepare(trafo_eval);
336
337 // fetch number of local dofs
338 int num_loc_dofs = space_eval.get_num_local_dofs();
339
340 // loop over all quadrature points and integrate
341 for(int k(0); k < cubature_rule.get_num_points(); ++k)
342 {
343 // compute trafo data
344 trafo_eval(trafo_data, cubature_rule.get_point(k));
345
346 // compute basis function data
347 space_eval(space_data, trafo_data);
348
349 // compute basis function values and derivatives
350 val.format();
351 jac.format();
352 for(int i(0); i < num_loc_dofs; ++i)
353 {
354 for(int bi(0); bi < dim_; ++bi)
355 {
356 val[bi] += basis_val[bi][i] * space_data.phi[i].value;
357 // J_ij = d/d_j f_i <==> J = (f_1,...,f_n)^T * (d/d_1, ..., d/d_n)
358 for(int bj(0); bj < dim_; ++bj)
359 {
360 // J_ij = d/d_j f_i
361 jac[bi][bj] += basis_val[bi][i] * space_data.phi[i].grad[bj];
362 }
363 }
364 }
365
366 // compute integration weight
367 DataType omega = trafo_data.jac_det * cubature_rule.get_weight(k);
368
369 // update info
370 for(int i(0); i < dim_; ++i)
371 {
372 // update H0 component norm
373 info.norm_h0_comp[i] += omega * Math::sqr(val[i]);
374
375 // update H1 component norm
376 info.norm_h1_comp[i] += omega * Tiny::dot(jac[i], jac[i]);
377 }
378
379 // update divergence
380 info.divergence += omega * Math::sqr(jac.trace());
381
382 // update vorticity
383 info.vorticity += omega * calc_vorticity(jac);
384
385 // continue with next cubature point
386 }
387
388 // finish evaluators
389 space_eval.finish();
390 trafo_eval.finish();
391
392 // continue with next cell
393 }
394
395 // finally, compute the rest
396 for(int i(0); i < dim_; ++i)
397 {
398 info.norm_h0 += info.norm_h0_comp[i];
399 info.norm_h1 += info.norm_h1_comp[i];
400 info.norm_h0_comp[i] = Math::sqrt(info.norm_h0_comp[i]);
401 info.norm_h1_comp[i] = Math::sqrt(info.norm_h1_comp[i]);
402 }
403
404 // take the roots
405 info.norm_h0 = Math::sqrt(info.norm_h0);
406 info.norm_h1 = Math::sqrt(info.norm_h1);
407 info.divergence = Math::sqrt(info.divergence);
408 info.vorticity = Math::sqrt(info.vorticity);
409
410 // okay
411 return info;
412 }

453 template<typename DataType_, typename IndexType_, int dim_, typename Space_>
454 static VelocityInfo<DataType_, dim_> compute(
455 const LAFEM::DenseVectorBlocked<DataType_, IndexType_, dim_>& vector,
456 const Space_& space, const Cubature::DynamicFactory& cubature_factory)
457 {
458 // first of all, verify the dimensions
459 static_assert(Space_::shape_dim == dim_, "invalid velocity field dimension");
460
461 // validate vector dimensions
462 XASSERTM(vector.size() == space.get_num_dofs(), "invalid vector size");
463
465 typedef Space_ SpaceType;
466
467 typedef LAFEM::DenseVectorBlocked<DataType_, IndexType_, dim_> VectorType;
468
470 typedef AsmTraits1<DataType_, SpaceType, TrafoTags::jac_det, SpaceTags::value|SpaceTags::grad> AsmTraits;
471
473 typedef typename AsmTraits::DataType DataType;
474
475 // create the cubature rule
476 typename AsmTraits::CubatureRuleType cubature_rule(Cubature::ctor_factory, cubature_factory);
477
478 // fetch the trafo
479 const typename AsmTraits::TrafoType& trafo = space.get_trafo();
480
481 // create a trafo evaluator
482 typename AsmTraits::TrafoEvaluator trafo_eval(trafo);
483
484 // create a space evaluator and evaluation data
485 typename AsmTraits::SpaceEvaluator space_eval(space);
486 typedef typename AsmTraits::SpaceEvaluator SpaceEvaluator;
487
488 // create a dof-mapping
489 typename AsmTraits::DofMapping dof_mapping(space);
490
491 // create trafo evaluation data
492 typename AsmTraits::TrafoEvalData trafo_data;
493
494 // create space evaluation data
495 typename AsmTraits::SpaceEvalData space_data;
496
497 // create matrix scatter-axpy
498 typename VectorType::GatherAxpy gather(vector);
499
500 // initialize result
501 VelocityInfo<DataType_, dim_> info;
502
503 // local velocity values
504 Tiny::Vector<Tiny::Vector<DataType, dim_>, SpaceEvaluator::max_local_dofs> basis_val;
505
506 // vector field value
507 Tiny::Vector<DataType_, dim_> val;
508
509 // vector field first derivatives as jacobian matrix
510 Tiny::Matrix<DataType_, dim_, dim_> jac;
511
512 // loop over all cells of the mesh
513 for(typename AsmTraits::CellIterator cell(trafo_eval.begin()); cell != trafo_eval.end(); ++cell)
514 {
515 // initialize dof-mapping
516 dof_mapping.prepare(cell);
517
518 // fetch local basis values
519 basis_val.format();
520 gather(basis_val, dof_mapping);
521
522 // finish dof-mapping
523 dof_mapping.finish();
524
525 // prepare trafo evaluator
526 trafo_eval.prepare(cell);
527
528 // prepare space evaluator
529 space_eval.prepare(trafo_eval);
530
531 // fetch number of local dofs
532 const int num_loc_dofs = space_eval.get_num_local_dofs();
533
534 // loop over all quadrature points and integrate
535 for(int k(0); k < cubature_rule.get_num_points(); ++k)
536 {
537 // compute trafo data
538 trafo_eval(trafo_data, cubature_rule.get_point(k));
539
540 // compute basis function data
541 space_eval(space_data, trafo_data);
542
543 // compute basis function values and derivatives
544 val.format();
545 jac.format();
546 for(int i(0); i < num_loc_dofs; ++i)
547 {
548 val.axpy(space_data.phi[i].value, basis_val[i]);
549 // J_ij = d/d_j f_i <==> J = (f_1,...,f_n)^T * (d/d_1, ..., d/d_n)
550 jac.add_outer_product(basis_val[i], space_data.phi[i].grad);
551 }
552
553 // compute integration weight
554 const DataType omega = trafo_data.jac_det * cubature_rule.get_weight(k);
555
556 // update norms
557 for(int i(0); i < dim_; ++i)
558 {
559 // update H0 component norm
560 info.norm_h0_comp[i] += omega * Math::sqr(val[i]);
561
562 // update H1 component norm
563 info.norm_h1_comp[i] += omega * Tiny::dot(jac[i], jac[i]);
564 }
565
566 // update divergence
567 info.divergence += omega * Math::sqr(jac.trace());
568
569 // update vorticity
570 info.vorticity += omega * calc_vorticity(jac);
571
572 // continue with next cubature point
573 }
574
575 // finish evaluators
576 space_eval.finish();
577 trafo_eval.finish();
578
579 // continue with next cell
580 }
581
582 // finally, compute the rest
583 for(int i(0); i < dim_; ++i)
584 {
585 info.norm_h0 += info.norm_h0_comp[i];
586 info.norm_h1 += info.norm_h1_comp[i];
587 info.norm_h0_comp[i] = Math::sqrt(info.norm_h0_comp[i]);
588 info.norm_h1_comp[i] = Math::sqrt(info.norm_h1_comp[i]);
589 }
590
591 // take the roots
592 info.norm_h0 = Math::sqrt(info.norm_h0);
593 info.norm_h1 = Math::sqrt(info.norm_h1);
594 info.divergence = Math::sqrt(info.divergence);
595 info.vorticity = Math::sqrt(info.vorticity);
596
597 // okay
598 return info;
599 }
