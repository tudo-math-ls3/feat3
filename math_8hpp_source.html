296 template<typename T_>
297 inline T_ sqrt(T_ x)
298 {
299 static_assert(Type::Traits<T_>::is_float, "sqrt can only be applied to floating point types");
300
301 if(x <= T_(0))
302 return T_(0);
303
304 // use Newton iteration: y_{k+1} := y_k/2 * (3 - (y_k^2)/x)
305 // we choose y_0 = min(1,x); this ensures that the sequence y_k is monotonically increasing
306 // if y_{k+1} is not greater than y_k, we return y_k
307 const T_ z = T_(1) / x;
308 T_ y(Math::min(T_(1), x));
309 T_ yn(y);
310 do
311 {
312 y = yn;
313 yn = T_(0.5)*y * (T_(3) - (y*y*z));
314 } while(yn > y);
315
316 return y;
317 }
318#ifdef FEAT_COMPILER_MICROSOFT
319#pragma warning(pop)
320#endif

333 template<typename T_>
334 inline T_ sin(T_ x)
335 {
336 static_assert(Type::Traits<T_>::is_float, "sin can only be applied to floating point types");
337
338 // use the exponential sum formula:
339 // infty x^(2*n+1)
340 // sin(x) := sum (-1)^n -----------
341 // n=0 (2*n+1)!
342 T_ y(x), yl(x+T_(1)), z(x);
343 T_ fn(1.0);
344 int n(1);
345 do
346 {
347 // update 1/(2*n+1)!
348 fn /= T_(++n);
349 fn /= T_(++n);
350 yl = y;
351 y += T_(1 - int(n&2)) * (z *= x*x) * T_(fn);
352 } while(yl != y);
353
354 return y;
355 }

511 template<typename T_>
512 inline T_ exp(T_ x)
513 {
514 static_assert(Type::Traits<T_>::is_float, "exp can only be applied to floating point types");
515
516 T_ y(T_(1)), yl(T_(0)), z(y);
517 int n(0);
518 do
519 {
520 yl = y;
521 y += ((z *= x) /= T_(++n));
522 // Note about the stopping criterion:
523 // For x > 0, the sequence y_k must be strictly increasing.
524 // For x < 0, the sequence y_k must be alternating.
525 // And encode this into the most beautiful crypto-expression C++ has to offer ^_^
526 } while(x > T_(0) ? yl < y : n & 1 ? y < yl : yl < y);
527 return yl;
528 }

541 template<typename T_>
542 inline T_ log(T_ x)
543 {
544 static_assert(Type::Traits<T_>::is_float, "log can only be applied to floating point types");
545
546 // use Newton iteration: y_{k+1} = y_k + 2*(x - exp(y_k))/(x + exp(y_k))
547 T_ y(T_(0)), yl(T_(0));
548 do
549 {
550 yl = y;
551 T_ ey(Math::exp(y));
552 y += T_(2) * (x - ey) / (x + ey);
553 // Note about the stopping criterion:
554 // For x > 1, the sequence y_k must be strictly increasing.
555 // For x < 1, the sequence y_k must be strictly decreasing.
556 // Again, encode this into one beautiful expression
557 } while(x < T_(1) ? y < yl : yl < y);
558 return yl;
559 }

611 template<typename T_>
612 inline T_ atan(T_ x)
613 {
614 static_assert(Type::Traits<T_>::is_float, "atan can only be applied to floating point types");
615
616 // the exponential sum converges only for |x| < 1, but we can reduce any |x| >= 1 by
617 // atan(x) = 2*atan( x / (1 + sqrt(1 + x^2)))
618 int k(0);
619 for(; Math::abs(x) >= T_(1); ++k)
620 x /= (T_(1) + Math::sqrt(T_(1) + x*x));
621
622 // use the exponential sum formula:
623 // infty x^(2*n+1)
624 // atan(x) := sum (-1)^n -----------
625 // n=0 (2*n+1)
626 T_ y(x), yl(x+T_(1)), z(x);
627 int n(1);
628 do
629 {
630 yl = y;
631 T_ t((z *= x*x) / T_(n += 2));
632 y += T_(1 - int(n&2)) * t;
633 } while(yl != y);
634
635 return T_(1<<k) * y;
636 }

666 template<typename T_>
667 inline T_ pi()
668 {
669 static_assert(Type::Traits<T_>::is_float, "pi can only be applied to floating point types");
670
671 // use the Bailey-Borwein-Plouffe formula:
672 // infty 1 ( 4 2 1 1 )
673 // pi := sum ---- * ( ---- - ---- - ---- - ---- )
674 // k=0 16^k ( 8k+1 8k+4 8k+5 8k+6 )
675 T_ y(T_(0)), yl(T_(0));
676 int k(0);
677 const T_ z(T_(1) / T_(16));
678 T_ t(T_(1));
679 do
680 {
681 yl = y;
682 y += t * (T_(4)/T_(8*k+1) - T_(2)/T_(8*k+4) - T_(1)/T_(8*k+5) - T_(1)/T_(8*k+6));
683 t *= z;
684 ++k;
685 } while(yl != y);
686
687 return y;
688 }

1073 inline T_ binomial(T_ n, T_ k)
1074 {
1075 static_assert(Type::Traits<T_>::is_int, "binomial can only be applied to integral types");
1076

1078 {
1079 return T_(0); // by definition
1080 }
1081 else if((k <= T_(0)) || (k == n))
1082 {
1083 return T_(1); // by definition
1084 }
1085
1086 // exploit symmetry: (n \choose k) = (n \choose n-k)
1087 k = min(k, n-k);
1088
1089 // use multiplicative formula: (n \choose k+1) = (n - k) * (n \choose k) / (k + 1)
1090 T_ m = n;
1091 for(T_ i(1); i < k; ++i)
1092 {
1093 m *= (n - i);
1094 m /= (i + 1);
1095 }
1096

1168 template<typename DT_, typename IT_>
1169 DT_ invert_matrix(const IT_ n, const IT_ stride, DT_ a[], IT_ p[])
1170 {
1171 // make sure that the parameters are valid
1172 if((n <= IT_(0)) || (stride < n) || (a == nullptr) || (p == nullptr))
1173 return DT_(0);
1174
1175 // invert 1x1 explicitly
1176 if(n == IT_(1))
1177 {
1178 DT_ det = a[0];
1179 a[0] = DT_(1) / det;
1180 return det;
1181 }
1182
1183 // initialize identity permutation
1184 for(IT_ i(0); i < n; ++i)
1185 {
1186 p[i] = i;
1187 }
1188
1189 // initialize determinant to 1
1190 DT_ det = DT_(1);
1191
1192 // primary column elimination loop
1193 for(IT_ k(0); k < n; ++k)
1194 {
1195 // step 1: find a pivot for the elimination of column k
1196 {
1197 // for this, we only check the rows p[j] with j >= k, as all
1198 // rows p[j] with j < k have already been eliminated and are
1199 // therefore not candidates for pivoting
1200 DT_ pivot = Math::abs(a[p[k]*stride + p[k]]);
1201 IT_ i = k;
1202
1203 // loop over all unprocessed rows
1204 for(IT_ j(k+1); j < n; ++j)
1205 {
1206 // get our matrix value and check whether it can be a pivot
1207 DT_ abs_val = Math::abs(a[p[j]*stride + p[j]]);
1208 if(abs_val > pivot)
1209 {
1210 pivot = abs_val;
1211 i = j;
1212 }
1213 }
1214
1215 // do we have to swap rows i and k?
1216 if(i > k)
1217 {
1218 // swap rows "virtually" by exchanging their permutation positions
1219 IT_ t = p[k];
1220 p[k] = p[i];
1221 p[i] = t;
1222 }
1223 }
1224
1225 // compute pivot row offset
1226 const IT_ pk_off = p[k]*stride;
1227
1228 // step 2: process pivot row
1229 {
1230 // update determinant by multiplying with the pivot element
1231 det *= a[pk_off + p[k]];
1232
1233 // get our inverted pivot element
1234 const DT_ pivot = DT_(1) / a[pk_off + p[k]];
1235
1236 // replace column entry by unit column entry
1237 a[pk_off + p[k]] = DT_(1);
1238
1239 // divide the whole row by the inverted pivot
1240 for(IT_ j(0); j < n; ++j)
1241 {
1242 a[pk_off+j] *= pivot;
1243 }
1244 }
1245
1246 // step 3: eliminate pivot column
1247
1248 // loop over all rows of the matrix
1249 for(IT_ i(0); i < n; ++i)
1250 {
1251 // skip the pivot row
1252 if(i == p[k])
1253 continue;
1254
1255 // compute row and pivot offsets
1256 const IT_ row_off = i*stride;
1257
1258 // fetch elimination factor
1259 const DT_ factor = a[row_off + p[k]];
1260
1261 // replace by unit column entry
1262 a[row_off + p[k]] = DT_(0);
1263
1264 // process the row
1265 for(IT_ j(0); j < n; ++j)
1266 {
1267 a[row_off + j] -= a[pk_off + j] * factor;
1268 }
1269 }
1270 }
1271
1272 // return determinant
1273 return det;
1274 }
