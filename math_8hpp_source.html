299 template<typename T_>
300 inline T_ sqrt(T_ x)
301 {
302 static_assert(Type::Traits<T_>::is_float, "sqrt can only be applied to floating point types");
303
304 if(x <= T_(0))
305 return T_(0);
306
307 // use Newton iteration: y_{k+1} := y_k/2 * (3 - (y_k^2)/x)
308 // we choose y_0 = min(1,x); this ensures that the sequence y_k is monotonically increasing
309 // if y_{k+1} is not greater than y_k, we return y_k
310 const T_ z = T_(1) / x;
311 T_ y(Math::min(T_(1), x));
312 T_ yn(y);
313 do
314 {
315 y = yn;
316 yn = T_(0.5)*y * (T_(3) - (y*y*z));
317 } while(yn > y);
318
319 return y;
320 }
321#ifdef FEAT_COMPILER_MICROSOFT
322#pragma warning(pop)
323#endif

343 template<typename T_>
344 inline T_ sin(T_ x)
345 {
346 static_assert(Type::Traits<T_>::is_float, "sin can only be applied to floating point types");
347
348 // use the exponential sum formula:
349 // infty x^(2*n+1)
350 // sin(x) := sum (-1)^n -----------
351 // n=0 (2*n+1)!
352 T_ y(x), yl(x+T_(1)), z(x);
353 T_ fn(1.0);
354 int n(1);
355 do
356 {
357 // update 1/(2*n+1)!
358 fn /= T_(++n);
359 fn /= T_(++n);
360 yl = y;
361 y += T_(1 - int(n&2)) * (z *= x*x) * T_(fn);
362 } while(yl != y);
363
364 return y;
365 }

549 template<typename T_>
550 inline T_ exp(T_ x)
551 {
552 static_assert(Type::Traits<T_>::is_float, "exp can only be applied to floating point types");
553
554 T_ y(T_(1)), yl(T_(0)), z(y);
555 int n(0);
556 do
557 {
558 yl = y;
559 y += ((z *= x) /= T_(++n));
560 // Note about the stopping criterion:
561 // For x > 0, the sequence y_k must be strictly increasing.
562 // For x < 0, the sequence y_k must be alternating.
563 // And encode this into the most beautiful crypto-expression C++ has to offer ^_^
564 } while(x > T_(0) ? yl < y : n & 1 ? y < yl : yl < y);
565 return yl;
566 }

579 template<typename T_>
580 inline T_ log(T_ x)
581 {
582 static_assert(Type::Traits<T_>::is_float, "log can only be applied to floating point types");
583
584 // use Newton iteration: y_{k+1} = y_k + 2*(x - exp(y_k))/(x + exp(y_k))
585 T_ y(T_(0)), yl(T_(0));
586 do
587 {
588 yl = y;
589 T_ ey(Math::exp(y));
590 y += T_(2) * (x - ey) / (x + ey);
591 // Note about the stopping criterion:
592 // For x > 1, the sequence y_k must be strictly increasing.
593 // For x < 1, the sequence y_k must be strictly decreasing.
594 // Again, encode this into one beautiful expression
595 } while(x < T_(1) ? y < yl : yl < y);
596 return yl;
597 }

661 template<typename T_>
662 inline T_ atan(T_ x)
663 {
664 static_assert(Type::Traits<T_>::is_float, "atan can only be applied to floating point types");
665
666 // the exponential sum converges only for |x| < 1, but we can reduce any |x| >= 1 by
667 // atan(x) = 2*atan( x / (1 + sqrt(1 + x^2)))
668 int k(0);
669 for(; Math::abs(x) >= T_(1); ++k)
670 x /= (T_(1) + Math::sqrt(T_(1) + x*x));
671
672 // use the exponential sum formula:
673 // infty x^(2*n+1)
674 // atan(x) := sum (-1)^n -----------
675 // n=0 (2*n+1)
676 T_ y(x), yl(x+T_(1)), z(x);
677 int n(1);
678 do
679 {
680 yl = y;
681 T_ t((z *= x*x) / T_(n += 2));
682 y += T_(1 - int(n&2)) * t;
683 } while(yl != y);
684
685 return T_(1<<k) * y;
686 }

723 template<typename T_>
724 inline T_ pi()
725 {
726 static_assert(Type::Traits<T_>::is_float, "pi can only be applied to floating point types");
727
728 // use the Bailey-Borwein-Plouffe formula:
729 // infty 1 ( 4 2 1 1 )
730 // pi := sum ---- * ( ---- - ---- - ---- - ---- )
731 // k=0 16^k ( 8k+1 8k+4 8k+5 8k+6 )
732 T_ y(T_(0)), yl(T_(0));
733 int k(0);
734 const T_ z(T_(1) / T_(16));
735 T_ t(T_(1));
736 do
737 {
738 yl = y;
739 y += t * (T_(4)/T_(8*k+1) - T_(2)/T_(8*k+4) - T_(1)/T_(8*k+5) - T_(1)/T_(8*k+6));
740 t *= z;
741 ++k;
742 } while(yl != y);
743
744 return y;
745 }

1187 template<typename T_>
1188 inline T_ binomial(T_ n, T_ k)
1189 {
1190 static_assert(Type::Traits<T_>::is_int, "binomial can only be applied to integral types");
1191
1192 if(k > n)
1193 {
1194 return T_(0); // by definition
1195 }
1196 else if((k <= T_(0)) || (k == n))
1197 {
1198 return T_(1); // by definition
1199 }
1200
1201 // exploit symmetry: (n \choose k) = (n \choose n-k)
1202 k = min(k, n-k);
1203
1204 // use multiplicative formula: (n \choose k+1) = (n - k) * (n \choose k) / (k + 1)
1205 T_ m = n;
1206 for(T_ i(1); i < k; ++i)
1207 {
1208 m *= (n - i);
1209 m /= (i + 1);
1210 }
1211
1212 return m;
1213 }

1283 template<typename DT_, typename IT_>
1284 DT_ invert_matrix(const IT_ n, const IT_ stride, DT_ a[], IT_ p[])
1285 {
1286 // make sure that the parameters are valid
1287 if((n <= IT_(0)) || (stride < n) || (a == nullptr) || (p == nullptr))
1288 return DT_(0);
1289
1290 // invert 1x1 explicitly
1291 if(n == IT_(1))
1292 {
1293 DT_ det = a[0];
1294 a[0] = DT_(1) / det;
1295 return det;
1296 }
1297
1298 // initialize identity permutation
1299 for(IT_ i(0); i < n; ++i)
1300 {
1301 p[i] = i;
1302 }
1303
1304 // initialize determinant to 1
1305 DT_ det = DT_(1);
1306
1307 // primary column elimination loop
1308 for(IT_ k(0); k < n; ++k)
1309 {
1310 // step 1: find a pivot for the elimination of column k
1311 {
1312 // for this, we only check the rows p[j] with j >= k, as all
1313 // rows p[j] with j < k have already been eliminated and are
1314 // therefore not candidates for pivoting
1315 DT_ pivot = Math::abs(a[p[k]*stride + p[k]]);
1316 IT_ i = k;
1317
1318 // loop over all unprocessed rows
1319 for(IT_ j(k+1); j < n; ++j)
1320 {
1321 // get our matrix value and check whether it can be a pivot
1322 DT_ abs_val = Math::abs(a[p[j]*stride + p[j]]);
1323 if(abs_val > pivot)
1324 {
1325 pivot = abs_val;
1326 i = j;
1327 }
1328 }
1329
1330 // do we have to swap rows i and k?
1331 if(i > k)
1332 {
1333 // swap rows "virtually" by exchanging their permutation positions
1334 IT_ t = p[k];
1335 p[k] = p[i];
1336 p[i] = t;
1337 }
1338 }
1339
1340 // compute pivot row offset
1341 const IT_ pk_off = p[k]*stride;
1342
1343 // step 2: process pivot row
1344 {
1345 // update determinant by multiplying with the pivot element
1346 det *= a[pk_off + p[k]];
1347
1348 // get our inverted pivot element
1349 const DT_ pivot = DT_(1) / a[pk_off + p[k]];
1350
1351 // replace column entry by unit column entry
1352 a[pk_off + p[k]] = DT_(1);
1353
1354 // divide the whole row by the inverted pivot
1355 for(IT_ j(0); j < n; ++j)
1356 {
1357 a[pk_off+j] *= pivot;
1358 }
1359 }
1360
1361 // step 3: eliminate pivot column
1362
1363 // loop over all rows of the matrix
1364 for(IT_ i(0); i < n; ++i)
1365 {
1366 // skip the pivot row
1367 if(i == p[k])
1368 continue;
1369
1370 // compute row and pivot offsets
1371 const IT_ row_off = i*stride;
1372
1373 // fetch elimination factor
1374 const DT_ factor = a[row_off + p[k]];
1375
1376 // replace by unit column entry
1377 a[row_off + p[k]] = DT_(0);
1378
1379 // process the row
1380 for(IT_ j(0); j < n; ++j)
1381 {
1382 a[row_off + j] -= a[pk_off + j] * factor;
1383 }
1384 }
1385 }
1386
1387 // return determinant
1388 return det;
1389 }

1405 template<typename DT_>
1406 DT_ calc_opening_angle_intern(DT_ cross_prod, DT_ dot_prod)
1407 {
1408 DT_ theta;
1409 // We always want to use the version that uses values nearer to 1 for better conditioning
1410 if(Math::abs(cross_prod) < DT_(0.5))
1411 {
1412 theta = Math::asin(cross_prod);
1413 // Transform to [0,2pi]
1414 if(theta >= DT_(0))
1415 theta = (dot_prod >= DT_(0)) ? theta : (Math::pi<DT_>() - theta);
1416 else
1417 theta = (dot_prod >= DT_(0)) ? (DT_(2) * Math::pi<DT_>() + theta) : (Math::pi<DT_>() - theta);
1418 }
1419 else
1420 {
1421 theta = Math::acos(dot_prod);
1422 //Transform to [0,2pi]
1423 theta = (cross_prod >= DT_(0)) ? theta : (DT_(2) * Math::pi<DT_>() - theta);
1424 }
1425 return theta;
