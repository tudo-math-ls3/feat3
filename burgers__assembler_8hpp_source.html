125 template<typename Space_, typename CubatureFactory_>
126 void assemble_matrix(
127 LAFEM::SparseMatrixBCSR<DataType_, IndexType_, dim_, dim_>& matrix,
128 const LAFEM::DenseVectorBlocked<DataType_, IndexType_, dim_>& convect,
129 const Space_& space,
130 const CubatureFactory_& cubature_factory,
131 const DataType_ scale = DataType_(1)
132 ) const
133 {
134 // validate matrix and vector dimensions
135 XASSERTM(matrix.rows() == space.get_num_dofs(), "invalid matrix dimensions");
136 XASSERTM(matrix.columns() == space.get_num_dofs(), "invalid matrix dimensions");
137 XASSERTM(convect.size() == space.get_num_dofs(), "invalid vector size");
138
139 typedef LAFEM::DenseVectorBlocked<DataType_, IndexType_, dim_> VectorType;
140 typedef LAFEM::SparseMatrixBCSR<DataType_, IndexType_, dim_, dim_> MatrixType;
141
142 // tolerance: sqrt(eps)
143 const DataType tol_eps = Math::sqrt(Math::eps<DataType>());
144
145 // first of all, let's see what we have to assemble
146 const bool need_diff = (Math::abs(nu) > DataType(0));
147 const bool need_conv = (Math::abs(beta) > DataType(0));
148 const bool need_conv_frechet = (Math::abs(frechet_beta) > DataType(0));
149 const bool need_reac = (Math::abs(theta) > DataType(0));
150 const bool need_streamdiff = (Math::abs(sd_delta) > DataType(0)) && (sd_v_norm > tol_eps);
151
152 // define our assembly traits
153 typedef AsmTraits1<DataType_, Space_, TrafoTags::jac_det, SpaceTags::value|SpaceTags::grad> AsmTraits;
154
155 // fetch our trafo
156 const typename AsmTraits::TrafoType& trafo = space.get_trafo();
157
158 // create a trafo evaluator
159 typename AsmTraits::TrafoEvaluator trafo_eval(trafo);
160
161 // create a space evaluator and evaluation data
162 typename AsmTraits::SpaceEvaluator space_eval(space);
163
164 // create a dof-mapping
165 typename AsmTraits::DofMapping dof_mapping(space);
166
167 // create trafo evaluation data
168 typename AsmTraits::TrafoEvalData trafo_data;
169
170 // create space evaluation data
171 typename AsmTraits::SpaceEvalData space_data;
172
173 // create cubature rule
174 typename AsmTraits::CubatureRuleType cubature_rule(Cubature::ctor_factory, cubature_factory);
175
176 // create matrix scatter-axpy
177 typename MatrixType::ScatterAxpy scatter_matrix(matrix);
178
179 // create convection gather-axpy
180 typename VectorType::GatherAxpy gather_conv(convect);
181
182 // get maximum number of local dofs
183 static constexpr int max_local_dofs = AsmTraits::max_local_test_dofs;
184
185 // create local matrix data
186 typedef Tiny::Matrix<DataType, dim_, dim_> MatrixValue;
187 typedef Tiny::Matrix<MatrixValue, max_local_dofs, max_local_dofs> LocalMatrixType;
188 LocalMatrixType local_matrix;
189
190 // create local vector data
191 typedef Tiny::Vector<DataType, dim_> VectorValue;
192 typedef Tiny::Vector<VectorValue, max_local_dofs> LocalVectorType;
193
194 // local convection field dofs
195 LocalVectorType local_conv_dofs;
196
197 // our local velocity value
198 Tiny::Vector<DataType, dim_> loc_v, mean_v;
199
200 // our local velocity gradient
201 Tiny::Matrix<DataType, dim_, dim_> loc_grad_v;
202
203 loc_v.format();
204 mean_v.format();
205 loc_grad_v.format();
206
207 // compute reference element barycentre
208 Tiny::Vector<DataType, dim_> barycentre;
209 for(int i(0); i < dim_; ++i)
210 barycentre[i] = Shape::ReferenceCell<typename Space_::ShapeType>::template centre<DataType>(i);
211
212 // our local streamline diffusion coefficients
213 Tiny::Vector<DataType, max_local_dofs> streamdiff_coeffs;
214 streamdiff_coeffs.format();
215
216 // our local delta for streamline diffusion
217 DataType local_delta = DataType(0);
218
219 // loop over all cells of the mesh
220 for(typename AsmTraits::CellIterator cell(trafo_eval.begin()); cell != trafo_eval.end(); ++cell)
221 {
222 // prepare trafo evaluator
223 trafo_eval.prepare(cell);
224
225 // prepare space evaluator
226 space_eval.prepare(trafo_eval);
227
228 // initialize dof-mapping
229 dof_mapping.prepare(cell);
230
231 // fetch number of local dofs
232 const int num_loc_dofs = space_eval.get_num_local_dofs();
233
234 // gather our local convection dofs
235 local_conv_dofs.format();
236 gather_conv(local_conv_dofs, dof_mapping);
237
238 // compute mesh width if necessary
239 if(need_streamdiff)
240 {
241 // reset local delta
242 local_delta = DataType(0);
243
244 // evaluate trafo and space at barycentre
245 trafo_eval(trafo_data, barycentre);
246 space_eval(space_data, trafo_data);
247
248 // compute velocity at barycentre
249 mean_v.format();
250 for(int i(0); i < num_loc_dofs; ++i)
251 mean_v.axpy(space_data.phi[i].value, local_conv_dofs[i]);
252
253 // compute norm of mean velocity
254 const DataType local_norm_v = mean_v.norm_euclid();
255
256 // do we have a non-zero velocity?
257 if(local_norm_v > tol_eps)
258 {
259 // compute local mesh width w.r.t. mean velocity
260 const DataType local_h = trafo_eval.width_directed(mean_v) * local_norm_v;
261
262 // compute local Re_T
263 const DataType local_re = (local_norm_v * local_h) / this->sd_nu;
264
265 // compute local delta
266 local_delta = this->sd_delta * (local_h / this->sd_v_norm) * (DataType(2)*local_re) / (DataType(1) + local_re);
267 }
268 }
269
270 // format our local matrix and vector
271 local_matrix.format();
272
273 // loop over all quadrature points and integrate
274 for(int point(0); point < cubature_rule.get_num_points(); ++point)
275 {
276 // compute trafo data
277 trafo_eval(trafo_data, cubature_rule.get_point(point));
278
279 // compute basis function data
280 space_eval(space_data, trafo_data);
281
282 // pre-compute cubature weight
283 const DataType weight = trafo_data.jac_det * cubature_rule.get_weight(point);
284
285 // evaluate convection function and its gradient (if required)
286 if(need_conv || need_streamdiff)
287 {
288 loc_v.format();
289 for(int i(0); i < num_loc_dofs; ++i)
290 {
291 // update velocity value
292 loc_v.axpy(space_data.phi[i].value, local_conv_dofs[i]);
293 }
294 }
295 if(need_conv_frechet)
296 {
297 loc_grad_v.format();
298 for(int i(0); i < num_loc_dofs; ++i)
299 {
300 // update velocity gradient
301 loc_grad_v.add_outer_product(local_conv_dofs[i], space_data.phi[i].grad);
302 }
303 }
304
305 // evaluate streamline diffusion coefficients
306 if(need_streamdiff)
307 {
308 for(int i(0); i < num_loc_dofs; ++i)
309 {
310 streamdiff_coeffs[i] = Tiny::dot(loc_v, space_data.phi[i].grad);
311 }
312 }
313
314 // assemble diffusion matrix?
315 if(need_diff && !deformation)
316 {
317 // assemble gradient-tensor diffusion
318
319 // test function loop
320 for(int i(0); i < num_loc_dofs; ++i)
321 {
322 // trial function loop
323 for(int j(0); j < num_loc_dofs; ++j)
324 {
325 // compute scalar value
326 const DataType value = nu * weight * Tiny::dot(space_data.phi[i].grad, space_data.phi[j].grad);
327
328 // update local matrix
329 local_matrix[i][j].add_scalar_main_diag(value);
330 }
331 }
332 }
333 else if(need_diff && deformation)
334 {
335 // assemble deformation-tensor diffusion
336
337 // test function loop
338 for(int i(0); i < num_loc_dofs; ++i)
339 {
340 // trial function loop
341 for(int j(0); j < num_loc_dofs; ++j)
342 {
343 // compute inner product of grad(phi) and grad(psi)
344 const DataType value = nu * weight * Tiny::dot(space_data.phi[j].grad, space_data.phi[i].grad);
345
346 // update local matrix
347 local_matrix[i][j].add_scalar_main_diag(value);
348
349 // add outer product of grad(phi) and grad(psi)
350 local_matrix[i][j].add_outer_product(space_data.phi[j].grad, space_data.phi[i].grad, nu * weight);
351 }
352 }
353 }
354
355 // assemble convection?
356 if(need_conv)
357 {
358 // test function loop
359 for(int i(0); i < num_loc_dofs; ++i)
360 {
361 // trial function loop
362 for(int j(0); j < num_loc_dofs; ++j)
363 {
364 // compute scalar value
365 const DataType value = beta * weight * space_data.phi[i].value * Tiny::dot(loc_v, space_data.phi[j].grad);
366
367 // update local matrix
368 local_matrix[i][j].add_scalar_main_diag(value);
369 }
370 }
371 }
372
373 // assemble convection Frechet?
374 if(need_conv_frechet)
375 {
376 // test function loop
377 for(int i(0); i < num_loc_dofs; ++i)
378 {
379 // trial function loop
380 for(int j(0); j < num_loc_dofs; ++j)
381 {
382 // compute scalar value
383 const DataType value = frechet_beta * weight * space_data.phi[i].value * space_data.phi[j].value;
384
385 // update local matrix
386 local_matrix[i][j].axpy(value, loc_grad_v);
387 }
388 }
389 }
390
391 // assemble reaction?
392 if(need_reac)
393 {
394 // test function loop
395 for(int i(0); i < num_loc_dofs; ++i)
396 {
397 // trial function loop
398 for(int j(0); j < num_loc_dofs; ++j)
399 {
400 // compute scalar value
401 const DataType value = theta * weight * space_data.phi[i].value * space_data.phi[j].value;
402
403 // update local matrix
404 local_matrix[i][j].add_scalar_main_diag(value);
405 }
406 }
407 }
408
409 // assemble streamline diffusion?
410 if(need_streamdiff && (local_delta > tol_eps))
411 {
412 // test function loop
413 for(int i(0); i < num_loc_dofs; ++i)
414 {
415 // trial function loop
416 for(int j(0); j < num_loc_dofs; ++j)
417 {
418 // compute scalar value
419 const DataType value = local_delta * weight * streamdiff_coeffs[i] * streamdiff_coeffs[j];
420
421 // update local matrix
422 local_matrix[i][j].add_scalar_main_diag(value);
423 }
424 }
425 }
426
427 // continue with next cubature point
428 }
429
430 // scatter into matrix
431 scatter_matrix(local_matrix, dof_mapping, dof_mapping, scale);
432
433 // finish dof mapping
434 dof_mapping.finish();
435
436 // finish evaluators
437 space_eval.finish();
438 trafo_eval.finish();
439 }
440 }

551 // loop over all cells of the mesh
552 for(typename AsmTraits::CellIterator cell(trafo_eval.begin()); cell != trafo_eval.end(); ++cell)
553 {
554 // prepare trafo evaluator
555 trafo_eval.prepare(cell);
556
557 // prepare space evaluator

564 const int num_loc_dofs = space_eval.get_num_local_dofs();
565
566 // gather our local convection dofs
567 local_conv_dofs.format();
568 gather_conv(local_conv_dofs, dof_mapping);
569
570 // compute mesh width if necessary
571 if(need_streamdiff)
572 {
573 // reset local delta
574 local_delta = DataType(0);
575
576 // evaluate trafo and space at barycentre
577 trafo_eval(trafo_data, barycentre);

584

586 const DataType local_norm_v = mean_v.norm_euclid();
587
588 // do we have a non-zero velocity?
589 if(local_norm_v > tol_eps)
590 {
591 // compute local mesh width w.r.t. mean velocity
592 const DataType local_h = trafo_eval.width_directed(mean_v) * local_norm_v;
593

595 const DataType local_re = (local_norm_v * local_h) / this->sd_nu;
596
597 // compute local delta
598 local_delta = this->sd_delta * (local_h / this->sd_v_norm) * (DataType(2)*local_re) / (DataType(1) + local_re);
599 }
600 }
601
602 // format our local matrix and vector
603 local_matrix.format();
604
605 // loop over all quadrature points and integrate
606 for(int point(0); point < cubature_rule.get_num_points(); ++point)
607 {
608 // compute trafo data
609 trafo_eval(trafo_data, cubature_rule.get_point(point));
610
611 // compute basis function data

613

689 }
690 }
691 }
692
693 // continue with next cubature point
694 }
695
696 // scatter into matrix
697 scatter_matrix(local_matrix, dof_mapping, dof_mapping, scale);
698
699 // finish dof mapping
700 dof_mapping.finish();
701
702 // finish evaluators
703 space_eval.finish();

774
775 // create vector-scatter-axpy (if needed)
776 typename VectorType::ScatterAxpy scatter_vector(vector);
777
778 // create convection gather-axpy

794

797

829
830 // format our local vector
831 local_vector.format();
832
833 // loop over all quadrature points and integrate
834 for(int point(0); point < cubature_rule.get_num_points(); ++point)
835 {
836 // compute trafo data
837 trafo_eval(trafo_data, cubature_rule.get_point(point));
838
839 // compute basis function data
840 space_eval(space_data, trafo_data);
841
842 // pre-compute cubature weight
843 const DataType weight = trafo_data.jac_det * cubature_rule.get_weight(point);
844
845 // evaluate convection function and its gradient (if required)
846 if(need_conv)
847 {
848 loc_v.format();
849 for(int i(0); i < num_loc_dofs; ++i)
850 {
851 // update velocity value
852 loc_v.axpy(space_data.phi[i].value, local_conv_dofs[i]);
853 }
854 }
855 /*if(need_conv_frechet)
856 {
857 loc_grad_v.format();
858 for(int i(0); i < num_loc_dofs; ++i)
859 {
860 // update velocity gradient
861 loc_grad_v.add_outer_product(local_conv_dofs[i], space_data.phi[i].grad);
862 }
863 }*/
864
865 // assemble diffusion matrix?
866 if(need_diff && !deformation)
867 {
868 // assemble gradient-tensor diffusion
869
870 // test function loop
871 for(int i(0); i < num_loc_dofs; ++i)
872 {
873 // trial function loop
874 for(int j(0); j < num_loc_dofs; ++j)
875 {
876 // compute scalar value
877 const DataType value = nu * weight * Tiny::dot(space_data.phi[i].grad, space_data.phi[j].grad);
878
879 // update local vector
880 local_vector[i].axpy(value, local_prim_dofs[j]);
881 }
882 }
883 }
884 else if(need_diff && deformation)
885 {
886 // assemble deformation-tensor diffusion
887
888 // test function loop
889 for(int i(0); i < num_loc_dofs; ++i)
890 {
891 // trial function loop
892 for(int j(0); j < num_loc_dofs; ++j)
893 {
894 // compute inner product of grad(phi) and grad(psi)
895 const DataType value1 = nu * weight * Tiny::dot(space_data.phi[i].grad, space_data.phi[j].grad);
896
897 // compute outer product of grad(phi) and grad(psi)
898 const DataType value2 = nu * weight * Tiny::dot(local_prim_dofs[j], space_data.phi[i].grad);
899
900 // update local vector
901 local_vector[i].axpy(value1, local_prim_dofs[j]);
902 local_vector[i].axpy(value2, space_data.phi[j].grad);
903 }
904 }
905 }
906
907 // assemble convection?
908 if(need_conv)
909 {
910 // test function loop
911 for(int i(0); i < num_loc_dofs; ++i)
912 {
913 // trial function loop
914 for(int j(0); j < num_loc_dofs; ++j)
915 {
916 // compute scalar value
917 const DataType value = beta * weight * space_data.phi[i].value * Tiny::dot(loc_v, space_data.phi[j].grad);
918
919 // update local vector
920 local_vector[i].axpy(value, local_prim_dofs[j]);
921 }
922 }
923 }
924
925 // assemble reaction?
926 if(need_reac)
927 {
928 // test function loop
929 for(int i(0); i < num_loc_dofs; ++i)
930 {
931 // trial function loop
932 for(int j(0); j < num_loc_dofs; ++j)
933 {
934 // compute scalar value
935 const DataType value = theta * weight * space_data.phi[i].value * space_data.phi[j].value;
936
937 // update local vector
938 local_vector[i].axpy(value, local_prim_dofs[j]);
939 }
940 }
941 }
942
943 // continue with next cubature point
944 }
945
946 // scatter into vector
947 scatter_vector(local_vector, dof_mapping, scale);
948
949 // finish dof mapping
950 dof_mapping.finish();
951
952 // finish evaluators
953 space_eval.finish();
954 trafo_eval.finish();
955 }
956 }
