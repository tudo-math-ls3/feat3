217 filename.erase(found);

221 {
222 directory = filename.substr(0, found + 1);
223 filename.erase(0, found + 1);

226 file << "%%MatrixMarket powerrowmatrix coordinate real general" << std::endl;
227 for (Index i(1); i <= blocks_; ++i)
228 {
229 file << filename << "_pr" << i << suffix << std::endl;

231

526
528 void restore_from_checkpoint_data(std::vector<char> & data)
529 {
530 std::uint64_t isize = *(std::uint64_t*) data.data(); //get size of checkpointed _first
531 std::vector<char>::iterator start = std::begin(data) + sizeof(std::uint64_t);
532 std::vector<char>::iterator last_of_first = std::begin(data) + sizeof(std::uint64_t) + (int) isize;
533 std::vector<char> buffer_first(start, last_of_first);
534 _first.restore_from_checkpoint_data(buffer_first);
535
536 data.erase(std::begin(data), last_of_first);
537 _rest.restore_from_checkpoint_data(data);
538 }
539
541 std::uint64_t set_checkpoint_data(std::vector<char>& data, SerialConfig& config)
542 {
543 std::size_t old_size = data.size();
544 data.insert(std::end(data), sizeof(std::uint64_t), 0); //add placeholder
545 std::uint64_t ireal_size = _first.set_checkpoint_data(data, config); //add data of _first to the overall checkpoint and save its size
546 char* csize = reinterpret_cast<char*>(&ireal_size);
547 for(std::size_t i(0) ; i < sizeof(std::uint64_t) ; ++i) //overwrite the guessed datalength
548 {
549 data[old_size +i] = csize[i];
550 }
551
552 return sizeof(std::uint64_t) + ireal_size + _rest.set_checkpoint_data(data, config); //generate and add checkpoint data for the _rest
553 }
