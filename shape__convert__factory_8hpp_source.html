63 template<
64 typename Shape_,
65 int num_coords_,
66 typename Coord_>
67 class ShapeConvertFactory<ConformalMesh<Shape_, num_coords_, Coord_> > :
68 public Factory<ConformalMesh<Shape_, num_coords_, Coord_> >
69 {
70 public:
71 typedef Factory<ConformalMesh<Shape_, num_coords_, Coord_> > BaseClass;
72 typedef Shape_ ShapeType;
73 typedef typename Intern::OtherShape<Shape_>::Type OtherShapeType;
74 typedef ConformalMesh<Shape_, num_coords_, Coord_> MeshType;
75 typedef ConformalMesh<OtherShapeType, num_coords_, Coord_> OtherMeshType;
77 typedef typename MeshType::VertexSetType VertexSetType;
79 typedef typename MeshType::IndexSetHolderType IndexSetHolderType;
80
82 static constexpr int shape_dim = ShapeType::dimension;
83
84 protected:
85 const OtherMeshType& _other_mesh;
87 Index _num_entities_in[shape_dim + 1];
89 Index _num_entities_out[shape_dim + 1];
90
91 public:
92 explicit ShapeConvertFactory(const OtherMeshType& other_mesh) :
93 _other_mesh(other_mesh)
94 {
95 // get number of entities in input mesh
96 for(int i(0); i <= shape_dim; ++i)
97 {
98 _num_entities_out[i] = _num_entities_in[i] = _other_mesh.get_num_entities(i);
99 }
100
101 // calculate number of entities in output mesh
102 Intern::EntityCountWrapper<Intern::ShapeConvertTraits, OtherShapeType>::query(_num_entities_out);
103 }
104
105 virtual Index get_num_entities(int dim) override
106 {
107 return _num_entities_out[dim];
108 }
109
110 virtual void fill_vertex_set(VertexSetType& vertex_set) override
111 {
112 // call wrapper
113 Intern::ShapeConvertVertexWrapper<OtherShapeType, VertexSetType>
114 ::refine(vertex_set, _other_mesh.get_vertex_set(), _other_mesh.get_index_set_holder());
115 }
116
117 virtual void fill_index_sets(IndexSetHolderType& index_set_holder) override
118 {
119 // call wrapper to build essential index sets
120 Intern::ShapeConvertIndexWrapper<ShapeType>
121 ::refine(index_set_holder, _num_entities_in, _other_mesh.get_index_set_holder());
122
123 // build redundant index sets
124 RedundantIndexSetBuilder<ShapeType>::compute(index_set_holder);
125 }
