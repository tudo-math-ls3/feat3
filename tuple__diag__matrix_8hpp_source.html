208 void write_out(FileMode mode, String filename) const
209 {
210 std::ofstream file(filename.c_str(), std::ofstream::out);
211 if (! file.is_open())
212 XABORTM("Unable to open Matrix file " + filename);

214 String suffix, directory;
215 auto found = filename.rfind(".");
216 if (found != std::string::npos)
217 {
218 suffix = filename.substr(found);
219 filename.erase(found);
220 }
221 found = filename.rfind("/");
222 if (found != std::string::npos)
223 {
224 directory = filename.substr(0, found + 1);
225 filename.erase(0, found + 1);
226 }
227
228 file << "%%MatrixMarket tuplediagmatrix coordinate real general" << "\n";
229 for (Index i(1); i <= num_row_blocks; ++i)
230 {
231 file << filename << "_td" << i << suffix << "\n";

543
545 void restore_from_checkpoint_data(std::vector<char> & data)
546 {
547 std::uint64_t isize = *(std::uint64_t*) data.data(); //get size of checkpointed _first
548 std::vector<char>::iterator start = std::begin(data) + sizeof(std::uint64_t);
549 std::vector<char>::iterator last_of_first = std::begin(data) + sizeof(std::uint64_t) + (int) isize;
550 std::vector<char> buffer_first(start, last_of_first);
551 _first.restore_from_checkpoint_data(buffer_first);
552
553 data.erase(std::begin(data), last_of_first);
554 _rest.restore_from_checkpoint_data(data);
555 }
556
558 std::uint64_t set_checkpoint_data(std::vector<char>& data, SerialConfig& config)
559 {
560 std::size_t old_size = data.size();
561 data.insert(std::end(data), sizeof(std::uint64_t), 0); //add placeholder
562 std::uint64_t ireal_size = _first.set_checkpoint_data(data, config); //add data of _first to the overall checkpoint and save its size
563 char* csize = reinterpret_cast<char*>(&ireal_size);
564 for(std::size_t i(0) ; i < sizeof(std::uint64_t) ; ++i) //overwrite the guessed datalength
565 {
566 data[old_size +i] = csize[i];
567 }
568
569 return sizeof(std::uint64_t) + ireal_size + _rest.set_checkpoint_data(data, config); //generate and add checkpoint data for the _rest
570 }
