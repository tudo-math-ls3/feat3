133void * FEAT::Util::cuda_get_static_memory(const Index bytes)
134{
135 if(Intern::cuda_workspace_size < bytes)
136 {
137 cudaFree(Intern::cuda_workspace);
138 auto status = cudaMalloc(&Intern::cuda_workspace, bytes);
139 if (status != cudaSuccess)
140 throw InternalError(__func__, __FILE__, __LINE__, "Util::cuda_get_static_memory allocation error\n" + stringify(cudaGetErrorString(status)));
141 if (Intern::cuda_workspace == nullptr)
142 throw InternalError(__func__, __FILE__, __LINE__, "Util::cuda_get_static_memory allocation error (null pointer returned)");
143 Intern::cuda_workspace_size = bytes;
144 }
145 return Intern::cuda_workspace;
146}

180void FEAT::Util::cuda_initialize(int rank, int /*ranks_per_node*/, int /*ranks_per_uma*/, int gpus_per_node)
181{
182 /// \todo enable non cuda ranks and ensure balance of ranks per numa section
183 FEAT::Util::cuda_device_number = rank % gpus_per_node;
184 if (cudaSuccess != cudaSetDevice(cuda_device_number))
185 throw InternalError(__func__, __FILE__, __LINE__, "cudaSetDevice failed!");
186
187 int mm_support = 0;
188 if (cudaSuccess != cudaDeviceGetAttribute(&mm_support, cudaDevAttrManagedMemory, cuda_device_number))
189 throw InternalError(__func__, __FILE__, __LINE__, "cudaGetAttribute failed!");
190 XASSERTM(mm_support == 1, "selected cuda device does not support managed memory!");
191
192 if (CUBLAS_STATUS_SUCCESS != cublasCreate(&Util::Intern::cublas_handle))
193 throw InternalError(__func__, __FILE__, __LINE__, "cublasCreate failed!");
194 if (CUSPARSE_STATUS_SUCCESS != cusparseCreate(&Util::Intern::cusparse_handle))
195 throw InternalError(__func__, __FILE__, __LINE__, "cusparseCreate failed!");
196 if (CUBLAS_STATUS_SUCCESS != cublasSetPointerMode(Util::Intern::cublas_handle, CUBLAS_POINTER_MODE_HOST))
197 throw InternalError(__func__, __FILE__, __LINE__, "cublasSetPointerMode failed!");
198 if (CUSPARSE_STATUS_SUCCESS != cusparseSetPointerMode(Util::Intern::cusparse_handle, CUSPARSE_POINTER_MODE_HOST))
199 throw InternalError(__func__, __FILE__, __LINE__, "cusparseSetPointerMode failed!");
200
201 if (CUBLAS_STATUS_SUCCESS != cublasSetMathMode(Util::Intern::cublas_handle, CUBLAS_TF32_TENSOR_OP_MATH))
202 throw InternalError(__func__, __FILE__, __LINE__, "cublasSetMathMode failed!");
203
204#ifdef FEAT_HAVE_CUDSS
205 if(CUDSS_STATUS_SUCCESS != cudssCreate(&Util::Intern::cudss_handle))
206 throw InternalError(__func__, __FILE__, __LINE__, "cudssCreate failed!");
207#endif
208
209 Util::Intern::cublas_lt_algo_matmat = new cublasLtMatmulAlgo_t[6];
210 Util::Intern::cublas_lt_algo_matmat_initialized = new bool[6];
211 for (int i(0) ; i < 6 ; ++i)
212 {
213 Util::Intern::cublas_lt_algo_matmat_initialized[i] = false;
214 }
215
216 //Util::Intern::cuda_workspace_size = 1024ul * 1024ul * 1024ul * 2ul;
217 Util::Intern::cuda_workspace_size = 0;
218 /*auto status = cudaMalloc(&(Util::Intern::cuda_workspace), Util::Intern::cuda_workspace_size);
219 if (status != cudaSuccess)
220 throw InternalError(__func__, __FILE__, __LINE__, "cudaMalloc failed: " + stringify(cudaGetErrorString(status)));*/
221}

223void FEAT::Util::cuda_finalize()
224{
225 if (cudaSuccess != cudaDeviceSynchronize())
226 throw InternalError(__func__, __FILE__, __LINE__, "cudaDeviceSynchronize failed!");
227
228#ifdef FEAT_HAVE_CUDSS
229 if(CUDSS_STATUS_SUCCESS != cudssDestroy(Util::Intern::cudss_handle))
230 throw InternalError(__func__, __FILE__, __LINE__, "cudssDestroy failed!");
231#endif
232 if (CUBLAS_STATUS_SUCCESS != cublasDestroy(Util::Intern::cublas_handle))
233 throw InternalError(__func__, __FILE__, __LINE__, "cublasDestroy failed!");
234 if (CUSPARSE_STATUS_SUCCESS != cusparseDestroy(Util::Intern::cusparse_handle))
235 throw InternalError(__func__, __FILE__, __LINE__, "cusparseDestroy failed!");
236
237 delete[] Util::Intern::cublas_lt_algo_matmat;
238 delete[] Util::Intern::cublas_lt_algo_matmat_initialized;
239
240 cuda_free_static_memory();
241
242 cudaError_t last_error(cudaGetLastError());
243 if (cudaSuccess != last_error)
244 throw InternalError(__func__, __FILE__, __LINE__, "Pending cuda errors occurred in execution!\n" + stringify(cudaGetErrorString(last_error)));
245}

322template <typename DT_>
323void FEAT::Util::cuda_set_memory(DT_ * address, const DT_ val, const Index count)
324{
325 Index blocksize = FEAT::Util::cuda_blocksize_misc;
326 dim3 grid;
327 dim3 block;
328 block.x = (unsigned)blocksize;
329 grid.x = (unsigned)ceil((count)/(double)(block.x));
330 FEAT::Util::Intern::cuda_set_memory<<<grid, block>>>(address, val, count);
331
332 cudaDeviceSynchronize();
333#ifdef FEAT_DEBUG_MODE
334 cudaError_t last_error(cudaGetLastError());
335 if (cudaSuccess != last_error)
336 throw InternalError(__func__, __FILE__, __LINE__, "Util::cuda_set_memory failed!\n" + stringify(cudaGetErrorString(last_error)));
337#endif
338}

403String FEAT::Util::cuda_get_visible_devices()
404{
405 String result("");
406 int numDevices(-1);
407 if (cudaSuccess != cudaGetDeviceCount(&numDevices))
408 throw InternalError(__func__, __FILE__, __LINE__, "cudaGetDeviceCount failed!");
409 result += "Number of visible cuda devices: " + stringify(numDevices) + "\n" ;
410
411 for (int idevice(0); idevice<numDevices; ++idevice)
412 {
413 // get device properties
414 cudaDeviceProp prop;
415 if (cudaSuccess != cudaGetDeviceProperties (&prop, idevice))
416 throw InternalError(__func__, __FILE__, __LINE__, "cudaGetDeviceProperties failed!");
417 // print out device name and compute capabilities
418 result += "Device " + stringify(idevice) + ": " + stringify(prop.name) + "\n";
419 }
420 return result;
421}
