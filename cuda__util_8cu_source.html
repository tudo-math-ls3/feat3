50 if (idx >= count)

52 ptr[idx] = val;

59 if (idx >= count)
60 return;
61 dest[idx] = src[idx];

151void FEAT::Util::cuda_initialize(int rank, int /*ranks_per_node*/, int /*ranks_per_uma*/, int gpus_per_node)
152{
153 /// \todo enable non cuda ranks and ensure balance of ranks per numa section
154 int device = rank % gpus_per_node;
155 if (cudaSuccess != cudaSetDevice(device))
156 throw InternalError(__func__, __FILE__, __LINE__, "cudaSetDevice failed!");
157
158 int mm_support = 0;
159 if (cudaSuccess != cudaDeviceGetAttribute(&mm_support, cudaDevAttrManagedMemory, device))
160 throw InternalError(__func__, __FILE__, __LINE__, "cudaGetAttribute failed!");
161 XASSERTM(mm_support == 1, "selected cuda device does not support managed memory!");
162
163 if (CUBLAS_STATUS_SUCCESS != cublasCreate(&Util::Intern::cublas_handle))
164 throw InternalError(__func__, __FILE__, __LINE__, "cublasCreate failed!");
165 if (CUSPARSE_STATUS_SUCCESS != cusparseCreate(&Util::Intern::cusparse_handle))
166 throw InternalError(__func__, __FILE__, __LINE__, "cusparseCreate failed!");
167 if (CUBLAS_STATUS_SUCCESS != cublasSetPointerMode(Util::Intern::cublas_handle, CUBLAS_POINTER_MODE_HOST))
168 throw InternalError(__func__, __FILE__, __LINE__, "cublasSetPointerMode failed!");
169 if (CUSPARSE_STATUS_SUCCESS != cusparseSetPointerMode(Util::Intern::cusparse_handle, CUSPARSE_POINTER_MODE_HOST))
170 throw InternalError(__func__, __FILE__, __LINE__, "cusparseSetPointerMode failed!");
171
172 if (CUBLAS_STATUS_SUCCESS != cublasSetMathMode(Util::Intern::cublas_handle, CUBLAS_TF32_TENSOR_OP_MATH))
173 throw InternalError(__func__, __FILE__, __LINE__, "cublasSetMathMode failed!");
174
175#ifdef FEAT_HAVE_CUDSS
176 if(CUDSS_STATUS_SUCCESS != cudssCreate(&Util::Intern::cudss_handle))
177 throw InternalError(__func__, __FILE__, __LINE__, "cudssCreate failed!");
178#endif
179
180 Util::Intern::cublas_lt_algo_matmat = new cublasLtMatmulAlgo_t[6];
181 Util::Intern::cublas_lt_algo_matmat_initialized = new bool[6];
182 for (int i(0) ; i < 6 ; ++i)
183 {
184 Util::Intern::cublas_lt_algo_matmat_initialized[i] = false;
185 }
186
187 //Util::Intern::cuda_workspace_size = 1024ul * 1024ul * 1024ul * 2ul;
188 Util::Intern::cuda_workspace_size = 0;
189 /*auto status = cudaMalloc(&(Util::Intern::cuda_workspace), Util::Intern::cuda_workspace_size);
190 if (status != cudaSuccess)
191 throw InternalError(__func__, __FILE__, __LINE__, "cudaMalloc failed: " + stringify(cudaGetErrorString(status)));*/
192}

194void FEAT::Util::cuda_finalize()
195{
196 if (cudaSuccess != cudaDeviceSynchronize())
197 throw InternalError(__func__, __FILE__, __LINE__, "cudaDeviceSynchronize failed!");
198
199#ifdef FEAT_HAVE_CUDSS
200 if(CUDSS_STATUS_SUCCESS != cudssDestroy(Util::Intern::cudss_handle))
201 throw InternalError(__func__, __FILE__, __LINE__, "cudssDestroy failed!");
202#endif
203 if (CUBLAS_STATUS_SUCCESS != cublasDestroy(Util::Intern::cublas_handle))
204 throw InternalError(__func__, __FILE__, __LINE__, "cublasDestroy failed!");
205 if (CUSPARSE_STATUS_SUCCESS != cusparseDestroy(Util::Intern::cusparse_handle))
206 throw InternalError(__func__, __FILE__, __LINE__, "cusparseDestroy failed!");
207
208 delete[] Util::Intern::cublas_lt_algo_matmat;
209 delete[] Util::Intern::cublas_lt_algo_matmat_initialized;
210
211 //if (cudaSuccess != cudaFree(Util::Intern::cuda_workspace))
212 // throw InternalError(__func__, __FILE__, __LINE__, "cudaFree failed!");
213
214 cudaError_t last_error(cudaGetLastError());
215 if (cudaSuccess != last_error)
216 throw InternalError(__func__, __FILE__, __LINE__, "Pending cuda errors occurred in execution!\n" + stringify(cudaGetErrorString(last_error)));
217
218 if (cudaSuccess != cudaDeviceReset())
219 throw InternalError(__func__, __FILE__, __LINE__, "cudaDeviceReset failed!");
220}

287template <typename DT_>
288void FEAT::Util::cuda_set_memory(DT_ * address, const DT_ val, const Index count)
289{
290 Index blocksize = FEAT::Util::cuda_blocksize_misc;
291 dim3 grid;
292 dim3 block;
293 block.x = (unsigned)blocksize;
294 grid.x = (unsigned)ceil((count)/(double)(block.x));
295 FEAT::Util::Intern::cuda_set_memory<<<grid, block>>>(address, val, count);
296
297 cudaDeviceSynchronize();
298#ifdef FEAT_DEBUG_MODE
299 cudaError_t last_error(cudaGetLastError());
300 if (cudaSuccess != last_error)
301 throw InternalError(__func__, __FILE__, __LINE__, "Util::cuda_set_memory failed!\n" + stringify(cudaGetErrorString(last_error)));
302#endif
303}

368String FEAT::Util::cuda_get_visible_devices()
369{
370 String result("");
371 int numDevices(-1);
372 if (cudaSuccess != cudaGetDeviceCount(&numDevices))
373 throw InternalError(__func__, __FILE__, __LINE__, "cudaGetDeviceCount failed!");
374 result += "Number of visible cuda devices: " + stringify(numDevices) + "\n" ;
375
376 for (int idevice(0); idevice<numDevices; ++idevice)
377 {
378 // get device properties
379 cudaDeviceProp prop;
380 if (cudaSuccess != cudaGetDeviceProperties (&prop, idevice))
381 throw InternalError(__func__, __FILE__, __LINE__, "cudaGetDeviceProperties failed!");
382 // print out device name and compute capabilities
383 result += "Device " + stringify(idevice) + ": " + stringify(prop.name) + "\n";
384 }
385 return result;
386}

388void FEAT::Util::cuda_set_max_cache_thread(const std::size_t bytes)
389{
390 std::size_t value = bytes;
391 auto status = cudaDeviceSetLimit(cudaLimit::cudaLimitStackSize, value);
392 if(cudaSuccess != status)
393 throw InternalError(__func__, __FILE__, __LINE__, "cudaDeviceSetLimit failed!");
394 status = cudaDeviceGetLimit(&value, cudaLimit::cudaLimitStackSize);
395 if(cudaSuccess != status)
396 throw InternalError(__func__, __FILE__, __LINE__, "cudaDeviceGetLimit failed!");
397 if(value != bytes)
398 XABORTM("Could not set max cache per thread (" +stringify(value)+") to expected amount(" + stringify(bytes) + ")");
399}
