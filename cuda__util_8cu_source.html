76void * FEAT::Util::cuda_malloc_managed(const Index bytes)
77{
78 void * memory(nullptr);
79 if (bytes == 0)
80 return memory;
81
82 auto status = cudaMallocManaged((void**)&memory, bytes);
83 if (status != cudaSuccess)
84 throw InternalError(__func__, __FILE__, __LINE__, "Util::cuda_malloc_managed allocation error\n" + stringify(cudaGetErrorString(status)));
85 if (memory == nullptr)
86 throw InternalError(__func__, __FILE__, __LINE__, "Util::cuda_malloc_managed allocation error (null pointer returned)");
87 return memory;
88}

100void FEAT::Util::cuda_initialize(int rank, int /*ranks_per_node*/, int /*ranks_per_uma*/, int gpus_per_node)
101{
102 /// \todo enable non cuda ranks and ensure balance of ranks per numa section
103 int device = rank % gpus_per_node;
104 if (cudaSuccess != cudaSetDevice(device))
105 throw InternalError(__func__, __FILE__, __LINE__, "cudaSetDevice failed!");
106
107 int mm_support = 0;
108 if (cudaSuccess != cudaDeviceGetAttribute(&mm_support, cudaDevAttrManagedMemory, device))
109 throw InternalError(__func__, __FILE__, __LINE__, "cudaGetAttribute failed!");
110 XASSERTM(mm_support == 1, "selected cuda device does not support managed memory!");
111
112 if (CUBLAS_STATUS_SUCCESS != cublasCreate(&Util::Intern::cublas_handle))
113 throw InternalError(__func__, __FILE__, __LINE__, "cublasCreate failed!");
114 if (CUSPARSE_STATUS_SUCCESS != cusparseCreate(&Util::Intern::cusparse_handle))
115 throw InternalError(__func__, __FILE__, __LINE__, "cusparseCreate failed!");
116 if (CUBLAS_STATUS_SUCCESS != cublasSetPointerMode(Util::Intern::cublas_handle, CUBLAS_POINTER_MODE_HOST))
117 throw InternalError(__func__, __FILE__, __LINE__, "cublasSetPointerMode failed!");
118 if (CUSPARSE_STATUS_SUCCESS != cusparseSetPointerMode(Util::Intern::cusparse_handle, CUSPARSE_POINTER_MODE_HOST))
119 throw InternalError(__func__, __FILE__, __LINE__, "cusparseSetPointerMode failed!");
120
121 if (CUBLAS_STATUS_SUCCESS != cublasSetMathMode(Util::Intern::cublas_handle, CUBLAS_TF32_TENSOR_OP_MATH))
122 throw InternalError(__func__, __FILE__, __LINE__, "cublasSetMathMode failed!");
123
124 Util::Intern::cublas_lt_algo_matmat = new cublasLtMatmulAlgo_t[6];
125 Util::Intern::cublas_lt_algo_matmat_initialized = new bool[6];
126 for (int i(0) ; i < 6 ; ++i)
127 {
128 Util::Intern::cublas_lt_algo_matmat_initialized[i] = false;
129 }
130
131 //Util::Intern::cuda_workspace_size = 1024ul * 1024ul * 1024ul * 2ul;
132 Util::Intern::cuda_workspace_size = 0;
133 /*auto status = cudaMalloc(&(Util::Intern::cuda_workspace), Util::Intern::cuda_workspace_size);
134 if (status != cudaSuccess)
135 throw InternalError(__func__, __FILE__, __LINE__, "cudaMalloc failed: " + stringify(cudaGetErrorString(status)));*/
136}

138void FEAT::Util::cuda_finalize()
139{
140 if (cudaSuccess != cudaDeviceSynchronize())
141 throw InternalError(__func__, __FILE__, __LINE__, "cudaDeviceSynchronize failed!");
142
143 if (CUBLAS_STATUS_SUCCESS != cublasDestroy(Util::Intern::cublas_handle))
144 throw InternalError(__func__, __FILE__, __LINE__, "cublasDestroy failed!");
145 if (CUSPARSE_STATUS_SUCCESS != cusparseDestroy(Util::Intern::cusparse_handle))
146 throw InternalError(__func__, __FILE__, __LINE__, "cusparseDestroy failed!");
147
148 delete[] Util::Intern::cublas_lt_algo_matmat;
149 delete[] Util::Intern::cublas_lt_algo_matmat_initialized;
150
151 //if (cudaSuccess != cudaFree(Util::Intern::cuda_workspace))
152 // throw InternalError(__func__, __FILE__, __LINE__, "cudaFree failed!");
153
154 cudaError_t last_error(cudaGetLastError());
155 if (cudaSuccess != last_error)
156 throw InternalError(__func__, __FILE__, __LINE__, "Pending cuda errors occurred in execution!\n" + stringify(cudaGetErrorString(last_error)));
157
158 if (cudaSuccess != cudaDeviceReset())
159 throw InternalError(__func__, __FILE__, __LINE__, "cudaDeviceReset failed!");
160}

202template <typename DT_>
203void FEAT::Util::cuda_set_memory(DT_ * address, const DT_ val, const Index count)
204{
205 Index blocksize = FEAT::Util::cuda_blocksize_misc;
206 dim3 grid;
207 dim3 block;
208 block.x = (unsigned)blocksize;
209 grid.x = (unsigned)ceil((count)/(double)(block.x));
210 FEAT::Util::Intern::cuda_set_memory<<<grid, block>>>(address, val, count);
211
212 cudaDeviceSynchronize();
213#ifdef FEAT_DEBUG_MODE
214 cudaError_t last_error(cudaGetLastError());
215 if (cudaSuccess != last_error)
216 throw InternalError(__func__, __FILE__, __LINE__, "Util::cuda_set_memory failed!\n" + stringify(cudaGetErrorString(last_error)));
217#endif
218}
