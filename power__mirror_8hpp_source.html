20 // PowerMirror helper class: this implements the actual recursion for the gather/scatter methods
21 template<int i_>
22 struct PowerMirrorHelper
23 {
24 template<typename SM_, typename BV_, typename PV_>
25 static void gather(const SM_& sm, BV_& bv, const PV_& pv, const Index bo)
26 {
27 sm.gather(bv, pv.first(), bo);
28 PowerMirrorHelper<i_-1>::gather(sm, bv, pv.rest(), bo + sm.buffer_size(pv.first()));
29 }
30 template<typename SM_, typename PV_, typename BV_, typename AT_>
31 static void scatter_axpy(const SM_& sm, PV_& pv, const BV_& bv, const AT_ a, const Index bo)
32 {
33 sm.scatter_axpy(pv.first(), bv, a, bo);
34 PowerMirrorHelper<i_-1>::scatter_axpy(sm, pv.rest(), bv, a, bo + sm.buffer_size(pv.first()));
35 }
36 template<typename SM_, typename PV_>
37 static Index buffer_size(const SM_& sm, PV_& pv)
38 {
39 return sm.buffer_size(pv.first()) + PowerMirrorHelper<i_-1>::buffer_size(sm, pv.rest());
40 }
41 template<Perspective perspective_, typename SM_, typename Pv_>
42 static Index mask_scatter(const SM_& sm, const Pv_& pv, std::vector<int>& mask, const int value, const Index offset)
43 {
44 Index nf = sm.template mask_scatter<perspective_>(pv.first(), mask, value, offset);
45 Index nr = PowerMirrorHelper<i_-1>::template mask_scatter<perspective_>(sm, pv.rest(), mask, value, offset + nf);
46 return nf + nr;
47 }
48 };
