65 template<typename SpaceHelp_, typename LocMatType_, typename LocVecType_, int dim_, int num_verts_>
66 CUDA_HOST_DEVICE void burgers_mat_assembly_kernel(LocMatType_& loc_mat, const LocVecType_& local_conv_dofs, const typename SpaceHelp_::DataType (&local_coeffs)[dim_][num_verts_],
67 const typename SpaceHelp_::DomainPointType* cub_pt, const typename SpaceHelp_::DataType* cub_wg, const int num_cubs,
68 const VoxelAssembly::AssemblyBurgersData<typename SpaceHelp_::DataType>& burgers_params,
69 const bool need_streamline, const bool need_convection, const typename SpaceHelp_::DataType tol_eps)
70 {
71 typedef SpaceHelp_ SpaceHelp;
72 constexpr int dim = SpaceHelp::dim;
73 typedef typename SpaceHelp::SpaceType SpaceType;
74 typedef typename SpaceHelp::DataType DataType;
75
76 const DataType& nu{burgers_params.nu};
77 const DataType& theta{burgers_params.theta};
78 const DataType& beta{burgers_params.beta};
79 const DataType& frechet_beta{burgers_params.frechet_beta};
80 const DataType& sd_delta{burgers_params.sd_delta};
81 const DataType& sd_nu{burgers_params.sd_nu};
82 const DataType& sd_v_norm{burgers_params.sd_v_norm};
83 const bool& deformation{burgers_params.deformation};
84
85 // #ifdef __CUDACC__
86 // const DataType tol_eps = CudaMath::cuda_get_sqrt_eps<DataType>();
87 // #else
88 // const DataType tol_eps = Math::sqrt(Math::eps<DataType>());
89 // #endif
90
91 // #ifdef __CUDACC__
92 // const bool need_streamline = (CudaMath::cuda_abs(sd_delta) > DataType(0)) && (sd_v_norm > tol_eps);
93 // const bool need_convection = CudaMath::cuda_abs(beta) > DataType(0);
94 // #else
95 // const bool need_streamline = (Math::abs(sd_delta) > DataType(0)) && (sd_v_norm > tol_eps);
96 // const bool need_convection = Math::abs(beta) > DataType(0);
97 // #endif
98
99
100 //define local sizes
101 constexpr int num_loc_dofs = SpaceType::DofMappingType::dof_count;
102 //get number of nodes per element
103 // constexpr int num_loc_verts = SpaceType::MeshType::template IndexSet<dim, 0>::Type::num_indices;
104 // define local arrays
105 typename SpaceHelp::JacobianMatrixType loc_jac, loc_jac_inv;
106 typename SpaceHelp::EvalData basis_data;
107
108 // local vector and matrix defines
109 typedef Tiny::Vector<DataType, dim> VecValueType;
110 typedef Tiny::Matrix<DataType, dim, dim> MatValueType;
111
112 VecValueType loc_v(DataType(0)), mean_v(DataType(0));
113 MatValueType loc_grad_v(DataType(0));
114 DataType local_delta(DataType(0));
115
116 loc_mat.format();
117
118 Tiny::Vector<DataType, num_loc_dofs> streamdiff_coeffs(DataType(0));
119
120 if(need_streamline) //need streamdiff?
121 {
122 //set domain point to barycenter, which is zero for Hypercubes
123 const VecValueType barycenter(DataType(0));
124 // NewSpaceHelp::_map_point(img_point, barycenter, local_coeffs);
125 //only reserve memory for reference values
126 SpaceHelp::eval_ref_values(basis_data, barycenter);
127 SpaceHelp::trans_values(basis_data);
128 for(int i(0); i < num_loc_dofs; ++i)
129 {
130 mean_v.axpy(basis_data.phi[i].value, local_conv_dofs[i]);
131 }
132 const DataType local_norm_v = mean_v.norm_euclid();
133
134 if(local_norm_v > tol_eps)
135 {
136 //we need this trafo data
137 SpaceHelp::calc_jac_mat(loc_jac, barycenter, local_coeffs);
138 loc_jac_inv.set_inverse(loc_jac);
139
140 const DataType local_h = SpaceHelp::width_directed(mean_v, local_coeffs) * local_norm_v;
141 const DataType local_re = (local_norm_v * local_h) / sd_nu;
142 local_delta = sd_delta * (local_h / sd_v_norm) * (DataType(2) * local_re) / (DataType(1) + local_re);
143 }
144 }
145
146 for(int cub_ind = 0; cub_ind < num_cubs; ++cub_ind)
147 {
148 const typename SpaceHelp::DomainPointType& dom_point = cub_pt[cub_ind];
149 SpaceHelp::eval_ref_values(basis_data, dom_point);
150 SpaceHelp::trans_values(basis_data);
151 // NewSpaceHelp::_map_point(img_point, dom_point, local_coeffs);//not needed?
152 SpaceHelp::calc_jac_mat(loc_jac, dom_point, local_coeffs);
153 loc_jac_inv.set_inverse(loc_jac);
154
155 SpaceHelp::eval_ref_gradients(basis_data, dom_point);
156 SpaceHelp::trans_gradients(basis_data, loc_jac_inv);
157
158 const DataType weight = loc_jac.det() * cub_wg[cub_ind];
159 if(need_convection || need_streamline) // need streamdiff or convection?
160 {
161 loc_v.format();
162 for(int i = 0; i < num_loc_dofs; ++i)
163 {
164 loc_v.axpy(basis_data.phi[i].value, local_conv_dofs[i]);
165 }
166 }
167 #ifdef __CUDACC__
168 if(CudaMath::cuda_abs(frechet_beta) > DataType(0)) // need frechet beta?
169 #else
170 if(Math::abs(frechet_beta) > DataType(0)) // need frechet beta?
171 #endif
172 {
173 loc_grad_v.format();
174 for(int i = 0; i < num_loc_dofs; ++i)
175 {
176 loc_grad_v.add_outer_product(local_conv_dofs[i], basis_data.phi[i].grad);
177 }
178 }
179
180 if(need_streamline) // need streamdiff?
181 {
182 for(int i = 0; i < num_loc_dofs; ++i)
183 {
184 streamdiff_coeffs[i] = Tiny::dot(loc_v, basis_data.phi[i].grad);
185 }
186 }
187
188 if(deformation)
189 {
190 for(int i(0); i < num_loc_dofs; ++i)
191 {
192 // trial function loop
193 for(int j(0); j < num_loc_dofs; ++j)
194 {
195 // compute scalar value
196 const DataType value = nu * weight * Tiny::dot(basis_data.phi[i].grad, basis_data.phi[j].grad);
197
198 // update local matrix
199 loc_mat[i][j].add_scalar_main_diag(value);
200 loc_mat[i][j].add_outer_product(basis_data.phi[j].grad, basis_data.phi[i].grad, nu * weight);
201 }
202 }
203 }
204 else
205 {
206 for(int i(0); i < num_loc_dofs; ++i)
207 {
208 // trial function loop
209 for(int j(0); j < num_loc_dofs; ++j)
210 {
211 // compute scalar value
212 const DataType value = nu * weight * Tiny::dot(basis_data.phi[i].grad, basis_data.phi[j].grad);
213
214 // update local matrix
215 loc_mat[i][j].add_scalar_main_diag(value);
216 }
217 }
218 }
219
220 if(need_convection) // assemble convection?
221 {
222 // test function loop
223 for(int i(0); i < num_loc_dofs; ++i)
224 {
225 // trial function loop
226 for(int j(0); j < num_loc_dofs; ++j)
227 {
228 // compute scalar value
229 const DataType value = beta * weight * basis_data.phi[i].value * Tiny::dot(loc_v, basis_data.phi[j].grad);
230
231 // update local matrix
232 loc_mat[i][j].add_scalar_main_diag(value);
233 }
234 }
235 }
236
237 // assemble convection Frechet?
238 #ifdef __CUDACC__
239 if(CudaMath::cuda_abs(frechet_beta) > DataType(0))
240 #else
241 if(Math::abs(frechet_beta) > DataType(0))
242 #endif
243 {
244 // test function loop
245 for(int i(0); i < num_loc_dofs; ++i)
246 {
247 // trial function loop
248 for(int j(0); j < num_loc_dofs; ++j)
249 {
250 // compute scalar value
251 const DataType value = frechet_beta * weight * basis_data.phi[i].value * basis_data.phi[j].value;
252
253 // update local matrix
254 loc_mat[i][j].axpy(value, loc_grad_v);
255 }
256 }
257 }
258
259 // assemble reaction?
260 #ifdef __CUDACC__
261 if(CudaMath::cuda_abs(theta) > DataType(0))
262 #else
263 if(Math::abs(theta) > DataType(0))
264 #endif
265 {
266 // test function loop
267 for(int i(0); i < num_loc_dofs; ++i)
268 {
269 // trial function loop
270 for(int j(0); j < num_loc_dofs; ++j)
271 {
272 // compute scalar value
273 const DataType value = theta * weight * basis_data.phi[i].value * basis_data.phi[j].value;
274
275 // update local matrix
276 loc_mat[i][j].add_scalar_main_diag(value);
277 }
278 }
279 }
280
281 // assemble streamline diffusion?
282 if((need_streamline) && (local_delta > tol_eps))
283 {
284 // test function loop
285 for(int i(0); i < num_loc_dofs; ++i)
286 {
287 // trial function loop
288 for(int j(0); j < num_loc_dofs; ++j)
289 {
290 // compute scalar value
291 const DataType value = local_delta * weight * streamdiff_coeffs[i] * streamdiff_coeffs[j];
292
293 // update local matrix
294 loc_mat[i][j].add_scalar_main_diag(value);
295 }
296 }
297 }
298 // next cubature point
299 }
