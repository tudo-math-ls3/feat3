104 class ScatterAxpy
105 {
106 public:
107 typedef LAFEM::DenseVectorBlocked<DT_, IT_, BlockSize_> VectorType;
108 typedef DT_ DataType;
109 typedef IT_ IndexType;
110 typedef Tiny::Vector<DT_, BlockSize_> ValueType;
111
112 private:
113 IT_ _num_entries;
114 ValueType * _data;
115
116 public:
117 explicit ScatterAxpy(VectorType & vector) :
118 _num_entries(vector.size()),
119 _data(vector.elements())
120 {
121 }
122
123 template <typename LocalVector_, typename Mapping_>
124 void operator()(const LocalVector_ & loc_vec, const Mapping_ & mapping, DT_ alpha = DT_(1))
125 {
126 // loop over all local entries
127 for (int i(0); i < mapping.get_num_local_dofs(); ++i)
128 {
129 // get dof index
130 Index dof_idx = mapping.get_index(i);
131 ASSERT(dof_idx < _num_entries);
132
133 // update vector data
134 _data[dof_idx] += alpha * loc_vec[i];
135 }
136 }
137 }; // class ScatterAxpy

442 template <typename DT2_, typename IT2_>

444 {
445 XASSERTM(other.size() % Index(BlockSize_) == 0, "DenseVector cannot be converted to given blocksize!");

448
449 this->_scalar_index.push_back(other.size() / Index(BlockSize_));
450
451 this->_elements.push_back(other.get_elements().at(0));

455 MemoryPool::increase_memory(this->_elements.at(i));
456 for (Index i(0) ; i < this->_indices.size() ; ++i)
457 MemoryPool::increase_memory(this->_indices.at(i));
458 }

591 void read_from(FileMode mode, String filename)
592 {
593 std::ios_base::openmode bin = std::ifstream::in | std::ifstream::binary;
594 if (mode == FileMode::fm_mtx)
595 bin = std::ifstream::in;
596 std::ifstream file(filename.c_str(), bin);
597 if (! file.is_open())
598 XABORTM("Unable to open Vector file " + filename);
599 read_from(mode, file);
600 file.close();
601 }
602
609 void read_from(FileMode mode, std::istream & file)
610 {
611 switch (mode)
612 {
613 case FileMode::fm_mtx:
614 {
615 this->clear();
616 this->_scalar_index.push_back(0);
617
618 Index rows;
619 String line;
620 std::getline(file, line);
621 if (line.find("%%MatrixMarket matrix array real general") == String::npos)
622 {
623 XABORTM("Input-file is not a compatible mtx-vector-file");
624 }
625 while (! file.eof())
626 {
627 std::getline(file, line);
628 if (file.eof())
629 XABORTM("Input-file is empty");
630
631 String::size_type begin(line.find_first_not_of(" "));
632 if (line.at(begin) != '%')
633 break;
634 }
635 {
636 String::size_type begin(line.find_first_not_of(" "));
637 line.erase(0, begin);
638 String::size_type end(line.find_first_of(" "));
639 String srows(line, 0, end);
640 rows = (Index)atol(srows.c_str());
641 line.erase(0, end);
642
643 begin = line.find_first_not_of(" ");
644 line.erase(0, begin);
645 end = line.find_first_of(" ");
646 String scols(line, 0, end);
647 Index cols((Index)atol(scols.c_str()));
648 line.erase(0, end);
649 if (cols != 1)
650 XABORTM("Input-file is no dense-vector-file");
651 }
652
653 DenseVectorBlocked<DT_, IT_, BlockSize_> tmp(rows / BlockSize_);
654 DT_ * pval(tmp.template elements<Perspective::pod>());
655
656 while (! file.eof())
657 {
658 std::getline(file, line);
659 if (file.eof())
660 break;
661
662 String::size_type begin(line.find_first_not_of(" "));
663 line.erase(0, begin);
664 String::size_type end(line.find_first_of(" "));
665 String sval(line, 0, end);
666 DT_ tval((DT_)atof(sval.c_str()));
667
668 *pval = tval;
669 ++pval;
670 }
671 this->assign(tmp);
672 break;
673 }
674 case FileMode::fm_exp:
675 {
676 this->clear();
677 this->_scalar_index.push_back(0);
678
679 std::vector<DT_> data;
680
681 while (! file.eof())
682 {
683 std::string line;
684 std::getline(file, line);
685 if (line.find("#", 0) < line.npos)
686 continue;
687 if (file.eof())
688 break;
689
690 std::string n_z_s;
691
692 std::string::size_type first_digit(line.find_first_not_of(" "));
693 line.erase(0, first_digit);
694 std::string::size_type eol(line.length());
695 for (std::string::size_type i(0); i < eol; ++i)
696 {
697 n_z_s.append(1, line[i]);
698 }
699
700 DT_ n_z((DT_)atof(n_z_s.c_str()));
701
702 data.push_back(n_z);
703 }
704
705 _size() = Index(data.size()) / Index(BlockSize_);
706 this->_elements.push_back(MemoryPool::template allocate_memory<DT_>(Index(data.size())));
707 this->_elements_size.push_back(Index(data.size()));
708 MemoryPool::template copy<DT_>(this->_elements.at(0), &data[0], Index(data.size()));
709 break;
710 }
711 case FileMode::fm_dvb:
712 case FileMode::fm_binary:
713 {
714 this->clear();
715
716 this->template _deserialize<double, std::uint64_t>(FileMode::fm_dvb, file);
717 break;
718 }
719 default:
720 XABORTM("Filemode not supported!");
721 }
722 }

730 void write_out(FileMode mode, String filename) const
731 {
732 std::ios_base::openmode bin = std::ofstream::out | std::ofstream::binary;
733 if (mode == FileMode::fm_mtx || mode == FileMode::fm_exp)
734 bin = std::ofstream::out;
735 std::ofstream file(filename.c_str(), bin);
736 if (! file.is_open())
737 XABORTM("Unable to open Vector file " + filename);
738 write_out(mode, file);
739 file.close();
740 }

748 void write_out(FileMode mode, std::ostream & file) const
749 {
750 switch (mode)
751 {
752 case FileMode::fm_mtx:
753 {
754 DenseVectorBlocked<DT_, IT_, BlockSize_> temp;
755 temp.convert(*this);
756
757 const Index tsize(temp.template size<Perspective::pod>());
758 file << "%%MatrixMarket matrix array real general" << std::endl;
759 file << tsize << " " << 1 << std::endl;
760
761 const DT_ * pval(temp.template elements<Perspective::pod>());
762 for (Index i(0); i < tsize; ++i, ++pval)
763 {
764 file << std::scientific << *pval << std::endl;
765 }
766 break;
767 }
768 case FileMode::fm_exp:
769 {
770 DT_ * temp = MemoryPool::template allocate_memory<DT_>((this->size<Perspective::pod>()));
771 MemoryPool::template copy<DT_>(temp, this->template elements<Perspective::pod>(), this->size<Perspective::pod>());
772
773 for (Index i(0); i < this->size<Perspective::pod>(); ++i)
774 {
775 file << std::scientific << temp[i] << std::endl;
776 }
777
778 MemoryPool::release_memory(temp);
779 break;
780 }
781 case FileMode::fm_dvb:
782 case FileMode::fm_binary:
783 this->template _serialize<double, std::uint64_t>(FileMode::fm_dvb, file);
784 break;
785 default:
786 XABORTM("Filemode not supported!");
787 }
788 }

799 void axpy(
800 const DenseVectorBlocked & x,
801 const DenseVectorBlocked & y,
802 const DT_ alpha = DT_(1))
803 {
804 XASSERTM(x.size() == y.size(), "Vector size does not match!");
805 XASSERTM(x.size() == this->size(), "Vector size does not match!");
806
807 if (Math::abs(alpha) < Math::eps<DT_>())
808 {
809 this->copy(y);
810 //y.scale(beta);
811 return;
812 }
813
814 TimeStamp ts_start;
815
816 Statistics::add_flops(this->size<Perspective::pod>() * 2);
817 Arch::Axpy::value(elements<Perspective::pod>(), alpha, x.template elements<Perspective::pod>(), y.template elements<Perspective::pod>(), this->size<Perspective::pod>());
818
819 TimeStamp ts_stop;
820 Statistics::add_time_axpy(ts_stop.elapsed(ts_start));
821 }
