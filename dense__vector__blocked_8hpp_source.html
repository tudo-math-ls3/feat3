106 class ScatterAxpy
107 {
108 public:
109 typedef LAFEM::DenseVectorBlocked<DT_, IT_, BlockSize_> VectorType;
110 typedef DT_ DataType;
111 typedef IT_ IndexType;
112 typedef Tiny::Vector<DT_, BlockSize_> ValueType;
113
114 private:
115 IT_ _num_entries;
116 ValueType * _data;
117
118 public:
119 explicit ScatterAxpy(VectorType & vector) :
120 _num_entries(IT_(vector.size())),
121 _data(vector.elements())
122 {
123 }
124
125 template <typename LocalVector_, typename Mapping_>
126 void operator()(const LocalVector_ & loc_vec, const Mapping_ & mapping, DT_ alpha = DT_(1))
127 {
128 // loop over all local entries
129 for (int i(0); i < mapping.get_num_local_dofs(); ++i)
130 {
131 // get dof index
132 Index dof_idx = mapping.get_index(i);
133 ASSERT(dof_idx < _num_entries);
134
135 // update vector data
136 _data[dof_idx] += alpha * loc_vec[i];
137 }
138 }
139 }; // class ScatterAxpy

459 MemoryPool::increase_memory(this->_indices.at(i));

593 void read_from(FileMode mode, const String& filename)
594 {
595 std::ios_base::openmode bin = std::ifstream::in | std::ifstream::binary;
596 if (mode == FileMode::fm_mtx)
597 bin = std::ifstream::in;
598 std::ifstream file(filename.c_str(), bin);
599 if (! file.is_open())
600 XABORTM("Unable to open Vector file " + filename);
601 read_from(mode, file);
602 file.close();
603 }
604
611 void read_from(FileMode mode, std::istream & file)
612 {
613 switch (mode)
614 {
615 case FileMode::fm_mtx:
616 {
617 this->clear();
618 this->_scalar_index.push_back(0);
619
620 Index rows;
621 String line;
622 std::getline(file, line);
623 if (line.find("%%MatrixMarket matrix array real general") == String::npos)
624 {
625 XABORTM("Input-file is not a compatible mtx-vector-file");
626 }
627 while (! file.eof())
628 {
629 std::getline(file, line);
630 if (file.eof())
631 XABORTM("Input-file is empty");
632
633 String::size_type begin(line.find_first_not_of(" "));
634 if (line.at(begin) != '%')
635 break;
636 }
637 {
638 String::size_type begin(line.find_first_not_of(" "));
639 line.erase(0, begin);
640 String::size_type end(line.find_first_of(" "));
641 String srows(line, 0, end);
642 rows = (Index)atol(srows.c_str());
643 line.erase(0, end);
644
645 begin = line.find_first_not_of(" ");
646 line.erase(0, begin);
647 end = line.find_first_of(" ");
648 String scols(line, 0, end);
649 Index cols((Index)atol(scols.c_str()));
650 line.erase(0, end);
651 if (cols != 1)
652 XABORTM("Input-file is no dense-vector-file");
653 }
654
655 DenseVectorBlocked<DT_, IT_, BlockSize_> tmp(rows / BlockSize_);
656 DT_ * pval(tmp.template elements<Perspective::pod>());
657
658 while (! file.eof())
659 {
660 std::getline(file, line);
661 if (file.eof())
662 break;
663
664 String::size_type begin(line.find_first_not_of(" "));
665 line.erase(0, begin);
666 String::size_type end(line.find_first_of(" "));
667 String sval(line, 0, end);
668 DT_ tval((DT_)atof(sval.c_str()));
669
670 *pval = tval;
671 ++pval;
672 }
673 this->move(std::move(tmp));
674 break;
675 }
676 case FileMode::fm_exp:
677 {
678 this->clear();
679 this->_scalar_index.push_back(0);
680
681 std::vector<DT_> data;
682
683 while (! file.eof())
684 {
685 std::string line;
686 std::getline(file, line);
687 if (line.find("#", 0) < line.npos)
688 continue;
689 if (file.eof())
690 break;
691
692 std::string n_z_s;
693
694 std::string::size_type first_digit(line.find_first_not_of(" "));
695 line.erase(0, first_digit);
696 std::string::size_type eol(line.length());
697 for (std::string::size_type i(0); i < eol; ++i)
698 {
699 n_z_s.append(1, line[i]);
700 }
701
702 DT_ n_z((DT_)atof(n_z_s.c_str()));
703
704 data.push_back(n_z);
705 }
706
707 _size() = Index(data.size()) / Index(BlockSize_);
708 this->_elements.push_back(MemoryPool::template allocate_memory<DT_>(Index(data.size())));
709 this->_elements_size.push_back(Index(data.size()));
710 MemoryPool::template copy<DT_>(this->_elements.at(0), &data[0], Index(data.size()));
711 break;
712 }
713 case FileMode::fm_dvb:
714 case FileMode::fm_binary:
715 {
716 this->clear();
717
718 this->template _deserialize<double, std::uint64_t>(FileMode::fm_dvb, file);
719 break;
720 }
721 default:
722 XABORTM("Filemode not supported!");
723 }
724 }

732 void write_out(FileMode mode, const String& filename) const
733 {
734 std::ios_base::openmode bin = std::ofstream::out | std::ofstream::binary;
735 if (mode == FileMode::fm_mtx || mode == FileMode::fm_exp)
736 bin = std::ofstream::out;
737 std::ofstream file;
738 char* buff = nullptr;
739 if(mode == FileMode::fm_mtx)
740 {
741 buff = new char[LAFEM::FileOutStreamBufferSize];
742 file.rdbuf()->pubsetbuf(buff, LAFEM::FileOutStreamBufferSize);
743 }
744 file.open(filename.c_str(), bin);
745 if(! file.is_open())
746 XABORTM("Unable to open Matrix file " + filename);
747 write_out(mode, file);
748 file.close();
749 delete[] buff;
750 }

758 void write_out(FileMode mode, std::ostream & file) const
759 {
760 switch (mode)
761 {
762 case FileMode::fm_mtx:
763 {
764 DenseVectorBlocked<DT_, IT_, BlockSize_> temp;
765 temp.convert(*this);
766
767 const Index tsize(temp.template size<Perspective::pod>());
768 file << "%%MatrixMarket matrix array real general" << "\n";
769 file << tsize << " " << 1 << "\n";
770
771 const DT_ * pval(temp.template elements<Perspective::pod>());
772 for (Index i(0); i < tsize; ++i, ++pval)
773 {
774 file << stringify_fp_sci(*pval) << "\n";
775 }
776 break;
777 }
778 case FileMode::fm_exp:
779 {
780 DT_ * temp = MemoryPool::template allocate_memory<DT_>((this->size<Perspective::pod>()));
781 MemoryPool::template copy<DT_>(temp, this->template elements<Perspective::pod>(), this->size<Perspective::pod>());
782
783 for (Index i(0); i < this->size<Perspective::pod>(); ++i)
784 {
785 file << stringify_fp_sci(temp[i]) << "\n";
786 }
787
788 MemoryPool::release_memory(temp);
789 break;
790 }
791 case FileMode::fm_dvb:
792 case FileMode::fm_binary:
793 this->template _serialize<double, std::uint64_t>(FileMode::fm_dvb, file);
794 break;
795 default:
796 XABORTM("Filemode not supported!");
797 }
798 }
