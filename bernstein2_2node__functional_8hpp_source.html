116 public NodeFunctionalBase<Space_, DataType_>
117 {
118 public:
119 typedef NodeFunctionalBase<Space_, DataType_> BaseClass;
120 static constexpr int max_assigned_dofs = 1;

123 template<typename Function_>
124 struct Value
125 {
126 typedef typename Analytic::EvalTraits<DataType_, Function_>::ValueType Type;
127 };
128
129 protected:
130 typedef typename Space_::TrafoType TrafoType;
131 typedef Shape::Hypercube<1> ShapeType;
132 typedef typename TrafoType::template Evaluator<ShapeType, DataType_>::Type TrafoEvalType;
133 typedef typename TrafoEvalType::EvalTraits TrafoEvalTraits;
134 typedef typename TrafoEvalTraits::DataType DataType;
135 typedef typename TrafoEvalTraits::DomainPointType DomainPointType;
136
137 static constexpr int image_dim = TrafoEvalTraits::image_dim;
138
139 // The coef Matrix gets calculatet in prepare.
140 // Every row represents the coefficients of a basis function.
141 // The i-th basis functions can be evaluated in (x,y) via Intern::base_q(x,y,coef[i])
142 Tiny::Matrix<DataType_, 3, 3> coef;
143
144 static constexpr TrafoTags trafo_tags = TrafoTags::img_point | TrafoTags::jac_det;
145 typedef typename TrafoEvalType::template ConfigTraits<trafo_tags>::EvalDataType TrafoEvalData;
146
147 typedef Cubature::Rule<ShapeType, DataType_, DataType_, DomainPointType> CubRuleType;
148
149 TrafoEvalType _trafo_eval;
150 CubRuleType _cub_rule;
151
152 public:
153 explicit NodeFunctional(const Space_& space) :
154 BaseClass(space),
155 _trafo_eval(space.get_trafo()),
156 _cub_rule(Cubature::ctor_factory, Cubature::DynamicFactory("gauss-legendre:3"))
157 {
158 coef.format();
159 }
160
161 // prepare() is called for every FE.
162 // In prepare the coefficientmatrix is calculated for every FE because otherwise the isoparametic trafo doesnt work.
163 void prepare(Index cell_index)
164 {
165 coef.format();
166 BaseClass::prepare(cell_index);
167 _trafo_eval.prepare(cell_index);
168
169 // declare trafo evaluation data
170 TrafoEvalData trafo_data;
171
172
173 Tiny::Matrix<DataType_, 3, 3> temp(DataType_(0));
174 DataType mean(DataType(0));
175
176 for (int i(0); i < _cub_rule.get_num_points(); ++i)
177 {
178 // get current cubature point
179 _trafo_eval(trafo_data, _cub_rule.get_point(i));
180
181 // calculate p0,p1,p2,q0,q1,q2 at the i-th cubature point[x] and temporary store them in temp like this:
182 //temp=[p0(x),q0(x),0;
183 // p1(x),q1(x),0;
184 // p2(x),q2(x),0;]
185 for (int j = 0; j < 3; ++j)
186 {
187 temp(j, 0) = Intern::p(j, trafo_data.dom_point[0]);
188 temp(j, 1) = Intern::q(j, trafo_data.dom_point[0]);
189 }
190
191 DataType_ mult = _cub_rule.get_weight(i) * trafo_data.jac_det;
192 mean += mult;
193 for (int px = 0; px < 3; ++px)
194 {
195 for (int qx = 0; qx < 3; ++qx)
196 {
197 coef(px, qx) +=
198 (mult * temp(px, 0) * temp(qx, 1));
199 } // end for qx
200 } // end for px
201 temp.format();
202 } // end cubature
203 // scale coefficient matrix
204 coef *= DataType(2) / mean;
205 temp.set_inverse(coef);
206 coef.set_transpose(temp);
207 } // end prepare
208
209 void finish()
210 {
211 _trafo_eval.finish();
212 BaseClass::finish();
213 }

216 {
217 return max_assigned_dofs;

220 template<typename NodeData_, typename Function_>
221 void operator()(NodeData_& node_data, const Function_& function) const
222 {
223 static_assert(std::is_base_of<Analytic::Function, Function_>::value, "invalid function object");
224 static_assert(Function_::can_value, "function cannot compute values");
225
226 // declare our evaluation traits
227 typedef Analytic::EvalTraits<DataType_, Function_> FuncEvalTraits;
228
229 // declare function evaluator
230 typename Function_::template Evaluator<FuncEvalTraits> func_eval(function);
231
232 // compute trafo data
233 typename FuncEvalTraits::ValueType value(DataType_(0));
234 DataType_ mean(DataType_(0));
235 TrafoEvalData trafo_data;
236
237 // integrate over facet with cubature factory
238 for (int i(0); i < _cub_rule.get_num_points(); ++i)
239 {
240 _trafo_eval(trafo_data, _cub_rule.get_point(i));
241 value += (_cub_rule.get_weight(i) * trafo_data.jac_det) * func_eval.value(trafo_data.img_point)
242 *Intern::base_q(trafo_data.dom_point[0], coef[2]);
243 mean += _cub_rule.get_weight(i) *trafo_data.jac_det;
244 }
245
246 // set node_data to the integral mean
247 node_data[0] = (DataType_(2) / mean)* value;
248 }
