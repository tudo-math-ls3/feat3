39 void query_memory_usage(
40 unsigned long long& work_set_size,
41 unsigned long long& work_set_size_peak,
42 unsigned long long& page_file_usage,
43 unsigned long long& page_file_usage_peak)
44 {
45 work_set_size = work_set_size_peak = page_file_usage = page_file_usage_peak = 0ull;
46
47 // initialize memory counters structure
48 PROCESS_MEMORY_COUNTERS pmc;
49 memset(&pmc, 0, sizeof(PROCESS_MEMORY_COUNTERS));
50
51 // get memory info
52 if(GetProcessMemoryInfo(GetCurrentProcess(), &pmc, sizeof(PROCESS_MEMORY_COUNTERS)) != FALSE)
53 {
54 work_set_size = pmc.WorkingSetSize;
55 work_set_size_peak = pmc.PeakWorkingSetSize;
56 page_file_usage = pmc.PagefileUsage;
57 page_file_usage_peak = pmc.PeakPagefileUsage;
58 }
59 }

61 void dump_call_stack(CONTEXT& ctx, FILE* stream)
62 {
63 DWORD line_dsp;
64 DWORD64 addr;
65 struct
66 {
67 SYMBOL_INFO si;
68 char name[512];
69 } sim;
70
71 // retrieve our process and thread id
72 HANDLE prc = GetCurrentProcess();
73 HANDLE thr = GetCurrentThread();
74
75 // initialize line info
76 IMAGEHLP_LINE64 img_line;
77 img_line.SizeOfStruct = sizeof(IMAGEHLP_LINE64);
78
79 // create a symbol information structure
80 memset(&sim, 0, sizeof(SYMBOL_INFO) + 512);
81 sim.si.MaxNameLen = 512;
82 sim.si.SizeOfStruct = sizeof(SYMBOL_INFO);
83
84 // initialize symbol table
85 SymInitialize(prc, nullptr, TRUE);
86
87 // initialize the stack frame stuff
88 STACKFRAME64 stack_frame;
89 memset(&stack_frame, 0, sizeof(stack_frame));
90
91#if defined(_WIN64)
92 int machine_type = IMAGE_FILE_MACHINE_AMD64;
93 stack_frame.AddrPC.Offset = ctx.Rip;
94 stack_frame.AddrFrame.Offset = ctx.Rbp;
95 stack_frame.AddrStack.Offset = ctx.Rsp;
96#else
97 int machine_type = IMAGE_FILE_MACHINE_I386;
98 stack_frame.AddrPC.Offset = ctx.Eip;
99 stack_frame.AddrFrame.Offset = ctx.Ebp;
100 stack_frame.AddrStack.Offset = ctx.Esp;
101#endif
102 stack_frame.AddrPC.Mode = AddrModeFlat;
103 stack_frame.AddrFrame.Mode = AddrModeFlat;
104 stack_frame.AddrStack.Mode = AddrModeFlat;
105
106 // flush the output stream
107 fflush(stream);
108
109 // dump call stack
110 fprintf(stream, "\nCall-Stack Back-Trace:\n");
111 fprintf(stream, "----------------------\n");
112
113 // walk the stack
114 while (StackWalk64(machine_type, prc, thr, &stack_frame, &ctx, nullptr,
115 &SymFunctionTableAccess64, &SymGetModuleBase64, nullptr) != FALSE)
116 {
117 // get stack address
118 addr = stack_frame.AddrPC.Offset;
119
120 // print stack address
121 fprintf(stream, "0x%0*I64X", 2*sizeof(void*), addr);
122
123 // get symbol info to obtain function name
124 if(SymFromAddr(prc, addr, 0, &sim.si) != FALSE)
125 {
126 // print function name
127 fprintf(stream, ": '%s'", sim.si.Name);
128 }
129 else
130 {
131 // cannot query function name
132 fprintf(stream, ": ???");
133 }
134
135 // get source file line info
136 if(SymGetLineFromAddr64(prc, addr, &line_dsp, &img_line) != FALSE)
137 {
138 // print source file and line
139 fprintf(stream, " ['%s' @ %i]", img_line.FileName, img_line.LineNumber);
140 }
141
142 // next stack entry
143 fprintf(stream, "\n");
144 }
145
146 // flush stream
147 fflush(stream);
148 }

160 LONG WINAPI FeatWinExceptionFilter(LPEXCEPTION_POINTERS p)
161 {
162 // print an error header
163 fprintf(stderr, "\n>>>>> FATAL ERROR <<<<<\n\n");
164 fprintf(stderr, "Exception at 0x%0*I64X: ", 2*sizeof(void*), (DWORD64)p->ExceptionRecord->ExceptionAddress);
165
166 // print exception info
167 switch(p->ExceptionRecord->ExceptionCode)
168 {
169 // note: There are more than 20 exceptions, but we do not handle all of them here explicitly.
170 case EXCEPTION_ACCESS_VIOLATION:
171 // This one is also known as 'segmentation fault'...
172 fprintf(stderr, "Access Violation when ");
173 if(p->ExceptionRecord->ExceptionInformation[0] == 0)
174 fprintf(stderr, "reading from");
175 else
176 fprintf(stderr, "writing to");
177 fprintf(stderr, " 0x%0*I64X\n", 2*sizeof(void*), (DWORD64)p->ExceptionRecord->ExceptionInformation[1]);
178 break;
179 case EXCEPTION_ILLEGAL_INSTRUCTION:
180 fprintf(stderr, "Illegal Instruction\n");
181 break;
182 case EXCEPTION_IN_PAGE_ERROR:
183 fprintf(stderr, "Page Error\n");
184 break;
185 case EXCEPTION_STACK_OVERFLOW:
186 fprintf(stderr, "Stack Overflow\n");
187 break;
188 default:
189 // We do not handle all error types, so just dump the error code here...
190 fprintf(stderr, "Error 0x%08X\n", p->ExceptionRecord->ExceptionCode);
191 break;
192 }
193
194 // dump call stack
195 CONTEXT ctx = *p->ContextRecord;
196 dump_call_stack(ctx, stderr);
197
198 // flush
199 fprintf(stderr, "\n>>>>> TERMINATING <<<<<\n");
200 fflush(stderr);
201
202 // return
203 return EXCEPTION_EXECUTE_HANDLER;
204 }
