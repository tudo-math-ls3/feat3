197 virtual Status _apply_intern(VectorType& vec_sol)
198 {
199 IterationStats pre_iter(*this);
200 Statistics::add_solver_expression(std::make_shared<ExpressionStartSolve>(this->name()));
201
202 const MatrixType& matrix(this->_system_matrix);
203 const FilterType& filter(this->_system_filter);
204 VectorType& vec_v(this->_vec_v);
205 VectorType& vec_vh(this->_vec_vh);
206 VectorType& vec_z(this->_vec_z);
207 VectorType& vec_s(this->_vec_s);
208 VectorType& vec_t(this->_vec_t);
209 VectorType& vec_th(this->_vec_th);
210 VectorType& vec_r(this->_vec_r);
211 VectorType& vec_r0(this->_vec_r0);
212
213 DataType rho(0);
214 DataType rho_old(0);
215 DataType delta(0);
216 DataType alpha(0);
217 DataType theta(0);
218 DataType omega(0);
219 DataType beta(0);
220 DataType phi(0);
221 DataType psi(0);
222
223 vec_r0.copy(vec_r);
224
225 // set initial defect:
226 // r[0] := b - A*x[0]
227 Status status = this->_set_initial_defect(vec_r, vec_sol);
228 if(status != Status::progress)
229 {
230 pre_iter.destroy();
231 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), status, this->get_num_iter()));
232 return status;
233 }
234
235 auto dot_rho = vec_r.dot_async(vec_r);
236
237 // apply preconditioner to defect vector
238 if(!this->_apply_precond(vec_z, vec_r, filter))
239 {
240 pre_iter.destroy();
241 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
242 return Status::aborted;
243 }
244
245 vec_vh.copy(vec_z);
246
247 rho = dot_rho.wait();
248 rho_old = rho;
249
250 pre_iter.destroy();
251
252 // start iterating
253 while(status == Status::progress)
254 {
255 IterationStats stat(*this);
256
257 matrix.apply(vec_v, vec_vh);
258 filter.filter_def(vec_v);
259
260 auto dot_delta = vec_v.dot_async(vec_r0);
261
262 // apply preconditioner
263 if(!this->_apply_precond(vec_s, vec_v, filter))
264 {
265 stat.destroy();
266 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
267 return Status::aborted;
268 }
269
270 delta = dot_delta.wait();
271 alpha = rho / delta;
272
273 vec_th.axpy(vec_s, vec_z, -alpha);
274
275 matrix.apply(vec_t, vec_th);
276 filter.filter_def(vec_t);
277
278 vec_sol.axpy(vec_vh, vec_sol, alpha);
279 vec_r.axpy(vec_v, vec_r, -alpha);
280 auto norm_def_half = vec_r.norm2_async();
281
282 auto dot_theta = vec_t.dot_async(vec_r);
283 auto dot_phi = vec_t.dot_async(vec_t);
284 auto dot_psi = vec_t.dot_async(vec_r0);
285
286 // apply preconditioner
287 if(!this->_apply_precond(vec_z, vec_t, filter))
288 {
289 stat.destroy();
290 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
291 return Status::aborted;
292 }
293
294 theta = dot_theta.wait();
295 phi = dot_phi.wait();
296 psi = dot_psi.wait();
297
298 // Check if we are already converged or failed after the "half" update
299 {
300 Status status_half(Status::progress);
301
302 DataType def_old(this->_def_cur);
303 DataType def_half(norm_def_half.wait());
304
305 // ensure that the defect is neither NaN nor infinity
306 if(!Math::isfinite(def_half))
307 {
308 status_half = Status::aborted;
309 }
310 // is diverged?
311 else if(this->is_diverged(def_half))
312 {
313 this->_def_cur = def_half;
314 status_half = Status::diverged;
315 }
316 // is converged?
317 else if(this->is_converged(def_half))
318 {
319 this->_def_cur = def_half;
320 status_half = Status::success;
321 }
322
323 // If we break early, we still count this iteration and plot it if necessary
324 if(status_half != Status::progress)
325 {
326 ++this->_num_iter;
327 this->_def_prev = def_old;
328 // plot?
329 if(this->_plot_iter())
330 this->_plot_iter_line(this->_num_iter, this->_def_cur, def_old);
331
332 stat.destroy();
333 Statistics::add_solver_expression(
334 std::make_shared<ExpressionEndSolve>(this->name(), status_half, this->get_num_iter()));
335
336 return status_half;
337 }
338 }
339
340 omega = theta / phi;
341
342 vec_sol.axpy(vec_th, vec_sol, omega);
343 vec_r.axpy(vec_t, vec_r, -omega);
344
345 auto norm_def_cur = vec_r.norm2_async();
346
347 rho = -omega * psi;
348 vec_z.axpy(vec_z, vec_th, -omega);
349 beta = (rho / rho_old) * (alpha / omega);
350 rho_old = rho;
351 vec_vh.axpy(vec_s, vec_vh, -omega);
352 vec_vh.axpy(vec_vh, vec_z, beta);
353
354 status = this->_update_defect(norm_def_cur.wait());
355 if(status != Status::progress)
356 {
357 stat.destroy();
358 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), status, this->get_num_iter()));
359 return status;
360 }
361 }
362
363 // we should never reach this point...
364 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::undefined, this->get_num_iter()));
365 return Status::undefined;
366 }
