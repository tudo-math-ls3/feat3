195 virtual Status _apply_intern(VectorType& vec_sol)
196 {
197 IterationStats pre_iter(*this);
198 Statistics::add_solver_expression(std::make_shared<ExpressionStartSolve>(this->name()));
199
200 const MatrixType& matrix(this->_system_matrix);
201 const FilterType& filter(this->_system_filter);
202 VectorType& vec_v(this->_vec_v);
203 VectorType& vec_vh(this->_vec_vh);
204 VectorType& vec_z(this->_vec_z);
205 VectorType& vec_s(this->_vec_s);
206 VectorType& vec_t(this->_vec_t);
207 VectorType& vec_th(this->_vec_th);
208 VectorType& vec_r(this->_vec_r);
209 VectorType& vec_r0(this->_vec_r0);
210
211 DataType rho(0);
212 DataType rho_old(0);
213 DataType delta(0);
214 DataType alpha(0);
215 DataType theta(0);
216 DataType omega(0);
217 DataType beta(0);
218 DataType phi(0);
219 DataType psi(0);
220
221 vec_r0.copy(vec_r);
222
223 // set initial defect:
224 // r[0] := b - A*x[0]
225 Status status = this->_set_initial_defect(vec_r, vec_sol);
226 if(status != Status::progress)
227 {
228 pre_iter.destroy();
229 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), status, this->get_num_iter()));
230 return status;
231 }
232
233 auto dot_rho = vec_r.dot_async(vec_r);
234
235 // apply preconditioner to defect vector
236 if(!this->_apply_precond(vec_z, vec_r, filter))
237 {
238 pre_iter.destroy();
239 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
240 return Status::aborted;
241 }
242
243 vec_vh.copy(vec_z);
244
245 rho = dot_rho.wait();
246 rho_old = rho;
247
248 pre_iter.destroy();
249
250 // start iterating
251 while(status == Status::progress)
252 {
253 IterationStats stat(*this);
254
255 matrix.apply(vec_v, vec_vh);
256 filter.filter_def(vec_v);
257
258 auto dot_delta = vec_v.dot_async(vec_r0);
259
260 // apply preconditioner
261 if(!this->_apply_precond(vec_s, vec_v, filter))
262 {
263 stat.destroy();
264 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
265 return Status::aborted;
266 }
267
268 delta = dot_delta.wait();
269 alpha = rho / delta;
270
271 vec_th.copy(vec_z);
272 vec_th.axpy(vec_s, -alpha);
273
274 matrix.apply(vec_t, vec_th);
275 filter.filter_def(vec_t);
276
277 vec_sol.axpy(vec_vh, alpha);
278 vec_r.axpy(vec_v, -alpha);
279 auto norm_def_half = vec_r.norm2_async();
280
281 auto dot_theta = vec_t.dot_async(vec_r);
282 auto dot_phi = vec_t.dot_async(vec_t);
283 auto dot_psi = vec_t.dot_async(vec_r0);
284
285 // apply preconditioner
286 if(!this->_apply_precond(vec_z, vec_t, filter))
287 {
288 stat.destroy();
289 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
290 return Status::aborted;
291 }
292
293 theta = dot_theta.wait();
294 phi = dot_phi.wait();
295 psi = dot_psi.wait();
296
297 // Check if we are already converged or failed after the "half" update
298 {
299 Status status_half(Status::progress);
300
301 DataType def_old(this->_def_cur);
302 DataType def_half(norm_def_half.wait());
303
304 // ensure that the defect is neither NaN nor infinity
305 if(!Math::isfinite(def_half))
306 {
307 status_half = Status::aborted;
308 }
309 // is diverged?
310 else if(this->is_diverged(def_half))
311 {
312 this->_def_cur = def_half;
313 status_half = Status::diverged;
314 }
315 // is converged?
316 else if(this->is_converged(def_half))
317 {
318 this->_def_cur = def_half;
319 status_half = Status::success;
320 }
321
322 // If we break early, we still count this iteration and plot it if necessary
323 if(status_half != Status::progress)
324 {
325 ++this->_num_iter;
326 this->_def_prev = def_old;
327 // plot?
328 if(this->_plot_iter())
329 this->_plot_iter_line(this->_num_iter, this->_def_cur, def_old);
330
331 stat.destroy();
332 Statistics::add_solver_expression(
333 std::make_shared<ExpressionEndSolve>(this->name(), status_half, this->get_num_iter()));
334
335 return status_half;
336 }
337 }
338
339 omega = theta / phi;
340
341 vec_sol.axpy(vec_th, omega);
342 vec_r.axpy(vec_t, -omega);
343
344 auto norm_def_cur = vec_r.norm2_async();
345
346 rho = -omega * psi;
347 vec_z.scale(vec_z, -omega);
348 vec_z.axpy(vec_th);
349 beta = (rho / rho_old) * (alpha / omega);
350 rho_old = rho;
351 vec_vh.axpy(vec_s, -omega);
352 vec_vh.scale(vec_vh, beta);
353 vec_vh.axpy(vec_z);
354
355 status = this->_update_defect(norm_def_cur.wait());
356 if(status != Status::progress)
357 {
358 stat.destroy();
359 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), status, this->get_num_iter()));
360 return status;
361 }
362 }
363
364 // we should never reach this point...
365 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::undefined, this->get_num_iter()));
366 return Status::undefined;
367 }
