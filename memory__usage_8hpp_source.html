65 void stamp()
66 {
67#if defined(__linux)
68 String line;
69 std::ifstream status_file("/proc/self/status");
70 if (!status_file.is_open())
71 XABORTM("could not open /proc/self/status!");
72
73 /* man 5 proc:
74 * /proc/[pid]/status
75 * Provides much of the information in /proc/[pid]/stat and
76 * /proc/[pid]/statm in a format that's easier for humans to parse.
77 * The fields are as follows:
78 * Name: Command run by this process.
79 * State: Current state of the process. One of "R (running)", "S (sleeping)", "D (disk sleep)", "T (stopped)", "T (trac-
80 * ing stop)", "Z (zombie)", or "X (dead)".
81 * Tgid: Thread group ID (i.e., Process ID).
82 * Pid: Thread ID (see gettid(2)).
83 * PPid: PID of parent process.
84 * TracerPid: PID of process tracing this process (0 if not being traced).
85 * Uid, Gid: Real, effective, saved set, and file system UIDs (GIDs).
86 * FDSize: Number of file descriptor slots currently allocated.
87 * Groups: Supplementary group list.
88 *! VmPeak: Peak virtual memory size.
89 *! VmSize: Virtual memory size.
90 * VmLck: Locked memory size (see mlock(3)).
91 *! VmHWM: Peak resident set size ("high water mark").
92 *! VmRSS: Resident set size.
93 * VmData, VmStk, VmExe: Size of data, stack, and text segments.
94 * VmLib: Shared library code size.
95 * VmPTE: Page table entries size (since Linux 2.6.10).
96 * Threads: Number of threads in process containing this thread.
97 * [...]
98 */
99 while (std::getline(status_file, line))
100 {
101 if (line.starts_with("VmRSS"))
102 {
103 std::deque<String> v = line.split_by_whitespaces();
104 XASSERTM(v.back() == "kB", "get_memory_usage: unit mismatch!");
105 v.at(v.size() - 2).parse(_current_physical);
106 _current_physical *= 1024;
107 continue;
108 }
109
110 if (line.starts_with("VmHWM"))
111 {
112 std::deque<String> v = line.split_by_whitespaces();
113 XASSERTM(v.back() == "kB", "get_memory_usage: unit mismatch!");
114 v.at(v.size() - 2).parse(_peak_physical);
115 _peak_physical *= 1024;
116 continue;
117 }
118
119 if (line.starts_with("VmSize"))
120 {
121 std::deque<String> v = line.split_by_whitespaces();
122 XASSERTM(v.back() == "kB", "get_memory_usage: unit mismatch!");
123 v.at(v.size() - 2).parse(_current_virtual);
124 _current_virtual *= 1024;
125 continue;
126 }
127
128 if (line.starts_with("VmPeak"))
129 {
130 std::deque<String> v = line.split_by_whitespaces();
131 XASSERTM(v.back() == "kB", "get_memory_usage: unit mismatch!");
132 v.at(v.size() - 2).parse(_peak_virtual);
133 _peak_virtual *= 1024;
134 continue;
135 }
136
137 if (line.starts_with("VmSwap"))
138 {
139 std::deque<String> v = line.split_by_whitespaces();
140 XASSERTM(v.back() == "kB", "get_memory_usage: unit mismatch!");
141 v.at(v.size() - 2).parse(_current_swap);
142 _current_swap *= 1024;
143 continue;
144 }
145 }
146 status_file.close();
147
148#elif defined(_WIN32)
149
150 unsigned long long work_set_size(0ull), work_set_size_peak(0ull), page_file_usage(0ull), page_file_usage_peak(0ull);
151
152 Windows::query_memory_usage(work_set_size, work_set_size_peak, page_file_usage, page_file_usage_peak);
153
154 _current_physical = std::size_t(work_set_size);
155 _current_virtual = std::size_t(page_file_usage);
156 _peak_physical = std::size_t(work_set_size_peak);
157 _peak_virtual = std::size_t(page_file_usage_peak);
158 _current_swap = std::size_t(0);
159
160#elif defined(__unix__)
161 // see:
162 // https://linux.die.net/man/2/getrusage
163 // https://www.freebsd.org/cgi/man.cgi?query=getrusage
164 struct rusage r_usage;
165 if (0 != getrusage(RUSAGE_SELF, &r_usage))
166 XABORTM("Error in getrusage call!");
167 _peak_physical = (std::size_t)r_usage.ru_maxrss;
168 _peak_virtual = (std::size_t)r_usage.ru_maxrss;
169 _current_physical = (std::size_t)r_usage.ru_maxrss;
170 _current_virtual = (std::size_t)r_usage.ru_maxrss;
171 _current_swap = std::size_t(0);
172 // 'ru_maxrss' is given in kilobytes
173 _peak_physical *= 1024u;
174 _peak_virtual *= 1024u;
175 _current_physical *= 1024u;
176 _current_virtual *= 1024u;
177#endif
178 }

211 String get_formatted_memory_usage() const
212 {
213 String r;
214 r += String("Current real:").pad_back(20) + stringify(_current_physical / 1024 / 1024) + " MByte\n";
215 r += String("Peak real:").pad_back(20) + stringify(_peak_physical / 1024 / 1024) + " MByte\n";
216 r += String("Current virtual:").pad_back(20) + stringify(_current_virtual / 1024 / 1024) + " MByte\n";
217 r += String("Peak virtual:").pad_back(20) + stringify(_peak_virtual / 1024 / 1024) + " MByte\n";
218 r += String("Current swap:").pad_back(20) + stringify(_current_swap / 1024 / 1024) + " MByte\n";
219
220 return r;
