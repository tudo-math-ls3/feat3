194 explicit BiCGStabL(const String& section_name, PropertyMap* section,
195 const MatrixType& matrix, const FilterType& filter,
196 std::shared_ptr<PrecondType> precond = nullptr)
197 :
198 BaseClass("BiCGStabL", section_name, section, precond),
199 _system_matrix(matrix),
200 _system_filter(filter),
201 _precon_variant(BiCGStabLPreconVariant::left)
202 {
203 // set communicator by system matrix
204 this->_set_comm_by_matrix(matrix);
205 // Check if we have set _p_variant
206 auto p_variant_p = section->query("precon_variant");
207 if(p_variant_p.second && !p_variant_p.first.parse(this->_precon_variant))
208 throw ParseError(section_name + ".precon_variant", p_variant_p.first, "one of: left, right");
209
210 //Check if we have set the solver parameter _l
211 std::pair<String , bool> l_pair = section->query("polynomial_degree");
212
213 int l_pm = 2; //use default l, if the parameter is not given
214 if (l_pair.second && (!l_pair.first.parse(l_pm) || (l_pm <= 0)))
215 throw ParseError(section_name + ".polynomial_degree", l_pair.first, "a positive integer");
216
217 this->_l = l_pm;
218 this->set_plot_name("BiCGStab(" + stringify(_l) + ")");
219 }

261 virtual Status correct(VectorType& vec_sol, const VectorType& vec_rhs) override
262 {
263 // compute initial defect
264 this->_system_matrix.apply(this->_vec_rj_hat.at(0), vec_sol, vec_rhs, -DataType(1));
265 this->_system_filter.filter_def(this->_vec_rj_hat.at(0));
266
267 // apply solver
268 this->_status = _apply_intern(vec_sol, vec_rhs);
269
270 // plot summary
271 this->plot_summary();
272
273 // return status
274 return this->_status;
275 }

286 // apply solver
287 this->_status = _apply_intern(vec_cor, vec_def);
288
289 // plot summary
290 this->plot_summary();
291
292 // return status
293 return this->_status;
294 }

303 virtual void set_precon_variant(BiCGStabLPreconVariant precon_variant)
304 {

309 protected:
321 Status _apply_intern(VectorType& vec_sol, const VectorType& vec_rhs)
322 {
323 IterationStats pre_iter(*this);
324 Statistics::add_solver_expression(std::make_shared<ExpressionStartSolve>(this->name()));
325
326
327 int l (_l);
328
329 //choose vec_r0_tilde as vec_r0_hat (vec_rj_hat[0])
330 _vec_r0_tilde.copy(_vec_rj_hat.at(0));
331
332 std::vector<DataType> tau ( Index( l*(l-1) ) , DataType(0)); //vector for tau_ij (j from 0 to l-1 , i from 0 to j-1, so LESS THAN HALF of the vector gets used)
333 //tau_ij = tau[j*(l-1) + i] (column major ordering)
334
335 std::vector<DataType> sigma ( Index(l) , DataType(0) ); //0-offset in implementation; 1-offset in pseudo-code
336 std::vector<DataType> gamma_j ( Index(l) , DataType(0) ); //0-offset in implementation; 1-offset in pseudo-code
337 std::vector<DataType> gamma_prime ( Index(l) , DataType(0) ); //0-offset in implementation; 1-offset in pseudo-code
338 std::vector<DataType> gamma_prime_prime ( Index(l) , DataType(0) ); //0-offset in implementation; 1-offset in pseudo-code
339
340 DataType rho_0(1), alpha(0), omega(1), rho_1(0), beta(0), gamma(0);
341
342
343 const MatrixType& mat_sys(this->_system_matrix);
344 const FilterType& fil_sys(this->_system_filter);
345 Status status(Status::progress);
346
347 // Compute initial defect
348 status = this->_set_initial_defect(_vec_rj_hat.at(0), vec_sol);
349
350 if (_precon_variant == BiCGStabLPreconVariant::left)
351 {
352 _vec_pc.copy(_vec_rj_hat.at(0));
353 if(!this->_apply_precond(_vec_rj_hat.at(0), _vec_pc, fil_sys))
354 {
355 Statistics::add_solver_expression(
356 std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
357 return Status::aborted;
358 }
359 }
360
361
362 pre_iter.destroy();
363 while(status == Status::progress)
364 {
365 IterationStats stat(*this);
366 rho_0 *= (-omega);
367
368
369 for( int j(0); j < l ; j++)
370 {
371 rho_1 = _vec_rj_hat.at( Index(j) ).dot(_vec_r0_tilde);
372 beta = alpha * rho_1 / rho_0;
373 rho_0 = rho_1;
374 for ( int i(0) ; i <= j ; i++)
375 {
376 _vec_uj_hat.at( Index(i) ).axpy(_vec_uj_hat.at( Index(i) ), _vec_rj_hat.at( Index(i) ), -beta );
377 }
378
379
380
381 if (_precon_variant == BiCGStabLPreconVariant::left)
382 {
383 mat_sys.apply(_vec_pc, _vec_uj_hat.at(Index(j)) );
384 fil_sys.filter_def(_vec_pc);
385 if(!this->_apply_precond(_vec_uj_hat.at( Index(j+1) ), _vec_pc, fil_sys))
386 {
387 Statistics::add_solver_expression(
388 std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
389 return Status::aborted;
390 }
391 }
392 else
393 {
394 if(!this->_apply_precond(_vec_pc, _vec_uj_hat.at( Index(j) ), fil_sys))
395 {
396 Statistics::add_solver_expression(
397 std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
398 return Status::aborted;
399 }
400 mat_sys.apply(_vec_uj_hat.at( Index(j+1) ), _vec_pc);
401 fil_sys.filter_def(_vec_uj_hat.at( Index(j+1) ));
402 }
403
404 gamma = _vec_uj_hat.at( Index(j+1) ).dot(_vec_r0_tilde);
405 alpha = rho_0 / gamma;
406
407 for ( int i(0) ; i <= j ; i++)
408 {
409 _vec_rj_hat.at(Index(i)).axpy( _vec_uj_hat.at( Index(i+1) ) , _vec_rj_hat.at( Index(i) ), -alpha );
410 }
411
412 if (_precon_variant == BiCGStabLPreconVariant::left)
413 {
414
415 mat_sys.apply( _vec_pc, _vec_rj_hat.at(Index(j)) );
416 fil_sys.filter_def(_vec_pc);
417 if(!this->_apply_precond(_vec_rj_hat.at( Index(j+1) ) , _vec_pc, fil_sys))
418 {
419 Statistics::add_solver_expression(
420 std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
421 return Status::aborted;
422 }
423 }
424 else
425 {
426 if(!this->_apply_precond(_vec_pc, _vec_rj_hat.at(Index(j)), fil_sys))
427 {
428 Statistics::add_solver_expression(
429 std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
430 return Status::aborted;
431 }
432 mat_sys.apply(_vec_rj_hat.at(Index(j+1)), _vec_pc);
433 fil_sys.filter_def( _vec_rj_hat.at(Index(j+1)) );
434 }
435
436 vec_sol.axpy(_vec_uj_hat.at(0), vec_sol, alpha);
437 }
438 for ( int j(0) ; j < l ; j++) //changed counter in comparison to pseudo code
439 {
440
441 for ( int i(0); i<j ; i++) //changed counter
442 {
443 tau.at( Index(j*(l-1) + i) ) = ( _vec_rj_hat.at( Index(i+1) ).dot( _vec_rj_hat.at( Index(j+1) ) ) ) / sigma.at(Index(i));
444 _vec_rj_hat.at(Index(j+1)).axpy(_vec_rj_hat.at(Index(i+1)), _vec_rj_hat.at(Index(j+1)), -tau.at( Index(j*(l-1) + i) ) );
445 }
446 sigma.at(Index(j)) = _vec_rj_hat.at( Index(j+1) ).dot( _vec_rj_hat.at(Index(j+1)) );
447 gamma_prime.at(Index(j)) = ( _vec_rj_hat.at(0).dot( _vec_rj_hat.at(Index(j+1)) ) ) / sigma.at(Index(j));
448 }
449 gamma_j.at(Index(l-1)) = gamma_prime.at(Index(l-1));
450 omega = gamma_j.at(Index(l-1));
451
452 for ( int j(l-2) ; j >= 0 ; j--) //changed counter
453 {
454 gamma_j.at(Index(j)) = gamma_prime.at(Index(j));
455 for ( int i(j+1) ; i < l ; i++)
456 {
457 gamma_j.at(Index(j)) -= tau.at( Index(i*(l-1) + j) ) * gamma_j.at(Index(i));
458 }
459 }
460
461 for ( int j(0); j < l-1 ; j++) //changed counter
462 {
463 gamma_prime_prime.at( Index(j) ) = gamma_j.at( Index(j+1) );
464 for ( int i(j+1) ; i < l-1 ; i++) //changed counter
465 {
466 gamma_prime_prime.at(Index(j)) += tau.at(Index(i*(l-1) + j)) * gamma_j.at(Index(i+1));
467 }
468 }
469 vec_sol.axpy(_vec_rj_hat.at(0), vec_sol, gamma_j.at(0));
470 _vec_rj_hat.at(0).axpy(_vec_rj_hat.at(Index(l)), _vec_rj_hat.at(0), -gamma_prime.at(Index(l-1)));
471 _vec_uj_hat.at(0).axpy(_vec_uj_hat.at(Index(l)), _vec_uj_hat.at(0), -gamma_j.at(Index(l-1)));
472
473 for ( int j(1) ; j<l; j++)
474 {
475 _vec_uj_hat.at(0).axpy(_vec_uj_hat.at(Index(j)), _vec_uj_hat.at(0), -gamma_j.at(Index(j-1)));
476 vec_sol.axpy(_vec_rj_hat.at(Index(j)), vec_sol, gamma_prime_prime.at(Index(j-1)));
477 _vec_rj_hat.at(0).axpy(_vec_rj_hat.at(Index(j)), _vec_rj_hat.at(0), -gamma_prime.at(Index(j-1)));
478 }
479 // Compute defect norm
480 if (_precon_variant == BiCGStabLPreconVariant::left)
481 {
482 mat_sys.apply(_vec_pc, vec_sol);
483 fil_sys.filter_def(_vec_pc);
484 _vec_pc.axpy( vec_rhs, _vec_pc , -1);
485 fil_sys.filter_def(_vec_pc);
486 status = this->_set_new_defect(_vec_pc, vec_sol);
487 }
488 else
489 {
490 status = this->_set_new_defect(_vec_rj_hat.at(0), vec_sol);
491 }
492
493 if(status != Status::progress)
494 {
495 stat.destroy();
496 Statistics::add_solver_expression(
497 std::make_shared<ExpressionEndSolve>(this->name(), status, this->get_num_iter()));
498 if (_precon_variant == BiCGStabLPreconVariant::right)
499 {
500 if(!this->_apply_precond(_vec_pc, vec_sol, fil_sys))
501 {
502 Statistics::add_solver_expression(
503 std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
504 return Status::aborted;
505 }
506 vec_sol.copy(_vec_pc);
507 }
508 return status;
509 }
510
511 }
512 // we should never reach this point...
513 Statistics::add_solver_expression(
514 std::make_shared<ExpressionEndSolve>(this->name(), Status::undefined, this->get_num_iter()));
515 return Status::undefined;
516 }
517 };
