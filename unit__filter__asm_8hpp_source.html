30 template<typename DomainLevel_>
31 Assembly::UnitFilterAssembler<typename DomainLevel_::MeshType> create_unit_filter_asm(const DomainLevel_& dom_level, const std::deque<String>& mesh_part_names)
32 {
33
34 Assembly::UnitFilterAssembler<typename DomainLevel_::MeshType> unit_asm;
35 // loop over all mesh parts
36 for(const auto& mp_name : mesh_part_names)
37 {
38 auto* mesh_part_node = dom_level.get_mesh_node()->find_mesh_part_node(mp_name);
39 XASSERT(mesh_part_node != nullptr);
40
41 // let's see if we have that mesh part
42 // if it is nullptr, then our patch is not adjacent to that boundary part
43 auto* mesh_part = mesh_part_node->get_mesh();
44 if (mesh_part != nullptr)
45 {
46 // add to boundary assembler
47 unit_asm.add_mesh_part(*mesh_part);
48 }
49 }
50 return unit_asm;
51 }

62 template<typename DomainLevel_>
63 Assembly::UnitFilterAssembler<typename DomainLevel_::MeshType> create_unit_filter_asm(const DomainLevel_& dom_level, const String& mesh_part_names)
64 {
65 std::deque<String> mp_names;
66 if(mesh_part_names == "*")
67 mp_names = dom_level.get_mesh_node()->get_mesh_part_names(true);
68 else
69 mp_names = mesh_part_names.split_by_whitespaces();
70
71 return create_unit_filter_asm(dom_level, mp_names);
72 }
