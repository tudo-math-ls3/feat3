294 class ImageIterator
295 {
296 // The CompositeAdjactor class needs to be a friend as it calls the protected constructors.
297 friend class CompositeAdjactor<Adj1_, Adj2_>;
298
299 private:
300
302 typedef typename Adj1_::ImageIterator Iterator1;
304 typedef typename Adj2_::ImageIterator Iterator2;
305
307 const Adj2_* _adj2;
308
310 mutable Iterator1 _cur1;
312 mutable Iterator1 _end1;
314 mutable Iterator2 _cur2;
316 mutable Iterator2 _end2;
317
318 protected:
320 // constructor for image_begin()
321 ImageIterator(
322 const Adj1_* adj1,
323 const Adj2_* adj2,
324 Index domain_node)
325 :
326 _adj2(adj2),
327 _cur1(adj1->image_begin(domain_node)),
328 _end1(adj1->image_end(domain_node)),
329 _cur2(),
330 _end2()
331 {
332 if(_cur1 != _end1)
333 {
334 _cur2 = adj2->image_begin(*_cur1);
335 _end2 = adj2->image_end(*_cur1);
336 }
337 }
338
339 // constructor for image_end()
340 ImageIterator(
341 const Adj1_* adj1,
342 Index domain_node)
343 :
344 _adj2(nullptr),
345 _cur1(adj1->image_end(domain_node)),
346 _end1(_cur1),
347 _cur2(),
348 _end2()
349 {
350 }
352
353 public:
355 inline ImageIterator() :
356 _adj2(nullptr),
357 _cur1(),
358 _end1(),
359 _cur2(),
360 _end2()
361 {
362 }
363
365 inline ImageIterator(const ImageIterator& other) :
366 _adj2(other._adj2),
367 _cur1(other._cur1),
368 _cur2(other._cur2),
369 _end1(other._end1),
370 _end2(other._end2)
371 {
372 }
373
375 inline Index operator*() const
376 {
377 return *_cur2;
378 }
379
381 inline bool operator!=(const ImageIterator& other) const
382 {
383 return (_cur1 != other._cur1) || (_cur2 != other._cur2);
384 }
385
387 ImageIterator& operator++()
388 {
389 // increment the second iterator; if it did not reach the end then we have a valid position
390 if(++_cur2 != _end2)
391 {
392 return *this;
393 }
394
395 // second iterator reached its end, so keep incrementing the first iterator until we find the next
396 // non-empty second adjacency list or until we reach its end
397 while(++_cur1 != _end1)
398 {
399 // reset second iterator
400 _cur2 = _adj2->image_begin(*_cur1);
401 _end2 = _adj2->image_end(*_cur1);
402
403 // check whether the second adjacency list is empty, if not then we have a valid position
404 if(_cur2 != _end2)
405 {
406 return *this;
407 }
408 }
409
410 // If we come out here, then there are no more adjacencies left, so we need to set
411 // this iterator to the 'end' position - this setting must be compatible to the iterator
412 // returned by the image_end() function of the CompositeAdjactor class!
413 // Note: see the second protected constructor of this class for the following definition.
414 _cur2 = Iterator2();
415 _end2 = _cur2;
416
417 return *this;
418 }
419 }; // class CompositeAdjactor::ImageIterator

430 inline CompositeAdjactor(
431 const Adj1_& adjactor1,
432 const Adj2_& adjactor2)
433 :
434 _adj1(adjactor1),
435 _adj2(adjactor2)
436 {
437 // Ensure that the number of image nodes in the first adjactor is not greater than the number of domain
438 // nodes in the second adjactor; otherwise the composite adjactor is ill-formed.
439 XASSERTM(_adj1.get_num_nodes_image() <= _adj2.get_num_nodes_domain(), "Composite Adjactor is ill-formed");
440 }
441
443 inline Index get_num_nodes_domain() const
444 {
445 // The number of domain nodes of the composite adjactor is given by the number of domain nodes of the
446 // first adjactor in the composition.
447 return _adj1.get_num_nodes_domain();
448 }
449
451 inline Index get_num_nodes_image() const
452 {
453 // The number of image nodes of the composite adjactor is given by the number of image nodes of the
454 // second adjactor in the composition.
455 return _adj2.get_num_nodes_image();
456 }
457
459 inline ImageIterator image_begin(Index domain_node) const
460 {
461 ASSERTM(domain_node < get_num_nodes_domain(), "domain_node out of valid range");
462 return ImageIterator(&_adj1, &_adj2, domain_node);
463 }
464
466 inline ImageIterator image_end(Index domain_node) const
467 {
468 ASSERTM(domain_node < get_num_nodes_domain(), "domain_node out of valid range");
469 return ImageIterator(&_adj1, domain_node);
470 }
471
477 inline const Adjactor1& get_adjactor1() const

480 }
481
487 inline const Adjactor2& get_adjactor2() const
