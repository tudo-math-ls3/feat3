42 static DataType fx00(Index i)
43 {
44 return DataType(0.2) * DataType(i);
45 }
46
47 static DataType fx01(Index i)
48 {
49 return DataType(2) - DataType(i);
50 }
51
52 static DataType fx1(Index i)
53 {
54 return DataType(0.01) * Math::sqr(DataType(i+1));
55 }
56
57 static DataType fy00(Index i)
58 {
59 return -DataType(3) + DataType(2*i);
60 }
61
62 static DataType fy01(Index i)
63 {
64 return DataType(1) + Math::sqrt(DataType(i+1));
65 }
66
67 static DataType fy1(Index i)
68 {
69 return DataType(1) + DataType(7) / DataType(i+1);
70 }
71
72 static DataType fz00(Index i)
73 {
74 return Math::sqr(DataType(i+1));
75 }
76
77 static DataType fz01(Index i)
78 {
79 return -DataType(5) - Math::sqrt(DataType(i+1));
80 }
81
82 static DataType fz1(Index i)
83 {
84 return - DataType(7) / DataType(i+1);
85 }

87 // generate test-vector x
88 static MetaVector gen_vector_x(Index n00, Index n01, Index n1)
89 {
90 DenseVector<DataType, IndexType> r00(n00), r01(n01), r1(n1);
91
92 // fill vectors
93 for(Index i(0); i < n00; ++i)
94 r00(i, fx00(i));
95 for(Index i(0); i < n01; ++i)
96 r01(i, fx01(i));
97 for(Index i(0); i < n1; ++i)
98 r1(i, fx1(i));
99
100 // construct data vector
101 MetaVector x;
102 x.template at<0>().template at<0>().convert(r00);
103 x.template at<0>().template at<1>().convert(r01);
104 x.template at<1>().convert(r1);
105 return x;
106 }
