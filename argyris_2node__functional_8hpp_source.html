86 void operator()(NodeData_& node_data, const Function_& function) const
87 {
88 static_assert(std::is_base_of<Analytic::Function, Function_>::value, "invalid function object");
89 static_assert(Function_::can_value, "function cannot compute values");
90 static_assert(Function_::can_grad, "function cannot compute gradients");
91 static_assert(Function_::can_hess, "function cannot compute hessians");
92
93 // declare our evaluation traits
94 typedef Analytic::EvalTraits<DataType_, Function_> FuncEvalTraits;
95
96 // declare function evaluator
97 typename Function_::template Evaluator<FuncEvalTraits> func_eval(function);
98
99 // compute trafo data
100 DomainPointType dom_point(DataType_(0));
101 TrafoEvalData trafo_data;
102 _trafo_eval(trafo_data, dom_point);
103
104 // evaluate function
105 const auto func_value = func_eval.value(trafo_data.img_point);
106 const auto func_grad = func_eval.gradient(trafo_data.img_point);
107 const auto func_hess = func_eval.hessian(trafo_data.img_point);
108
109 // set node functional values
110 node_data[0] = func_value;
111 node_data[1] = func_grad[0];
112 node_data[2] = func_grad[1];
113 node_data[3] = func_hess[0][0];
114 node_data[4] = func_hess[1][1];
115 node_data[5] = func_hess[0][1];
116 }

178 template<typename NodeData_, typename Function_>
179 void operator()(NodeData_& node_data, const Function_& function) const
180 {
181 static_assert(std::is_base_of<Analytic::Function, Function_>::value, "invalid function object");
182 static_assert(Function_::can_grad, "function cannot compute gradients");
183
184 // declare our evaluation traits
185 typedef Analytic::EvalTraits<DataType_, Function_> FuncEvalTraits;
186
187 // declare function evaluator
188 typename Function_::template Evaluator<FuncEvalTraits> func_eval(function);
189
190 DomainPointType dom_point(DataType_(0.5));
191 TrafoEvalData trafo_data;
192
193 _trafo_eval(trafo_data, dom_point);
194
195 // compute edge normal
196 DataType_ dnx = +trafo_data.jac_mat(1,0);
197 DataType_ dny = -trafo_data.jac_mat(0,0);
198 DataType_ dnl = DataType_(1) / Math::sqrt(dnx*dnx + dny*dny);
199 dnx *= dnl;
200 dny *= dnl;
201
202 // evaluate function gradient
203 const auto func_grad = func_eval.gradient(trafo_data.img_point);
204
205 // set edge normal derivative
206 node_data[0] = dnx * func_grad[0] + dny * func_grad[1];
207 }
