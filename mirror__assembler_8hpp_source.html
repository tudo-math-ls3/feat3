28 template<
29 typename Space_,
30 typename MeshPart_,
31 int shape_dim_>
32 struct DofMirrorHelper
33 {
34 static Index count(const Space_& space, const MeshPart_& mesh_part)
35 {
36 // fetch the target set for this dimension
37 const typename MeshPart_::template TargetSet<shape_dim_>::Type&
38 target_set(mesh_part.template get_target_set<shape_dim_>());
39 if(target_set.get_num_entities() <= 0)
40 return 0;
41
42 // create a dof-assignment object
43 typename Space_::template DofAssignment<shape_dim_>::Type dof_assign(space);
44 if(dof_assign.get_max_assigned_dofs() <= 0)
45 return 0;
46
47 // loop over all target indices
48 Index num_entities = target_set.get_num_entities();
49 Index count(0);
50 for(Index i(0); i < num_entities; ++i)
51 {
52 dof_assign.prepare(target_set[i]);
53 count += Index(dof_assign.get_num_assigned_dofs());
54 dof_assign.finish();
55 }
56
57 return count;
58 }
59
60 template<typename IT_>
61 static Index fill(IT_ idx[], Index offset, const Space_& space, const MeshPart_& mesh_part)
62 {
63 // fetch the target set for this dimension
64 const typename MeshPart_::template TargetSet<shape_dim_>::Type&
65 target_set(mesh_part.template get_target_set<shape_dim_>());
66 if(target_set.get_num_entities() <= 0)
67 return offset;
68
69 // create a dof-assignment object
70 typename Space_::template DofAssignment<shape_dim_>::Type dof_assign(space);
71 if(dof_assign.get_max_assigned_dofs() <= 0)
72 return offset;
73
74 // loop over all target indices
75 Index num_entities = target_set.get_num_entities();
76 for(Index i(0); i < num_entities; ++i)
77 {
78 dof_assign.prepare(target_set[i]);
79 int num_assign(dof_assign.get_num_assigned_dofs());
80 for(int j(0); j < num_assign; ++j, ++offset)
81 {
82 idx[offset] = IT_(dof_assign.get_index(j));
83 }
84 dof_assign.finish();
85 }
86
87 return offset;
88 }

91 template<
92 typename Space_,
93 typename MeshPart_,
94 int shape_dim_ = Space_::shape_dim>
95 struct DofMirrorHelpWrapper
96 {
97 static Index count(const Space_& space, const MeshPart_& mesh_part)
98 {
99 // recursive call
100 return DofMirrorHelpWrapper<Space_, MeshPart_, shape_dim_ - 1>::count(space, mesh_part) +
101 DofMirrorHelper<Space_, MeshPart_, shape_dim_>::count(space, mesh_part);
102 }
103
104 template<typename IT_>
105 static Index fill(IT_ idx[], const Space_& space, const MeshPart_& mesh_part)
106 {
107 Index offset = DofMirrorHelpWrapper<Space_, MeshPart_, shape_dim_ - 1>::fill(idx, space, mesh_part);
108 return DofMirrorHelper<Space_, MeshPart_, shape_dim_>::fill(idx, offset, space, mesh_part);
109 }
110 };

151 template<
152 typename DataType_,
153 typename IndexType_,
154 typename Space_,
155 typename MeshPart_>
156 static void assemble_mirror(
157 LAFEM::VectorMirror<DataType_, IndexType_>& vec_mirror,
158 const Space_& space, const MeshPart_& mesh_part)
159 {
160 // count number of dofs in mirror
161 const Index count = Intern::DofMirrorHelpWrapper<Space_, MeshPart_>::count(space, mesh_part);
162
163 // allocate mirror in main memory
164 vec_mirror = LAFEM::VectorMirror<DataType_, IndexType_>(space.get_num_dofs(), count);
165
166 // fill mirror indices
167 if(count > Index(0))
168 {
169 Intern::DofMirrorHelpWrapper<Space_, MeshPart_>::fill(vec_mirror.indices(), space, mesh_part);
170 }
171 }
