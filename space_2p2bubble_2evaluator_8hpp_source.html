109 template<typename EvalData_>
110 void eval_ref_values(
111 EvalData_& data,
112 const DomainPointType& point) const
113 {
114 const DataType x = point[0];
115 const DataType y = point[1];
116 // vertex dofs
117 data.phi[0].ref_value = (DataType(1) + DataType(3)*x*y - DataType(2)*(x + y))*(DataType(1) - x - y);
118 data.phi[1].ref_value = x*(DataType(2)*x + DataType(3)*y*(DataType(1) - x - y) - DataType(1));
119 data.phi[2].ref_value = y*(DataType(2)*y + DataType(3)*x*(DataType(1) - x - y) - DataType(1));
120 // edge dofs
121 data.phi[3].ref_value = DataType(4)*x*y*(DataType(3)*(x+y)-DataType(2));
122 data.phi[4].ref_value = DataType(4)*y*(DataType(3)*x-DataType(1))*(x+y-DataType(1));
123 data.phi[5].ref_value = DataType(4)*x*(DataType(3)*y-DataType(1))*(x+y-DataType(1));
124 // centre dof
125 data.phi[6].ref_value = DataType(27)*x*y*(DataType(1)-x-y);
126 }

137 template<typename EvalData_>
138 void eval_ref_gradients(
139 EvalData_& data,
140 const DomainPointType& point) const
141 {
142 const DataType x = point[0];
143 const DataType y = point[1];
144 // vertex dofs
145 data.phi[0].ref_grad[0] = y*(DataType(7)-DataType(3)*y-DataType(6)*x)+DataType(4)*x-DataType(3);
146 data.phi[0].ref_grad[1] = x*(DataType(7)-DataType(3)*x-DataType(6)*y)+DataType(4)*y-DataType(3);
147 data.phi[1].ref_grad[0] = DataType(3)*y*(DataType(1)-y-DataType(2)*x)+DataType(4)*x-DataType(1);
148 data.phi[1].ref_grad[1] = DataType(3)*x*(DataType(1)-x-DataType(2)*y);
149 data.phi[2].ref_grad[0] = DataType(3)*y*(DataType(1)-y-DataType(2)*x);
150 data.phi[2].ref_grad[1] = DataType(3)*x*(DataType(1)-x-DataType(2)*y)+DataType(4)*y-DataType(1);
151 // edge dofs
152 data.phi[3].ref_grad[0] = DataType(4)*y*(DataType(6)*x+DataType(3)*y-DataType(2));
153 data.phi[3].ref_grad[1] = DataType(4)*x*(DataType(3)*x+DataType(6)*y-DataType(2));
154 data.phi[4].ref_grad[0] = DataType(4)*y*(DataType(6)*x+DataType(3)*y-DataType(4));
155 data.phi[4].ref_grad[1] = DataType(4)*(DataType(3)*x-DataType(1))*(x+DataType(2)*y-DataType(1));
156 data.phi[5].ref_grad[0] = DataType(4)*(DataType(3)*y-DataType(1))*(y+DataType(2)*x-DataType(1));
157 data.phi[5].ref_grad[1] = DataType(4)*x*(DataType(3)*x+DataType(6)*y-DataType(4));
158 // centre dof
159 data.phi[6].ref_grad[0] = DataType(27)*y*(DataType(1)-y-DataType(2)*x);
160 data.phi[6].ref_grad[1] = DataType(27)*x*(DataType(1)-x-DataType(2)*y);
161 }

172 template<typename EvalData_>
173 void eval_ref_hessians(
174 EvalData_& data,
175 const DomainPointType& point) const
176 {
177 const DataType x = point[0];
178 const DataType y = point[1];
179 // vertex dofs
180 data.phi[0].ref_hess[0][0] = -DataType(6)*y+DataType(4);
181 data.phi[0].ref_hess[1][1] =-DataType(6)*x+DataType(4);
182 data.phi[0].ref_hess[0][1] = data.phi[0].ref_hess[1][0] = DataType(7)-DataType(6)*(x+y);
183 data.phi[1].ref_hess[0][0] = -DataType(6)*y+DataType(4);
184 data.phi[1].ref_hess[1][1] = -DataType(6)*x;
185 data.phi[1].ref_hess[0][1] = data.phi[1].ref_hess[1][0] = DataType(3)-DataType(6)*(x+y);
186 data.phi[2].ref_hess[0][0] = -DataType(6)*y;
187 data.phi[2].ref_hess[1][1] = -DataType(6)*x+DataType(4);
188 data.phi[2].ref_hess[0][1] = data.phi[2].ref_hess[1][0] = DataType(3)-DataType(6)*(x+y);
189 // edge dofs
190 data.phi[3].ref_hess[0][0] = DataType(24)*y;
191 data.phi[3].ref_hess[1][1] = DataType(24)*x;
192 data.phi[3].ref_hess[0][1] = data.phi[3].ref_hess[1][0] = DataType(8)*(DataType(3)*(x+y)-DataType(1));
193 data.phi[4].ref_hess[0][0] = DataType(24)*y;
194 data.phi[4].ref_hess[1][1] = DataType(24)*x-DataType(8);
195 data.phi[4].ref_hess[0][1] = data.phi[4].ref_hess[1][0] = DataType(8)*(DataType(3)*(x+y)-DataType(2));
196 data.phi[5].ref_hess[0][0] = DataType(24)*y-DataType(8);
197 data.phi[5].ref_hess[1][1] = DataType(24)*x;
198 data.phi[5].ref_hess[0][1] = data.phi[5].ref_hess[1][0] = DataType(8)*(DataType(3)*(x+y)-DataType(2));
199 // centre dof
200 data.phi[6].ref_hess[0][0] = -DataType(54)*y;
201 data.phi[6].ref_hess[1][1] = -DataType(54)*x;
202 data.phi[6].ref_hess[0][1] = data.phi[6].ref_hess[1][0] = DataType(27)*(DataType(1)-DataType(2)*(x+y));
203 }
