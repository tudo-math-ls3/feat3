68 template<typename DT_, typename IT_, typename SpaceV_, typename SpaceS_, int dim_, int nsc_>
69 static void assemble_matrix(
70 LAFEM::SparseMatrixBCSR<DT_, IT_, nsc_, nsc_>& matrix,
71 const LAFEM::DenseVectorBlocked<DT_, IT_, dim_>& convect,
72 const SpaceV_& space_v,
73 const SpaceS_& space_s,
74 const Cubature::DynamicFactory& cubature_factory,
75 const DT_ lambda,
76 const DT_ gamma = DT_(1),
77 const DT_ zeta = DT_(1)
78 )
79 {
80 // validate matrix and vector dimensions
81 XASSERTM(matrix.rows() == space_s.get_num_dofs(), "invalid matrix dimensions");
82 XASSERTM(matrix.columns() == space_s.get_num_dofs(), "invalid matrix dimensions");
83 XASSERTM(convect.size() == space_v.get_num_dofs(), "invalid vector size");
84
85 typedef LAFEM::DenseVectorBlocked<DT_, IT_, dim_> VectorType;
86 typedef LAFEM::SparseMatrixBCSR<DT_, IT_, nsc_, nsc_> MatrixType;
87
88 // define our assembly traits
89 // we will "abuse" the 'trial space' of this assembly space for the velocity space,
90 // the stress space is both the test and the 'real' trial space here
91 typedef AsmTraits2<DT_, SpaceS_, SpaceV_, TrafoTags::jac_det,
92 SpaceTags::value|SpaceTags::grad, SpaceTags::value|SpaceTags::grad> AsmTraits;
93
94 // out datatype
95 typedef typename AsmTraits::DataType DataType;
96
97 // fetch our trafo
98 const typename AsmTraits::TrafoType& trafo = space_s.get_trafo();
99
100 // create a trafo evaluator
101 typename AsmTraits::TrafoEvaluator trafo_eval(trafo);
102
103 // create the space evaluators
104 typename AsmTraits::TestEvaluator space_eval_s(space_s);
105 typename AsmTraits::TrialEvaluator space_eval_v(space_v);
106
107 // create the dof-mappings
108 typename AsmTraits::TestDofMapping dof_mapping_s(space_s);
109 typename AsmTraits::TrialDofMapping dof_mapping_v(space_v);
110
111 // create trafo evaluation data
112 typename AsmTraits::TrafoEvalData trafo_data;
113
114 // create space evaluation data
115 typename AsmTraits::TestEvalData space_data_s;
116 typename AsmTraits::TrialEvalData space_data_v;
117
118 // create cubature rule
119 typename AsmTraits::CubatureRuleType cubature_rule(Cubature::ctor_factory, cubature_factory);
120
121 // create matrix scatter-axpy
122 typename MatrixType::ScatterAxpy scatter_matrix(matrix);
123
124 // create convection gather-axpy
125 typename VectorType::GatherAxpy gather_conv(convect);
126
127 // get maximum number of local dofs
128 static constexpr int max_local_dofs_s = AsmTraits::max_local_test_dofs;
129 static constexpr int max_local_dofs_v = AsmTraits::max_local_trial_dofs;
130
131 // create local matrix data
132 typedef Tiny::Matrix<DataType, nsc_, nsc_> MatrixValue;
133 typedef Tiny::Matrix<MatrixValue, max_local_dofs_s, max_local_dofs_s> LocalMatrixType;
134 LocalMatrixType local_matrix;
135
136 // create local vector data
137 typedef Tiny::Vector<DataType, dim_> VectorValue;
138 typedef Tiny::Vector<VectorValue, max_local_dofs_v> LocalVectorType;
139
140 // local convection field dofs
141 LocalVectorType local_conv_dofs;
142
143 // our local velocity value
144 Tiny::Vector<DataType, dim_> loc_v;
145
146 // our local velocity gradient
147 Tiny::Matrix<DataType, dim_, dim_> loc_grad_v;
148
149 loc_v.format();
150 loc_grad_v.format();
151
152 // loop over all cells of the mesh
153 for(typename AsmTraits::CellIterator cell(trafo_eval.begin()); cell != trafo_eval.end(); ++cell)
154 {
155 // prepare trafo evaluator
156 trafo_eval.prepare(cell);
157
158 // prepare space evaluator
159 space_eval_s.prepare(trafo_eval);
160 space_eval_v.prepare(trafo_eval);
161
162 // initialize dof-mapping
163 dof_mapping_s.prepare(cell);
164 dof_mapping_v.prepare(cell);
165
166 // fetch number of local dofs
167 const int num_loc_dofs_s = space_eval_s.get_num_local_dofs();
168 const int num_loc_dofs_v = space_eval_v.get_num_local_dofs();
169
170 // gather our local convection dofs
171 local_conv_dofs.format();
172 gather_conv(local_conv_dofs, dof_mapping_v);
173
174 // format our local matrix and vector
175 local_matrix.format();
176
177 // loop over all quadrature points and integrate
178 for(int point(0); point < cubature_rule.get_num_points(); ++point)
179 {
180 // compute trafo data
181 trafo_eval(trafo_data, cubature_rule.get_point(point));
182
183 // compute basis function data
184 space_eval_s(space_data_s, trafo_data);
185 space_eval_v(space_data_v, trafo_data);
186
187 // pre-compute cubature weight
188 const DataType weight = trafo_data.jac_det * cubature_rule.get_weight(point);
189
190 // evaluate convection function and its gradient (if required)
191 loc_v.format();
192 loc_grad_v.format();
193 for(int i(0); i < num_loc_dofs_v; ++i)
194 {
195 loc_v.axpy(space_data_v.phi[i].value, local_conv_dofs[i]);
196 loc_grad_v.add_outer_product(local_conv_dofs[i], space_data_v.phi[i].grad);
197 }
198
199 // test function loop
200 for(int i(0); i < num_loc_dofs_s; ++i)
201 {
202 // trial function loop
203 for(int j(0); j < num_loc_dofs_s; ++j)
204 {
205 // add reactive term 'lambda*sigma'
206 local_matrix[i][j].add_scalar_main_diag(lambda * weight * space_data_s.phi[i].value * space_data_s.phi[j].value);
207
208 // add scalar term 'gamma*v*grad(sigma)' for all components of sigma
209 local_matrix[i][j].add_scalar_main_diag(gamma * weight * space_data_s.phi[i].value * Tiny::dot(loc_v, space_data_s.phi[j].grad));
210
211 // evaluate actual Oldroyd operator
212 core_eval(local_matrix[i][j], loc_grad_v, -zeta * weight * space_data_s.phi[i].value * space_data_s.phi[j].value);
213 }
214 }
215
216 // continue with next cubature point
217 }
218
219 // scatter into matrix
220 scatter_matrix(local_matrix, dof_mapping_s, dof_mapping_s, DataType(1));
221
222 // finish dof mapping
223 dof_mapping_v.finish();
224 dof_mapping_s.finish();
225
226 // finish evaluators
227 space_eval_v.finish();
228 space_eval_s.finish();
229 trafo_eval.finish();
230 }
231 }

242 template<typename DT_>
243 static void core_eval(Tiny::Matrix<DT_, 4, 4, 4, 4>& M, const Tiny::Matrix<DT_, 2, 2, 2, 2>& G, const DT_ omega)
244 {
245 // we have
246 //
247 // X := grad(v) * sigma + sigma * grad(v)^T
248 //
249 // = [ dx(v1) dy(v1) ] * [ sigma_11 sigma_12 ] + [ sigma_11 sigma_12 ] * [ dx(v1) dx(v2) ]
250 // [ dx(v2) dy(v2) ] [ sigma_21 sigma_22 ] [ sigma_21 sigma_22 ] [ dy(v1) dy(v2) ]
251 // \_________________/ \_____________________/ \_____________________/ \_________________/
252 // =: G =: S = S = G^T
253 //
254 // = [ G_11 G_12 ] * [ S_1 S_2 ] + [ S_1 S_2 ] * [ G_11 G_21 ]
255 // [ G_21 G_22 ] [ S_3 S_4 ] [ S_3 S_4 ] [ G_12 G_22 ]
256 //
257 // = [ G_11*S_1 + G_12*S_3 G_11*S_2 + G_12*S_4 ] + [ S_1*G_11 + S_2*G_12 S_1*G_21 + S_2*G_22 ]
258 // [ G_21*S_1 + G_22*S_3 G_21*S_2 + G_22*S_4 ] + [ S_3*G_11 + S_4*G_12 S_3*G_21 + S_4*G_22 ]
259 //
260 // = [ 2*G_11*S_1 + G_12*S_2 + G_12*S_3 G_21*S_1 + (G_11+G_22)*S_2 + G_12*S_4 ]
261 // [ G_21*S_1 + (G_11+G_22)*S_3 + G_12*S_4 G_21*S_2 + G_21*S_3 + 2*G_22*S_4 ]
262 //
263 // = [ X_1 X_2 ]
264 // [ X_3 X_4 ]
265 //
266 // Now re-write definition of X as 4x4 matrix-vector-product of M and S
267 //
268 // [ X_1 ] = [ M_11 M_12 M_13 M_14 ] * [ S_1 ]
269 // [ X_2 ] [ M_21 M_22 M_23 M_24 ] [ S_2 ]
270 // [ X_3 ] [ M_31 M_32 M_33 M_34 ] [ S_3 ]
271 // [ X_4 ] [ M_41 M_42 M_43 M_44 ] [ S_4 ]
272
273 // X_1 = M_1. * S
274 M(0,0) += omega * DT_(2) * G(0,0); // 2 * G_11 * S_1
275 M(0,1) += omega * G(0,1); // G_12 * S_2
276 M(0,2) += omega * G(0,1); // G_12 * S_3
277 //M(0,3) += DT_(0); // 0 * S_4
278
279 // X_2 = M_2. * S
280 M(1,0) += omega * G(1,0); // G_21 * S_1
281 M(1,1) += omega * (G(0,0) + G(1,1)); // (G_11+G_22) * S_2
282 //M(1,2) += DT_(0); // 0 * S_3
283 M(1,3) += omega * G(0,1); // G_12 * S_4
284
285 // X_3 = M_3. * S
286 M(2,0) += omega * G(1,0); // G_21 * S_1
287 //M(2,1) += DT_(0); // 0 * S_2
288 M(2,2) += omega * (G(0,0) + G(1,1)); // (G_11+G_22) * S_3
289 M(2,3) += omega * (G(0,1)); // G_12 * S_4
290
291 // X_4 = M_4. * S
292 //M(3,0) += DT_(0); // 0 * S_1
293 M(3,1) += omega * G(1,0); // G_21 * S_2
294 M(3,2) += omega * G(1,0); // G_21 * S_3
295 M(3,3) += omega * DT_(2) * G(1,1); // 2 * G_22 * S_4
296 }

306 template<typename DT_>
307 static void core_eval(Tiny::Matrix<DT_, 3, 3, 3, 3>& M, const Tiny::Matrix<DT_, 2, 2, 2, 2>& G, const DT_ omega)
308 {
309 // we have
310 //
311 // X := grad(v) * sigma + sigma * grad(v)^T
312 //
313 // = [ dx(v1) dy(v1) ] * [ sigma_11 sigma_12 ] + [ sigma_11 sigma_12 ] * [ dx(v1) dx(v2) ]
314 // [ dx(v2) dy(v2) ] [ sigma_21 sigma_22 ] [ sigma_21 sigma_22 ] [ dy(v1) dy(v2) ]
315 // \_________________/ \_____________________/ \_____________________/ \_________________/
316 // =: G =: S = S = G^T
317 //
318 // = [ G_11 G_12 ] * [ S_1 S_3 ] + [ S_1 S_3 ] * [ G_11 G_21 ]
319 // [ G_21 G_22 ] [ S_3 S_2 ] [ S_3 S_2 ] [ G_12 G_22 ]
320 //
321 // At this point we exploit the symmetry of sigma, i.e. we have S_3 := sigma_12 = sigma_21, then:
322 //
323 // = [ G_11*S_1 + G_12*S_3 G_11*S_3 + G_12*S_2 ] + [ S_1*G_11 + S_3*G_12 S_1*G_21 + S_3*G_22 ]
324 // [ G_21*S_1 + G_22*S_3 G_21*S_3 + G_22*S_2 ] + [ S_3*G_11 + S_2*G_12 S_3*G_21 + S_2*G_22 ]
325 //
326 // = [ 2*G_11*S_1 + 2*G_12*S_3 G_21*S_1 + G_12*S_2 + (G_11+G_22)*S_3 ]
327 // [ G_21*S_1 + G_12*S_2 + (G_11+G_22)*S_3 2*G_22*S_2 + 2*G_21*S_3 ]
328 //
329 // = [ X_1 X_3 ]
330 // [ X_3 X_2 ]
331 //
332 // At this point, we exploit the symmetry of X and it as 3x3 matrix-vector-product of M and S
333 //
334 // [ X_1 ] = [ M_11 M_12 M_13 ] * [ S_1 ]
335 // [ X_2 ] [ M_21 M_22 M_23 ] [ S_3 ]
336 // [ X_3 ] [ M_31 M_32 M_33 ] [ S_3 ]
337
338 // X_1 = M_1. * S
339 M(0,0) += omega * DT_(2) * G(0,0); // 2 * G_11 * S_1
340 //M(0,1) += DT_(0); // 0 * S_2
341 M(0,2) += omega * DT_(2) * G(0,1); // 2 * G_12 * S_3
342
343 // X_2 = M_2. * S
344 //M(1,0) += DT_(0); // 0 * S_1
345 M(1,1) += omega * DT_(2) * G(1,1); // 2 * G_22 * S_2
346 M(1,2) += omega * DT_(2) * G(1,0); // 2 * G_21 * S_3
347
348 // X_2 = M_2. * S
349 M(2,0) += omega * G(1,0); // G_21 * S_1
350 M(2,1) += omega * G(0,1); // G_12 * S_2
351 M(2,2) += omega * (G(0,0) + G(1,1)); // (G_11+G_22) * S_3
352 }
