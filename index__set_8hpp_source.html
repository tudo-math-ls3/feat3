294 void permute(const Adjacency::Permutation& perm, const Adjacency::Permutation& inv_perm_face)
295 {
296 if(_indices.empty())
297 return;
298
299 // first, permute the actual index tuples by the forward permutation
300 if(!perm.empty())
301 {
302 XASSERT(perm.size() == get_num_entities());
303 perm.apply(_indices.data());
304 }
305
306 // now, permute all indices by the inverse face permutation
307 if(!inv_perm_face.empty())
308 {
309 XASSERT(inv_perm_face.size() == get_index_bound());
310 const Index n = Index(_indices.size());
311 for(Index i(0); i < n; ++i)
312 _indices.at(i).permute_map(inv_perm_face);
313 }
314 }

902 // Replace the corresponding IndexSet in the IndexSetHolder by the just generated IndexSet of the right
903 // size
904 ish.template get_index_set<shape_dim_, shape_dim_-2>() = std::move(subshape_at_shape_index_set);
