346 virtual void _create(std::unique_ptr<MeshNodeType> base_mesh_node)
347 {
348 // ensure that the domain control is still empty
349 XASSERTM(!this->_was_created, "domain has already been created");
350 XASSERT(this->size_physical() == std::size_t(0));
351 XASSERT(this->size_virtual() == std::size_t(0));
352
353 TimeStamp stamp_create;
354
355 // create the domain control
356#ifdef FEAT_HAVE_MPI
357 if(this->_comm.size() == 1)
358 {
359 // We've got just one process, so it's a simple choice:
360 this->_create_single_process(std::move(base_mesh_node));
361 }
362 else if(this->_support_multi_layered && (this->_desired_levels.size() > std::size_t(2)))
363 {
364 // The application supports multi-layered domain controls and
365 // the user wants that, so create a multi-layered one:
366 this->_create_multi_layered(std::move(base_mesh_node));
367 }
368 else
369 {
370 // Create a single-layered domain control:
371 this->_create_single_layered(std::move(base_mesh_node));
372 }
373#else // not FEAT_HAVE_MPI
374 {
375 // In the non-MPI case, we always have only one process:
376 this->_create_single_process(std::move(base_mesh_node));
377 }
378#endif // FEAT_HAVE_MPI
379
380 // compile virtual levels
381 this->compile_virtual_levels();
382
383 // collect partition statistics
384 FEAT::Statistics::toe_partition = stamp_create.elapsed_now();
385
386 // domain created
387 this->_was_created = true;
388 }

401 virtual void _create_single_process(std::unique_ptr<MeshNodeType> base_mesh_node)
402 {
403 // create and push single layer
404 this->push_layer(std::make_shared<LayerType>(this->_comm.comm_dup(), 0));
405
406 // create single ancestry
407 this->_create_ancestry_single();
408 Ancestor& ancestor = this->_ancestry.front();
409
410 // use the whole base mesh here
411 ancestor.parti_info = "Using base-mesh";
412
413 // refine up to desired minimum level
414 int lvl = 0;
415 for(; lvl < ancestor.desired_level_min; ++lvl)
416 {
417 // refine the patch mesh
418 base_mesh_node = base_mesh_node->refine_unique(this->_adapt_mode);
419 }
420
421 // save chosen minimum level if it is not equal to the desired maximum level
422 if(lvl < ancestor.desired_level_max)
423 this->_chosen_levels.push_front(std::make_pair(lvl, 0));
424
425 // refine up to maximum level and push to control
426 for(; lvl < ancestor.desired_level_max; ++lvl)
427 {
428 // refine the base mesh
429 auto refined_node = base_mesh_node->refine_unique(this->_adapt_mode);
430
431 // push this level
432 this->push_level_front(0, std::make_shared<LevelType>(lvl, std::move(base_mesh_node)));
433
434 // continue with refined node
435 base_mesh_node = std::move(refined_node);
436 }
437
438 // save chosen maximum level
439 this->_chosen_levels.push_front(std::make_pair(lvl, 1));
440
441 // push finest level
442 this->push_level_front(0, std::make_shared<LevelType>(lvl, std::move(base_mesh_node)));
443 }

456 virtual void _create_single_layered(std::unique_ptr<MeshNodeType> base_mesh_node)
457 {
458 // create and push single layer
459 std::shared_ptr<LayerType> layer = std::make_shared<LayerType>(this->_comm.comm_dup(), 0);
460 this->push_layer(layer);
461
462 // create single-layered ancestry
463 this->_create_ancestry_single();
464 Ancestor& ancestor = this->_ancestry.front();
465
466 // do we have to keep the base-mesh levels on this process?
467 const bool keep_base = this->_keep_base_levels && (this->_comm.rank() == 0);
468
469 // choose a partitioning strategy
470 this->_check_parti(ancestor, *base_mesh_node, true);
471
472 // refine up to chosen partition level
473 int lvl = 0;
474 for(; lvl < ancestor.parti_level; ++lvl)
475 {
476 // refine the base mesh
477 base_mesh_node = base_mesh_node->refine_unique(this->_adapt_mode);
478 }
479
480 // apply partitioner
481 if(!this->_apply_parti(ancestor, *base_mesh_node))
482 {
483 XABORTM("Failed to find a suitable partitioning");
484 }
485
486 // extract our patch
487 std::vector<int> neighbor_ranks;
488 std::unique_ptr<MeshNodeType> patch_mesh_node(
489 base_mesh_node->extract_patch(neighbor_ranks, ancestor.parti_graph, this->_comm.rank()));
490
491 // set the neighbor ranks of our child layer
492 layer->set_neighbor_ranks(neighbor_ranks);
493
494 // if we have to keep the base-mesh levels, then we also need the patches
495 if(keep_base)
496 {
497 // Note: patch mesh-part for rank = 0 was already created by 'extract_patch' call
498 for(int i(1); i < ancestor.num_parts; ++i)
499 {
500 base_mesh_node->create_patch_meshpart(ancestor.parti_graph, i);
501 }
502 }
503
504 // refine up to minimum level
505 for(; lvl < ancestor.desired_level_min; ++lvl)
506 {
507 // refine the patch mesh
508 patch_mesh_node = patch_mesh_node->refine_unique(this->_adapt_mode);
509
510 // keep base mesh node?
511 if(keep_base)
512 base_mesh_node = base_mesh_node->refine_unique(this->_adapt_mode);
513 }
514
515 // save chosen minimum level
516 if(lvl < ancestor.desired_level_max)
517 this->_chosen_levels.push_front(std::make_pair(lvl, 0));
518
519 // refine up to maximum level
520 for(; lvl < ancestor.desired_level_max; ++lvl)
521 {
522 // refine the patch mesh
523 auto refined_node = patch_mesh_node->refine_unique(this->_adapt_mode);
524
525 // push this (unrefined) level
526 this->push_level_front(0, std::make_shared<LevelType>(lvl, std::move(patch_mesh_node)));
527
528 // continue with refined node
529 patch_mesh_node = std::move(refined_node);
530
531 // keep base mesh node?
532 if(keep_base)
533 {
534 // refine base-mesh node
535 auto refined_base_mesh_node = base_mesh_node->refine_unique(this->_adapt_mode);
536
537 // push this (unrefined) level
538 this->_base_levels.push_front(std::make_shared<LevelType>(lvl, std::move(base_mesh_node)));
539
540 // continue with refined node
541 base_mesh_node = std::move(refined_base_mesh_node);
542 }
543 }
544
545 // save chosen maximum level
546 this->_chosen_levels.push_front(std::make_pair(lvl, this->_comm.size()));
547
548 // push finest level
549 this->push_level_front(0, std::make_shared<LevelType>(lvl, std::move(patch_mesh_node)));
550
551 // keep base mesh node?
552 if(keep_base)
553 this->_base_levels.push_front(std::make_shared<LevelType>(lvl, std::move(base_mesh_node)));
554 }

562 virtual void _create_multi_layered(std::unique_ptr<MeshNodeType> base_mesh_node)
563 {
564 // create layers
565 this->_create_multi_layers_scattered();
566
567 // create ancestry
568 this->_create_ancestry_scattered();
569
570 // do we have to keep the base-mesh levels on this process?
571 const bool keep_base = this->_keep_base_levels && (this->_comm.rank() == 0);
572
573 // make sure we have at most 2 layers if we want to keep the base-mesh levels
574 XASSERTM(!keep_base || (this->_ancestry.size() <= std::size_t(1)), "cannot keep base levels for more than 2 domain layers (yet)");
575
576 // we start counting at level 0
577 int lvl = 0;
578
579 // move the base-mesh pointer to a new parent-mesh pointer;
580 // the base-mesh is always the one on the layer with only 1 process
581 std::unique_ptr<MeshNodeType> parent_mesh_node = std::move(base_mesh_node);
582 base_mesh_node.reset();
583
584 // loop over all global layers in reverse order (coarse to fine)
585 for(std::size_t slayer = this->_ancestry.size(); slayer > std::size_t(0); )
586 {
587 // is this the base-mesh layer aka the 1-process layer?
588 const bool is_base_layer = (slayer == this->_ancestry.size());
589
590 --slayer;
591
592 // get the ancestor object
593 Ancestor& ancestor = this->_ancestry.at(slayer);
594
595 // determine the minimum desired level of our parent layer
596 int parent_min_lvl = -1;
597 if(!is_base_layer)
598 parent_min_lvl = this->_chosen_levels.front().first;
599 else if(this->_ancestry.size() + 1u < this->_desired_levels.size())
600 parent_min_lvl = this->_desired_levels.back().first;
601
602 // check available partitioning strategies
603 this->_check_parti(ancestor, *parent_mesh_node, is_base_layer);
604
605 // the check_parti function returns the partitioning level w.r.t. the current
606 // level (which may be > 0), so we have to compensate that by adding our current level:
607 ancestor.parti_level += lvl;
608
609 // Note: each progeny group within the main communicator may have chosen a different
610 // partitioning level at this point. We will compensate this by adjusting the minimum
611 // refinement level of the child layer after the partitioning step below.
612
613 // refine up to the chosen partitioning level
614 for(; lvl < ancestor.parti_level; ++lvl)
615 {
616 // refine the base-mesh node
617 auto refined_node = parent_mesh_node->refine_unique(this->_adapt_mode);
618
619 // push the base mesh into our parent layer if desired
620 if((ancestor.layer_p >= 0) && (parent_min_lvl >= 0) && (lvl >= parent_min_lvl))
621 {
622 this->push_level_front(ancestor.layer_p, std::make_shared<LevelType>(lvl, std::move(parent_mesh_node)));
623 }
624
625 // continue with refined node
626 parent_mesh_node = std::move(refined_node);
627
628 // refine base-mesh node if we have one
629 if(base_mesh_node)
630 {
631 // refine base-mesh node
632 auto refined_base_mesh_node = base_mesh_node->refine_unique(this->_adapt_mode);
633
634 // push this (unrefined) level
635 this->_base_levels.push_front(std::make_shared<LevelType>(lvl, std::move(base_mesh_node)));
636
637 // continue with refined node
638 base_mesh_node = std::move(refined_base_mesh_node);
639 }
640 }
641
642 // we're now at the partitioning level, so apply the partitioner
643 if(!this->_apply_parti(ancestor, *parent_mesh_node))
644 {
645 XABORTM("Failed to find a suitable partitioning");
646 }
647
648 // extract our patch
649 std::vector<int> neighbor_ranks;
650 std::unique_ptr<MeshNodeType> patch_mesh_node(
651 parent_mesh_node->extract_patch(neighbor_ranks, ancestor.parti_graph, ancestor.progeny_child));
652
653 // translate neighbor ranks by progeny group to obtain the neighbor ranks
654 // w.r.t. this layer's communicator
655 {
656 std::map<int,int> halo_map;
657 for(auto& i : neighbor_ranks)
658 {
659 int old_i(i);
660 halo_map.emplace(old_i, i += ancestor.progeny_group);
661 }
662 patch_mesh_node->rename_halos(halo_map);
663 }
664
665 // does this process participate in the parent layer or do we need to keep the base-meshes anyways?
666 if((ancestor.layer_p >= 0) || (keep_base && is_base_layer))
667 {
668 // Note: patch mesh-part for rank = 0 was already created by 'extract_patch' call
669 for(int i(1); i < ancestor.num_parts; ++i)
670 {
671 parent_mesh_node->create_patch_meshpart(ancestor.parti_graph, i);
672 }
673 }
674
675 // make sure we choose the same minimum level for all processes, because we may
676 // have chosen different partitioning levels for each patch
677 int global_level_min = Math::max(ancestor.desired_level_min, ancestor.parti_level);
678 this->_comm.allreduce(&global_level_min, &global_level_min, std::size_t(1), Dist::op_max);
679
680 // make sure our minimum level is greater than the minimum level of the previous layer,
681 // because each layer must contain at least one non-ghost level
682 if(!is_base_layer)
683 global_level_min = Math::max(global_level_min, this->_chosen_levels.front().first+1);
684
685 // refine up to desired minimum level of this layer
686 for(; lvl < global_level_min; ++lvl)
687 {
688 // refine the base mesh node
689 std::unique_ptr<MeshNodeType> coarse_mesh_node(std::move(parent_mesh_node));
690 parent_mesh_node = coarse_mesh_node->refine_unique(this->_adapt_mode);
691
692 // push base mesh to parent layer if desired
693 if((ancestor.layer_p >= 0) && (parent_min_lvl >= 0) && (lvl >= parent_min_lvl))
694 {
695 // clear patches before pushing this node as they are redundant here
696 coarse_mesh_node->clear_patches();
697 this->push_level_front(ancestor.layer_p, std::make_shared<LevelType>(lvl, std::move(coarse_mesh_node)));
698 }
699
700 // refine the patch mesh
701 patch_mesh_node = patch_mesh_node->refine_unique(this->_adapt_mode);
702
703 // refine base-mesh node if we have one
704 if(base_mesh_node)
705 {
706 // refine base-mesh node
707 auto refined_base_mesh_node = base_mesh_node->refine_unique(this->_adapt_mode);
708
709 // push this (unrefined) level
710 this->_base_levels.push_front(std::make_shared<LevelType>(lvl, std::move(base_mesh_node)));
711
712 // continue with refined node
713 base_mesh_node = std::move(refined_base_mesh_node);
714 }
715 }
716
717 // split the halos of our base-mesh and compute the halos of our patches from that
718 this->_split_basemesh_halos(ancestor, *parent_mesh_node, *patch_mesh_node, neighbor_ranks);
719
720 // does this process participate in the child layer?
721 if(ancestor.layer >= 0)
722 {
723 // set the neighbor ranks in our child layer
724 this->_layers.at(std::size_t(ancestor.layer))->set_neighbor_ranks(neighbor_ranks);
725 }
726
727 // set chosen minimum level for this layer
728 if(!is_base_layer)
729 this->_chosen_levels.push_front(std::make_pair(lvl, this->_ancestry.at(slayer+1u).num_procs));
730 else if(parent_min_lvl < 0)
731 this->_chosen_levels.push_front(std::make_pair(lvl, 0));
732 else
733 {
734 this->_chosen_levels.push_front(std::make_pair(parent_min_lvl, 0));
735 this->_chosen_levels.push_front(std::make_pair(lvl, 1));
736 }
737
738 // if we have to keep the base-mesh, then we have to create a clone here if we're still on the base layer;
739 // we don't need to worry that this is a clone, because it will get refined in the next layer anyways
740 if(keep_base && is_base_layer)
741 {
742 XASSERT(base_mesh_node.get() == nullptr);
743 base_mesh_node = parent_mesh_node->clone_unique();
744 }
745
746 // push the finest base-mesh
747 if(ancestor.layer_p >= 0)
748 {
749 this->push_level_front(ancestor.layer_p, std::make_shared<LevelType>(lvl, std::move(parent_mesh_node)));
750 }
751
752 // continue with the next layer
753 parent_mesh_node = std::move(patch_mesh_node);
754 }
755
756 // get the desired maximum level
757 // if we have more than one layer, make sure that the finest one contains at
758 // least one level, as otherwise the finest global level would be a ghost level
759 int desired_level_max = Math::max(this->_ancestry.front().desired_level_max, lvl+1);
760
761 for(; lvl < desired_level_max; ++lvl)
762 {
763 // refine the patch mesh
764 auto refined_node = parent_mesh_node->refine_unique(this->_adapt_mode);
765
766 // push patch mesh to this level
767 this->push_level_front(0, std::make_shared<LevelType>(lvl, std::move(parent_mesh_node)));
768
769 // continue with refined node
770 parent_mesh_node = std::move(refined_node);
771
772 // refine base-mesh node if we have one
773 if(base_mesh_node)
774 {
775 // refine base-mesh node
776 auto refined_base_mesh_node = base_mesh_node->refine_unique(this->_adapt_mode);
777
778 // push this (unrefined) level
779 this->_base_levels.push_front(std::make_shared<LevelType>(lvl, std::move(base_mesh_node)));
780
781 // continue with refined node
782 base_mesh_node = std::move(refined_base_mesh_node);
783 }
784 }
785
786 // push finest level
787 this->push_level_front(0, std::make_shared<LevelType>(lvl, std::move(parent_mesh_node)));
788
789 // set chosen maximum level for finest layer
790 this->_chosen_levels.push_front(std::make_pair(lvl, this->_comm.size()));
791
792 // refine base-mesh node if we have one
793 if(base_mesh_node)
794 this->_base_levels.push_front(std::make_shared<LevelType>(lvl, std::move(base_mesh_node)));
795 }

822 virtual bool _check_parti(Ancestor& ancestor, const MeshNodeType& mesh_node, bool is_base_layer) override
823 {
824 // Try to find an appropriate a-priori partitioning first:
825 if(is_base_layer && this->_check_parti_extern(ancestor))
826 return true;
827
828 if(this->_check_parti_2level(ancestor, mesh_node))
829 return true;
830
831 // call base class
832 return BaseClass::_check_parti(ancestor, mesh_node, is_base_layer);
833 }
834
844 bool _check_parti_extern(Ancestor& ancestor)
845 {
846 // is this even allowed?
847 if(!this->_allow_parti_extern)
848 return false;
849
850 // check whether we have a suitable partition
