228 bool parse_args(SimpleArgParser& args)
229 {
230 XASSERTM(!_was_created, "This function has to be called before domain control creation!");
231 // Try to parse --parti-type <types...>
232 {
233 auto it = args.query("parti-type");
234 if(it != nullptr)
235 {
236 // disallow all strategies by default
237 _allow_parti_extern = _allow_parti_2level = false;
238 _allow_parti_metis = _allow_parti_naive = false;
239 _allow_parti_genetic = _allow_parti_zoltan = false;
240
241 // loop over all allowed strategies
242 for(const auto& t : it->second)
243 {
244 if (t.compare_no_case("extern") == 0)
245 _allow_parti_extern = true;
246 else if(t.compare_no_case("2level") == 0)
247 _allow_parti_2level = true;
248 else if(t.compare_no_case("metis") == 0)
249 _allow_parti_metis = true;
250 else if(t.compare_no_case("genetic") == 0)
251 _allow_parti_genetic = true;
252 else if(t.compare_no_case("naive") == 0)
253 _allow_parti_naive = true;
254 else if(t.compare_no_case("zoltan") == 0)
255 _allow_parti_zoltan = true;
256 else
257 {
258 this->_comm.print("ERROR: unknown partitioner type '" + t + "'");
259 return false;
260 }
261 }
262 }
263 }
264
265 // parse --parti-extern-name <names...>
266 {
267 auto it = args.query("parti-extern-name");
268 if(it != nullptr)
269 _extern_parti_names = it->second;
270 }
271
272 // parse --parti-rank-elems <count>
273 args.parse("parti-rank-elems", _required_elems_per_rank);
274
275 // parse --parti-genetic-time <time-init> <time-mutate>
276 args.parse("parti-genetic-time", _genetic_time_init, _genetic_time_mutate);
277
278 // okay
279 return true;
280 }

427 void set_desired_levels(const std::deque<String>& slvls)
428 {
429 XASSERTM(!_was_created, "This function has to be called before domain control creation!");
430
431 const int nranks = this->_comm.size();
432 std::deque<String> sv;
433
434 for(std::size_t i(0); i < slvls.size(); ++i)
435 {
436 int ilvl = -1, nprocs = -1;
437 sv = slvls.at(i).split_by_string(":");
438
439 if((sv.size() < std::size_t(1)) || (sv.size() > std::size_t(2)))
440 throw ParseError("Invalid input format", slvls.at(i), "an int-pair 'level:patches'");
441
442 if(!sv.front().parse(ilvl))
443 throw ParseError("Failed to parse level index", slvls.at(i), "an integer");
444 if((sv.size() > std::size_t(1)) && !sv.back().parse(nprocs))
445 throw ParseError("Failed to parse process count" , slvls.at(i), "an integer");
446
447 // level must be non-negative
448 if(ilvl < 0)
449 throw ParseError("Invalid negative level index", slvls.at(i), "a non-negative level index");
450
451 // first level index?
452 if(i == std::size_t(0))
453 {
454 if((nprocs >= 0) && (nprocs != this->_comm.size()))
455 throw ParseError("Invalid number of processes for global level: '" + slvls.at(i) +
456 "', expected " + stringify(this->_comm.size()) + " but got " + stringify(nprocs));
457 _desired_levels.push_back(std::make_pair(ilvl, nranks));
458 continue;
459 }
460
461 // make sure the level is non-ascending
462 if(_desired_levels.back().first < ilvl)
463 throw ParseError("Invalid non-descending level index: '" + slvls.at(i) +
464 "', expected <= " + stringify(_desired_levels.back().first) + " but got " + stringify(ilvl));
465
466 // make sure process count is valid
467 if((i + 1) == slvls.size())
468 nprocs = 0;
469 else
470 {
471 // the process count must be descending
472 if(_desired_levels.back().second <= nprocs)
473 throw ParseError("Invalid non-descending process count: '" + slvls.at(i) +
474 "', expected < " + stringify(_desired_levels.back().second) + " but got " + stringify(nprocs));
475
476 // the previous process count must be a multiple
477 if(_desired_levels.back().second % nprocs != 0)
478 throw ParseError("Invalid indivisible process count: '" + slvls.at(i) +
479 "', expected a divisor of " + stringify(_desired_levels.back().second) + " but got " + stringify(nprocs));
480 }
481
482 // push the level-proc pair
483 _desired_levels.push_back(std::make_pair(ilvl, nprocs));
484 }

715 void create(std::unique_ptr<MeshNodeType> base_mesh_node)
716 {
717 // ensure that the domain control is still empty
718 XASSERTM(!_was_created, "domain has already been created");
719 XASSERT(this->size_physical() == std::size_t(0));
720 XASSERT(this->size_virtual() == std::size_t(0));
721
722 TimeStamp stamp_create;
723
724 // create the domain control
725#ifdef FEAT_HAVE_MPI
726 if(this->_comm.size() == 1)
727 {
728 // We've got just one process, so it's a simple choice:
729 this->_create_single_process(std::move(base_mesh_node));
730 }
731 else if(_support_multi_layered && (_desired_levels.size() > std::size_t(2)))
732 {
733 // The application supports multi-layered domain controls and
734 // the user wants that, so create a multi-layered one:
735 this->_create_multi_layered(std::move(base_mesh_node));
736 }
737 else
738 {
739 // Create a single-layered domain control:
740 this->_create_single_layered(std::move(base_mesh_node));
741 }
742#else // not FEAT_HAVE_MPI
743 {
744 // In the non-MPI case, we always have only one process:
745 this->_create_single_process(std::move(base_mesh_node));
746 }
747#endif // FEAT_HAVE_MPI
748
749 // cleanup
750 _parti_set.clear();
751
752 // compile virtual levels
753 this->compile_virtual_levels();
754
755 // finally, apply mesh permutation
756 this->create_mesh_permutations();
757
758 // collect partition statistics
759 FEAT::Statistics::toe_partition = stamp_create.elapsed_now();
760
761 // domain created
762 _was_created = true;
763 }

859 std::vector<char> serialize_partitioning() const
860 {
861 BinaryStream bs;
862 typedef std::uint64_t u64;
863
864 // serialization magic number: "F3PaDoCo"
865 const std::uint64_t magic = 0x6F436F4461503346;
866
867 XASSERT(this->_num_global_layers >= this->_ancestry.size());
868
869 if(this->_comm.rank() == 0)
870 {
871 // dump magic
872 bs.write((const char*)&magic, 8u);
873
874 // write finest level
875 const u64 lev = u64(this->_virt_levels.front()->get_level_index());
876 bs.write((const char*)&lev, 8u);
877
878 // serialize ancestry/layers
879 const u64 na = u64(this->_ancestry.size());
880 bs.write((const char*)&na, 8u);
881
882 // write number of ranks per layer in reverse order
883 for(auto it = this->_ancestry.rbegin(); it != this->_ancestry.rend(); ++it)
884 {
885 // write ranks
886 const u64 np = u64(it->num_procs);
887 const u64 pl = u64(it->parti_level);
888 bs.write((const char*)&np, 8u);
889 bs.write((const char*)&pl, 8u);
890 }
891 }
892
893 // loop over all ancestry graphs
894 for(auto it = this->_ancestry.rbegin(); it != this->_ancestry.rend(); ++it)
895 {
896 if(it == this->_ancestry.rbegin())
897 {
898 std::vector<char> buf = it->parti_graph.serialize();
899 bs.write(buf.data(), std::streamsize(buf.size()));
900 continue;
901 }
902
903 // get layer index
904 if(it->layer_p < 0)
905 continue;
906
907 // get parent layer communicator
908 const std::size_t ilp = std::size_t(it->layer_p);
909 XASSERT(ilp < this->_layers.size());
910 const Dist::Comm& comm_p = this->_layers.at(ilp)->comm();
911
912 // serialize graph
913 std::vector<char> buf = it->parti_graph.serialize();
914
915 // choose maximum size
916 u64 buf_size = buf.size();
917
918 // gather individual buffer sizes on rank 0
919 std::vector<u64> all_sizes;
920 if(comm_p.rank() == 0)
921 all_sizes.resize(std::size_t(comm_p.size()));
922 comm_p.gather(&buf_size, 1u, all_sizes.data(), 1u, 0);
923
924 // allreduce maximum size
925 comm_p.allreduce(&buf_size, &buf_size, std::size_t(1), Dist::op_max);
926
927 // adjust buffer size
928 if(buf_size > u64(buf.size()))
929 buf.resize(buf_size);
930
931 // on rank 0, allocate common buffer
932 std::vector<char> com_buf;
933 if(comm_p.rank() == 0)
934 com_buf.resize(std::size_t(buf_size * u64(comm_p.size())));
935
936 // gather all buffers on rank 0
937 comm_p.gather(buf.data(), buf_size, com_buf.data(), buf_size, 0);
938
939 // write each individual buffer
940 if(comm_p.rank() == 0)
941 {
942 char* x = com_buf.data();
943 for(u64 k(0); k < u64(comm_p.size()); ++k)
944 bs.write(&x[k*buf_size], std::streamsize(all_sizes[k]));
945 }
946 }
947
948 return bs.container();
949 }

1162 void _create_multi_layers_scattered()
1163 {
1164 // create and push global layer
1165 this->push_layer(std::make_shared<LayerType>(this->_comm.comm_dup(), 0));
1166
1167 // loop over all desired layers
1168 for(std::size_t ilay(1u); (ilay+1u) < this->_desired_levels.size(); ++ilay)
1169 {
1170 // get child layer
1171 std::shared_ptr<LayerType> layer_c = this->_layers.back();
1172
1173 // get child layer communicator
1174 const Dist::Comm& comm_c = layer_c->comm();
1175
1176 // get number of processes in child comm
1177 const int nprocs_c = comm_c.size();
1178
1179 // get desired number of processes for parent comm
1180 const int nprocs_p = this->_desired_levels.at(ilay).second;
1181 XASSERT(nprocs_p < nprocs_c);
1182 XASSERT(nprocs_p > 0);
1183 XASSERT(nprocs_c % nprocs_p == 0); // same number of children for each parent
1184
1185 // compute number of siblings = children per parent
1186 const int num_sibs = nprocs_c / nprocs_p;
1187
1188 // get my rank in child comm
1189 const int my_rank_c = comm_c.rank();
1190
1191 // compute my parent's rank in child comm
1192 const int parent_rank_c = (my_rank_c / num_sibs) * num_sibs;
1193
1194 // create our sibling communicator and set the parent rank
1195 layer_c->set_parent(comm_c.comm_create_range_incl(num_sibs, parent_rank_c), 0);
1196
1197 // next, create the actual parent communicator
1198 Dist::Comm comm_p = comm_c.comm_create_range_incl(nprocs_p, 0, num_sibs);
1199
1200 // Are we the parent?
1201 if(parent_rank_c == my_rank_c)
1202 {
1203 // make sure we have a valid communicator
1204 XASSERT(!comm_p.is_null());
1205
1206 // push the parent layer
1207 this->push_layer(std::make_shared<LayerType>(std::move(comm_p), int(ilay)));
1208 }
1209 else
1210 {
1211 // We are not a parent, so we must have received a null communicator
1212 XASSERT(comm_p.is_null());
1213
1214 // Exit the loop, as we are not part of the party anymore...
1215 break;
1216 }
1217 }

1223 void _create_ancestry_scattered()
1224 {
1225 // the layers must have been created already
1226 XASSERT(!this->_layers.empty());
1227
1228 // create a deque with the number of processes for each layer
1229 std::deque<int> num_procs;
1230 for(auto it = this->_desired_levels.begin(); it != this->_desired_levels.end(); ++it)
1231 {
1232 if(it->second > 1)
1233 num_procs.push_back(it->second);
1234 }
1235 // manually add the base-layer
1236 num_procs.push_back(1);
1237
1238 // allocate and create ancestry
1239 this->_ancestry.resize(num_procs.size()-std::size_t(1));
1240
1241 // set up the ancestry
1242 const int main_rank = this->_comm.rank();
1243 const int main_size = this->_comm.size();
1244 for(std::size_t i(0); i < this->_ancestry.size(); ++i)
1245 {
1246 // get our ancestor info
1247 Ancestor& ancestor = this->_ancestry.at(i);
1248
1249 // set the layer index (or -1, if this process is not part of that layer)
1250 ancestor.layer = (i < this->_layers.size() ? int(i) : -1);
1251
1252 // set the parent layer index (or -1, if this process is not in the parent layer)
1253 ancestor.layer_p = ((i+1u) < this->_layers.size() ? int(i)+1 : -1);
1254
1255 // set the total number of processes for this layer
1256 ancestor.num_procs = num_procs.at(i);
1257
1258 // set the total number of partitions per progeny group
1259 ancestor.num_parts = num_procs.at(i) / num_procs.at(i+1u);
1260
1261 // set desired maximum level
1262 XASSERT(i < this->_desired_levels.size());
1263 ancestor.desired_level_max = this->_desired_levels.at(i).first;
1264
1265 // set desired minimum level
1266 if((i+1u) < this->_desired_levels.size())
1267 ancestor.desired_level_min = this->_desired_levels.at(i+1).first;
1268 else
1269 ancestor.desired_level_min = ancestor.desired_level_max;
1270
1271 // set the progeny group
1272 ancestor.progeny_group = ((main_rank * num_procs.at(i+1u)) / main_size) * ancestor.num_parts;
1273 ancestor.progeny_child = ((main_rank * num_procs.at(i)) / main_size) % ancestor.num_parts;
1274
1275 // create the progeny communicator
1276 ancestor.progeny_count = main_size / num_procs.at(i+1u);
1277 ancestor.progeny_first = (main_rank / ancestor.progeny_count) * ancestor.progeny_count;
1278 ancestor.progeny_comm = this->_comm.comm_create_range_incl(ancestor.progeny_count, ancestor.progeny_first);
1279 }

1288 void _create_multi_layered(std::unique_ptr<MeshNodeType> base_mesh_node)
1289 {
1290 // create layers
1291 this->_create_multi_layers_scattered();
1292
1293 // create ancestry
1294 this->_create_ancestry_scattered();
1295
1296 // do we have to keep the base-mesh levels on this process?
1297 const bool keep_base = this->_keep_base_levels && (this->_comm.rank() == 0);
1298
1299 // make sure we have at most 2 layers if we want to keep the base-mesh levels
1300 XASSERTM(!keep_base || (this->_ancestry.size() <= std::size_t(1)), "cannot keep base levels for more than 2 domain layers (yet)");
1301
1302 // we start counting at level 0
1303 int lvl = 0;
1304
1305 // move the base-mesh pointer to a new parent-mesh pointer;
1306 // the base-mesh is always the one on the layer with only 1 process
1307 std::unique_ptr<MeshNodeType> parent_mesh_node = std::move(base_mesh_node);
1308 base_mesh_node.reset();
1309
1310 // loop over all global layers in reverse order (coarse to fine)
1311 for(std::size_t slayer = this->_ancestry.size(); slayer > std::size_t(0); )
1312 {
1313 // is this the base-mesh layer aka the 1-process layer?
1314 const bool is_base_layer = (slayer == this->_ancestry.size());
1315
1316 --slayer;
1317
1318 // get the ancestor object
1319 Ancestor& ancestor = this->_ancestry.at(slayer);
1320
1321 // determine the minimum desired level of our parent layer
1322 int parent_min_lvl = -1;
1323 if(!is_base_layer)
1324 parent_min_lvl = this->_chosen_levels.front().first;
1325 else if(_ancestry.size()+1u < _desired_levels.size())
1326 parent_min_lvl = this->_desired_levels.back().first;
1327
1328 // check available partitioning strategies
1329 this->_check_parti(ancestor, *parent_mesh_node, is_base_layer);
1330
1331 // the check_parti function returns the partitioning level w.r.t. the current
1332 // level (which may be > 0), so we have to compensate that by adding our current level:
1333 ancestor.parti_level += lvl;
1334
1335 // Note: each progeny group within the main communicator may have chosen a different
1336 // partitioning level at this point. We will compensate this by adjusting the minimum
1337 // refinement level of the child layer after the partitioning step below.
1338
1339 // refine up to the chosen partitioning level
1340 for(; lvl < ancestor.parti_level; ++lvl)
1341 {
1342 // refine the base-mesh node
1343 auto refined_node = parent_mesh_node->refine_unique(this->_adapt_mode);
1344
1345 // push the base mesh into our parent layer if desired
1346 if((ancestor.layer_p >= 0) && (parent_min_lvl >= 0) && (lvl >= parent_min_lvl))
1347 {
1348 this->push_level_front(ancestor.layer_p, std::make_shared<LevelType>(lvl, std::move(parent_mesh_node)));
1349 }
1350
1351 // continue with refined node
1352 parent_mesh_node = std::move(refined_node);
1353
1354 // refine base-mesh node if we have one
1355 if(base_mesh_node)
1356 {
1357 // refine base-mesh node
1358 auto refined_base_mesh_node = base_mesh_node->refine_unique(this->_adapt_mode);
1359
1360 // push this (unrefined) level
1361 this->_base_levels.push_front(std::make_shared<LevelType>(lvl, std::move(base_mesh_node)));
1362
1363 // continue with refined node
1364 base_mesh_node = std::move(refined_base_mesh_node);
1365 }
1366 }
1367
1368 // we're now at the partitioning level, so apply the partitioner
1369 if(!this->_apply_parti(ancestor, *parent_mesh_node))
1370 {
1371 XABORTM("Failed to find a suitable partitioning");
1372 }
1373
1374 // extract our patch
1375 std::vector<int> neighbor_ranks;
1376 std::unique_ptr<MeshNodeType> patch_mesh_node(
1377 parent_mesh_node->extract_patch(neighbor_ranks, ancestor.parti_graph, ancestor.progeny_child));
1378
1379 // translate neighbor ranks by progeny group to obtain the neighbor ranks
1380 // w.r.t. this layer's communicator
1381 {
1382 std::map<int,int> halo_map;
1383 for(auto& i : neighbor_ranks)
1384 {
1385 int old_i(i);
1386 halo_map.emplace(old_i, i += ancestor.progeny_group);
1387 }
1388 patch_mesh_node->rename_halos(halo_map);
1389 }
1390
1391 // does this process participate in the parent layer or do we need to keep the base-meshes anyways?
1392 if((ancestor.layer_p >= 0) || (keep_base && is_base_layer))
1393 {
1394 // Note: patch mesh-part for rank = 0 was already created by 'extract_patch' call
1395 for(int i(1); i < ancestor.num_parts; ++i)
1396 {
1397 parent_mesh_node->create_patch_meshpart(ancestor.parti_graph, i);
1398 }
1399 }
1400
1401 // make sure we choose the same minimum level for all processes, because we may
1402 // have chosen different partitioning levels for each patch
1403 int global_level_min = Math::max(ancestor.desired_level_min, ancestor.parti_level);
1404 this->_comm.allreduce(&global_level_min, &global_level_min, std::size_t(1), Dist::op_max);
1405
1406 // make sure our minimum level is greater than the minimum level of the previous layer,
1407 // because each layer must contain at least one non-ghost level
1408 if(!is_base_layer)
1409 global_level_min = Math::max(global_level_min, this->_chosen_levels.front().first+1);
1410
1411 // refine up to desired minimum level of this layer
1412 for(; lvl < global_level_min; ++lvl)
1413 {
1414 // refine the base mesh node
1415 std::unique_ptr<MeshNodeType> coarse_mesh_node(std::move(parent_mesh_node));
1416 parent_mesh_node = coarse_mesh_node->refine_unique(this->_adapt_mode);
1417
1418 // push base mesh to parent layer if desired
1419 if((ancestor.layer_p >= 0) && (parent_min_lvl >= 0) && (lvl >= parent_min_lvl))
1420 {
1421 // clear patches before pushing this node as they are redundant here
1422 coarse_mesh_node->clear_patches();
1423 this->push_level_front(ancestor.layer_p, std::make_shared<LevelType>(lvl, std::move(coarse_mesh_node)));
1424 }
1425
1426 // refine the patch mesh
1427 patch_mesh_node = patch_mesh_node->refine_unique(this->_adapt_mode);
1428
1429 // refine base-mesh node if we have one
1430 if(base_mesh_node)
1431 {
1432 // refine base-mesh node
1433 auto refined_base_mesh_node = base_mesh_node->refine_unique(this->_adapt_mode);
1434
1435 // push this (unrefined) level
1436 this->_base_levels.push_front(std::make_shared<LevelType>(lvl, std::move(base_mesh_node)));
1437
1438 // continue with refined node
1439 base_mesh_node = std::move(refined_base_mesh_node);
1440 }
1441 }
1442
1443 // split the halos of our base-mesh and compute the halos of our patches from that
1444 this->_split_basemesh_halos(ancestor, *parent_mesh_node, *patch_mesh_node, neighbor_ranks);
1445
1446 // does this process participate in the child layer?
1447 if(ancestor.layer >= 0)
1448 {
1449 // set the neighbor ranks in our child layer
1450 this->_layers.at(std::size_t(ancestor.layer))->set_neighbor_ranks(neighbor_ranks);
1451 }
1452
1453 // set chosen minimum level for this layer
1454 if(!is_base_layer)
1455 this->_chosen_levels.push_front(std::make_pair(lvl, this->_ancestry.at(slayer+1u).num_procs));
1456 else if(parent_min_lvl < 0)
1457 this->_chosen_levels.push_front(std::make_pair(lvl, 0));
1458 else
1459 {
1460 this->_chosen_levels.push_front(std::make_pair(parent_min_lvl, 0));
1461 this->_chosen_levels.push_front(std::make_pair(lvl, 1));
1462 }
1463
1464 // if we have to keep the base-mesh, then we have to create a clone here if we're still on the base layer;
1465 // we don't need to worry that this is a clone, because it will get refined in the next layer anyways
1466 if(keep_base && is_base_layer)
1467 {
1468 XASSERT(base_mesh_node.get() == nullptr);
1469 base_mesh_node = parent_mesh_node->clone_unique();
1470 }
1471
1472 // push the finest base-mesh
1473 if(ancestor.layer_p >= 0)
1474 {
1475 this->push_level_front(ancestor.layer_p, std::make_shared<LevelType>(lvl, std::move(parent_mesh_node)));
1476 }
1477
1478 // continue with the next layer
1479 parent_mesh_node = std::move(patch_mesh_node);
1480 }
1481
1482 // get the desired maximum level
1483 // if we have more than one layer, make sure that the finest one contains at
1484 // least one level, as otherwise the finest global level would be a ghost level
1485 int desired_level_max = Math::max(this->_ancestry.front().desired_level_max, lvl+1);
1486
1487 for(; lvl < desired_level_max; ++lvl)
1488 {
1489 // refine the patch mesh
1490 auto refined_node = parent_mesh_node->refine_unique(this->_adapt_mode);
1491
1492 // push patch mesh to this level
1493 this->push_level_front(0, std::make_shared<LevelType>(lvl, std::move(parent_mesh_node)));
1494
1495 // continue with refined node
1496 parent_mesh_node = std::move(refined_node);
1497
1498 // refine base-mesh node if we have one
1499 if(base_mesh_node)
1500 {
1501 // refine base-mesh node
1502 auto refined_base_mesh_node = base_mesh_node->refine_unique(this->_adapt_mode);
1503
1504 // push this (unrefined) level
1505 this->_base_levels.push_front(std::make_shared<LevelType>(lvl, std::move(base_mesh_node)));
1506
1507 // continue with refined node
1508 base_mesh_node = std::move(refined_base_mesh_node);
1509 }
1510 }
1511
1512 // push finest level
1513 this->push_level_front(0, std::make_shared<LevelType>(lvl, std::move(parent_mesh_node)));
1514
1515 // set chosen maximum level for finest layer
1516 this->_chosen_levels.push_front(std::make_pair(lvl, this->_comm.size()));
1517
1518 // refine base-mesh node if we have one
1519 if(base_mesh_node)
1520 this->_base_levels.push_front(std::make_shared<LevelType>(lvl, std::move(base_mesh_node)));
1521 }

1540 void _split_basemesh_halos(
1541 const Ancestor& ancestor,
1542 const MeshNodeType& base_mesh_node,
1543 MeshNodeType& patch_mesh_node,
1544 std::vector<int>& neighbor_ranks)
1545 {
1546 // get the map of the base-mesh halos
1547 const std::map<int, std::unique_ptr<MeshPartType>>& base_halo_map = base_mesh_node.get_halo_map();
1548
1549 // if the base mesh has no halos, then we can jump out of here
1550 if(base_halo_map.empty())
1551 return;
1552
1553 // get number of halos
1554 const std::size_t num_halos = base_halo_map.size();

1853 bool _check_parti(Ancestor& ancestor, const MeshNodeType& mesh_node, bool check_extern)
1854 {
1855 // Try to find an appropriate a-priori partitioning first:
1856 if(check_extern && this->_check_parti_extern(ancestor))
1857 return true;
1858
1859 if(this->_check_parti_2level(ancestor, mesh_node))
1860 return true;
1861
1862 // No a-priori partitioning found, so we need to determine the
1863 // required partitioning level for a-posteriori partitioning.
1864 // For this, first determine the factor by which the number of elements
1865 // increases for each refinement:
1866 const Index factor = Index(Geometry::Intern::StandardRefinementTraits<typename BaseClass::ShapeType, BaseClass::ShapeType::dimension>::count);
1867
1868 // compute minimum number of elements for a-posteriori partitioning
1869 const Index min_elems = Index(ancestor.num_parts) * Index(_required_elems_per_rank);
1870
1871 // Okay, get the number of elements on base-mesh level 0
1872 Index num_elems = mesh_node.get_mesh()->get_num_elements();
1873
1874 // compute refinement level on which we have enough elements
1875 int level = 0;
1876 for(; num_elems < min_elems; ++level)
1877 {
1878 num_elems *= factor;
1879 }
1880
1881 // finally, the user may have ancestor a greater level for partitioning:
1882 ancestor.parti_level = Math::max(ancestor.parti_level, level);
1883
1884 // No a-priori partitioning; try a-posteriori partitioner later
1885 return false;
1886 }
