169 void assemble()
170 {
171 // format local vector
172 local_vector.format();
173
174 // fetch number of local dofs
175 const int num_loc_dofs = space_eval.get_num_local_dofs();
176
177 // loop over all quadrature points and integrate
178 for(int k(0); k < cubature_rule.get_num_points(); ++k)
179 {
180 // compute trafo data
181 trafo_eval(trafo_data, cubature_rule.get_point(k));
182
183 // compute basis function data
184 space_eval(space_data, trafo_data);
185
186 // evaluate operator
187 static_cast<Derived_&>(*this).set_point(trafo_data);
188
189 // test function loop
190 for(int i(0); i < num_loc_dofs; ++i)
191 {
192 // evaluate functional and integrate
193 static_cast<Derived_&>(*this).eval(local_vector(i),
194 trafo_data.jac_det * cubature_rule.get_weight(k),
195 space_data.phi[i]);
196 // continue with next test function
197 }
198 // continue with next cubature point
199 }
200 }
