167 void assemble()
168 {
169 // format local vector
170 local_vector.format();
171
172 // fetch number of local dofs
173 const int num_loc_dofs = space_eval.get_num_local_dofs();
174
175 // loop over all quadrature points and integrate
176 for(int k(0); k < cubature_rule.get_num_points(); ++k)
177 {
178 // compute trafo data
179 trafo_eval(trafo_data, cubature_rule.get_point(k));
180
181 // compute basis function data
182 space_eval(space_data, trafo_data);
183
184 // evaluate operator
185 static_cast<Derived_&>(*this).set_point(trafo_data);
186
187 // test function loop
188 for(int i(0); i < num_loc_dofs; ++i)
189 {
190 // evaluate functional and integrate
191 static_cast<Derived_&>(*this).eval(local_vector(i),
192 trafo_data.jac_det * cubature_rule.get_weight(k),
193 space_data.phi[i]);
194 // continue with next test function
195 }
196 // continue with next cubature point
197 }
198 }

217 // finish dof mapping
