42 template<typename DataType_, typename IndexType_>
43 static SparseMatrixBanded<DataType_, IndexType_> value(const Index fe_degree,
44 const std::vector<IndexType_> & num_of_subintervalls)
45 {
46 const IndexType_ * const pnos(num_of_subintervalls.data());
47
48 // get number of dimensions
49 const Index d(Index(num_of_subintervalls.size()));
50
51 // output of errors if wrong input
52 XASSERTM(d >= IndexType_(1), "You need at least 1 dimension");
53
54 for (IndexType_ i(0); i < d; ++i)
55 {
56 XASSERTM(pnos[i] >= IndexType_(3), "You need at least 3 subintervalls per dimension");
57 }
58
59 // calculate size of matrix and number of offsets
60 IndexType_ size(1);
61 Index noo(1);
62 if (fe_degree > Index(0))
63 {
64 for (Index i(0); i < d; ++i)
65 {
66 size *= pnos[i] * IndexType_(fe_degree) + 1;
67 noo *= 2 * fe_degree + 1;
68 }
69 }
70 else
71 {
72 for (Index i(0); i < d; ++i)
73 {
74 size *= pnos[i];
75 }
76 }
77
78 // allocate memory for vectors of matrix
79 DenseVector<IndexType_, IndexType_> vec_offsets(noo);
80 DenseVector<DataType_, IndexType_> vec_val(noo * Index(size));
81
82 // fill offsets-vector
83 IndexType_ * const poffsets(vec_offsets.elements());
84 const Index h_off((noo - 1) / 2);
85
86 // save position of main-diagonal
87 poffsets[h_off] = size - IndexType_(1);
88
89 for (Index i(0), k(1), m(1); i < d; ++i, k *= 2 * fe_degree + 1, m *= pnos[i - 1] * fe_degree + 1)
90 {
91 Index k1((k - 1) / 2);
92
93 for (IndexType_ j(1); j <= fe_degree; ++j)
94 {
95 for (IndexType_ l(0); l < k; ++l)
96 {
97 poffsets[h_off - k1 + l + k * j] = poffsets[h_off - k1 + l] + IndexType_(j * m);
98 poffsets[h_off - k1 + l - k * j] = poffsets[h_off - k1 + l] - IndexType_(j * m);
99 }
100 }
101 }
102
103 // return the matrix
104 return SparseMatrixBanded<DataType_, IndexType_>(Index(size), Index(size), vec_val, vec_offsets);
105 }

115 struct PointstarStructureFD
116 {
126 template<typename DataType_, typename IndexType_>
127 static SparseMatrixBanded<DataType_, IndexType_> value(const std::vector<IndexType_> & num_of_subintervalls)
128 {
129 const IndexType_ * const pnos(num_of_subintervalls.data());
130
131 const Index d(Index(num_of_subintervalls.size()) - 1);
132
133 // calculate dimension of the matrix
134 IndexType_ size(1);
135 for (Index i(1); i <= d; ++i)
136 {
137 size *= pnos[i] - 1;
138 }
139
140 // calculate number of offsets
141 const Index num_of_offsets(2 * d + 1);
142
143 // allocate memory for vectors of matrix
144 DenseVector<DataType_, IndexType_> vec_val(Index(size) * num_of_offsets);
145 DenseVector<IndexType_, IndexType_> vec_offsets(num_of_offsets);
146
147 // fill vec_offsets
148 IndexType_ * const poffsets(vec_offsets.elements());
149
150 poffsets[d] = IndexType_(size - 1);
151
152 Index tmp(1);
153 for (Index i(0); i < d; ++i)
154 {
155 tmp *= pnos[i] - 1;
156 poffsets[d - 1 - i] = size - IndexType_(1 + tmp);
157 poffsets[d + 1 + i] = size + IndexType_(tmp - 1);
158 }
159
160 // return the matrix
161 return SparseMatrixBanded<DataType_, IndexType_>(Index(size), Index(size), vec_val, vec_offsets);
162 }
