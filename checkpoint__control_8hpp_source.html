165 std::uint64_t _collect_checkpoint_data(std::vector<char> & buffer)
166 {
167 std::uint64_t checkpoint_size(0u);
168 std::uint64_t real_size(0u);
169 std::map<String, std::tuple<std::uint64_t, std::uint64_t>> sizes;
170
171 for (auto const & it : _checkpointable_by_identifier)
172 {
173 std::uint64_t identifierlength = (std::uint64_t)it.first.length();
174 std::uint64_t datalength = it.second->get_checkpoint_size(_config);
175
176 checkpoint_size += identifierlength + datalength + sizeof(std::uint64_t) + sizeof(std::uint64_t);
177 real_size += identifierlength + sizeof(std::uint64_t) + sizeof(std::uint64_t);
178 sizes[it.first] = std::make_tuple(identifierlength, datalength);
179 }
180
181 buffer.reserve(checkpoint_size);
182
183 for (auto const & it : _checkpointable_by_identifier)
184 {
185 char * cidentifierlength = reinterpret_cast<char *>(&std::get<0>(sizes[it.first]));
186 buffer.insert(std::end(buffer), cidentifierlength, cidentifierlength + sizeof(std::uint64_t));
187 buffer.insert(std::end(buffer), it.first.begin(), it.first.end());
188 std::uint64_t old_size = buffer.size();
189 buffer.insert(std::end(buffer), sizeof(std::uint64_t), 0); //set datalength to an arbitrary value
190 std::uint64_t ireal_size = it.second->set_checkpoint_data(buffer, _config);
191 char * csize = reinterpret_cast<char *>(&ireal_size);
192 for(std::uint64_t i(0) ; i < sizeof(std::uint64_t) ; ++i) //overwrite the the datalength
193 {
194 buffer[old_size + i] = csize[i];
195 }
196 real_size += ireal_size;
197 }
198
199 buffer.resize(real_size);
200 return real_size;
201 }
202
210 void _restore_checkpoint_data()
211 {
212 std::uint64_t stringsize;
213 std::uint64_t datasize(0);
214 std::size_t i = 0;
215 std::size_t size = _input_array.size();
216
217 // Loop over all objects stored in the checkpoint file
218 while (i < size)
219 {
220 // Get size of identifier string
221 ::memcpy(&stringsize, _input_array.data() + i, sizeof(std::uint64_t));
222 i += sizeof(std::uint64_t);
223
224 // Read the identifier string and put it as key for the calculated offset to the map
225 _offset_by_identifier[String(_input_array.data() + i, stringsize)] = i + stringsize;
226 i += stringsize;
227
228 // Get the size of the Data holding the array, to get the start point of the next checkpointable object in the array
229 ::memcpy(&datasize, _input_array.data() + i, sizeof(std::uint64_t));
230 i += sizeof(std::uint64_t) + datasize;
231 }
232 }

347 template <typename OT_>
348 void add_object(String identifier, OT_ & object)
349 {
350 XASSERTM(_checkpointable_by_identifier.count(identifier) == 0, "a object with the given identifier is already registered; choose a unique identifier");
351 auto checkpointable_object = std::make_shared<CheckpointableWrapper<OT_>>(object);

381 template <typename OT_>
382 void restore_object(String identifier, OT_ & object, bool add_to_checkpoint_control = true)
383 {
384 XASSERTM(_input_array.size() > 0u, "no input file has been loaded before");
385 XASSERTM(_offset_by_identifier.count(identifier) == 1, "input file doesn't contain data for the given identifier");
386
387 auto checkpointable_object = std::make_shared<CheckpointableWrapper<OT_>>(object);
388
389 std::uint64_t size(0);
390 char * in_data = _input_array.data() + _offset_by_identifier[identifier];
391 ::memcpy(&size, in_data, sizeof(std::uint64_t));
392 std::vector<char> buffer(in_data + sizeof(std::uint64_t), in_data + sizeof(std::uint64_t) + size);
393 checkpointable_object->restore_from_checkpoint_data(buffer);
394
395 if (add_to_checkpoint_control)
396 {
397 add_object(identifier, object);
398 }
399 }

409 void save(const String filename)
410 {
411 size_t pos = filename.rfind('.');
412 String extension = filename.substr(pos + 1);
413 String name = filename.substr(0, pos);
414
415 XASSERTM(name != "", "no complete filename consisting of name.extension given");
416
417 if (extension == "cp")
418 {
419 _save(name);
420 }
421 else
422 {
423 XASSERTM(extension != "zcp", "no zlib support, activate zlib or write uncompressed files with .cp extension instead");
424 XASSERTM(extension == "zcp", "no valid checkpoint filename choosen");
425 }
426 }

491 {
492 XASSERTM(_input_array.size() == 0, "another input file was read before");

496
497 _input_array.resize(size);
