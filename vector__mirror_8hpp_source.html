541 template<Perspective perspective_, typename DT2_, typename IT2_>
542 Index mask_scatter(const DenseVector<DT2_, IT2_>& vector, std::vector<int>& mask,
543 const int value, const Index offset = Index(0)) const
544 {
545 XASSERT(Index(mask.size()) >= vector.template size<perspective_>());
546 XASSERT(Index(mask.size()) >= this->size() + offset);
547 const Index n = this->num_indices();
548 const IT_* idx = this->indices();
549
550 for(Index i(0); i < n; ++i)
551 mask[offset + idx[i]] = value;
552
553 return vector.template size<perspective_>();
554 }
555
578 template<Perspective perspective_, typename DT2_, typename IT2_, int block_size_>
579 Index mask_scatter(const DenseVectorBlocked<DT2_, IT2_, block_size_>& vector, std::vector<int>& mask,
580 const int value, const Index offset = Index(0)) const
581 {
582 XASSERT(Index(mask.size()) >= vector.template size<perspective_>());
583 const Index n = this->num_indices();
584 const IT_* idx = this->indices();
585
586 if(perspective_ == LAFEM::Perspective::native)
587 {
588 XASSERT(Index(mask.size()) >= this->size() + offset);
589 for(Index i(0); i < n; ++i)
590 mask[offset + idx[i]] = value;
591 }
592 else // POD
593 {
594 XASSERT(Index(mask.size()) >= Index(block_size_)*this->size() + offset);
595 for(Index i(0); i < n; ++i)
596 {
597 const Index ibs = idx[i] * Index(block_size_);
598 for(int k(0); k < block_size_; ++k)
599 mask[offset + ibs + Index(k)] = value;
600 }
601 }
602 return vector.template size<perspective_>();
603 }
