317 void filter_mat(SparseMatrixBCSR<DT_, IT_, BlockSize_, BlockWidth_> & matrix) const
318 {
319 if(_sv.used_elements() == Index(0))
320 return;

324 Arch::UnitFilterBlocked::filter_unit_mat

326 _sv.template elements<LAFEM::Perspective::pod>(), _sv.indices(), _sv.used_elements(), _ignore_nans);

365 template<int BlockWidth_>
366 void filter_weak_matrix_rows(SparseMatrixBCSR<DT_, IT_, BlockSize_, BlockWidth_> & matrix_a,
367 const SparseMatrixBCSR<DT_, IT_, BlockSize_, BlockWidth_>& matrix_m) const
368 {
369 if(_sv.used_elements() == Index(0))
370 return;
371
372 if(matrix_a.val() == matrix_m.val())
373 {
374 XABORTM("Matrices are not allowed to hold the same data");
375 }
376
377 XASSERTM(_sv.size() == matrix_a.rows(), "Matrix size does not match!");
378 XASSERTM(_sv.size() == matrix_m.rows(), "Matrix size does not match!");
379
380 const IndexType* row_ptr(matrix_a.row_ptr());
381 const IndexType* col_idx(matrix_m.col_ind());
382 XASSERTM(row_ptr == matrix_m.row_ptr(), "matrix A and M must share their layout");
383 XASSERTM(col_idx == matrix_m.col_ind(), "matrix A and M must share their layout");
384
385 Arch::UnitFilterBlocked::filter_weak_matrix_rows
386 (matrix_a.template val<LAFEM::Perspective::pod>(), matrix_m.template val<LAFEM::Perspective::pod>(), row_ptr, BlockSize_, BlockWidth_,
387 _sv.template elements<LAFEM::Perspective::pod>(), _sv.indices(), _sv.used_elements());
388 }
