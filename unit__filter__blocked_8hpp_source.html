312 template<int BlockWidth_>
313 void filter_mat(SparseMatrixBCSR<DT_, IT_, BlockSize_, BlockWidth_> & matrix) const
314 {
315 if(_sv.used_elements() == Index(0))
316 return;
317
318 XASSERTM(_sv.size() == matrix.rows(), "Matrix size does not match!");
319
320 const IT_* row_ptr(matrix.row_ptr());
321 const IT_* col_idx(matrix.col_ind());
322 typename SparseMatrixBCSR<DT_, IT_, BlockSize_, BlockWidth_>::ValueType* v(matrix.val());
323
324 for(Index i(0); i < _sv.used_elements(); ++i)
325 {
326 const IT_ ix(_sv.indices()[i]);
327 const ValueType vx(_sv.elements()[i]);
328
329 // replace by unit row
330 for(IT_ j(row_ptr[ix]); j < row_ptr[ix + 1]; ++j)
331 {
332 // loop over rows in the block
333 for(int k(0); k < BlockSize_; ++k)
334 {
335 // possibly skip row if filter value is NaN
336 if(_ignore_nans && Math::isnan(vx[k]))
337 continue;
338 for(int l(0); l < BlockWidth_; ++l)
339 v[j][k][l] = DT_(0);
340 if((col_idx[j] == ix) && (k < BlockWidth_))
341 v[j][k][k] = DT_(1);
342 }
343 }
344 }
345 }

402 template<int BlockWidth_>
403 void filter_weak_matrix_rows(SparseMatrixBCSR<DT_, IT_, BlockSize_, BlockWidth_> & matrix_a,
404 const SparseMatrixBCSR<DT_, IT_, BlockSize_, BlockWidth_>& matrix_m) const
405 {
406 if(_sv.used_elements() == Index(0))
407 return;
408
409 XASSERTM(_sv.size() == matrix_a.rows(), "Matrix size does not match!");
410 XASSERTM(_sv.size() == matrix_m.rows(), "Matrix size does not match!");
411
412 const IndexType* row_ptr(matrix_a.row_ptr());
413 const IndexType* col_idx(matrix_m.col_ind());
414 XASSERTM(row_ptr == matrix_m.row_ptr(), "matrix A and M must share their layout");
415 XASSERTM(col_idx == matrix_m.col_ind(), "matrix A and M must share their layout");
416
417 Tiny::Matrix<DataType, BlockSize_, BlockWidth_>* val_a(matrix_a.val());
418 const Tiny::Matrix<DataType, BlockSize_, BlockWidth_>* val_m(matrix_m.val());
419
420 const IT_* idx = get_indices();
421 const Tiny::Vector<DataType, BlockSize_>* val = get_values();
422
423 // loop over all filter entries
424 for(Index i(0); i < _sv.used_elements(); ++i)
425 {
426 // replace row of A by scaled row of M
427 Index row(idx[i]);
428 for(IndexType j(row_ptr[row]); j < row_ptr[row + 1]; ++j)
429 {
430 //val_a[j] = val[i] * val_m[j];
431 for(int k(0); k < BlockSize_; ++k)
432 {
433 for(int l(0); l < BlockWidth_; ++l)
434 {
435 val_a[j][k][l] = val[i][k] * val_m[j][k][l];
436 }
437 }
438 }
439 }
