191 void init()
192 {
193 _nonlinear_functional.init_pre_sync();
194 // Synchronize the necessary parts
195 while(!_nonlinear_functional.sync_scalars.empty())
196 {
197 auto it = _nonlinear_functional.sync_scalars.begin();
198 if(_col_gate != nullptr)
199 {
200 *((*it).second) = _col_gate->sum(*((*it).second));
201 }
202 _nonlinear_functional.sync_scalars.erase(it);
203 }
204
205 if(_col_gate != nullptr)
206 {
207 for(auto* it:_nonlinear_functional.sync_vecs)
208 {
209 _col_gate->sync_0(*it);
210 }
211 }
212 _nonlinear_functional.init_post_sync();
213 }

225 void prepare(const VectorTypeR& vec_state, FilterType& filter)
226 {
227
228 // Prepare the patch local nonlinear functional, computing everything that does need synchronising
229 _nonlinear_functional.prepare_pre_sync(vec_state.local(), filter.local());
230
231 // Synchronize the necessary parts
232 while(!_nonlinear_functional.sync_scalars.empty())
233 {
234 auto it = _nonlinear_functional.sync_scalars.begin();
235 if(_col_gate != nullptr)
236 {
237 *((*it).second) = _col_gate->sum(*((*it).second));
238 }
239 _nonlinear_functional.sync_scalars.erase(it);
240 }
241
242 if(_col_gate != nullptr)
243 {
244 for(auto* it:_nonlinear_functional.sync_vecs)
245 {
246 _col_gate->sync_0(*it);
247 }
248 }
249
250 // Prepare the rest of the functional
251 _nonlinear_functional.prepare_post_sync(vec_state.local(), filter.local());
252
253 // Sync the filter vector in the SlipFilter
254 if(_col_gate != nullptr )
255 {
256 // For all slip filters...
257 for(auto& it : filter.local().template at<0>())
258 {
259 // get the filter vector
260 auto& slip_filter_vector = it.second.get_filter_vector();
261
262 if(slip_filter_vector.used_elements() > 0)
263 {
264 // Temporary DenseVector for syncing
265 LocalVectorTypeR tmp(slip_filter_vector.size(), DataType(0));
266
267 auto* tmp_elements = tmp.template elements<LAFEM::Perspective::native>();
268 auto* sfv_elements = slip_filter_vector.template elements<LAFEM::Perspective::native>();
269
270 // Copy sparse filter vector contents to DenseVector
271 for(Index isparse(0); isparse < slip_filter_vector.used_elements(); ++isparse)
272 {
273 Index idense(slip_filter_vector.indices()[isparse]);
274 tmp_elements[idense] = sfv_elements[isparse];
275 }
276
277 _col_gate->sync_0(tmp);
278 // Copy sparse filter vector contents to DenseVector
279 for(Index isparse(0); isparse < slip_filter_vector.used_elements(); ++isparse)
280 {
281 Index idense(slip_filter_vector.indices()[isparse]);
282 tmp_elements[idense].normalize();
283 sfv_elements[isparse] = tmp_elements[idense];
284
285 }
286 }
287
288 else
289 {
290 // Temporary DenseVector for syncing
291 LocalVectorTypeR tmp(slip_filter_vector.size(), DataType(0));
292 _col_gate->sync_0(tmp);
293 }
294 }
295 } // col_gate
296 } // prepare()

340 void eval_fval_grad(DataType& fval, VectorTypeL& grad)
341 {
342 // As the penalty term is quadratic, we have to compute the local functional values separately for summing
343 // them up over all ranks
344 const bool add_penalty_fval(false);
345 _nonlinear_functional.eval_fval_grad(fval, grad.local(), add_penalty_fval);
346
347 // Sum up over all patches
348 if(_row_gate != nullptr)
349 fval = _row_gate->sum(fval);
350
351 // Add the penalty term
352 if(get_penalty_param() > DataType(0))
353 {
354 DataType constraint(get_constraint());
355 fval += get_penalty_param()*DataType(0.5)*Math::sqr(constraint);
356 }
357
358 // Synchronize the gradient vector
359 grad.sync_0();
360 }
