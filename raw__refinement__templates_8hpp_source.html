89 RawTemplate& axis_aligned(const VertexType& a, const VertexType& b)
90 {
91 static_assert(Shape_::dimension <= 3);
92
93 static constexpr int num_verts = Shape::FaceTraits<Shape_, 0>::count;
94
95 static std::array<std::array<Real, 3>, 8> coeffs {{
96 {0, 0, 0},
97 {1, 0, 0},
98 {0, 1, 0},
99 {1, 1, 0},
100 {0, 0, 1},
101 {1, 0, 1},
102 {0, 1, 1},
103 {1, 1, 1},
104 }};
105
106 VertexType diff = b - a;
107
108 std::array<VertexType, num_verts> vertices;
109 for(int i(0); i < num_verts; ++i)
110 {
111 auto& c = coeffs[i];
112 VertexType v = a;
113 for(int j(0); j < Shape_::dimension; ++j)
114 {
115 v[j] += c[j] * diff[j];
116 }
117 for (int j(0); j < v.n; ++j)
118 {
119 ASSERTM(v[j] >= 0.0 && v[j] <= 1.0, "Invalid vertex in axis_aligned");
120 }
121 vertices[i] = v;
122 }
123 add_entity(vertices);
124
125 return *this;
126 }

128 RawTemplate& grid(std::array<Index, Shape_::dimension> size, VertexType stepsize)
129 {
130 std::array<Index, Shape_::dimension> coords = {};
131 _grid(size, stepsize, 0, coords);
132 return *this;

141 RawTemplate& recurse(Index face, const RawTemplate& tmplt)
142 {
143 const EntityType parent = entities[face];
144 entities.erase(entities.begin() + face);
145
146 for(const EntityType& entity : tmplt.entities)
147 {
148 EntityType mapped{};
149
150 for(Index i(0); i < entity.coords.size(); ++i)
151 {
152 const auto coeffs = Intern::vertex_coefficients<Shape_>(entity.coords[i]);
153 mapped.coords[i] = Intern::interpolate(parent.coords, coeffs);
154 }
155 entities.push_back(mapped);
156 }
157
158 return *this;
159 }
160
161 private:
162 void _grid(
163 const std::array<Index, Shape_::dimension>& size,
164 VertexType stepsize,
165 int dim,
166 std::array<Index, Shape_::dimension>& coords)
167 {
168 if (dim == Shape_::dimension)
169 {
170 VertexType v;
171 for (int i = 0; i < Shape_::dimension; ++i)
172 {
173 v[i] = coords[i] * stepsize[i];
174 }
175 axis_aligned(v, v + stepsize);
176 }
177 else
178 {
179 for(Index i(0); i < size[dim]; i++)
180 {
181 coords[dim] = i;
182 _grid(size, stepsize, dim + 1, coords);
183 }
184 }
185 }

189 template<typename Shape_, typename FnVert, typename FnEntity>
190 static RawTemplate<Shape_> transform_template_vertices(FnVert vertex_transform, FnEntity entity_transform, RawTemplate<Shape_>& tmplt)
191 {
192
193 RawTemplate<Shape_> result;
194 for(auto& entity : tmplt.entities)
195 {
196 RawEntity<Shape_> new_entity;
197
198 for(int i(0); i < entity.num_vertices; ++i)
199 {
200 new_entity.coords.at(i) = vertex_transform(entity.coords.at(i));
201 }
202
203 entity_transform(new_entity);
204 result.entities.push_back(new_entity);
205 }
206 return result;
207 }

212 template<typename Shape_>
213 static RawTemplate<Shape_> rotate_template_2d(RawTemplate<Shape_>& tmplt)
214 {
215 static_assert(Shape_::dimension == 2, "rotate_template_2d called for non 2D template!");
216
217 // Rotate vertex 90 degrees counterclockwise around (0.5, 0.5)
218 auto rotate = [](Tiny::Vector<Real, 2> vertex)
219 {
220 return Tiny::Vector<Real, 2>{-vertex[1] + 1, vertex[0]};
221 };
222
223 auto rotate_indices = [](RawEntity<Shape_>& entity)
224 {
225 entity.coords = Intern::rotate_arraylike_2d(entity.coords);
226 };
227
228 return transform_template_vertices(rotate, rotate_indices, tmplt);
229 }

307 template<typename TemplateMap, typename RefinementType_>
308 int create_2dtemplate_rotations(TemplateMap& map, const RefinementType_& base_type)
309 {
310 // Count base template towards created templates
311 int templates_created = 1;
312
313 RefinementType_ type = base_type;
314 auto tmplt = map[type];
315
316 for(int i = 0; i < 4; i++)
317 {
318 if(map.find(type) == map.end())
319 {
320 map[type] = tmplt;
321 templates_created += 1;
322 }
323
324 type = type.rotate_2d();
325 tmplt = rotate_template_2d(tmplt);
326 }
327
328 return templates_created;
329 }

331 template<typename TemplateMap, typename RefinementType_>
332 int create_3dtemplate_rotations(TemplateMap& map, const RefinementType_& base_type)
333 {
334 enum class Rotation
335 {
336 None,
337 X,
338 Y,
339 };
340
341 static constexpr std::array<Rotation, 6> rotations = {{
342 Rotation::None,
343 Rotation::X,
344 Rotation::Y,
345 Rotation::X,
346 Rotation::Y,
347 Rotation::X,
348 }};
349
350 // Count base template towards created templates
351 int templates_created = 1;
352
353 RefinementType_ outer_type = base_type;
354 auto outer_template = map[outer_type];
355
356 for(int outer_rotation(0); outer_rotation < 6; outer_rotation++)
357 {
358 const Rotation next_outer_rotation = rotations[outer_rotation];
359 if (next_outer_rotation == Rotation::X)
360 {
361 outer_type = outer_type.rotate_xaxis();
362 outer_template = rotate_template_xaxis(outer_template);
363 }
364 else if (next_outer_rotation == Rotation::Y)
365 {
366 outer_type = outer_type.rotate_yaxis();
367 outer_template = rotate_template_yaxis(outer_template);
368 }
369
370 RefinementType_ inner_type = outer_type;
371 auto inner_template = outer_template;
372 for(int inner_rotation(0); inner_rotation < 4; inner_rotation++)
373 {
374 if(map.find(inner_type) == map.end())
375 {
376 map[inner_type] = inner_template;
377 templates_created += 1;
378 }
379
380 inner_type = inner_type.rotate_zaxis();
381 inner_template = rotate_template_zaxis(inner_template);
382 }
383 }
384 return templates_created;
385 }
