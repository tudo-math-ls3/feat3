44 template<typename NeighborIndexSetType_, typename FacetIndexSetType_>
45 static void compute(NeighborIndexSetType_& neighbors, const FacetIndexSetType_& facet_idx)
46 {
47 Index num_cells(facet_idx.get_num_entities());
48 Index num_facets(facet_idx.get_index_bound());
49
50 XASSERT(neighbors.get_num_entities() == num_cells);
51
52 // A facet is shared by exactly 2 cells if it is interiour, and is present in exactly one cell if it is at
53 // the boundary
54 typedef Index SharedBy[2];
55 std::vector<Index> shared_by_vec(size_t(2)*num_facets);
56 auto shared_by = reinterpret_cast<SharedBy*>(shared_by_vec.data());
57
58 // ~Index(0) is the marker for "no neighbor"
59 for(Index l(0); l < num_facets; ++l)
60 {
61 shared_by[l][0] = ~Index(0);
62 shared_by[l][1] = ~Index(0);
63 }
64
65 // For each facet, find the cells sharing it
66 for(Index k(0); k < num_cells; ++k)
67 {
68 for(int j(0); j < facet_idx.num_indices; ++j)
69 {
70 // Index of the facet
71 Index l(facet_idx[k][j]);
72
73 if(shared_by[l][0] == ~Index(0))
74 shared_by[l][0] = k;
75 else if(shared_by[l][1] == ~Index(0))
76 shared_by[l][1] = k;
77 else
78 XABORTM("Facet "+stringify(l)+" is shared by cells "+stringify(shared_by[l][0])+", "+stringify(shared_by[l][1])+" and again by "+stringify(k));
79 }
80 }
81
82 // For every cell and for every facet of that cell, the neighbor at a face is the OTHER cell sharing it
83 // (if any)
84 for(Index k(0); k < num_cells; ++k)
85 {
86 for(int j(0); j < facet_idx.num_indices; ++j)
87 {
88 // Index of the facet
89 Index l(facet_idx[k][j]);
90
91 if(shared_by[l][0] == k)
92 neighbors[k][j] = shared_by[l][1];
93 else if(shared_by[l][1] == k)
94 neighbors[k][j] = shared_by[l][0];
95 else
96 XABORTM("Facet "+stringify(l)+" found at cell "+stringify(k)+" but is shared by cells "+stringify(shared_by[l][0])+", "+stringify(shared_by[l][1]));
97 }
98 }
99
100 } // compute()
101
102 }; // struct FacetNeighbors
