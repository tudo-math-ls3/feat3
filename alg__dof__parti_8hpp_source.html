363 void assemble_by_gate(const GateType& gate)
364 {
365 // set our communicator
366 this->_comm = gate.get_comm();
367 XASSERTM(this->_comm != nullptr, "need a communicator here!");
368
369 // get the communicator
370 const Dist::Comm& comm = *this->_comm;
371
372 // get my rank and neighbor ranks
373 const int my_rank = comm.rank();
374 const std::vector<int>& neighbor_ranks = gate._ranks;
375
376 // get number of local DOFs on our patch
377 const IndexType num_local_dofs = IndexType(gate._freqs.size());
378
379 // As a very first step, we have to decide which process will become the
380 // owner of each DOF. In this implementation, each DOF is owned by by the
381 // process with the lowest rank, which simplifies a lot of things.
382
383 // So, first create a vector, which stores the rank of the owner process
384 // of each of our local DOFs and format the vector to our rank, i.e. at
385 // startup assume that we own all local DOFs.
386 std::vector<int> dof_owners(num_local_dofs, my_rank);
387
388 // Now loop over all our neighbor processes
389 for(std::size_t i(0); i < neighbor_ranks.size(); ++i)
390 {
391 // Check whether the neighbors rank is less than our rank, as otherwise
392 // that particular neighbor can not own any of our local DOFs.
393 const int neighbor_rank = neighbor_ranks.at(i);
394 if(neighbor_rank < my_rank)
395 {
396 // loop over all DOFs in the mirror of that neighbor
397 const MirrorType& mirror = gate._mirrors.at(i);
398 const Index num_indices = mirror.num_indices();
399 const IndexType* idx = mirror.indices();
400 for(Index j(0); j < num_indices; ++j)
401 {
402 // get a reference to the current owner rank of that DOF and
403 // update the DOF owner if that neighbor has a lower rank
404 // than the previous "owner candidate"
405 int& dof_owner = dof_owners.at(idx[j]);
406 if(neighbor_rank < dof_owner)
407 dof_owner = neighbor_rank;
408 }
409 }
410 }
411
412 // Okay, at this point we (and all other processes) know the owners of
413 // each of our local DOFs. Now, we have to generate a vector of all
414 // local DOF indices of the DOFs that we own:
415 std::vector<IndexType> owned_dofs;
416 owned_dofs.reserve(num_local_dofs);
417
418 // Furthermore, we need the 'inverse' information, i.e. for each of our
419 // local DOFs, we have to store the "owned DOF index" if we own this
420 // particular DOF. We initialize the vector with ~0, which stands for
421 // "not my DOF":
422 std::vector<IndexType> own_dof_idx(num_local_dofs, ~IndexType(0));
423
424 // Loop over all local DOFs
425 for(IndexType i(0); i < num_local_dofs; ++i)
426 {
427 // Am I the owner of this DOF?
428 if(dof_owners[i] == my_rank)
429 {
430 // Yes, that's my DOF, so let's give it a new 'my owned DOF' index:
431 own_dof_idx.at(i) = IndexType(owned_dofs.size());
432 // And remember that I own this DOF:
433 owned_dofs.push_back(i);
434 }
435 }
436
438 XASSERTM(!owned_dofs.empty(), "this process has no DOFs!");
439
440 // get number of my owned DOFS
441 const IndexType num_owned_dofs = IndexType(owned_dofs.size());
442
443 // Next, we have to determine the global offset of our first owned DOF, which
444 // is easily obtained by an exclusive-scan over each processes owned DOF count:
445 comm.exscan(&num_owned_dofs, &this->_glob_dof_offset, 1, Dist::op_sum);
446
447 // Furthermore, we also perform an allreduce to obtain the total number of
448 // global DOFs, which is generally helpful and can be used for sanity checks.
449 comm.allreduce(&num_owned_dofs, &this->_glob_dof_count, 1, Dist::op_sum);
450
451 // Next, we have to set up the mirror for all of our owned DOFs, which we can
452 // easily generate from our "owned_dofs" vector:
453 this->_owned_mirror = _vidx2mirror(num_local_dofs, owned_dofs);
454
455 // Now we also have to create the mirrors for each of our neighbor processes:
456 for(std::size_t i(0); i < neighbor_ranks.size(); ++i)
457 {
458 // get the neighbor rank and its mirror
459 const int neighbor_rank = neighbor_ranks.at(i);
460 const MirrorType& mirror = gate._mirrors.at(i);
461
462 // There are two cases now:
463 // 1) The neighbor process has a lower rank, so it is a (potential) owner
464 // of at least one of our local DOFs.
465 // 2) The neighbor process has a greater rank, so we are an owner of at least
466 // one of the neighbor process's local DOFs.
467 if(neighbor_rank < my_rank)
468 {
469 // This is a potential "owner-neighbor", which may own one or several of
470 // our local DOFs. We call a helper function, which will create a vector
471 // of all *local* DOF indices, which are owned by that neighbor process.
472 // Note that it is perfectly legit if the vector is empty, as this simply
473 // means that all our local DOFs, which are shared with that particular
474 // neighbor, are owned by some *other* neighbor process(es) and not by
475 // the particular neighbor that we are currently considering.
476 std::vector<IndexType> v = _owner_vidx(mirror, dof_owners, neighbor_rank);
477 if(!v.empty())
478 {
479 // That neighbor owns at least one of our local DOFs, so create a mirror
480 // from the index-vector and push it into our list of "owner-neighbors".
481 // Note that the indices in the mirror are *local* DOF indices.
482 _neighbors_owner.emplace_back(std::make_pair(neighbor_rank, _vidx2mirror(num_local_dofs, v)));
483 }
484 }
485 else
486 {
487 // This is a potential "donee-neighbor", which shares one or several of
488 // our local DOFs, which we might own. We call a helper function, which
489 // will create a vector of all our *owned* DOF indices, which are shared
490 // with that particular neighbor process.
491 // Note that it is perfectly legit if the vector is empty, as this simply
492 // means that all our local DOFs, which are shared with that particular
493 // neighbor, are owned by some *other* neighbor process(es) and not by
494 // this process (i.e. us).
495 std::vector<IndexType> v = _donee_vidx(mirror, own_dof_idx);
496 if(!v.empty())
497 {
498 // We own at least of one of the neighbor's local DOFs, so create a mirror
499 // from the index-vector and push it into our list of "donee-neighbors".
500 // Note that the indices in the mirror are *owned* DOF indices.
501 _neighbors_donee.emplace_back(std::make_pair(neighbor_rank, _vidx2mirror(num_owned_dofs, v)));
502 }
503 }
504 }
505
506 // Finally, we have to determine the global DOF index for each of our local DOFs,
507 // so that we can perform a "local-to-global" DOF index lookup. This information
508 // is required for the assembly of matrices later on, so we have to store this in
509 // a member-variable vector.
510
511 // Let's create the vector of the required length, initialize all its indices to
512 // ~0, which will can be used for a sanity check later on to ensure that we know
513 // the global indices of all our local DOFs:
514 this->_glob_dof_idx.resize(num_local_dofs, ~IndexType(0));
515
516 // Next, let's loop over all DOFs that we own and assign their corresponding
517 // global DOF indices, starting with our global DOF offset, which we have already
518 // determined before:
519 for(std::size_t i(0); i < owned_dofs.size(); ++i)
520 this->_glob_dof_idx[owned_dofs[i]] = this->_glob_dof_offset + IndexType(i);
521
522 // Finally, we also need to query the global DOF indices of all our local DOFs,
523 // which are owned by some neighbor process (i.e. we are the donee for these DOFs).
524 // Of course, we also have to send the global DOF indices of all DOFs that we own
525 // to all of our "donee-neighbors".
526
527 // Allocate index buffers and post receives for all of our owner-neighbors:
528 const IndexType num_neigh_owner = IndexType(_neighbors_owner.size());
529 Dist::RequestVector recv_reqs(num_neigh_owner);
530 std::vector<std::vector<IndexType>> recv_bufs(num_neigh_owner);
531 for(IndexType i(0); i < num_neigh_owner; ++i)
532 {
533 const std::size_t num_idx = _neighbors_owner.at(i).second.num_indices();
534 recv_bufs.at(i).resize(num_idx);
535 recv_reqs[i] = comm.irecv(recv_bufs.at(i).data(), num_idx, _neighbors_owner.at(i).first);
536 }
537
538 // Allocate index buffers, fill them with the global DOF indices of our owned
539 // DOFs and send them to the donee-neighbors:
540 const IndexType num_neigh_donee = IndexType(_neighbors_donee.size());
541 Dist::RequestVector send_reqs(num_neigh_donee);
542 std::vector<std::vector<IndexType>> send_bufs(num_neigh_donee);
543 for(IndexType i(0); i < num_neigh_donee; ++i)
544 {
545 // Get the mirror for this donee-neighbor:
546 const MirrorType& mirror = _neighbors_donee.at(i).second;
547 const Index num_idx = mirror.num_indices();
548 const IndexType* mir_idx = mirror.indices();
549 // Allocate buffer of required size and translate the indices of our mirror,
550 // which are "owned DOF indices", to global DOF indices, where:
551 // global_dof_index := global_dof_offset + owned_dof_index
552 send_bufs.at(i).resize(num_idx);
553 IndexType* jdx = send_bufs.at(i).data();
554 for(IndexType j(0); j < num_idx; ++j)
555 jdx[j] = this->_glob_dof_offset + mir_idx[j];
556 send_reqs[i] = comm.isend(send_bufs.at(i).data(), num_idx, _neighbors_donee.at(i).first);
557 }
558
559 // Now process all receive requests from our owner-neighbors:
560 for(std::size_t i(0u); recv_reqs.wait_any(i); )
561 {
562 const MirrorType& mirror = _neighbors_owner.at(i).second;
563 const Index num_idx = mirror.num_indices();
564 const IndexType* mir_idx = mirror.indices();
565 const IndexType* jdx = recv_bufs.at(i).data();
566 // Scatter the global DOF indices, which our friendly owner-neighbor has
567 // provided us with, into our global DOF index vector:
568 for(IndexType j(0); j < num_idx; ++j)
569 this->_glob_dof_idx[mir_idx[j]] = jdx[j];
570 }
571
572 // wait for all sends to finish
573 send_reqs.wait_all();
574
575#ifdef DEBUG
576 // Last but not least: debug-mode sanity check
577 // We now should know the global DOF indices for all of our local DOFs:
578 for(IndexType gdi : this->_glob_dof_idx)
579 {
580 XASSERTM(gdi != ~IndexType(0), "invalid global DOF index");
581 }
582#endif // DEBUG
583 }

585 private:
586 // auxiliary function: create a mirror from a vector of indices
587 static MirrorType _vidx2mirror(const IndexType size, const std::vector<IndexType>& vidx)
588 {
589 const IndexType nidx = IndexType(vidx.size());
590 MirrorType mirror(size, nidx);
591 IndexType* idx = mirror.indices();
592 for(IndexType i(0); i < nidx; ++i)
593 idx[i] = vidx[i];
594 return mirror;
595 }
596
597 // auxiliary function: create an index vector of all *local DOFs*
598 // which are owned by the process with the given neighbor rank:
599 static std::vector<IndexType> _owner_vidx(const MirrorType& old_mir,
600 const std::vector<int>& dof_owners, const int neighbor_rank)
601 {
602 std::vector<IndexType> v;
603 v.reserve(old_mir.num_indices());
604
605 const IndexType* old_idx = old_mir.indices();
606 for(IndexType i(0); i < old_mir.num_indices(); ++i)
607 {
608 // is this neighbor the owner?
609 if(neighbor_rank == dof_owners[old_idx[i]])
610 v.push_back(old_idx[i]);
611 }
612 return v;
613 }
614
615 // auxiliary function: create an index vector all *owned DOFs*
616 // which are local DOFs of the process whose mirror is given:
617 static std::vector<IndexType> _donee_vidx(const MirrorType& old_mir,
618 const std::vector<IndexType>& own_dof_idx)
619 {
620 std::vector<IndexType> v;
621 v.reserve(old_mir.num_indices());
622
623 const IndexType* old_idx = old_mir.indices();
624 for(IndexType i(0); i < old_mir.num_indices(); ++i)
625 {
626 // do we own the DOF in this mirror?
627 if(own_dof_idx[old_idx[i]] != ~IndexType(0))
628 v.push_back(own_dof_idx[old_idx[i]]);
629 }
630 return v;
631 }

819 void download(LocalVectorType& local_vector) const
820 {
821 XASSERT(this->_alg_dof_parti != nullptr);
822 XASSERT(this->_vector.size() == this->_alg_dof_parti->get_num_owned_dofs());
823 XASSERT(local_vector.size() == this->_alg_dof_parti->get_num_local_dofs());
824
825 // get our algebraic dof partitioning object
826 const AlgDofPartiType& adp = *this->_alg_dof_parti;
827
828 // get the communicator
829 const Dist::Comm& comm = *this->get_comm();
830
831 // get the number of owner- and donee-neighbors
832 const IndexType num_neigh_owner = adp.get_num_owner_neighbors();
833 const IndexType num_neigh_donee = adp.get_num_donee_neighbors();
834
835 // The basic idea is simple:
836 // 1) Send the shared DOFs to all of our donee-neighbors
837 // 2) Receive the shared DOFs from all of our owner-neighbors
838 // 3) Scatter our own owned DOFs into our local vector
839
840 // Create receive buffers and post receives for all owner-neighbors
841 Dist::RequestVector recv_reqs(num_neigh_owner);
842 std::vector<BufferVectorType> recv_bufs(num_neigh_owner);
843 for(IndexType i(0); i < num_neigh_owner; ++i)
844 {
845 // create a vector buffer
846 // note: owner mirrors relate to local DOF indices
847 recv_bufs.at(i) = adp.get_owner_mirror(i).create_buffer(local_vector);
848 // post receive from owner neighbor
849 recv_reqs[i] = comm.irecv(recv_bufs.at(i).elements(), recv_bufs.at(i).size(), adp.get_owner_rank(i));
850 }
851
852 // Create send buffers, fill them with our owned DOFs and send this
853 // to our donee-neighbors
854 Dist::RequestVector send_reqs(num_neigh_donee);
855 std::vector<BufferVectorType> send_bufs(num_neigh_donee);
856 for(IndexType i(0); i < num_neigh_donee; ++i)
857 {
858 // get mirror, create buffer and gather the shared DOFs
859 // note: donee mirrors relate to owned DOF indices
860 const MirrorType& mir = adp.get_donee_mirror(i);
861 send_bufs.at(i) = mir.create_buffer(this->_vector);
862 mir.gather(send_bufs.at(i), this->_vector);
863 // post send to donee neighbor
864 send_reqs[i] = comm.isend(send_bufs.at(i).elements(), send_bufs.at(i).size(), adp.get_donee_rank(i));
865 }
866
867 // format the output vector
868 local_vector.format();
869
870 // scatter our own owned DOFs into our output vector
871 adp.get_owned_mirror().scatter_axpy(local_vector, this->_vector);
872
873 // process receives from our owner-neighbors
874 for(std::size_t idx(0u); recv_reqs.wait_any(idx); )
875 {
876 // scatter received DOFs into our output vector
877 adp.get_owner_mirror(IndexType(idx)).scatter_axpy(local_vector, recv_bufs.at(idx));
878 }
879
880 // at this point, all receives should have finished
881 XASSERT(recv_reqs.is_null());

1242 const IndexType off_glob = adp.get_global_dof_offset();

1244
1245 // get local filter indices
1246 const Index num_idx = filter.used_elements();

1250 const IndexType* row_ptr = this->_matrix.row_ptr();
1251 const IndexType* col_idx = this->_matrix.col_ind();

1258 const IndexType gidx = adp.map_local_to_global_index(fil_idx[i]);

1262 continue; // not my DOF
1263

1266 for(IndexType j(row_ptr[row]); j < row_ptr[row+1]; ++j)
1267 {

1342 void _assemble_buffers(const LocalMatrixType& local_matrix)
1343 {
1344 // get our partitioning
1345 const AlgDofPartiType& adp = *this->_alg_dof_parti;
1346
1347 // get out communicator
1348 const Dist::Comm& comm = *this->get_comm();
1349
1350 // get number of neighbors and allocate buffer vectors
1351 const IndexType num_neigh_owner = adp.get_num_owner_neighbors();
1352 const IndexType num_neigh_donee = adp.get_num_donee_neighbors();
1353 _owner_bufs.resize(num_neigh_owner);
1354 _donee_bufs.resize(num_neigh_donee);
1355
1356 // create the owner buffers by using the auxiliary helper function.
1357 for(IndexType i(0); i < num_neigh_owner; ++i)
1358 {
1359 this->_owner_bufs.at(i) = _asm_mat_buf(local_matrix, adp.get_owner_mirror(i),
1360 adp.get_global_dof_indices(), adp.get_num_global_dofs());
1361 }
1362
1363 // We have assembled our owner-neighbor matrix buffers, however, we cannot
1364 // assemble the donee-neighbor matrix buffers on our own. Instead, each owner
1365 // neighbor has to receive its donee-buffers from its donee-neighbors, because
1366 // only the donee knows the matrix layout of those buffers.
1367 // Yes, this is brain-twisting, but believe me, it cannot be realized otherwise.
1368
1369 // Note:
1370 // The following code is effectively just copy-&-paste from the SynchMatrix::init()
1371 // function, as the same buffer problem also arises when converting global type-0
1372 // matrices to local type-1 matrices.
1373
1374 Dist::RequestVector send_reqs(num_neigh_owner);
1375 Dist::RequestVector recv_reqs(num_neigh_donee);
1376
1377 // receive buffer dimensions vector
1378 std::vector<std::array<IndexType,4>> send_dims(num_neigh_owner);
1379 std::vector<std::array<IndexType,4>> recv_dims(num_neigh_donee);
1380
1381 // post send-buffer dimension receives
1382 for(IndexType i(0); i < num_neigh_donee; ++i)
1383 {
1384 recv_reqs[i] = comm.irecv(recv_dims.at(i).data(), std::size_t(4), adp.get_donee_rank(i));
1385 }
1386
1387 // send owner-buffer dimensions
1388 for(IndexType i(0); i < num_neigh_owner; ++i)
1389 {
1390 const BufferMatrixType& sbuf = this->_owner_bufs.at(i);
1391 send_dims.at(i)[0] = IndexType(sbuf.rows());
1392 send_dims.at(i)[1] = IndexType(sbuf.columns());
1393 send_dims.at(i)[2] = IndexType(sbuf.entries_per_nonzero());
1394 send_dims.at(i)[3] = IndexType(sbuf.used_elements());
1395 send_reqs[i] = comm.isend(send_dims.at(i).data(), std::size_t(4), adp.get_owner_rank(i));
1396 }
1397
1398 // wait for all receives to finish
1399 recv_reqs.wait_all();
1400
1401 // allocate donee-buffers
1402 for(IndexType i(0); i < num_neigh_donee; ++i)
1403 {
1404 // get the receive buffer dimensions
1405 IndexType nrows = recv_dims.at(i)[0];
1406 IndexType ncols = recv_dims.at(i)[1];
1407 IndexType nepnz = recv_dims.at(i)[2];
1408 IndexType nnze = recv_dims.at(i)[3];
1409
1410 // allocate receive buffer
1411 this->_donee_bufs.at(i) = BufferMatrixType(nrows, ncols, nnze, nepnz);
1412 }
1413
1414 // post donee-buffer row-pointer array receives
1415 for(IndexType i(0); i < num_neigh_donee; ++i)
1416 {
1417 recv_reqs[i] = comm.irecv(this->_donee_bufs.at(i).row_ptr(),
1418 this->_donee_bufs.at(i).rows()+std::size_t(1), adp.get_donee_rank(i));
1419 }
1420
1421 // wait for all previous sends to finish
1422 send_reqs.wait_all();
1423
1424 // post owner-buffer row-pointer array sends
1425 for(IndexType i(0); i < num_neigh_owner; ++i)
1426 {
1427 send_reqs[i] = comm.isend(this->_owner_bufs.at(i).row_ptr(),
1428 this->_owner_bufs.at(i).rows()+std::size_t(1), adp.get_owner_rank(i));
1429 }
1430
1431 // wait for all previous receives to finish
1432 recv_reqs.wait_all();
1433
1434 // post donee-buffer column-index array receives
1435 for(IndexType i(0); i < num_neigh_donee; ++i)
1436 {
1437 recv_reqs[i] = comm.irecv(this->_donee_bufs.at(i).col_ind(),
1438 this->_donee_bufs.at(i).used_elements(), adp.get_donee_rank(i));
1439 }
1440
1441 // wait for all previous sends to finish
1442 send_reqs.wait_all();
1443
1444 // post owner-buffer column-index array sends
1445 for(IndexType i(0); i < num_neigh_owner; ++i)
1446 {
1447 send_reqs[i] = comm.isend(this->_owner_bufs.at(i).col_ind(),
1448 this->_owner_bufs.at(i).used_elements(), adp.get_owner_rank(i));
1449 }
1450
1451 // wait for all receives and sends to finish
1452 recv_reqs.wait_all();
1453 send_reqs.wait_all();
1454 }

1463 void _assemble_structure(const LocalMatrixType& local_matrix)
1464 {
1465 // get our partitioning
1466 const AlgDofPartiType& adp = *this->_alg_dof_parti;
1467
1468 // get our matrix dimensions:
1469 // * each row of our matrix corresponds to one owned DOF
1470 // * each column corresponds to one global DOF
1471 const IndexType num_rows = adp.get_num_owned_dofs();
1472 const IndexType num_cols = adp.get_num_global_dofs();
1473
1474 // Unfortunately, assembling the matrix structure is not that easy,
1475 // because it is a union of our owned matrix structure and all of
1476 // our donee matrix buffers. We don't have a chance to determine how
1477 // many duplicate (i,j) pairs we will encounter here, so we use the
1478 // DynamicGraph class here to keep things simple.
1479
1480 // create a dynamic graph for our matrix
1481 Adjacency::DynamicGraph dynamic_graph(num_rows, num_cols);
1482
1483 // First, let us add all local matrix rows which correspond to the
1484 // owned DOFs of this process, which are given by the "owned mirror"
1485 const IndexType* loc_row_ptr = local_matrix.row_ptr();
1486 const IndexType* loc_col_idx = local_matrix.col_ind();
1487 const IndexType* own_mir_idx = adp.get_owned_mirror().indices();
1488 for(IndexType i(0); i < num_rows; ++i)
1489 {
1490 // get the local DOF index of our i-th owned DOF
1491 const IndexType lrow = own_mir_idx[i];
1492 for(IndexType j(loc_row_ptr[lrow]); j < loc_row_ptr[lrow + 1]; ++j)
1493 {
1494 // translate the local column indices into global DOF indices
1495 dynamic_graph.insert(i, adp.map_local_to_global_index(loc_col_idx[j]));
1496 }
1497 }
1498
1499 // process all donee-neighbor buffers
1500 const IndexType num_neigh_donee = adp.get_num_donee_neighbors();
1501 for(IndexType ineigh(0); ineigh < num_neigh_donee; ++ineigh)
1502 {
1503 // get the mirror and the matrix buffer of that donee neighbor
1504 // note: the donee mirror indices are owned DOF indices
1505 const MirrorType& mir = adp.get_donee_mirror(ineigh);
1506 const BufferMatrixType& buf = this->_donee_bufs.at(ineigh);
1507 XASSERT(mir.num_indices() == buf.rows());
1508 const Index num_idx = mir.num_indices();
1509 const IndexType* mir_idx = mir.indices();
1510 const IndexType* buf_ptr = buf.row_ptr();
1511 const IndexType* buf_idx = buf.col_ind();
1512
1513 // loop over all matrix buffer rows
1514 for(IndexType i(0); i < num_idx; ++i)
1515 {
1516 // the owned DOF index of our the buffer matrix row
1517 const IndexType row = mir_idx[i];
1518
1519 // loop over all buffer indices
1520 for(IndexType j(buf_ptr[i]); j < buf_ptr[i + 1]; ++j)
1521 {
1522 // insert entry into our local matrix
1523 dynamic_graph.insert(row, buf_idx[j]);
1524 }
1525 }
1526 }
1527
1528 // render into a standard graph
1529 Adjacency::Graph graph(Adjacency::RenderType::as_is, dynamic_graph);
1530
1531 // create our dof-partitoned matrix object
1532 this->_matrix = OwnedMatrixType(graph);
1533 }
