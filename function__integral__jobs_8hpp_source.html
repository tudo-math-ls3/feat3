291 template<int sm_, int sn_>
292 static DataType _calc_vorticity(const Tiny::Matrix<DataType, 2, 2, sm_, sn_>& g)
293 {
294 // 2D: (dx u2 - dy u1)^2
295 // J_ij = d/d_j f_i ==> (J_01 - J_10)^2
296 return Math::sqr(g[1][0] - g[0][1]);
297 }
298
299 template<int sm_, int sn_>
300 static DataType _calc_vorticity(const Tiny::Matrix<DataType, 3, 3, sm_, sn_>& g)
301 {
302 // 3D: (dy u3 - dz u2)^2 + (dz u1 - dx u3)^2 + (dx u2 - dy u1)
303 return
304 Math::sqr(g[2][1] - g[1][2]) + // dy u3 - dz u2 = J_21 - J_12
305 Math::sqr(g[0][2] - g[2][0]) + // dz u1 - dx u3 = J_02 - J_20
306 Math::sqr(g[1][0] - g[0][1]); // dx u2 - dy u1 = J_10 - J_01
307 }

508 void synchronize(const Dist::Comm& comm)
509 {
510 // nothing to do here?
511 if(comm.size() <= 1)
512 return;
513
514 static constexpr std::size_t max_len = std::size_t(7) +
515 (6u*sizeof(ValueType) + sizeof(GradientType) + sizeof(HessianType)) / sizeof(DataType);
516 DataType v[max_len];
517
518 // write all stuff to array
519 std::size_t k = 0u;
520 _write_to(v, k, value);
521 _write_to(v, k, grad);
522 _write_to(v, k, hess);
523 _write_to(v, k, norm_h0_sqr);
524 _write_to(v, k, norm_h1_sqr);
525 _write_to(v, k, norm_h2_sqr);
526 _write_to(v, k, norm_l1);
527 _write_to(v, k, norm_h0_sqr_comp);
528 _write_to(v, k, norm_h1_sqr_comp);
529 _write_to(v, k, norm_h2_sqr_comp);
530 _write_to(v, k, norm_l1_comp);
531 _write_to(v, k, divergence_l2_sqr);
532 _write_to(v, k, vorticity_l2_sqr);
533 XASSERT(k <= max_len);
534
535 // synchronize
536 comm.allreduce(v, v, k, Dist::op_sum);
537
538 // read stuff form array
539 k = std::size_t(0u);
540 _read_from(v, k, value);
541 _read_from(v, k, grad);
542 _read_from(v, k, hess);
543 _read_from(v, k, norm_h0_sqr);
544 _read_from(v, k, norm_h1_sqr);
545 _read_from(v, k, norm_h2_sqr);
546 _read_from(v, k, norm_l1);
547 _read_from(v, k, norm_h0_sqr_comp);
548 _read_from(v, k, norm_h1_sqr_comp);
549 _read_from(v, k, norm_h2_sqr_comp);
550 _read_from(v, k, norm_l1_comp);
551 _read_from(v, k, divergence_l2_sqr);
552 _read_from(v, k, vorticity_l2_sqr);
553
554 // write lmax values
555 k = std::size_t(0u);
556 _write_to(v, k, norm_lmax);
557 _write_to(v, k, norm_lmax_comp);
558
559 // synchronize
560 comm.allreduce(v, v, k, Dist::op_max);
561
562 // read lmax values
563 k = std::size_t(0u);
564 _read_from(v, k, norm_lmax);
565 _read_from(v, k, norm_lmax_comp);
566 }
