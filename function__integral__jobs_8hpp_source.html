294 template<int sm_, int sn_>
295 static DataType _calc_vorticity(const Tiny::Matrix<DataType, 2, 2, sm_, sn_>& g)
296 {
297 // 2D: (dx u2 - dy u1)^2
298 // J_ij = d/d_j f_i ==> (J_01 - J_10)^2
299 return Math::sqr(g[1][0] - g[0][1]);
300 }
301
302 template<int sm_, int sn_>
303 static DataType _calc_vorticity(const Tiny::Matrix<DataType, 3, 3, sm_, sn_>& g)
304 {
305 // 3D: (dy u3 - dz u2)^2 + (dz u1 - dx u3)^2 + (dx u2 - dy u1)
306 return
307 Math::sqr(g[2][1] - g[1][2]) + // dy u3 - dz u2 = J_21 - J_12
308 Math::sqr(g[0][2] - g[2][0]) + // dz u1 - dx u3 = J_02 - J_20
309 Math::sqr(g[1][0] - g[0][1]); // dx u2 - dy u1 = J_10 - J_01
310 }

399 void add_value(DataType omega, const DataType& v)
400 {
401 value += omega * v;
402 const DataType av = Math::abs(v);
403 const DataType sv = Math::sqr(v);

420 template<int n_, int sn_>
421 void add_value(DataType omega, const Tiny::Vector<DataType, n_, sn_>& v)

423 value += omega * v;
424 for(int i(0); i < n_; ++i)
425 {
426 const DataType av = Math::abs(v[i]);
427 const DataType sv = Math::sqr(v[i]);
428 norm_h0_sqr += omega * sv;

466 template<int m_, int n_, int sm_, int sn_>
467 void add_grad(DataType omega, const Tiny::Matrix<DataType, m_, n_, sm_, sn_>& g)

471 {

473 norm_h1_sqr += omega * sv;

534 void synchronize(const Dist::Comm& comm)
535 {
536 // nothing to do here?
537 if(comm.size() <= 1)
538 return;
539
540 static constexpr std::size_t max_len = std::size_t(7) +
541 (6u*sizeof(ValueType) + sizeof(GradientType) + sizeof(HessianType)) / sizeof(DataType);
542 DataType v[max_len];
543
544 // write all stuff to array
545 std::size_t k = 0u;
546 _write_to(v, k, value);
547 _write_to(v, k, grad);
548 _write_to(v, k, hess);
549 _write_to(v, k, norm_h0_sqr);
550 _write_to(v, k, norm_h1_sqr);
551 _write_to(v, k, norm_h2_sqr);
552 _write_to(v, k, norm_l1);
553 _write_to(v, k, norm_h0_sqr_comp);
554 _write_to(v, k, norm_h1_sqr_comp);
555 _write_to(v, k, norm_h2_sqr_comp);
556 _write_to(v, k, norm_l1_comp);
557 _write_to(v, k, divergence_l2_sqr);
558 _write_to(v, k, vorticity_l2_sqr);
559 XASSERT(k <= max_len);
560
561 // synchronize
562 comm.allreduce(v, v, k, Dist::op_sum);
563
564 // read stuff form array
565 k = std::size_t(0u);
566 _read_from(v, k, value);
567 _read_from(v, k, grad);
568 _read_from(v, k, hess);
569 _read_from(v, k, norm_h0_sqr);
570 _read_from(v, k, norm_h1_sqr);
571 _read_from(v, k, norm_h2_sqr);
572 _read_from(v, k, norm_l1);
573 _read_from(v, k, norm_h0_sqr_comp);
574 _read_from(v, k, norm_h1_sqr_comp);
575 _read_from(v, k, norm_h2_sqr_comp);
576 _read_from(v, k, norm_l1_comp);
577 _read_from(v, k, divergence_l2_sqr);
578 _read_from(v, k, vorticity_l2_sqr);
579
580 // write lmax values
581 k = std::size_t(0u);
582 _write_to(v, k, norm_lmax);
583 _write_to(v, k, norm_lmax_comp);
584
585 // synchronize
586 comm.allreduce(v, v, k, Dist::op_max);
587
588 // read lmax values
589 k = std::size_t(0u);
590 _read_from(v, k, norm_lmax);
591 _read_from(v, k, norm_lmax_comp);
592 }
