263 virtual Status _apply_intern(VectorType& vec_sol)
264 {
265 Statistics::add_solver_expression(std::make_shared<ExpressionStartSolve>(this->name()));
266
267 const MatrixType& matrix(this->_system_matrix);
268 const MatrixType& transp(this->_transp_matrix);
269 const FilterType& filter(this->_system_filter);
270
271 VectorType& vec_p(this->_vec_p);
272 VectorType& vec_q(this->_vec_q);
273 VectorType& vec_r(this->_vec_r);
274 VectorType& vec_s(this->_vec_s);
275 VectorType& vec_t(this->_vec_t);
276 VectorType& vec_y(this->_vec_s); // same as s
277 VectorType& vec_z(this->_vec_t); // same as t
278
279 // set initial defect
280 // r[0] := b - A*x[0]
281 Status status = this->_set_initial_defect(vec_r, vec_sol);
282 if(status != Status::progress)
283 {
284 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), status, this->get_num_iter()));
285 return status;
286 }
287
288 // apply left preconditioner to defect vector
289 // p[0] := M_L^{-1} * r[0]
290 if(!this->_apply_precond_l(vec_p, vec_r))
291 {
292 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
293 return Status::aborted;
294 }
295
296 // apply transposed matrix
297 // s[0] := A^T * p[0]
298 transp.apply(vec_s, vec_p);
299
300 // apply right preconditioner
301 // q[0] := M_R^{-1} * s[0]
302 if(!this->_apply_precond_r(vec_q, vec_s))
303 {
304 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
305 return Status::aborted;
306 }
307
308 // compute initial gamma
309 // gamma[0] := < s[0], q[0] >
310 DataType gamma = vec_s.dot(vec_q);
311
312 // start iterating
313 while(status == Status::progress)
314 {
315 IterationStats stat(*this);
316
317 // y[k] := A * q[k]
318 matrix.apply(vec_y, vec_q);
319 filter.filter_def(vec_y);
320
321 // z[k] := M_L^{-1} * y[k]
322 if(!this->_apply_precond_l(vec_z, vec_y))
323 {
324 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
325 return Status::aborted;
326 }
327
328 // compute alpha
329 // alpha[k] := gamma[k] / < y[k], z[k] >
330 DataType alpha = gamma / vec_y.dot(vec_z);
331
332 // update solution vector
333 // x[k+1] := x[k] + alpha[k] * q[k]
334 vec_sol.axpy(vec_q, vec_sol, alpha);
335
336 // update defect vector
337 // r[k+1] := r[k] - alpha[k] * y[k]
338 vec_r.axpy(vec_y, vec_r, -alpha);
339
340 // compute defect norm and check for convergence
341 status = this->_set_new_defect(vec_r, vec_sol);
342 if(status != Status::progress)
343 {
344 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), status, this->get_num_iter()));
345 return status;
346 }
347
348 // p[k+1] := p[k] - alpha[k] * z[k]
349 vec_p.axpy(vec_z, vec_p, -alpha);
350
351 // s[k+1] := A^T * p[k+1]
352 transp.apply(vec_s, vec_p);
353 filter.filter_def(vec_s);
354
355 // t[k+1] := M_R^{-1} * s[k+1]
356 if(!this->_apply_precond_r(vec_t, vec_s))
357 {
358 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
359 return Status::aborted;
360 }
361
362 // compute new gamma
363 // gamma[k+1] := < s[k+1], t[k+1] >
364 DataType gamma2 = gamma;
365 gamma = vec_s.dot(vec_t);
366
367 // compute beta
368 // beta[k] := gamma[k+1] / gamma[k]
369 DataType beta = gamma / gamma2;
370
371 // update direction vector
372 // q[k+1] := t[k+1] + beta * q[k]
373 vec_q.axpy(vec_q, vec_t, beta);
374 }
375
376 // we should never reach this point...
377 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::undefined, this->get_num_iter()));
378 return Status::undefined;
379 }
