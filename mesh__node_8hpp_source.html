50 enum class AdaptMode
51 {
52 none = 0x0,
53 chart = 0x1,
54 dual = 0x2
55 };
56
58 AdaptMode inline operator|(AdaptMode x, AdaptMode y)
59 {
60 return static_cast<AdaptMode>(int(x) | int(y));
61 }
62
63 AdaptMode inline operator&(AdaptMode x, AdaptMode y)
64 {
65 return static_cast<AdaptMode>(int(x) & int(y));
66 }
67
71 inline std::ostream& operator<<(std::ostream& os, AdaptMode mode)
72 {
73 switch(mode)
74 {
75 case AdaptMode::none:
76 return os << "none";
77 case AdaptMode::chart:
78 return os << "chart";
79 case AdaptMode::dual:
80 return os << "dual";
81 default:
82 return os << "-unknown-";
83 }
84 }

86 inline void operator<<(AdaptMode& mode, const String& mode_name)
87 {
88 if(mode_name == "none")
89 mode = AdaptMode::none;

91 mode = AdaptMode::chart;
92 else if(mode_name == "dual")

94 else
95 XABORTM("Unknown AdaptMode identifier string " + mode_name);

479 void adapt(bool recursive = true)
480 {
481 // loop over all mesh_part nodes
482 MeshPartNodeIterator it(_mesh_part_nodes.begin());
483 MeshPartNodeIterator jt(_mesh_part_nodes.end());
484 for(; it != jt; ++it)
485 {
486 // adapt child node
487 if(recursive)
488 {
489 it->second.node->adapt(true);
490 }
491
492 // adapt this node if a chart is given
493 if(it->second.chart != nullptr)
494 {
495 // we need to check whether the mesh part really exists
496 // because it may be nullptr in a parallel run, which
497 // indicates that the current patch is not adjacent to
498 // the boundary represented by the mesh part.
499 const MeshPartType* mesh_part = it->second.node->get_mesh();
500 if(mesh_part != nullptr)
501 it->second.chart->adapt(*_mesh, *mesh_part);
502 }
503 }
504 }
505
521 bool adapt_by_name(const String& part_name, bool recursive = false)
522 {
523 // Try to find the corresponding mesh_part node
524 MeshPartNodeIterator it(_mesh_part_nodes.find(part_name));
525 // Do not proceed if the mesh_part node does not exist or exists but is empty on the current patch
526 if(it == _mesh_part_nodes.end() || it->second.node->get_mesh() == nullptr)
527 return false;
528
529 // adapt child node
530 if(recursive)
531 {
532 it->second.node->adapt(true);
533 }
534
535 // Adapt this node if we have a chart
536 if(it->second.chart != nullptr)
537 {
538 it->second.chart->adapt(*_mesh, *(it->second.node->get_mesh()));
539 return true;
540 }
541
542 // no chart associated
543 return false;
544 }

987 std::unique_ptr<RootMeshNode> refine_unique(AdaptMode adapt_mode = AdaptMode::chart) const
988 {
989 // create a refinery
990 StandardRefinery<MeshType> refinery(*this->_mesh);
991
992 // create a new root mesh node
993 std::unique_ptr<RootMeshNode> fine_node = make_unique(refinery.make_unique(), this->_atlas);
994
995 // refine our children
996 this->refine_children(*fine_node);
997
998 // refine our halos
999 for(const auto& v : _halos)
1000 {
1001 if(v.second)
1002 {
1003 StandardRefinery<MeshPartType> halo_refinery(*v.second, *this->_mesh);
1004 fine_node->add_halo(v.first, halo_refinery.make_unique());
1005 }
1006 else
1007 fine_node->add_halo(v.first, nullptr);
1008 }
1009
1010 // refine our patch mesh-parts
1011 for(const auto& v : _patches)
1012 {
1013 if(v.second)
1014 {
1015 StandardRefinery<MeshPartType> patch_refinery(*v.second, *this->_mesh);
1016 fine_node->add_patch(v.first, patch_refinery.make_unique());
1017 }
1018 else
1019 fine_node->add_patch(v.first, nullptr);
1020 }
1021
1022 // adapt by chart?
1023 if((adapt_mode & AdaptMode::chart) != AdaptMode::none)
1024 {
1025 fine_node->adapt(true);
1026 }
1027
1028 // adapt dual?
1029 if((adapt_mode & AdaptMode::dual) != AdaptMode::none)
1030 {
1031 Intern::DualAdaptor<MeshType>::adapt(*fine_node->get_mesh(), *this->get_mesh());
1032 }
1033
1034 // okay
1035 return fine_node;
1036 }

1059 std::unique_ptr<RootMeshNode> extract_patch(std::vector<Index>&& elements,
1060 bool split_meshparts, bool split_halos, bool split_patches)
1061 {
1062 XASSERTM(!elements.empty(), "cannot create empty patch!");
1063
1064 // get our mesh
1065 const MeshType* base_root_mesh = this->get_mesh();
1066 XASSERTM(base_root_mesh != nullptr, "mesh node has no mesh");
1067
1068 // Step 1: create mesh part of the patch
1069 MeshPartType* patch_mesh_part = nullptr;
1070 {
1071 // create a factory for our partition
1072 PatchMeshPartFactory<MeshType> part_factory(std::forward<std::vector<Index>>(elements));
1073
1074 // create patch mesh part and add it to our map
1075 patch_mesh_part = this->add_patch(-1, part_factory.make_unique());
1076
1077 // deduct the target sets
1078 patch_mesh_part->template deduct_target_sets_from_top<MeshType::shape_dim>(
1079 base_root_mesh->get_index_set_holder());
1080 }
1081
1082 // Step 2: Create root mesh node of partition by using PatchMeshFactory
1083 std::unique_ptr<RootMeshNode> patch_node;
1084 {
1085 // create patch root mesh
1086 PatchMeshFactory<MeshType> patch_factory(*base_root_mesh, *patch_mesh_part);
1087 patch_node = RootMeshNode::make_unique(patch_factory.make_unique(), this->_atlas);
1088 }
1089
1090 // Step 3: intersect boundary and other base mesh parts
1091 if(split_meshparts)
1092 {
1093 // create mesh part splitter
1094 PatchMeshPartSplitter<MeshType> part_splitter(*base_root_mesh, *patch_mesh_part);
1095
1096 // get all mesh part names
1097 std::deque<String> part_names = this->get_mesh_part_names();
1098
1099 // loop over all base-mesh mesh parts
1100 for(auto it = part_names.begin(); it != part_names.end(); ++it)
1101 {
1102 // get base mesh part node
1103 auto* base_part_node = this->find_mesh_part_node(*it);
1104 XASSERTM(base_part_node != nullptr, String("base-mesh part '") + (*it) + "' not found!");
1105
1106 // our split mesh part
1107 std::unique_ptr<MeshPartType> split_part;
1108
1109 // get base-mesh part
1110 MeshPartType* base_part = base_part_node->get_mesh();
1111
1112 // build
1113 if((base_part != nullptr) && part_splitter.build(*base_part))
1114 {
1115 // create our mesh part
1116 split_part = part_splitter.make_unique();
1117 }
1118
1119 // Insert patch mesh part
1120 patch_node->add_mesh_part(*it, std::move(split_part), this->find_mesh_part_chart_name(*it), this->find_mesh_part_chart(*it));
1121 }
1122 }
1123
1124 // Step 4: intersect halos
1125 if(split_halos)
1126 {
1127 // create mesh part splitter
1128 PatchMeshPartSplitter<MeshType> part_splitter(*base_root_mesh, *patch_mesh_part);
1129
1130 // loop over all halo mesh parts
1131 for(auto it = this->_halos.begin(); it != this->_halos.end(); ++it)
1132 {
1133 // our split halo
1134 std::unique_ptr<MeshPartType> split_halo;
1135
1136 // get base-mesh halo
1137 MeshPartType* base_halo = it->second.get();
1138
1139 // build
1140 if((base_halo != nullptr) && part_splitter.build(*base_halo))
1141 {
1142 // create our mesh halo
1143 split_halo = part_splitter.make_unique();
1144 }
1145
1146 // Insert patch mesh halo
1147 patch_node->add_halo(it->first, std::move(split_halo));
1148 }
1149 }
1150
1151 // Step 5: intersect patches
1152 if(split_patches)
1153 {
1154 // create mesh part splitter
1155 PatchMeshPartSplitter<MeshType> part_splitter(*base_root_mesh, *patch_mesh_part);
1156
1157 // loop over all halo mesh parts
1158 for(auto it = this->_patches.begin(); it != this->_patches.end(); ++it)
1159 {

1169

1172 {
1173 // create our mesh patch
1174 split_patch = part_splitter.make_unique();
1175 }
1176
1177 // Insert patch mesh halo

1179 }
1180 }

1204 std::unique_ptr<RootMeshNode> extract_patch(
1205 std::vector<int>& comm_ranks,
1206 const Adjacency::Graph& elems_at_rank,
1207 const int rank)
1208 {
1209 // get dimensions of graph
1210 const Index num_elems = elems_at_rank.get_num_nodes_image();
1211
1212 // get our mesh
1213 const MeshType* base_root_mesh = this->get_mesh();
1214 XASSERTM(base_root_mesh != nullptr, "mesh node has no mesh");
1215
1216 // validate element count
1217 XASSERTM(num_elems == base_root_mesh->get_num_elements(), "mesh vs partition: element count mismatch");
1218
1219 // transpose the partitioning graph here, as we will need it multiple times
1220 const Adjacency::Graph ranks_at_elem(Adjacency::RenderType::transpose, elems_at_rank);
1221
1222 // Step 1: compute neighbor ranks
1223 {
1224 // get vertices-at-element
1225 const auto& verts_at_elem = base_root_mesh->template get_index_set<MeshType::shape_dim, 0>();
1226
1227 // Note:
1228 // In the following, we are building the ranks-at-rank adjacency graph in an apparently
1229 // cumbersome manner in five steps (including the ranks-at-elem graph assembly above).
1230 // Let "R>E" be the elements-at-rank graph, that is given as a parameter to this function,
1231 // and let "E>V" be the vertices-at-element graph, then we perform the following steps:
1232 //
1233 // 1: compute ranks-at-element by transposition: E>R := (R>E)^T
1234 // 2: compute elements-at-vertex by transposition: V>E := (E>V)^T
1235 // 3: compute ranks-at-vertex by composition: V>R := (V>E) * (E>R)
1236 // 4: compute vertices-at-rank by transposition: R>V := (V>R)^T
1237 // 5: compute ranks-at-rank by composition: R>R := (R>V) * (V>R)
1238 //
1239 // As said before, this looks as if it is more complicated than it has to be, because
1240 // one might also try the following three-step approach (which was, in fact, implemented
1241 // here before):
1242 //
1243 // 1. compute vertices-at-rank by composition: R>V := (R>E) * (E>V)
1244 // 2. compute ranks-at-vertex by transposition: V>R := (R>V)^T
1245 // 3: compute ranks-at-rank by composition: R>R := (R>V) * (V>R)
1246 //
1247 // The million dollar question is: why don't use the 3-step approach?
1248 // Answer: Assume that we have N elements in the base-mesh and P processes, then it
1249 // can be shown that the first step "R>V := (R>E) * (E>V)" has a runtime of O(N^2/P),
1250 // which results in quadratic runtime in N unless N = O(P). So the practically important
1251 // scenario of "many elements on few processors" lead to this bottleneck. However, it is
1252 // worth mentioning that the 3-step approach has linear runtime in the case N = O(P).
1253 // On the other hand, the 5-step approach implemented here can be shown to have
1254 // linear runtime in both N and P in any relevant case, which is the reason why it
1255 // was chosen here.
1256
1257 // build elements-at-vertex graph
1258 Adjacency::Graph elems_at_vert(Adjacency::RenderType::transpose, verts_at_elem);
1259
1260 // build ranks-at-vertex graph and transpose it
1261 Adjacency::Graph ranks_at_vert(Adjacency::RenderType::injectify, elems_at_vert, ranks_at_elem);
1262 Adjacency::Graph verts_at_rank(Adjacency::RenderType::transpose, ranks_at_vert);
1263
1264 // build ranks-at-rank (via vertices) graph
1265 Adjacency::Graph ranks_at_rank(Adjacency::RenderType::injectify, verts_at_rank, ranks_at_vert);
1266
1267 // build comm neighbor ranks vector
1268 for(auto it = ranks_at_rank.image_begin(Index(rank)); it != ranks_at_rank.image_end(Index(rank)); ++it)
1269 {
1270 if(int(*it) != rank)
1271 comm_ranks.push_back(int(*it));
1272 }
1273 }
1274
1275 // Step 2: create mesh part of the patch
1276 MeshPartType* patch_mesh_part = nullptr;
1277 {
1278 // create a factory for our partition
1279 PatchMeshPartFactory<MeshType> part_factory(Index(rank), elems_at_rank);
1280
1281 // ensure that the partition is not empty
1282 if(part_factory.empty())
1283 {
1284 String msg("Rank ");
1285 msg += stringify(rank);
1286 msg += " received empty patch from partitioner!";
1287 XABORTM(msg);
1288 }
1289
1290 // create patch mesh part and add it to our map
1291 patch_mesh_part = this->add_patch(rank, part_factory.make_unique());
1292
1293 // deduct the target sets
1294 patch_mesh_part->template deduct_target_sets_from_top<MeshType::shape_dim>(
1295 base_root_mesh->get_index_set_holder());
1296 }
1297
1298 // Step 3: Create root mesh node of partition by using PatchMeshFactory
1299 std::unique_ptr<RootMeshNode> patch_node;
1300 {
1301 // create patch root mesh
1302 PatchMeshFactory<MeshType> patch_factory(*base_root_mesh, *patch_mesh_part);
1303 patch_node = RootMeshNode::make_unique(patch_factory.make_unique(), this->_atlas);
1304 }
1305
1306 // Step 4: intersect boundary and other base mesh parts
1307 {
1308 // create mesh part splitter
1309 PatchMeshPartSplitter<MeshType> part_splitter(*base_root_mesh, *patch_mesh_part);
1310
1311 // get all mesh part names
1312 std::deque<String> part_names = this->get_mesh_part_names();
1313
1314 // loop over all base-mesh mesh parts
1315 for(auto it = part_names.begin(); it != part_names.end(); ++it)
1316 {
1317 // get base mesh part node
1318 auto* base_part_node = this->find_mesh_part_node(*it);
1319 XASSERTM(base_part_node != nullptr, String("base-mesh part '") + (*it) + "' not found!");
1320
1321 // our split mesh part
1322 std::unique_ptr<MeshPartType> split_part;
1323
1324 // get base-mesh part
1325 MeshPartType* base_part = base_part_node->get_mesh();
1326
1327 // build
1328 if((base_part != nullptr) && part_splitter.build(*base_part))
1329 {
1330 // create our mesh part
1331 split_part = part_splitter.make_unique();
1332 }
1333
1334 // Insert patch mesh part
1335 patch_node->add_mesh_part(*it, std::move(split_part), this->find_mesh_part_chart_name(*it), this->find_mesh_part_chart(*it));
1336 }
1337 }
1338
1339 // Step 5: Create halos
1340 {
1341 // create halo factory
1342 PatchHaloFactory<MeshType> halo_factory(ranks_at_elem, *base_root_mesh, *patch_mesh_part);
1343
1344 // loop over all comm ranks
1345 for(auto it = comm_ranks.begin(); it != comm_ranks.end(); ++it)
1346 {
1347 // build halo
1348 halo_factory.build(Index(*it));
1349
1350 // create halo mesh part and add to our map
1351 patch_node->add_halo(*it, halo_factory.make_unique());
1352 }
1353 }
1354
1355 // return patch mesh part
1356 return patch_node;
1357 }
