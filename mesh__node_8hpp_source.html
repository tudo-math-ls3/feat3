52 enum class AdaptMode
53 {
54 none = 0x0,
55 chart = 0x1,
56 dual = 0x2
57 };
58
60 AdaptMode inline operator|(AdaptMode x, AdaptMode y)
61 {
62 return static_cast<AdaptMode>(int(x) | int(y));
63 }
64
65 AdaptMode inline operator&(AdaptMode x, AdaptMode y)
66 {
67 return static_cast<AdaptMode>(int(x) & int(y));
68 }

73 inline std::ostream& operator<<(std::ostream& os, AdaptMode mode)
74 {
75 switch(mode)
76 {
77 case AdaptMode::none:
78 return os << "none";
79 case AdaptMode::chart:
80 return os << "chart";
81 case AdaptMode::dual:
82 return os << "dual";
83 default:
84 return os << "-unknown-";
85 }

475 void adapt(bool recursive = true)
476 {
477 // loop over all mesh_part nodes
478 MeshPartNodeIterator it(_mesh_part_nodes.begin());
479 MeshPartNodeIterator jt(_mesh_part_nodes.end());
480 for(; it != jt; ++it)
481 {
482 // adapt child node
483 if(recursive)
484 {
485 it->second.node->adapt(true);
486 }
487
488 // adapt this node if a chart is given
489 if(it->second.chart != nullptr)
490 {
491 // we need to check whether the mesh part really exists
492 // because it may be nullptr in a parallel run, which
493 // indicates that the current patch is not adjacent to
494 // the boundary represented by the mesh part.
495 const MeshPartType* mesh_part = it->second.node->get_mesh();
496 if(mesh_part != nullptr)
497 it->second.chart->adapt(*_mesh, *mesh_part);
498 }
499 }
500 }
501
517 bool adapt_by_name(const String& part_name, bool recursive = false)
518 {
519 // Try to find the corresponding mesh_part node
520 MeshPartNodeIterator it(_mesh_part_nodes.find(part_name));
521 // Do not proceed if the mesh_part node does not exist or exists but is empty on the current patch
522 if(it == _mesh_part_nodes.end() || it->second.node->get_mesh() == nullptr)
523 return false;
524
525 // adapt child node
526 if(recursive)
527 {
528 it->second.node->adapt(true);
529 }
530
531 // Adapt this node if we have a chart
532 if(it->second.chart != nullptr)
533 {
534 it->second.chart->adapt(*_mesh, *(it->second.node->get_mesh()));
535 return true;
536 }
537
538 // no chart associated
539 return false;
540 }

983 std::unique_ptr<RootMeshNode> refine_unique(AdaptMode adapt_mode = AdaptMode::chart) const
984 {
985 // create a refinery
986 StandardRefinery<MeshType> refinery(*this->_mesh);
987
988 // create a new root mesh node
989 std::unique_ptr<RootMeshNode> fine_node = make_unique(refinery.make_unique(), this->_atlas);
990
991 // refine our children
992 this->refine_children(*fine_node);
993
994 // refine our halos
995 for(const auto& v : _halos)
996 {
997 StandardRefinery<MeshPartType> halo_refinery(*v.second, *this->_mesh);
998 fine_node->add_halo(v.first, halo_refinery.make_unique());
999 }
1000
1001 // refine our patch mesh-parts
1002 for(const auto& v : _patches)
1003 {
1004 StandardRefinery<MeshPartType> patch_refinery(*v.second, *this->_mesh);
1005 fine_node->add_patch(v.first, patch_refinery.make_unique());
1006 }
1007
1008 // adapt by chart?
1009 if((adapt_mode & AdaptMode::chart) != AdaptMode::none)
1010 {
1011 fine_node->adapt(true);
1012 }
1013
1014 // adapt dual?
1015 if((adapt_mode & AdaptMode::dual) != AdaptMode::none)
1016 {
1017 Intern::DualAdaptor<MeshType>::adapt(*fine_node->get_mesh(), *this->get_mesh());
1018 }
1019
1020 // okay
1021 return fine_node;
1022 }

1042 std::unique_ptr<RootMeshNode> extract_patch(
1043 std::vector<int>& comm_ranks,
1044 const Adjacency::Graph& elems_at_rank,
1045 const int rank)
1046 {
1047 // get dimensions of graph
1048 const Index num_elems = elems_at_rank.get_num_nodes_image();
1049
1050 // get our mesh
1051 const MeshType* base_root_mesh = this->get_mesh();
1052 XASSERTM(base_root_mesh != nullptr, "mesh node has no mesh");
1053
1054 // validate element count
1055 XASSERTM(num_elems == base_root_mesh->get_num_elements(), "mesh vs partition: element count mismatch");
1056
1057 // transpose the partitioning graph here, as we will need it multiple times
1058 const Adjacency::Graph ranks_at_elem(Adjacency::RenderType::transpose, elems_at_rank);
1059
1060 // Step 1: compute neighbor ranks
1061 {
1062 // get vertices-at-element
1063 const auto& verts_at_elem = base_root_mesh->template get_index_set<MeshType::shape_dim, 0>();
1064
1065 // Note:
1066 // In the following, we are building the ranks-at-rank adjacency graph in an apparently
1067 // cumbersome manner in five steps (including the ranks-at-elem graph assembly above).
1068 // Let "R>E" be the elements-at-rank graph, that is given as a parameter to this function,
1069 // and let "E>V" be the vertices-at-element graph, then we perform the following steps:
1070 //
1071 // 1: compute ranks-at-element by transposition: E>R := (R>E)^T
1072 // 2: compute elements-at-vertex by transposition: V>E := (E>V)^T
1073 // 3: compute ranks-at-vertex by composition: V>R := (V>E) * (E>R)
1074 // 4: compute vertices-at-rank by transposition: R>V := (V>R)^T
1075 // 5: compute ranks-at-rank by composition: R>R := (R>V) * (V>R)
1076 //
1077 // As said before, this looks as if it is more complicated than it has to be, because
1078 // one might also try the following three-step approach (which was, in fact, implemented
1079 // here before):
1080 //
1081 // 1. compute vertices-at-rank by composition: R>V := (R>E) * (E>V)
1082 // 2. compute ranks-at-vertex by transposition: V>R := (R>V)^T
1083 // 3: compute ranks-at-rank by composition: R>R := (R>V) * (V>R)
1084 //
1085 // The million dollar question is: why don't use the 3-step approach?
1086 // Answer: Assume that we have N elements in the base-mesh and P processes, then it
1087 // can be shown that the first step "R>V := (R>E) * (E>V)" has a runtime of O(N^2/P),
1088 // which results in quadratic runtime in N unless N = O(P). So the practically important
1089 // scenario of "many elements on few processors" lead to this bottleneck. However, it is
1090 // worth mentioning that the 3-step approach has linear runtime in the case N = O(P).
1091 // On the other hand, the 5-step approach implemented here can be shown to have
1092 // linear runtime in both N and P in any relevant case, which is the reason why it
1093 // was chosen here.
1094
1095 // build elements-at-vertex graph
1096 Adjacency::Graph elems_at_vert(Adjacency::RenderType::transpose, verts_at_elem);
1097
1098 // build ranks-at-vertex graph and transpose it
1099 Adjacency::Graph ranks_at_vert(Adjacency::RenderType::injectify, elems_at_vert, ranks_at_elem);
1100 Adjacency::Graph verts_at_rank(Adjacency::RenderType::transpose, ranks_at_vert);
1101
1102 // build ranks-at-rank (via vertices) graph
1103 Adjacency::Graph ranks_at_rank(Adjacency::RenderType::injectify, verts_at_rank, ranks_at_vert);
1104
1105 // build comm neighbor ranks vector
1106 for(auto it = ranks_at_rank.image_begin(Index(rank)); it != ranks_at_rank.image_end(Index(rank)); ++it)
1107 {
1108 if(int(*it) != rank)
1109 comm_ranks.push_back(int(*it));
1110 }
1111 }
1112
1113 // Step 2: create mesh part of the patch
1114 MeshPartType* patch_mesh_part = nullptr;
1115 {
1116 // create a factory for our partition
1117 PatchMeshPartFactory<MeshType> part_factory(Index(rank), elems_at_rank);
1118
1119 // ensure that the partition is not empty
1120 if(part_factory.empty())
1121 {
1122 String msg("Rank ");
