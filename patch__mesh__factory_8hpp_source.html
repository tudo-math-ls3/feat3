61 virtual void fill_vertex_set(VertexSetType& vertex_set) override
62 {
63 // fetch vertex-target-indices of the patch mesh
64 const typename MeshPartType::template TargetSet<0>::Type& idx(_patch_part.template get_target_set<0>());
65 typedef typename VertexSetType::VertexType VertexType;
66
67 // fetch base-mesh vertex set
68 const VertexSetType& vertex_set_in(_base_mesh.get_vertex_set());
69
70 // loop over all vertices
71 const Index num_verts(idx.get_num_entities());
72 for(Index i(0); i < num_verts; ++i)
73 {
74 VertexType& vo(vertex_set[i]);
75 const VertexType& vi(vertex_set_in[idx[i]]);
76 for(int j(0); j < num_coords_; ++j)
77 {
78 vo[j] = vi[j];
79 }
80 }
81 }
82
83 virtual void fill_index_sets(IndexSetHolderType& index_set_holder) override
84 {
85 // set up base-mesh num_entities
86 Index num_entities[Shape_::dimension + 1];
87 for(int i(0); i <= Shape_::dimension; ++i)
88 {
89 num_entities[i] = _base_mesh.get_num_entities(i);
90 }
91
92 // build index sets
93 Intern::PatchIndexMapping<Shape_>::apply(
94 index_set_holder,
95 _base_mesh.get_index_set_holder(),
96 _patch_part.get_target_set_holder(),
97 num_entities);
98 }
