194 void add_variable(const String& name, double value = 0.0)
195 {
196 XASSERTM(_extra_vars.find(name) == _extra_vars.end(), "extra variable '" + name + "' already added");
197 XASSERTM(name.compare_no_case("x") != 0, "variable name 'x' is reserved and cannot be used as extra variable");
198 XASSERTM(name.compare_no_case("y") != 0, "variable name 'y' is reserved and cannot be used as extra variable");
199 XASSERTM(name.compare_no_case("z") != 0, "variable name 'z' is reserved and cannot be used as extra variable");
200 _extra_vars.emplace(name, value);
201 }

212 void set_variable(const String& name, double value)
213 {
214 auto it = _extra_vars.find(name);
215 XASSERTM(it != _extra_vars.end(), "no variable named '" + name + "' found");
216 it->second = value;
217 }

231 void parse(const String& function)
232 {
233 // add variables to our parser
234 String vars("x");
235 if(dim_ > 1) vars += ",y";
236 if(dim_ > 2) vars += ",z";
237
238 // add extra variables
239 for(auto& v : _extra_vars)
240 (vars += ",") += v.first;
241
242 // try to parse the function
243 const int ret = _parser.Parse(function.c_str(), vars.c_str());
244 if(ret >= 0)
245 {
246 String msg(_parser.ErrorMsg());
247 msg.append("\n>>> '");
248 msg.append(function);
249 msg.append("'");
250 if(ret < int(function.size()))
251 {
252 // ret contains the index of the first invalid character in the input string
253 // append an additional line to mark the faulty character
254 msg.append("\n>>>");
255 msg.append(String(std::size_t(ret+2), '-'));
256 msg.append("^");
257 }
258 throw ParsedFunctionParseError(msg);
259 }
260
261 // optimize the parsed function
262 _parser.Optimize();
263 }

265 template<typename Traits_>
266 class Evaluator :
267 public Analytic::Function::Evaluator<Traits_>
268 {
269 public:
271 typedef typename Traits_::DataType DataType;
273 typedef typename Traits_::PointType PointType;
275 typedef typename Traits_::ValueType ValueType;
276
277 private:
278 // Note: We need to create a copy of the parser rather than just a reference to it,
279 // as the 'Eval' function of the parser is not thread-safe, whereas our Evaluators
280 // are always meant to be so!
282 ::FunctionParser _parser;
284 std::vector<double> _vars;
285
286 public:
287 explicit Evaluator(const ParsedScalarFunction& function) :
288 _parser(function._parser),
289 _vars(std::size_t(dim_), 0.0)
290 {
291 // force deep copy to avoid race conditions in case of multi-threading
292 _parser.ForceDeepCopy();
293 // set extra vars
294 for(const auto& v : function._extra_vars)
295 _vars.push_back(v.second);
296 }
297
298 ValueType value(const PointType& point)
299 {
300 // copy coordinates
301 for(int i(0); i < dim_; ++i)
302 _vars[std::size_t(i)] = double(point[i]);
303
304 // evaluate the parser
305 const double val = _parser.Eval(_vars.data());
306
307 // check for errors
308 switch(_parser.EvalError())
309 {
310 case 0: // no error
311 break;
312
313 case 1: // division by zero
314 throw ParsedFunctionEvalError("Error in ParsedScalarFunction evaluation: division by zero");
315
316 case 2: // sqrt error
317 case 3: // log error
318 case 4: // trigonometric error
319 throw ParsedFunctionEvalError("Error in ParsedScalarFunction evaluation: illegal input value");
320
321 case 5: // recursion error
322 throw ParsedFunctionEvalError("Error in ParsedScalarFunction evaluation: maximum recursion depth reached");
323
324 default: // ???
325 throw ParsedFunctionEvalError("Error in ParsedScalarFunction evaluation: unknown error");
326 }
327
328 // return value
329 return ValueType(val);
330 }
331 }; // class ParsedScalarFunction::Evaluator<...>
332 }; // class ParsedScalarFunction

487 void parse(const String& function)
488 {
489 // add variables to our parser
490 String vars("x");
491 if(dom_dim_ > 1) vars += ",y";
492 if(dom_dim_ > 2) vars += ",z";
493
494 // add extra variables
495 for(auto& v : _extra_vars)
496 (vars += ",") += v.first;
497
498 String sfunc(function);
499 if(sfunc.starts_with('['))
500 sfunc.pop_front();
501 if(sfunc.ends_with(']'))
502 sfunc.pop_back();
503 sfunc.trim_me();
504
505 // split function string
506 std::deque<String> sfuncs = sfunc.split_by_charset("'");
507 if(sfuncs.size() != _parsers.size())
508 {
509 String msg("Invalid number of formulae: expected ");
510 msg.append(stringify(_parsers.size()));
511 msg.append(" but got ");
512 msg.append(stringify(sfuncs.size()));
513 throw ParsedFunctionParseError(msg);
514 }
515
516 // parse all functions
517 for(std::size_t i(0); i < _parsers.size(); ++i)
518 {
519 // try to parse the function
520 const int ret = _parsers.at(i).Parse(sfuncs.at(i).c_str(), vars.c_str());
521 if(ret >= 0)
522 {
523 String msg(_parsers.at(i).ErrorMsg());
524 msg.append("\n>>> '");
525 msg.append(sfuncs.at(i));
526 msg.append("'");
527 if(ret < int(sfuncs.at(i).size()))
528 {
529 // ret contains the index of the first invalid character in the input string
530 // append an additional line to mark the faulty character
531 msg.append("\n>>>");
532 msg.append(String(std::size_t(ret+2), '-'));
533 msg.append("^");
534 }
535 throw ParsedFunctionParseError(msg);
536 }
537
538 // optimize the parsed function
539 _parsers.at(i).Optimize();
540 }
