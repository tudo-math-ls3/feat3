269 virtual Status _set_initial_defect(const VectorType& vec_def, const VectorType& vec_sol) override
270 {
271 // store new defect
272 this->_def_init = this->_def_cur = this->_calc_def_norm(vec_def, vec_sol);
273 this->_num_iter = Index(0);
274 this->_num_stag_iter = Index(0);
275
276 _fval_init = _fval;
277 _fval_prev = Math::huge<DataType>();
278 _steplength = DataType(1);
279 _ls_its = Index(0);
280
281 Statistics::add_solver_expression(
282 std::make_shared<ExpressionDefect>(this->name(), this->_def_init, this->get_num_iter()));
283
284 if(this->_plot_iter())
285 {
286 String msg = this->_plot_name
287 + ": " + stringify(this->_num_iter).pad_front(this->_iter_digits)
288 + " (" + stringify(this->_ls_its).pad_front(_ls_iter_digits) + ")"
289 + " : " + stringify_fp_sci(this->_def_cur)
290 + " / " + stringify_fp_sci(this->_def_cur / this->_def_init)
291 + " : " + stringify_fp_sci(this->_fval);
292
293 this->_print_line(msg);
294 }
295
296 // Ensure that the initial fval and defect are neither NaN nor infinity
297 if( !(Math::isfinite(this->_def_init) && Math::isfinite(this->_fval_init)) )
298 {
299 return Status::aborted;
300 }
301
302 // Check if the initial defect is already low enough
303 if(this->_def_init <= Math::sqr(Math::eps<DataType>()))
304 {
305 return Status::success;
306 }
307
308 // continue iterating
309 return Status::progress;
310 }

327 virtual Status _set_new_defect(const VectorType& vec_r, const VectorType& vec_sol) override
328 {
329 // increase iteration count
330 ++this->_num_iter;
331
332 // first, let's see if we have to compute the defect at all
333 bool calc_def = false;
334 calc_def = calc_def || (this->_min_iter < this->_max_iter);
335 calc_def = calc_def || this->_plot_iter();
336 calc_def = calc_def || (this->_min_stag_iter > Index(0));
337
338 // compute new defect
339 if(calc_def)
340 {
341 this->_def_cur = this->_calc_def_norm(vec_r, vec_sol);
342 Statistics::add_solver_expression(
343 std::make_shared<ExpressionDefect>(this->name(), this->_def_cur, this->get_num_iter()));
344 }
345
346 // plot?
347 if(this->_plot_iter())
348 {
349 String msg = this->_plot_name
350 + ": " + stringify(this->_num_iter).pad_front(this->_iter_digits)
351 + " (" + stringify(this->_ls_its).pad_front(_ls_iter_digits) + ")"
352 + " : " + stringify_fp_sci(this->_def_cur)
353 + " / " + stringify_fp_sci(this->_def_cur / this->_def_init)
354 + " : " + stringify_fp_sci(this->_fval)
355 + " : " + stringify_fp_sci(this->_steplength);
356
357 // print message line via comm (if available)
358 this->_print_line(msg);
359 }
360
361 // ensure that the defect is neither NaN nor infinity
362 if(!Math::isfinite(this->_def_cur))
363 {
364 return Status::aborted;
365 }
366
367 // is diverged?
368 if(this->is_diverged())
369 {
370 return Status::diverged;
371 }
372
373 // minimum number of iterations performed?
374 if(this->_num_iter < this->_min_iter)
375 {
376 return Status::progress;
377 }
378
379 // maximum number of iterations performed?
380 if(this->_num_iter >= this->_max_iter)
381 {
382 return Status::max_iter;
383 }
384
385 // Check for convergence of the gradient norm (relative AND absolute criterion)
386 if(this->is_converged())
387 {
388 return Status::success;
389 }
390
391 // Check for convergence wrt. the function value improvement if _tol_fval says so
392 if(_tol_fval > DataType(0))
393 {
394 // This is the factor for the relative function value
395 DataType scale(Math::max(_fval, _fval_prev));
396 // Make sure it is at least 1
397 scale = Math::max(scale, DataType(1));
398 // Check for success
399 if(Math::abs(_fval_prev - _fval) <= _tol_fval*scale)
400 {
401 return Status::success;
402 }
403 }
404
405 if(_steplength <= _tol_step)
406 {
407 return Status::success;
408 }
409
410 // If there were too many stagnated iterations, the solver is stagnated
411 if(this->_min_stag_iter > 0 && this->_num_stag_iter > this->_min_stag_iter)
412 {
413 return Status::stagnated;
414 }
415
416 // continue iterating
417 return Status::progress;
418 }
