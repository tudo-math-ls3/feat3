185 template<typename IndexType_, int conv_dim_>
186 static DataType calc_sd_v_norm(const LAFEM::DenseVectorBlocked<DataType_, IndexType_, conv_dim_>& convect)
187 {
188 const auto* vals = convect.elements();
189 DataType_ r = DataType_(0);
190 for(Index i(0); i < convect.size(); ++i)
191 r = Math::max(r, vals[i].norm_euclid());
192 return r;
193 }

207 template<typename LocalVector_, typename Mirror_>
208 static DataType calc_sd_v_norm(const Global::Vector<LocalVector_, Mirror_>& convect)
209 {
210 const auto* gate = convect.get_gate();
211 if(gate != nullptr)
212 return gate->max(calc_sd_v_norm(convect.local()));
213 else
214 return calc_sd_v_norm(convect.local());

371 explicit BurgersAssemblyTaskBase(const Job_& job_) :
372 job(job_),
373 tol_eps(Math::sqrt(Math::eps<DataType>())),
374 deformation(job_.deformation),
375 nu(job_.nu),
376 theta(job_.theta),
377 beta(job_.beta),
378 frechet_beta(job_.frechet_beta),
379 sd_delta(job_.sd_delta),
380 sd_nu(job_.sd_nu),
381 sd_v_norm(job_.sd_v_norm),
382 need_diff(Math::abs(nu) > DataType(0)),
383 need_conv(Math::abs(beta) > DataType(0)),
384 need_conv_frechet(Math::abs(frechet_beta) > DataType(0)),
385 need_reac(Math::abs(theta) > DataType(0)),
386 need_streamdiff((Math::abs(sd_delta) > DataType(0)) && (sd_v_norm > tol_eps)),
387 need_loc_v(need_conv),
388 need_loc_grad_v(need_conv_frechet),
389 need_mean_v(need_streamdiff),
390 need_local_h(need_streamdiff),
391 space(job_.space),
392 trafo(space.get_trafo()),
393 trafo_eval(trafo),
394 space_eval(space),
395 dof_mapping(space),
396 cubature_rule(Cubature::ctor_factory, job_.cubature_factory),
397 trafo_data(),
398 space_data(),
399 gather_conv(job_.convection_vector),
400 num_local_dofs(0),
401 local_conv_dofs(),
402 loc_v(),
403 mean_v(),
404 loc_grad_v(),
405 streamdiff_coeffs(),
406 barycenter(),
407 local_h(DataType(0)),
408 local_delta(DataType(0))
409 {
410 // compute reference element barycenter
411 for(int i(0); i < shape_dim; ++i)
412 barycenter[i] = Shape::ReferenceCell<typename SpaceType::ShapeType>::template centre<DataType>(i);
413 }
414
421 void prepare(const Index cell)
422 {
423 // prepare dof mapping
424 dof_mapping.prepare(cell);
425
426 // prepare trafo evaluator
427 trafo_eval.prepare(cell);
428
429 // prepare space evaluator
430 space_eval.prepare(trafo_eval);
431
432 // fetch number of local dofs
433 num_local_dofs = space_eval.get_num_local_dofs();
434
435 // gather our local convection dofs
436 local_conv_dofs.format();
437 gather_conv(local_conv_dofs, dof_mapping);
438
439 // compute mesh width if necessary
440 if(need_mean_v || need_local_h || need_streamdiff)
441 {
442 // reset local h and delta
443 local_h = local_delta = DataType(0);
444
445 // evaluate trafo and space at barycenter
446 trafo_eval(trafo_data, barycenter);
447 space_eval(space_data, trafo_data);
448
449 // compute velocity at barycenter
450 mean_v.format();
451 for(int i(0); i < num_local_dofs; ++i)
452 mean_v.axpy(space_data.phi[i].value, local_conv_dofs[i]);
453
454 // compute norm of mean velocity
455 const DataType local_norm_v = mean_v.norm_euclid();
456
457 // do we have a non-zero velocity?
458 if(local_norm_v > tol_eps)
459 {
460 // compute local mesh width w.r.t. mean velocity
461 local_h = trafo_eval.width_directed(mean_v) * local_norm_v;
462
463 // compute local Re_T
464 const DataType local_re = (local_norm_v * local_h) / this->sd_nu;
465
466 // compute local delta
467 local_delta = this->sd_delta * (local_h / this->sd_v_norm) * (DataType(2)*local_re) / (DataType(1) + local_re);
468 }
469 }
470 }
471
482 void prepare_point(int cubature_point)
483 {
484 // compute trafo data
485 trafo_eval(trafo_data, cubature_rule.get_point(cubature_point));
486
487 // compute basis function data
488 space_eval(space_data, trafo_data);
489
490 // evaluate convection function and its gradient (if required)
491 if(need_loc_v || need_conv || need_streamdiff)
492 {
493 loc_v.format();
494 for(int i(0); i < num_local_dofs; ++i)
495 {
496 // update velocity value
497 loc_v.axpy(space_data.phi[i].value, local_conv_dofs[i]);
498 }
499 }
500 if(need_loc_grad_v || need_conv_frechet)
501 {
502 loc_grad_v.format();
503 for(int i(0); i < num_local_dofs; ++i)
504 {
505 // update velocity gradient
506 loc_grad_v.add_outer_product(local_conv_dofs[i], space_data.phi[i].grad);
507 }
508 }
509
510 // evaluate streamline diffusion coefficients
511 if(need_streamdiff)
512 {
513 for(int i(0); i < num_local_dofs; ++i)
514 {
515 streamdiff_coeffs[i] = Tiny::dot(loc_v, space_data.phi[i].grad);
516 }
517 }
518 }
519
523 void finish()
524 {
525 // finish evaluators
526 space_eval.finish();
527 trafo_eval.finish();
528
529 // finish dof mapping
530 dof_mapping.finish();
531 }
532
536 void combine()
537 {
538 // nothing to do here
539 }

561 template<typename Job_, typename DataType_, int block_size_>
562 class BurgersBlockedAssemblyTaskBase :
563 public BurgersAssemblyTaskBase<Job_, DataType_>
564 {
565 public:
566 typedef BurgersAssemblyTaskBase<Job_, DataType_> BaseClass;
567
568 typedef DataType_ DataType;
569
570 using BaseClass::max_local_dofs;
571 using BaseClass::num_local_dofs;
572
573 protected:
575 typedef Tiny::Matrix<DataType_, block_size_, block_size_> MatrixValue;
576 typedef Tiny::Matrix<MatrixValue, max_local_dofs, max_local_dofs> LocalMatrixType;
577 LocalMatrixType local_matrix;
578
579 public:
586 explicit BurgersBlockedAssemblyTaskBase(const Job_& job_) :
587 BaseClass(job_),
588 local_matrix()
589 {
590 }
591
598 void assemble_burgers_point(const DataType weight)
599 {
600 // assemble diffusion matrix?
601 if(this->need_diff)
602 {
603 // assemble gradient-tensor diffusion
604
605 // test function loop
606 for(int i(0); i < this->num_local_dofs; ++i)
607 {
608 // trial function loop
609 for(int j(0); j < this->num_local_dofs; ++j)
610 {
611 // compute scalar value
612 const DataType value = this->nu * weight * Tiny::dot(this->space_data.phi[i].grad, this->space_data.phi[j].grad);
613
614 // update local matrix
615 local_matrix[i][j].add_scalar_main_diag(value);
616 }
617 }
618
619 // assemble deformation tensor?
620 if(this->deformation)
621 {
622 // test function loop
623 for(int i(0); i < this->num_local_dofs; ++i)
624 {
625 // trial function loop
626 for(int j(0); j < this->num_local_dofs; ++j)
627 {
628 // add outer product of grad(phi) and grad(psi)
629 local_matrix[i][j].add_outer_product(this->space_data.phi[j].grad, this->space_data.phi[i].grad, this->nu * weight);
630 }
631 }
632 }
633 }
634
635 // assemble convection?
636 if(this->need_conv)
637 {
638 // test function loop
639 for(int i(0); i < this->num_local_dofs; ++i)
640 {
641 // trial function loop
642 for(int j(0); j < this->num_local_dofs; ++j)
643 {
644 // compute scalar value
645 const DataType value = this->beta * weight * this->space_data.phi[i].value * Tiny::dot(this->loc_v, this->space_data.phi[j].grad);
646
647 // update local matrix
648 local_matrix[i][j].add_scalar_main_diag(value);
649 }
650 }
651 }
652
653 // assemble convection Frechet?
654 if(this->need_conv_frechet)
655 {
656 // test function loop
657 for(int i(0); i < this->num_local_dofs; ++i)
658 {
659 // trial function loop
660 for(int j(0); j < this->num_local_dofs; ++j)
661 {
662 // compute scalar value
663 const DataType value = this->frechet_beta * weight * this->space_data.phi[i].value * this->space_data.phi[j].value;
664
665 // update local matrix
666 local_matrix[i][j].axpy(value, this->loc_grad_v);
667 }
668 }
669 }
670
671 // assemble reaction?
672 if(this->need_reac)
673 {
674 // test function loop
675 for(int i(0); i < this->num_local_dofs; ++i)
676 {
677 // trial function loop
678 for(int j(0); j < this->num_local_dofs; ++j)
679 {
680 // compute scalar value
681 const DataType value = this->theta * weight * this->space_data.phi[i].value * this->space_data.phi[j].value;
682
683 // update local matrix
684 local_matrix[i][j].add_scalar_main_diag(value);
685 }
686 }
687 }
688 }
689
696 void assemble_streamline_diffusion(const DataType weight)
697 {
698 // assemble streamline diffusion?
699 if(this->need_streamdiff && (this->local_delta > this->tol_eps))
700 {
701 // test function loop
702 for(int i(0); i < this->num_local_dofs; ++i)
703 {
704 // trial function loop
705 for(int j(0); j < this->num_local_dofs; ++j)
706 {
707 // compute scalar value
708 const DataType value = this->local_delta * weight * this->streamdiff_coeffs[i] * this->streamdiff_coeffs[j];
709
710 // update local matrix
711 local_matrix[i][j].add_scalar_main_diag(value);
712 }
713 }
714 }
715 }
716
720 void assemble()
721 {
722 local_matrix.format();
723
724 // loop over all quadrature points and integrate
725 for(int point(0); point < this->cubature_rule.get_num_points(); ++point)
726 {
727 // prepare trafo and space for cubature point
728 this->prepare_point(point);
729
730 // precompute cubature weight
731 const DataType weight = this->trafo_data.jac_det * this->cubature_rule.get_weight(point);
732
733 // assemble the Burgers operator
734 this->assemble_burgers_point(weight);
735
736 // assemble the streamline diffusion
737 this->assemble_streamline_diffusion(weight);
738 } // continue with next cubature point
739 }
740 }; // class BurgersBlockedAssemblyTaskBase<...>
