25 template<typename Vector_, typename Mirror_>
26 static Vector_ _deslag_vector(const Vector_& vector, const Mirror_& mirror)
27 {
28 const Index num_idx = mirror.num_indices();
29 Vector_ vector_x(num_idx);
30 const auto* v = vector.elements();
31 auto* x = vector_x.elements();
32 const auto* idx_f = mirror.indices();
33 for(Index i(0); i < num_idx; ++i)
34 x[i] = v[idx_f[i]];
35 return vector_x;
36 }
37
39 template<typename Matrix_, typename Mirror_>
40 static Matrix_ _deslag_matrix_rows(const Matrix_& matrix, const Mirror_& row_mirror)
41 {
42 const Index num_idx = row_mirror.num_indices();
43 const auto* idx_f = row_mirror.indices();
44
45 const auto* row_ptr_s = matrix.row_ptr();
46 const auto* col_idx_s = matrix.col_ind();
47 Index num_nze = 0u;
48 for(Index i(0); i < num_idx; ++i)
49 num_nze += row_ptr_s[idx_f[i] + 1u] - row_ptr_s[idx_f[i]];
50
51 Matrix_ matrix_x(num_idx, matrix.columns(), num_nze);
52 auto* row_ptr_x = matrix_x.row_ptr();
53 auto* col_idx_x = matrix_x.col_ind();
54 auto* val_x = matrix_x.val();
55 const auto* val_s = matrix.val();
56 row_ptr_x[0] = 0u;
57 for(Index i(0); i < num_idx; ++i)
58 {
59 Index row_s = idx_f[i];
60 Index k(row_ptr_x[i]);
61 for(auto j(row_ptr_s[row_s]); j < row_ptr_s[row_s+1u]; ++j, ++k)
62 {
63 col_idx_x[k] = col_idx_s[j];
64 val_x[k] = val_s[j];
65 }
66 row_ptr_x[i+1u] = k;
67 }
68 return matrix_x;
69 }
70
72 template<typename Matrix_, typename Mirror_>
73 static Matrix_ _deslag_matrix_cols(const Matrix_& matrix, const Mirror_& col_mirror)
74 {
75 const Index num_idx = col_mirror.num_indices();
76 const auto* idx_f = col_mirror.indices();
77
78 std::vector<Index> col_map(matrix.columns(), ~Index(0));
79 for(Index i(0); i < num_idx; ++i)
80 col_map[idx_f[i]] = i;
81
82 const Index num_rows = matrix.rows();
83 const auto* row_ptr_s = matrix.row_ptr();
84 const auto* col_idx_s = matrix.col_ind();
85 Index used_elems = matrix.used_elements();
86 Index num_nze = 0u;
87 for(Index i(0); i < used_elems; ++i)
88 num_nze += (col_map[col_idx_s[i]] != ~Index(0));
89
90 Matrix_ matrix_x(num_rows, num_idx, num_nze);
91 auto* row_ptr_x = matrix_x.row_ptr();
92 auto* col_idx_x = matrix_x.col_ind();
93 auto* val_x = matrix_x.val();
94 const auto* val_s = matrix.val();
95 row_ptr_x[0] = 0u;
96 for(Index i(0); i < num_rows; ++i)
97 {
98 Index k(row_ptr_x[i]);
99 for(auto j(row_ptr_s[i]); j < row_ptr_s[i+1u]; ++j)
100 {
101 if(col_map[col_idx_s[j]] != ~Index(0))
102 {
103 col_idx_x[k] = col_map[col_idx_s[j]];
104 val_x[k] = val_s[j];
105 ++k;
106 }
107 }
108 row_ptr_x[i+1u] = k;
109 }
110 return matrix_x;
111 }

157 template<typename SystemLevel_, typename DomainLevel_, typename SpaceLambda_,
158 typename TransferLambda_, typename GateLambda_, typename MuxerLambda_>
159 static void _assemble_scalar_transfer(SystemLevel_& sys_lvl_f, const SystemLevel_& sys_lvl_c,
160 const Control::Domain::VirtualLevel<Control::Domain::VoxelDomainLevelWrapper<DomainLevel_>>& virt_lvl_f,
161 const Control::Domain::VirtualLevel<Control::Domain::VoxelDomainLevelWrapper<DomainLevel_>>& virt_lvl_c,
162 const String& cubature, bool trunc, bool shrink, SpaceLambda_&& space_lambda,
163 TransferLambda_&& transfer_lambda, GateLambda_&& gate_lambda, MuxerLambda_&& muxer_lambda)
164 {
165 typedef LAFEM::VectorMirror<typename SystemLevel_::DataType, typename SystemLevel_::IndexType> MirrorType;
166 typedef LAFEM::DenseVector<typename SystemLevel_::DataType, typename SystemLevel_::IndexType> ScalarVectorType;
167
168 // do we have a slag level?
169 const bool have_slag(virt_lvl_f->slag_level);
170
171 // we have: C -> S >> F, where '->' is the prolongation and '>>' is the deslagging operator
172 const DomainLevel_& level_f = have_slag ? *virt_lvl_f->slag_level : static_cast<const DomainLevel_&>(*virt_lvl_f);
173 const DomainLevel_& level_c = virt_lvl_c.is_child() ? virt_lvl_c.level_c() : *virt_lvl_c;
174
175 //const auto& space_f = level_f.space;
176 const auto& space_f = *space_lambda(level_f);
177 const auto& space_c = *space_lambda(level_c);
178
179 // get local transfer operator
180 auto& loc_trans = *transfer_lambda(sys_lvl_f);
181
182 // get local transfer matrices
183 auto& loc_prol = loc_trans.get_mat_prol();
184 auto& loc_rest = loc_trans.get_mat_rest();
185 auto& loc_trunc = loc_trans.get_mat_trunc();
186
187 // assemble structure?
188 if (loc_prol.empty())
189 Assembly::SymbolicAssembler::assemble_matrix_2lvl(loc_prol, space_f, space_c);
190
191 // if we need a truncation matrix, then compute its sparsity pattern now
192 if(trunc)
193 loc_trunc.transpose(loc_prol);
194
195 // create local weight vectors
196 auto loc_vec_weight_p = loc_prol.create_vector_l();
197 auto loc_vec_weight_t = loc_prol.create_vector_r();
198
199 // assemble prolongation matrix
200 loc_prol.format();
201 loc_vec_weight_p.format();
202 loc_vec_weight_t.format();
203
204 // assemble prolongation matrix
205 Assembly::GridTransfer::assemble_prolongation(loc_prol, loc_vec_weight_p, space_f, space_c, cubature);
206
207 // assemble truncation matrix if desired
208 if(trunc)
209 Assembly::GridTransfer::assemble_truncation(loc_trunc, loc_vec_weight_t, space_f, space_c, cubature);
210
211 // do we have a slag level?
212 if(have_slag)
213 {
214 // assemble a slag mirror
215 const auto* slag_part = level_f.find_patch_part(-1);
216 XASSERTM(slag_part != nullptr, "slag patch part not found!");
217 MirrorType slag_mirror;
218 Assembly::MirrorAssembler::assemble_mirror(slag_mirror, space_f, *slag_part);
219
220 loc_vec_weight_p = _deslag_vector(loc_vec_weight_p, slag_mirror);
221 loc_prol = _deslag_matrix_rows(loc_prol, slag_mirror);
222 if(trunc)
223 {
224 loc_trunc = _deslag_matrix_cols(loc_trunc, slag_mirror);
225 }
226 }
227
228 // synchronize weight vector using the gate
229 gate_lambda(sys_lvl_f)->sync_0(loc_vec_weight_p);
230
231 // invert components
232 loc_vec_weight_p.component_invert(loc_vec_weight_p);
233
234 // scale prolongation matrix
235 loc_prol.scale_rows(loc_prol, loc_vec_weight_p);
236
237 // shrink prolongation matrix if desired
238 if(shrink)
239 loc_prol.shrink(1E-3 * loc_prol.max_abs_element());
240
241 // copy and transpose
242 loc_rest = loc_prol.transpose();
243
244 // do we need a truncation operator?
245 if(!trunc)
246 return;
247
248 // We now need to synchronize the weight vector in analogy to the prolongation matrix assembly.
249 // Note that the weight vector is now a coarse-level vector, so we need to synchronize over
250 // the coarse-level gate. This may be a bit more complicated if the coarse level is a ghost
251 // level, as in this case we have to join/split around the synchronize operation.
252
253 // synchronize weight vector using the muxer/gate
254 if(!virt_lvl_c.is_child())
255 {
256 XASSERT(&sys_lvl_f != &sys_lvl_c);
257
258 // The coarse level is a simple (non-child) level that exists on all processes,
259 // so simply synchronize over the coarse-level gate:
260 gate_lambda(sys_lvl_c)->sync_0(loc_vec_weight_t);
261 }
262 else if(virt_lvl_c.is_parent())
263 {
264 XASSERT(&sys_lvl_f != &sys_lvl_c);
265
266 // The coarse level is a child level and this is one of the parent processes which contain
267 // the coarse-level gate. So we first need to join the weight vector onto the parent processes
268 // first, then synch that joined vector over the parent gate and finally split the result
269 // to all child processes -- this "emulates" a synch over the (non-existent) coarse-level
270 // child gate, which is what we actually require here...
271
272 // create temporary vector on parent partitioning
273 ScalarVectorType loc_tmp(gate_lambda(sys_lvl_c)->get_freqs().size());
274
275 // join child weights over muxer
276 muxer_lambda(sys_lvl_f)->join(loc_vec_weight_t, loc_tmp);
277
278 // sync over coarse gate
279 gate_lambda(sys_lvl_c)->sync_0(loc_tmp);
280
281 // split over muxer
282 muxer_lambda(sys_lvl_f)->split(loc_vec_weight_t, loc_tmp);
283 }
284 else // ghost
285 {
286 // The coarse level is a ghost level, i.e. a child but not a parent. In this case, we
287 // only have to participate in the join/send operations of the muxer, which are part
288 // of the operations executed on the parents handled by the else-if case above.
289
290 muxer_lambda(sys_lvl_f)->join_send(loc_vec_weight_t);
291
292 // parent performs sync over its gate here (see above else-if)
293
294 muxer_lambda(sys_lvl_f)->split_recv(loc_vec_weight_t);
295 }
296
297 // invert components
298 loc_vec_weight_t.component_invert(loc_vec_weight_t);
299
300 // scale prolongation matrix
301 loc_trunc.scale_rows(loc_trunc, loc_vec_weight_t);
302
303 // shrink prolongation matrix if desired
304 if(shrink)
305 loc_trunc.shrink(1E-3 * loc_trunc.max_abs_element());
306 }

352 template<typename SystemLevel_, typename DomainLevel_, typename SpaceLambda_,
353 typename TransferLambda_, typename GateLambda_, typename MuxerLambda_>
354 static void _assemble_blocked_transfer(SystemLevel_& sys_lvl_f, const SystemLevel_& sys_lvl_c,
355 const Control::Domain::VirtualLevel<Control::Domain::VoxelDomainLevelWrapper<DomainLevel_>>& virt_lvl_f,
356 const Control::Domain::VirtualLevel<Control::Domain::VoxelDomainLevelWrapper<DomainLevel_>>& virt_lvl_c,
357 const String& cubature, bool trunc, bool shrink, SpaceLambda_&& space_lambda,
358 TransferLambda_&& transfer_lambda, GateLambda_&& gate_lambda, MuxerLambda_&& muxer_lambda)
359 {
360 typedef LAFEM::VectorMirror<typename SystemLevel_::DataType, typename SystemLevel_::IndexType> MirrorType;
361 typedef LAFEM::DenseVector<typename SystemLevel_::DataType, typename SystemLevel_::IndexType> ScalarVectorType;
362
363 // do we have a slag level?
364 const bool have_slag(virt_lvl_f->slag_level);
365
366 // we have: C -> S >> F, where '->' is the prolongation and '>>' is the deslagging operator
367 const DomainLevel_& level_f = have_slag ? *virt_lvl_f->slag_level : static_cast<const DomainLevel_&>(*virt_lvl_f);
368 const DomainLevel_& level_c = virt_lvl_c.is_child() ? virt_lvl_c.level_c() : *virt_lvl_c;
369
370 //const auto& space_f = level_f.space;
371 const auto& space_f = *space_lambda(level_f);
372 const auto& space_c = *space_lambda(level_c);
373
374 // get local transfer operator
375 auto& loc_trans = *transfer_lambda(sys_lvl_f);
376
377 // get local transfer matrices
378 auto& loc_prol_wrapped = loc_trans.get_mat_prol();
379 auto& loc_rest_wrapped = loc_trans.get_mat_rest();
380 auto& loc_trunc_wrapped = loc_trans.get_mat_trunc();
381
382 // get the unwrapped types
383 auto& loc_prol = loc_prol_wrapped.unwrap();
384 auto& loc_rest = loc_rest_wrapped.unwrap();
385 auto& loc_trunc = loc_trunc_wrapped.unwrap();
386
387 // assemble structure?
388 if (loc_prol.empty())
389 Assembly::SymbolicAssembler::assemble_matrix_2lvl(loc_prol, space_f, space_c);
390
391 // if we need a truncation matrix, then compute its sparsity pattern now
392 if(trunc)
393 loc_trunc.transpose(loc_prol);
394
395 // create local weight vector
396 auto loc_vec_weight_p = loc_prol.create_vector_l();
397 auto loc_vec_weight_t = loc_prol.create_vector_r();
398
399 // assemble prolongation matrix
400 loc_prol.format();
401 loc_vec_weight_p.format();
402 loc_vec_weight_t.format();
403
404 // assemble prolongation matrix
405 Assembly::GridTransfer::assemble_prolongation(loc_prol, loc_vec_weight_p, space_f, space_c, cubature);
406
407 // assemble truncation matrix if desired
408 if(trunc)
409 Assembly::GridTransfer::assemble_truncation(loc_trunc, loc_vec_weight_t, space_f, space_c, cubature);
410
411 // do we have a slag level?
412 if(have_slag)
413 {
414 // assemble a slag mirror
415 const auto* slag_part = level_f.find_patch_part(-1);
416 XASSERTM(slag_part != nullptr, "slag patch part not found!");
417 MirrorType slag_mirror;
418 Assembly::MirrorAssembler::assemble_mirror(slag_mirror, space_f, *slag_part);
419
420 loc_vec_weight_p = _deslag_vector(loc_vec_weight_p, slag_mirror);
421 loc_prol = _deslag_matrix_rows(loc_prol, slag_mirror);
422 if(trunc)
423 {
424 loc_trunc = _deslag_matrix_cols(loc_trunc, slag_mirror);
425 }
426 }
427
428 // create a temporary scalar gate
429 Global::Gate<ScalarVectorType, MirrorType> scalar_gate;
430 scalar_gate.convert(*gate_lambda(sys_lvl_f), loc_vec_weight_p.clone(LAFEM::CloneMode::Allocate));
431
432 // synchronize weight vector using the scalar gate
433 scalar_gate.sync_0(loc_vec_weight_p);
434
435 // invert components
436 loc_vec_weight_p.component_invert(loc_vec_weight_p);
437
438 // scale prolongation matrix
439 loc_prol.scale_rows(loc_prol, loc_vec_weight_p);
440
441 // shrink prolongation matrix if desired
442 if(shrink)
443 loc_prol.shrink(1E-3 * loc_prol.max_abs_element());
444
445 // copy and transpose
446 loc_rest = loc_prol.transpose();
447
448 // do we need a truncation operator?
449 if(!trunc)
450 return;
451
452 // We now need to synchronize the weight vector in analogy to the prolongation matrix assembly.
453 // Note that the weight vector is now a coarse-level vector, so we need to synchronize over
454 // the coarse-level gate. This may be a bit more complicated if the coarse level is a ghost
455 // level, as in this case we have to join/split around the synchronize operation.
456
457 // synchronize weight vector using the muxer/gate
458 if(!virt_lvl_c.is_child())
459 {
460 XASSERT(&sys_lvl_f != &sys_lvl_c);
461
462 // The coarse level is a simple (non-child) level that exists on all processes,
463 // so simply synchronize over the coarse-level gate:
464
465 // create a temporary scalar gate on the coarse level
466 Global::Gate<ScalarVectorType, MirrorType> scalar_gate_c;
467 scalar_gate_c.convert(*gate_lambda(sys_lvl_c), ScalarVectorType(space_c.get_num_dofs()));
468
469 // sync over scalar coarse gate
470 scalar_gate_c.sync_0(loc_vec_weight_t);
471 }
472 else if(virt_lvl_c.is_parent())
473 {
474 XASSERT(&sys_lvl_f != &sys_lvl_c);
475
476 // The coarse level is a child level and this is one of the parent processes which contain
477 // the coarse-level gate. So we first need to join the weight vector onto the parent processes
478 // first, then synch that joined vector over the parent gate and finally split the result
479 // to all child processes -- this "emulates" a synch over the (non-existent) coarse-level
480 // child gate, which is what we actually require here...
481
482 // get the number of the coarse level dofs from the gate's frequency vector, which is different
483 // than space_c.get_num_dofs(), because the first one is defined on the parent patch, whereas
484 // the latter one is defined on the child patch
485 const Index num_dofs_c = gate_lambda(sys_lvl_c)->get_freqs().size();
486
487 // create a temporary scalar gate on the coarse level
488 Global::Gate<ScalarVectorType, MirrorType> scalar_gate_c;
489 scalar_gate_c.convert(*gate_lambda(sys_lvl_c), ScalarVectorType(num_dofs_c));
490
491 // create a temporary scalar muxer on the fine level
492 Global::Muxer<ScalarVectorType, MirrorType> scalar_muxer_f;
493 scalar_muxer_f.convert(*muxer_lambda(sys_lvl_f), ScalarVectorType(space_f.get_num_dofs()));
494
495 // create temporary vector on parent partitioning
496 ScalarVectorType loc_tmp(num_dofs_c);
497
498 // join child weights over muxer
499 scalar_muxer_f.join(loc_vec_weight_t, loc_tmp);
500
501 // sync over scalar coarse gate
502 scalar_gate_c.sync_0(loc_tmp);
503
504 // split over muxer
505 scalar_muxer_f.split(loc_vec_weight_t, loc_tmp);
506 }
507 else // ghost
508 {
509 // The coarse level is a ghost level, i.e. a child but not a parent. In this case, we
510 // only have to participate in the join/send operations of the muxer, which are part
511 // of the operations executed on the parents handled by the else-if case above.
512
513 // create a temporary scalar muxer on the fine level
514 Global::Muxer<ScalarVectorType, MirrorType> scalar_muxer_f;
515 scalar_muxer_f.convert(*muxer_lambda(sys_lvl_f), ScalarVectorType(space_f.get_num_dofs()));
516
517 scalar_muxer_f.join_send(loc_vec_weight_t);
518
519 // parent performs sync over its gate here (see above else-if)
520
521 scalar_muxer_f.split_recv(loc_vec_weight_t);
522 }
523
524 // invert components
525 loc_vec_weight_t.component_invert(loc_vec_weight_t);
526
527 // scale prolongation matrix
528 loc_trunc.scale_rows(loc_trunc, loc_vec_weight_t);
529
530 // shrink prolongation matrix if desired
531 if(shrink)
532 loc_trunc.shrink(1E-3 * loc_trunc.max_abs_element());
533 }
