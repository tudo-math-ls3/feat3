72 static constexpr SpaceTags eval_caps =
73 // reference values, gradients and hessians are supplied by our derived CRTP class
74 // (if available), so we mask that out from the reference capabilities
75 (ref_caps_ & (SpaceTags::ref_value | SpaceTags::ref_grad | SpaceTags::ref_hess)) |
76 // we supply values if the derived class supplies reference values and the
77 // trafo supplies domain point coordinates:
78 ((*(ref_caps_ & SpaceTags::ref_value) && *(trafo_caps & TrafoTags::dom_point)) ?
79 SpaceTags::value : SpaceTags::none) |
80 // we supply gradients if the derived class supplies reference gradients and
81 // the trafo supplies domain point coordinates and jacobian inverse matrices
82 // (which we require for the first-order chain rule)
83 ((*(ref_caps_ & SpaceTags::ref_grad) && *(trafo_caps & (TrafoTags::dom_point | TrafoTags::jac_inv))) ?
84 SpaceTags::grad : SpaceTags::none) |
85 // we supply hessians if the derived class supplies reference hessians and
86 // the trafo supplies domain point coordinates, jacobian inverse matrices and
87 // hessian inverse tensors (which we require for the second-order chain rule)
88 ((*(ref_caps_ & SpaceTags::ref_hess) && *(trafo_caps & (TrafoTags::dom_point | TrafoTags::jac_inv | TrafoTags::hess_inv))) ?
89 SpaceTags::hess : SpaceTags::none);

101 template<SpaceTags cfg_>
102 struct ConfigTraits
103 {
105 static constexpr SpaceTags config =
106 ( cfg_ & (SpaceTags::value | SpaceTags::grad | SpaceTags::hess)) |
107 // we need reference hessians if hessians are required
108 (*(cfg_ & (SpaceTags::ref_hess | SpaceTags::hess)) ? SpaceTags::ref_hess : SpaceTags::none) |
109 // we need reference gradients if gradients or hessians are required
110 (*(cfg_ & (SpaceTags::ref_grad | SpaceTags::grad | SpaceTags::hess)) ? SpaceTags::ref_grad : SpaceTags::none) |
111 // we need reference values if values are required
112 (*(cfg_ & (SpaceTags::ref_value | SpaceTags::value)) ? SpaceTags::ref_value : SpaceTags::none);
113
115 static constexpr TrafoTags trafo_config = TrafoTags::dom_point |
116 // we need jacobian inverse matrices for gradient or hessian computation by the chain rule
117 (*(config & (SpaceTags::grad | SpaceTags::hess)) ? TrafoTags::jac_inv : TrafoTags::none) |
118 // we need hessan inverse tensors for hessian computation by the chain rule
119 (*(config & SpaceTags::hess) ? TrafoTags::hess_inv : TrafoTags::none);

229 template<typename SpaceData_, typename TrafoData_>
230 static void trans_values(SpaceData_& space_data, const TrafoData_&)
231 {
232 // loop over all basis functions
233 for(int i(0); i < SpaceData_::max_local_dofs; ++i)
234 {
235 // and copy the basis function value
236 space_data.phi[i].value = space_data.phi[i].ref_value;
237 }
238 }
239
240 template<typename SpaceData_, typename TrafoData_>
241 static void trans_gradients(SpaceData_& space_data, const TrafoData_& trafo_data)
242 {
243 // loop over all basis functions
244 for(int i(0); i < SpaceData_::max_local_dofs; ++i)
245 {
246 // and apply the first-order chain rule
247 space_data.phi[i].grad.set_vec_mat_mult(space_data.phi[i].ref_grad, trafo_data.jac_inv);
248 }
249 }
250
251 template<typename SpaceData_, typename TrafoData_>
252 static void trans_hessians(SpaceData_& space_data, const TrafoData_& trafo_data)
253 {
254 // loop over all basis functions
255 for(int i(0); i < SpaceData_::max_local_dofs; ++i)
256 {
257 // and apply the second-order chain rule
258 space_data.phi[i].hess.set_double_mat_mult(space_data.phi[i].ref_hess, trafo_data.jac_inv, trafo_data.jac_inv);
259 space_data.phi[i].hess.add_vec_tensor_mult(space_data.phi[i].ref_grad, trafo_data.hess_inv);
260 }
261 }
