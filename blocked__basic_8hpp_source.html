123 template<typename DomainLevel_>
124 void assemble_gate(const Domain::VirtualLevel<DomainLevel_>& virt_dom_lvl)
125 {
126 const auto& dom_level = virt_dom_lvl.level();
127 const auto& dom_layer = virt_dom_lvl.layer();
128 const auto& space = dom_level.space;
129
130 // set the gate comm
131 this->gate_sys.set_comm(dom_layer.comm_ptr());
132
133 // loop over all ranks
134 for(Index i(0); i < dom_layer.neighbor_count(); ++i)
135 {
136 int rank = dom_layer.neighbor_rank(i);
137
138 // try to find our halo
139 auto* halo = dom_level.find_halo_part(rank);
140 XASSERT(halo != nullptr);
141
142 // assemble the mirror
143 SystemMirror mirror_sys;
144 Assembly::MirrorAssembler::assemble_mirror(mirror_sys, space, *halo);
145
146 // push mirror into gate
147 this->gate_sys.push(rank, std::move(mirror_sys));
148 }
149
150 // create local template vector
151 LocalSystemVector tmpl_s(space.get_num_dofs());
152
153 // compile gate
154 this->gate_sys.compile(std::move(tmpl_s));
155 }

157 template<typename DomainLevel_>
158 void assemble_coarse_muxer(const Domain::VirtualLevel<DomainLevel_>& virt_lvl_coarse)
159 {
160 // assemble muxer parent
161 if(virt_lvl_coarse.is_parent())
162 {
163 XASSERT(virt_lvl_coarse.is_child());
164
165 const auto& layer_c = virt_lvl_coarse.layer_c();
166 const DomainLevel_& level_p = virt_lvl_coarse.level_p();
167
168 // loop over all children
169 for(Index i(0); i < layer_c.child_count(); ++i)
170 {
171 const auto* child = level_p.find_patch_part(int(i));
172 XASSERT(child != nullptr);
173 SystemMirror child_mirror;
174 Assembly::MirrorAssembler::assemble_mirror(child_mirror, level_p.space, *child);
175 this->coarse_muxer_sys.push_child(std::move(child_mirror));
176 }
177 }
178
179 // assemble muxer child
180 if(virt_lvl_coarse.is_child())
181 {
182 const auto& layer_c = virt_lvl_coarse.layer_c();
183 const DomainLevel_& level_c = virt_lvl_coarse.level_c();
184
185 // manually set up an identity gather/scatter matrix
186 Index n = level_c.space.get_num_dofs();
187 SystemMirror parent_mirror(n, n);
188 auto* idx = parent_mirror.indices();
189 for(Index i(0); i < n; ++i)
190 idx[i] = i;
191
192 // set parent and sibling comms
193 this->coarse_muxer_sys.set_parent(
194 layer_c.sibling_comm_ptr(),
195 layer_c.get_parent_rank(),
196 std::move(parent_mirror)
197 );
198
199 // compile muxer
200 LocalSystemVector vec_tmp(level_c.space.get_num_dofs());
201 this->coarse_muxer_sys.compile(vec_tmp);
202 }

205 template<typename DomainLevel_, typename Cubature_>
206 void assemble_transfer(
207 const Domain::VirtualLevel<DomainLevel_>& virt_lvl_fine,
208 const Domain::VirtualLevel<DomainLevel_>& virt_lvl_coarse,
209 const Cubature_& cubature)
210 {
211 // get fine and coarse domain levels
212 const DomainLevel_& level_f = *virt_lvl_fine;
213 const DomainLevel_& level_c = virt_lvl_coarse.is_child() ? virt_lvl_coarse.level_c() : *virt_lvl_coarse;
214
215 const auto& space_f = level_f.space;
216 const auto& space_c = level_c.space;
217
218 // get local transfer operator
219 LocalSystemTransfer& loc_trans = this->transfer_sys.local();
220
221 // get local transfer matrices
222 LocalSystemTransferMatrix& loc_prol_wrapped = loc_trans.get_mat_prol();
223 LocalSystemTransferMatrix& loc_rest_wrapped = loc_trans.get_mat_rest();
224
225 // get the unwrapped types
226 auto& loc_prol = loc_prol_wrapped.unwrap();
227 auto& loc_rest = loc_rest_wrapped.unwrap();
228
229 // assemble structure?
230 if (loc_prol.empty())
231 {
232 Assembly::SymbolicAssembler::assemble_matrix_2lvl(loc_prol, space_f, space_c);
233 }
234
235 // create local blocked weight vector
236 LocalSystemVector loc_vec_weight = loc_prol_wrapped.create_vector_l();
237
238 // create local weight vector
239 auto loc_scal_vec_weight = loc_prol.create_vector_l();
240
241 // get the data arrays of the weight vectors
242 auto* v_wb = loc_vec_weight.elements();
243 auto* v_ws = loc_scal_vec_weight.elements();
244
245 // assemble prolongation matrix
246 loc_prol.format();
247 loc_scal_vec_weight.format();
248
249 // assemble prolongation matrix
250 Assembly::GridTransfer::assemble_prolongation(loc_prol, loc_scal_vec_weight, space_f, space_c, cubature);
251
252 // copy weights from scalar to blocked
253 for(Index i(0); i < loc_prol.rows(); ++i)
254 v_wb[i] = v_ws[i];
255
256 // synchronize blocked weight vector
257 this->gate_sys.sync_0(loc_vec_weight);
258
259 // copy weights from blocked to scalar
260 for(Index i(0); i < loc_prol.rows(); ++i)
261 v_ws[i] = v_wb[i][0];
262
263 // invert components
264 loc_scal_vec_weight.component_invert(loc_scal_vec_weight);
265
266 // scale prolongation matrix
267 loc_prol.scale_rows(loc_prol, loc_scal_vec_weight);
268
269 // copy and transpose
270 loc_rest = loc_prol.transpose();
271
272 // compile global transfer
273 transfer_sys.compile();
274 }

276 template<typename DomainLevel_, typename Cubature_>
277 void assemble_truncation(
278 const Domain::VirtualLevel<DomainLevel_>& virt_lvl_fine,
279 const Domain::VirtualLevel<DomainLevel_>& virt_lvl_coarse,
280 const Cubature_& cubature,
281 const BlockedBasicSystemLevel* sys_lvl_coarse = nullptr)
282 {
283 // if the coarse level is a parent, then we need the coarse system level
284 XASSERT((sys_lvl_coarse != nullptr) || !virt_lvl_coarse.is_parent());
285
286 // get fine and coarse domain levels
287 const DomainLevel_& level_f = *virt_lvl_fine;
288 const DomainLevel_& level_c = virt_lvl_coarse.is_child() ? virt_lvl_coarse.level_c() : *virt_lvl_coarse;
289
290 const auto& space_f = level_f.space;
291 const auto& space_c = level_c.space;
292
293 // get local transfer operator
294 LocalSystemTransfer& loc_trans = this->transfer_sys.local();
295
296 // get local transfer matrices
297 const LocalSystemTransferMatrix& loc_rest_wrapped = loc_trans.get_mat_rest();
298 LocalSystemTransferMatrix& loc_trunc_wrapped = loc_trans.get_mat_trunc();
299
300 const auto& loc_rest = loc_rest_wrapped.unwrap();
301 auto& loc_trunc = loc_trunc_wrapped.unwrap();
302
303 // restriction matrix must be already assembled
304 XASSERTM(loc_rest.size() > Index(0), "you need to call 'assemble_transfer' first");
305
306 // clone sparsity pattern of restriction matrix
307 loc_trunc = loc_rest.clone(LAFEM::CloneMode::Layout);
308
309 // create local weight vector
310 LocalSystemVector loc_vec_weight = loc_trunc_wrapped.create_vector_l();
311
312 // create a scalar weight vector for the assembly
313 auto loc_scal_vec_weight = loc_trunc.create_vector_l();
314
315 // get the data arrays of the weight vectors
316 auto* v_wb = loc_vec_weight.elements();
317 auto* v_ws = loc_scal_vec_weight.elements();
318
319 // format
320 loc_trunc.format();
321 loc_scal_vec_weight.format();
322
323 // assemble truncation matrix
324 Assembly::GridTransfer::assemble_truncation(loc_trunc, loc_scal_vec_weight, space_f, space_c, cubature);
325
326 // copy weights from scalar to blocked
327 for(Index i(0); i < loc_trunc.rows(); ++i)
328 v_wb[i] = v_ws[i];
329
330 // We now need to synchronize the weight vector in analogy to the prolongation matrix assembly.
331 // Note that the weight vector is now a coarse-level vector, so we need to synchronize over
332 // the coarse-level gate. This may be a bit more complicated if the coarse level is a ghost
333 // level, as in this case we have to join/split around the synch operation.
334
335 // synchronize weight vector using the muxer/gate
336 if(!virt_lvl_coarse.is_child())
337 {
338 // The coarse level is a simple (non-child) level that exists on all processes,
339 // so simply synch over the coarse-level gate:
340 sys_lvl_coarse->gate_sys.sync_0(loc_vec_weight);
341 }
342 else if(virt_lvl_coarse.is_parent())
343 {
344 // The coarse level is a child level and this is one of the parent processes which contain
345 // the coarse-level gate. So we first need to join the weight vector onto the parent processes
346 // first, then synch that joined vector over the parent gate and finally split the result
347 // to all child processes -- this "emulates" a synch over the (non-existent) coarse-level
348 // child gate, which is what we actually require here...
349
350 // create temporary vector on parent partitioning
351 LocalSystemVector loc_tmp = sys_lvl_coarse->gate_sys._freqs.clone(LAFEM::CloneMode::Allocate);
352
353 // join child weights over muxer
354 this->coarse_muxer_sys.join(loc_vec_weight, loc_tmp);
355
356 // sync over coarse gate
357 sys_lvl_coarse->gate_sys.sync_0(loc_tmp);
358
359 // split over muxer
360 this->coarse_muxer_sys.split(loc_vec_weight, loc_tmp);
361 }
362 else // ghost
363 {
364 // The coarse level is a ghost level, i.e. a child but not a parent. In this case, we
365 // only have to participate in the join/send operations of the muxer, which are part
366 // of the operations executed on the parents handled by the else-if case above.
367
368 this->coarse_muxer_sys.join_send(loc_vec_weight);
369
370 // parent performs sync over its gate here (see above else-if)
371
372 this->coarse_muxer_sys.split_recv(loc_vec_weight);
373 }
374
375 // copy weights from blocked to scalar
376 for(Index i(0); i < loc_trunc.rows(); ++i)
377 v_ws[i] = v_wb[i][0];
378
379 // invert components
380 loc_scal_vec_weight.component_invert(loc_scal_vec_weight);
381
382 // scale reduction matrix
383 loc_trunc.scale_rows(loc_trunc, loc_scal_vec_weight);
384 }
385 }; // class BlockedBasicSystemLevel<...>

447 template<typename DomainLevel_, typename Space_>
448 void assemble_homogeneous_unit_filter(const DomainLevel_& dom_level, const Space_& space)
449 {
450 auto& loc_filter = this->filter_sys.local();
451
452 // create unit-filter assembler
453 Assembly::UnitFilterAssembler<typename DomainLevel_::MeshType> unit_asm;
454
455 std::deque<String> part_names = dom_level.get_mesh_node()->get_mesh_part_names(true);
456 for(const auto& name : part_names)
457 {
458 auto* mesh_part_node = dom_level.get_mesh_node()->find_mesh_part_node(name);
459 XASSERT(mesh_part_node != nullptr);
460
461 // let's see if we have that mesh part
462 // if it is nullptr, then our patch is not adjacent to that boundary part
463 auto* mesh_part = mesh_part_node->get_mesh();
464 if (mesh_part != nullptr)
465 {
466 // add to boundary assembler
467 unit_asm.add_mesh_part(*mesh_part);
468 }
469 }
470
471 // finally, assemble the filter
472 unit_asm.assemble(loc_filter, space);
473 }
