226 virtual Status apply(VectorType& vec_cor, const VectorType& vec_def) override
227 {
228 Statistics::add_solver_expression(std::make_shared<ExpressionStartSolve>(this->name()));
229 // fetch the references
230 VectorTypeV& tmp_v = this->_vec_tmp_v;
231 VectorTypeP& tmp_p = this->_vec_tmp_p;
232 VectorTypeV& sol_v = vec_cor.template at<0>();
233 VectorTypeP& sol_p = vec_cor.template at<1>();
234 const VectorTypeV& rhs_v = vec_def.template at<0>();
235 const VectorTypeP& rhs_p = vec_def.template at<1>();
236
237 // now let's check the preconditioner type
238 switch(_uzawa_type)
239 {
240 case UzawaType::diagonal:
241 // solve A*u_v = f_v
242 Statistics::add_solver_expression(std::make_shared<ExpressionCallUzawaA>(this->name(), this->_solver_a->name()));
243 if(!status_success(_solver_a->apply(sol_v, rhs_v)))
244 {
245 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, 0));
246 return Status::aborted;
247 }
248
249 // solve S*u_p = f_p
250 Statistics::add_solver_expression(std::make_shared<ExpressionCallUzawaS>(this->name(), this->_solver_s->name()));
251 if(!status_success(_solver_s->apply(sol_p, rhs_p)))
252 {
253 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, 0));
254 return Status::aborted;
255 }
256
257 // okay
258 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::success, 0));
259 return Status::success;
260
261 case UzawaType::lower:
262 // solve A*u_v = f_v
263 Statistics::add_solver_expression(std::make_shared<ExpressionCallUzawaA>(this->name(), this->_solver_a->name()));
264 if(!status_success(_solver_a->apply(sol_v, rhs_v)))
265 {
266 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, 0));
267 return Status::aborted;
268 }
269
270 // compute g_p := f_p - D*u_v
271 this->_matrix_d.apply(tmp_p, sol_v, rhs_p, -DataType(1));
272
273 // apply pressure defect filter
274 this->_filter_p.filter_def(tmp_p);
275
276 // solve S*u_p = g_p
277 Statistics::add_solver_expression(std::make_shared<ExpressionCallUzawaS>(this->name(), this->_solver_s->name()));
278 if(!status_success(_solver_s->apply(sol_p, tmp_p)))
279 {
280 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, 0));
281 return Status::aborted;
282 }
283
284 // okay
285 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::success, 0));
286 return Status::success;
287
288 case UzawaType::upper:
289 // solve S*u_p = f_p
290 Statistics::add_solver_expression(std::make_shared<ExpressionCallUzawaS>(this->name(), this->_solver_s->name()));
291 if(!status_success(_solver_s->apply(sol_p, rhs_p)))
292 {
293 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, 0));
294 return Status::aborted;
295 }
296
297 // compute g_v := f_v - B*u_p
298 this->_matrix_b.apply(tmp_v, sol_p, rhs_v, -DataType(1));
299
300 // apply velocity defect filter
301 this->_filter_v.filter_def(tmp_v);
302
303 // solve A*u_v = g_v
304 Statistics::add_solver_expression(std::make_shared<ExpressionCallUzawaA>(this->name(), this->_solver_a->name()));
305 if(!status_success(_solver_a->apply(sol_v, tmp_v)))
306 {
307 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, 0));
308 return Status::aborted;
309 }
310
311 // okay
312 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::success, 0));
313 return Status::success;
314
315 case UzawaType::full:
316 // Note: We will use the first component of the solution vector here.
317 // It will be overwritten by the third solution step below.
318 // solve A*u_v = f_v
319 Statistics::add_solver_expression(std::make_shared<ExpressionCallUzawaA>(this->name(), this->_solver_a->name()));
320 if(!status_success(_solver_a->apply(sol_v, rhs_v)))
321 {
322 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, 0));
323 return Status::aborted;
324 }
325
326 // compute g_p := f_p - D*u_v
327 this->_matrix_d.apply(tmp_p, sol_v, rhs_p, -DataType(1));
328
329 // apply pressure defect filter
330 this->_filter_p.filter_def(tmp_p);
331
332 // solve S*u_p = g_p
333 Statistics::add_solver_expression(std::make_shared<ExpressionCallUzawaS>(this->name(), this->_solver_s->name()));
334 if(!status_success(_solver_s->apply(sol_p, tmp_p)))
335 {
336 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, 0));
337 return Status::aborted;
338 }
339
340 // compute g_v := f_v - B*u_p
341 this->_matrix_b.apply(tmp_v, sol_p, rhs_v, -DataType(1));
342
343 // apply velocity defect filter
344 this->_filter_v.filter_def(tmp_v);
345
346 // solve A*u_v = g_v
347 Statistics::add_solver_expression(std::make_shared<ExpressionCallUzawaA>(this->name(), this->_solver_a->name()));
348 if(!status_success(_solver_a->apply(sol_v, tmp_v)))
349 {
350 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, 0));
351 return Status::aborted;
352 }
353
354 // okay
355 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::success, 0));
356 return Status::success;
357 }
358
359 // we should never come out here...
360 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, 0));
361 return Status::aborted;
362 }
