55 template<
56 typename Weight_,
57 typename Coord_,
58 typename Point_>
59 static void fill(Rule<Shape::Simplex<dim_>, Weight_, Coord_, Point_>& rule)
60 {
61 // auxiliary variables
62 Weight_ V = Weight_(1) / Weight_(MetaMath::Factorial<dim_>::value);
63 Weight_ B = Weight_(2 - dim_)/Weight_((dim_ + 1)*(dim_ + 2)) * V;
64 Weight_ C = Weight_(4)/Weight_((dim_ + 1)*(dim_ + 2)) * V;
65
66 // B-points
67 for(int i(0); i <= dim_; ++i)
68 {
69 // set weight
70 rule.get_weight(i) = B;
71
72 // set point coords
73 for(int j(0); j < dim_; ++j)
74 {
75 rule.get_coord(i,j) = (i == j) ? Coord_(1) : Coord_(0);
76 }
77 }
78
79 // counter
80 int count = dim_;
81
82 // C-points
83 for(int i(1); i <= dim_; ++i)
84 {
85 for(int k(0); k < i; ++k)
86 {
87 ++count;
88 rule.get_weight(count) = C;
89
90 // set point coords
91 for(int j(0); j < dim_; ++j)
92 {
93 rule.get_coord(count,j) = ((j == k) || (j == i)) ? Coord_(1)/Coord_(2) : Coord_(0);
94 } // j-loop
95 } // k-loop
96 } // i-loop
97
98 }
99 }; // class LaufferD2Driver<Simplex<...>,...>

118 class LaufferD4Driver<Shape::Simplex<3> > :
119 public DriverBase<Shape::Simplex<3> >

121 public:
123 static constexpr bool variadic = 0;
124 static constexpr int dim = 3;
125 static constexpr int num_points = (MetaMath::Factorial<dim + 4>::value)/(24*MetaMath::Factorial<dim>::value);

139 template<
140 typename Weight_,
141 typename Coord_,
142 typename Point_>
143 static void fill(Rule<Shape::Simplex<3>, Weight_, Coord_, Point_>& rule)
144 {
145 // auxiliary variables
146 Weight_ V = Weight_(1) / Weight_(6);
147 Weight_ B1 =
148 Weight_(-3*dim*dim*dim + 17*dim*dim - 58*dim + 72)/
149 Weight_(3*(dim + 1)*(dim + 2)*(dim + 3)*(dim + 4)) * V;
150 Weight_ B2 =
151 Weight_(16*(dim*dim - 5*dim + 12))/
152 Weight_(3*(dim + 1)*(dim + 2)*(dim + 3)*(dim + 4)) * V;
153 Weight_ B3 =
154 Weight_(4*(dim*dim - 9*dim + 12))/
155 Weight_((dim + 1)*(dim + 2)*(dim + 3)*(dim + 4)) * V;
156 Weight_ B4 =
157 Weight_(64*(4 - dim))/
158 Weight_(2*(dim + 1)*(dim + 2)*(dim + 3)*(dim + 4)) * V;
159 Weight_ B5 =
160 Weight_(256)/
161 Weight_((dim + 1)*(dim + 2)*(dim + 3)*(dim + 4)) * V;
162
163 int count = 0;
164
165 // B1-points
166 for(int i(0); i <= dim; ++i)
167 {
168 // set weight
169 rule.get_weight(count) = B1;
170
171 // set point coords
172 for(int j(0); j < dim; ++j)
173 {
174 rule.get_coord(count,j) = (i == j) ? Coord_(1) : Coord_(0);
175 }
176 ++count;
177 }
178
179 // B2-points
180 for(int i(0); i <= dim; ++i)
181 {
182 for(int j(0); j <= dim; ++j)
183 {
184 if(i != j)
185 {
186 // set weight
187 rule.get_weight(count) = B2;
188
189 // set point coords
190 for(int k(0); k < dim; ++k)
191 {
192 if(k == i)
193 {
194 rule.get_coord(count,k) = Coord_(1)/Coord_(4);
195 }
196 else if (k == j)
197 {
198 rule.get_coord(count,k) = Coord_(3)/Coord_(4);
199 }
200 else
201 {
202 rule.get_coord(count,k) = Coord_(0);
203 }
204 }
205 ++count;
206 }
207 }
208 }
209
210 // B3-points
211 for(int i(1); i <= dim; ++i)
212 {
213 for(int j(0); j < i; ++j)
214 {
215 rule.get_weight(count) = B3;
216
217 // set point coords
218 for(int k(0); k < dim; ++k)
219 {
220 rule.get_coord(count,k) = ((k == j) || (k == i)) ? Coord_(1)/Coord_(2) : Coord_(0);
221 } // k-loop
222 ++count;
223 } // j-loop
224 } // i-loop
225
226 // B4-points
227 for(int i(0); i <= dim; ++i)
228 {
229 for(int j(0); j <= dim; ++j)
230 {
231 for(int k(0); k < j; ++k)
232 {
233 if(i != j && i != k && j != k)
234 {
235 rule.get_weight(count) = B4;
236
237 // set point coords
238 for(int l(0); l < dim; ++l)
239 {
240 if(l == j || l == k)
241 {
242 rule.get_coord(count,l) = Coord_(1)/Coord_(4);
243 }
244 else if(l == i)
245 {
246 rule.get_coord(count,l) = Coord_(1)/Coord_(2);
247 }
248 else
249 {
250 rule.get_coord(count,l) = Coord_(0);
251 }
252 } // l-loop
253 ++count;
254 } // if
255 } // k-loop
256 } // j-loop
257 } //i-loop
258
259 // B5-points
260 for(int i(1); i <= dim; ++i)
261 {
262 for(int j(0); j < i; ++j)
263 {
264 for(int k(0); k < j; ++k)
265 {
266 for(int l(0); l < k; ++l)
267 {
268 rule.get_weight(count) = B5;
269
270 // set point coords
271 for(int m(0); m < dim; ++m)
272 {
273 if(m == i || m == j || m == k || m == l)
274 {
275 rule.get_coord(count,m) = Coord_(1)/Coord_(4);
276 }
277 else
278 {
279 rule.get_coord(count,m) = Coord_(0);
280 }
281 } // m-loop
282 ++count;
283 } // l-loop
284 } // k-loop
285 } // j-loop
286 } // i-loop
287 }
