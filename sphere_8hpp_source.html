89 virtual void transform(const WorldPoint& origin, const WorldPoint& angles, const WorldPoint& offset) override
90 {
91 // create rotation matrix
92 Tiny::Matrix<CoordType, 3, 3> rot;
93 rot.set_rotation_3d(angles(0), angles(1), angles(2));
94
95 // transform midpoint
96 WorldPoint tmp;
97 tmp = _midpoint - origin;
98 _midpoint.set_mat_vec_mult(rot, tmp) += offset;
99 }

108 void project_point(WorldPoint& point) const
109 {
110 WorldPoint grad_dist(point - _midpoint);
111 CoordType distance(grad_dist.norm_euclid());
112
113 if(distance < Math::eps<CoordType>())
114 {
115 grad_dist(0) = _radius;
116 point += grad_dist;
117 }
118 else
119 {
120 point = _midpoint + (_radius / distance)*grad_dist;
121 }
122 }

134 void project_meshpart(Mesh_& mesh, const MeshPart<Mesh_>& meshpart) const
135 {
136 auto& vtx = mesh.get_vertex_set();
137 const auto& target_vtx = meshpart.template get_target_set<0>();
138
139 for(Index i(0); i < meshpart.get_num_entities(0); ++i)
140 {
141 project_point(reinterpret_cast<WorldPoint&>(vtx[target_vtx[i]]));
142 }
143 }

146 CoordType compute_dist(const WorldPoint& point) const
147 {
148 return Math::abs(compute_signed_dist(point));
149 }
150
152 CoordType compute_dist(const WorldPoint& point, WorldPoint& grad_dist) const
153 {
154 WorldPoint projected(point);
155 project_point(projected);
156
157 CoordType my_dist(_radius - (point - _midpoint).norm_euclid());
158
159 grad_dist = (point - projected);
160 grad_dist.normalize();
161
162 return Math::abs(my_dist);
163 }
164
166 CoordType compute_signed_dist(const WorldPoint& point) const
167 {
168 return (point - _midpoint).norm_euclid() - _radius;
169 }
170
172 CoordType compute_signed_dist(const WorldPoint& point, WorldPoint& grad_dist) const
173 {
174 WorldPoint projected(point);
175 project_point(projected);
176
177 CoordType my_dist(_radius - (point - _midpoint).norm_euclid());
178
179 grad_dist = (point - projected);
180 grad_dist.normalize();
181 grad_dist *= Math::signum(my_dist);
182
183 return my_dist;
184 }

218 virtual void create(
219 int iline,
220 const String& sline,
221 const String&,
222 const std::map<String, String>& attrs,
223 bool) override
224 {
225 CoordType radius = CoordType(0);
226 CoordType mid_x = CoordType(0);
227 CoordType mid_y = CoordType(0);
228 CoordType mid_z = CoordType(0);
229
230 // try to parse the radius
231 if(!attrs.find("radius")->second.parse(radius))
232 throw Xml::GrammarError(iline, sline, "Failed to parse sphere radius");
233 if(radius < CoordType(1E-5))
234 throw Xml::GrammarError(iline, sline, "Invalid sphere radius");
235
236 // try to parse midpoint
237 std::deque<String> mids = attrs.find("midpoint")->second.split_by_whitespaces();
238 if(mids.size() != std::size_t(3))
239 throw Xml::GrammarError(iline, sline, "Invalid sphere midpoint string");
240 if(!mids.front().parse(mid_x) || !mids.at(1).parse(mid_y) || !mids.back().parse(mid_z))
241 throw Xml::GrammarError(iline, sline, "Failed to parse sphere midpoint");
242
243 _chart.reset(new ChartType(mid_x, mid_y, mid_z, radius));
244 }
