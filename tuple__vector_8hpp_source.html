187 void restore_from_checkpoint_data(std::vector<char> & data)
188 {
189 std::uint64_t isize = *(std::uint64_t*) data.data(); //get size of checkpointed _first
190 std::vector<char>::iterator start = std::begin(data) + sizeof(std::uint64_t);
191 std::vector<char>::iterator last_of_first = std::begin(data) + sizeof(std::uint64_t) + (int) isize;
192 std::vector<char> buffer_first(start, last_of_first);
193 _first.restore_from_checkpoint_data(buffer_first);
194
195 data.erase(std::begin(data), last_of_first);
196 _rest.restore_from_checkpoint_data(data);
197 }
198
200 std::uint64_t set_checkpoint_data(std::vector<char>& data, SerialConfig& config)
201 {
202 std::size_t old_size = data.size();
203 data.insert(std::end(data), sizeof(std::uint64_t), 0); //add placeholder
204 std::uint64_t ireal_size = _first.set_checkpoint_data(data, config); //add data of _first to the overall checkpoint and save its size
205 char* csize = reinterpret_cast<char*>(&ireal_size);
206 for(std::size_t i(0) ; i < sizeof(std::uint64_t) ; ++i) //overwrite the guessed datalength
207 {
208 data[old_size +i] = csize[i];
209 }
210
211 return sizeof(std::uint64_t) + ireal_size + _rest.set_checkpoint_data(data, config); //generate and add checkpoint data for the _rest
212 }
