235 template<typename ParentMesh_>
236 class GlobalMaskedBoundaryFactory :
237 public Factory<MeshPart<ParentMesh_>>
238 {
239 public:
241 typedef Factory<MeshPart<ParentMesh_>> BaseClass;
243 typedef ParentMesh_ ParentMeshType;
245 typedef MeshPart<ParentMeshType> MeshType;
247 typedef typename MeshType::ShapeType ShapeType;
249 typedef typename MeshType::TargetSetHolderType TargetSetHolderType;
251 static constexpr int shape_dim = ShapeType::dimension;
253 static constexpr int facet_dim = shape_dim - 1;
254
255 private:
257 const ParentMeshType& _mesh_in;
259 Intern::BoundaryFaceComputer<ShapeType> _face_computer;
261 std::vector<int> _facet_mask;
263 std::array<std::vector<int>, std::size_t(shape_dim)> _bnd_masks;
265 std::array<std::vector<Index>, std::size_t(shape_dim)> _face_idx;
267 std::vector<std::pair<int, const MeshType*>> _halos;
268
269 public:
276 explicit GlobalMaskedBoundaryFactory(const ParentMeshType& mesh_in) :
277 _mesh_in(mesh_in),
278 _facet_mask(mesh_in.get_num_entities(facet_dim), 0)
279 {
280 }
281
282 // Adds a single facet to the mask.
283 void add_mask_facet(Index facet_idx)
284 {
285 _facet_mask.at(facet_idx) = 1;
286 }
287
294 void add_mask_meshpart(const MeshPart<ParentMeshType>& part)
295 {
296 const TargetSet& trg_set = part.template get_target_set<facet_dim>();
297 for(Index i(0); i < trg_set.get_num_entities(); ++i)
298 _facet_mask.at(trg_set[i]) = 1;
299 }
300
310 void add_halo(int rank, const MeshPart<ParentMeshType>& halo)
311 {
312 // add the halo as a mask meshpart
313 add_mask_meshpart(halo);
314 // and save the halo
315 //_halos.emplace(rank, &halo);
316 _halos.push_back(std::make_pair(rank, &halo));
317 }
318
328 void compile(const Dist::Comm& comm)
329 {
330 _face_computer.compute_masks(_mesh_in.get_index_set_holder(), _bnd_masks, _face_idx, _facet_mask);
331 _sync_bnd_masks(comm);
332 _face_computer.compute_faces(_bnd_masks, _face_idx);
333 }
334
336 virtual Index get_num_entities(int dim) override
337 {
338 if(dim < shape_dim)
339 return Index(_face_idx.at(std::size_t(dim)).size());
340 else
341 return Index(0);
342 }
343
345 virtual void fill_attribute_sets(typename BaseClass::AttributeSetContainer&) override
346 {
347 }
348
350 virtual void fill_index_sets(std::unique_ptr<typename BaseClass::IndexSetHolderType>&) override
351 {
352 }
353
355 virtual void fill_target_sets(TargetSetHolderType& target_set_holder) override
356 {
357 _face_computer.fill_target_sets(target_set_holder, _face_idx);
358 }
359
360 protected:
364 void _sync_bnd_masks(const Dist::Comm& comm)
365 {
366 if(_halos.empty())
367 return;
368
369 // allocate halo buffers
370 std::vector<std::vector<int>> halo_send_bufs(_halos.size()), halo_recv_bufs(_halos.size());
371
372 // create send buffers and allocate receive buffers
373 for(std::size_t i(0); i < _halos.size(); ++i)
374 {
375 // count total number of entries in halo
376 Index count = 0u;
377 for(int dim = 0; dim < shape_dim; ++dim) // we don't need to sync dim=shape_dim here
378 count += _halos[i].second->get_num_entities(dim);
379
380 halo_recv_bufs[i].resize(count);
381 halo_send_bufs[i].resize(count); // build_halo_buffer uses push_back
382 _face_computer.build_halo_buffer(halo_send_bufs[i], _bnd_masks, _halos[i].second->get_target_set_holder());
383 }
384
385 // post receive requests
386 Dist::RequestVector recv_reqs(_halos.size());
387 for(std::size_t i(0); i < _halos.size(); ++i)
388 recv_reqs.push_back(comm.irecv(halo_recv_bufs[i].data(), halo_recv_bufs[i].size(), _halos[i].first));
389
390 // post send requests
391 Dist::RequestVector send_reqs(_halos.size());
392 for(std::size_t i(0); i < _halos.size(); ++i)
393 send_reqs.push_back(comm.isend(halo_send_bufs[i].data(), halo_send_bufs[i].size(), _halos[i].first));
394
395 // wait for all receives to finish
396 recv_reqs.wait_all();
397
398 // process receives halo buffers
399 for(std::size_t i(0); i < _halos.size(); ++i)
400 {
401 _face_computer.mask_halo_buffer(_bnd_masks, halo_recv_bufs[i], _halos[i].second->get_target_set_holder());
402 }
403
404 // wait for all sends to finish
405 send_reqs.wait_all();
406 }
