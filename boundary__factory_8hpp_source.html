257 private:

265 std::array<std::vector<int>, std::size_t(shape_dim)> _bnd_masks;
267 std::array<std::vector<Index>, std::size_t(shape_dim)> _face_idx;
269 std::vector<std::pair<int, const MeshType*>> _halos;
270
271 public:
278 explicit GlobalMaskedBoundaryFactory(const ParentMeshType& mesh_in) :
279 _mesh_in(mesh_in),
280 _facet_mask(mesh_in.get_num_entities(facet_dim), 0)
281 {
282 }
283
284 // Adds a single facet to the mask.
285 void add_mask_facet(Index facet_idx)
286 {
287 _facet_mask.at(facet_idx) = 1;
288 }
289
296 void add_mask_meshpart(const MeshPart<ParentMeshType>& part)
297 {
298 const TargetSet& trg_set = part.template get_target_set<facet_dim>();
299 for(Index i(0); i < trg_set.get_num_entities(); ++i)
300 _facet_mask.at(trg_set[i]) = 1;
301 }
302
312 void add_halo(int rank, const MeshPart<ParentMeshType>& halo)
313 {
314 // add the halo as a mask meshpart
315 add_mask_meshpart(halo);
316 // and save the halo
317 //_halos.emplace(rank, &halo);
318 _halos.push_back(std::make_pair(rank, &halo));
319 }
320
330 void compile(const Dist::Comm& comm)
331 {
332 _face_computer.compute_masks(_mesh_in.get_index_set_holder(), _bnd_masks, _face_idx, _facet_mask);
333 _sync_bnd_masks(comm);
334 _face_computer.compute_faces(_bnd_masks, _face_idx);
335 }
336
338 virtual Index get_num_entities(int dim) override
339 {
340 if(dim < shape_dim)
341 return Index(_face_idx.at(std::size_t(dim)).size());
342 else
343 return Index(0);
344 }
345
347 virtual void fill_attribute_sets(typename BaseClass::AttributeSetContainer&) override
348 {
349 }
350
352 virtual void fill_index_sets(std::unique_ptr<typename BaseClass::IndexSetHolderType>&) override
353 {
354 }
355
357 virtual void fill_target_sets(TargetSetHolderType& target_set_holder) override
358 {
359 _face_computer.fill_target_sets(target_set_holder, _face_idx);
360 }
361
362 protected:
366 void _sync_bnd_masks(const Dist::Comm& comm)
367 {
368 if(_halos.empty())
369 return;
370
371 // allocate halo buffers
372 std::vector<std::vector<int>> halo_send_bufs(_halos.size()), halo_recv_bufs(_halos.size());
373
374 // create send buffers and allocate receive buffers
375 for(std::size_t i(0); i < _halos.size(); ++i)
376 {
377 // count total number of entries in halo
378 Index count = 0u;
379 for(int dim = 0; dim < shape_dim; ++dim) // we don't need to sync dim=shape_dim here
380 count += _halos[i].second->get_num_entities(dim);
381
382 halo_recv_bufs[i].resize(count);
383 halo_send_bufs[i].resize(count); // build_halo_buffer uses push_back
384 _face_computer.build_halo_buffer(halo_send_bufs[i], _bnd_masks, _halos[i].second->get_target_set_holder());
385 }
386
387 // post receive requests
388 Dist::RequestVector recv_reqs(_halos.size());
389 for(std::size_t i(0); i < _halos.size(); ++i)
390 recv_reqs.push_back(comm.irecv(halo_recv_bufs[i].data(), halo_recv_bufs[i].size(), _halos[i].first));
391
392 // post send requests
393 Dist::RequestVector send_reqs(_halos.size());
394 for(std::size_t i(0); i < _halos.size(); ++i)
395 send_reqs.push_back(comm.isend(halo_send_bufs[i].data(), halo_send_bufs[i].size(), _halos[i].first));
396
397 // wait for all receives to finish
398 recv_reqs.wait_all();
399
400 // process receives halo buffers
401 for(std::size_t i(0); i < _halos.size(); ++i)
402 {
403 _face_computer.mask_halo_buffer(_bnd_masks, halo_recv_bufs[i], _halos[i].second->get_target_set_holder());
404 }
405
406 // wait for all sends to finish
407 send_reqs.wait_all();
408 }
