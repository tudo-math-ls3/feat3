237 class GlobalMaskedBoundaryFactory :
238 public Factory<MeshPart<ParentMesh_>>
239 {
240 public:
242 typedef Factory<MeshPart<ParentMesh_>> BaseClass;
244 typedef ParentMesh_ ParentMeshType;
246 typedef MeshPart<ParentMeshType> MeshType;
248 typedef typename MeshType::ShapeType ShapeType;
250 typedef typename MeshType::TargetSetHolderType TargetSetHolderType;
252 static constexpr int shape_dim = ShapeType::dimension;
254 static constexpr int facet_dim = shape_dim - 1;
255
256 private:
258 const ParentMeshType& _mesh_in;
260 Intern::BoundaryFaceComputer<ShapeType> _face_computer;
262 std::vector<int> _facet_mask;
264 std::array<std::vector<int>, std::size_t(shape_dim)> _bnd_masks;
266 std::array<std::vector<Index>, std::size_t(shape_dim)> _face_idx;
268 std::vector<std::pair<int, const MeshType*>> _halos;
269
270 public:
277 explicit GlobalMaskedBoundaryFactory(const ParentMeshType& mesh_in) :
278 _mesh_in(mesh_in),
279 _facet_mask(mesh_in.get_num_entities(facet_dim), 0)
280 {
281 }
282
283 // Adds a single facet to the mask.
284 void add_mask_facet(Index facet_idx)
285 {
286 _facet_mask.at(facet_idx) = 1;
287 }
288
295 void add_mask_meshpart(const MeshPart<ParentMeshType>& part)
296 {
297 const TargetSet& trg_set = part.template get_target_set<facet_dim>();
298 for(Index i(0); i < trg_set.get_num_entities(); ++i)
299 _facet_mask.at(trg_set[i]) = 1;
300 }
301
311 void add_halo(int rank, const MeshPart<ParentMeshType>& halo)
312 {
313 // add the halo as a mask meshpart
314 add_mask_meshpart(halo);
315 // and save the halo
316 //_halos.emplace(rank, &halo);
317 _halos.push_back(std::make_pair(rank, &halo));
318 }
319
329 void compile(const Dist::Comm& comm)
330 {
331 _face_computer.compute_masks(_mesh_in.get_index_set_holder(), _bnd_masks, _face_idx, _facet_mask);
332 _sync_bnd_masks(comm);
333 _face_computer.compute_faces(_bnd_masks, _face_idx);
334 }
335
337 virtual Index get_num_entities(int dim) override
338 {
339 if(dim < shape_dim)
340 return Index(_face_idx.at(std::size_t(dim)).size());
341 else
342 return Index(0);
343 }
344
346 virtual void fill_attribute_sets(typename BaseClass::AttributeSetContainer&) override
347 {
348 }
349
351 virtual void fill_index_sets(std::unique_ptr<typename BaseClass::IndexSetHolderType>&) override
352 {
353 }
354
356 virtual void fill_target_sets(TargetSetHolderType& target_set_holder) override
357 {
358 _face_computer.fill_target_sets(target_set_holder, _face_idx);
359 }
360
361 protected:
365 void _sync_bnd_masks(const Dist::Comm& comm)
366 {
367 if(_halos.empty())
368 return;
369
370 // allocate halo buffers
371 std::vector<std::vector<int>> halo_send_bufs(_halos.size()), halo_recv_bufs(_halos.size());
372
373 // create send buffers and allocate receive buffers
374 for(std::size_t i(0); i < _halos.size(); ++i)
375 {
376 // count total number of entries in halo
377 Index count = 0u;
378 for(int dim = 0; dim < shape_dim; ++dim) // we don't need to sync dim=shape_dim here
379 count += _halos[i].second->get_num_entities(dim);
380
381 halo_recv_bufs[i].resize(count);
382 halo_send_bufs[i].resize(count); // build_halo_buffer uses push_back
383 _face_computer.build_halo_buffer(halo_send_bufs[i], _bnd_masks, _halos[i].second->get_target_set_holder());
384 }
385
386 // post receive requests
387 Dist::RequestVector recv_reqs(_halos.size());
388 for(std::size_t i(0); i < _halos.size(); ++i)
389 recv_reqs.push_back(comm.irecv(halo_recv_bufs[i].data(), halo_recv_bufs[i].size(), _halos[i].first));
390
391 // post send requests
392 Dist::RequestVector send_reqs(_halos.size());
393 for(std::size_t i(0); i < _halos.size(); ++i)
394 send_reqs.push_back(comm.isend(halo_send_bufs[i].data(), halo_send_bufs[i].size(), _halos[i].first));
395
396 // wait for all receives to finish
397 recv_reqs.wait_all();
398
399 // process receives halo buffers
400 for(std::size_t i(0); i < _halos.size(); ++i)
401 {
402 _face_computer.mask_halo_buffer(_bnd_masks, halo_recv_bufs[i], _halos[i].second->get_target_set_holder());
403 }
404
405 // wait for all sends to finish
406 send_reqs.wait_all();
407 }
