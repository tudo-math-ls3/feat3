143 void add_mask_meshpart(const MeshPart<ParentMeshType>& part)
144 {
145 const TargetSet& trg_set = part.template get_target_set<facet_dim>();
146 for(Index i(0); i < trg_set.get_num_entities(); ++i)
147 _facet_mask.at(trg_set[i]) = 1;
148 }
