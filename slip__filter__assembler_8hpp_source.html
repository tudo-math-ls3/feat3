108 static void compute_orientations(CoordType* orientation, const Index* facets, const MeshType& mesh)
109 {
110 // Various index sets for the parent
111 auto& idx_vert_at_facet(mesh.template get_index_set<facet_dim, 0>());
112 auto& idx_vert_at_shape(mesh.template get_index_set<ShapeType::dimension, 0>());
113 auto& idx_facet_at_shape(mesh.template get_index_set<ShapeType::dimension, facet_dim>());
114
115 // This will contain the local vertex indices in the reference cell numbering
116 typename MeshType::template IndexSet<facet_dim, 0>::Type::IndexTupleType reference_numbering;
117 // This will contain the local vertex indices as they are numbered in the mesh
118 typename MeshType::template IndexSet<facet_dim, 0>::Type::IndexTupleType my_numbering;
119
120 // Go check all cells in the mesh
121 for(Index k(0); k < mesh.get_num_entities(ShapeType::dimension); ++k)
122 {
123 // Check all facets
124 for(int l(0); l < Shape::FaceTraits<ShapeType, facet_dim>::count; ++l)
125 {
126 // Index of the l-th facet in the parent mesh
127 Index parent_l(idx_facet_at_shape[k][l]);
128
129 // If the facet is in the meshpart ...
130 if(facets[parent_l] != ~Index(0))
131 {
132 // ... check the numbering
133 for(int i(0); i < nvt_loc; ++i)
134 {
135 // The FaceIndexMapping gives us the reference cell numbering of the facet
136 int j(Geometry::Intern::FaceIndexMapping<ShapeType, facet_dim, 0>::map(l,i));
137 reference_numbering[i] = idx_vert_at_shape[k][j];
138
139 // This is how the facet is numbered in the mesh
140 my_numbering[i] = idx_vert_at_facet[parent_l][i];
141 }
142
143 // Get the orientation code. This describes if the facet given by my_numbering is a rotation of the
144 // reference facet or a rotation of an inverted reference facet
145 int orientation_code(
146 Geometry::Intern::CongruencySampler<FacetType>::compare(reference_numbering, my_numbering));
147
148 // The orientation of the facet is the (possibly inverted) orientation of the appropriate facet in
149 // the reference cell
150 orientation[parent_l] = CoordType(
151 Geometry::Intern::CongruencySampler<FacetType>::orientation(orientation_code)
152 *Shape::ReferenceCell<ShapeType>::facet_orientation(l));
153 }
154 } // facets
155 } // cells
156 } // compute_orientations

187 template<typename VectorType_, typename TrafoType_>
188 static void compute_outer_unit_normal(
189 VectorType_& nu,
190 const Index* facets,
191 const CoordType* orientation,
192 const TrafoType_& trafo)
193 {
194 typedef typename VectorType_::DataType DataType;
195
196 nu.format(DataType(0));
197 // Vertex at facet index set from the parent
198 auto& idx(trafo.get_mesh().template get_index_set<facet_dim,0>());
199 // Temporary vector for holding one normal at a time
200 Tiny::Vector<DataType, world_dim> tmp(DataType(0));
201
202 // create a facet trafo evaluator
203 typename TrafoType_::template Evaluator<FacetType, CoordType>::Type facet_evaluator(trafo);
204
205 // For every cell in the meshpart, compute the outer normal vectors in all local Lagrange points of the
206 // element's transformation
207 for(Index k(0); k < trafo.get_mesh().get_num_entities(facet_dim); ++k)
208 {
209 if(facets[k] != ~Index(0) )
210 {
211 facet_evaluator.prepare(k);
212 CoordType vol = facet_evaluator.volume();
213 facet_evaluator.finish();
214 //CoordType vol(trafo.template compute_vol<FacetType>(k));

216 // Compute all normals
217 Tiny::Matrix<DataType, nvt_loc, world_dim> nu_loc(compute_oriented_normals(trafo, k));
218
219 // Add the local contributions to the vector that is numbered according to the parent
220 for(int l(0); l < nvt_loc; ++l)
221 {
222 tmp = nu(idx[k][l]) + nu_loc[l]*DataType(orientation[k]*vol);
223 nu(idx[k][l], tmp);
224 }
225 }

249 template<typename TrafoType_>
250 static Tiny::Matrix<CoordType, nvt_loc, MeshType::world_dim> compute_oriented_normals(const TrafoType_& trafo, Index k)
251 {
252 // Evaluator on Faces
253 typedef typename TrafoType_::template Evaluator<FacetType, CoordType>::Type FaceEvaluator;
254 // Type of the Jacobian it returns
255 typedef typename FaceEvaluator::JacobianMatrixType JacobianMatrixType;
256
257 // Evaluator for the trafo
258 FaceEvaluator trafo_eval(trafo);
259 trafo_eval.prepare(k);
260
261 // This will be the coordinates of the Lagrange points on the reference cell the face is parametrized over
262 Tiny::Matrix<CoordType, nvt_loc, ShapeType::dimension - 1> xloc(CoordType(0));
263 for(int i(0); i < nvt_loc; ++i)
264 {
265 for(int d(0); d < ShapeType::dimension - 1 ; ++d)
266 xloc(i,d) = Shape::ReferenceCell<FacetType>::template vertex<CoordType>(i,d);
267 }
268
269 // The Jacobian matrix of the transformation
270 JacobianMatrixType jac_mat;
271 // This will hold the local tangential surface coordinate system at one Lagrange point of the local face
272 Tiny::Matrix<CoordType, MeshType::world_dim-1, MeshType::world_dim> tau(CoordType(0));
273
274 // This will hold the oriented unit normal vector for all Lagrange points of the face
275 Tiny::Matrix<CoordType, nvt_loc, MeshType::world_dim> nu(CoordType(0));
276 for(int i(0); i < nvt_loc; ++i)
277 {
278 tau.format();
279
280 // Compute the tangential vectors at the Lagrange points: They are the gradients wrt. the parametrization
281 // of the transformation evaluated in the Lagrange points
282 trafo_eval.calc_jac_mat(jac_mat, xloc[i]);
283
284 // Now write the cross product of the tangentials into nu[i]
285 nu[i] = Tiny::orthogonal(jac_mat);
286 // Normalize nu
287 nu[i].normalize();
288 }
289
290 return nu;
291
292 } // compute_oriented_normals

384 void add_mesh_part(const Geometry::MeshPart<MeshType>& meshpart)
385 {
386 // Adding a MeshPart changes the structures
387 this->_recompute = true;
388 const auto& facet_target(meshpart.template get_target_set<facet_dim>());
389
390 for(Index l(0); l < facet_target.get_num_entities(); l++)
391 {
392 // Add the new facet if it is not already there
393 if(_facets[facet_target[l]] == ~Index(0))
394 {
395 _facets[facet_target[l]] = _num_entities[facet_dim];
396 ++_num_entities[facet_dim];
397 }
398 }
399 }

408 void recompute_target_set_holder(const MeshType& mesh)
409 {
410 // We know how many facets we have from add_mesh_part
411 _target_set_holder = TargetSetHolderType(_num_entities);
412
413 auto& facet_targets = _target_set_holder.template get_target_set<facet_dim>();
414
415 // Copy the facet information from the marker array
416 for(Index l(0); l < mesh.get_num_entities(facet_dim); ++l)
417 {
418 if(_facets[l] != ~Index(0))
419 facet_targets[_facets[l]] = l;
420 }
421
422 // Now we deduct the vertices/edges/whatever at the slip boundary from the facet target mapping
423 Geometry::Intern::template TargetSetComputer<facet_dim, 0>::top_to_bottom(
424 _target_set_holder, mesh.get_index_set_holder());
425
426 // Update num_entities information from the possibly modified _target_set_holder
427 for(int d(0); d <= facet_dim; ++d)
428 _num_entities[d] = _target_set_holder.get_num_entities(d);
429 }

449 template<typename DataType_, typename IndexType_>
450 void assemble(LAFEM::SlipFilter<DataType_, IndexType_, world_dim>& filter, const Space::Lagrange1::Element<Trafo_>& space)
451 {
452 // Allocate the filter if necessary
453 if(filter.get_nu().size() == Index(0))
454 {
455 filter = LAFEM::SlipFilter<DataType_, IndexType_, world_dim>
456 (space.get_trafo().get_mesh().get_num_entities(0), space.get_num_dofs());
457 }
458
459 // Compute orientations if necessary
460 if(_recompute)
461 {
462 OuterNormalComputer<Trafo_>::compute_orientations(_orientation.data(), _facets.data(), space.get_trafo().get_mesh());
463 }
464
465 // Compute the weighted outer unit normal field
466 OuterNormalComputer<Trafo_>::compute_outer_unit_normal(
467 filter.get_nu(), _facets.data(), _orientation.data(), space.get_trafo());
468
469 // Generate the filter vector. For the Lagrange 1 with standard trafo case, this is just a clone operation
470 filter.get_filter_vector().clone(filter.get_nu());
471 }

498 template<typename DataType_, typename IndexType_>
499 void assemble(LAFEM::SlipFilter<DataType_, IndexType_, world_dim>& filter, const Space::Lagrange2::Element<Trafo_>& space)
500 {
501 typedef Space::Lagrange2::Element<Trafo_> SpaceType;
502
503 // The mesh the FE space lives on
504 const auto& mesh = space.get_trafo().get_mesh();
505
506 // Allocate the filter if necessary
507 if(filter.get_nu().size() == Index(0))
508 {
509 filter = LAFEM::SlipFilter<DataType_, IndexType_, world_dim>
510 (mesh.get_num_entities(0), space.get_num_dofs());
511 }
512
513 // Compute orientations if necessary
514 if(_recompute)
515 {
516 recompute_target_set_holder(space.get_trafo().get_mesh());
517 OuterNormalComputer<Trafo_>::compute_orientations(_orientation.data(), _facets.data(), mesh);
518 }
519
520 // Compute the weighted outer unit normal field
521 OuterNormalComputer<Trafo_>::compute_outer_unit_normal(
522 filter.get_nu(), _facets.data(), _orientation.data(), space.get_trafo());
523
524 // We only have something to do if the filter is not empty after recompute_target_set_holder()
525 if(filter.get_nu().used_elements() > Index(0))
526 {
527 Intern::Lagrange2InterpolatorWrapper<SpaceType>::project(
528 filter.get_filter_vector(), filter.get_nu(), space, &_target_set_holder);
529
530 // re-normalize
531 auto* nu = filter.get_values();
532 for(Index i(0); i < filter.used_elements(); ++i)
533 nu[i].normalize();
534 }
535 }

541 /*
542 * This is a hackish implementation of a Lagrange 1 to Lagrange 2 interpolator and mostly copied from
543 * interpolator.hpp. The critical thing is that it requires the application of node functionals of one space
544 * to FE functions from another space, so in general this will not be well-defined (like applying Q2 node
545 * functionals (which are point evaluations) to Q1~ functions (which are not continuous). In the case of
546 * conforming Lagrange elements, it is well-defined however, and implemented below.
547 */
548 template<typename Vector_, typename Space_, int shape_dim_>
549 struct Lagrange2InterpolatorCore
550 {
551 public:
552 template< typename TSH_>
553 static void project(Vector_& lagrange_2_vector, const Vector_& lagrange_1_vector, const Space_& space, const TSH_* tsh)
554 {
555 typedef typename Vector_::DataType DataType;
556 typedef typename Vector_::ValueType ValueType;
557
558 // define dof assignment
559 typedef typename Space_::template DofAssignment<shape_dim_, DataType>::Type DofAssignType;
560 DofAssignType dof_assign(space);
561
562 // skip empty dof assignments
563 if(dof_assign.get_max_assigned_dofs() < 1)
564 return;
565
566 // Vertex at shape index set
567 auto& idx(space.get_trafo().get_mesh().template get_index_set<shape_dim_, 0>());
568
569 // loop over all entities
570 if(tsh == nullptr)
571 {
572 const Index num_shapes = space.get_mesh().get_num_entities(shape_dim_);
573 for(Index shape(0); shape < num_shapes; ++shape)
574 {
575 ValueType tmp(DataType(0));
576
577 // evaluate the "node functional"
578 for(int j(0); j < idx.get_num_indices(); ++j)
579 tmp += lagrange_1_vector(idx(shape, j));
580
581 tmp *= DataType(1)/DataType(idx.get_num_indices());
582 tmp.normalize();
583
584 // prepare dof-assignment
585 dof_assign.prepare(shape);
586
587 // loop over all contributions
588 Index dof_index(dof_assign.get_index(0));
589 lagrange_2_vector(dof_index, tmp);
590
591 // finish
592 dof_assign.finish();
593 }
594 }
595 else
596 {
597 auto& ts = tsh->template get_target_set<shape_dim_>();
598
599 // loop over all entities
600 const Index num_shapes = ts.get_num_entities();
601
602 for(Index shape(0); shape < num_shapes; ++shape)
603 {
604 ValueType tmp(DataType(0));
605
606 // evaluate the "node functional"
607 for(int j(0); j < idx.get_num_indices(); ++j)
608 tmp += lagrange_1_vector(idx(ts[shape], j));
609
610 tmp *= DataType(1)/DataType(idx.get_num_indices());
611 tmp.normalize();
612
613 // prepare dof-assignment
614 dof_assign.prepare(ts[shape]);
615
616 // loop over all contributions
617 Index dof_index(dof_assign.get_index(0));
618 lagrange_2_vector(dof_index, tmp);
619
620 // finish
621 dof_assign.finish();
622 }
623 }
624 }
625 };

627 template<typename Vector_, typename Space_>
628 struct Lagrange2InterpolatorCore<Vector_, Space_, 0>
629 {
630 public:
631 template<typename TSH_>
632 static void project(Vector_& lagrange_2_vector, const Vector_& lagrange_1_vector, const Space_& space, const TSH_* tsh)
633 {
634 typedef typename Vector_::DataType DataType;
635 typedef typename Vector_::ValueType ValueType;
636
637 // define dof assignment
638 typedef typename Space_::template DofAssignment<0, DataType>::Type DofAssignType;
639 DofAssignType dof_assign(space);
640
641 if(tsh == nullptr)
642 {
643 // loop over all entities
644 const Index num_verts = space.get_mesh().get_num_entities(0);
645 for(Index vert(0); vert < num_verts; ++vert)
646 {
647 // prepare dof-assignment
648 dof_assign.prepare(vert);
649
650 // "loop over all contributions"
651 Index dof_index(dof_assign.get_index(0));
652
653 // evaluate the "node functional"
654 ValueType tmp(lagrange_1_vector(vert));
655 lagrange_2_vector(dof_index, tmp);
656
657 // finish
658 dof_assign.finish();
659 }
660 }
661 else
662 {
663 auto& ts = tsh->template get_target_set<0>();
664 // loop over all entities
665 const Index num_verts = ts.get_num_entities();
666 for(Index vert(0); vert < num_verts; ++vert)
667 {
668 // prepare dof-assignment
669 dof_assign.prepare(ts[vert]);
670
671 // "loop over all contributions"
672 Index dof_index(dof_assign.get_index(0));
673
674 // evaluate the "node functional"
675 ValueType tmp(lagrange_1_vector(ts[vert]));
676 lagrange_2_vector(dof_index, tmp);
677
678 // finish
679 dof_assign.finish();
680 }
681 }
682 }

685 template<typename Space_, int shape_dim_>
686 class Lagrange2InterpolatorWrapper
687 {
688 public:
689 template<typename Vector_, typename TSH_>
690 static void project(Vector_& lagrange_2_vector, const Vector_& lagrange_1_vector, const Space_& space, const TSH_* tsh)
691 {
692 // recurse down
693 Lagrange2InterpolatorWrapper<Space_, shape_dim_ - 1>::project(lagrange_2_vector, lagrange_1_vector, space, tsh);
694
695 // call interpolator core
696 Lagrange2InterpolatorCore<Vector_, Space_, shape_dim_>::project(lagrange_2_vector, lagrange_1_vector, space, tsh);
697 }
698 };
