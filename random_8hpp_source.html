234 // specialization for integers
235 template<typename T_>
236 class RandomNumber<T_, Type::IntegralClass> :
237 public RandomInteger<T_>
238 {
239 public:
240 static T_ gen_ranged(Random& rng, T_ a, T_ b)
241 {
242 // call gen() prior to the if-clause to advance the rng in any case
243 // also, ensure that the value is non-negative
244 T_ x = NonNegInt<T_>::make(RandomInteger<T_>::gen(rng));
245 if(a < b)
246 // Note: The following casting orgy is necessary on some compilers if the type 'T_' is
247 // actually shorter than 'int', as the arithmetic operations return values of type 'int'
248 // in this case.
249 return T_(a + T_(x % T_(T_(b - a) + T_(1))));
250 else
251 return a;
252 }
253 };
254
255 // specialization for floating point numbers
256 template<typename T_>
257 class RandomNumber<T_, Type::FloatingClass>
258 {
259 public:
260 static T_ gen(Random& rng)
261 {
262 static constexpr std::uint64_t mask = std::uint64_t(0xFFFFFFFFull);
263 const double d = 1.0 / double(mask);
264 return T_(double(rng.next() & mask) * d);
265 }
266
267 static T_ gen_ranged(Random& rng, T_ a, T_ b)
268 {
269 // machine exactness; we do not want to divide by anything smaller than that...
270 static const T_ eps = std::numeric_limits<T_>::epsilon();
271
272 // call gen() prior to the if-clause to advance the rng in any case
273 T_ x(gen(rng));
274 if(a + eps < b)
275 return a + x * (b - a);
276 else
277 return a;
278 }
279 };

291 static T_ gen_ranged(Random& rng, T_ a, T_ b)
292 {
293 // call gen() prior to the if-clause to advance the rng in any case
294 T_ x(gen(rng));
295 // The following case may look pointless for bool, but if a and b are equal
296 // we want to return that value here - otherwise its either true or false.
297 if(a != b)
298 return x;
299 else
300 return a;
301 }
