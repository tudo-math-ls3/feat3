236 // specialization for integers
237 template<typename T_>
238 class RandomNumber<T_, Type::IntegralClass> :
239 public RandomInteger<T_>
240 {
241 public:
242 static T_ gen_ranged(Random& rng, T_ a, T_ b)
243 {
244 // call gen() prior to the if-clause to advance the rng in any case
245 // also, ensure that the value is non-negative
246 T_ x = NonNegInt<T_>::make(RandomInteger<T_>::gen(rng));
247 if(a < b)
248 // Note: The following casting orgy is necessary on some compilers if the type 'T_' is
249 // actually shorter than 'int', as the arithmetic operations return values of type 'int'
250 // in this case.
251 return T_(a + T_(x % T_(T_(b - a) + T_(1))));
252 else
253 return a;
254 }
255 };
256
257 // specialization for floating point numbers
258 template<typename T_>
259 class RandomNumber<T_, Type::FloatingClass>
260 {
261 public:
262 static T_ gen(Random& rng)
263 {
264 static constexpr std::uint64_t mask = std::uint64_t(0xFFFFFFFFull);
265 const double d = 1.0 / double(mask);
266 return T_(double(rng.next() & mask) * d);
267 }
268
269 static T_ gen_ranged(Random& rng, T_ a, T_ b)
270 {
271 // machine exactness; we do not want to divide by anything smaller than that...
272 static const T_ eps = std::numeric_limits<T_>::epsilon();
273
274 // call gen() prior to the if-clause to advance the rng in any case
275 T_ x(gen(rng));
276 if(a + eps < b)
277 return a + x * (b - a);
278 else
279 return a;
280 }
281 };
282
283 // specialization for bool
284 template<typename T_>
285 class RandomNumber<T_, Type::BooleanClass>
286 {
287 public:
288 static T_ gen(Random& rng)
289 {
290 return T_(int(rng.next() & 0x8ull) != 0);
291 }
292
293 static T_ gen_ranged(Random& rng, T_ a, T_ b)
294 {
295 // call gen() prior to the if-clause to advance the rng in any case
296 T_ x(gen(rng));
297 // The following case may look pointless for bool, but if a and b are equal
298 // we want to return that value here - otherwise its either true or false.
299 if(a != b)
300 return x;
301 else
302 return a;
303 }
304 };
