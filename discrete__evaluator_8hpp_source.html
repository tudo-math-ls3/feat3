33 template<typename DT_, int dim_>
34 class ScalarDiscreteEvalData
35 {
36 public:
38 static constexpr int dim = dim_;
40 typedef DT_ DataType;
42 typedef DataType ValueType;
43
45 std::vector<ValueType> values;
46
50 bool empty() const
51 {
52 return values.empty();
53 }
54
65 ValueType mean_value() const
66 {
67 if(values.empty())
68 return DT_(0);
69
70 // compute and return mean value
71 DT_ r = DT_(0);
72 for(const auto& v : values)
73 r += v;
74 return r / DT_(values.size());
75 }
76
86 ValueType mean_value_dist(const Dist::Comm& comm) const
87 {
88 // In a distributed domain, we need to compute the mean value
89 // over all patches, as the evaluation point may intersect
90 // with several patches. Therefore, we also need to sum up
91 // the number of values in addition to the values themselves,
92 // so that we can compute the mean value over all patches.
93 DT_ val[2] =
94 {
95 DT_(values.size()),
96 DT_(0)
97 };
98
99 // add all values from this patch
100 for(const auto& v : values)
101 val[1] += v;
102
103 // sum up over all patches
104 comm.allreduce(val, val, std::size_t(2), Dist::op_sum);
105
106 // compute global mean value
107 return (val[0] > DT_(0)) ? (val[1] / val[0]) : DT_(0);
108 }
109 }; // class ScalarDiscreteEvalData

122 template<typename DT_, int dim_>
123 class VectorDiscreteEvalData
124 {
125 public:
127 static constexpr int dim = dim_;
129 typedef DT_ DataType;
131 typedef Tiny::Vector<DT_, dim_> ValueType;
132
134 std::vector<ValueType> values;
135
139 bool empty() const
140 {
141 return values.empty();
142 }
143
154 ValueType mean_value() const
155 {
156 ValueType r;
157 r.format();
158 if(values.empty())
159 return r;
160
161 for(const auto& v : values)
162 r += v;
163 r *= (DT_(1) / DT_(values.size()));
164
165 return r;
166 }
167
177 ValueType mean_value_dist(const Dist::Comm& comm) const
178 {
179 ValueType r;
180 r.format();
181
182 DT_ val[1+dim_] =
183 {
184 DT_(values.size()),
185 };
186
187 for(int i(0); i < dim_; ++i)
188 val[i+1] = DT_(0);
189
190 for(const auto& v : values)
191 {
192 for(int i(0); i < dim_; ++i)
193 val[i+1] += v[i];
194 }
195
196 comm.allreduce(val, val, std::size_t(dim_+1), Dist::op_sum);
197
198 for(int i(0); i < dim_; ++i)
199 r[i] = val[i+1];
200
201 if(val[0] > DT_(0))
202 r *= (DT_(1) / val[0]);
203
204 return r;
205 }
206 }; // class VectorDiscreteEvalData

219 template<typename DT_, int dim_i_, int dim_j_>
220 class MatrixDiscreteEvalData
221 {
222 public:
224 static constexpr int dim_i = dim_i_;
225 static constexpr int dim_j = dim_j_;
227 typedef DT_ DataType;
229 typedef Tiny::Matrix<DT_, dim_i_, dim_j_> ValueType;
230
232 std::vector<ValueType> values;
233
237 bool empty() const
238 {
239 return values.empty();
240 }
241
252 ValueType mean_value() const
253 {
254 ValueType r;
255 r.format();
256 if(values.empty())
257 return r;
258
259 for(const auto& v : values)
260 r += v;
261 r *= (DT_(1) / DT_(values.size()));
262

264 }
265
275 ValueType mean_value_dist(const Dist::Comm& comm) const
276 {
277 ValueType r;
278 r.format();
279
280 DT_ val[1+dim_i_*dim_j_] =
281 {
282 DT_(values.size()),
283 };
284
285 for(int i(0); i < dim_i_; ++i)
286 {
287 for(int j(0); j < dim_j_; ++j)
288 {
289 val[i*dim_j_ + j + 1] = DT_(0);
290 }
291 }
292
293 for(const auto& v : values)
294 {
295 for(int i(0); i < dim_i_; ++i)
296 {
297 for(int j(0); j < dim_j_; ++j)
298 {
299 val[i*dim_j_ + j + 1] += v[i][j];
300 }
301 }
302 }
303
304 comm.allreduce(val, val, std::size_t(dim_i_*dim_j_+1), Dist::op_sum);
305
306 for(int i(0); i < dim_i_; ++i)
307 {
308 for(int j(0); j < dim_j_; ++j)
309 {
310 r[i][j] = val[i*dim_j_ + j + 1];
311 }
312 }
313
314 if(val[0] > DT_(0))
315 r *= (DT_(1) / val[0]);
316
317 return r;
318 }

412 template<typename DT_, typename DTP_, typename IT_, int dim_, typename Space_>
413 static ScalarDiscreteEvalData<DT_, dim_> eval_fe_function(
414 const Trafo::InverseMappingData<DTP_, dim_, dim_>& inv_map_data,
415 const LAFEM::DenseVector<DT_, IT_>& vector,
416 const Space_& space)
417 {
418 // vector type
419 typedef LAFEM::DenseVector<DT_, IT_> VectorType;
420 // space type
421 typedef Space_ SpaceType;
422 // assembly traits
423 typedef AsmTraits1<DT_, SpaceType, TrafoTags::none, SpaceTags::value> AsmTraits;
424 // data type
425 typedef typename AsmTraits::DataType DataType;
426
427 // fetch the trafo
428 const typename AsmTraits::TrafoType& trafo = space.get_trafo();
429
430 // create a trafo evaluator
431 typename AsmTraits::TrafoEvaluator trafo_eval(trafo);
432
433 // create a space evaluator and evaluation data
434 typename AsmTraits::SpaceEvaluator space_eval(space);
435
436 // create a dof-mapping
437 typename AsmTraits::DofMapping dof_mapping(space);
438
439 // create trafo evaluation data
440 typename AsmTraits::TrafoEvalData trafo_data;
441
442 // create space evaluation data
443 typename AsmTraits::SpaceEvalData space_data;
444
445 // create local vector data
446 typename AsmTraits::template TLocalVector<DT_> local_vector;
447
448 // create matrix scatter-axpy
449 typename VectorType::GatherAxpy gather_axpy(vector);
450
451 // create evaluation data
452 ScalarDiscreteEvalData<DT_, dim_> eval_data;
453
454 // loop over all cells of the mesh
455 for(std::size_t i(0); i < inv_map_data.size(); ++i)
456 {
457 const Index& cell = inv_map_data.cells.at(i);
458 const auto& dom_point = inv_map_data.dom_points.at(i);
459
460 // format local vector
461 local_vector.format();
462
463 // initialize dof-mapping
464 dof_mapping.prepare(cell);
465
466 // gather local vector data
467 gather_axpy(local_vector, dof_mapping);
468
469 // finish dof-mapping
470 dof_mapping.finish();
471
472 // prepare trafo evaluator
473 trafo_eval.prepare(cell);
474
475 // prepare space evaluator
476 space_eval.prepare(trafo_eval);
477
478 // compute trafo data
479 trafo_eval(trafo_data, dom_point);
480
481 // compute basis function data
482 space_eval(space_data, trafo_data);
483
484 // fetch number of local dofs
485 const int num_loc_dofs = space_eval.get_num_local_dofs();
486
487 // compute function value
488 DataType value = DataType(0);
489 for(int j(0); j < num_loc_dofs; ++j)
490 value += local_vector[j] * space_data.phi[j].value;
491
492 // finally, add this result to the eval data
493 eval_data.values.push_back(value);
494
495 // finish evaluators
496 space_eval.finish();
497 trafo_eval.finish();
498
499 // continue with next cell
500 }
501
502 // return evaluation data
503 return eval_data;
504 }
