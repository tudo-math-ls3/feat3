109 template<typename EvalData_>
110 void eval_ref_values(
111 EvalData_& data,
112 const DomainPointType& point) const
113 {
114 const DataType x = point[0];
115 const DataType y = point[1];
116 const DataType x2 = x*x;
117 const DataType y2 = y*y;
118 // edge midpoints dofs
119 data.phi[0].ref_value = DataType(0.25) - DataType(0.5)*y + ( DataType(0.375) - DataType(0.625)*x2)*x2 + (-DataType(0.375) + DataType(0.625)*y2)*y2;
120 data.phi[1].ref_value = DataType(0.25) + DataType(0.5)*y + ( DataType(0.375) - DataType(0.625)*x2)*x2 + (-DataType(0.375) + DataType(0.625)*y2)*y2;
121 data.phi[2].ref_value = DataType(0.25) - DataType(0.5)*x + (-DataType(0.375) + DataType(0.625)*x2)*x2 + ( DataType(0.375) - DataType(0.625)*y2)*y2;
122 data.phi[3].ref_value = DataType(0.25) + DataType(0.5)*x + (-DataType(0.375) + DataType(0.625)*x2)*x2 + ( DataType(0.375) - DataType(0.625)*y2)*y2;
123 // bubble dof
124 data.phi[4].ref_value = DataType(2.25)*x*y;
125 }

136 template<typename EvalData_>
137 void eval_ref_gradients(
138 EvalData_& data,
139 const DomainPointType& point) const
140 {
141 const DataType x = point[0];
142 const DataType y = point[1];
143 const DataType x2 = x*x;
144 const DataType y2 = y*y;
145 // edge midpoint dofs
146 data.phi[0].ref_grad[0] = ( DataType(0.75) - DataType(2.5)*x2)*x;
147 data.phi[0].ref_grad[1] = (-DataType(0.75) + DataType(2.5)*y2)*y - DataType(0.5);
148 data.phi[1].ref_grad[0] = ( DataType(0.75) - DataType(2.5)*x2)*x;
149 data.phi[1].ref_grad[1] = (-DataType(0.75) + DataType(2.5)*y2)*y + DataType(0.5);
150 data.phi[2].ref_grad[0] = (-DataType(0.75) + DataType(2.5)*x2)*x - DataType(0.5);
151 data.phi[2].ref_grad[1] = ( DataType(0.75) - DataType(2.5)*y2)*y;
152 data.phi[3].ref_grad[0] = (-DataType(0.75) + DataType(2.5)*x2)*x + DataType(0.5);
153 data.phi[3].ref_grad[1] = ( DataType(0.75) - DataType(2.5)*y2)*y;
154 // bubble dof
155 data.phi[4].ref_grad[0] = DataType(2.25)*y;
156 data.phi[4].ref_grad[1] = DataType(2.25)*x;
157 }
