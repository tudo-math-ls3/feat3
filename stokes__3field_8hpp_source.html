240 void compile_system_matrix()

267 template<typename DomainLevel_>
268 void assemble_gates(const Domain::VirtualLevel<DomainLevel_>& virt_dom_lvl)
269 {
270 Asm::asm_gate(virt_dom_lvl, virt_dom_lvl->space_velo, this->gate_velo, true);
271 Asm::asm_gate(virt_dom_lvl, virt_dom_lvl->space_pres, this->gate_pres, true);
272 Asm::asm_gate(virt_dom_lvl, virt_dom_lvl->space_stress, this->gate_stress, true);
273 Asm::build_gate_tuple(this->gate_sys, this->gate_velo, this->gate_pres, this->gate_stress);
274 this->gate_scalar_velo.convert(this->gate_velo, LocalScalarVector(virt_dom_lvl->space_velo.get_num_dofs()));
275 this->gate_scalar_stress.convert(this->gate_stress, LocalScalarVector(virt_dom_lvl->space_stress.get_num_dofs()));
276 }

278 template<typename DomainLevel_>
279 void assemble_coarse_muxers(const Domain::VirtualLevel<DomainLevel_>& virt_lvl_coarse)
280 {
281 Asm::asm_muxer(virt_lvl_coarse, [](const DomainLevel_& dl){return &dl.space_velo;}, this->coarse_muxer_velo);
282 Asm::asm_muxer(virt_lvl_coarse, [](const DomainLevel_& dl){return &dl.space_pres;}, this->coarse_muxer_pres);

284 Asm::build_muxer_tuple(this->coarse_muxer_sys, this->gate_sys.get_freqs(), this->coarse_muxer_velo, this->coarse_muxer_pres, this->coarse_muxer_stress);

286 this->coarse_muxer_scalar_stress.convert(this->coarse_muxer_stress, this->gate_scalar_stress.get_freqs().clone(LAFEM::CloneMode::Shallow));

289 template<typename DomainLevel_>
290 void assemble_transfers(
291 const Stokes3FieldSystemLevel& sys_lvl_coarse,
292 const Domain::VirtualLevel<DomainLevel_>& virt_lvl_fine,
293 const Domain::VirtualLevel<DomainLevel_>& virt_lvl_coarse,
294 const String& cubature, bool trunc_v = false, bool trunc_p = false, bool trunc_s = false, bool shrink = true)
295 {
296 Asm::asm_transfer_blocked(virt_lvl_fine, virt_lvl_coarse, cubature, trunc_v, shrink,
297 [](const DomainLevel_& dl) {return &dl.space_velo;},
298 this->transfer_velo.local(), this->coarse_muxer_velo, this->gate_velo, sys_lvl_coarse.gate_velo);
299 Asm::asm_transfer_scalar(virt_lvl_fine, virt_lvl_coarse, cubature, trunc_p, shrink,
300 [](const DomainLevel_& dl) {return &dl.space_pres;},
301 this->transfer_pres.local(), this->coarse_muxer_pres, this->gate_pres, sys_lvl_coarse.gate_pres);
302 Asm::asm_transfer_blocked(virt_lvl_fine, virt_lvl_coarse, cubature, trunc_s, shrink,
303 [](const DomainLevel_& dl) {return &dl.space_stress;},
304 this->transfer_stress.local(), this->coarse_muxer_stress, this->gate_stress, sys_lvl_coarse.gate_stress);
305
306 this->transfer_velo.compile();
307 this->transfer_pres.compile();
308 this->transfer_stress.compile();
309 this->compile_system_transfer();
310 }
