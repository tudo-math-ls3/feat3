247 template<typename DomainLevel_>
248 void assemble_gates(const Domain::VirtualLevel<DomainLevel_>& virt_dom_lvl)
249 {
250 const auto& dom_level = virt_dom_lvl.level();
251 const auto& dom_layer = virt_dom_lvl.layer();
252 const auto& space_velo = dom_level.space_velo;
253 const auto& space_pres = dom_level.space_pres;
254 const auto& space_stress = dom_level.space_stress;
255
256 // set the gate comm
257 this->gate_velo.set_comm(dom_layer.comm_ptr());
258 this->gate_pres.set_comm(dom_layer.comm_ptr());
259 this->gate_stress.set_comm(dom_layer.comm_ptr());
260 this->gate_sys.set_comm(dom_layer.comm_ptr());
261
262 // loop over all ranks
263 for(Index i(0); i < dom_layer.neighbor_count(); ++i)
264 {
265 int rank = dom_layer.neighbor_rank(i);
266
267 // try to find our halo
268 auto* halo = dom_level.find_halo_part(rank);
269 XASSERT(halo != nullptr);
270
271 // create (empty) velocity mirror
272 VeloMirror mirror_velo;
273 Assembly::MirrorAssembler::assemble_mirror(mirror_velo, space_velo, *halo);
274
275 // create (empty) pressure mirror
276 PresMirror mirror_pres;
277 Assembly::MirrorAssembler::assemble_mirror(mirror_pres, space_pres, *halo);
278
279 // create (empty) stress mirror
280 StressMirror mirror_stress;
281 Assembly::MirrorAssembler::assemble_mirror(mirror_stress, space_stress, *halo);
282
283 // create a system mirror
284 SystemMirror mirror_sys(mirror_velo.clone(), mirror_pres.clone(), mirror_stress.clone());
285
286 // push mirrors into gates
287 if(!mirror_velo.empty())
288 this->gate_velo.push(rank, std::move(mirror_velo));
289 if(!mirror_pres.empty())
290 this->gate_pres.push(rank, std::move(mirror_pres));
291 if(!mirror_stress.empty())
292 this->gate_stress.push(rank, std::move(mirror_stress));
293 if(!mirror_sys.empty())
294 this->gate_sys.push(rank, std::move(mirror_sys));
295 }
296
297 // create local template vectors
298 LocalVeloVector tmpl_v(space_velo.get_num_dofs());
299 LocalPresVector tmpl_p(space_pres.get_num_dofs());
300 LocalStressVector tmpl_s(space_stress.get_num_dofs());
301 LocalSystemVector tmpl_sys(tmpl_v.clone(), tmpl_p.clone(), tmpl_s.clone());
302
303 // compile gates
304 this->gate_velo.compile(std::move(tmpl_v));
305 this->gate_pres.compile(std::move(tmpl_p));
306 this->gate_stress.compile(std::move(tmpl_s));
307 this->gate_sys.compile(std::move(tmpl_sys));
308 }

310 template<typename DomainLevel_>
311 void assemble_coarse_muxers(const Domain::VirtualLevel<DomainLevel_>& virt_lvl_coarse)
312 {
313 // assemble muxer parent
314 if(virt_lvl_coarse.is_parent())
315 {
316 XASSERT(virt_lvl_coarse.is_child());
317
318 const auto& layer_c = virt_lvl_coarse.layer_c();
319 const DomainLevel_& level_p = virt_lvl_coarse.level_p();
320
321 // loop over all children
322 for(Index i(0); i < layer_c.child_count(); ++i)
323 {
324 const auto* child = level_p.find_patch_part(int(i));
325 XASSERT(child != nullptr);
326 SystemMirror child_mirror_sys;
327 VeloMirror& child_mirror_v = child_mirror_sys.template at<0>();
328 PresMirror& child_mirror_p = child_mirror_sys.template at<1>();
329 StressMirror& child_mirror_s = child_mirror_sys.template at<2>();
330 Assembly::MirrorAssembler::assemble_mirror(child_mirror_v, level_p.space_velo, *child);
331 Assembly::MirrorAssembler::assemble_mirror(child_mirror_p, level_p.space_pres, *child);
332 Assembly::MirrorAssembler::assemble_mirror(child_mirror_s, level_p.space_stress, *child);
333 this->coarse_muxer_velo.push_child(child_mirror_v.clone(LAFEM::CloneMode::Shallow));
334 this->coarse_muxer_pres.push_child(child_mirror_p.clone(LAFEM::CloneMode::Shallow));
335 this->coarse_muxer_stress.push_child(child_mirror_s.clone(LAFEM::CloneMode::Shallow));
336 this->coarse_muxer_sys.push_child(std::move(child_mirror_sys));
337 }
338 }
339
340 // assemble muxer child
341 if(virt_lvl_coarse.is_child())
342 {
343 const auto& layer_c = virt_lvl_coarse.layer_c();
344 const DomainLevel_& level_c = virt_lvl_coarse.level_c();
345
346 SystemMirror parent_mirror_sys;
347 VeloMirror& parent_mirror_v = parent_mirror_sys.template at<0>();
348 PresMirror& parent_mirror_p = parent_mirror_sys.template at<1>();
349 StressMirror& parent_mirror_s = parent_mirror_sys.template at<2>();
350
351 // manually set up an identity gather/scatter matrix
352 {
353 Index n = level_c.space_velo.get_num_dofs();
354 parent_mirror_v = ScalarMirror(n, n);
355 auto* idx = parent_mirror_v.indices();
356 for(Index i(0); i < n; ++i)
357 idx[i] = i;
358 }
359 {
360 Index n = level_c.space_pres.get_num_dofs();
361 parent_mirror_p = ScalarMirror(n, n);
362 auto* idx = parent_mirror_p.indices();
363 for(Index i(0); i < n; ++i)
364 idx[i] = i;
365 }
366 {
367 Index n = level_c.space_stress.get_num_dofs();
368 parent_mirror_s = ScalarMirror(n, n);
369 auto* idx = parent_mirror_s.indices();
370 for(Index i(0); i < n; ++i)
371 idx[i] = i;
372 }
373
374 // set parent and sibling comms
375 this->coarse_muxer_velo.set_parent(
376 layer_c.sibling_comm_ptr(),
377 layer_c.get_parent_rank(),
378 parent_mirror_v.clone(LAFEM::CloneMode::Shallow)
379 );
380 this->coarse_muxer_pres.set_parent(
381 layer_c.sibling_comm_ptr(),
382 layer_c.get_parent_rank(),
383 parent_mirror_p.clone(LAFEM::CloneMode::Shallow)
384 );
385 this->coarse_muxer_stress.set_parent(
386 layer_c.sibling_comm_ptr(),
387 layer_c.get_parent_rank(),
388 parent_mirror_s.clone(LAFEM::CloneMode::Shallow)
389 );
390 this->coarse_muxer_sys.set_parent(
391 layer_c.sibling_comm_ptr(),
392 layer_c.get_parent_rank(),
393 std::move(parent_mirror_sys)
394 );
395
396 // compile muxer
397 LocalVeloVector tmpl_v(level_c.space_velo.get_num_dofs());
398 LocalPresVector tmpl_p(level_c.space_pres.get_num_dofs());
399 LocalStressVector tmpl_s(level_c.space_stress.get_num_dofs());
400 LocalSystemVector tmpl_sys(tmpl_v.clone(), tmpl_p.clone(), tmpl_s.clone());
401 this->coarse_muxer_velo.compile(tmpl_v);
402 this->coarse_muxer_pres.compile(tmpl_p);
403 this->coarse_muxer_stress.compile(tmpl_s);
404 this->coarse_muxer_sys.compile(tmpl_sys);
405 }
406 }

408 template<typename DomainLevel_, typename Cubature_>
409 void assemble_velocity_transfer(
410 const Domain::VirtualLevel<DomainLevel_>& virt_lvl_fine,
411 const Domain::VirtualLevel<DomainLevel_>& virt_lvl_coarse,
412 const Cubature_& cubature)
413 {
414 // get fine and coarse domain levels
415 const DomainLevel_& level_f = *virt_lvl_fine;
416 const DomainLevel_& level_c = virt_lvl_coarse.is_child() ? virt_lvl_coarse.level_c() : *virt_lvl_coarse;
417
418 const auto& space_f = level_f.space_velo;
419 const auto& space_c = level_c.space_velo;
420
421 // get local transfer operator
422 LocalVeloTransfer& loc_trans = this->transfer_velo.local();
423
424 // get local transfer matrices
425 LocalVeloTransferMatrix& loc_prol_wrapped = loc_trans.get_mat_prol();
426 LocalVeloTransferMatrix& loc_rest_wrapped = loc_trans.get_mat_rest();
427
428 // get the unwrapped types
429 typename LocalVeloTransferMatrix::BaseClass& loc_prol = loc_prol_wrapped;
430 typename LocalVeloTransferMatrix::BaseClass& loc_rest = loc_rest_wrapped;
431
432 // assemble structure?
433 if (loc_prol.empty())
434 {
435 Assembly::SymbolicAssembler::assemble_matrix_2lvl(loc_prol, space_f, space_c);
436 }
437
438 // create a local weight vector
439 LocalVeloVector loc_vec_weight = loc_prol_wrapped.create_vector_l();
440
441 // create a scalar weight vector for the assembly
442 LocalScalarVector loc_scal_vec_weight = loc_prol.create_vector_l();
443
444 // get the data arrays of the weight vectors
445 auto* v_wb = loc_vec_weight.elements();
446 auto* v_ws = loc_scal_vec_weight.elements();
447
448 // assemble prolongation matrix
449 {
450 loc_prol.format();
451 loc_scal_vec_weight.format();
452
453 // assemble prolongation matrix
454 Assembly::GridTransfer::assemble_prolongation(loc_prol, loc_scal_vec_weight,
455 space_f, space_c, cubature);
456
457 // copy weights from scalar to blocked
458 for(Index i(0); i < loc_prol.rows(); ++i)
459 v_wb[i] = v_ws[i];
460
461 // synchronize blocked weight vector
462 this->gate_velo.sync_0(loc_vec_weight);
463
464 // copy weights from blocked to scalar
465 for(Index i(0); i < loc_prol.rows(); ++i)
466 v_ws[i] = v_wb[i][0];
467
468 // invert weight components
469 loc_scal_vec_weight.component_invert(loc_scal_vec_weight);
470
471 // scale prolongation matrix
472 loc_prol.scale_rows(loc_prol, loc_scal_vec_weight);
473
474 // copy and transpose
475 loc_rest = loc_prol.transpose();
476 }
477
478 // compile velocity transfer
479 this->transfer_velo.compile();
480 }

497
498 // get local transfer matrices
499 LocalPresTransferMatrix& loc_prol = loc_trans.get_mat_prol();

501
502 // assemble structure?
503 if (loc_prol.empty())
504 {
505 Assembly::SymbolicAssembler::assemble_matrix_2lvl(loc_prol, space_f, space_c);
506 }
507
508 // get local pressure weight vector
509 LocalPresVector loc_vec_weight = loc_prol.create_vector_l();

525
526 // scale prolongation matrix

528
529 // copy and transpose

531 }
532

537 template<typename DomainLevel_, typename Cubature_>
538 void assemble_stress_transfer(
539 const Domain::VirtualLevel<DomainLevel_>& virt_lvl_fine,
540 const Domain::VirtualLevel<DomainLevel_>& virt_lvl_coarse,
541 const Cubature_& cubature)
542 {
543 // get fine and coarse domain levels
544 const DomainLevel_& level_f = *virt_lvl_fine;
545 const DomainLevel_& level_c = virt_lvl_coarse.is_child() ? virt_lvl_coarse.level_c() : *virt_lvl_coarse;
546
547 const auto& space_f = level_f.space_stress;
548 const auto& space_c = level_c.space_stress;
549
550 // get local transfer operator
551 LocalStressTransfer& loc_trans = this->transfer_stress.local();
552
553 // get local transfer matrices
554 LocalStressTransferMatrix& loc_prol_wrapped = loc_trans.get_mat_prol();
555 LocalStressTransferMatrix& loc_rest_wrapped = loc_trans.get_mat_rest();
556
557 // get the unwrapped types
558 typename LocalStressTransferMatrix::BaseClass& loc_prol = loc_prol_wrapped;
559 typename LocalStressTransferMatrix::BaseClass& loc_rest = loc_rest_wrapped;
560
561 // assemble structure?
562 if (loc_prol.empty())
563 {
564 Assembly::SymbolicAssembler::assemble_matrix_2lvl(loc_prol, space_f, space_c);
565 }
566
567 // create a local weight vector
568 LocalStressVector loc_vec_weight = loc_prol_wrapped.create_vector_l();
569
570 // create a scalar weight vector for the assembly
571 LocalScalarVector loc_scal_vec_weight = loc_prol.create_vector_l();
572

584 space_f, space_c, cubature);

624 template<typename DomainLevel_, typename Cubature_>
625 void assemble_velocity_truncation(
626 const Domain::VirtualLevel<DomainLevel_>& virt_lvl_fine,
627 const Domain::VirtualLevel<DomainLevel_>& virt_lvl_coarse,
628 const Cubature_& cubature,
629 const Stokes3FieldSystemLevel* sys_lvl_coarse = nullptr)
630 {
631 // if the coarse level is a parent, then we need the coarse system level
632 XASSERT((sys_lvl_coarse != nullptr) || !virt_lvl_coarse.is_parent());
633
634 // get fine and coarse domain levels
635 const DomainLevel_& level_f = *virt_lvl_fine;
636 const DomainLevel_& level_c = virt_lvl_coarse.is_child() ? virt_lvl_coarse.level_c() : *virt_lvl_coarse;
637
638 const auto& space_f = level_f.space_velo;
639 const auto& space_c = level_c.space_velo;
640
641 // get local transfer operator
642 LocalVeloTransfer& loc_trans = this->transfer_velo.local();
643
644 // get local transfer matrices
645 const LocalVeloTransferMatrix& loc_rest_wrapped = loc_trans.get_mat_rest();
646 LocalVeloTransferMatrix& loc_trunc_wrapped = loc_trans.get_mat_trunc();
647
648 // get the matrix blocks
649 const typename LocalVeloTransferMatrix::BaseClass& loc_rest = loc_rest_wrapped;
650 typename LocalVeloTransferMatrix::BaseClass& loc_trunc = loc_trunc_wrapped;
651
652 // restriction matrix must be already assembled
653 XASSERTM(loc_rest.size() > Index(0), "you need to call 'assemble_velocity_transfer' first");
654
655 // clone sparsity pattern of restriction matrix
656 loc_trunc = loc_rest.clone(LAFEM::CloneMode::Layout);
657
658 // create local weight vector
659 LocalVeloVector loc_vec_weight = loc_rest_wrapped.create_vector_l();
660
661 // create a scalar weight vector for the assembly
662 LocalScalarVector loc_scal_vec_weight = loc_rest.create_vector_l();
663
664 // get the data arrays of the weight vectors
665 auto* v_wb = loc_vec_weight.elements();
666 auto* v_ws = loc_scal_vec_weight.elements();
667
668 // format
669 loc_trunc.format();
670 loc_scal_vec_weight.format();
671
672 // assemble truncation matrix
673 Assembly::GridTransfer::assemble_truncation(loc_trunc, loc_scal_vec_weight, space_f, space_c, cubature);
674
675 // copy weights from scalar to blocked
676 for(Index i(0); i < loc_trunc.rows(); ++i)
677 v_wb[i] = v_ws[i];
678
679 // We now need to synchronize the weight vector in analogy to the prolongation matrix assembly.
680 // Note that the weight vector is now a coarse-level vector, so we need to synchronize over
681 // the coarse-level gate. This may be a bit more complicated if the coarse level is a ghost
682 // level, as in this case we have to join/split around the synch operation.
683
684 // synchronize weight vector using the muxer/gate
685 if(!virt_lvl_coarse.is_child())
686 {
687 // The coarse level is a simple (non-child) level that exists on all processes,
688 // so simply synch over the coarse-level gate:
689 sys_lvl_coarse->gate_velo.sync_0(loc_vec_weight);
690 }
691 else if(virt_lvl_coarse.is_parent())
692 {
693 // The coarse level is a child level and this is one of the parent processes which contain
694 // the coarse-level gate. So we first need to join the weight vector onto the parent processes
695 // first, then synch that joined vector over the parent gate and finally split the result
696 // to all child processes -- this "emulates" a synch over the (non-existent) coarse-level
697 // child gate, which is what we actually require here...
698
699 // create temporary vector on parent partitioning
700 LocalVeloVector loc_tmp = sys_lvl_coarse->gate_velo._freqs.clone(LAFEM::CloneMode::Allocate);
701
702 // join child weights over muxer
703 this->coarse_muxer_velo.join(loc_vec_weight, loc_tmp);
704
705 // sync over coarse gate
706 sys_lvl_coarse->gate_velo.sync_0(loc_tmp);
707
708 // split over muxer
709 this->coarse_muxer_velo.split(loc_vec_weight, loc_tmp);
710 }
711 else // ghost
712 {
713 // The coarse level is a ghost level, i.e. a child but not a parent. In this case, we
714 // only have to participate in the join/send operations of the muxer, which are part
715 // of the operations executed on the parents handled by the else-if case above.
716
717 this->coarse_muxer_velo.join_send(loc_vec_weight);
718
719 // parent performs sync over its gate here (see above else-if)
720
721 this->coarse_muxer_velo.split_recv(loc_vec_weight);
722 }
723
724 // copy weights from blocked to scalar
725 for(Index i(0); i < loc_trunc.rows(); ++i)
726 v_ws[i] = v_wb[i][0];
727
728 // invert components
729 loc_scal_vec_weight.component_invert(loc_scal_vec_weight);
730
731 // scale reduction matrix
732 loc_trunc.scale_rows(loc_trunc, loc_scal_vec_weight);
733 }
