97 // Mapping for structured meshes
98 template<typename FineMesh_, typename CoarseMesh_>
99 class CoarseFineCellMapping<FineMesh_, CoarseMesh_, true>
100 {
101 static_assert(FineMesh_::is_structured == true, "Do not provide the third template parameter manually!");
102
103 Index _num_elements;
104 Index _num_children;
105 Index _dim[3]={1,1,1}; // fine mesh dimension
106
107 public:
108 explicit CoarseFineCellMapping(const FineMesh_& fine_mesh, const CoarseMesh_& coarse_mesh) :
109 _num_elements(coarse_mesh.get_num_entities(CoarseMesh_::ShapeType::dimension)),
110 _num_children(fine_mesh.get_num_entities(FineMesh_::ShapeType::dimension) / _num_elements)
111 {
112 for (int d(0); d < FineMesh_::ShapeType::dimension; ++d)
113 {
114 _dim[d] = fine_mesh.get_num_slices(d);
115 }
116 }
117
118 Index calc_fcell(Index ccell, Index child=0) const
119 {
120 // extract cell position based on structured numbering
121 Index xy_plane_size = _dim[0]/2;
122 if (FineMesh_::shape_dim > 1)
123 {
124 xy_plane_size *= _dim[1]/2;
125 }
126 Index zpos = ccell / xy_plane_size; // yes, it is ok to point to a non-existent slice.
127 // see also the implementation of the operator++ of ImageIterator
128 Index ypos = (ccell - zpos * xy_plane_size) / (_dim[0]/2);
129 Index xpos = ccell - zpos * xy_plane_size - ypos * (_dim[0]/2);
130
131 // coarse position -> fine position
132 zpos = 2*zpos + (child / 4);
133 ypos = 2*ypos;
134 if (child == 2 || child == 3 || child == 6 || child == 7) { ++ypos; }
135 xpos = 2*xpos + (child % 2);
136
137 return zpos * _dim[1] * _dim[0] + ypos * _dim[0] + xpos;
138 }
139
140 Index get_num_children() const
141 {
142 return _num_children;
143 }

145 class ImageIterator
146 {
147 protected:
148 // the current index of the iterator
149 Index _index;
150 // the dimensions of the fine mesh
151 Index _dim[3]={1,1,1};
152
153 public:
154 ImageIterator() :
155 _index(0)
156 {
157 }
158
159 explicit ImageIterator(Index index, const Index dim[]) :
160 _index(index)
161 {
162 _dim[0] = dim[0];
163 _dim[1] = dim[1];
164 _dim[2] = dim[2];
165 }
166
167 ImageIterator(const ImageIterator& other) :
168 _index(other._index),
169 _dim{other._dim[0], other._dim[1], other._dim[2]}
170 {
171 }
172
173 ImageIterator& operator=(const ImageIterator& other)
174 {
175 _index = other._index;
176 _dim[0] = other._dim[0];
177 _dim[1] = other._dim[1];
178 _dim[2] = other._dim[2];
179 return *this;
180 }
181
182 ImageIterator& operator++()
183 {
184 // extract cell position based on structured numbering
185 Index zpos = _index / (_dim[0] * _dim[1]);
186 Index ypos = (_index - zpos * _dim[0] * _dim[1]) / _dim[0];
187 Index xpos = _index - zpos * _dim[0] * _dim[1] - ypos * _dim[0];
188
189 // update cell position
190 if (xpos % 2 == 0)
191 {
192 // in-element jump
193 ++xpos;
194 }
195 else
196 {
197 --xpos;
198 if (FineMesh_::ShapeType::dimension > 1 && ypos % 2 == 0)
199 {
200 // still in-element jump
201 ++ypos;
202 }
203 else
204 {
205 if (FineMesh_::ShapeType::dimension > 1) --ypos;
206 if (FineMesh_::ShapeType::dimension > 2 && zpos % 2 == 0)
207 {
208 // still in-element jump
209 ++zpos;
210 }
211 else
212 {
213 // jump to next element
214 if (FineMesh_::ShapeType::dimension > 2) --zpos; // now we point to the first fine element of the coarse element
215 if (xpos + 2 < _dim[0])
216 {
217 xpos += 2;
218 }
219 else if (ypos + 2 < _dim[1])
220 {
221 xpos = 0;
222 ypos += 2;
223 }
224 else // dont check for limit. this means we can also jump into a non-existent slice.
225 // that's needed to have the iterator at the correct endpoint.
226 // see also the implementation of calc_fcell
227 {
228 xpos = 0;
229 ypos = 0;
230 zpos += 2;
231 }
232 }
233 }
234 }
235 _index = zpos * _dim[0] * _dim[1] + ypos * _dim[0] + xpos;
236 return *this;
237 }
238
239 Index operator*() const
240 {
241 return _index;
242 }
243
244 bool operator!=(const ImageIterator& other) const
245 {
246 return _index != other._index;
247 }
248 }; // class CoarseFineCellMapping<FineMesh_, CoarseMesh_, true>::ImageIterator
