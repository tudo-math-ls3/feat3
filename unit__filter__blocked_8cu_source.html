19 template <typename DT_, typename IT_, int BlockSize_>
20 __global__ void cuda_unit_filter_blocked_rhs(DT_ * v, const DT_ * sv_elements, const IT_ * sv_indices, const Index ue, bool ign_nans)
21 {
22 Index idx = threadIdx.x + blockDim.x * blockIdx.x;
23 if (idx >= ue)
24 return;
25
26 Index block_size = Index(BlockSize_);
27 if(ign_nans)
28 {
29 for(Index j(0) ; j < block_size; ++j)
30 {
31 if(!isnan(sv_elements[block_size * idx + j]))
32 v[block_size* sv_indices[idx] + j] = sv_elements[block_size * idx + j];
33 }
34 }
35 else
36 {
37 for(Index j(0) ; j < block_size; ++j)
38 v[block_size* sv_indices[idx] + j] = sv_elements[block_size * idx + j];
39 }
40 }

42 template <typename DT_, typename IT_, int BlockSize_>
43 __global__ void cuda_unit_filter_blocked_def(DT_ * v, const DT_ * sv_elements, const IT_ * sv_indices, const Index ue, bool ign_nans)
44 {
45 Index idx = threadIdx.x + blockDim.x * blockIdx.x;
46 if (idx >= ue)
47 return;
48
49 Index block_size = Index(BlockSize_);
50 if(ign_nans)
51 {
52 for(Index j(0) ; j < block_size; ++j)
53 {
54 if(!isnan(sv_elements[block_size * idx + j]))
55 v[block_size* sv_indices[idx] + j] = DT_(0);
56 }
57 }
58 else
59 {
60 for(Index j(0) ; j < block_size; ++j)
61 v[block_size * sv_indices[idx] + j] = DT_(0);
62 }
63 }
64 }
65 }

73template <int BlockSize_, typename DT_, typename IT_>
74void UnitFilterBlocked::filter_rhs_cuda(DT_ * v, const DT_ * const sv_elements, const IT_ * const sv_indices, const Index ue, bool ign_nans)
75{
76 Index blocksize = Util::cuda_blocksize_misc;
77 dim3 grid;
78 dim3 block;
79 block.x = (unsigned)blocksize;
80 grid.x = (unsigned)ceil((ue)/(double)(block.x));
81
82 FEAT::LAFEM::Intern::cuda_unit_filter_blocked_rhs<DT_, IT_, BlockSize_><<<grid, block>>>(v, sv_elements, sv_indices, ue, ign_nans);
83
84 cudaDeviceSynchronize();
85#ifdef FEAT_DEBUG_MODE
86 cudaError_t last_error(cudaGetLastError());
87 if (cudaSuccess != last_error)
88 throw InternalError(__func__, __FILE__, __LINE__, "CUDA error occurred in execution!\n" + stringify(cudaGetErrorString(last_error)));
89#endif
90}
