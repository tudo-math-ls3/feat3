444 void write(const String& filename, const int rank, const int nparts)
445 {
446 // call the standard serial write version if nparts < 1
447 if(nparts <= 1)
448 {
449 write(filename);
450 return;
451 }
452
453 // verify rank
454 XASSERTM((rank >= 0) && (rank < nparts), "Invalid rank");
455
456 // Add rank cell array since we're parallel if we come to here
457 std::vector<double> rank_array((std::size_t(_num_cells)), double(rank));
458 add_cell_scalar("rank", rank_array.data());
459
460 // compute number of non-zero digits in (nparts-1) for padding
461 const std::size_t ndigits = Math::ilog10(std::size_t(nparts-1));
462
463 // write serial VTU file: "filename.#rank.vtu"
464 write(filename + "." + stringify(rank).pad_front(ndigits, '0'));
465
466 // we're done unless we have rank = 0
467 if(rank != 0)
468 return;
469
470 // try to open our output file
471 String pvtu_name(filename + ".pvtu");
472 std::ofstream ofs(pvtu_name.c_str());
473 if(!(ofs.is_open() && ofs.good()))
474 throw FileError("Failed to create '" + pvtu_name + "'");
475
476 // extract the file title from our filename
477 std::size_t p = filename.find_last_of("\\/");
478 String file_title = filename.substr(p == filename.npos ? 0 : ++p);
479
480 // write PVTU file
481 write_pvtu(ofs, file_title, nparts);
482
483 // and close
484 ofs.close();
485 }

499 void write(const String& filename, const Dist::Comm& comm)
500 {
501 // Add rank cell array since we're parallel if we come to here
502 std::vector<double> rank_array(std::size_t(_num_cells), double(comm.rank()));
503 add_cell_scalar("rank", rank_array.data());
504
505 // compute number of non-zero digits in (nparts-1) for padding
506 const std::size_t ndigits = std::size_t(Math::max(Math::ilog10(comm.size()-1), 1));
507
508 // write serial VTU file into a stringstream
509 std::stringstream stream;
510 write_vtu(stream);
511
512 // generate pattern for filename: "filename.#rank.vtu"
513 String pattern = filename + "." + String(ndigits, '*') + ".vtu";
514
515 // write distributed VTU files
516 DistFileIO::write_sequence(stream, pattern, comm);
517
518 // we're done unless we have rank = 0
519 if(comm.rank() != 0)
520 return;
521
522 // try to open our output file
523 String pvtu_name(filename + ".pvtu");
524 std::ofstream ofs(pvtu_name.c_str());
525 if(!(ofs.is_open() && ofs.good()))
526 throw FileError("Failed to create '" + pvtu_name + "'");
527
528 // extract the file title from our filename
529 std::size_t p = filename.find_last_of("\\/");
530 String file_title = filename.substr(p == filename.npos ? 0 : ++p);
531
532 // write PVTU file
533 write_pvtu(ofs, file_title, comm.size());
534
535 // and close
536 ofs.close();
537 }

545 void write_vtu(std::ostream& os) const
546 {
547 // fetch basic information
548 const int num_coords = MeshType::world_dim;
549 const int verts_per_cell = Shape::FaceTraits<ShapeType,0>::count;
550
551 // write VTK header
552 os << "<VTKFile type=\"UnstructuredGrid\" version=\"0.1\">" << std::endl;
553 os << "<!-- Generated by FEAT v" << version_major << "." << version_minor;
554 os << "." << version_patch << " -->" << std::endl;
555
556 // write mesh header
557 os << "<UnstructuredGrid>" << std::endl;
558 os << "<Piece NumberOfPoints=\"" << _num_verts << "\" NumberOfCells=\"" << _num_cells << "\">" << std::endl;
559
560 // write point data
561 if((!_vertex_scalars.empty()) || (!_vertex_vectors.empty()))
562 {
563 os << "<PointData>" << std::endl;
564
565 // write vertex variables
566 for(Index i(0); i < Index(_vertex_scalars.size()); ++i)
567 {
568 const auto& var(_vertex_scalars[i]);
569 os << "<DataArray type=\"Float64\" Name=\"" << var.first <<"\" Format=\"ascii\">" << std::endl;
570 for(Index j(0); j < _num_verts; ++j)
571 {
572 os << stringify_fp_sci(var.second[j], _var_prec) << std::endl;
573 }
574 os << "</DataArray>" << std::endl;
575 }
576 // write vertex fields
577 for(Index i(0); i < Index(_vertex_vectors.size()); ++i)
578 {
579 const auto& var(_vertex_vectors[i]);
580 os << "<DataArray type=\"Float64\" Name=\"" << var.first;
581 os <<"\" NumberOfComponents=\"3\" Format=\"ascii\">" << std::endl;
582 for(Index j(0); j < _num_verts; ++j)
583 {
584 os << stringify_fp_sci(var.second[3*j+0], _var_prec) << " ";
585 os << stringify_fp_sci(var.second[3*j+1], _var_prec) << " ";
586 os << stringify_fp_sci(var.second[3*j+2], _var_prec) << std::endl;
587 }
588 os << "</DataArray>" << std::endl;
589 }
590
591 os << "</PointData>" << std::endl;
592 }
593
594 // write cell data
595 if(!_cell_scalars.empty() || !_cell_vectors.empty())
596 {
597 os << "<CellData>" << std::endl;
598 if(!_cell_scalars.empty())
599 {
600 for(Index i(0); i < Index(_cell_scalars.size()); ++i)
601 {
602 const auto& var(_cell_scalars[i]);
603 os << "<DataArray type=\"Float64\" Name=\"" << var.first <<"\" Format=\"ascii\">" << std::endl;
604 for(Index j(0); j < _num_cells; ++j)
605 {
606 os << stringify_fp_sci(var.second[j], _var_prec) << std::endl;
607 }
608 os << "</DataArray>" << std::endl;
609 }
610 }
611
612 if(!_cell_vectors.empty())
613 {
614 // write cell fields
615 for(Index i(0); i < Index(_cell_vectors.size()); ++i)
616 {
617 const auto& var(_cell_vectors[i]);
618 os << "<DataArray type=\"Float64\" Name=\"" << var.first;
619 os <<"\" NumberOfComponents=\"3\" Format=\"ascii\">" << std::endl;
620 for(Index j(0); j < _num_cells; ++j)
621 {
622 os << stringify_fp_sci(var.second[3*j+0], _var_prec) << " ";
623 os << stringify_fp_sci(var.second[3*j+1], _var_prec) << " ";
624 os << stringify_fp_sci(var.second[3*j+2], _var_prec) << std::endl;
625 }
626 os << "</DataArray>" << std::endl;
627 }
628 }
629 os << "</CellData>" << std::endl;
630 }
631
632 // write vertices
633 const auto& vtx = _mesh.get_vertex_set();
634 os << "<Points>" << std::endl;
635 os << "<DataArray type=\"Float32\" NumberOfComponents=\"3\" Format=\"ascii\">" << std::endl;
636 for(Index i(0); i < _num_verts; ++i)
637 {
638 os << vtx[i][0];
639 for(int j(1); j < num_coords; ++j)
640 {
641 os << " " << vtx[i][j];
642 }
643 for(int j(num_coords); j < 3; ++j)
644 {
645 os << " 0";
646 }
647 os << std::endl;
648 }
649 os << "</DataArray>" << std::endl;
650 os << "</Points>" << std::endl;
651
652 // write cells
653 const auto& idx = _mesh.template get_index_set<MeshType::shape_dim, 0>();
654 os << "<Cells>" << std::endl;
655 os << "<DataArray type=\"UInt32\" Name=\"connectivity\">" << std::endl;
656 for(Index i(0); i < _num_cells; ++i)
657 {
658 os << idx(i, VTKShapeType::map(0));
659 for(int j(1); j < verts_per_cell; ++j)
660 {
661 os << " " << idx(i, VTKShapeType::map(j));
662 }
663 os << std::endl;
664 }
665 os << "</DataArray>" << std::endl;
666 os << "<DataArray type=\"UInt32\" Name=\"offsets\">" << std::endl;
667 for(Index i(0); i < _num_cells; ++i)
668 {
669 os << ((i+1) * verts_per_cell) << std::endl;
670 }
671 os << "</DataArray>" << std::endl;
672 os << "<DataArray type=\"UInt32\" Name=\"types\">" << std::endl;
673 for(Index i(0); i < _num_cells; ++i)
674 {
675 os << VTKShapeType::type << std::endl;
676 }
677 os << "</DataArray>" << std::endl;
678 os << "</Cells>" << std::endl;
679
680 // finish
681 os << "</Piece>" << std::endl;
682 os << "</UnstructuredGrid>" << std::endl;
683 os << "</VTKFile>" << std::endl;
684 }

698 void write_pvtu(std::ostream& os, const String& file_title, const int nparts) const
699 {
700 // write VTK header
701 os << "<VTKFile type=\"PUnstructuredGrid\" version=\"0.1\">" << std::endl;
702 os << "<!-- Generated by FEAT v" << version_major << "." << version_minor;
703 os << "." << version_patch << " -->" << std::endl;
704 os << "<PUnstructuredGrid GhostLevel=\"0\">" << std::endl;
705
706 // write vertex data
707 if((!_vertex_scalars.empty()) || (!_vertex_vectors.empty()))
708 {
709 os << "<PPointData>" << std::endl;
710
711 // write vertex variables
712 for(Index i(0); i < Index(_vertex_scalars.size()); ++i)
713 {
714 os << "<PDataArray type=\"Float64\" Name=\"" << _vertex_scalars[i].first <<"\" />" << std::endl;
715 }
716 // write vertex fields
717 for(Index i(0); i < Index(_vertex_vectors.size()); ++i)
718 {
719 os << "<PDataArray type=\"Float64\" Name=\"" << _vertex_vectors[i].first;
720 os <<"\" NumberOfComponents=\"3\" />" << std::endl;
721 }
722
723 os << "</PPointData>" << std::endl;
724 }
725
726 // write cell variables
727 if(!_cell_scalars.empty() || !_cell_vectors.empty())
728 {
729 os << "<PCellData>" << std::endl;
730 // write cell scalars
731 for(Index i(0); i < Index(_cell_scalars.size()); ++i)
732 {
733 os << "<PDataArray type=\"Float64\" Name=\"" << _cell_scalars[i].first <<"\" />" << std::endl;
734 }
735 // write cell fields
736 for(Index i(0); i < Index(_cell_vectors.size()); ++i)
737 {
738 os << "<PDataArray type=\"Float64\" Name=\"" << _cell_vectors[i].first;
739 os <<"\" NumberOfComponents=\"3\" />" << std::endl;
740 }
741 os << "</PCellData>" << std::endl;
742 }
743
744 // write vertices
745 os << "<PPoints>" << std::endl;
746 os << "<PDataArray type=\"Float32\" NumberOfComponents=\"3\" />" << std::endl;
747 os << "</PPoints>" << std::endl;
748
749 // compute number of non-zero digits in (nparts-1) for padding
750 const std::size_t ndigits = Math::ilog10(std::size_t(nparts-1));
751
752 // now let's write our piece data
753 for(int i(0); i < nparts; ++i)
754 {
755 os << "<Piece Source=\"" << file_title << "." << stringify(i).pad_front(ndigits, '0');
756 os << ".vtu\" />" << std::endl;
757 }
758
759 // finish
760 os << "</PUnstructuredGrid>" << std::endl;
761 os << "</VTKFile>" << std::endl;
762 }
