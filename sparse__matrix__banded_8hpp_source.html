134 template<typename LocalMatrix_, typename RowMapping_, typename ColMapping_>
135 void operator()(const LocalMatrix_& loc_mat, const RowMapping_& row_map,
136 const ColMapping_& col_map, DT_ alpha = DT_(1))
137 {
138 // loop over all local row entries
139 for(int i(0); i < row_map.get_num_local_dofs(); ++i)
140 {
141 // fetch row index
142 const Index ix = row_map.get_index(i);
143
144 // build column pointer for this row entry contribution
145 for(IT_ k(0); k < _num_of_offsets; ++k)
146 {
147 if(_offsets[k] + ix + 1 >= _num_rows && _offsets[k] + ix + 1 < 2 * _num_rows)
148 {
149 _col_ptr[_offsets[k] + ix + 1 - _num_rows] = IT_(k * _num_rows + ix);
150 }
151 }
152
153 // loop over all local column entries
154 for(int j(0); j < col_map.get_num_local_dofs(); ++j)
155 {
156 // fetch column index
157 const Index jx = col_map.get_index(j);
158
159 // ensure that the column pointer is valid for this index
160 ASSERTM(_col_ptr[jx] != _deadcode, "invalid column index");
161
162 // incorporate data into global matrix
163 _data[_col_ptr[jx]] += alpha * loc_mat[i][j];
164
165 // continue with next column entry
166 }
167
168#ifdef DEBUG
169 // reformat column-pointer array
170 for(IT_ k(0); k < _num_of_offsets; ++k)
171 {
172 if(_offsets[k] + ix + 1 >= _num_rows && _offsets[k] + ix + 1 < 2 * _num_rows)
173 {
174 _col_ptr[_offsets[k] + ix + 1 - _num_rows] = _deadcode;
175 }
176 }
177#endif
178 // continue with next row entry
179 }
180 }

414 explicit SparseMatrixBanded(const Index rows_in, const Index columns_in,
415 DenseVector<DT_, IT_> & val_in,
416 DenseVector<IT_, IT_> & offsets_in) :
417 Container<DT_, IT_>(rows_in * columns_in)
418 {
419 if (val_in.size() != rows_in * offsets_in.size())
420 {
421 XABORTM("Size of values does not match to number of offsets and row count!");
422 }
423
424 this->_scalar_index.push_back(rows_in);
425 this->_scalar_index.push_back(columns_in);
426
427 Index tused_elements(0);
428
429 for (Index i(0); i < offsets_in.size(); ++i)
430 {
431 const Index toffset(Index(offsets_in(i)));
432
433 if (toffset + Index(2) > rows_in + columns_in)
434 {
435 XABORTM("Offset out of matrix!");
436 }
437
438 tused_elements += columns_in + Math::min(rows_in, columns_in + rows_in - toffset - 1) - Math::max(columns_in + rows_in - toffset - 1, columns_in);
439 }
440
441 this->_scalar_index.push_back(tused_elements);
442 this->_scalar_index.push_back(offsets_in.size());
443
444 this->_elements.push_back(val_in.elements());
445 this->_elements_size.push_back(val_in.size());
446 this->_indices.push_back(offsets_in.elements());
447 this->_indices_size.push_back(offsets_in.size());
448
449 for (Index i(0) ; i < this->_elements.size() ; ++i)
450 MemoryPool::increase_memory(this->_elements.at(i));
451 for (Index i(0) ; i < this->_indices.size() ; ++i)
452 MemoryPool::increase_memory(this->_indices.at(i));
453 }

476 explicit SparseMatrixBanded(const Adjacency::Graph & graph) :
477 Container<DT_, IT_>(0)
478 {
479 Index num_rows = graph.get_num_nodes_domain();
480 Index num_cols = graph.get_num_nodes_image();
481
482 const Index * dom_ptr(graph.get_domain_ptr());
483 const Index * img_idx(graph.get_image_idx());
484
485 std::set<IT_> moffsets;
486
487 for(Index row(0); row < num_rows; ++row)
488 {
489 for(Index j(dom_ptr[row]); j < dom_ptr[row+1]; ++j)
490 {
491 moffsets.insert(IT_(num_rows - 1 + img_idx[j] - row));
492 }
493 }
494
495 DenseVector<IT_, IT_> toffsets(Index(moffsets.size()));
496 auto * ptoffsets = toffsets.elements();
497
498 IT_ idx(0);
499 for (auto off : moffsets)
500 {
501 ptoffsets[idx] = off;
502 ++idx;
503 }
504
505 DenseVector<IT_, IT_> toffsets_mem;
506 toffsets_mem.convert(toffsets);
507 DenseVector<DT_, IT_> tval_mem(Index(moffsets.size()) * num_rows, DT_(0));
508
509 this->assign(SparseMatrixBanded(num_rows, num_cols, tval_mem, toffsets_mem));
510 }

655 template <typename DT2_, typename IT2_>
656 void convert(const SparseMatrixCSR<DT2_, IT2_> & csr)
657 {
658 XASSERT(csr.template used_elements<Perspective::pod>() > 0);
659
660 std::set<IT_> offset_set;
661 Index nrows(csr.rows());
662 Index ncolumns(csr.columns());
663 for (Index row(0) ; row < nrows ; ++row)
664 {
665 for (Index i(csr.row_ptr()[row]) ; i < csr.row_ptr()[row+1] ; ++i)
666 {
667 //compute band offset for each non zero entry
668 IT_ off = IT_(csr.col_ind()[i] - row + nrows - 1);
669 offset_set.insert(off);
670 }
671 }
672
673 DenseVector<DT_, IT_> val_new(Index(offset_set.size()) * nrows, DT_(0));
674 for (Index row(0) ; row < nrows ; ++row)
675 {
676 for (Index i(csr.row_ptr()[row]) ; i < csr.row_ptr()[row+1] ; ++i)
677 {
678 Index col = csr.col_ind()[i];
679 Index offset = (col - row + nrows - 1);
680 Index band = Index(std::distance(offset_set.begin(), offset_set.find(IT_(offset))));
681 val_new.elements()[band * nrows + row] = csr.val()[i];
682 }
683 }
684
685 DenseVector<IT_, IT_> offsets_new(Index(offset_set.size()));
686 auto it_offset = offset_set.begin();
687 for (Index i(0) ; i < offset_set.size() ; ++i, ++it_offset)
688 {
689 offsets_new(i, *it_offset);
690 }
691 offset_set.clear();
692
693 SparseMatrixBanded<DT_, IT_> temp(nrows, ncolumns, val_new, offsets_new);
694 this->assign(temp);
695 }

745 DT_ operator()(Index row, Index col) const
746 {
747 ASSERT(row < rows());
748 ASSERT(col < columns());
749
750 MemoryPool::synchronize();
751
752 const Index trows(this->_scalar_index.at(1));
753
754 for (Index i(0); i < this->_scalar_index.at(4); ++i)
755 {
756 const Index toffset(this->offsets()[i]);
757 if (row + toffset + Index(1) == col + trows)
758 {
759 return this->val()[i * trows + row];
760 }
761 }
762 return DT_(0.);
763 }

878 void axpy(
879 const SparseMatrixBanded & x,
880 const SparseMatrixBanded & y,
881 const DT_ alpha = DT_(1))
882 {
883 XASSERTM(x.rows() == y.rows(), "Matrix rows do not match!");
884 XASSERTM(x.rows() == this->rows(), "Matrix rows do not match!");
885 XASSERTM(x.columns() == y.columns(), "Matrix columns do not match!");
886 XASSERTM(x.columns() == this->columns(), "Matrix columns do not match!");
887 XASSERTM(x.num_of_offsets() == y.num_of_offsets(), "Matrix num_of_offsets do not match!");
888 XASSERTM(x.num_of_offsets() == this->num_of_offsets(), "Matrix num_of_offsets do not match!");
889 XASSERTM(x.used_elements() == y.used_elements(), "Matrix used_elements do not match!");
890 XASSERTM(x.used_elements() == this->used_elements(), "Matrix used_elements do not match!");
891
892 if (Math::abs(alpha) < Math::eps<DT_>())
893 {
894 this->copy(y);
895 //y.scale(beta);
896 return;
897 }
898
899 TimeStamp ts_start;
900
901 Statistics::add_flops(this->used_elements() * 2);
902 Arch::Axpy::value(this->val(), alpha, x.val(), y.val(), this->rows() * this->num_of_offsets());
903
904 TimeStamp ts_stop;
905 Statistics::add_time_axpy(ts_stop.elapsed(ts_start));
906 }

956 void apply(DenseVector<DT_, IT_>& r, const DenseVector<DT_, IT_>& x) const
957 {
958 XASSERTM(r.size() == this->rows(), "Vector size of r does not match!");

972 this->offsets(),
973 this->num_of_offsets(),
974 this->rows(),
975 this->columns());
976
