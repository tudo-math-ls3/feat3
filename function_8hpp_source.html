275 template<typename Function_, typename DT_, int dim_, int s_>
276 typename Analytic::EvalTraits<DT_, Function_>::ValueType
277 eval_value(const Function_& function, const Tiny::Vector<DT_, dim_, s_>& point)
278 {
279 typedef Analytic::EvalTraits<DT_, Function_> Traits;
280 static_assert(Function_::can_value, "function does not support evaluation of values");
281 static_assert(dim_ == Traits::domain_dim, "invalid point dimension");
282 typename Function_::template Evaluator<Traits> evaluator(function);
