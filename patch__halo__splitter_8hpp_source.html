83 explicit PatchHaloSplitter(const MeshType& base_mesh, const MeshPartType& patch_mesh_part) :
84 _base_mesh(base_mesh),
85 _patch_mesh_part(patch_mesh_part),
86 _patch_inv_map(std::size_t(shape_dim+1)),
87 _isect_halo_idx(std::size_t(shape_dim+1))
88 {
89 // allocate inverse patch mapping

93 _patch_inv_map.at(std::size_t(dim)).resize(_base_mesh.get_num_entities(dim), ~Index(0));
94 }

96 Intern::PatchInvMapWrapper<ShapeType>::build(_patch_inv_map, patch_mesh_part.get_target_set_holder());

112 std::size_t add_halo(int halo_rank, const MeshPartType& halo)
113 {
114 // create new split halo object
115 auto sh = std::make_shared<SplitHaloType>();
116
117 // If the following function returns false, then this patch is not adjacent to that halo
118 if(!sh->split(_patch_inv_map, halo.get_target_set_holder()))
119 return std::size_t(0);
120
121 // okay, store halo and return split size
122 return _split_halos.emplace(halo_rank, sh).first->second->size();
123 }

163 bool intersect_split_halo(int halo_rank, const std::vector<Index>& buffer, const Index buffer_offset)
164 {
165 // clear all intersection indices
166 for(auto& x : _isect_halo_idx)
167 x.clear();
168
169 const auto it = _split_halos.find(halo_rank);
170 if(it == _split_halos.end())
171 return false;
172
173 // try to intersect
174 return it->second->intersect(_isect_halo_idx, buffer, std::size_t(buffer_offset));
175 }

206 //template<int shape_dim_>
207 class PatchInvMap
208 {
209 public:
210 static void build(std::vector<Index>& pim, const TargetSet& ts)
211 {
212 for(Index i(0); i < ts.get_num_entities(); ++i)
213 pim[ts[i]] = i;
214 }
215
216 static bool split(std::vector<Index>& shi, std::vector<Index>& spi, const std::vector<Index>& pim, const TargetSet& ts)
217 {
218 shi.reserve(ts.get_num_entities());
219 spi.reserve(ts.get_num_entities());
220
221 for(Index i(0); i < ts.get_num_entities(); ++i)
222 {
223 const Index patch_idx = pim[ts[i]];
224 if(patch_idx != ~Index(0))
225 {
226 spi.push_back(patch_idx);
227 shi.push_back(i);
228 }
229 }
230
231 return !spi.empty();
232 }
233
234 static void fill(TargetSet& trg, const std::vector<Index>& idx)
235 {
236 const Index n = trg.get_num_entities();
237 XASSERT(n == Index(idx.size()));
238 for(Index i(0); i < n; ++i)
239 trg[i] = idx[std::size_t(i)];
240 }
241 }; // class PatchInvMap

292 template<typename Shape_>
293 class PatchHaloSplitPart
294 {
295 protected:
296 static constexpr int shape_dim = Shape_::dimension;
297
299 std::vector<std::vector<Index>> _halo_idx;
301 std::vector<std::vector<Index>> _patch_idx;
302
303 public:
304 explicit PatchHaloSplitPart()
305 {
306 std::size_t sdim = std::size_t(shape_dim+1);
307 _halo_idx.resize(sdim);
308 _patch_idx.resize(sdim);
309 }
310
311 bool split(const std::vector<std::vector<Index>>& pim, const TargetSetHolder<Shape_>& tsh)
312 {
313 return PatchInvMapWrapper<Shape_>::split(_halo_idx, _patch_idx, pim, tsh);
314 }
315
316 std::size_t size() const
317 {
318 std::size_t s = std::size_t(0);
319 for(const auto& x : _halo_idx)
320 s += x.size();
321
322 // s == 0 ==> patch and halo are disjoint
323 return s == std::size_t(0) ? s : ++s + _halo_idx.size();
324 }
325
326 void serialize(std::vector<Index>& buffer, int child_rank)
327 {
328 const std::size_t s = this->size();
329 if(this->size() == std::size_t(0))
330 return;
331
332 if(buffer.capacity() < buffer.size() + s)
333 buffer.reserve(buffer.size() + s);
334
335 buffer.push_back(Index(child_rank));
336
337 for(std::size_t i(0); i < _halo_idx.size(); ++i)
338 buffer.push_back(Index(_halo_idx.at(i).size()));
339
340 for(std::size_t i(0); i < _halo_idx.size(); ++i)
341 {
342 const std::vector<Index>& v = _halo_idx.at(i);
343 for(std::size_t j(0); j < v.size(); ++j)
344 buffer.push_back(v[j]);
345 }
346 }
347
348 bool intersect(std::vector<std::vector<Index>>& isect, const std::vector<Index>& other, const std::size_t offset) const
349 {
350 // offset of first index within other halo
351 std::size_t off = offset + std::size_t(shape_dim+2);
352
353 bool ret = false;
354
355 // compute offsets from sizes
356 for(std::size_t i(0); i <= std::size_t(shape_dim); ++i)
357 {
358 // get index counts
359 const std::size_t n_1 = _halo_idx.at(i).size();
360 const std::size_t n_2 = other.at(offset+i+1u);
361
362 // get the minimum of both
363 const std::size_t n_min = Math::min(n_1, n_2);
364 if(n_min == std::size_t(0))
365 {
366 off += n_2;
367 continue;
368 }
369
370 // get my halo indices
371 const Index* idx_1 = _halo_idx.at(i).data();
372 const Index* idx_2 = &other.data()[off];
373 const Index* idx_p = _patch_idx.at(i).data();
374
375 // intersect idx_1 and idx_2
376 std::vector<Index>& idx = isect.at(i);
377 idx.reserve(n_min);
378 for(std::size_t j_1(0), j_2(0); (j_1 < n_1) && (j_2 < n_2); )
379 {
380 if(idx_1[j_1] < idx_2[j_2])
381 ++j_1;
382 else if(idx_1[j_1] > idx_2[j_2])
383 ++j_2;
384 else
385 {
386 // match; store halo index w.r.t. patch
387 idx.push_back(idx_p[j_1]);
388 ++j_1;
389 ++j_2;
390 ret = true;
391 }
392 }
393
394 off += n_2;
395 }
396
397 // okay
398 return ret;
399 }
