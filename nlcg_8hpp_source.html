310 virtual Status apply(VectorType& vec_cor, const VectorType& vec_def) override
311 {
312 // clear solution vector
313 vec_cor.format();
314
315 // Evaluate the functional at the new state
316 this->_functional.prepare(vec_cor, this->_filter);
317 this->_functional.eval_fval_grad(this->_fval, this->_vec_r);
318
319 // Copy back given defect
320 this->_vec_r.copy(vec_def);
321 //this->_system_filter.filter_def(this->_vec_r);
322
323 // Prepare the preconditioner (if any)
324 if(this->_precond != nullptr)
325 {
326 this->_precond->prepare(vec_cor, this->_filter);
327 }
328
329 // apply
330 this->_status = _apply_intern(vec_cor);
331 this->plot_summary();
332 return this->_status;
333 }

336 virtual Status correct(VectorType& vec_sol, const VectorType& DOXY(vec_rhs)) override
337 {
338 //std::cout << std::scientific;
339 //std::cout << std::setprecision(16);
340 // Evaluate the functional at the new state
341 this->_functional.prepare(vec_sol, this->_filter);
342 // Compute defect
343 this->_functional.eval_fval_grad(this->_fval, this->_vec_r);
344 this->_vec_r.scale(this->_vec_r,DataType(-1));
345 this->_filter.filter_def(this->_vec_r);
346
347 // Prepare the preconditioner (if any)
348 if(this->_precond != nullptr)
349 {
350 this->_precond->prepare(vec_sol, this->_filter);
351 }
352
353 // apply
354 this->_status = _apply_intern(vec_sol);
355 this->plot_summary();
356 return this->_status;
357 }

423 virtual Status _apply_intern(VectorType& vec_sol)
424 {
425 IterationStats pre_iter(*this);
426 Statistics::add_solver_expression(std::make_shared<ExpressionStartSolve>(this->name()));
427
428 // p[k+1] <- r[k+1] + _beta * p[k+1]
429 DataType beta;
430 // eta[k] = <r[k], p[k]>
431 DataType eta;
432 // Reset member variables in the LineSearch
433 _linesearch->reset();
434
435 if(iterates != nullptr)
436 {
437 iterates->push_back(std::move(vec_sol.clone()));
438 }
439
440 // Set initial defect. The defect vector was calculated in the calling function
441 Status status = this->_set_initial_defect(this->_vec_r, vec_sol);
442 if(status != Status::progress)
443 {
444 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), status, this->get_num_iter()));
445 return status;
446 }
447
448 // apply preconditioner to defect vector
449 if(!this->_apply_precond(this->_vec_z, this->_vec_r, this->_filter))
450 {
451 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
452 return Status::aborted;
453 }
454 //this->_vec_z.copy(this->_vec_r);
455
456 // The first direction has to be the (preconditioned) steepest descent direction
457 this->_vec_p.copy(this->_vec_z);
458
459
461 //DataType max_elem(this->_vec_p.max_element());
462 //DataType scale1 = DataType(1)/max_elem;
463 //std::cout << "max_elem " << max_elem << " " << scale1 << std::endl;
464 //DataType scale2(DataType(1)/new_norm);
465 //std::cout << " new norm " << new_norm << " " << scale2 << std::endl;
466
467 // Compute initial eta = < p, r>
468 eta = this->_vec_p.dot(this->_vec_r);
469
470 // If the preconditioner was not pd in the first step, reset the search direction to steepest descent
471 if(eta <= DataType(0))
472 {
473 this->_vec_p.copy(this->_vec_r);
474 }
475
476 // compute initial gamma = < z[0], r[0] >
477 DataType gamma(this->_vec_z.dot(this->_vec_r));
478 DataType gamma_prev(0);
479
480 if(this->_def_init <= this->_tol_rel)
481 {
482 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::success, this->get_num_iter()));
483 return Status::success;
484 }
485
486 Index its_since_restart(0);
487 _num_restarts = Index(0);
488 Index first_restart(_restart_freq+Index(1));
489
490 pre_iter.destroy();
491
492 // start iterating
493 while(status == Status::progress)
494 {
495 IterationStats stat(*this);
496
497 // Clear statistics if we have a preconditioner. Otherwise this tends to overflow the main memory if the
498 // preconditioner does too many iterations
499 // \todo: fix this
500 //FEAT::Statistics::reset_solver_statistics();
501
502 this->_fval_prev = this->_fval;
503
504 this->_vec_y.copy(this->_vec_r);
505
506 // Copy information to the linesearch
507 _linesearch->set_initial_fval(this->_fval);
508 _linesearch->set_grad_from_defect(this->_vec_r);
509 // If we are using a preconditioner, use the additional information about the preconditioned step length in the line search
510 if(this->_precond != nullptr)
511 {
512 _linesearch->set_dir_scaling(true);
513 }
514
515 // Call the linesearch to update vec_sol
516 Status linesearch_status = _linesearch->correct(vec_sol, this->_vec_p);
517
518 // Copy back information from the linesearch
519 this->_fval = _linesearch->get_final_fval();
520 this->_ls_its = _linesearch->get_num_iter();
521 this->_steplength = _linesearch->get_rel_update();
522 _linesearch->get_defect_from_grad(this->_vec_r);
523 this->_vec_y.axpy(this->_vec_y, this->_vec_r, -DataType(1));
524
525 // Log iterates if necessary
526 if(iterates != nullptr)
527 {
528 iterates->push_back(vec_sol.clone());
529 }
530
531 // Compute defect norm. This also performs the convergence/divergence checks.
532 status = this->_set_new_defect(this->_vec_r, vec_sol);
533
534 if(status != Status::progress)
535 {
536 stat.destroy();
537 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), status, this->get_num_iter()));
538 return status;
539 }
540
541 // Update preconditioner if necessary
542 if(this->_precond != nullptr)
543 {
544 this->_precond->prepare(vec_sol, this->_filter);
545 }
546
547 // apply preconditioner
548 if(!this->_apply_precond(_vec_z, _vec_r, this->_filter))
549 {
550 stat.destroy();
551 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
552 return Status::aborted;
553 }
554
555 // Save old gamma and compute new
556 gamma_prev = gamma;
557 gamma = this->_vec_r.dot(this->_vec_z);
558
559 //String info("");
560 bool restart(false);
561 // We need to perform a steepest descent step if the Wolfe conditions do not hold (linesearch status
562 // != success).
563 if(linesearch_status != Status::success)
564 {
565 //info += " Wolfe Conditions do not hold";
566 restart = true;
567 _num_restarts++;
568 }
569 else
570 {
571 _num_restarts = Index(0);
572 }
573
574 if(_restart_freq > Index(0) && this->_num_iter >= first_restart && its_since_restart%_restart_freq == 0)
575 {
576 //info += " Scheduled restart";
577 restart = true;
578 its_since_restart = Index(0);
579 }
580 // This needs to be done after all the checks
581 ++its_since_restart;
582
583 // Set beta = 0 or compute depending on the restart flag
584 restart ? beta = DataType(0) : beta = compute_beta(gamma, gamma_prev);
585
586 //std::cout << "Beta " << beta << info << std::endl;
587
588 // We need to check beta again here as some direction updates might set it to zero
589 // Discard the old search direction and perform (preconditioned) steepest descent
590 if(beta == DataType(0))
591 {
592 this->_vec_p.copy(this->_vec_z);
593 }
594 else
595 {
596 this->_vec_p.axpy(this->_vec_p, this->_vec_z, beta);
597 }
598
599
600 //max_elem = this->_vec_p.max_element();
601 //scale1 = DataType(1)/max_elem;
602 //std::cout << "max_elem " << max_elem << " " << scale1 << std::endl;
603 //scale2 = DataType(1)/new_norm;
604 //std::cout << " new norm " << new_norm << " " << scale2 << std::endl;
605
606 eta = this->_vec_p.dot(this->_vec_r);
607
608 // Safeguard as this should not happen
609 // TODO: Correct the output of the preconditioner if it turns out to not have been positive definite
610 if(eta <= DataType(0))
611 {
612 this->_vec_p.copy(this->_vec_r);
613 //return Status::aborted;
614 }
615
616 }
617
618 // We should never come to this point
619 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::undefined, this->get_num_iter()));
620 return Status::undefined;
621 }

636 DataType compute_beta(const DataType& gamma, const DataType& gamma_prev) const
637 {
638 switch(_direction_update)
639 {
640 case NLCGDirectionUpdate::DaiYuan:
641 return dai_yuan(gamma);
642 case NLCGDirectionUpdate::DYHSHybrid:
643 return dy_hs_hybrid(gamma);
644 case NLCGDirectionUpdate::FletcherReeves:
645 return fletcher_reeves(gamma, gamma_prev);
646 case NLCGDirectionUpdate::HagerZhang:
647 // Warning: This update is not very efficient in its current implementation.
648 return hager_zhang(gamma);
649 case NLCGDirectionUpdate::HestenesStiefel:
650 return hestenes_stiefel();
651 case NLCGDirectionUpdate::PolakRibiere:
652 return polak_ribiere(gamma_prev);
653 default:
654 XABORTM("Unhandled direction update: "+stringify(_direction_update));
655 }
656 }

673 virtual Status _set_new_defect(const VectorType& vec_r, const VectorType& vec_sol) override
674 {
675
676 Status st = BaseClass::_set_new_defect(vec_r, vec_sol);
677
678 if(st != Status::progress)
679 {
680 return st;
681 }
682
683 // If there were too many subsequent restarts, the solver is stagnated
684 if(_max_num_restarts > Index(0) && _num_restarts > _max_num_restarts)
685 {
686 return Status::stagnated;
687 }
688
689 // continue iterating
690 return Status::progress;
691 }

713 DataType dai_yuan(const DataType gamma) const
714 {
715 // <r[k+1] - r[k], p[k]>
716 DataType eta_dif = this->_vec_p.dot(this->_vec_y);
717
718 return gamma/(-eta_dif);
719 }
720
742 DataType dy_hs_hybrid(const DataType gamma) const
743 {
744
745 // <r[k+1] - r[k], z[k+1]>
746 DataType gamma_dif = this->_vec_z.dot(this->_vec_y);
747 // <r[k+1] - r[k], p[k]>
748 DataType eta_dif = this->_vec_p.dot(this->_vec_y);
749
750 DataType beta(Math::min(gamma/(-eta_dif), gamma_dif/(-eta_dif)));
751 beta = Math::max(DataType(0), beta);
752
753 return beta;
754
755 }
756
777 DataType fletcher_reeves(const DataType gamma, const DataType gamma_prev) const
778 {
779 return gamma/gamma_prev;
780 }
781
812 DataType hager_zhang(const DataType gamma) const
813 {
814
815 DataType thresh = DataType(0.01);
816
817 // <r[k+1] - r[k], p[k]>
818 DataType eta_dif = this->_vec_p.dot(this->_vec_y);
819 // <r[k+1] - r[k], z[k+1]>
820 DataType gamma_dif = this->_vec_y.dot(this->_vec_z);
821 // <p[k], z[k+1]>
822 DataType omega_mid = this->_vec_p.dot(this->_vec_z);
823 // || r[k+1] - r[k]
824 DataType norm_y = this->_vec_y.norm2();
825
826 DataType beta = -( gamma_dif/(eta_dif) + DataType(2)*(norm_y)*omega_mid/Math::sqr(eta_dif));
827 beta = Math::max(beta,-DataType(1)/(this->_vec_p.norm2()*Math::min(thresh, Math::sqrt(gamma))));
828
829 return beta;
830 }
831
851 DataType hestenes_stiefel() const
852 {
853 // <r[k+1] - r[k], p[k]>
854 DataType eta_dif = this->_vec_p.dot(this->_vec_y);
855 // <r[k+1] - r[k], z[k+1]>
856 DataType gamma_dif = this->_vec_z.dot(this->_vec_y);
857
858 return gamma_dif/( -eta_dif);
859 }
860
876 DataType polak_ribiere(const DataType gamma_prev) const
877 {
878 // <r[k+1] - r[k], z[k+1]>
879 DataType gamma_dif = this->_vec_z.dot(this->_vec_y);
880
881 return Math::max(gamma_dif/gamma_prev, DataType(0));
882 }
