228 virtual Status _apply_intern(VectorType& vec_sol, const VectorType& vec_rhs)
229 {
230 Statistics::add_solver_expression(std::make_shared<ExpressionStartSolve>(this->name()));
231
232 const Index inner_iter_digits(Math::ilog10(_inner_solver->get_max_iter()));
233
234 Status st(_set_initial_defect(_functional.get_constraint()));
235
236 IterationStats stat(*this);
237
238 // The Penalty Iteration(TM)
239 while(st == Status::progress)
240 {
241 ++(this->_num_iter);
242
243 DataType def_old(this->_def_cur);
244 //DataType penalty_param_old(penalty_param);
245
246 Status inner_st(_inner_solver->correct(vec_sol, vec_rhs));
247 if(inner_st == Status::aborted)
248 {
249 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
250 return Status::aborted;
251 }
252
253 // After the inner solver was called, the constraint is already there so we can just get() it
254 this->_def_cur = _functional.get_constraint();
255 DataType penalty_param(_functional.get_penalty_param());
256
257 if(this->_plot_iter())
258 {
259 String msg = this->_plot_name
260 + ": " + stringify(this->_num_iter).pad_front(this->_iter_digits)
261 + " (" + stringify(this->_inner_solver->get_num_iter()).pad_front(inner_iter_digits)
262 + ":" + stringify(inner_st) + ")"
263 + " : " + stringify_fp_sci(this->_def_cur)
264 + " / " + stringify_fp_sci(this->_def_cur / this->_def_init)
265 + " / " + stringify_fp_sci(DataType(0.5)*Math::sqr(this->_def_cur))
266 + " : " + stringify_fp_sci(penalty_param);
267
268 this->_print_line(msg);
269 }
270
271 // ensure that the defect is neither NaN nor infinity
272 if(!Math::isfinite(this->_def_cur))
273 {
274 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
275 return Status::aborted;
276 }
277
278 // is diverged?
279 if(this->is_diverged())
280 {
281 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::diverged, this->get_num_iter()));
282 return Status::diverged;
283 }
284
285 // minimum number of iterations performed?
286 if(this->_num_iter < this->_min_iter)
287 {
288 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::progress, this->get_num_iter()));
289 return Status::progress;
290 }
291
292 // maximum number of iterations performed?
293 if(this->_num_iter >= this->_max_iter)
294 {
295 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::max_iter, this->get_num_iter()));
296 return Status::max_iter;
297 }
298
299 // Check for convergence
300 if(this->is_converged())
301 {
302 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::success, this->get_num_iter()));
303 return Status::success;
304 }
305
306 // Increment for the penalty factor by at least factor 5, very arbitrary
307 penalty_param = DataType(5)*Math::sqr(penalty_param)
308 *Math::max(Math::sqr( def_old/this->_def_cur),DataType(1));
309
310 if(penalty_param >= _tol_penalty)
311 {
312 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::stagnated, this->get_num_iter()));
313 return Status::stagnated;
314 }
315
316 _functional.set_penalty_param(penalty_param);
317 }
318
319 // We should never come to this point
320 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::undefined, this->get_num_iter()));
321 return Status::undefined;
322 }

333 Status _set_initial_defect(const DataType def_init_)
334 {
335 // store new defect
336 this->_def_init = this->_def_cur = def_init_;
337 // insert special toe to signal new start of solver
338 //Statistics::add_solver_toe(this->_branch, double(-1));
339 //Statistics::add_solver_mpi_execute(this->_branch, double(-1));
340 //Statistics::add_solver_mpi_wait(this->_branch, double(-1));
341 //insert -1 as first defect, to signalize a new starting solver iteration run
342 //Statistics::add_solver_defect(this->_branch, double(-1));
343 //Statistics::add_solver_defect(this->_branch, double(this->_def_init));
344 this->_num_iter = Index(0);
345 this->_num_stag_iter = Index(0);
346 Statistics::add_solver_expression(std::make_shared<ExpressionDefect>(this->name(), this->_def_init, this->get_num_iter()));
347
348 // Plot?
349 if(this->_plot_iter())
350 {
351 String msg = this->_plot_name
352 + ": " + stringify(0).pad_front(this->_iter_digits)
353 + " : " + stringify_fp_sci(this->_def_init);
354
355 this->_print_line(msg);
356 }
357
358 // Ensure that the defect is neither NaN nor infinity
359 if(!Math::isfinite(this->_def_init))
360 {
361 return Status::aborted;
362 }
363
364 // Check if the initial defect lower than the absolute tolerance
365 if(this->_def_init <= this->_tol_abs)
366 {
367 return Status::success;
368 }
369
370 // continue iterating
371 return Status::progress;
372 }
373
