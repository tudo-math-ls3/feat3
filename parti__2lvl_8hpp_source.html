70 class Parti2Lvl<ConformalMesh<Shape_, num_coords_, Coord_>>

98 explicit Parti2Lvl(const MeshType& mesh, Index num_ranks) :
99 _num_elems(mesh.get_num_elements()),
100 _num_ranks(num_ranks),
101 _success(false),
102 _ref_lvl(0),
103 _ref_elems(0)
104 {
105 XASSERT(num_ranks > Index(0));
106
107 const Index factor = Intern::Parti2LvlHelper<Shape_>::factor;
108 const Index lvlinc = Intern::Parti2LvlHelper<Shape_>::lvlinc;
109 Index count(_num_elems);
110 Index power(0);
111
112 while(count < _num_ranks)
113 {
114 count *= factor;
115 ++power;
116 }
117
118 if(count != _num_ranks)
119 {
120 // no 2-level partitioning possible
121 return;
122 }
123
124 _success = true;
125
126 // compute refinement level
127 _ref_lvl = (power + lvlinc - Index(1)) / lvlinc;
128
129 // compute refined element count
130 const Index ref_fac(Intern::StandardRefinementTraits<Shape_, Shape_::dimension>::count);
131 _ref_elems = _num_elems;
132 for(Index i(0); i < _ref_lvl; ++i)
133 _ref_elems *= ref_fac;
134 }
135
142 bool success() const
143 {
144 return _success;
145 }
146
150 Index parti_level() const
151 {
152 return _ref_lvl;
153 }

161 Adjacency::Graph build_elems_at_rank() const
162 {
163 Adjacency::Graph graph(_num_ranks, _ref_elems, _ref_elems);
164 Index* ptr = graph.get_domain_ptr();
165 Index* idx = graph.get_image_idx();
166
167 // build pointer array
168 const Index elems_per_rank = _ref_elems / _num_ranks;
169 for(Index i(0); i <= _num_ranks; ++i)
170 ptr[i] = elems_per_rank * i;
171
172 // build index array
173 for(Index i(0); i < _ref_elems; ++i)
174 idx[i] = i;
175
176 return graph;
177 }
