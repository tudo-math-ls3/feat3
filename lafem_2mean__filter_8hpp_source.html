214 void filter_rhs(VectorType& vector) const
215 {
216 // subtract to dual integral mean zero
217 if(!_vec_prim.empty())

227 void filter_sol(VectorType& vector) const
228 {
229 // given: v=vec_prim, w=vec_dual, vol=v*w, input: x
230 // note that int(v) = v*w and int(x) = x*w
231 // we want:
232 // 1/vol * int (x + alpha*v) = sol_mean
233 // <==> int(x) + alpha*int(v) = sol_mean*vol
234 // <==> x*w + alpha*v*w = sol_mean*v*w
235 // <==> alpha*v*w = sol_mean*v*w - x*w
236 // <==> alpha = sol_mean - (x*w/v*w)
237 if(!_vec_prim.empty())
238 vector.axpy(_vec_prim, vector, _sol_mean - vector.dot(_vec_dual) / _volume);
239 }
