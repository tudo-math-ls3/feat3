231 template<typename DomainLevel_>
232 void assemble_gates(const Domain::VirtualLevel<DomainLevel_>& virt_dom_lvl)
233 {
234 const auto& dom_level = virt_dom_lvl.level();
235 const auto& dom_layer = virt_dom_lvl.layer();
236 const auto& space_velo = dom_level.space_velo;
237 const auto& space_pres = dom_level.space_pres;
238
239 // set the gate comm
240 this->gate_velo.set_comm(dom_layer.comm_ptr());
241 this->gate_pres.set_comm(dom_layer.comm_ptr());
242 this->gate_sys.set_comm(dom_layer.comm_ptr());
243
244 // loop over all ranks
245 for(Index i(0); i < dom_layer.neighbor_count(); ++i)
246 {
247 int rank = dom_layer.neighbor_rank(i);
248
249 // try to find our halo
250 auto* halo = dom_level.find_halo_part(rank);
251 XASSERT(halo != nullptr);
252
253 // create (empty) velocity mirror
254 VeloMirror mirror_velo;
255 Assembly::MirrorAssembler::assemble_mirror(mirror_velo, space_velo, *halo);
256
257 // create (empty) pressure mirror
258 PresMirror mirror_pres;
259 Assembly::MirrorAssembler::assemble_mirror(mirror_pres, space_pres, *halo);
260
261 // create a system mirror
262 SystemMirror mirror_sys(mirror_velo.clone(), mirror_pres.clone());
263
264 // push mirrors into gates
265 if(!mirror_velo.empty())
266 this->gate_velo.push(rank, std::move(mirror_velo));
267 if(!mirror_pres.empty())
268 this->gate_pres.push(rank, std::move(mirror_pres));
269 if(!mirror_sys.empty())
270 this->gate_sys.push(rank, std::move(mirror_sys));
271 }
272
273 // create local template vectors
274 LocalVeloVector tmpl_v(space_velo.get_num_dofs());
275 LocalPresVector tmpl_p(space_pres.get_num_dofs());
276 LocalSystemVector tmpl_s(tmpl_v.clone(), tmpl_p.clone());
277
278 // compile gates
279 this->gate_velo.compile(std::move(tmpl_v));
280 this->gate_pres.compile(std::move(tmpl_p));
281 this->gate_sys.compile(std::move(tmpl_s));
282 }

284 template<typename DomainLevel_>
285 void assemble_coarse_muxers(const Domain::VirtualLevel<DomainLevel_>& virt_lvl_coarse)
286 {
287 // assemble muxer parent
288 if(virt_lvl_coarse.is_parent())
289 {
290 XASSERT(virt_lvl_coarse.is_child());
291
292 const auto& layer_c = virt_lvl_coarse.layer_c();
293 const DomainLevel_& level_p = virt_lvl_coarse.level_p();
294
295 // loop over all children
296 for(Index i(0); i < layer_c.child_count(); ++i)
297 {
298 const auto* child = level_p.find_patch_part(int(i));
299 XASSERT(child != nullptr);
300 SystemMirror child_mirror_sys;
301 VeloMirror& child_mirror_v = child_mirror_sys.template at<0>();
302 PresMirror& child_mirror_p = child_mirror_sys.template at<1>();
303 Assembly::MirrorAssembler::assemble_mirror(child_mirror_v, level_p.space_velo, *child);
304 Assembly::MirrorAssembler::assemble_mirror(child_mirror_p, level_p.space_pres, *child);
305 this->coarse_muxer_velo.push_child(child_mirror_v.clone(LAFEM::CloneMode::Shallow));
306 this->coarse_muxer_pres.push_child(child_mirror_p.clone(LAFEM::CloneMode::Shallow));
307 this->coarse_muxer_sys.push_child(std::move(child_mirror_sys));
308 }
309 }
310
311 // assemble muxer child
312 if(virt_lvl_coarse.is_child())
313 {
314 const auto& layer_c = virt_lvl_coarse.layer_c();
315 const DomainLevel_& level_c = virt_lvl_coarse.level_c();
316
317 SystemMirror parent_mirror_sys;
318 VeloMirror& parent_mirror_v = parent_mirror_sys.template at<0>();
319 PresMirror& parent_mirror_p = parent_mirror_sys.template at<1>();
320
321 // manually set up an identity gather/scatter matrix
322 parent_mirror_v = VeloMirror::make_identity(level_c.space_velo.get_num_dofs());
323 parent_mirror_p = PresMirror::make_identity(level_c.space_pres.get_num_dofs());
324
325 // set parent and sibling comms
326 this->coarse_muxer_velo.set_parent(
327 layer_c.sibling_comm_ptr(),
328 layer_c.get_parent_rank(),
329 parent_mirror_v.clone(LAFEM::CloneMode::Shallow)
330 );
331 this->coarse_muxer_pres.set_parent(
332 layer_c.sibling_comm_ptr(),
333 layer_c.get_parent_rank(),
334 parent_mirror_p.clone(LAFEM::CloneMode::Shallow)
335 );
336 this->coarse_muxer_sys.set_parent(
337 layer_c.sibling_comm_ptr(),
338 layer_c.get_parent_rank(),
339 std::move(parent_mirror_sys)
340 );
341
342 // compile muxer
343 LocalVeloVector tmpl_v(level_c.space_velo.get_num_dofs());
344 LocalPresVector tmpl_p(level_c.space_pres.get_num_dofs());
345 LocalSystemVector tmpl_s(tmpl_v.clone(), tmpl_p.clone());
346 this->coarse_muxer_velo.compile(tmpl_v);
347 this->coarse_muxer_pres.compile(tmpl_p);
348 this->coarse_muxer_sys.compile(tmpl_s);
349 }

352 template<typename DomainLevel_>
353 void assemble_base_splitters(const Domain::VirtualLevel<DomainLevel_>& virt_lvl)
354 {
355 // get the layer
356 const auto& layer = virt_lvl.layer();
357
358 // nothing to assemble?
359 if(layer.comm().size() <= 1)
360 return;
361
362 // ensure that this virtual level has a base-mesh
363 XASSERTM(virt_lvl.has_base(), "cannot assemble base splitter because base level is missing");
364
365 // is this the root process?
366 if(layer.comm().rank() == 0)
367 {
368 const DomainLevel_& level_b = virt_lvl.level_b();
369
370 LocalVeloVector tmpl_v(level_b.space_velo.get_num_dofs());
371 LocalPresVector tmpl_p(level_b.space_pres.get_num_dofs());
372 LocalSystemVector tmpl_s(tmpl_v.clone(), tmpl_p.clone());
373
374 // set parent vector template
375 this->base_splitter_velo.set_base_vector_template(std::move(tmpl_v));
376 this->base_splitter_pres.set_base_vector_template(std::move(tmpl_p));
377 this->base_splitter_sys.set_base_vector_template(std::move(tmpl_s));
378
379 // assemble patch mirrors on root process
380 for(int i(0); i < layer.comm().size(); ++i)
381 {
382 const auto* patch = level_b.find_patch_part(i);
383 XASSERT(patch != nullptr);
384 SystemMirror patch_mirror;
385 VeloMirror& patch_mirror_v = patch_mirror.template at<0>();
386 PresMirror& patch_mirror_p = patch_mirror.template at<1>();
387 Assembly::MirrorAssembler::assemble_mirror(patch_mirror_v, level_b.space_velo, *patch);
388 Assembly::MirrorAssembler::assemble_mirror(patch_mirror_p, level_b.space_pres, *patch);
389 this->base_splitter_velo.push_patch(patch_mirror_v.clone(LAFEM::CloneMode::Shallow));
390 this->base_splitter_pres.push_patch(patch_mirror_p.clone(LAFEM::CloneMode::Shallow));
391 this->base_splitter_sys.push_patch(std::move(patch_mirror));
392 }
393 }
394
395 // assemble muxer child
396 const DomainLevel_& level = virt_lvl.level();
397
398 SystemMirror root_mirror_sys;
399 VeloMirror& root_mirror_v = root_mirror_sys.template at<0>();
400 PresMirror& root_mirror_p = root_mirror_sys.template at<1>();
401
402 // manually set up an identity gather/scatter matrix
403 root_mirror_v = VeloMirror::make_identity(level.space_velo.get_num_dofs());
404 root_mirror_p = PresMirror::make_identity(level.space_pres.get_num_dofs());
405
406 // set parent and sibling comms
407 this->base_splitter_velo.set_root(layer.comm_ptr(), 0, root_mirror_v.clone(LAFEM::CloneMode::Shallow));
408 this->base_splitter_pres.set_root(layer.comm_ptr(), 0, root_mirror_p.clone(LAFEM::CloneMode::Shallow));
409 this->base_splitter_sys.set_root(layer.comm_ptr(), 0, std::move(root_mirror_sys));
410
411 // compile muxer
412 LocalVeloVector tmpl_v(level.space_velo.get_num_dofs());
413 LocalPresVector tmpl_p(level.space_pres.get_num_dofs());
414 LocalSystemVector tmpl_s(tmpl_v.clone(), tmpl_p.clone());
415 this->base_splitter_velo.compile(tmpl_v);
416 this->base_splitter_pres.compile(tmpl_p);
417 this->base_splitter_sys.compile(tmpl_s);
418 }

420 template<typename DomainLevel_, typename Cubature_>
421 void assemble_velocity_transfer(
422 const Domain::VirtualLevel<DomainLevel_>& virt_lvl_fine,
423 const Domain::VirtualLevel<DomainLevel_>& virt_lvl_coarse,
424 const Cubature_& cubature)
425 {
426 // get fine and coarse domain levels
427 const DomainLevel_& level_f = *virt_lvl_fine;
428 const DomainLevel_& level_c = virt_lvl_coarse.is_child() ? virt_lvl_coarse.level_c() : *virt_lvl_coarse;
429
430 const auto& space_f = level_f.space_velo;
431 const auto& space_c = level_c.space_velo;
432
433 // get local transfer operator
434 LocalVeloTransfer& loc_trans = this->transfer_velo.local();
435
436 // get local transfer matrices
437 LocalVeloTransferMatrix& loc_prol_wrapped = loc_trans.get_mat_prol();
438 LocalVeloTransferMatrix& loc_rest_wrapped = loc_trans.get_mat_rest();
439
440 // get the unwrapped types
441 typename LocalVeloTransferMatrix::BaseClass& loc_prol = loc_prol_wrapped;
442 typename LocalVeloTransferMatrix::BaseClass& loc_rest = loc_rest_wrapped;
443
444 // assemble structure?
445 if (loc_prol.empty())
446 {
447 Assembly::SymbolicAssembler::assemble_matrix_2lvl(loc_prol, space_f, space_c);
448 }
449
450 // create a local weight vector
451 LocalVeloVector loc_vec_weight = loc_prol_wrapped.create_vector_l();
452
453 // create a scalar weight vector for the assembly
454 LocalScalarVector loc_scal_vec_weight = loc_prol.create_vector_l();
455
456 // get the data arrays of the weight vectors
457 auto* v_wb = loc_vec_weight.elements();
458 auto* v_ws = loc_scal_vec_weight.elements();
459
460 // assemble prolongation matrix
461 {
462 loc_prol.format();
463 loc_scal_vec_weight.format();
464
465 // assemble prolongation matrix
466 Assembly::GridTransfer::assemble_prolongation(loc_prol, loc_scal_vec_weight,
467 space_f, space_c, cubature);
468
469 // copy weights from scalar to blocked
470 for(Index i(0); i < loc_prol.rows(); ++i)
471 v_wb[i] = v_ws[i];
472
473 // synchronize blocked weight vector
474 this->gate_velo.sync_0(loc_vec_weight);
475
476 // copy weights from blocked to scalar
477 for(Index i(0); i < loc_prol.rows(); ++i)
478 v_ws[i] = v_wb[i][0];
479
480 // invert weight components
481 loc_scal_vec_weight.component_invert(loc_scal_vec_weight);
482
483 // scale prolongation matrix
484 loc_prol.scale_rows(loc_prol, loc_scal_vec_weight);
485
486 // copy and transpose
487 loc_rest = loc_prol.transpose();
488 }
489
490 // compile velocity transfer
491 this->transfer_velo.compile();
492 }

494 template<typename DomainLevel_, typename Cubature_>
495 void assemble_pressure_transfer(
496 const Domain::VirtualLevel<DomainLevel_>& virt_lvl_fine,
497 const Domain::VirtualLevel<DomainLevel_>& virt_lvl_coarse,
498 const Cubature_& cubature)
499 {
500 // get fine and coarse domain levels
501 const DomainLevel_& level_f = *virt_lvl_fine;
502 const DomainLevel_& level_c = virt_lvl_coarse.is_child() ? virt_lvl_coarse.level_c() : *virt_lvl_coarse;
503
504 const auto& space_f = level_f.space_pres;
505 const auto& space_c = level_c.space_pres;
506
507 // get local transfer operator
508 LocalPresTransfer& loc_trans = this->transfer_pres.local();
509
510 // get local transfer matrices
511 LocalPresTransferMatrix& loc_prol = loc_trans.get_mat_prol();
512 LocalPresTransferMatrix& loc_rest = loc_trans.get_mat_rest();
513
514 // assemble structure?
515 if (loc_prol.empty())
516 {
517 Assembly::SymbolicAssembler::assemble_matrix_2lvl(loc_prol, space_f, space_c);
518 }
519
520 // get local pressure weight vector
521 LocalPresVector loc_vec_weight = loc_prol.create_vector_l();
522
523 // assemble prolongation matrix
524 {
525 loc_prol.format();
526 loc_vec_weight.format();
527
528 // assemble prolongation matrix
529 Assembly::GridTransfer::assemble_prolongation(loc_prol, loc_vec_weight,
530 space_f, space_c, cubature);
531
532 // synchronize weight vector
533 this->gate_pres.sync_0(loc_vec_weight);
534
535 // invert components
536 loc_vec_weight.component_invert(loc_vec_weight);
537
538 // scale prolongation matrix
539 loc_prol.scale_rows(loc_prol, loc_vec_weight);
540
541 // copy and transpose
542 loc_rest = loc_prol.transpose();
543 }
544
545 // compile pressure transfer
546 this->transfer_pres.compile();
547 }

579 LocalVeloTransfer& loc_trans = this->transfer_velo.local();
580
581 // get local transfer matrices
582 const LocalVeloTransferMatrix& loc_rest_wrapped = loc_trans.get_mat_rest();
583 LocalVeloTransferMatrix& loc_trunc_wrapped = loc_trans.get_mat_trunc();
584
585 // get the matrix blocks
586 const typename LocalVeloTransferMatrix::BaseClass& loc_rest = loc_rest_wrapped;
587 typename LocalVeloTransferMatrix::BaseClass& loc_trunc = loc_trunc_wrapped;
588
589 // restriction matrix must be already assembled
590 XASSERTM(loc_rest.size() > Index(0), "you need to call 'assemble_velocity_transfer' first");
591
592 // clone sparsity pattern of restriction matrix
593 loc_trunc = loc_rest.clone(LAFEM::CloneMode::Layout);
594
595 // create local weight vector
596 LocalVeloVector loc_vec_weight = loc_rest_wrapped.create_vector_l();
597
598 // create a scalar weight vector for the assembly
599 LocalScalarVector loc_scal_vec_weight = loc_rest.create_vector_l();
600
601 // get the data arrays of the weight vectors
602 auto* v_wb = loc_vec_weight.elements();
603 auto* v_ws = loc_scal_vec_weight.elements();
604
605 // format
606 loc_trunc.format();
607 loc_scal_vec_weight.format();
608
609 // assemble truncation matrix
610 Assembly::GridTransfer::assemble_truncation(loc_trunc, loc_scal_vec_weight, space_f, space_c, cubature);
611
612 // copy weights from scalar to blocked
613 for(Index i(0); i < loc_trunc.rows(); ++i)
614 v_wb[i] = v_ws[i];
615
616 // We now need to synchronize the weight vector in analogy to the prolongation matrix assembly.
617 // Note that the weight vector is now a coarse-level vector, so we need to synchronize over
618 // the coarse-level gate. This may be a bit more complicated if the coarse level is a ghost
619 // level, as in this case we have to join/split around the synch operation.
620
621 // synchronize weight vector using the muxer/gate
622 if(!virt_lvl_coarse.is_child())
623 {
624 // The coarse level is a simple (non-child) level that exists on all processes,
625 // so simply synch over the coarse-level gate:
626 sys_lvl_coarse->gate_velo.sync_0(loc_vec_weight);
627 }
628 else if(virt_lvl_coarse.is_parent())
629 {
630 // The coarse level is a child level and this is one of the parent processes which contain
631 // the coarse-level gate. So we first need to join the weight vector onto the parent processes
632 // first, then synch that joined vector over the parent gate and finally split the result
633 // to all child processes -- this "emulates" a synch over the (non-existent) coarse-level
634 // child gate, which is what we actually require here...
635
636 // create temporary vector on parent partitioning
637 LocalVeloVector loc_tmp = sys_lvl_coarse->gate_velo._freqs.clone(LAFEM::CloneMode::Allocate);
638
639 // join child weights over muxer
640 this->coarse_muxer_velo.join(loc_vec_weight, loc_tmp);
641
642 // sync over coarse gate
643 sys_lvl_coarse->gate_velo.sync_0(loc_tmp);
644
645 // split over muxer
646 this->coarse_muxer_velo.split(loc_vec_weight, loc_tmp);
647 }
648 else // ghost
649 {
650 // The coarse level is a ghost level, i.e. a child but not a parent. In this case, we
651 // only have to participate in the join/send operations of the muxer, which are part
652 // of the operations executed on the parents handled by the else-if case above.
653
654 this->coarse_muxer_velo.join_send(loc_vec_weight);
655
656 // parent performs sync over its gate here (see above else-if)
657
658 this->coarse_muxer_velo.split_recv(loc_vec_weight);
659 }
660
661 // copy weights from blocked to scalar
662 for(Index i(0); i < loc_trunc.rows(); ++i)
663 v_ws[i] = v_wb[i][0];
664
665 // invert components
666 loc_scal_vec_weight.component_invert(loc_scal_vec_weight);
667
668 // scale reduction matrix
669 loc_trunc.scale_rows(loc_trunc, loc_scal_vec_weight);
670 }
