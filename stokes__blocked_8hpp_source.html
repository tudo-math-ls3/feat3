231 template<typename DomainLevel_>
232 void assemble_gates(const Domain::VirtualLevel<DomainLevel_>& virt_dom_lvl)
233 {
234 const auto& dom_level = virt_dom_lvl.level();
235 const auto& dom_layer = virt_dom_lvl.layer();
236 const auto& space_velo = dom_level.space_velo;
237 const auto& space_pres = dom_level.space_pres;
238
239 // set the gate comm
240 this->gate_velo.set_comm(dom_layer.comm_ptr());
241 this->gate_pres.set_comm(dom_layer.comm_ptr());
242 this->gate_sys.set_comm(dom_layer.comm_ptr());
243
244 // loop over all ranks
245 for(Index i(0); i < dom_layer.neighbor_count(); ++i)
246 {
247 int rank = dom_layer.neighbor_rank(i);
248
249 // try to find our halo
250 auto* halo = dom_level.find_halo_part(rank);
251 if(halo == nullptr) continue;
252 XASSERT(halo != nullptr);
253
254 // create (empty) velocity mirror
255 VeloMirror mirror_velo;
256 Assembly::MirrorAssembler::assemble_mirror(mirror_velo, space_velo, *halo);
257
258 // create (empty) pressure mirror
259 PresMirror mirror_pres;
260 Assembly::MirrorAssembler::assemble_mirror(mirror_pres, space_pres, *halo);
261
262 // create a system mirror
263 SystemMirror mirror_sys(mirror_velo.clone(), mirror_pres.clone());
264
265 // push mirrors into gates
266 if(!mirror_velo.empty())
267 this->gate_velo.push(rank, std::move(mirror_velo));
268 if(!mirror_pres.empty())
269 this->gate_pres.push(rank, std::move(mirror_pres));
270 if(!mirror_sys.empty())
271 this->gate_sys.push(rank, std::move(mirror_sys));
272 }
273
274 // create local template vectors
275 LocalVeloVector tmpl_v(space_velo.get_num_dofs());
276 LocalPresVector tmpl_p(space_pres.get_num_dofs());
277 LocalSystemVector tmpl_s(tmpl_v.clone(), tmpl_p.clone());
278
279 // compile gates
280 this->gate_velo.compile(std::move(tmpl_v));
281 this->gate_pres.compile(std::move(tmpl_p));
282 this->gate_sys.compile(std::move(tmpl_s));
283 }

285 template<typename DomainLevel_>
286 void assemble_coarse_muxers(const Domain::VirtualLevel<DomainLevel_>& virt_lvl_coarse)
287 {
288 // assemble muxer parent
289 if(virt_lvl_coarse.is_parent())
290 {
291 XASSERT(virt_lvl_coarse.is_child());
292
293 const auto& layer_c = virt_lvl_coarse.layer_c();
294 const DomainLevel_& level_p = virt_lvl_coarse.level_p();
295
296 // loop over all children
297 for(Index i(0); i < layer_c.child_count(); ++i)
298 {
299 const auto* child = level_p.find_patch_part(int(i));
300 XASSERT(child != nullptr);
301 SystemMirror child_mirror_sys;
302 VeloMirror& child_mirror_v = child_mirror_sys.template at<0>();
303 PresMirror& child_mirror_p = child_mirror_sys.template at<1>();
304 Assembly::MirrorAssembler::assemble_mirror(child_mirror_v, level_p.space_velo, *child);
305 Assembly::MirrorAssembler::assemble_mirror(child_mirror_p, level_p.space_pres, *child);
306 this->coarse_muxer_velo.push_child(child_mirror_v.clone(LAFEM::CloneMode::Shallow));
307 this->coarse_muxer_pres.push_child(child_mirror_p.clone(LAFEM::CloneMode::Shallow));
308 this->coarse_muxer_sys.push_child(std::move(child_mirror_sys));
309 }
310 }
311
312 // assemble muxer child
313 if(virt_lvl_coarse.is_child())
314 {
315 const auto& layer_c = virt_lvl_coarse.layer_c();
316 const DomainLevel_& level_c = virt_lvl_coarse.level_c();
317
318 SystemMirror parent_mirror_sys;
319 VeloMirror& parent_mirror_v = parent_mirror_sys.template at<0>();
320 PresMirror& parent_mirror_p = parent_mirror_sys.template at<1>();
321
322 // manually set up an identity gather/scatter matrix
323 parent_mirror_v = VeloMirror::make_identity(level_c.space_velo.get_num_dofs());
324 parent_mirror_p = PresMirror::make_identity(level_c.space_pres.get_num_dofs());
325
326 // set parent and sibling comms
327 this->coarse_muxer_velo.set_parent(
328 layer_c.sibling_comm_ptr(),
329 layer_c.get_parent_rank(),
330 parent_mirror_v.clone(LAFEM::CloneMode::Shallow)
331 );
332 this->coarse_muxer_pres.set_parent(
333 layer_c.sibling_comm_ptr(),
334 layer_c.get_parent_rank(),
335 parent_mirror_p.clone(LAFEM::CloneMode::Shallow)
336 );
337 this->coarse_muxer_sys.set_parent(
338 layer_c.sibling_comm_ptr(),
339 layer_c.get_parent_rank(),
340 std::move(parent_mirror_sys)
341 );
342
343 // compile muxer
344 LocalVeloVector tmpl_v(level_c.space_velo.get_num_dofs());
345 LocalPresVector tmpl_p(level_c.space_pres.get_num_dofs());
346 LocalSystemVector tmpl_s(tmpl_v.clone(), tmpl_p.clone());
347 this->coarse_muxer_velo.compile(tmpl_v);
348 this->coarse_muxer_pres.compile(tmpl_p);
349 this->coarse_muxer_sys.compile(tmpl_s);
350 }

353 template<typename DomainLevel_>
354 void assemble_base_splitters(const Domain::VirtualLevel<DomainLevel_>& virt_lvl)
355 {
356 // get the layer
357 const auto& layer = virt_lvl.layer();
358
359 // nothing to assemble?
360 if(layer.comm().size() <= 1)
361 return;
362
363 // ensure that this virtual level has a base-mesh
364 XASSERTM(virt_lvl.has_base(), "cannot assemble base splitter because base level is missing");
365
366 // is this the root process?
367 if(layer.comm().rank() == 0)
368 {
369 const DomainLevel_& level_b = virt_lvl.level_b();
370
371 LocalVeloVector tmpl_v(level_b.space_velo.get_num_dofs());
372 LocalPresVector tmpl_p(level_b.space_pres.get_num_dofs());
373 LocalSystemVector tmpl_s(tmpl_v.clone(), tmpl_p.clone());
374
375 // set parent vector template
376 this->base_splitter_velo.set_base_vector_template(std::move(tmpl_v));
377 this->base_splitter_pres.set_base_vector_template(std::move(tmpl_p));
378 this->base_splitter_sys.set_base_vector_template(std::move(tmpl_s));
379
380 // assemble patch mirrors on root process
381 for(int i(0); i < layer.comm().size(); ++i)
382 {
383 const auto* patch = level_b.find_patch_part(i);
384 XASSERT(patch != nullptr);
385 SystemMirror patch_mirror;
386 VeloMirror& patch_mirror_v = patch_mirror.template at<0>();
387 PresMirror& patch_mirror_p = patch_mirror.template at<1>();
388 Assembly::MirrorAssembler::assemble_mirror(patch_mirror_v, level_b.space_velo, *patch);
389 Assembly::MirrorAssembler::assemble_mirror(patch_mirror_p, level_b.space_pres, *patch);
390 this->base_splitter_velo.push_patch(patch_mirror_v.clone(LAFEM::CloneMode::Shallow));
391 this->base_splitter_pres.push_patch(patch_mirror_p.clone(LAFEM::CloneMode::Shallow));
392 this->base_splitter_sys.push_patch(std::move(patch_mirror));
393 }
394 }
395
396 // assemble muxer child
397 const DomainLevel_& level = virt_lvl.level();
398
399 SystemMirror root_mirror_sys;
400 VeloMirror& root_mirror_v = root_mirror_sys.template at<0>();
401 PresMirror& root_mirror_p = root_mirror_sys.template at<1>();
402
403 // manually set up an identity gather/scatter matrix
404 root_mirror_v = VeloMirror::make_identity(level.space_velo.get_num_dofs());
405 root_mirror_p = PresMirror::make_identity(level.space_pres.get_num_dofs());
406
407 // set parent and sibling comms
408 this->base_splitter_velo.set_root(layer.comm_ptr(), 0, root_mirror_v.clone(LAFEM::CloneMode::Shallow));
409 this->base_splitter_pres.set_root(layer.comm_ptr(), 0, root_mirror_p.clone(LAFEM::CloneMode::Shallow));
410 this->base_splitter_sys.set_root(layer.comm_ptr(), 0, std::move(root_mirror_sys));
411
412 // compile muxer
413 LocalVeloVector tmpl_v(level.space_velo.get_num_dofs());
414 LocalPresVector tmpl_p(level.space_pres.get_num_dofs());
415 LocalSystemVector tmpl_s(tmpl_v.clone(), tmpl_p.clone());
416 this->base_splitter_velo.compile(tmpl_v);
417 this->base_splitter_pres.compile(tmpl_p);
418 this->base_splitter_sys.compile(tmpl_s);
419 }

421 template<typename DomainLevel_, typename Cubature_>
422 void assemble_velocity_transfer(
423 const Domain::VirtualLevel<DomainLevel_>& virt_lvl_fine,
424 const Domain::VirtualLevel<DomainLevel_>& virt_lvl_coarse,
425 const Cubature_& cubature)
426 {
427 // get fine and coarse domain levels
428 const DomainLevel_& level_f = *virt_lvl_fine;
429 const DomainLevel_& level_c = virt_lvl_coarse.is_child() ? virt_lvl_coarse.level_c() : *virt_lvl_coarse;
430
431 const auto& space_f = level_f.space_velo;
432 const auto& space_c = level_c.space_velo;
433
434 // get local transfer operator
435 LocalVeloTransfer& loc_trans = this->transfer_velo.local();
436
437 // get local transfer matrices
438 LocalVeloTransferMatrix& loc_prol_wrapped = loc_trans.get_mat_prol();
439 LocalVeloTransferMatrix& loc_rest_wrapped = loc_trans.get_mat_rest();
440
441 // get the unwrapped types
442 typename LocalVeloTransferMatrix::BaseClass& loc_prol = loc_prol_wrapped;
443 typename LocalVeloTransferMatrix::BaseClass& loc_rest = loc_rest_wrapped;
444
445 // assemble structure?
446 if (loc_prol.empty())
447 {
448 Assembly::SymbolicAssembler::assemble_matrix_2lvl(loc_prol, space_f, space_c);
449 }
450
451 // create a local weight vector
452 LocalVeloVector loc_vec_weight = loc_prol_wrapped.create_vector_l();
453
454 // create a scalar weight vector for the assembly
455 LocalScalarVector loc_scal_vec_weight = loc_prol.create_vector_l();
456
457 // get the data arrays of the weight vectors
458 auto* v_wb = loc_vec_weight.elements();
459 auto* v_ws = loc_scal_vec_weight.elements();
460
461 // assemble prolongation matrix
462 {
463 loc_prol.format();
464 loc_scal_vec_weight.format();
465
466 // assemble prolongation matrix
467 Assembly::GridTransfer::assemble_prolongation(loc_prol, loc_scal_vec_weight,
468 space_f, space_c, cubature);
469
470 // copy weights from scalar to blocked
471 for(Index i(0); i < loc_prol.rows(); ++i)
472 v_wb[i] = v_ws[i];
473
474 // synchronize blocked weight vector
475 this->gate_velo.sync_0(loc_vec_weight);
476
477 // copy weights from blocked to scalar
478 for(Index i(0); i < loc_prol.rows(); ++i)
479 v_ws[i] = v_wb[i][0];
480
481 // invert weight components
482 loc_scal_vec_weight.component_invert(loc_scal_vec_weight);
483
484 // scale prolongation matrix
485 loc_prol.scale_rows(loc_prol, loc_scal_vec_weight);
486
487 // copy and transpose
488 loc_rest = loc_prol.transpose();
489 }
490
491 // compile velocity transfer
492 this->transfer_velo.compile();
493 }

495 template<typename DomainLevel_, typename Cubature_>
496 void assemble_pressure_transfer(
497 const Domain::VirtualLevel<DomainLevel_>& virt_lvl_fine,
498 const Domain::VirtualLevel<DomainLevel_>& virt_lvl_coarse,
499 const Cubature_& cubature)
500 {
501 // get fine and coarse domain levels
502 const DomainLevel_& level_f = *virt_lvl_fine;
503 const DomainLevel_& level_c = virt_lvl_coarse.is_child() ? virt_lvl_coarse.level_c() : *virt_lvl_coarse;
504
505 const auto& space_f = level_f.space_pres;
506 const auto& space_c = level_c.space_pres;
507
508 // get local transfer operator
509 LocalPresTransfer& loc_trans = this->transfer_pres.local();
510
511 // get local transfer matrices
512 LocalPresTransferMatrix& loc_prol = loc_trans.get_mat_prol();
513 LocalPresTransferMatrix& loc_rest = loc_trans.get_mat_rest();
514
515 // assemble structure?
516 if (loc_prol.empty())
517 {
518 Assembly::SymbolicAssembler::assemble_matrix_2lvl(loc_prol, space_f, space_c);
519 }
520
521 // get local pressure weight vector
522 LocalPresVector loc_vec_weight = loc_prol.create_vector_l();
523
524 // assemble prolongation matrix
525 {
526 loc_prol.format();
527 loc_vec_weight.format();
528
529 // assemble prolongation matrix
530 Assembly::GridTransfer::assemble_prolongation(loc_prol, loc_vec_weight,
531 space_f, space_c, cubature);
532
533 // synchronize weight vector
534 this->gate_pres.sync_0(loc_vec_weight);
535
536 // invert components
537 loc_vec_weight.component_invert(loc_vec_weight);
538
539 // scale prolongation matrix
540 loc_prol.scale_rows(loc_prol, loc_vec_weight);
541
542 // copy and transpose
543 loc_rest = loc_prol.transpose();
544 }
545
546 // compile pressure transfer
547 this->transfer_pres.compile();
548 }

579 // get local transfer operator
580 LocalVeloTransfer& loc_trans = this->transfer_velo.local();
581
582 // get local transfer matrices
583 const LocalVeloTransferMatrix& loc_rest_wrapped = loc_trans.get_mat_rest();
584 LocalVeloTransferMatrix& loc_trunc_wrapped = loc_trans.get_mat_trunc();
585
586 // get the matrix blocks
587 const typename LocalVeloTransferMatrix::BaseClass& loc_rest = loc_rest_wrapped;
588 typename LocalVeloTransferMatrix::BaseClass& loc_trunc = loc_trunc_wrapped;
589
590 // restriction matrix must be already assembled
591 XASSERTM(loc_rest.size() > Index(0), "you need to call 'assemble_velocity_transfer' first");
592
593 // clone sparsity pattern of restriction matrix
594 loc_trunc = loc_rest.clone(LAFEM::CloneMode::Layout);
595
596 // create local weight vector
597 LocalVeloVector loc_vec_weight = loc_rest_wrapped.create_vector_l();
598
599 // create a scalar weight vector for the assembly
600 LocalScalarVector loc_scal_vec_weight = loc_rest.create_vector_l();
601
602 // get the data arrays of the weight vectors
603 auto* v_wb = loc_vec_weight.elements();
604 auto* v_ws = loc_scal_vec_weight.elements();
605
606 // format
607 loc_trunc.format();
608 loc_scal_vec_weight.format();
609
610 // assemble truncation matrix
611 Assembly::GridTransfer::assemble_truncation(loc_trunc, loc_scal_vec_weight, space_f, space_c, cubature);
612
613 // copy weights from scalar to blocked
614 for(Index i(0); i < loc_trunc.rows(); ++i)
615 v_wb[i] = v_ws[i];
616
617 // We now need to synchronize the weight vector in analogy to the prolongation matrix assembly.
618 // Note that the weight vector is now a coarse-level vector, so we need to synchronize over
619 // the coarse-level gate. This may be a bit more complicated if the coarse level is a ghost
620 // level, as in this case we have to join/split around the synch operation.
621
622 // synchronize weight vector using the muxer/gate
623 if(!virt_lvl_coarse.is_child())
624 {
625 // The coarse level is a simple (non-child) level that exists on all processes,
626 // so simply synch over the coarse-level gate:
627 sys_lvl_coarse->gate_velo.sync_0(loc_vec_weight);
628 }
629 else if(virt_lvl_coarse.is_parent())
630 {
631 // The coarse level is a child level and this is one of the parent processes which contain
632 // the coarse-level gate. So we first need to join the weight vector onto the parent processes
633 // first, then synch that joined vector over the parent gate and finally split the result
634 // to all child processes -- this "emulates" a synch over the (non-existent) coarse-level
635 // child gate, which is what we actually require here...
636
637 // create temporary vector on parent partitioning
638 LocalVeloVector loc_tmp = sys_lvl_coarse->gate_velo._freqs.clone(LAFEM::CloneMode::Allocate);
639
640 // join child weights over muxer
641 this->coarse_muxer_velo.join(loc_vec_weight, loc_tmp);
642
643 // sync over coarse gate
644 sys_lvl_coarse->gate_velo.sync_0(loc_tmp);
645
646 // split over muxer
647 this->coarse_muxer_velo.split(loc_vec_weight, loc_tmp);
648 }
649 else // ghost
650 {
651 // The coarse level is a ghost level, i.e. a child but not a parent. In this case, we
652 // only have to participate in the join/send operations of the muxer, which are part
653 // of the operations executed on the parents handled by the else-if case above.
654
655 this->coarse_muxer_velo.join_send(loc_vec_weight);
656
657 // parent performs sync over its gate here (see above else-if)
658
659 this->coarse_muxer_velo.split_recv(loc_vec_weight);
660 }
661
662 // copy weights from blocked to scalar
663 for(Index i(0); i < loc_trunc.rows(); ++i)
664 v_ws[i] = v_wb[i][0];
665
666 // invert components
667 loc_scal_vec_weight.component_invert(loc_scal_vec_weight);
668
669 // scale reduction matrix
670 loc_trunc.scale_rows(loc_trunc, loc_scal_vec_weight);
671 }
