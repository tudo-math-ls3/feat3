250 virtual Status _apply_intern(VectorType& vec_x)
251 {
252 Statistics::add_solver_expression(std::make_shared<ExpressionStartSolve>(this->name()));
253
254 const MatrixType& matrix(this->_system_matrix);
255 const MatrixType& transp(this->_transp_matrix);
256 const FilterType& filter(this->_system_filter);
257
258 VectorType& vec_p(this->_vec_p);
259 VectorType& vec_q(this->_vec_q);
260 VectorType& vec_r(this->_vec_r);
261 VectorType& vec_s(this->_vec_s);
262 VectorType& vec_t(this->_vec_t);
263 VectorType& vec_y(this->_vec_s); // same as s
264 VectorType& vec_z(this->_vec_t); // same as t
265
266 // compute initial defect:
267 // r[0] := b - A*x[0]
268 Status status = this->_set_initial_defect(vec_r, vec_x);
269 if(status != Status::progress)
270 {
271 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), status, this->get_num_iter()));
272 return status;
273 }
274
275 // apply left preconditioner to defect vector
276 // p[0] := (L*L^T)^{-1} r[0]
277 this->_precond_l(vec_p, vec_r);
278 filter.filter_cor(vec_p);
279
280 // apply transposed matrix
281 // s[0] := A^T * p[0]
282 transp.apply(vec_s, vec_p);
283 filter.filter_def(vec_s);
284
285 // apply right preconditioner
286 // q[0] := (R^T*R)^{-1} * s[0]
287 this->_precond_r(vec_q, vec_s);
288 filter.filter_cor(vec_q);
289
290 // compute initial gamma:
291 // gamma[0] := < s[0], q[0] >
292 DataType gamma = vec_s.dot(vec_q);
293
294 // start iterating
295 while(status == Status::progress)
296 {
297 IterationStats stat(*this);
298
299 // y[k] := A * q[k]
300 matrix.apply(vec_y, vec_q);
301 filter.filter_def(vec_y);
302
303 // z[k] := (L*L^T)^{-1} * y[k]
304 this->_precond_l(vec_z, vec_y);
305 filter.filter_cor(vec_z);
306
307 // compute alpha
308 // alpha[k] := gamma[k] / < y[k], z[k] >
309 DataType alpha = gamma / vec_y.dot(vec_z);
310
311 // update solution vector
312 // x[k+1] := x[k] + alpha[k] * q[k]
313 vec_x.axpy(vec_q, vec_x, alpha);
314
315 // update defect vector
316 // r[k+1] := r[k] - alpha[k] * y[k]
317 vec_r.axpy(vec_y, vec_r, -alpha);
318
319 // compute defect norm
320 status = this->_set_new_defect(vec_r, vec_x);
321 if(status != Status::progress)
322 {
323 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), status, this->get_num_iter()));
324 return status;
325 }
326
327 // p[k+1] := p[k] - alpha[k] * z[k]
328 vec_p.axpy(vec_z, vec_p, -alpha);
329
330 // s[k+1] := A^T * p[k+1]
331 transp.apply(vec_s, vec_p);
332 filter.filter_def(vec_s);
333
334 // t[k+1] := (R^T*R)^{-1} * s[k+1]
335 this->_precond_r(vec_t, vec_s);
336 filter.filter_cor(vec_t);
337
338 // compute new gamma
339 // gamma[k+1] := < s[k+1], t[k+1] >
340 DataType gamma2 = gamma;
341 gamma = vec_s.dot(vec_t);
342
343 // compute beta
344 // beta[k] := gamma[k+1] / gamma[k]
345 DataType beta = gamma / gamma2;
346
347 // update direction vector
348 // q[k+1] := t[k+1] + beta * q[k]
349 vec_q.axpy(vec_q, vec_t, beta);
350 }
351
352 // we should never reach this point...
353 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::undefined, this->get_num_iter()));
354 return Status::undefined;
355 }
