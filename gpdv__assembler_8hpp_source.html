91 template<
92 typename DataType_, typename IndexType_, int dim_,
93 typename SpaceVelo_, typename SpacePres_>
94 static void assemble(
95 LAFEM::SparseMatrixBCSR<DataType_, IndexType_, dim_, 1>& matrix_b,
96 LAFEM::SparseMatrixBCSR<DataType_, IndexType_, 1, dim_>& matrix_d,
97 const SpaceVelo_& space_velo,
98 const SpacePres_& space_pres,
99 const Cubature::DynamicFactory& cubature_factory,
100 const DataType_ scale_b = -DataType_(1),
101 const DataType_ scale_d = -DataType_(1)
102 )
103 {
104 typedef DataType_ DataType;
105
106 // ensure that the matrices have the correct dimensions
107 static_assert(SpaceVelo_::shape_dim == dim_, "invalid matrix block sizes");
108
109 // make sure that velocity and pressure have the same trafo
110 XASSERTM((&space_velo.get_trafo()) == (&space_pres.get_trafo()),
111 "Velocity and Pressure spaces must be defined on the same trafo!");
112
113 typedef LAFEM::SparseMatrixBCSR<DataType_, IndexType_, dim_, 1> MatrixB;
114 typedef LAFEM::SparseMatrixBCSR<DataType_, IndexType_, 1, dim_> MatrixD;
115
116 // assembly traits
117 typedef AsmTraits2<
118 DataType,
119 SpaceVelo_,
120 SpacePres_,
121 TrafoTags::jac_det,
122 SpaceTags::grad,
123 SpaceTags::value> AsmTraits;
124
125 // fetch the trafo
126 const typename AsmTraits::TrafoType& trafo = space_velo.get_trafo();
127
128 // create a trafo evaluator
129 typename AsmTraits::TrafoEvaluator trafo_eval(trafo);
130
131 // create space evaluators
132 typename AsmTraits::TestEvaluator velo_eval(space_velo);
133 typename AsmTraits::TrialEvaluator pres_eval(space_pres);
134
135 // create dof-mappings
136 typename AsmTraits::TestDofMapping velo_dof_mapping(space_velo);
137 typename AsmTraits::TrialDofMapping pres_dof_mapping(space_pres);
138
139 // create trafo evaluation data
140 typename AsmTraits::TrafoEvalData trafo_data;
141
142 // create space evaluation data
143 typename AsmTraits::TestEvalData velo_data;
144 typename AsmTraits::TrialEvalData pres_data;
145
146 // maximum number of dofs
147 static constexpr int max_velo_dofs = AsmTraits::max_local_test_dofs;
148 static constexpr int max_pres_dofs = AsmTraits::max_local_trial_dofs;
149
150 // create local matrix data
151 //typename AsmTraits::LocalMatrixType lmd;
152 typedef Tiny::Matrix<DataType, dim_, 1> BEntryType;
153 typedef Tiny::Matrix<DataType, 1, dim_> DEntryType;
154 Tiny::Matrix<BEntryType, max_velo_dofs, max_pres_dofs> local_b;
155 Tiny::Matrix<DEntryType, max_pres_dofs, max_velo_dofs> local_d;
156
157 // create cubature rule
158 typename AsmTraits::CubatureRuleType cubature_rule(Cubature::ctor_factory, cubature_factory);
159
160 // first of all, check whether we need to assemble the matrix structures
161 if(matrix_b.empty())
162 {
163 Adjacency::Graph graph_b(SymbolicAssembler::assemble_graph_std2(space_velo, space_pres));
164 matrix_b = MatrixB(graph_b);
165 }
166 matrix_b.format();
167 if(matrix_d.empty())
168 {
169 Adjacency::Graph graph_d(SymbolicAssembler::assemble_graph_std2(space_pres, space_velo));
170 matrix_d = MatrixD(graph_d);
171 matrix_d.format();
172 }
173 matrix_d.format();
174
175 // create matrix scatter-axpy
176 typename MatrixB::ScatterAxpy scatter_b(matrix_b);
177 typename MatrixD::ScatterAxpy scatter_d(matrix_d);
178
179 // loop over all cells of the mesh
180 for(typename AsmTraits::CellIterator cell(trafo_eval.begin()); cell != trafo_eval.end(); ++cell)
181 {
182 // prepare trafo evaluator
183 trafo_eval.prepare(cell);
184
185 // prepare space evaluators
186 velo_eval.prepare(trafo_eval);
187 pres_eval.prepare(trafo_eval);
188
189 // fetch number of local dofs
190 const int num_loc_velo_dofs = velo_eval.get_num_local_dofs();
191 const int num_loc_pres_dofs = pres_eval.get_num_local_dofs();
192
193 // format local matrix
194 local_b.format();
195 local_d.format();
196
197 // loop over all quadrature points and integrate
198 for(int pt(0); pt < cubature_rule.get_num_points(); ++pt)
199 {
200 // compute trafo data
201 trafo_eval(trafo_data, cubature_rule.get_point(pt));
202
203 // compute basis function data
204 velo_eval(velo_data, trafo_data);
205 pres_eval(pres_data, trafo_data);
206
207 // test function loop
208 for(int i(0); i < num_loc_velo_dofs; ++i)
209 {
210 // trial function loop
211 for(int j(0); j < num_loc_pres_dofs; ++j)
212 {
213 const DataType pv = trafo_data.jac_det * cubature_rule.get_weight(pt) * pres_data.phi[j].value;
214
215 // dimension loop
216 for(int k(0); k < dim_; ++k)
217 {
218 local_b[i][j][k][0] += velo_data.phi[i].grad[k] * pv;
219 local_d[j][i][0][k] += velo_data.phi[i].grad[k] * pv;
220 }
221
222 // continue with next trial function
223 }
224 // continue with next test function
225 }
226 // continue with next cubature point
227 }
228
229 // finish evaluators
230 pres_eval.finish();
231 velo_eval.finish();
232 trafo_eval.finish();
233
234 // initialize dof-mappings
235 velo_dof_mapping.prepare(cell);
236 pres_dof_mapping.prepare(cell);
237
238 // incorporate local matrix
239 scatter_b(local_b, velo_dof_mapping, pres_dof_mapping, scale_b);
240 scatter_d(local_d, pres_dof_mapping, velo_dof_mapping, scale_d);
241
242 // finish dof mapping
243 pres_dof_mapping.finish();
244 velo_dof_mapping.finish();
245
246 // continue with next cell
247 }
248 }
249 };
