197 virtual Status _apply_intern(VectorType& vec_sol, const VectorType& vec_dir)
198 {
199 Statistics::add_solver_expression(std::make_shared<ExpressionStartSolve>(this->name()));
200
201 // The step length wrt. to the NORMALISED search direction
202 DataType alpha(0);
203 // The functional value
204 DataType fval(0);
206 DataType df(0);
207
208 // Perform initializations and checks
209 Status st = this->_startup(alpha, fval, df, vec_sol, vec_dir);
210 // Use the additional information about the preconditioned search direction's length?
211 if(this->_dir_scaling)
212 {
213 alpha *= this->_norm_dir;
214 }
215
216 DataType alpha_update(alpha);
217
218 // The second secant point in the first iteration is x + _secant_step * vec_dir
219 // The first "other" point for the secant
220 vec_sol.axpy(this->_vec_pn, this->_vec_initial_sol, alpha);
221
222 // start iterating
223 while(st == Status::progress)
224 {
225 IterationStats stat(*this);
226
227 // Increase iteration count
228 ++this->_num_iter;
229
230 this->_functional.prepare(vec_sol, this->_filter);
231 this->_functional.eval_fval_grad(fval, this->_vec_grad);
232 // DO NOT call trim() here, as this simple linesearch accepts bad steps too, which would then be trimmed
233 // and used.
234 //this->trim_func_grad(fval);
235 this->_filter.filter_def(this->_vec_grad);
236
237 if(fval < this->_fval_min)
238 {
239 this->_fval_min = fval;
240 this->_alpha_min = alpha;
241 }
242
243 // Set new defect, do convergence checks and compute the new _alpha
244 DataType df_prev = df;
245 df = this->_vec_pn.dot(this->_vec_grad);
246
247 st = this->_check_convergence(fval, df, alpha);
248
249 // Check if the diffence in etas is too small, thus leading to a huge update of relative size > sqrt(eps)
250 if(Math::abs(df - df_prev) < Math::abs(alpha_update*df)*Math::sqrt(Math::eps<DataType>()))
251 {
252 st = Status::stagnated;
253 }
254
255 if(st != Status::progress)
256 {
257 break;
258 }
259
260 // Update alpha according to secant formula
261 alpha_update *= df/(df_prev - df);
262 alpha += alpha_update;
263
264 // Update the solution
265 vec_sol.axpy(this->_vec_pn, this->_vec_initial_sol, alpha);
266 }
267
268 // If we are successful, we could save the last step length as the new initial step length. This is
269 // disabled by default, as it can lead to stagnation e.g. for minimising the Rosenbrock function.
270 //if(st == Status::success)
271 //{
272 // this->_alpha_0 = this->_alpha_min/this->_norm_dir;
273 //}
274 // If we are not successful, we update the best step length and need to re-evaluate everything for that
275 // step
276 if(st != Status::success)
277 {
278 vec_sol.axpy(this->_vec_pn, this->_vec_initial_sol, this->_alpha_min);
279
280 // Prepare and evaluate
281 this->_functional.prepare(vec_sol, this->_filter);
282 this->_functional.eval_fval_grad(fval, this->_vec_grad);
283 // DO NOT call trim() here, as this simple linesearch accepts bad steps too, which would then be trimmed
284 // and used.
285 //this->trim_func_grad(fval);
286 this->_filter.filter_def(this->_vec_grad);
287 }
288 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::undefined, this->get_num_iter()));
289 return st;
290 }
