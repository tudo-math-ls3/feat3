381 template<typename... Prms_>
382 int parse(const String& option, Prms_&&... prms) const
383 {
384 // try to find the option
385 auto it = _opts.find(option);
386 if(it == _opts.end())
387 return 0;
388
389 // try to parse arguments
390 return _parse(it->second, std::size_t(0), std::forward<Prms_>(prms)...);
391 }
392
393 private:
395 void _process()
396 {
397 // get an option iterator
398 auto opt = _opts.end();
399
400 // we definitely skip the first argument
401 _num_skipped_args = 1;
402
403 // loop over all arguments
404 for(int i(1); i < int(_args.size()); ++i)
405 {
406 // get the trimmed argument
407 String arg = _args.at(std::size_t(i)).trim();
408
409 // does it start with '--' ?
410 if((arg.size() > std::size_t(1)) && (arg[0] == '-') && (arg[1] == '-'))
411 {
412 // it's an option
413 String optname = arg.substr(2).trim();
414
415 // let's insert it
416 auto ik = _opts.insert(std::make_pair(optname, std::make_pair(i, std::deque<String>())));
417 opt = ik.first;
418
419 // clear existing deque
420 if(!ik.second)
421 {
422 // option was already given, so overwrite it
423 opt->second.first = i;
424 opt->second.second.clear();
425 }
426 }
427 else if(opt != _opts.end())
428 {
429 // push this argument as a parameter for the last processed option
430 opt->second.second.push_back(arg);
431 }
432 else
433 {
434 // We did not process any options yet, so we skip this argument
435 _num_skipped_args = i + 1;
436 }
437 }
438 }

440 template<typename Prm_>
441 int _parse_prm(const std::pair<int,std::deque<String> >& isd, std::size_t offset, Prm_&& prm) const
442 {
443 // check whether we have another parameter
444 if(offset >= isd.second.size())
445 return 0;
446
447 // try to parse the argument
448 if(isd.second.at(offset).parse(std::forward<Prm_>(prm)))
449 return +1; // success
450 else
451 return -isd.first - int(offset) - 1; // fail; return negative argument index
452 }

454 // overload for StringMapped
455 template<typename ValueType_>
456 int _parse_prm(const std::pair<int,std::deque<String> >& isd, std::size_t offset, StringMapped<ValueType_>&& prm) const
457 {
458 // check whether we have another parameter
459 if(offset >= isd.second.size())
460 return 0;
461
462 // try to parse the argument
463 if(prm.lookup(isd.second.at(offset)))
464 return +1; // success
465 else
466 return -isd.first - int(offset) - 1; // fail; return negative argument index
467 }

469 template<typename Prm1_>
470 int _parse(const std::pair<int,std::deque<String> >& isd, std::size_t offset, Prm1_&& prm1) const
471 {
472 int rtn = _parse_prm(isd, offset, std::forward<Prm1_>(prm1));
473 if(rtn > 0)
474 {
475 // parse successful; no more arguments to be parsed
476 return int(offset) + 1;
477 }
478 else if(rtn == 0)
479 {
480 // parameter missing
481 return int(offset);
482 }
483 else
484 {
485 // failed to parse
486 return rtn;
487 }
488 }

490 template<typename Prm1_, typename... Prms_>
491 int _parse(const std::pair<int,std::deque<String> >& isd, std::size_t offset, Prm1_&& prm1, Prms_&&... prms) const
492 {
493 int rtn = _parse_prm(isd, offset, std::forward<Prm1_>(prm1));
494 if(rtn > 0)
495 {
496 // parse successful; continue parsing remaining parameters
497 return _parse(isd, ++offset, std::forward<Prms_>(prms)...);
498 }
499 else if(rtn == 0)
500 {
501 // no more parameters, quit here
502 return int(offset);
503 }
504 else
505 {
506 // failed to parse
507 return rtn;
508 }
