56 template<typename Shape_, typename DataType_>
57 static void write(
58 std::ostream& os,
59 const ConformalMesh<Shape_, 2, DataType_>& mesh,
60 const double width = 100.0,
61 const double height = 100.0,
62 const double stroke = 0.1,
63 const double extra = 0.0)
64 {
65 XASSERTM(width >= 1.0, "invalid bounding box width");
66 XASSERTM(height >= 1.0, "invalid bounding box height");
67 XASSERTM(stroke > 0.01, "invalid stroke width");
68 XASSERTM(extra >= 0.0, "invalid bounding box extra");
69
70 // get vertices-at-edge
71 const auto& vtx = mesh.get_vertex_set();
72 const auto& idx = mesh.template get_index_set<1,0>();
73
74 // compute minimal and maximal x/y coords
75 DataType_ x_min, x_max, y_min, y_max;
76 x_min = x_max = vtx[0][0];
77 y_min = y_max = vtx[0][1];
78 for(Index i(1); i < vtx.get_num_vertices(); ++i)
79 {
80 const auto& v = vtx[i];
81 Math::minimax(v[0], x_min, x_max);
82 Math::minimax(v[1], y_min, y_max);
83 }
84
85 // compute x/y scale:
86 // Note: PostScript units are "points"
87 // 1 inch = 25.4 millimeters
88 // 1 inch = 72 points
89 // ==> 1 millimeter = 72 / 25.4 points
90 const double x_sc = (72.0 / 25.4) * width / double(x_max - x_min);
91 const double y_sc = (72.0 / 25.4) * height / double(y_max - y_min);
92
93 // compute scaling factor
94 const double scale = Math::min(x_sc, y_sc);
95
96 // compute bounding box (rounding up)
97 const int box_x = int(scale * double(x_max - x_min) + 2.0*extra) + 1;
98 const int box_y = int(scale * double(y_max - y_min) + 2.0*extra) + 1;
99
100 // compute offset from extra
101 const double box_o = 72.0 * extra / 25.4;
102
103 // write EPS header
104 os << "%!!PS-Adobe-3.0 EPSF-3.0" << std::endl;
105
106 // write bounding box
107 os << "%%BoundingBox: 0 0 " << box_x << " " << box_y << std::endl;
108
109 // set stroke width
110 os << (72.0 * stroke / 25.4) << " setlinewidth" << std::endl;
111
112 // begin a new path
113 os << "newpath" << std::endl;
114
115 // write all edges
116 for(Index i(0); i < idx.get_num_entities(); ++i)
117 {
118 // get the vertices
119 const auto& v0 = vtx[idx[i][0]];
120 const auto& v1 = vtx[idx[i][1]];
121
122 // transform vertices
123 double v0x = box_o + (double(v0[0]) - x_min) * scale;
124 double v0y = box_o + (double(v0[1]) - y_min) * scale;
125 double v1x = box_o + (double(v1[0]) - x_min) * scale;
126 double v1y = box_o + (double(v1[1]) - y_min) * scale;
127
128 // write vertices
129 os << v0x << " " << v0y << " moveto" << std::endl;
130 os << v1x << " " << v1y << " lineto" << std::endl;
131 }
132
133 // draw path and show page
134 os << "stroke" << std::endl;
135 os << "showpage" << std::endl;
136 }
