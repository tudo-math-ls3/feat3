289 virtual void upload_vector(SolverVectorType& vector, const VectorTypeV& vector_v, const VectorTypeP& vector_p) const
290 {
291 const IT_ nv = IT_(vector_v.template size<LAFEM::Perspective::pod>());
292 const IT_ np = IT_(vector_p.size());
293 const IT_ nx = IT_(vector.size());
294
295 XASSERTM(nx == IT_(this->_solver_matrix.rows()), "invalid solver vector size");
296 XASSERTM(nv + np <= nx, "invalid velocity/pressure vector size");
297
298 SolverDataType* vec = vector.elements();
299 const DT_* vec_v = vector_v.template elements<LAFEM::Perspective::pod>();
300 const DT_* vec_p = vector_p.elements();
301 IT_ j(0u);
302 // copy velocity DOFs
303 for(IT_ i(0); i < nv; ++i, ++j)
304 vec[j] = SolverDataType(vec_v[i]);
305 // copy pressure DOFs
306 for(IT_ i(0); i < np; ++i, ++j)
307 vec[j] = SolverDataType(vec_p[i]);
308 // format Lagrange multipliers
309 for(; j < nx; ++j)
310 vec[j] = SolverDataType(0);
311 }

384 virtual void init_symbolic()
385 {
386 // get the number of DOFs
387 const IT_ idim = IT_(dim_);
388 const IT_ num_dofs_v = IT_(_matrix_a.rows());
389 const IT_ num_dofs_p = IT_(_matrix_d.rows());
390
391 XASSERT(_matrix_a.columns() == num_dofs_v);
392 XASSERT(_matrix_b.columns() == num_dofs_p);
393 XASSERT(_matrix_d.columns() == num_dofs_v);
394 XASSERT(_matrix_b.rows() == num_dofs_v);
395 XASSERT(_matrix_d.rows() == num_dofs_p);
396
397 if(_matrix_s != nullptr)
398 {
399 XASSERT(_matrix_s->rows() == num_dofs_p);
400 XASSERT(_matrix_s->columns() == num_dofs_p);
401 }
402
403 // is the mean filter dual vector non-empty?
404 this->_have_mean_filter_p = !this->_mean_filter_p_dual.empty();
405
406 // compute number of slip DOFs
407 const IT_ num_slip_dofs = this->_compute_slip_matrix_layout();
408
409 // mask velocity and pressure unit dofs
410 this->_compute_unit_mask_v();
411 this->_compute_unit_mask_p();
412
413 // compute total number of scalar DOFs
414 const IT_ num_total_dofs = idim*num_dofs_v + num_dofs_p + num_slip_dofs + IT_(_have_mean_filter_p ? 1 : 0);
415
416 // get number of matrix NZEs
417 const IT_ num_nze_a = IT_(_matrix_a.used_elements());
418 const IT_ num_nze_b = IT_(_matrix_b.used_elements());
419 const IT_ num_nze_d = IT_(_matrix_d.used_elements());
420 const IT_ num_nze_s = IT_(_matrix_s != nullptr ? _matrix_s->used_elements() : 0u);
421
422 // compute upper bound for non-zero entry count
423 IT_ max_nze = idim*idim*num_nze_a + idim*(num_nze_b + num_nze_d) + num_nze_s;
424
425 // Lagrange multiplier for velocity slip filters
426 max_nze += IT_(2)*num_slip_dofs*idim;
427
428 // Lagrange multiplier for pressure mean filter (when necessary)
429 if(this->_have_mean_filter_p)
430 max_nze += IT_(2)*num_dofs_p;
431
432 // need a main diagonal?
433 if(this->_need_main_diagonal)
434 max_nze += num_dofs_p + IT_(2)*num_slip_dofs*idim + IT_(1);
435
436 // allocate row-pointer and column index arrays
437 std::vector<IT_> row_ptr, col_idx;
438 row_ptr.resize(num_total_dofs + IT_(1u), ~IT_(0));
439 col_idx.resize(max_nze, ~IT_(0));
440 row_ptr[0u] = IT_(0u);
441
442 // the system has the following structure:
443 // | A B Q 0 |
444 // | D S 0 P |
445 // | Q^T 0 0 0 |
446 // | 0 P^T 0 0 |
447 //
448 // where:
449 // - A, B, D and S are the original matrices from the Stokes system
450 // - Q is the Lagrange multiplier for the velocity slip DOFs
451 // - P is the Lagrange multiplier for the pressure mean filter
452
453 // get all the matrix arrays
454 const IT_* row_ptr_a = _matrix_a.row_ptr();
455 const IT_* col_idx_a = _matrix_a.col_ind();
456 const IT_* row_ptr_b = _matrix_b.row_ptr();
457 const IT_* col_idx_b = _matrix_b.col_ind();
458 const IT_* row_ptr_d = _matrix_d.row_ptr();
459 const IT_* col_idx_d = _matrix_d.col_ind();
460 const IT_* row_ptr_s = (_matrix_s != nullptr ? _matrix_s->row_ptr() : nullptr);
461 const IT_* col_idx_s = (_matrix_s != nullptr ? _matrix_s->col_ind() : nullptr);
462
463 IT_ row(0u), op(0u);
464
465 // okay, loop over all rows of A/B/Q
466 for(IT_ i(0); i < num_dofs_v; ++i)
467 {
468 // is this a unit row?
469 if(_unit_mask_v[i] != 0)
470 {
471 // set the entire block to an identity block
472 for(IT_ k(0); k < idim; ++k)
473 {
474 col_idx[op] = i*idim + k;
475 row_ptr[++row] = ++op;
476 }
477 continue;
478 }
479
480 // not a unit row, so copy rows of A, B and Q
481 for(IT_ j(0); j < idim; ++j)
482 {
483 // copy row of A
484 for(IT_ ip(row_ptr_a[i]); ip < row_ptr_a[i+1]; ++ip)
485 for(IT_ k(0); k < idim; ++k, ++op)
486 col_idx[op] = idim* col_idx_a[ip] + k;
487 // copy row of B
488 for(IT_ ip(row_ptr_b[i]); ip < row_ptr_b[i+1]; ++ip, ++op)
489 col_idx[op] = idim*num_dofs_v + col_idx_b[ip];
490 // copy row of Q
491 for(IT_ ip(_slip_row_ptr[i]); ip < _slip_row_ptr[i+1]; ++ip, ++op)
492 col_idx[op] = idim*num_dofs_v + num_dofs_p + _slip_col_idx[ip];
493 // set row pointer
494 row_ptr[++row] = op;
495 }
496 }
497
498 XASSERT(row == idim*num_dofs_v);
499
500 // next, loop over all rows of D/S/P
501 for(IT_ i(0); i < num_dofs_p; ++i)
502 {
503 // is this a unit row?
504 if(_unit_mask_p[i] != 0)
505 {
506 col_idx[op] = idim*num_dofs_v + i;
507 row_ptr[++row] = ++op;
508 continue;
509 }
510
511 // not a unit row, so copy rows of D, S and P
512 // copy row of D
513 for(IT_ ip(row_ptr_d[i]); ip < row_ptr_d[i+1]; ++ip)
514 for(IT_ k(0); k < idim; ++k, ++op)
515 col_idx[op] = idim*col_idx_d[ip] + k;
516 // copy row of S (if it exists)
517 if(row_ptr_s != nullptr)
518 {
519 if(this->_need_main_diagonal)
520 {
521 // copy row i of S and insert a main diagonal entry if it is not present in S
522 bool have_diag = false;
523 for(IT_ ip(row_ptr_s[i]); ip < row_ptr_s[i+1]; ++ip, ++op)
524 {
525 if(!have_diag && (col_idx_s[ip] > i))
526 {
527 col_idx[op++] = row;
528 have_diag = true;
529 }
530 col_idx[op] = idim*num_dofs_v + col_idx_s[ip];
531 have_diag = have_diag || (col_idx_s[ip] == i);
532 }
533 }
534 else
535 {
536 for(IT_ ip(row_ptr_s[i]); ip < row_ptr_s[i+1]; ++ip, ++op)
537 col_idx[op] = idim*num_dofs_v + col_idx_s[ip];
538 }
539 }
540 else if(this->_need_main_diagonal)
541 {
542 // insert a main diagonal entry
543 col_idx[op++] = row;
544 }
545 // copy row of P (if we have a pressure mean filter)
546 if(this->_have_mean_filter_p)
547 col_idx[op++] = idim*num_dofs_v + num_slip_dofs + num_dofs_p;
548 // set row pointer
549 row_ptr[++row] = op;
550 }
551
552 // next, add all slip filter entries Q^T
553 for(auto it(_slip_filters_v.begin()); it != _slip_filters_v.end(); ++it)
554 {
555 const IT_ nsx = IT_(it->used_elements());
556 const IT_* idx = it->get_indices();
557 for(IT_ j(0); j < nsx; ++j)
558 {
559 for(IT_ k(0); k < idim; ++k, ++op)
560 col_idx[op] = idim*idx[j] + k;
561 if(this->_need_main_diagonal)
562 col_idx[op++] = row;
563 // set row pointer
564 row_ptr[++row] = op;
565 }
566 }
567
568 // finally, add the pressure mean filter vector P^T (if it exists)
569 if(this->_have_mean_filter_p)
570 {
571 for(IT_ j(0); j < num_dofs_p; ++j, ++op)
572 col_idx[op] = idim*num_dofs_v + j;
573 if(this->_need_main_diagonal)
574 col_idx[op++] = row;
575 row_ptr[++row] = op;
576 }
577
578 // the number of rows must match, but we may have less non-zeros due to unit filtering
579 XASSERT(row == num_total_dofs);
580 XASSERT(op <= max_nze);
581
582 const Index num_nze = op;
583
584 // ok, allocate the matrix
585 this->_solver_matrix = SolverMatrixType(num_total_dofs, num_total_dofs, num_nze);
586
587 // copy row pointer
588 SolverIndexType* row_ptr_x = this->_solver_matrix.row_ptr();
589 for(IT_ i(0); i <= num_total_dofs; ++i)
590 row_ptr_x[i] = SolverIndexType(row_ptr[i]);
591
592 // copy column indices
593 SolverIndexType* col_idx_x = this->_solver_matrix.col_ind();
594 for(IT_ i(0); i < num_nze; ++i)
595 col_idx_x[i] = SolverIndexType(col_idx[i]);
596 }

603 virtual void init_numeric()
604 {
605 // get the number of DOFs
606 const IT_ num_dofs_v = IT_(_matrix_a.rows());
607 const IT_ num_dofs_p = IT_(_matrix_d.rows());
608
609 // the system has the following structure:
610 // | A B Q 0 |
611 // | D S 0 P |
612 // | Q^T 0 0 0 |
613 // | 0 P^T 0 0 |
614 //
615 // where:
616 // - A, B, D and S are the original matrices from the Stokes system
617 // - Q is the Lagrange multiplier for the velocity slip DOFs
618 // - P is the Lagrange multiplier for the pressure mean filter
619
620 // get all the matrix arrays
621 const IT_* row_ptr_a = _matrix_a.row_ptr();
622 const auto* val_a = _matrix_a.val();
623 const IT_* row_ptr_b = _matrix_b.row_ptr();
624 const auto* val_b = _matrix_b.val();
625 const IT_* row_ptr_d = _matrix_d.row_ptr();
626 const auto* val_d = _matrix_d.val();
627 const IT_* row_ptr_s = (_matrix_s != nullptr ? _matrix_s->row_ptr() : nullptr);
628 const IT_* col_idx_s = (_matrix_s != nullptr ? _matrix_s->col_ind() : nullptr);
629 const auto* val_s = (_matrix_s != nullptr ? _matrix_s->val() : nullptr);
630 SolverDataType* val_x = _solver_matrix.val();
631
632 IT_ op(0u);
633
634 // okay, loop over all rows of A/B/Q
635 for(IT_ i(0); i < num_dofs_v; ++i)
636 {
637 // is this a unit row?
638 if(_unit_mask_v[i] != 0)
639 {
640 // set the entire block to an identity block
641 for(int k(0); k < dim_; ++k, ++op)
642 val_x[op] = SolverDataType(1);
643 continue;
644 }
645
646 // not a unit row, so copy rows of A, B and Q
647 for(int j(0); j < dim_; ++j)
648 {
649 // copy row of A
650 for(IT_ ip(row_ptr_a[i]); ip < row_ptr_a[i+1]; ++ip)
651 for(int k(0); k < dim_; ++k, ++op)
652 val_x[op] = SolverDataType(val_a[ip][j][k]);
653 // copy row of B
654 for(IT_ ip(row_ptr_b[i]); ip < row_ptr_b[i+1]; ++ip, ++op)
655 val_x[op] = SolverDataType(val_b[ip][j][0]);
656 // copy row of Q
657 for(IT_ ip(_slip_row_ptr[i]); ip < _slip_row_ptr[i+1]; ++ip, ++op)
658 {
659 // get the normal vector of this slip filter entry
660 const auto& nu = this->_slip_filters_v.at(_slip_fil_dqu[ip]).get_values()[_slip_fil_idx[ip]];
661 // set the normal vector component
662 val_x[op] = SolverDataType(nu[j]);
663 }
664 }
665 }
666
667 // next, loop over all rows of D/S/P
668 for(IT_ i(0); i < num_dofs_p; ++i)
669 {
670 // is this a unit row?
671 if(_unit_mask_p[i] != 0)
672 {
673 val_x[op++] = SolverDataType(1);
674 continue;
675 }
676
677 // not a unit row, so copy rows of D, S and P
678 // copy row of D
679 for(IT_ ip(row_ptr_d[i]); ip < row_ptr_d[i+1]; ++ip)
680 for(int k(0); k < dim_; ++k, ++op)
681 val_x[op] = SolverDataType(val_d[ip][0][k]);
682 // copy row of S (if it exists)
683 if(row_ptr_s != nullptr)
684 {
685 if(this->_need_main_diagonal)
686 {
687 bool have_diag = false;
688 for(IT_ ip(row_ptr_s[i]); ip < row_ptr_s[i+1]; ++ip, ++op)
689 {
690 if(!have_diag && (col_idx_s[ip] > i))
691 {
692 val_x[op++] = SolverDataType(0);
693 have_diag = true;
694 }
695 val_x[op] = SolverDataType(val_s[ip]);
696 have_diag = have_diag || (col_idx_s[ip] == i);
697 }
698 }
699 else
700 {
701 for(IT_ ip(row_ptr_s[i]); ip < row_ptr_s[i+1]; ++ip, ++op)
702 val_x[op] = SolverDataType(val_s[ip]);
703 }
704 }
705 else if(this->_need_main_diagonal)
706 {
707 // insert a main diagonal entry
708 val_x[op++] = SolverDataType(0);
709 }
710
711 // copy row of P (if we have a pressure mean filter)
712 if(this->_have_mean_filter_p)
713 {
714 const auto* v = _mean_filter_p_dual.elements();
715 val_x[op++] = SolverDataType(v[i]);
716 }
717 }
718
719 // next, add all slip filter entries Q^T
720 for(auto it(_slip_filters_v.begin()); it != _slip_filters_v.end(); ++it)
721 {
722 const IT_ nsx = IT_(it->used_elements());
723 const auto* nu = it->get_values();
724 for(IT_ j(0); j < nsx; ++j)
725 {
726 for(int k(0); k < dim_; ++k, ++op)
727 val_x[op] = SolverDataType(nu[j][k]);
728 if(this->_need_main_diagonal)
729 val_x[op++] = SolverDataType(0);
730 }
731 }
732
733 // finally, add the pressure mean filter vector P^T (if it exists)
734 if(this->_have_mean_filter_p)
735 {
736 const auto* v = _mean_filter_p_dual.elements();
737 for(IT_ j(0); j < num_dofs_p; ++j, ++op)
738 val_x[op] = SolverDataType(v[j]);
739 if(this->_need_main_diagonal)
740 val_x[op++] = SolverDataType(0);
741 }
742
743 // sanity check
744 XASSERT(op == _solver_matrix.used_elements());
745 }

778 void _add_filter_velo(const UnitFilterTypeV& filter, FilterUpdateCounter* fuc)
779 {
780 if(fuc != nullptr)
781 {
782 // get the old unit filter
783 UnitFilterTypeV& old_filter = this->_unit_filters_v.at(fuc->at(0u)++);
784
785 // compare layout of old and new filter
786 if(!old_filter.get_filter_vector().compare_layout(filter.get_filter_vector()))
787 XABORTM("velocity unit filter layout has changed");
788
789 // update the filter
790 old_filter.clone(filter, LAFEM::CloneMode::Shallow);
791 }
792 else

939 IT_ _compute_slip_matrix_layout()
940 {
941 const IT_ nv = IT_(_matrix_a.rows());
942
943 _slip_row_ptr.clear();
944 _slip_col_idx.clear();
945 _slip_fil_dqu.clear();
946 _slip_fil_idx.clear();
947 _slip_row_ptr.resize(nv+1u, IT_(0));
948
949 if(this->_slip_filters_v.empty())
950 return IT_(0);
951
952 // count the total number of slip filter entries
953 IT_ count(0u);
954 for(std::size_t it(0u); it < this->_slip_filters_v.size(); ++it)
955 {
956 const IT_ ni = IT_(this->_slip_filters_v.at(it).used_elements());
957 const IT_* idx = this->_slip_filters_v.at(it).get_indices();
958 for(IT_ i(0); i < ni; ++i)
959 ++_slip_row_ptr.at(idx[i]+1);
960 count += ni;
961 }
962
963 // only empty filters?
964 if(count == IT_(0))
965 return count;
966
967 // compute row pointer array from aux
968 for(IT_ i(0); i < nv; ++i)
969 {
970 _slip_row_ptr[i+1u] += _slip_row_ptr[i];
971 }
972
973 // allocate vectors
974 _slip_col_idx.resize(count);
975 _slip_fil_dqu.resize(count);
976 _slip_fil_idx.resize(count);
977
978 // copy row pointer
979 std::vector<IT_> aux(_slip_row_ptr);
980
981 // loop over all slip filters
982 IT_ offset(0u);
983 for(std::size_t it(0u); it < this->_slip_filters_v.size(); ++it)
984 {
985 const IT_ ni = IT_(this->_slip_filters_v.at(it).used_elements());
986 const IT_* idx = this->_slip_filters_v.at(it).get_indices();
987
988 // and emplace them
989 for(IT_ i(0); i < ni; ++i)
990 {
991 IT_& px = aux[idx[i]];
992 _slip_col_idx[px] = offset + i;
993 _slip_fil_dqu[px] = IT_(it);
994 _slip_fil_idx[px] = i;
995 ++px;
996 }
997 offset += ni;
998 }
999
1000 XASSERT(count == offset);
1001 return count;
1002 }
