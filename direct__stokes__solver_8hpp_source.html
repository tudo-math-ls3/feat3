270 virtual void upload_vector(SolverVectorType& vector, const VectorTypeV& vector_v, const VectorTypeP& vector_p) const
271 {
272 const IT_ nv = vector_v.template size<LAFEM::Perspective::pod>();
273 const IT_ np = vector_p.size();
274 const IT_ nx = vector.size();
275
276 XASSERTM(nx == this->_solver_matrix.rows(), "invalid solver vector size");
277 XASSERTM(nv + np <= nx, "invalid velocity/pressure vector size");
278
279 SolverDataType* vec = vector.elements();
280 const DT_* vec_v = vector_v.template elements<LAFEM::Perspective::pod>();
281 const DT_* vec_p = vector_p.elements();
282 IT_ j(0u);
283 // copy velocity DOFs
284 for(IT_ i(0); i < nv; ++i, ++j)
285 vec[j] = SolverDataType(vec_v[i]);
286 // copy pressure DOFs
287 for(IT_ i(0); i < np; ++i, ++j)
288 vec[j] = SolverDataType(vec_p[i]);
289 // format Lagrange multipliers
290 for(; j < nx; ++j)
291 vec[j] = SolverDataType(0);
292 }

306 const IT_ np = vector_p.size();

313 DT_* vec_v = vector_v.template elements<LAFEM::Perspective::pod>();
314 DT_* vec_p = vector_p.elements();
315 IT_ j(0u);
316 // copy velocity DOFs
317 for(IT_ i(0); i < nv; ++i, ++j)
318 vec_v[i] = DT_(vec[j]);
319 // copy pressure DOFs
320 for(IT_ i(0); i < np; ++i, ++j)
321 vec_p[i] = DT_(vec[j]);
322 // Lagrange multipliers are ignored
323 }

365 virtual void init_symbolic()
366 {
367 // get the number of DOFs
368 const IT_ idim = IT_(dim_);
369 const IT_ num_dofs_v = _matrix_a.rows();
370 const IT_ num_dofs_p = _matrix_d.rows();
371
372 XASSERT(_matrix_a.columns() == num_dofs_v);
373 XASSERT(_matrix_b.columns() == num_dofs_p);
374 XASSERT(_matrix_d.columns() == num_dofs_v);
375 XASSERT(_matrix_b.rows() == num_dofs_v);
376 XASSERT(_matrix_d.rows() == num_dofs_p);
377
378 if(_matrix_s != nullptr)
379 {
380 XASSERT(_matrix_s->rows() == num_dofs_p);
381 XASSERT(_matrix_s->columns() == num_dofs_p);
382 }
383
384 // is the mean filter dual vector non-empty?
385 this->_have_mean_filter_p = !this->_mean_filter_p_dual.empty();
386
387 // compute number of slip DOFs
388 const IT_ num_slip_dofs = this->_compute_slip_matrix_layout();
389
390 // mask velocity and pressure unit dofs
391 this->_compute_unit_mask_v();
392 this->_compute_unit_mask_p();
393
394 // compute total number of scalar DOFs
395 const IT_ num_total_dofs = idim*num_dofs_v + num_dofs_p + num_slip_dofs + IT_(_have_mean_filter_p ? 1 : 0);
396
397 // get number of matrix NZEs
398 const IT_ num_nze_a = _matrix_a.used_elements();
399 const IT_ num_nze_b = _matrix_b.used_elements();
400 const IT_ num_nze_d = _matrix_d.used_elements();
401 const IT_ num_nze_s = IT_(_matrix_s != nullptr ? _matrix_s->used_elements() : 0u);
402
403 // compute upper bound for non-zero entry count
404 IT_ max_nze = idim*idim*num_nze_a + idim*(num_nze_b + num_nze_d) + num_nze_s;
405
406 // Lagrange multiplier for velocity slip filters
407 max_nze += IT_(2)*num_slip_dofs*idim;
408
409 // Lagrange multiplier for pressure mean filter (when necessary)
410 if(this->_have_mean_filter_p)
411 max_nze += IT_(2)*num_dofs_p;
412
413 // allocate row-pointer and column index arrays
414 std::vector<IT_> row_ptr, col_idx;
415 row_ptr.resize(num_total_dofs + IT_(1u), ~IT_(0));
416 col_idx.resize(max_nze, ~IT_(0));
417 row_ptr[0u] = IT_(0u);
418
419 // the system has the following structure:
420 // | A B Q 0 |
421 // | D S 0 P |
422 // | Q^T 0 0 0 |
423 // | 0 P^T 0 0 |
424 //
425 // where:
426 // - A, B, D and S are the original matrices from the Stokes system
427 // - Q is the Lagrange multiplier for the velocity slip DOFs
428 // - P is the Lagrange multiplier for the pressure mean filter
429
430 // get all the matrix arrays
431 const IT_* row_ptr_a = _matrix_a.row_ptr();
432 const IT_* col_idx_a = _matrix_a.col_ind();
433 const IT_* row_ptr_b = _matrix_b.row_ptr();
434 const IT_* col_idx_b = _matrix_b.col_ind();
435 const IT_* row_ptr_d = _matrix_d.row_ptr();
436 const IT_* col_idx_d = _matrix_d.col_ind();
437 const IT_* row_ptr_s = (_matrix_s != nullptr ? _matrix_s->row_ptr() : nullptr);
438 const IT_* col_idx_s = (_matrix_s != nullptr ? _matrix_s->col_ind() : nullptr);
439
440 IT_ row(0u), op(0u);
441
442 // okay, loop over all rows of A/B/Q
443 for(IT_ i(0); i < num_dofs_v; ++i)
444 {
445 // is this a unit row?
446 if(_unit_mask_v[i] != 0)
447 {
448 // set the entire block to an identity block
449 for(IT_ k(0); k < idim; ++k)
450 {
451 col_idx[op] = i*idim + k;
452 row_ptr[++row] = ++op;
453 }
454 continue;
455 }
456
457 // not a unit row, so copy rows of A, B and Q
458 for(IT_ j(0); j < idim; ++j)
459 {
460 // copy row of A
461 for(IT_ ip(row_ptr_a[i]); ip < row_ptr_a[i+1]; ++ip)
462 for(IT_ k(0); k < idim; ++k, ++op)
463 col_idx[op] = idim* col_idx_a[ip] + k;
464 // copy row of B
465 for(IT_ ip(row_ptr_b[i]); ip < row_ptr_b[i+1]; ++ip, ++op)
466 col_idx[op] = idim*num_dofs_v + col_idx_b[ip];
467 // copy row of Q
468 for(IT_ ip(_slip_row_ptr[i]); ip < _slip_row_ptr[i+1]; ++ip, ++op)
469 col_idx[op] = idim*num_dofs_v + num_dofs_p + _slip_col_idx[ip];
470 // set row pointer
471 row_ptr[++row] = op;
472 }
473 }
474
475 XASSERT(row == idim*num_dofs_v);
476
477 // next, loop over all rows of D/S/P
478 for(IT_ i(0); i < num_dofs_p; ++i)
479 {
480 // is this a unit row?
481 if(_unit_mask_p[i] != 0)
482 {
483 col_idx[op] = idim*num_dofs_v + i;
484 row_ptr[++row] = ++op;
485 continue;
486 }
487
488 // not a unit row, so copy rows of D, S and P
489 // copy row of D
490 for(IT_ ip(row_ptr_d[i]); ip < row_ptr_d[i+1]; ++ip)
491 for(IT_ k(0); k < idim; ++k, ++op)
492 col_idx[op] = idim*col_idx_d[ip] + k;
493 // copy row of S (if it exists)
494 if(row_ptr_s != nullptr)
495 for(IT_ ip(row_ptr_s[i]); ip < row_ptr_s[i+1]; ++ip, ++op)
496 col_idx[op] = idim*num_dofs_v + col_idx_s[ip];
497 // copy row of P (if we have a pressure mean filter)
498 if(this->_have_mean_filter_p)
499 col_idx[op++] = idim*num_dofs_v + num_slip_dofs + num_dofs_p;
500 // set row pointer
501 row_ptr[++row] = op;
502 }
503
504 // next, add all slip filter entries Q^T
505 for(auto it(_slip_filters_v.begin()); it != _slip_filters_v.end(); ++it)
506 {
507 const IT_ nsx = it->used_elements();
508 const IT_* idx = it->get_indices();
509 for(IT_ j(0); j < nsx; ++j)
510 {
511 for(IT_ k(0); k < idim; ++k, ++op)
512 col_idx[op] = idim*idx[j] + k;
513 // set row pointer
514 row_ptr[++row] = op;
515 }
516 }
517
518 // finally, add the pressure mean filter vector P^T (if it exists)
519 if(this->_have_mean_filter_p)
520 {
521 for(IT_ j(0); j < num_dofs_p; ++j, ++op)
522 col_idx[op] = idim*num_dofs_v + j;
523 row_ptr[++row] = op;
524 }
525
526 // the number of rows must match, but we may have less non-zeros due to unit filtering
527 XASSERT(row == num_total_dofs);
528 XASSERT(op <= max_nze);
529
530 const Index num_nze = op;
531
532 // ok, allocate the matrix
533 this->_solver_matrix = SolverMatrixType(num_total_dofs, num_total_dofs, num_nze);
534
535 // copy row pointer
536 SolverIndexType* row_ptr_x = this->_solver_matrix.row_ptr();
537 for(IT_ i(0); i <= num_total_dofs; ++i)
538 row_ptr_x[i] = SolverIndexType(row_ptr[i]);
539
540 // copy column indices
541 SolverIndexType* col_idx_x = this->_solver_matrix.col_ind();
542 for(IT_ i(0); i < num_nze; ++i)
543 col_idx_x[i] = SolverIndexType(col_idx[i]);
544 }

551 virtual void init_numeric()
552 {
553 // get the number of DOFs
554 const IT_ num_dofs_v = _matrix_a.rows();
555 const IT_ num_dofs_p = _matrix_d.rows();
556
557 // the system has the following structure:
558 // | A B Q 0 |
559 // | D S 0 P |
560 // | Q^T 0 0 0 |
561 // | 0 P^T 0 0 |
562 //
563 // where:
564 // - A, B, D and S are the original matrices from the Stokes system
565 // - Q is the Lagrange multiplier for the velocity slip DOFs
566 // - P is the Lagrange multiplier for the pressure mean filter
567
568 // get all the matrix arrays
569 const IT_* row_ptr_a = _matrix_a.row_ptr();
570 const auto* val_a = _matrix_a.val();
571 const IT_* row_ptr_b = _matrix_b.row_ptr();
572 const auto* val_b = _matrix_b.val();
573 const IT_* row_ptr_d = _matrix_d.row_ptr();
574 const auto* val_d = _matrix_d.val();
575 const IT_* row_ptr_s = (_matrix_s != nullptr ? _matrix_s->row_ptr() : nullptr);
576 const auto* val_s = (_matrix_s != nullptr ? _matrix_s->val() : nullptr);
577 SolverDataType* val_x = _solver_matrix.val();
578
579 IT_ op(0u);
580
581 // okay, loop over all rows of A/B/Q
582 for(IT_ i(0); i < num_dofs_v; ++i)
583 {
584 // is this a unit row?
585 if(_unit_mask_v[i] != 0)
586 {
587 // set the entire block to an identity block
588 for(int k(0); k < dim_; ++k, ++op)
589 val_x[op] = SolverDataType(1);
590 continue;
591 }
592
593 // not a unit row, so copy rows of A, B and Q
594 for(int j(0); j < dim_; ++j)
595 {
596 // copy row of A
597 for(IT_ ip(row_ptr_a[i]); ip < row_ptr_a[i+1]; ++ip)
598 for(int k(0); k < dim_; ++k, ++op)
599 val_x[op] = SolverDataType(val_a[ip][j][k]);
600 // copy row of B
601 for(IT_ ip(row_ptr_b[i]); ip < row_ptr_b[i+1]; ++ip, ++op)
602 val_x[op] = SolverDataType(val_b[ip][j][0]);
603 // copy row of Q
604 for(IT_ ip(_slip_row_ptr[i]); ip < _slip_row_ptr[i+1]; ++ip, ++op)
605 {
606 // get the normal vector of this slip filter entry
607 const auto& nu = this->_slip_filters_v.at(_slip_fil_dqu[ip]).get_values()[_slip_fil_idx[ip]];
608
609 // set the normal vector component
610 val_x[op] = SolverDataType(nu[j]);
611 }
612 }
613 }
614
615 // next, loop over all rows of D/S/P
616 for(IT_ i(0); i < num_dofs_p; ++i)
617 {
618 // is this a unit row?
619 if(_unit_mask_p[i] != 0)
620 {
621 val_x[op++] = SolverDataType(1);
622 continue;
623 }
624
625 // not a unit row, so copy rows of D, S and P
626 // copy row of D
627 for(IT_ ip(row_ptr_d[i]); ip < row_ptr_d[i+1]; ++ip)
628 for(int k(0); k < dim_; ++k, ++op)
629 val_x[op] = SolverDataType(val_d[ip][0][k]);
630 // copy row of S (if it exists)
631 if(row_ptr_s != nullptr)
632 for(IT_ ip(row_ptr_s[i]); ip < row_ptr_s[i+1]; ++ip, ++op)
633 val_x[op] = SolverDataType(val_s[ip]);
634 // copy row of P (if we have a pressure mean filter)
635 if(this->_have_mean_filter_p)
636 {
637 const auto* v = _mean_filter_p_dual.elements();
638 val_x[op++] = SolverDataType(v[i]);
639 }
640 }
641
642 // next, add all slip filter entries Q^T
643 for(auto it(_slip_filters_v.begin()); it != _slip_filters_v.end(); ++it)
644 {
645 const IT_ nsx = it->used_elements();
646 const auto* nu = it->get_values();
647 for(IT_ j(0); j < nsx; ++j)
648 {
649 for(int k(0); k < dim_; ++k, ++op)
650 val_x[op] = SolverDataType(nu[j][k]);
651 }
652 }
653
654 // finally, add the pressure mean filter vector P^T (if it exists)
655 if(this->_have_mean_filter_p)
656 {
657 const auto* v = _mean_filter_p_dual.elements();
658 for(IT_ j(0); j < num_dofs_p; ++j, ++op)
659 val_x[op] = SolverDataType(v[j]);
660 }
661 }

694 void _add_filter_velo(const UnitFilterTypeV& filter, FilterUpdateCounter* fuc)
695 {
696 if(fuc != nullptr)
697 {
698 // get the old unit filter
699 UnitFilterTypeV& old_filter = this->_unit_filters_v.at(fuc->at(0u)++);
700
701 // compare layout of old and new filter
702 if(!old_filter.get_filter_vector().compare_layout(filter.get_filter_vector()))
703 XABORTM("velocity unit filter layout has changed");
704
705 // update the filter
706 old_filter.clone(filter, LAFEM::CloneMode::Shallow);
707 }
708 else
709 {
710 // push new filter
711 this->_unit_filters_v.push_back(filter.clone(LAFEM::CloneMode::Shallow));
712 }
713 }

856 {
857 const IT_ nv = _matrix_a.rows();
858
859 _slip_row_ptr.clear();
860 _slip_col_idx.clear();
861 _slip_fil_dqu.clear();
862 _slip_fil_idx.clear();
863 _slip_row_ptr.resize(nv+1u, IT_(0));
864
865 if(this->_slip_filters_v.empty())
866 return IT_(0);
867
868 // count the total number of slip filter entries
869 IT_ count(0u);
870 for(std::size_t it(0u); it < this->_slip_filters_v.size(); ++it)
871 {
872 const IT_ ni = IT_(this->_slip_filters_v.at(it).used_elements());
873 const IT_* idx = this->_slip_filters_v.at(it).get_indices();
874 for(IT_ i(0); i < ni; ++i)
875 ++_slip_row_ptr.at(idx[i]+1);
876 count += ni;
877 }
878
879 // only empty filters?
880 if(count == IT_(0))
881 return count;
882
883 // compute row pointer array from aux
884 for(IT_ i(0); i < nv; ++i)
885 {
886 _slip_row_ptr[i+1u] += _slip_row_ptr[i];
887 }
888
889 // allocate vectors
890 _slip_col_idx.resize(count);
891 _slip_fil_dqu.resize(count);
892 _slip_fil_idx.resize(count);
893
894 // copy row pointer
895 std::vector<IT_> aux(_slip_row_ptr);
896
897 // loop over all slip filters
898 IT_ offset(0u);
899 for(std::size_t it(0u); it < this->_slip_filters_v.size(); ++it)
900 {
901 const IT_ ni = IT_(this->_slip_filters_v.at(it).used_elements());
902 const IT_* idx = this->_slip_filters_v.at(it).get_indices();
903
904 // and emplace them
905 for(IT_ i(0); i < ni; ++i)
906 {
907 IT_& px = aux[idx[i]];
908 _slip_col_idx[px] = offset + i;
909 _slip_fil_dqu[px] = IT_(it);
910 _slip_fil_idx[px] = i;
911 ++px;
912 }
913 offset += ni;
914 }
915
916 XASSERT(count == offset);
917 return count;
918 }
