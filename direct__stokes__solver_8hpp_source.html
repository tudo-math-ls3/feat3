271 virtual void upload_vector(SolverVectorType& vector, const VectorTypeV& vector_v, const VectorTypeP& vector_p) const
272 {
273 const IT_ nv = vector_v.template size<LAFEM::Perspective::pod>();
274 const IT_ np = vector_p.size();
275 const IT_ nx = vector.size();
276
277 XASSERTM(nx == this->_solver_matrix.rows(), "invalid solver vector size");
278 XASSERTM(nv + np <= nx, "invalid velocity/pressure vector size");
279
280 SolverDataType* vec = vector.elements();
281 const DT_* vec_v = vector_v.template elements<LAFEM::Perspective::pod>();
282 const DT_* vec_p = vector_p.elements();
283 IT_ j(0u);
284 // copy velocity DOFs
285 for(IT_ i(0); i < nv; ++i, ++j)
286 vec[j] = SolverDataType(vec_v[i]);
287 // copy pressure DOFs
288 for(IT_ i(0); i < np; ++i, ++j)
289 vec[j] = SolverDataType(vec_p[i]);
290 // format Lagrange multipliers
291 for(; j < nx; ++j)
292 vec[j] = SolverDataType(0);
293 }

366 virtual void init_symbolic()
367 {
368 // get the number of DOFs
369 const IT_ idim = IT_(dim_);
370 const IT_ num_dofs_v = _matrix_a.rows();
371 const IT_ num_dofs_p = _matrix_d.rows();
372
373 XASSERT(_matrix_a.columns() == num_dofs_v);
374 XASSERT(_matrix_b.columns() == num_dofs_p);
375 XASSERT(_matrix_d.columns() == num_dofs_v);
376 XASSERT(_matrix_b.rows() == num_dofs_v);
377 XASSERT(_matrix_d.rows() == num_dofs_p);
378
379 if(_matrix_s != nullptr)
380 {
381 XASSERT(_matrix_s->rows() == num_dofs_p);
382 XASSERT(_matrix_s->columns() == num_dofs_p);
383 }
384
385 // is the mean filter dual vector non-empty?
386 this->_have_mean_filter_p = !this->_mean_filter_p_dual.empty();
387
388 // compute number of slip DOFs
389 const IT_ num_slip_dofs = this->_compute_slip_matrix_layout();
390
391 // mask velocity and pressure unit dofs
392 this->_compute_unit_mask_v();
393 this->_compute_unit_mask_p();
394
395 // compute total number of scalar DOFs
396 const IT_ num_total_dofs = idim*num_dofs_v + num_dofs_p + num_slip_dofs + IT_(_have_mean_filter_p ? 1 : 0);
397
398 // get number of matrix NZEs
399 const IT_ num_nze_a = _matrix_a.used_elements();
400 const IT_ num_nze_b = _matrix_b.used_elements();
401 const IT_ num_nze_d = _matrix_d.used_elements();
402 const IT_ num_nze_s = IT_(_matrix_s != nullptr ? _matrix_s->used_elements() : 0u);
403
404 // compute upper bound for non-zero entry count
405 IT_ max_nze = idim*idim*num_nze_a + idim*(num_nze_b + num_nze_d) + num_nze_s;
406
407 // Lagrange multiplier for velocity slip filters
408 max_nze += IT_(2)*num_slip_dofs*idim;
409
410 // Lagrange multiplier for pressure mean filter (when necessary)
411 if(this->_have_mean_filter_p)
412 max_nze += IT_(2)*num_dofs_p;
413
414 // allocate row-pointer and column index arrays
415 std::vector<IT_> row_ptr, col_idx;
416 row_ptr.resize(num_total_dofs + IT_(1u), ~IT_(0));
417 col_idx.resize(max_nze, ~IT_(0));
418 row_ptr[0u] = IT_(0u);
419
420 // the system has the following structure:
421 // | A B Q 0 |
422 // | D S 0 P |
423 // | Q^T 0 0 0 |
424 // | 0 P^T 0 0 |
425 //
426 // where:
427 // - A, B, D and S are the original matrices from the Stokes system
428 // - Q is the Lagrange multiplier for the velocity slip DOFs
429 // - P is the Lagrange multiplier for the pressure mean filter
430
431 // get all the matrix arrays
432 const IT_* row_ptr_a = _matrix_a.row_ptr();
433 const IT_* col_idx_a = _matrix_a.col_ind();
434 const IT_* row_ptr_b = _matrix_b.row_ptr();
435 const IT_* col_idx_b = _matrix_b.col_ind();
436 const IT_* row_ptr_d = _matrix_d.row_ptr();
437 const IT_* col_idx_d = _matrix_d.col_ind();
438 const IT_* row_ptr_s = (_matrix_s != nullptr ? _matrix_s->row_ptr() : nullptr);
439 const IT_* col_idx_s = (_matrix_s != nullptr ? _matrix_s->col_ind() : nullptr);
440
441 IT_ row(0u), op(0u);
442
443 // okay, loop over all rows of A/B/Q
444 for(IT_ i(0); i < num_dofs_v; ++i)
445 {
446 // is this a unit row?
447 if(_unit_mask_v[i] != 0)
448 {
449 // set the entire block to an identity block
450 for(IT_ k(0); k < idim; ++k)
451 {
452 col_idx[op] = i*idim + k;
453 row_ptr[++row] = ++op;
454 }
455 continue;
456 }
457
458 // not a unit row, so copy rows of A, B and Q
459 for(IT_ j(0); j < idim; ++j)
460 {
461 // copy row of A
462 for(IT_ ip(row_ptr_a[i]); ip < row_ptr_a[i+1]; ++ip)
463 for(IT_ k(0); k < idim; ++k, ++op)
464 col_idx[op] = idim* col_idx_a[ip] + k;
465 // copy row of B
466 for(IT_ ip(row_ptr_b[i]); ip < row_ptr_b[i+1]; ++ip, ++op)
467 col_idx[op] = idim*num_dofs_v + col_idx_b[ip];
468 // copy row of Q
469 for(IT_ ip(_slip_row_ptr[i]); ip < _slip_row_ptr[i+1]; ++ip, ++op)
470 col_idx[op] = idim*num_dofs_v + num_dofs_p + _slip_col_idx[ip];
471 // set row pointer
472 row_ptr[++row] = op;
473 }
474 }
475
476 XASSERT(row == idim*num_dofs_v);
477
478 // next, loop over all rows of D/S/P
479 for(IT_ i(0); i < num_dofs_p; ++i)
480 {
481 // is this a unit row?
482 if(_unit_mask_p[i] != 0)
483 {
484 col_idx[op] = idim*num_dofs_v + i;
485 row_ptr[++row] = ++op;
486 continue;
487 }
488
489 // not a unit row, so copy rows of D, S and P
490 // copy row of D
491 for(IT_ ip(row_ptr_d[i]); ip < row_ptr_d[i+1]; ++ip)
492 for(IT_ k(0); k < idim; ++k, ++op)
493 col_idx[op] = idim*col_idx_d[ip] + k;
494 // copy row of S (if it exists)
495 if(row_ptr_s != nullptr)
496 for(IT_ ip(row_ptr_s[i]); ip < row_ptr_s[i+1]; ++ip, ++op)
497 col_idx[op] = idim*num_dofs_v + col_idx_s[ip];
498 // copy row of P (if we have a pressure mean filter)
499 if(this->_have_mean_filter_p)
500 col_idx[op++] = idim*num_dofs_v + num_slip_dofs + num_dofs_p;
501 // set row pointer
502 row_ptr[++row] = op;
503 }
504
505 // next, add all slip filter entries Q^T
506 for(auto it(_slip_filters_v.begin()); it != _slip_filters_v.end(); ++it)
507 {
508 const IT_ nsx = it->used_elements();
509 const IT_* idx = it->get_indices();
510 for(IT_ j(0); j < nsx; ++j)
511 {
512 for(IT_ k(0); k < idim; ++k, ++op)
513 col_idx[op] = idim*idx[j] + k;
514 // set row pointer
515 row_ptr[++row] = op;
516 }
517 }
518
519 // finally, add the pressure mean filter vector P^T (if it exists)
520 if(this->_have_mean_filter_p)
521 {
522 for(IT_ j(0); j < num_dofs_p; ++j, ++op)
523 col_idx[op] = idim*num_dofs_v + j;
524 row_ptr[++row] = op;
525 }
526
527 // the number of rows must match, but we may have less non-zeros due to unit filtering
528 XASSERT(row == num_total_dofs);
529 XASSERT(op <= max_nze);
530
531 const Index num_nze = op;
532
533 // ok, allocate the matrix
534 this->_solver_matrix = SolverMatrixType(num_total_dofs, num_total_dofs, num_nze);
535
536 // copy row pointer
537 SolverIndexType* row_ptr_x = this->_solver_matrix.row_ptr();
538 for(IT_ i(0); i <= num_total_dofs; ++i)
539 row_ptr_x[i] = SolverIndexType(row_ptr[i]);
540
541 // copy column indices
542 SolverIndexType* col_idx_x = this->_solver_matrix.col_ind();
543 for(IT_ i(0); i < num_nze; ++i)
544 col_idx_x[i] = SolverIndexType(col_idx[i]);
545 }

552 virtual void init_numeric()
553 {
554 // get the number of DOFs
555 const IT_ num_dofs_v = _matrix_a.rows();
556 const IT_ num_dofs_p = _matrix_d.rows();
557
558 // the system has the following structure:
559 // | A B Q 0 |
560 // | D S 0 P |
561 // | Q^T 0 0 0 |
562 // | 0 P^T 0 0 |
563 //
564 // where:
565 // - A, B, D and S are the original matrices from the Stokes system
566 // - Q is the Lagrange multiplier for the velocity slip DOFs
567 // - P is the Lagrange multiplier for the pressure mean filter
568
569 // get all the matrix arrays
570 const IT_* row_ptr_a = _matrix_a.row_ptr();
571 const auto* val_a = _matrix_a.val();
572 const IT_* row_ptr_b = _matrix_b.row_ptr();
573 const auto* val_b = _matrix_b.val();
574 const IT_* row_ptr_d = _matrix_d.row_ptr();
575 const auto* val_d = _matrix_d.val();
576 const IT_* row_ptr_s = (_matrix_s != nullptr ? _matrix_s->row_ptr() : nullptr);
577 const auto* val_s = (_matrix_s != nullptr ? _matrix_s->val() : nullptr);
578 SolverDataType* val_x = _solver_matrix.val();
579
580 IT_ op(0u);
581
582 // okay, loop over all rows of A/B/Q
583 for(IT_ i(0); i < num_dofs_v; ++i)
584 {
585 // is this a unit row?
586 if(_unit_mask_v[i] != 0)
587 {
588 // set the entire block to an identity block
589 for(int k(0); k < dim_; ++k, ++op)
590 val_x[op] = SolverDataType(1);
591 continue;
592 }
593
594 // not a unit row, so copy rows of A, B and Q
595 for(int j(0); j < dim_; ++j)
596 {
597 // copy row of A
598 for(IT_ ip(row_ptr_a[i]); ip < row_ptr_a[i+1]; ++ip)
599 for(int k(0); k < dim_; ++k, ++op)
600 val_x[op] = SolverDataType(val_a[ip][j][k]);
601 // copy row of B
602 for(IT_ ip(row_ptr_b[i]); ip < row_ptr_b[i+1]; ++ip, ++op)
603 val_x[op] = SolverDataType(val_b[ip][j][0]);
604 // copy row of Q
605 for(IT_ ip(_slip_row_ptr[i]); ip < _slip_row_ptr[i+1]; ++ip, ++op)
606 {
607 // get the normal vector of this slip filter entry
608 const auto& nu = this->_slip_filters_v.at(_slip_fil_dqu[ip]).get_values()[_slip_fil_idx[ip]];
609
610 // set the normal vector component
611 val_x[op] = SolverDataType(nu[j]);
612 }
613 }
614 }
615
616 // next, loop over all rows of D/S/P
617 for(IT_ i(0); i < num_dofs_p; ++i)
618 {
619 // is this a unit row?
620 if(_unit_mask_p[i] != 0)
621 {
622 val_x[op++] = SolverDataType(1);
623 continue;
624 }
625
626 // not a unit row, so copy rows of D, S and P
627 // copy row of D
628 for(IT_ ip(row_ptr_d[i]); ip < row_ptr_d[i+1]; ++ip)
629 for(int k(0); k < dim_; ++k, ++op)
630 val_x[op] = SolverDataType(val_d[ip][0][k]);
631 // copy row of S (if it exists)
632 if(row_ptr_s != nullptr)
633 for(IT_ ip(row_ptr_s[i]); ip < row_ptr_s[i+1]; ++ip, ++op)
634 val_x[op] = SolverDataType(val_s[ip]);
635 // copy row of P (if we have a pressure mean filter)
636 if(this->_have_mean_filter_p)
637 {
638 const auto* v = _mean_filter_p_dual.elements();
639 val_x[op++] = SolverDataType(v[i]);
640 }
641 }
642
643 // next, add all slip filter entries Q^T
644 for(auto it(_slip_filters_v.begin()); it != _slip_filters_v.end(); ++it)
645 {
646 const IT_ nsx = it->used_elements();
647 const auto* nu = it->get_values();
648 for(IT_ j(0); j < nsx; ++j)
649 {
650 for(int k(0); k < dim_; ++k, ++op)
651 val_x[op] = SolverDataType(nu[j][k]);
652 }
653 }
654
655 // finally, add the pressure mean filter vector P^T (if it exists)
656 if(this->_have_mean_filter_p)
657 {
658 const auto* v = _mean_filter_p_dual.elements();
659 for(IT_ j(0); j < num_dofs_p; ++j, ++op)
660 val_x[op] = SolverDataType(v[j]);
661 }
662 }

697 if(fuc != nullptr)
698 {
699 // get the old unit filter
700 UnitFilterTypeV& old_filter = this->_unit_filters_v.at(fuc->at(0u)++);
701
702 // compare layout of old and new filter

713 }

856 IT_ _compute_slip_matrix_layout()
857 {
858 const IT_ nv = _matrix_a.rows();
859
860 _slip_row_ptr.clear();
861 _slip_col_idx.clear();
862 _slip_fil_dqu.clear();
863 _slip_fil_idx.clear();
864 _slip_row_ptr.resize(nv+1u, IT_(0));
865
866 if(this->_slip_filters_v.empty())
867 return IT_(0);
868
869 // count the total number of slip filter entries
870 IT_ count(0u);
871 for(std::size_t it(0u); it < this->_slip_filters_v.size(); ++it)
872 {
873 const IT_ ni = IT_(this->_slip_filters_v.at(it).used_elements());
874 const IT_* idx = this->_slip_filters_v.at(it).get_indices();
875 for(IT_ i(0); i < ni; ++i)
876 ++_slip_row_ptr.at(idx[i]+1);
877 count += ni;
878 }
879
880 // only empty filters?
881 if(count == IT_(0))
882 return count;
883
884 // compute row pointer array from aux
885 for(IT_ i(0); i < nv; ++i)
886 {
887 _slip_row_ptr[i+1u] += _slip_row_ptr[i];
888 }
889
890 // allocate vectors
891 _slip_col_idx.resize(count);
892 _slip_fil_dqu.resize(count);
893 _slip_fil_idx.resize(count);
894
895 // copy row pointer
896 std::vector<IT_> aux(_slip_row_ptr);
897
898 // loop over all slip filters
899 IT_ offset(0u);
900 for(std::size_t it(0u); it < this->_slip_filters_v.size(); ++it)
901 {
902 const IT_ ni = IT_(this->_slip_filters_v.at(it).used_elements());
903 const IT_* idx = this->_slip_filters_v.at(it).get_indices();
904
905 // and emplace them
906 for(IT_ i(0); i < ni; ++i)
907 {
908 IT_& px = aux[idx[i]];
909 _slip_col_idx[px] = offset + i;
910 _slip_fil_dqu[px] = IT_(it);
911 _slip_fil_idx[px] = i;
912 ++px;
913 }
914 offset += ni;
915 }
916
917 XASSERT(count == offset);
918 return count;
919 }
