466 virtual Status _startup(DataType& alpha, DataType& fval, DataType& delta, const VectorType& vec_sol, const VectorType& vec_dir)
467 {
468 Status status(Status::progress);
469
470 this->_num_iter = Index(0);
471 this->_vec_initial_sol.copy(vec_sol);
472
473 this->_vec_pn.copy(vec_dir);
474 this->_norm_dir = this->_vec_pn.norm2();
475
476 // First scale so that all entries are |.| < 1
477 this->_vec_pn.scale(this->_vec_pn, DataType(1)/this->_vec_pn.max_abs_element());
478 // Now scale so that vec_pn.norm2() == 1. Note that this will only be approximately true
479 this->_vec_pn.scale(this->_vec_pn, DataType(1)/this->_vec_pn.norm2());
480
481 // It is critical that fval_0 was set from outside using set_initial_fval!
482 this->_fval_min = this->_fval_0;
483 this->_delta_0 = this->_vec_pn.dot(this->_vec_grad);
484
485 if(this->_delta_0 > DataType(0))
486 {
487 XABORTM("Search direction is not a descent direction: " +stringify_fp_sci(this->_delta_0));
488 }
489
490 // Compute initial defect. We want to minimize d^T * grad(_functional)
491 this->_def_init = Math::abs(this->_delta_0);
492
493 // Norm of the initial guess
494 this->_norm_sol = vec_sol.norm2();
495
496 this->_alpha_min = DataType(0);
497
498 // plot?
499 if(this->_plot_iter())
500 {
501 String msg = this->_plot_name
502 + ": " + stringify(this->_num_iter).pad_front(this->_iter_digits)
503 + " : " + stringify_fp_sci(this->_def_init)
504 + " : " + stringify_fp_sci(this->_fval_0)
505 + ", ||dir|| = " + stringify_fp_sci(this->_norm_dir);
506 this->_print_line(msg);
507 }
508
509 if(!Math::isfinite(this->_fval_0) || !Math::isfinite(this->_delta_0) || !Math::isfinite(this->_norm_dir))
510 {
511 status = Status::aborted;
512 }
513
514 // Set initial step size
515 alpha = _alpha_0;
516 // Other intitial values
517 fval = _fval_0;
518 delta = _delta_0;
519
520 return status;
521 }

545 virtual Status _check_convergence(const DataType fval, const DataType df, const DataType alpha)
546 {
547 Status status(Status::progress);
548
549 this->_def_cur = Math::abs(df);
550
551 Statistics::add_solver_expression(
552 std::make_shared<ExpressionDefect>(this->name(), this->_def_cur, this->get_num_iter()));
553
554 // plot?
555 if(this->_plot_iter())
556 {
557 String msg = this->_plot_name
558 + ": " + stringify(this->_num_iter).pad_front(this->_iter_digits)
559 + " : " + stringify_fp_sci(this->_def_cur)
560 + " / " + stringify_fp_sci(this->_def_cur / this->_def_init)
561 + " : " + stringify_fp_sci(fval) + " : " + stringify_fp_sci(alpha);
562 this->_print_line(msg);
563 }
564
565 // ensure that the defect is neither NaN nor infinity
566 if(!Math::isfinite(this->_def_cur))
567 {
568 status = Status::aborted;
569 }
570
571 // is diverged?
572 if(this->is_diverged())
573 {
574 status = Status::diverged;
575 }
576
577 // If the maximum number of iterations was performed, return the iterate for the best step so far
578 if(this->_num_iter >= this->_max_iter)
579 {
580 status = Status::max_iter;
581 }
582
583 // If the strong Wolfe conditions hold, we are successful
584 if((fval < this->_fval_0 +_tol_decrease*alpha*_delta_0) && (Math::abs(df) < -_tol_curvature*_delta_0))
585 {
586 status = Status::success;
587 }
588
589 return status;
590 }
