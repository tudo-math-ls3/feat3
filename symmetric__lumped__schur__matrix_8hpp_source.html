257 void extract_diag(VectorTypeL& diag, bool sync=true) const
258 {
259 // If we have a gate and it contains neighbors, we have to do it the complicated way
260 if(diag.get_gate() != nullptr && !diag.get_gate()->_ranks.empty() )
261 {
262 diag.format();
263
264 typename MatrixB_::LocalMatrixType matrix_b1(matrix_b.convert_to_1());
265
266 matrix_b1.add_trace_double_mat_mult(diag.local(), matrix_d.local(), inv_lumped_matrix_a.local(), DataType(1));
267 }
268 else
269 {
270 matrix_d.local().row_norm2sqr(diag.local(), inv_lumped_matrix_a.local());
271 }
272
273 if(sync)
274 {
275 diag.sync_0();
276 }
277
278 }
