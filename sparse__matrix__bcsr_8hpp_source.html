186 template<typename LocalMatrix_, typename RowMapping_, typename ColMapping_>
187 void operator()(const LocalMatrix_& loc_mat, const RowMapping_& row_map,
188 const ColMapping_& col_map, DT_ alpha = DT_(1))
189 {
190 // loop over all local row entries
191 for(int i(0); i < row_map.get_num_local_dofs(); ++i)
192 {
193 // fetch row index
194 const Index ix = row_map.get_index(i);
195
196 // build column pointer for this row entry contribution
197 for(IT_ k(_row_ptr[ix]); k < _row_ptr[ix + 1]; ++k)
198 {
199 _col_ptr[_col_idx[k]] = k;
200 }
201
202 // loop over all local column entries
203 for(int j(0); j < col_map.get_num_local_dofs(); ++j)
204 {
205 // fetch column index
206 const Index jx = col_map.get_index(j);
207
208 // ensure that the column pointer is valid for this index
209 ASSERTM(_col_ptr[jx] != _deadcode, "invalid column index");
210
211 // incorporate data into global matrix
212 _data[_col_ptr[jx]] += alpha * loc_mat[i][j];
213
214 // continue with next column entry
215 }
216
217#ifdef DEBUG
218 // reformat column-pointer array
219 for(IT_ k(_row_ptr[ix]); k < _row_ptr[ix + 1]; ++k)
220 {
221 _col_ptr[_col_idx[k]] = _deadcode;
222 }
223#endif
224 // continue with next row entry
225 }
226 }

343 explicit SparseMatrixBCSR(const Adjacency::Graph & graph) :
344 Container<DT_, IT_>(0)
345 {
346 Index num_rows = graph.get_num_nodes_domain();
347 Index num_cols = graph.get_num_nodes_image();
348 Index num_nnze = graph.get_num_indices();
349
350 if (num_nnze == 0)
351 {
352 this->move(SparseMatrixBCSR(num_rows, num_cols));
353 return;
354 }
355
356 // Create temporary vectors. Row and column pointer are block wise
357 LAFEM::DenseVector<IT_, IT_> vrow_ptr(num_rows+1);
358 LAFEM::DenseVector<IT_, IT_> vcol_idx(num_nnze);
359 // The data array has to account for the block size
360 LAFEM::DenseVector<DT_, IT_> vdata(num_nnze*Index(BlockHeight_*BlockWidth_), DT_(0));
361
362 const Index * dom_ptr(graph.get_domain_ptr());
363 const Index * img_idx(graph.get_image_idx());
364 IT_ * prow_ptr(vrow_ptr.elements());
365 IT_ * pcol_idx(vcol_idx.elements());
366
367 // build row-end
368 prow_ptr[0] = IT_(dom_ptr[0]);
369 for(Index i(0); i < num_rows; ++i)
370 prow_ptr[i+1] = IT_(dom_ptr[i+1]);
371
372 // build col-idx
373 for(Index i(0); i < num_nnze; ++i)
374 pcol_idx[i] = IT_(img_idx[i]);
375
376 // build the matrix
377 this->move(SparseMatrixBCSR<DT_, IT_, BlockHeight_, BlockWidth_>(num_rows, num_cols, vcol_idx, vdata, vrow_ptr));
378 }

1200 void transpose(const SparseMatrixBCSR<DT_, IT_, BlockWidth_, BlockHeight_> & x)
1201 {
1202 using XType = SparseMatrixBCSR<DT_, IT_, BlockWidth_, BlockHeight_>;
1203 if (x.used_elements() == 0)
1204 {
1205 this->move(SparseMatrixBCSR(x.rows(), x.columns()));
1206 return;
1207 }
1208
1209 const Index txrows(x.rows());
1210 const Index txcolumns(x.columns());
1211 const Index txused_elements(x.used_elements());
1212
1213 const IT_ * ptxcol_ind(x.col_ind());
1214 const IT_ * ptxrow_ptr(x.row_ptr());
1215 const typename XType::ValueType * ptxval(x.val());
1216
1217 DenseVector<IT_, IT_> tcol_ind(txused_elements);
1218 DenseVector<DT_, IT_> tval(txused_elements * BlockHeight_ * BlockWidth_);
1219 DenseVector<IT_, IT_> trow_ptr(txcolumns + 1, IT_(0));
1220
1221 IT_ * ptcol_ind(tcol_ind.elements());
1222 ValueType * ptval((ValueType*)tval.elements());
1223 IT_ * ptrow_ptr(trow_ptr.elements());
1224
1225 ptrow_ptr[0] = 0;
1226
1227 for (Index i(0); i < txused_elements; ++i)
1228 {
1229 ++ptrow_ptr[ptxcol_ind[i] + 1];
1230 }
1231
1232 for (Index i(1); i < txcolumns - 1; ++i)
1233 {
1234 ptrow_ptr[i + 1] += ptrow_ptr[i];
1235 }
1236
1237 for (Index i(0); i < txrows; ++i)
1238 {
1239 for (IT_ k(ptxrow_ptr[i]); k < ptxrow_ptr[i+1]; ++k)
1240 {
1241 const IT_ l(ptxcol_ind[k]);
1242 const IT_ j(ptrow_ptr[l]);
1243 ptval[j].set_transpose(ptxval[k]);
1244 ptcol_ind[j] = IT_(i);
1245 ++ptrow_ptr[l];
1246 }
1247 }
1248
1249 for (Index i(txcolumns); i > 0; --i)
1250 {
1251 ptrow_ptr[i] = ptrow_ptr[i - 1];
1252 }
1253 ptrow_ptr[0] = 0;
1254
1255 this->move(SparseMatrixBCSR<DT_, IT_, BlockHeight_, BlockWidth_> (txcolumns, txrows, tcol_ind, tval, trow_ptr));
1256 }

2026 void permute(Adjacency::Permutation & perm_row, Adjacency::Permutation & perm_col)
2027 {
2028 if (perm_row.size() == 0 && perm_col.size() == 0)
2029 return;
2030
2031 XASSERTM(perm_row.size() == this->rows(), "Container rows does not match permutation size");
2032 XASSERTM(perm_col.size() == this->columns(), "Container columns does not match permutation size");
2033
2034 // http://de.mathworks.com/help/matlab/math/sparse-matrix-operations.html#f6-13070
2035 IT_ * temp_row_ptr = new IT_[rows() + 1];
2036 IT_ * temp_col_ind = new IT_[used_elements()];
2037 ValueType * temp_val = new ValueType[used_elements()];
2038
2039 Index * perm_pos;
2040 perm_pos = perm_row.get_perm_pos();
2041
2042 //permute rows from source to temp_*
2043 Index new_start(0);
2044 temp_row_ptr[0] = 0;
2045 for (Index row(0) ; row < this->rows() ; ++row)
2046 {
2047 Index row_size(this->row_ptr()[perm_pos[row] + 1] - this->row_ptr()[perm_pos[row]]);
2048
2049 //iterate over all elements in single one new and old row
2050 for (Index i(new_start), j(this->row_ptr()[perm_pos[row]]) ; i < new_start + row_size ; ++i, ++j)
2051 {
2052 temp_col_ind[i] = this->col_ind()[j];
2053 temp_val[i] = this->val()[j];
2054 }
2055
2056 new_start += row_size;
2057 temp_row_ptr[row+1] = (IT_)new_start;
2058 }
2059
2060 //use inverse col permutation as lookup table: i -> new location of i
2061 Adjacency::Permutation perm_col_inv = perm_col.inverse();
2062 perm_pos = perm_col_inv.get_perm_pos();
2063
2064 //permute columns from temp_* to source
2065 ::memcpy(this->row_ptr(), temp_row_ptr, (rows() + 1) * sizeof(IT_));
2066 ::memcpy(this->val(), temp_val, used_elements() * sizeof(ValueType));
2067 for (Index i(0) ; i < used_elements() ; ++i)
2068 {
2069 this->col_ind()[i] = (IT_)perm_pos[temp_col_ind[i]];
2070 }
2071
2072 delete[] temp_row_ptr;
2073 delete[] temp_col_ind;
2074 delete[] temp_val;
2075
2076 //sort columns in every row by column index
2077 IT_ swap_key;
2078 ValueType swap_val;
2079 for (Index row(0) ; row < rows() ; ++row)
2080 {
2081 Index offset(this->row_ptr()[row]);
2082 Index row_size(this->row_ptr()[row+1] - this->row_ptr()[row]);
2083 for (Index i(1), j ; i < row_size ; ++i)
2084 {
2085 swap_key = this->col_ind()[i + offset];
2086 swap_val = this->val()[i + offset];
2087 j = i;
2088 while (j > 0 && this->col_ind()[j - 1 + offset] > swap_key)
2089 {
2090 this->col_ind()[j + offset] = this->col_ind()[j - 1 + offset];
2091 this->val()[j + offset] = this->val()[j - 1 + offset];
2092 --j;
2093 }
2094 this->col_ind()[j + offset] = swap_key;
2095 this->val()[j + offset] = swap_val;
2096 }
2097 }
