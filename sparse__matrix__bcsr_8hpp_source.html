186 template<typename LocalMatrix_, typename RowMapping_, typename ColMapping_>
187 void operator()(const LocalMatrix_& loc_mat, const RowMapping_& row_map,
188 const ColMapping_& col_map, DT_ alpha = DT_(1))
189 {
190 // loop over all local row entries
191 for(int i(0); i < row_map.get_num_local_dofs(); ++i)
192 {
193 // fetch row index
194 const Index ix = row_map.get_index(i);
195
196 // build column pointer for this row entry contribution
197 for(IT_ k(_row_ptr[ix]); k < _row_ptr[ix + 1]; ++k)
198 {
199 _col_ptr[_col_idx[k]] = k;
200 }
201
202 // loop over all local column entries
203 for(int j(0); j < col_map.get_num_local_dofs(); ++j)
204 {
205 // fetch column index
206 const Index jx = col_map.get_index(j);
207
208 // ensure that the column pointer is valid for this index
209 ASSERTM(_col_ptr[jx] != _deadcode, "invalid column index");
210
211 // incorporate data into global matrix
212 _data[_col_ptr[jx]] += alpha * loc_mat[i][j];
213
214 // continue with next column entry
215 }
216
217#ifdef DEBUG
218 // reformat column-pointer array
219 for(IT_ k(_row_ptr[ix]); k < _row_ptr[ix + 1]; ++k)
220 {
221 _col_ptr[_col_idx[k]] = _deadcode;
222 }
223#endif
224 // continue with next row entry
225 }
226 }

343 explicit SparseMatrixBCSR(const Adjacency::Graph & graph) :
344 Container<DT_, IT_>(0)
345 {
346 Index num_rows = graph.get_num_nodes_domain();
347 Index num_cols = graph.get_num_nodes_image();
348 Index num_nnze = graph.get_num_indices();
349
350 if (num_nnze == 0)
351 {
352 this->move(SparseMatrixBCSR(num_rows, num_cols));
353 return;
354 }
355
356 // Create temporary vectors. Row and column pointer are block wise
357 LAFEM::DenseVector<IT_, IT_> vrow_ptr(num_rows+1);
358 LAFEM::DenseVector<IT_, IT_> vcol_idx(num_nnze);
359 // The data array has to account for the block size
360 LAFEM::DenseVector<DT_, IT_> vdata(num_nnze*Index(BlockHeight_*BlockWidth_), DT_(0));
361
362 const Index * dom_ptr(graph.get_domain_ptr());
363 const Index * img_idx(graph.get_image_idx());
364 IT_ * prow_ptr(vrow_ptr.elements());
365 IT_ * pcol_idx(vcol_idx.elements());
366
367 // build row-end
368 prow_ptr[0] = IT_(dom_ptr[0]);
369 for(Index i(0); i < num_rows; ++i)
370 prow_ptr[i+1] = IT_(dom_ptr[i+1]);
371
372 // build col-idx
373 for(Index i(0); i < num_nnze; ++i)
374 pcol_idx[i] = IT_(img_idx[i]);
375
376 // build the matrix
377 this->move(SparseMatrixBCSR<DT_, IT_, BlockHeight_, BlockWidth_>(num_rows, num_cols, vcol_idx, vdata, vrow_ptr));
378 }

1278 void transpose(const SparseMatrixBCSR<DT_, IT_, BlockWidth_, BlockHeight_> & x)
1279 {
1280 using XType = SparseMatrixBCSR<DT_, IT_, BlockWidth_, BlockHeight_>;
1281 if (x.used_elements() == 0)
1282 {
1283 this->move(SparseMatrixBCSR(x.rows(), x.columns()));
1284 return;
1285 }
1286
1287 const Index txrows(x.rows());
1288 const Index txcolumns(x.columns());
1289 const Index txused_elements(x.used_elements());
1290
1291 const IT_ * ptxcol_ind(x.col_ind());
1292 const IT_ * ptxrow_ptr(x.row_ptr());
1293 const typename XType::ValueType * ptxval(x.val());
1294
1295 DenseVector<IT_, IT_> tcol_ind(txused_elements);
1296 DenseVector<DT_, IT_> tval(txused_elements * BlockHeight_ * BlockWidth_);
1297 DenseVector<IT_, IT_> trow_ptr(txcolumns + 1, IT_(0));
1298
1299 IT_ * ptcol_ind(tcol_ind.elements());
1300 ValueType * ptval((ValueType*)tval.elements());
1301 IT_ * ptrow_ptr(trow_ptr.elements());
1302
1303 ptrow_ptr[0] = 0;
1304
1305 for (Index i(0); i < txused_elements; ++i)
1306 {
1307 ++ptrow_ptr[ptxcol_ind[i] + 1];
1308 }
1309
1310 for (Index i(1); i < txcolumns - 1; ++i)
1311 {
1312 ptrow_ptr[i + 1] += ptrow_ptr[i];
1313 }
1314
1315 for (Index i(0); i < txrows; ++i)
1316 {
1317 for (IT_ k(ptxrow_ptr[i]); k < ptxrow_ptr[i+1]; ++k)
1318 {
1319 const IT_ l(ptxcol_ind[k]);
1320 const IT_ j(ptrow_ptr[l]);
1321 ptval[j].set_transpose(ptxval[k]);
1322 ptcol_ind[j] = IT_(i);
1323 ++ptrow_ptr[l];
1324 }
1325 }
1326
1327 for (Index i(txcolumns); i > 0; --i)
1328 {
1329 ptrow_ptr[i] = ptrow_ptr[i - 1];
1330 }
1331 ptrow_ptr[0] = 0;
1332
1333 this->move(SparseMatrixBCSR<DT_, IT_, BlockHeight_, BlockWidth_> (txcolumns, txrows, tcol_ind, tval, trow_ptr));
1334 }

2459 IT_ * temp_col_ind = new IT_[used_elements()];

2463 perm_pos = perm_row.get_perm_pos();
2464
2465 //permute rows from source to temp_*
2466 Index new_start(0);
2467 temp_row_ptr[0] = 0;
2468 for (Index row(0) ; row < this->rows() ; ++row)
2469 {
2470 Index row_size(this->row_ptr()[perm_pos[row] + 1] - this->row_ptr()[perm_pos[row]]);
2471
2472 //iterate over all elements in single one new and old row
2473 for (Index i(new_start), j(this->row_ptr()[perm_pos[row]]) ; i < new_start + row_size ; ++i, ++j)
2474 {
2475 temp_col_ind[i] = this->col_ind()[j];
2476 temp_val[i] = this->val()[j];
2477 }
2478
2479 new_start += row_size;
2480 temp_row_ptr[row+1] = (IT_)new_start;
2481 }
2482

2484 Adjacency::Permutation perm_col_inv = perm_col.inverse();
2485 perm_pos = perm_col_inv.get_perm_pos();
2486
2487 //permute columns from temp_* to source
2488 ::memcpy(this->row_ptr(), temp_row_ptr, (rows() + 1) * sizeof(IT_));
2489 ::memcpy(this->val(), temp_val, used_elements() * sizeof(ValueType));
2490 for (Index i(0) ; i < used_elements() ; ++i)
2491 {
2492 this->col_ind()[i] = (IT_)perm_pos[temp_col_ind[i]];
2493 }
2494

2496 delete[] temp_col_ind;
2497 delete[] temp_val;

2499 //sort columns in every row by column index
2500 IT_ swap_key;
2501 ValueType swap_val;
2502 for (Index row(0) ; row < rows() ; ++row)
2503 {
2504 Index offset(this->row_ptr()[row]);
2505 Index row_size(this->row_ptr()[row+1] - this->row_ptr()[row]);
2506 for (Index i(1), j ; i < row_size ; ++i)
2507 {
2508 swap_key = this->col_ind()[i + offset];
2509 swap_val = this->val()[i + offset];

2512 {

2514 this->val()[j + offset] = this->val()[j - 1 + offset];
2515 --j;
2516 }
2517 this->col_ind()[j + offset] = swap_key;
2518 this->val()[j + offset] = swap_val;
2519 }

2562 void add_trace_double_mat_mult(
2563 typename LAFEM::SparseMatrixBCSR< DT_, IT_, BHD_, BlockHeight>::VectorTypeL& v,

2580 XASSERT(d.columns() == a.size());
2581 XASSERT(a.size() == b.rows());
2582 XASSERT(b.columns() == this->columns());

2584 // Fetch matrix arrays
2585 // Use POD here to avoid hassle with 1x1 matrices and stuff
2586 DT_* data_v = v.template elements<LAFEM::Perspective::pod>();
2587 const DT_* data_d = d.template val<LAFEM::Perspective::pod>();
2588 const DT_* data_a = a.template elements<LAFEM::Perspective::pod>();
2589 const DT_* data_b = b.template val<LAFEM::Perspective::pod>();

2595 // Loop over all rows i of v
2596 for(IT_ row(0); row < IT_(v.size()); ++row)
2597 {
2598 // For all non-zeros D_ik in row i of D
2599 for(IT_ pos_d(row_ptr_d[row]); pos_d < row_ptr_d[row+1]; ++pos_d)
2600 {
2601 // Get column index k
2602 const IT_ col_d(col_idx_d[pos_d]);
2603
2604 // Pre-compute block factor (alpha * D_ik * A_kk)
2605 typename MatrixD::ValueType omega(0);
2606 for(IT_ i(0); i < bhd; ++i)
2607 {
2608 for(IT_ j(0); j < bwd; ++j)
2609 {
2610 omega[int(i)][int(j)] = alpha * data_d[bhd*bwd*pos_d + bwd*i + j] * data_a[bwd*col_d + j];
2611 }
2612 }
2613
2614 // Loop over all non-zero blocks B_kj in row j of B and perform a "sparse axpy" of B_l onto v_i, i.e.:
2615 // v_i += alpha * (D_ik * A_kk) * B_k.
2616 IT_ pos_b(row_ptr_b[col_d]);
2617 while(pos_b < row_ptr_b[col_d+1])
2618 {
2619 if(col_idx_b[pos_b] == row)
2620 {
2621 // B_kj contributes to v_i here
2622 for(IT_ i(0); i < bhd; ++i)
2623 {
2624 for(IT_ j(0); j < bh; ++j)
2625 {
2626 data_v[bhd*row + i] += omega[int(i)][int(j)]*data_b[bh*bw*pos_b + bw*j + i];
2627 }
2628 }
2629 break;
2630 }
2631 ++pos_b;
2632 }
2633 }
2634 }
