184 template<typename LocalMatrix_, typename RowMapping_, typename ColMapping_>
185 void operator()(const LocalMatrix_& loc_mat, const RowMapping_& row_map,
186 const ColMapping_& col_map, DT_ alpha = DT_(1))
187 {
188 // loop over all local row entries
189 for(int i(0); i < row_map.get_num_local_dofs(); ++i)
190 {
191 // fetch row index
192 const Index ix = row_map.get_index(i);
193
194 // build column pointer for this row entry contribution
195 for(IT_ k(_row_ptr[ix]); k < _row_ptr[ix + 1]; ++k)
196 {
197 _col_ptr[_col_idx[k]] = k;
198 }
199
200 // loop over all local column entries
201 for(int j(0); j < col_map.get_num_local_dofs(); ++j)
202 {
203 // fetch column index
204 const Index jx = col_map.get_index(j);
205
206 // ensure that the column pointer is valid for this index
207 ASSERTM(_col_ptr[jx] != _deadcode, "invalid column index");
208
209 // incorporate data into global matrix
210 _data[_col_ptr[jx]] += alpha * loc_mat[i][j];
211
212 // continue with next column entry
213 }
214
215#ifdef DEBUG
216 // reformat column-pointer array
217 for(IT_ k(_row_ptr[ix]); k < _row_ptr[ix + 1]; ++k)
218 {
219 _col_ptr[_col_idx[k]] = _deadcode;
220 }
221#endif
222 // continue with next row entry
223 }
224 }

1258 void transpose(const SparseMatrixBCSR<DT_, IT_, BlockWidth_, BlockHeight_> & x)
1259 {
1260 using XType = SparseMatrixBCSR<DT_, IT_, BlockWidth_, BlockHeight_>;
1261 if (x.used_elements() == 0)
1262 {
1263 this->move(SparseMatrixBCSR(x.rows(), x.columns()));
1264 return;
1265 }
1266
1267 const Index txrows(x.rows());
1268 const Index txcolumns(x.columns());
1269 const Index txused_elements(x.used_elements());
1270
1271 const IT_ * ptxcol_ind(x.col_ind());
1272 const IT_ * ptxrow_ptr(x.row_ptr());
1273 const typename XType::ValueType * ptxval(x.val());
1274
1275 DenseVector<IT_, IT_> tcol_ind(txused_elements);
1276 DenseVector<DT_, IT_> tval(txused_elements * BlockHeight_ * BlockWidth_);
1277 DenseVector<IT_, IT_> trow_ptr(txcolumns + 1, IT_(0));
1278
1279 IT_ * ptcol_ind(tcol_ind.elements());
1280 ValueType * ptval((ValueType*)tval.elements());
1281 IT_ * ptrow_ptr(trow_ptr.elements());
1282
1283 ptrow_ptr[0] = 0;
1284
1285 for (Index i(0); i < txused_elements; ++i)
1286 {
1287 ++ptrow_ptr[ptxcol_ind[i] + 1];
1288 }
1289
1290 for (Index i(1); i < txcolumns - 1; ++i)
1291 {
1292 ptrow_ptr[i + 1] += ptrow_ptr[i];
1293 }
1294
1295 for (Index i(0); i < txrows; ++i)
1296 {
1297 for (IT_ k(ptxrow_ptr[i]); k < ptxrow_ptr[i+1]; ++k)
1298 {
1299 const IT_ l(ptxcol_ind[k]);
1300 const IT_ j(ptrow_ptr[l]);
1301 ptval[j].set_transpose(ptxval[k]);
1302 ptcol_ind[j] = IT_(i);
1303 ++ptrow_ptr[l];
1304 }
1305 }
1306
1307 for (Index i(txcolumns); i > 0; --i)
1308 {
1309 ptrow_ptr[i] = ptrow_ptr[i - 1];
1310 }
1311 ptrow_ptr[0] = 0;
1312
1313 this->move(SparseMatrixBCSR<DT_, IT_, BlockHeight_, BlockWidth_> (txcolumns, txrows, tcol_ind, tval, trow_ptr));
1314 }

2099 void add_double_mat_product(
2100 const LAFEM::SparseMatrixBCSR<DT_, IT_, BlockHeight_, BlockWidth_>& d,
2101 const LAFEM::SparseMatrixBCSR<DT_, IT_, BlockHeight_, BlockWidth_>& a,
2102 const LAFEM::SparseMatrixBCSR<DT_, IT_, BlockHeight_, BlockWidth_>& b,
2103 const DT_ alpha = DT_(1),
2104 const bool allow_incomplete = false)
2105 {
2106 // validate matrix dimensions
2107 XASSERT(BlockHeight_ == BlockWidth_);
2108 XASSERT(this->rows() == d.rows());
2109 XASSERT(d.columns() == a.rows());
2110 XASSERT(a.columns() == b.rows());
2111 XASSERT(b.columns() == this->columns());
2112
2113 // fetch matrix arrays:
2114 ValueType* data_x = this->val();
2115 const ValueType* data_d = d.val();
2116 const ValueType* data_a = a.val();
2117 const ValueType* data_b = b.val();
2118 const IT_* row_ptr_x = this->row_ptr();
2119 const IT_* col_idx_x = this->col_ind();
2120 const IT_* row_ptr_d = d.row_ptr();
2121 const IT_* col_idx_d = d.col_ind();
2122 const IT_* row_ptr_a = a.row_ptr();
2123 const IT_* col_idx_a = a.col_ind();
2124 const IT_* row_ptr_b = b.row_ptr();
2125 const IT_* col_idx_b = b.col_ind();
2126
2127 // loop over all rows of D and X, resp.
2128 for(IT_ i(0); i < IT_(this->rows()); ++i)
2129 {
2130 // loop over all non-zeros D_ik in row i of D
2131 for(IT_ ik(row_ptr_d[i]); ik < row_ptr_d[i+1]; ++ik)
2132 {
2133 // get column index k
2134 const IT_ k = col_idx_d[ik];
2135
2136 // loop over all non-zeros A_kl in row k of A
2137 for(IT_ kl(row_ptr_a[k]); kl < row_ptr_a[k+1]; ++kl)
2138 {
2139 // get column index l
2140 const IT_ l = col_idx_a[kl];
2141
2142 // pre-compute factor (alpha * D_ik * A_kl)
2143 ValueType omega;
2144 omega.set_mat_mat_mult(data_d[ik], data_a[kl]);
2145 omega *= alpha;
2146
2147 // loop over all non-zeros B_lj in row j of B and
2148 // loop over all non-zeros X_ij in row i of X and
2149 // perform a "sparse axpy" of B_l onto X_i, i.e.:
2150 // X_i. += (alpha * D_ik * A_kl) * B_l.
2151 IT_ ij = row_ptr_x[i];
2152 IT_ lj = row_ptr_b[l];

2161 break; // continue with next row

2163 XABORTM("Incomplete output matrix structure");
2164 }

2166 {
2167 // okay: B_lj contributes to X_ij
2168 ValueType temp;
2169 temp.set_mat_mat_mult(omega, data_b[lj]);
2170 data_x[ij] += temp;
2171 ++ij;
2172 ++lj;
2173 }
2174 else if(col_idx_x[ij] < col_idx_b[lj])
2175 {
2176 // entry X_ij exists, but B_lj is missing:
2177 // this is a perfectly valid case, so continue with the next non-zero of X_i
2178 ++ij;
2179 }
2180 else //if(col_idx_x[ij] > col_idx_b[lj])
2181 {
2182 // If we come out here, then the sparsity pattern of X is incomplete:
