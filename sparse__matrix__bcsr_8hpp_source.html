186 template<typename LocalMatrix_, typename RowMapping_, typename ColMapping_>
187 void operator()(const LocalMatrix_& loc_mat, const RowMapping_& row_map,
188 const ColMapping_& col_map, DT_ alpha = DT_(1))
189 {
190 // loop over all local row entries
191 for(int i(0); i < row_map.get_num_local_dofs(); ++i)
192 {
193 // fetch row index
194 const Index ix = row_map.get_index(i);
195
196 // build column pointer for this row entry contribution
197 for(IT_ k(_row_ptr[ix]); k < _row_ptr[ix + 1]; ++k)
198 {
199 _col_ptr[_col_idx[k]] = k;
200 }
201
202 // loop over all local column entries
203 for(int j(0); j < col_map.get_num_local_dofs(); ++j)
204 {
205 // fetch column index
206 const Index jx = col_map.get_index(j);
207
208 // ensure that the column pointer is valid for this index
209 ASSERTM(_col_ptr[jx] != _deadcode, "invalid column index");
210
211 // incorporate data into global matrix
212 _data[_col_ptr[jx]] += alpha * loc_mat[i][j];
213
214 // continue with next column entry
215 }
216
217#ifdef DEBUG
218 // reformat column-pointer array
219 for(IT_ k(_row_ptr[ix]); k < _row_ptr[ix + 1]; ++k)
220 {
221 _col_ptr[_col_idx[k]] = _deadcode;
222 }
223#endif
224 // continue with next row entry
225 }
226 }

343 explicit SparseMatrixBCSR(const Adjacency::Graph & graph) :
344 Container<DT_, IT_>(0)
345 {
346 Index num_rows = graph.get_num_nodes_domain();
347 Index num_cols = graph.get_num_nodes_image();
348 Index num_nnze = graph.get_num_indices();
349
350 if (num_nnze == 0)
351 {
352 this->move(SparseMatrixBCSR(num_rows, num_cols));
353 return;
354 }
355
356 // Create temporary vectors. Row and column pointer are block wise
357 LAFEM::DenseVector<IT_, IT_> vrow_ptr(num_rows+1);
358 LAFEM::DenseVector<IT_, IT_> vcol_idx(num_nnze);
359 // The data array has to account for the block size
360 LAFEM::DenseVector<DT_, IT_> vdata(num_nnze*Index(BlockHeight_*BlockWidth_), DT_(0));
361
362 const Index * dom_ptr(graph.get_domain_ptr());
363 const Index * img_idx(graph.get_image_idx());
364 IT_ * prow_ptr(vrow_ptr.elements());
365 IT_ * pcol_idx(vcol_idx.elements());
366
367 // build row-end
368 prow_ptr[0] = IT_(dom_ptr[0]);
369 for(Index i(0); i < num_rows; ++i)
370 prow_ptr[i+1] = IT_(dom_ptr[i+1]);
371
372 // build col-idx
373 for(Index i(0); i < num_nnze; ++i)
374 pcol_idx[i] = IT_(img_idx[i]);
375
376 // build the matrix
377 this->move(SparseMatrixBCSR<DT_, IT_, BlockHeight_, BlockWidth_>(num_rows, num_cols, vcol_idx, vdata, vrow_ptr));
378 }

1290 void transpose(const SparseMatrixBCSR<DT_, IT_, BlockWidth_, BlockHeight_> & x)
1291 {
1292 using XType = SparseMatrixBCSR<DT_, IT_, BlockWidth_, BlockHeight_>;
1293 if (x.used_elements() == 0)
1294 {
1295 this->move(SparseMatrixBCSR(x.rows(), x.columns()));
1296 return;
1297 }
1298
1299 const Index txrows(x.rows());
1300 const Index txcolumns(x.columns());
1301 const Index txused_elements(x.used_elements());
1302
1303 const IT_ * ptxcol_ind(x.col_ind());
1304 const IT_ * ptxrow_ptr(x.row_ptr());
1305 const typename XType::ValueType * ptxval(x.val());
1306
1307 DenseVector<IT_, IT_> tcol_ind(txused_elements);
1308 DenseVector<DT_, IT_> tval(txused_elements * BlockHeight_ * BlockWidth_);
1309 DenseVector<IT_, IT_> trow_ptr(txcolumns + 1, IT_(0));
1310
1311 IT_ * ptcol_ind(tcol_ind.elements());
1312 ValueType * ptval((ValueType*)tval.elements());
1313 IT_ * ptrow_ptr(trow_ptr.elements());
1314
1315 ptrow_ptr[0] = 0;
1316
1317 for (Index i(0); i < txused_elements; ++i)
1318 {
1319 ++ptrow_ptr[ptxcol_ind[i] + 1];
1320 }
1321
1322 for (Index i(1); i < txcolumns - 1; ++i)
1323 {
1324 ptrow_ptr[i + 1] += ptrow_ptr[i];
1325 }
1326
1327 for (Index i(0); i < txrows; ++i)
1328 {
1329 for (IT_ k(ptxrow_ptr[i]); k < ptxrow_ptr[i+1]; ++k)
1330 {
1331 const IT_ l(ptxcol_ind[k]);
1332 const IT_ j(ptrow_ptr[l]);
1333 ptval[j].set_transpose(ptxval[k]);
1334 ptcol_ind[j] = IT_(i);
1335 ++ptrow_ptr[l];
1336 }
1337 }
1338
1339 for (Index i(txcolumns); i > 0; --i)
1340 {
1341 ptrow_ptr[i] = ptrow_ptr[i - 1];
1342 }
1343 ptrow_ptr[0] = 0;
1344
1345 this->move(SparseMatrixBCSR<DT_, IT_, BlockHeight_, BlockWidth_> (txcolumns, txrows, tcol_ind, tval, trow_ptr));
1346 }

1775 void add_double_mat_product(
1776 const LAFEM::SparseMatrixBCSR<DT_, IT_, BlockHeight_, BlockWidth_>& d,
1777 const LAFEM::SparseMatrixBCSR<DT_, IT_, BlockHeight_, BlockWidth_>& a,
1778 const LAFEM::SparseMatrixBCSR<DT_, IT_, BlockHeight_, BlockWidth_>& b,
1779 const DT_ alpha = DT_(1),
1780 const bool allow_incomplete = false)
1781 {
1782 // validate matrix dimensions
1783 XASSERT(BlockHeight_ == BlockWidth_);

1785 XASSERT(d.columns() == a.rows());
1786 XASSERT(a.columns() == b.rows());
1787 XASSERT(b.columns() == this->columns());
1788
1789 // fetch matrix arrays:
1790 ValueType* data_x = this->val();
1791 const ValueType* data_d = d.val();
1792 const ValueType* data_a = a.val();
1793 const ValueType* data_b = b.val();
1794 const IT_* row_ptr_x = this->row_ptr();
1795 const IT_* col_idx_x = this->col_ind();
1796 const IT_* row_ptr_d = d.row_ptr();
1797 const IT_* col_idx_d = d.col_ind();
1798 const IT_* row_ptr_a = a.row_ptr();
1799 const IT_* col_idx_a = a.col_ind();
1800 const IT_* row_ptr_b = b.row_ptr();
1801 const IT_* col_idx_b = b.col_ind();
1802
1803 // loop over all rows of D and X, resp.
1804 for(IT_ i(0); i < IT_(this->rows()); ++i)
1805 {
1806 // loop over all non-zeros D_ik in row i of D
1807 for(IT_ ik(row_ptr_d[i]); ik < row_ptr_d[i+1]; ++ik)
1808 {
1809 // get column index k
1810 const IT_ k = col_idx_d[ik];
1811
1812 // loop over all non-zeros A_kl in row k of A
1813 for(IT_ kl(row_ptr_a[k]); kl < row_ptr_a[k+1]; ++kl)
1814 {
1815 // get column index l
1816 const IT_ l = col_idx_a[kl];
1817
1818 // pre-compute factor (alpha * D_ik * A_kl)
1819 ValueType omega;
1820 omega.set_mat_mat_mult(data_d[ik], data_a[kl]);
1821 omega *= alpha;
1822
1823 // loop over all non-zeros B_lj in row j of B and
1824 // loop over all non-zeros X_ij in row i of X and
1825 // perform a "sparse axpy" of B_l onto X_i, i.e.:
1826 // X_i. += (alpha * D_ik * A_kl) * B_l.
1827 IT_ ij = row_ptr_x[i];
1828 IT_ lj = row_ptr_b[l];
1829 while(lj < row_ptr_b[l+1])
1830 {
1831 if(ij >= row_ptr_x[i+1])
1832 {
1833 // we have reached the end of row X_i, but there is at least
1834 // one entry in row B_l left, so the pattern of X is incomplete
1835 // We let the caller decide whether this is a valid case or not:
1836 if(allow_incomplete)
1837 break; // continue with next row
1838 else
1839 XABORTM("Incomplete output matrix structure");
1840 }
1841 else if(col_idx_x[ij] == col_idx_b[lj])
1842 {
1843 // okay: B_lj contributes to X_ij
1844 ValueType temp;
1845 temp.set_mat_mat_mult(omega, data_b[lj]);
1846 data_x[ij] += temp;
1847 ++ij;
1848 ++lj;
1849 }
1850 else if(col_idx_x[ij] < col_idx_b[lj])
1851 {
1852 // entry X_ij exists, but B_lj is missing:
1853 // this is a perfectly valid case, so continue with the next non-zero of X_i
1854 ++ij;
1855 }
1856 else //if(col_idx_x[ij] > col_idx_b[lj])
1857 {
1858 // If we come out here, then the sparsity pattern of X is incomplete:
1859 // B_lj is meant to be added onto X_ij, but the entry X_ij is missing
1860 // We let the caller decide whether this is a valid case or not:
1861 if(allow_incomplete)
1862 ++lj;
1863 else
1864 XABORTM("Incomplete output matrix structure");
1865 }
1866 }

2105 void permute(Adjacency::Permutation & perm_row, Adjacency::Permutation & perm_col)
2106 {
2107 if (perm_row.size() == 0 && perm_col.size() == 0)
2108 return;
2109
2110 XASSERTM(perm_row.size() == this->rows(), "Container rows does not match permutation size");
2111 XASSERTM(perm_col.size() == this->columns(), "Container columns does not match permutation size");
2112
2113 // http://de.mathworks.com/help/matlab/math/sparse-matrix-operations.html#f6-13070
2114 IT_ * temp_row_ptr = new IT_[rows() + 1];
2115 IT_ * temp_col_ind = new IT_[used_elements()];
2116 ValueType * temp_val = new ValueType[used_elements()];
2117
2118 Index * perm_pos;
2119 perm_pos = perm_row.get_perm_pos();
2120
2121 //permute rows from source to temp_*
2122 Index new_start(0);
2123 temp_row_ptr[0] = 0;
2124 for (Index row(0) ; row < this->rows() ; ++row)
2125 {
2126 Index row_size(this->row_ptr()[perm_pos[row] + 1] - this->row_ptr()[perm_pos[row]]);
2127
2128 //iterate over all elements in single one new and old row
2129 for (Index i(new_start), j(this->row_ptr()[perm_pos[row]]) ; i < new_start + row_size ; ++i, ++j)
2130 {
2131 temp_col_ind[i] = this->col_ind()[j];
2132 temp_val[i] = this->val()[j];
2133 }
2134
2135 new_start += row_size;
2136 temp_row_ptr[row+1] = (IT_)new_start;
2137 }
2138
2139 //use inverse col permutation as lookup table: i -> new location of i
2140 Adjacency::Permutation perm_col_inv = perm_col.inverse();
2141 perm_pos = perm_col_inv.get_perm_pos();
2142
2143 //permute columns from temp_* to source
2144 ::memcpy(this->row_ptr(), temp_row_ptr, (rows() + 1) * sizeof(IT_));
2145 ::memcpy(this->val(), temp_val, used_elements() * sizeof(ValueType));
2146 for (Index i(0) ; i < used_elements() ; ++i)
2147 {
2148 this->col_ind()[i] = (IT_)perm_pos[temp_col_ind[i]];
2149 }
2150
2151 delete[] temp_row_ptr;
2152 delete[] temp_col_ind;
2153 delete[] temp_val;
2154
2155 //sort columns in every row by column index
2156 IT_ swap_key;
2157 ValueType swap_val;
2158 for (Index row(0) ; row < rows() ; ++row)
2159 {
2160 Index offset(this->row_ptr()[row]);
2161 Index row_size(this->row_ptr()[row+1] - this->row_ptr()[row]);
2162 for (Index i(1), j ; i < row_size ; ++i)
2163 {
2164 swap_key = this->col_ind()[i + offset];
2165 swap_val = this->val()[i + offset];
2166 j = i;
2167 while (j > 0 && this->col_ind()[j - 1 + offset] > swap_key)
2168 {
2169 this->col_ind()[j + offset] = this->col_ind()[j - 1 + offset];
2170 this->val()[j + offset] = this->val()[j - 1 + offset];
2171 --j;
2172 }
2173 this->col_ind()[j + offset] = swap_key;
2174 this->val()[j + offset] = swap_val;
2175 }
2176 }

2218 void add_trace_double_mat_mult(
2219 typename LAFEM::SparseMatrixBCSR< DT_, IT_, BHD_, BlockHeight>::VectorTypeL& v,
2220 const LAFEM::SparseMatrixBCSR<DT_, IT_, BHD_, BlockHeight>& d,

2222 const DT_ alpha = DT_(1)) const

2225
2226 static constexpr IT_ bhd = IT_(BHD_);
2227 static constexpr IT_ bwd = IT_(BlockHeight);
2228
2229 static constexpr IT_ bh = IT_(BlockHeight);
2230 static constexpr IT_ bw = IT_(BlockWidth);
2231
2232 const auto& b(*this);

2236 XASSERT(d.columns() == a.size());
2237 XASSERT(a.size() == b.rows());
2238 XASSERT(b.columns() == this->columns());
2239
2240 // Fetch matrix arrays
2241 // Use POD here to avoid hassle with 1x1 matrices and stuff
2242 DT_* data_v = v.template elements<LAFEM::Perspective::pod>();
2243 const DT_* data_d = d.template val<LAFEM::Perspective::pod>();
2244 const DT_* data_a = a.template elements<LAFEM::Perspective::pod>();
2245 const DT_* data_b = b.template val<LAFEM::Perspective::pod>();

2247 const IT_* col_idx_d = d.col_ind();
2248 const IT_* row_ptr_b = b.row_ptr();
2249 const IT_* col_idx_b = b.col_ind();
2250
2251 // Loop over all rows i of v
2252 for(IT_ row(0); row < IT_(v.size()); ++row)
2253 {
2254 // For all non-zeros D_ik in row i of D
2255 for(IT_ pos_d(row_ptr_d[row]); pos_d < row_ptr_d[row+1]; ++pos_d)
2256 {
2257 // Get column index k
2258 const IT_ col_d(col_idx_d[pos_d]);
2259
2260 // Pre-compute block factor (alpha * D_ik * A_kk)
2261 typename MatrixD::ValueType omega(0);
2262 for(IT_ i(0); i < bhd; ++i)
2263 {
2264 for(IT_ j(0); j < bwd; ++j)

2268 }

2275 if(col_idx_b[pos_b] == row)

2279 {
2280 for(IT_ j(0); j < bh; ++j)
2281 {
2282 data_v[bhd*row + i] += omega[int(i)][int(j)]*data_b[bh*bw*pos_b + bw*j + i];
2283 }
2284 }

2286 }
2287 ++pos_b;
2288 }
