184 template<typename LocalMatrix_, typename RowMapping_, typename ColMapping_>
185 void operator()(const LocalMatrix_& loc_mat, const RowMapping_& row_map,
186 const ColMapping_& col_map, DT_ alpha = DT_(1))
187 {
188 // loop over all local row entries
189 for(int i(0); i < row_map.get_num_local_dofs(); ++i)
190 {
191 // fetch row index
192 const Index ix = row_map.get_index(i);
193
194 // build column pointer for this row entry contribution
195 for(IT_ k(_row_ptr[ix]); k < _row_ptr[ix + 1]; ++k)
196 {
197 _col_ptr[_col_idx[k]] = k;
198 }
199
200 // loop over all local column entries
201 for(int j(0); j < col_map.get_num_local_dofs(); ++j)
202 {
203 // fetch column index
204 const Index jx = col_map.get_index(j);
205
206 // ensure that the column pointer is valid for this index
207 ASSERTM(_col_ptr[jx] != _deadcode, "invalid column index");
208
209 // incorporate data into global matrix
210 _data[_col_ptr[jx]] += alpha * loc_mat[i][j];
211
212 // continue with next column entry
213 }
214
215#ifdef DEBUG
216 // reformat column-pointer array
217 for(IT_ k(_row_ptr[ix]); k < _row_ptr[ix + 1]; ++k)
218 {
219 _col_ptr[_col_idx[k]] = _deadcode;
220 }
221#endif
222 // continue with next row entry
223 }
224 }

341 explicit SparseMatrixBCSR(const Adjacency::Graph & graph) :
342 Container<DT_, IT_>(0)
343 {
344 Index num_rows = graph.get_num_nodes_domain();
345 Index num_cols = graph.get_num_nodes_image();
346 Index num_nnze = graph.get_num_indices();
347
348 if (num_nnze == 0)
349 {
350 this->assign(SparseMatrixBCSR(num_rows, num_cols));
351 return;
352 }
353
354 // Create temporary vectors. Row and column pointer are block wise
355 LAFEM::DenseVector<IT_, IT_> vrow_ptr(num_rows+1);
356 LAFEM::DenseVector<IT_, IT_> vcol_idx(num_nnze);
357 // The data array has to account for the block size
358 LAFEM::DenseVector<DT_, IT_> vdata(num_nnze*Index(BlockHeight_*BlockWidth_), DT_(0));
359
360 const Index * dom_ptr(graph.get_domain_ptr());
361 const Index * img_idx(graph.get_image_idx());
362 IT_ * prow_ptr(vrow_ptr.elements());
363 IT_ * pcol_idx(vcol_idx.elements());
364
365 // build row-end
366 prow_ptr[0] = IT_(dom_ptr[0]);
367 for(Index i(0); i < num_rows; ++i)
368 prow_ptr[i+1] = IT_(dom_ptr[i+1]);
369
370 // build col-idx
371 for(Index i(0); i < num_nnze; ++i)
372 pcol_idx[i] = IT_(img_idx[i]);
373
374 // build the matrix
375 this->assign(SparseMatrixBCSR<DT_, IT_, BlockHeight_, BlockWidth_>(num_rows, num_cols, vcol_idx, vdata, vrow_ptr));
376 }

1200 {
1201 using XType = SparseMatrixBCSR<DT_, IT_, BlockWidth_, BlockHeight_>;
1202 if (x.used_elements() == 0)
1203 {
1204 SparseMatrixBCSR r(x.rows(), x.columns());
1205 this->assign(r);
1206 return;
1207 }
1208
1209 const Index txrows(x.rows());
1210 const Index txcolumns(x.columns());
1211 const Index txused_elements(x.used_elements());
1212
1213 const IT_ * ptxcol_ind(x.col_ind());
1214 const IT_ * ptxrow_ptr(x.row_ptr());
1215 const typename XType::ValueType * ptxval(x.val());

1218 DenseVector<DT_, IT_> tval(txused_elements * BlockHeight_ * BlockWidth_);
1219 DenseVector<IT_, IT_> trow_ptr(txcolumns + 1, IT_(0));

1221 IT_ * ptcol_ind(tcol_ind.elements());
1222 ValueType * ptval((ValueType*)tval.elements());
1223 IT_ * ptrow_ptr(trow_ptr.elements());
1224
1225 ptrow_ptr[0] = 0;
1226
1227 for (Index i(0); i < txused_elements; ++i)
1228 {
1229 ++ptrow_ptr[ptxcol_ind[i] + 1];
1230 }
1231
1232 for (Index i(1); i < txcolumns - 1; ++i)
1233 {
1234 ptrow_ptr[i + 1] += ptrow_ptr[i];
1235 }
1236
1237 for (Index i(0); i < txrows; ++i)
1238 {
1239 for (IT_ k(ptxrow_ptr[i]); k < ptxrow_ptr[i+1]; ++k)
1240 {

1242 const IT_ j(ptrow_ptr[l]);

1246 }
1247 }

1250 {

1252 }

1629 void add_double_mat_product(
1630 const LAFEM::SparseMatrixBCSR<DT_, IT_, BlockHeight_, BlockWidth_>& d,
1631 const LAFEM::SparseMatrixBCSR<DT_, IT_, BlockHeight_, BlockWidth_>& a,
1632 const LAFEM::SparseMatrixBCSR<DT_, IT_, BlockHeight_, BlockWidth_>& b,
1633 const DT_ alpha = DT_(1),
1634 const bool allow_incomplete = false)
1635 {
1636 // validate matrix dimensions
1637 XASSERT(BlockHeight_ == BlockWidth_);
1638 XASSERT(this->rows() == d.rows());
1639 XASSERT(d.columns() == a.rows());
1640 XASSERT(a.columns() == b.rows());
1641 XASSERT(b.columns() == this->columns());
1642
1643 // fetch matrix arrays:
1644 ValueType* data_x = this->val();
1645 const ValueType* data_d = d.val();
1646 const ValueType* data_a = a.val();
1647 const ValueType* data_b = b.val();
1648 const IT_* row_ptr_x = this->row_ptr();
1649 const IT_* col_idx_x = this->col_ind();
1650 const IT_* row_ptr_d = d.row_ptr();
1651 const IT_* col_idx_d = d.col_ind();
1652 const IT_* row_ptr_a = a.row_ptr();
1653 const IT_* col_idx_a = a.col_ind();
1654 const IT_* row_ptr_b = b.row_ptr();
1655 const IT_* col_idx_b = b.col_ind();
1656
1657 // loop over all rows of D and X, resp.
1658 for(IT_ i(0); i < IT_(this->rows()); ++i)
1659 {
1660 // loop over all non-zeros D_ik in row i of D
1661 for(IT_ ik(row_ptr_d[i]); ik < row_ptr_d[i+1]; ++ik)
1662 {
1663 // get column index k
1664 const IT_ k = col_idx_d[ik];
1665
1666 // loop over all non-zeros A_kl in row k of A
1667 for(IT_ kl(row_ptr_a[k]); kl < row_ptr_a[k+1]; ++kl)
1668 {
1669 // get column index l
1670 const IT_ l = col_idx_a[kl];
1671
1672 // pre-compute factor (alpha * D_ik * A_kl)
1673 ValueType omega;
1674 omega.set_mat_mat_mult(data_d[ik], data_a[kl]);
1675 omega *= alpha;
1676
1677 // loop over all non-zeros B_lj in row j of B and
1678 // loop over all non-zeros X_ij in row i of X and
1679 // perform a "sparse axpy" of B_l onto X_i, i.e.:
1680 // X_i. += (alpha * D_ik * A_kl) * B_l.
1681 IT_ ij = row_ptr_x[i];
1682 IT_ lj = row_ptr_b[l];
1683 while(lj < row_ptr_b[l+1])
1684 {
1685 if(ij >= row_ptr_x[i+1])
1686 {
1687 // we have reached the end of row X_i, but there is at least
1688 // one entry in row B_l left, so the pattern of X is incomplete
1689 // We let the caller decide whether this is a valid case or not:
1690 if(allow_incomplete)
1691 break; // continue with next row
1692 else
1693 XABORTM("Incomplete output matrix structure");
1694 }
1695 else if(col_idx_x[ij] == col_idx_b[lj])
1696 {
1697 // okay: B_lj contributes to X_ij
1698 ValueType temp;
1699 temp.set_mat_mat_mult(omega, data_b[lj]);
1700 data_x[ij] += temp;
1701 ++ij;
1702 ++lj;
1703 }
1704 else if(col_idx_x[ij] < col_idx_b[lj])
1705 {
1706 // entry X_ij exists, but B_lj is missing:
1707 // this is a perfectly valid case, so continue with the next non-zero of X_i
1708 ++ij;
1709 }
1710 else //if(col_idx_x[ij] > col_idx_b[lj])
1711 {
1712 // If we come out here, then the sparsity pattern of X is incomplete:
1713 // B_lj is meant to be added onto X_ij, but the entry X_ij is missing
1714 // We let the caller decide whether this is a valid case or not:
1715 if(allow_incomplete)
1716 ++lj;
1717 else
1718 XABORTM("Incomplete output matrix structure");
1719 }
1720 }
1721 }
1722 }
1723 }

1773 XASSERT(a.columns() == b.rows());

1775
1776 // fetch matrix arrays:
1777 ValueType* data_x = this->val();
1778 const DT_* data_d = d.val();
1779 const ValueType* data_a = a.val();
1780 const DT_* data_b = b.val();
1781 const IT_* row_ptr_x = this->row_ptr();
1782 const IT_* col_idx_x = this->col_ind();
1783 const IT_* row_ptr_d = d.row_ptr();

1785 const IT_* row_ptr_a = a.row_ptr();

1787 const IT_* row_ptr_b = b.row_ptr();
1788 const IT_* col_idx_b = b.col_ind();
1789
1790 // loop over all rows of D and X, resp.
1791 for(IT_ i(0); i < IT_(this->rows()); ++i)
1792 {
1793 // loop over all non-zeros D_ik in row i of D
1794 for(IT_ ik(row_ptr_d[i]); ik < row_ptr_d[i+1]; ++ik)
1795 {
1796 // get column index k
1797 const IT_ k = col_idx_d[ik];
1798
1799 // loop over all non-zeros A_kl in row k of A
1800 for(IT_ kl(row_ptr_a[k]); kl < row_ptr_a[k+1]; ++kl)
1801 {
1802 // get column index l
1803 const IT_ l = col_idx_a[kl];
1804
1805 // pre-compute factor (alpha * D_ik * A_kl)
1806 ValueType omega;
1807 omega.set_mat_mat_mult(ValueType(data_d[ik]), data_a[kl]);
1808 omega *= alpha;
1809
1810 // loop over all non-zeros B_lj in row j of B and
1811 // loop over all non-zeros X_ij in row i of X and
1812 // perform a "sparse axpy" of B_l onto X_i, i.e.:
1813 // X_i. += (alpha * D_ik * A_kl) * B_l.
1814 IT_ ij = row_ptr_x[i];
1815 IT_ lj = row_ptr_b[l];
1816 while(lj < row_ptr_b[l+1])
1817 {
1818 if(ij >= row_ptr_x[i+1])
1819 {
1820 // we have reached the end of row X_i, but there is at least
1821 // one entry in row B_l left, so the pattern of X is incomplete
1822 // We let the caller decide whether this is a valid case or not:
1823 if(allow_incomplete)
1824 break; // continue with next row
1825 else
1826 XABORTM("Incomplete output matrix structure");
1827 }
1828 else if(col_idx_x[ij] == col_idx_b[lj])
1829 {
1830 // okay: B_lj contributes to X_ij
1831 ValueType temp;
1832 temp.set_mat_mat_mult(omega, ValueType(data_b[lj]));
1833 data_x[ij] += temp;
1834 ++ij;
1835 ++lj;
1836 }
1837 else if(col_idx_x[ij] < col_idx_b[lj])
1838 {
1839 // entry X_ij exists, but B_lj is missing:
1840 // this is a perfectly valid case, so continue with the next non-zero of X_i
1841 ++ij;
1842 }
1843 else //if(col_idx_x[ij] > col_idx_b[lj])
1844 {
1845 // If we come out here, then the sparsity pattern of X is incomplete:
1846 // B_lj is meant to be added onto X_ij, but the entry X_ij is missing
1847 // We let the caller decide whether this is a valid case or not:
1848 if(allow_incomplete)
1849 ++lj;
1850 else
1851 XABORTM("Incomplete output matrix structure");
1852 }
1853 }
1854 }
1855 }

1970 void permute(Adjacency::Permutation & perm_row, Adjacency::Permutation & perm_col)
1971 {
1972 if (perm_row.size() == 0 && perm_col.size() == 0)
1973 return;
1974
1975 XASSERTM(perm_row.size() == this->rows(), "Container rows does not match permutation size");
1976 XASSERTM(perm_col.size() == this->columns(), "Container columns does not match permutation size");
1977
1978 // http://de.mathworks.com/help/matlab/math/sparse-matrix-operations.html#f6-13070
1979 IT_ * temp_row_ptr = new IT_[rows() + 1];
1980 IT_ * temp_col_ind = new IT_[used_elements()];
1981 ValueType * temp_val = new ValueType[used_elements()];
1982
1983 Index * perm_pos;
1984 perm_pos = perm_row.get_perm_pos();
1985
1986 //permute rows from source to temp_*
1987 Index new_start(0);
1988 temp_row_ptr[0] = 0;
1989 for (Index row(0) ; row < this->rows() ; ++row)
1990 {
1991 Index row_size(this->row_ptr()[perm_pos[row] + 1] - this->row_ptr()[perm_pos[row]]);
1992
1993 //iterate over all elements in single one new and old row
1994 for (Index i(new_start), j(this->row_ptr()[perm_pos[row]]) ; i < new_start + row_size ; ++i, ++j)
1995 {
1996 temp_col_ind[i] = this->col_ind()[j];
1997 temp_val[i] = this->val()[j];
1998 }
1999
2000 new_start += row_size;
2001 temp_row_ptr[row+1] = (IT_)new_start;
2002 }
2003
2004 //use inverse col permutation as lookup table: i -> new location of i
2005 Adjacency::Permutation perm_col_inv = perm_col.inverse();
2006 perm_pos = perm_col_inv.get_perm_pos();
2007
2008 //permute columns from temp_* to source
2009 ::memcpy(this->row_ptr(), temp_row_ptr, (rows() + 1) * sizeof(IT_));
2010 ::memcpy(this->val(), temp_val, used_elements() * sizeof(ValueType));
2011 for (Index i(0) ; i < used_elements() ; ++i)
2012 {
2013 this->col_ind()[i] = (IT_)perm_pos[temp_col_ind[i]];
2014 }
2015
2016 delete[] temp_row_ptr;
2017 delete[] temp_col_ind;
2018 delete[] temp_val;
2019
2020 //sort columns in every row by column index
2021 IT_ swap_key;
2022 ValueType swap_val;
2023 for (Index row(0) ; row < rows() ; ++row)
2024 {
2025 Index offset(this->row_ptr()[row]);
2026 Index row_size(this->row_ptr()[row+1] - this->row_ptr()[row]);
2027 for (Index i(1), j ; i < row_size ; ++i)
2028 {
2029 swap_key = this->col_ind()[i + offset];
2030 swap_val = this->val()[i + offset];
2031 j = i;
2032 while (j > 0 && this->col_ind()[j - 1 + offset] > swap_key)
2033 {
2034 this->col_ind()[j + offset] = this->col_ind()[j - 1 + offset];
2035 this->val()[j + offset] = this->val()[j - 1 + offset];
2036 --j;
2037 }
2038 this->col_ind()[j + offset] = swap_key;
2039 this->val()[j + offset] = swap_val;
2040 }
2041 }
