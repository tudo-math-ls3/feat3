192 protected:
193 void _apply_intern(const LAFEM::SparseMatrixCSR<DataType, IndexType>& matrix, VectorType& vec_cor, const VectorType& vec_def)
194 {
195 // create pointers
196 DataType * pout(vec_cor.elements());
197 const DataType * pin(vec_def.elements());
198 const DataType * pval(matrix.val());
199 const IndexType * pcol_ind(matrix.col_ind());
200 const IndexType * prow_ptr(matrix.row_ptr());
201 const IndexType n((IndexType(matrix.rows())));
202
203 // __forward-insertion__
204 // iteration over all rows
205 for (Index i(0); i < n; ++i)
206 {
207 IndexType col;
208 DataType d(0);
209 // iteration over all elements on the left side of the main-diagonal
210 for (col = prow_ptr[i]; pcol_ind[col] < i; ++col)
211 {
212 d += pval[col] * pout[pcol_ind[col]];
213 }
214 pout[i] = (pin[i] - _omega * d) / pval[col];
215 }
216
217 // __backward-insertion__
218 // iteration over all rows
219 for (Index i(n); i > 0;)
220 {
221 --i;
222 IndexType col;
223 DataType d(0);
224 // iteration over all elements on the right side of the main-diagonal
225 for (col = prow_ptr[i+1] - IndexType(1); pcol_ind[col] > i; --col)
226 {
227 d += pval[col] * pout[pcol_ind[col]];
228 }
229 pout[i] -= _omega * d / pval[col];
230 }
231 }

251 void _apply_intern(const MatrixType & matrix, VectorType& vec_cor, const VectorType& vec_def)
252 {
253 // create pointers
254 auto* pout(vec_cor.elements());
255 const auto* pin(vec_def.elements());
256 const auto* pval(matrix.val());
257 const IndexType * pcol_ind(matrix.col_ind());
258 const IndexType * prow_ptr(matrix.row_ptr());
259 const IndexType n((IndexType(matrix.rows())));
260 typename MatrixType::ValueType inverse;
261
262 // __forward-insertion__
263 // iteration over all rows
264 for (Index i(0); i < n; ++i)
265 {
266 IndexType col;
267 typename VectorType::ValueType d(0);
268 // iteration over all elements on the left side of the main-diagonal
269 for (col = prow_ptr[i]; pcol_ind[col] < i; ++col)
270 {
271 d += pval[col] * pout[pcol_ind[col]];
272 }
273 //pout[i] = (pin[i] - _omega * d) / pval[col];
274 inverse.set_inverse(pval[col]);
275 pout[i] = inverse * (pin[i] - _omega * d);
276 }
277
278 // __backward-insertion__
279 // iteration over all rows
280 for (Index i(n); i > 0;)
281 {
282 --i;
283 IndexType col;
284 typename VectorType::ValueType d(0);
285 // iteration over all elements on the right side of the main-diagonal
286 for (col = prow_ptr[i+1] - IndexType(1); pcol_ind[col] > i; --col)
287 {
288 d += pval[col] * pout[pcol_ind[col]];
289 }
290 //pout[i] -= _omega * d / pval[col];
291 inverse.set_inverse(pval[col]);
292 pout[i] = pout[i] - (_omega * inverse * d);
293 }
294 }
