247 void _set_shadow_space()
248 {
249 //select set of random vectors
250
251 Random::SeedType seed;
252 if (_random)
253 {
254 seed = Random::get_seed();
255 }
256 else
257 {
258 auto comm = Dist::Comm::world();
259 seed = Random::SeedType(comm.rank());
260 }
261 Random rng(seed);
262
263 for (Index l(0); l<_krylov_dim; ++l)
264 _vec_P.at(l).format(rng, DataType(0), DataType(1));
265
266 //orthonormalize vector set P (modified Gram-Schmidt)
267 for (Index j(0); j<_krylov_dim; ++j)
268 {
269 for (Index i(0); i<j; ++i)
270 {
271 _vec_P.at(j).axpy(_vec_P.at(i), _vec_P.at(j), -_vec_P.at(j).dot(_vec_P.at(i)));
272 }
273 _vec_P.at(j).scale(_vec_P.at(j), DataType(1)/_vec_P.at(j).norm2());
274 }
275 _shadow_space_setup = true;
276 }

281 virtual Status _apply_intern(VectorType& vec_sol, const VectorType& DOXY(vec_rhs))
282 {
283 IterationStats pre_iter(*this);
284 Statistics::add_solver_expression(std::make_shared<ExpressionStartSolve>(this->name()));
285 const MatrixType& matrix(this->_system_matrix);
286 const FilterType& filter(this->_system_filter);
287
288 // compute initial defect
289 Status status = this->_set_initial_defect(this->_vec_res, vec_sol);
290 _vec_t.copy(_vec_res);
291 if(!this->_apply_precond(this->_vec_r, this->_vec_t, filter))
292 {
293 pre_iter.destroy();
294 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
295 return Status::aborted;
296 }
297 //select random vector set (shadow space)
298 if (!_shadow_space_setup)
299 _set_shadow_space();
300
301 pre_iter.destroy();
302
303 //produce start vectors
304 IterationStats first_iter(*this);
305 DataType om(0);
306 for (Index k(0); k < _krylov_dim; ++k)
307 {
308 matrix.apply(_vec_v, _vec_r);
309 filter.filter_def(_vec_v);
310 //save unpreconditioned residual update
311 _vec_t.copy(_vec_v);
312 if(!this->_apply_precond(this->_vec_v, this->_vec_t, filter))
313 {
314 first_iter.destroy();
315 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
316 return Status::aborted;
317 }
318 om = _vec_v.dot(_vec_r) / _vec_v.dot(_vec_v);
319 _vec_dX.at(k).scale(_vec_r, om);
320 _vec_dR.at(k).scale(_vec_v, -om);
321 vec_sol.axpy(vec_sol, _vec_dX.at(k));
322 _vec_r.axpy(_vec_r, _vec_dR.at(k));
323 _vec_res.axpy(_vec_t, _vec_res, -om);
324
325 // push our new defect
326 status = this->_set_new_defect(_vec_res, vec_sol);
327
328 //check for early convergence
329 if (status != Status::progress)
330 {
331 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), status, this->get_num_iter()));
332 return status;
333 }
334 // update k-th column of M
335 for (Index l(0); l < _krylov_dim; ++l)
336 _dmat_M(l,k, _vec_P.at(l).dot(_vec_dR.at(k)) );
337
338 }
339
340 Index oldest(0);
341 //calculate m
342 for (Index l(0); l< _krylov_dim; ++l)
343 _dvec_m(l, _vec_P.at(l).dot(_vec_r) );
344
345 first_iter.destroy();
346 // outer loop
347 while(status == Status::progress)
348 {
349 IterationStats stat(*this);
350
351 //inner loop
352 for (Index k(0); k <= _krylov_dim; ++k)
353 {
354 //solve c=M\m
355 _dmat_Minv = _dmat_M.inverse();
356 _dmat_Minv.apply(_dvec_c, _dvec_m);
357
358 _vec_q.scale(_vec_dR.at(0), -_dvec_c(0) );
359 for (Index l(1); l < _krylov_dim; ++l)
360 _vec_q.axpy(_vec_dR.at(l), _vec_q, -_dvec_c(l));
361
362 _vec_v.axpy(_vec_r, _vec_q);
363
364 if (k == 0)
365 {
366 matrix.apply(_vec_dR.at(oldest), _vec_v);
367 filter.filter_def(_vec_dR.at(oldest));
368 if(!this->_apply_precond(this->_vec_t, this->_vec_dR.at(oldest), filter))
369 {
370 stat.destroy();
371 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
372 return Status::aborted;
373 }
374 om = _vec_v.dot(_vec_t) / _vec_t.dot(_vec_t);
375 _vec_dR.at(oldest).axpy(_vec_t, _vec_q, -om);
376
377 //dX(oldest)=-dX*c+om*v
378 _vec_dX.at(oldest).scale(_vec_dX.at(oldest), -_dvec_c(oldest));
379 for (Index l(0); l < oldest; ++l)
380 _vec_dX.at(oldest).axpy(_vec_dX.at(l), _vec_dX.at(oldest), -_dvec_c(l));
381 for (Index l(oldest+1); l < _krylov_dim; ++l)
382 _vec_dX.at(oldest).axpy(_vec_dX.at(l), _vec_dX.at(oldest), -_dvec_c(l));
383 _vec_dX.at(oldest).axpy(_vec_v, _vec_dX.at(oldest), om);
384
385 matrix.apply(_vec_t, _vec_dX.at(oldest));
386 filter.filter_def(_vec_t);
387 }
388 else
389 {
390 //dX(oldest)=-dX*c+om*v
391 _vec_dX.at(oldest).scale(_vec_dX.at(oldest), -_dvec_c(oldest));
392 for (Index l(0); l < oldest; ++l)
393 _vec_dX.at(oldest).axpy(_vec_dX.at(l), _vec_dX.at(oldest), -_dvec_c(l));
394 for (Index l(oldest+1); l < _krylov_dim; ++l)
395 _vec_dX.at(oldest).axpy(_vec_dX.at(l), _vec_dX.at(oldest), -_dvec_c(l));
396 _vec_dX.at(oldest).axpy(_vec_v, _vec_dX.at(oldest), om);
397
398 matrix.apply( _vec_dR.at(oldest), _vec_dX.at(oldest));
399 filter.filter_def(_vec_dR.at(oldest));
400 _vec_t.copy(_vec_dR.at(oldest));
401 if(!this->_apply_precond(this->_vec_dR.at(oldest), this->_vec_t, filter))
402 {
403 stat.destroy();
404 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
405 return Status::aborted;
406 }
407 _vec_dR.at(oldest).scale(_vec_dR.at(oldest), DataType(-1));
408 }
409
410
411 _vec_r.axpy( _vec_dR.at(oldest), _vec_r );
412 vec_sol.axpy(_vec_dX.at(oldest), vec_sol);
413 _vec_res.axpy(_vec_t, _vec_res, -1);
414
415 status = this->_set_new_defect(_vec_res, vec_sol);
416 if (status != Status::progress)
417 break;
418
419 for (Index l(0); l < _krylov_dim; ++l)
420 {
421 _dvec_dm(l, _vec_P.at(l).dot(_vec_dR.at(oldest)) );
422 _dmat_M( l, oldest, _dvec_dm(l));
423 }
424 _dvec_m.axpy(_dvec_dm, _dvec_m);
425 oldest = (oldest+1) % _krylov_dim;
426 } //end inner loop
427
428 } //end outer loop
429 // finished
430 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), status, this->get_num_iter()));
431 return status;
432 } //_apply_intern(...)
433 }; // class IDRS<...>
