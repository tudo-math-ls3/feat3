135PointType FEAT::Geometry::CGALWrapper::closest_point(const PointType& point, PointType& primitive_grad) const
136{
137 Point_ query{(double)point[0], (double)point[1], (double)point[2]};
138 CGALWrapperData * cd = (CGALWrapperData*)_cgal_data;
139 const Point_and_primitive_id_& cl_p_query = cd->_tree->closest_point_and_primitive(query);
140 Polyhedron_::Face_handle f = cl_p_query.second;
141 auto v = CGAL::Polygon_mesh_processing::compute_face_normal(f,*(cd->_polyhedron));
142 for(int i = 0; i < 3; ++i)
143 primitive_grad[i] = (double)v[i];
144 return PointType{{(double)cl_p_query.first[0], (double)cl_p_query.first[1], (double)cl_p_query.first[2]}};
145}

147void FEAT::Geometry::CGALWrapper::_parse_mesh(std::istream & file, CGALFileMode file_mode)
148{
149 delete (CGALWrapperData*)_cgal_data;
150
151 _cgal_data = new CGALWrapperData;
152 CGALWrapperData * cd = (CGALWrapperData*)_cgal_data;
153
154 cd->_polyhedron = new Polyhedron_;
155
156 bool status(false);
157 switch(file_mode)
158 {
159 case CGALFileMode::fm_off:
160 status = CGAL::IO::read_OFF(file, *(cd->_polyhedron));
161 XASSERTM(status == true, "CGAL::IO::read_OFF: read/parse error !");
162 break;
163 case CGALFileMode::fm_obj:
164 status = CGAL::IO::read_OBJ(file, *(cd->_polyhedron));
165 XASSERTM(status == true, "CGAL::IO::read_OBJ: read/parse error !");
166 break;
167 default:
168 XASSERTM(false, "CGAL FileMode not supported!");
169 }
170
171 _init_wrapper();
172
173}

175void FEAT::Geometry::CGALWrapper::transform(const TransformMatrix& scale_rot, const PointType& translation, double scale)
176{
177 CGALWrapperData * cd = (CGALWrapperData*)_cgal_data;
178 //delete tree and _inside_tester
179 _delete_tree();
180 //build affine transformation
181 Transformation_ trafo_mat(scale_rot[0][0], scale_rot[0][1], scale_rot[0][2], translation[0],
182 scale_rot[1][0], scale_rot[1][1], scale_rot[1][2], translation[1],
183 scale_rot[2][0], scale_rot[2][1], scale_rot[2][2], translation[2],
184 scale);
185 //apply affine transformation on polyhedron
186 CGAL::Polygon_mesh_processing::transform(trafo_mat, *(cd->_polyhedron));
187 //create new trees
188 _init_wrapper();
189
190
191}

199void FEAT::Geometry::CGALWrapper::_delete_tree()
200{
201 CGALWrapperData * cd = (CGALWrapperData*)_cgal_data;
202 XASSERTM(cd->_polyhedron != nullptr, "ERROR: Polyhedron is not initialized!");
203 delete cd->_inside_tester;
204 cd->_inside_tester = nullptr;
205 delete cd->_tree;
206 cd->_tree = nullptr;
207}

209void FEAT::Geometry::CGALWrapper::_init_wrapper()
210{
211 CGALWrapperData * cd = (CGALWrapperData*)_cgal_data;
212 XASSERTM(cd->_polyhedron != nullptr, "ERROR: Polyhedron is not initialized!");
213 XASSERTM(cd->_tree == nullptr && cd->_inside_tester == nullptr, "ERROR: Tree or Inside Tester are already initialized");
214
215 // Construct AABB tree with a KdTree
216 cd->_tree = new Tree_(faces(*(cd->_polyhedron)).first, faces(*(cd->_polyhedron)).second, *(cd->_polyhedron));
217 cd->_tree->accelerate_distance_queries();
218 // Initialize the point-in-polyhedron tester
219 cd->_inside_tester = new Point_inside_(*(cd->_tree));
220}
