97bool FEAT::Geometry::CGALWrapper::point_inside(double x, double y, double z) const

114void FEAT::Geometry::CGALWrapper::_parse_mesh(std::istream & file, CGALFileMode file_mode)
115{
116 delete (CGALWrapperData*)_cgal_data;
117
118 _cgal_data = new CGALWrapperData;
119 CGALWrapperData * cd = (CGALWrapperData*)_cgal_data;
120
121 cd->_polyhedron = new Polyhedron_;
122
123 bool status(false);
124 switch(file_mode)
125 {
126 case CGALFileMode::fm_off:
127 status = CGAL::IO::read_OFF(file, *(cd->_polyhedron));
128 XASSERTM(status == true, "CGAL::IO::read_OFF: read/parse error !");
129 break;
130 case CGALFileMode::fm_obj:
131 status = CGAL::IO::read_OBJ(file, *(cd->_polyhedron));
132 XASSERTM(status == true, "CGAL::IO::read_OBJ: read/parse error !");
133 break;
134 default:
135 XASSERTM(false, "CGAL FileMode not supported!");
136 }
137
138 // Construct AABB tree with a KdTree
139 cd->_tree = new Tree_(faces(*(cd->_polyhedron)).first, faces(*(cd->_polyhedron)).second, *(cd->_polyhedron));
140 cd->_tree->accelerate_distance_queries();
141 // Initialize the point-in-polyhedron tester
142 cd->_inside_tester = new Point_inside_(*(cd->_tree));
143}
