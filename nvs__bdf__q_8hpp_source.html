187 NvsBdfQ(PropertyMap* config_section, const DataType reynolds, const bool use_deformation,
188 const bool startup):
189 _num_steps(~Index(0)),
190 _num_steps_startup(1),
191 _p_extrapolation_steps(~Index(0)),
192 _p_extrapolation_steps_startup(~Index(0)),
193 _reynolds(reynolds),
194 _delta_t(0),
195 _use_deformation(use_deformation),
196 _use_rotational_form(false),
197 _startup(startup),
198 coeff_lhs_v(3),
199 coeff_rhs_v_1(3),
200 coeff_rhs_v_phi(0),
201 coeff_extrapolation_p(0),
202 coeff_rhs_f(0),
203 coeff_rhs_proj_D_v(0),
204 coeff_rhs_phi(0),
205 coeff_rhs_v_2(0),
206 coeff_rhs_v_p(0),
207 ale_handling(TermHandling::off),
208 conv_handling(TermHandling::off),
209 visc_handling(TermHandling::off)
210 {
211 XASSERT(config_section != nullptr);
212 // Get time discretization order
213 auto num_steps_p = config_section->query("num_steps");
214 XASSERTM(num_steps_p.second, "TimeDiscretization section is missing the mandatory num_steps entry!");
215 _num_steps = Index(std::stoul(num_steps_p.first));
216 XASSERTM(_num_steps == Index(1) || _num_steps == Index(2),"Only BDF1 and BDF2 are implemented!");
217
218 // Get the order for extrapolation of the pressure in time
219 auto p_extrapolation_p = config_section->query("p_extrapolation_steps");
220 XASSERTM(p_extrapolation_p.second,
221 "TimeDiscretization section is missing the mandatory p_extrapolation_steps entry!");
222 _p_extrapolation_steps = Index(std::stoul(p_extrapolation_p.first));
223 XASSERTM(_p_extrapolation_steps < Index(3),"p extrapolation is only implemented for order 0, 1, 2!");
224 _p_extrapolation_steps_startup = Math::min(_p_extrapolation_steps, Index(1));
225
226 // Use the rotational form?
227 auto use_rotational_form_p = config_section->query("use_rotational_form");
228 XASSERTM(use_rotational_form_p.second,
229 "TimeDiscretization section is missing the mandatory use_rotational_form entry!");
230 XASSERTM(std::stoi(use_rotational_form_p.first) == 0 || std::stoi(use_rotational_form_p.first) == 1,
231 "use_rotational has to be set to 0 or 1");
232 _use_rotational_form = std::stoi(use_rotational_form_p.first) == 1;
233
234 // Get timestep size
235 auto delta_t_p = config_section->query("delta_t");
236 XASSERTM(delta_t_p.second, "TimeDiscretization section is missing the mandatory delta_t entry!");
237 _delta_t = DataType(std::stod(delta_t_p.first));
238 XASSERT(_delta_t > DataType(0));
239
240 // Use ale (meaning solve Navier-Stokes instead of plain Stokes)?
241 auto ale_p = config_section->query("ALE");
242 XASSERTM(ale_p.second,
243 "TimeDiscretization section is missing the mandatory ALE entry!");
244 ale_p.first.parse(ale_handling);
245
246 // Use convection (meaning solve Navier-Stokes instead of plain Stokes)?
247 auto convection_p = config_section->query("convection");
248 XASSERTM(convection_p.second,
249 "TimeDiscretization section is missing the mandatory convection entry!");
250 convection_p.first.parse(conv_handling);
251
252 // Use viscous (meaning solve Navier-Stokes instead of plain Stokes)?
253 auto viscous_p = config_section->query("viscous");
254 XASSERTM(viscous_p.second,
255 "TimeDiscretization section is missing the mandatory viscous entry!");
256 viscous_p.first.parse(visc_handling);
257
258 coeff_rhs_v_phi.clear();
259 coeff_rhs_v_phi = std::vector<DataType>(_num_steps, DataType(0));
260
261 coeff_extrapolation_p.clear();
262 coeff_extrapolation_p = std::vector<DataType>(_p_extrapolation_steps, DataType(0));
263
264 if(startup)
265 {
266 set_coeffs(_num_steps_startup, _p_extrapolation_steps_startup);
267 }
268 else
269 {
270 set_coeffs(_num_steps, _p_extrapolation_steps);
271 }

370 void set_coeffs(const Index num_steps, const Index p_extrapolation_steps)
371 {
372
373 if(num_steps == Index(1))
374 {
375 // Mass matrix
376 coeff_lhs_v.at(0) = DataType(1);
377 // Convection matrix
378 coeff_lhs_v.at(1) = (conv_handling == TermHandling::impl || ale_handling == TermHandling::impl)
379 ? _delta_t : DataType(0);
380 // Stiffness matrix
381 coeff_lhs_v.at(2) = (visc_handling == TermHandling::impl) ? _delta_t/_reynolds : DataType(0);
382
383 // Mass matrix
384 coeff_rhs_v_1.at(0) = DataType(1);
385 // Convection matrix
386 coeff_rhs_v_1.at(1) = (conv_handling == TermHandling::expl || ale_handling == TermHandling::expl)
387 ? -_delta_t : DataType(0);
388 // Stiffness matrix
389 coeff_rhs_v_1.at(2) = (visc_handling == TermHandling::expl) ? -_delta_t/_reynolds : DataType(0);
390
391 // Mass matrix coefficient for previous time step vector
392 coeff_rhs_v_2 = DataType(0);
393
394 coeff_rhs_f = _delta_t;
395
396 // coefficient for -(p*[k], div phi) on the rhs for the velocity eq.
397 coeff_rhs_v_p = -_delta_t;
398 // coefficient for -(phi[k], div phi) on the rhs for the velocity eq.
399 coeff_rhs_v_phi.at(0) = -_delta_t;
400
401 coeff_rhs_phi = DataType(1)/_delta_t;
402
403 }
404 else if(num_steps == Index(2))
405 {
406 // Mass matrix
407 coeff_lhs_v.at(0) = DataType(1);
408 // Convection matrix
409 coeff_lhs_v.at(1) = (conv_handling == TermHandling::impl) ? DataType(2)/DataType(3)*_delta_t : DataType(0);
410 // Stiffness matrix
411 coeff_lhs_v.at(2) = (visc_handling == TermHandling::impl || ale_handling == TermHandling::impl)
412 ? DataType(2)/DataType(3)*_delta_t/_reynolds : DataType(0);
413
414 // Mass matrix
415 coeff_rhs_v_1.at(0) = DataType(4)/DataType(3);
416 // Convection matrix
417 coeff_rhs_v_1.at(1) = (conv_handling == TermHandling::expl || ale_handling == TermHandling::expl)
418 ? -DataType(2)/DataType(3)*_delta_t : DataType(0);
419 // Stiffness matrix
420 coeff_rhs_v_1.at(2) =
421 (visc_handling == TermHandling::expl) ? -DataType(2)/DataType(3)*_delta_t/_reynolds : DataType(0);
422
423 coeff_rhs_f = DataType(2)/DataType(3)*_delta_t;
424
425 // Mass matrix coefficient for previous time step vector
426 coeff_rhs_v_2 = -DataType(1)/DataType(3);
427
428 // coefficient for -(p*[k], div phi) on the rhs for the velocity eq.
429 coeff_rhs_v_p = -DataType(2)/DataType(3)*_delta_t;
430
431 // coefficient for -(phi[k], div phi) on the rhs for the velocity eq.
432 coeff_rhs_v_phi.at(0) = -DataType(8)/DataType(9)*_delta_t;
433 // coefficient for -(phi[k-1], div phi) on the rhs for the velocity eq.
434 coeff_rhs_v_phi.at(1) = DataType(2)/DataType(9)*_delta_t;
435
436 coeff_rhs_phi = DataType(3)/(DataType(2)*_delta_t);
437
438 }
439 else
440 {
441 XABORTM("Invalid number of BDF steps: "+stringify(num_steps));
442 }
443
444 // Coefficient for the div v term for the projection step
445 coeff_rhs_proj_D_v = _use_rotational_form ? DataType(1)/_reynolds : DataType(0);
446
447 if(_use_deformation)
448 {
449 coeff_rhs_proj_D_v *= DataType(2);
450 }
451
452 // Coefficients for the extrapolation of the pressure
453 if(p_extrapolation_steps == Index(1))
454 {
455 coeff_extrapolation_p.at(0) = DataType(1);
456 }
457 else if(p_extrapolation_steps == Index(2))
458 {
459 coeff_extrapolation_p.at(0) = DataType(2);
460 coeff_extrapolation_p.at(1) = -DataType(1);
461 }
462 else
463 {
464 XABORTM("Invalid number of pressure extrapolation steps: "+stringify(p_extrapolation_steps));
465 }
466
467 }
