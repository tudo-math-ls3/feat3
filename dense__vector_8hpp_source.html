68 public:
69 typedef LAFEM::DenseVector<DT_, IT_> VectorType;

74 Index _num_entries;

78 explicit ScatterAxpy(VectorType & vector) :
79 _num_entries(vector.size()),
80 _data(vector.elements())
81 {
82 }
83
84 template <typename LocalVector_, typename Mapping_>
85 void operator()(const LocalVector_ & loc_vec, const Mapping_ & mapping, DT_ alpha = DT_(1))
86 {
87 // loop over all local entries
88 for (int i(0); i < mapping.get_num_local_dofs(); ++i)
89 {
90 // get dof index
91 Index dof_idx = mapping.get_index(i);
92 ASSERT(dof_idx < _num_entries);
93
94 // update vector entry
95 _data[dof_idx] += alpha * loc_vec[i];
96 }
97 }
98 }; // class ScatterAxpy

532 {
533 std::ios_base::openmode bin = std::ifstream::in | std::ifstream::binary;
534 if (mode == FileMode::fm_mtx)
535 bin = std::ifstream::in;
536 std::ifstream file(filename.c_str(), bin);

549 void read_from(FileMode mode, std::istream & file)
550 {
551 switch (mode)
552 {
553 case FileMode::fm_mtx:
554 {
555 this->clear();
556 this->_scalar_index.push_back(0);
557
558 Index rows;
559 String line;
560 std::getline(file, line);
561 if (line.find("%%MatrixMarket matrix array real general") == String::npos)
562 {
563 XABORTM("Input-file is not a compatible mtx-vector-file");
564 }
565 while (! file.eof())
566 {
567 std::getline(file, line);
568 if (file.eof())
569 XABORTM("Input-file is empty");
570
571 String::size_type begin(line.find_first_not_of(" "));
572 if (line.at(begin) != '%')
573 break;
574 }
575 {
576 String::size_type begin(line.find_first_not_of(" "));
577 line.erase(0, begin);
578 String::size_type end(line.find_first_of(" "));
579 String srows(line, 0, end);
580 rows = (Index)atol(srows.c_str());
581 line.erase(0, end);
582
583 begin = line.find_first_not_of(" ");
584 line.erase(0, begin);
585 end = line.find_first_of(" ");
586 String scols(line, 0, end);
587 Index cols((Index)atol(scols.c_str()));
588 line.erase(0, end);
589 XASSERTM(cols == 1, "Input-file is no dense-vector-file");
590 }
591
592 DenseVector<DT_, IT_> tmp(rows);
593 DT_ * pval(tmp.elements());
594
595 while (! file.eof())
596 {
597 std::getline(file, line);
598 if (file.eof())
599 break;
600
601 String::size_type begin(line.find_first_not_of(" "));
602 line.erase(0, begin);
603 String::size_type end(line.find_first_of(" "));
604 String sval(line, 0, end);
605 DT_ tval((DT_)atof(sval.c_str()));
606
607 *pval = tval;
608 ++pval;
609 }
610 this->assign(tmp);
611 break;
612 }
613 case FileMode::fm_exp:
614 {
615 this->clear();
616 this->_scalar_index.push_back(0);
617
618 std::vector<DT_> data;
619
620 while (! file.eof())
621 {
622 std::string line;
623 std::getline(file, line);
624 if (line.find("#", 0) < line.npos)
625 continue;
626 if (file.eof())
627 break;
628
629 std::string n_z_s;
630
631 std::string::size_type first_digit(line.find_first_not_of(" "));
632 line.erase(0, first_digit);
633 std::string::size_type eol(line.length());
634 for (std::string::size_type i(0); i < eol; ++i)
635 {
636 n_z_s.append(1, line[i]);
637 }
638
639 DT_ n_z((DT_)atof(n_z_s.c_str()));
640
641 data.push_back(n_z);
642 }
643
644 this->_scalar_index.at(0) = Index(data.size());
645 this->_elements.push_back(MemoryPool::template allocate_memory<DT_>(Index(data.size())));
646 this->_elements_size.push_back(Index(data.size()));
647 MemoryPool::copy(this->_elements.at(0), &data[0], Index(data.size()));
648 break;
649 }
650 case FileMode::fm_dv:
651 case FileMode::fm_binary:
652 this->template _deserialize<double, std::uint64_t>(FileMode::fm_dv, file);
653 break;
654 default:
655 XABORTM("Filemode not supported!");
656 }
657 }

684 void write_out(FileMode mode, std::ostream & file) const
685 {
686 switch (mode)
687 {
688 case FileMode::fm_mtx:
689 {
690 DenseVector<DT_, IT_> temp;
691 temp.convert(*this);
692
693 const Index tsize(temp.size());
694 file << "%%MatrixMarket matrix array real general" << std::endl;
695 file << tsize << " " << 1 << std::endl;
696
697 const DT_ * pval(temp.elements());
698 for (Index i(0); i < tsize; ++i, ++pval)
699 {
700 file << std::scientific << *pval << std::endl;
701 }
702 break;
703 }
704 case FileMode::fm_exp:
705 {
706 DT_ * temp = MemoryPool::template allocate_memory<DT_>((this->size()));
707 MemoryPool::copy(temp, this->_elements.at(0), this->size());
708
709 for (Index i(0); i < this->size(); ++i)
710 {
711 file << std::scientific << temp[i] << std::endl;
712 }
713 MemoryPool::release_memory(temp);
714 break;
715 }
716 case FileMode::fm_dv:
717 case FileMode::fm_binary:
718 this->template _serialize<double, std::uint64_t>(FileMode::fm_dv, file);
719 break;
720 default:
721 XABORTM("Filemode not supported!");
722 }
723 }

1083 friend bool operator== (const DenseVector & a, const DenseVector<DT_, IT_> & b)
1084 {
1085 if (a.size() != b.size())
1086 return false;
1087 if (a.get_elements().size() != b.get_elements().size())
1088 return false;
1089 if (a.get_indices().size() != b.get_indices().size())
1090 return false;
1091
1092 if (a.size() == 0 && b.size() == 0 && a.get_elements().size() == 0 && b.get_elements().size() == 0)
1093 return true;
1094
1095 bool ret(true);
1096
1097 for (Index i(0); i < a.size(); ++i)
1098 {
1099 if (a(i) != b(i))
1100 {
1101 ret = false;
1102 break;
1103 }
1104 }
