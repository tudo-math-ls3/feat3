95 template<typename Matrix_, typename Filter_, typename SolverBase_ = Solver::SolverBase<typename Matrix_::VectorTypeL>>
96 class HypreSolverBase :
97 public ADPSolverBase<Matrix_, Filter_, SolverBase_>
98 {
99 public:
101 typedef ADPSolverBase<Matrix_, Filter_, SolverBase_> BaseClass;
102 // the vector type
103 typedef typename BaseClass::VectorType VectorType;
104
105 protected:
107 void* _core;
108
109 explicit HypreSolverBase(const Matrix_& matrix, const Filter_& filter) :
110 BaseClass(matrix, filter),
111 _core(nullptr)
112 {
113 }
114
124 void _upload_def(const VectorType& vec_def)
125 {
126 // upload defect to ADP vector
127 this->_upload_vec_def(vec_def);
128
129 // set HYPRE defect vector values
130 Hypre::set_vec_def_values(this->_core, this->_get_vec_def_vals(vec_def));
131 }
132
136 void _format_cor()
137 {
138 // format HYPRE correction vector
139 Hypre::set_vec_cor_values(this->_core, nullptr);
140 }
141
151 void _download_cor(VectorType& vec_cor)
152 {
153 // get HYPRE correction vector values
154 Hypre::get_vec_cor_values(this->_core, this->_get_vec_cor_vals(vec_cor));
155
156 // download correction from APD vector
157 this->_download_vec_cor(vec_cor);
158
159 // apply correction filter
160 this->_system_filter.filter_cor(vec_cor);
161 }
162
163 public:
173 virtual void init_symbolic() override
174 {
175 BaseClass::init_symbolic();
176
177 XASSERT(this->_core == nullptr);
178
179 // create our HYPRE core wrapper object
180 this->_core = Hypre::create_core(
181#ifdef FEAT_HAVE_MPI
182 &this->_get_comm()->mpi_comm(),
183#else
184 nullptr, // no communicator for non-MPI builds
185#endif
186 this->_get_global_dof_offset(), this->_get_num_owned_dofs(), this->_get_mat_row_ptr(), this->_get_mat_col_idx());
187 }
188
195 virtual void init_numeric() override
196 {
197 BaseClass::init_numeric();
198
199 XASSERT(this->_core != nullptr);
200
201 // update matrix values of HYPRE matrix
202 Hypre::set_matrix_values(this->_core, this->_get_mat_vals());
203 }
204
211 virtual void done_symbolic() override
212 {
213 XASSERT(this->_core != nullptr);
214
215 Hypre::destroy_core(this->_core);
216 this->_core = nullptr;
217
218 BaseClass::done_symbolic();
219 }

233 template<typename Matrix_, typename Filter_>
234 class ParaSailsPrecond :
235 public HypreSolverBase<Matrix_, Filter_>
236 {
237 public:
239 typedef HypreSolverBase<Matrix_, Filter_> BaseClass;
241 typedef typename BaseClass::VectorType VectorType;
242
243 protected:
245 void* _solver;
246
248 // 0: num levels, third argument for HYPRE_ParaSailsSetParams, default: 1
249 // 1: symmetry, second argument for HYPRE_ParaSailsSetSym, default: 2
250 int _iparam[2];
251
253 // 0: threshold, second argument for HYPRE_ParaSailsSetParams, default: 0.1
254 // 1: filter, second argument for HYPRE_ParaSailsSetFilter, default: 0.05
255 double _dparam[2];
256
257 public:
258 explicit ParaSailsPrecond(const Matrix_& matrix, const Filter_& filter) :
259 BaseClass(matrix, filter),
260 _solver(nullptr)
261 {
262 _iparam[0] = 1;
263 _iparam[1] = 2;
264
265 _dparam[0] = 0.1;
266 _dparam[1] = 0.05;
267 }
268
269 explicit ParaSailsPrecond(const String& section_name, const PropertyMap* section,
270 const Matrix_& matrix, const Filter_& filter)
271 :
272 ParaSailsPrecond(matrix, filter)
273 {
274 auto level_p = section->query("level");
275 if(level_p.second && !level_p.first.parse(_iparam[0]))
276 throw ParseError(section_name + ".level", level_p.first, "a non-negative integer");
277
278 auto sym_p = section->query("sym");
279 if(sym_p.second && !sym_p.first.parse(_iparam[1]))
280 throw ParseError(section_name + ".sym", sym_p.first, "a non-negative integer");
281
282 auto thresh_p = section->query("thresh");
283 if(thresh_p.second && !thresh_p.first.parse(_dparam[0]))
284 throw ParseError(section_name + ".thresh", thresh_p.first, "a positive float");
285
286 auto filter_p = section->query("filter");
287 if(filter_p.second && !filter_p.first.parse(_dparam[1]))
288 throw ParseError(section_name + ".filter", filter_p.first, "a positive float");
289 }
290
291 virtual String name() const override
292 {
293 return "ParaSailsPrecond";
294 }
295
296 virtual void init_numeric() override
297 {
298 BaseClass::init_numeric();
299
300 // create ParaSails preconditioner
301 this->_solver = Hypre::create_parasails(this->_core, this->_iparam, this->_dparam);
302 }
303
304 virtual void done_numeric() override
305 {
306 if(_solver != nullptr)
307 {
308 Hypre::destroy_parasails(_solver);
309 _solver = nullptr;
310 }
311
312 BaseClass::done_numeric();
313 }
314
315 virtual Status apply(VectorType& vec_cor, const VectorType& vec_def) override
316 {
317 // upload defect vector and format correction
318 this->_upload_def(vec_def);
319 this->_format_cor();
320
321 // apply ParaSails preconditioner
322 Hypre::solve_parasails(this->_core, this->_solver);
323
324 // download correction
325 this->_download_cor(vec_cor);
326
327 // okay
328 return Status::success;
329 }
330 }; // class ParaSailsPrecond
