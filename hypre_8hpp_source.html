97 template<typename Matrix_, typename Filter_, typename SolverBase_ = Solver::SolverBase<typename Matrix_::VectorTypeL>>
98 class HypreSolverBase :
99 public ADPSolverBase<Matrix_, Filter_, SolverBase_>
100 {
101 public:
103 typedef ADPSolverBase<Matrix_, Filter_, SolverBase_> BaseClass;
104 // the vector type
105 typedef typename BaseClass::VectorType VectorType;
106
107 protected:
109 void* _core;
110
111 explicit HypreSolverBase(const Matrix_& matrix, const Filter_& filter) :
112 BaseClass(matrix, filter),
113 _core(nullptr)
114 {
115 }
116
126 void _upload_def(const VectorType& vec_def)
127 {
128 // upload defect to ADP vector
129 this->_upload_vec_def(vec_def);
130
131 // set HYPRE defect vector values
132 Hypre::set_vec_def_values(this->_core, this->_get_vec_def_vals(vec_def));
133 }
134
138 void _format_cor()
139 {
140 // format HYPRE correction vector
141 Hypre::set_vec_cor_values(this->_core, nullptr);
142 }
143
153 void _download_cor(VectorType& vec_cor)
154 {
155 // get HYPRE correction vector values
156 Hypre::get_vec_cor_values(this->_core, this->_get_vec_cor_vals(vec_cor));
157
158 // download correction from APD vector
159 this->_download_vec_cor(vec_cor);
160
161 // apply correction filter
162 this->_system_filter.filter_cor(vec_cor);
163 }
164
165 public:
175 virtual void init_symbolic() override
176 {
177 BaseClass::init_symbolic();
178
179 XASSERT(this->_core == nullptr);
180
181 // create our HYPRE core wrapper object
182 this->_core = Hypre::create_core(
183#ifdef FEAT_HAVE_MPI
184 &this->_get_comm()->mpi_comm(),
185#else
186 nullptr, // no communicator for non-MPI builds
187#endif
188 this->_get_global_dof_offset(), this->_get_num_owned_dofs(), this->_get_mat_row_ptr(), this->_get_mat_col_idx());
189 }
190
197 virtual void init_numeric() override
198 {
199 BaseClass::init_numeric();
200
201 XASSERT(this->_core != nullptr);
202
203 // update matrix values of HYPRE matrix
204 Hypre::set_matrix_values(this->_core, this->_get_mat_vals());
205 }
206
213 virtual void done_symbolic() override
214 {
215 XASSERT(this->_core != nullptr);
216
217 Hypre::destroy_core(this->_core);
218 this->_core = nullptr;
219
220 BaseClass::done_symbolic();
221 }

235 template<typename Matrix_, typename Filter_>
236 class ParaSailsPrecond :
237 public HypreSolverBase<Matrix_, Filter_>
238 {
239 public:
241 typedef HypreSolverBase<Matrix_, Filter_> BaseClass;
243 typedef typename BaseClass::VectorType VectorType;
244
245 protected:
247 void* _solver;
248
250 // 0: num levels, third argument for HYPRE_ParaSailsSetParams, default: 1
251 // 1: symmetry, second argument for HYPRE_ParaSailsSetSym, default: 2
252 int _iparam[2];
253
255 // 0: threshold, second argument for HYPRE_ParaSailsSetParams, default: 0.1
256 // 1: filter, second argument for HYPRE_ParaSailsSetFilter, default: 0.05
257 double _dparam[2];
258
259 public:
260 explicit ParaSailsPrecond(const Matrix_& matrix, const Filter_& filter) :
261 BaseClass(matrix, filter),
262 _solver(nullptr)
263 {
264 _iparam[0] = 1;
265 _iparam[1] = 2;
266
267 _dparam[0] = 0.1;
268 _dparam[1] = 0.05;
269 }
270
271 explicit ParaSailsPrecond(const String& section_name, PropertyMap* section,
272 const Matrix_& matrix, const Filter_& filter)
273 :
274 ParaSailsPrecond(matrix, filter)
275 {
276 auto level_p = section->query("level");
277 if(level_p.second && !level_p.first.parse(_iparam[0]))
278 throw ParseError(section_name + ".level", level_p.first, "a non-negative integer");
279
280 auto sym_p = section->query("sym");
281 if(sym_p.second && !sym_p.first.parse(_iparam[1]))
282 throw ParseError(section_name + ".sym", sym_p.first, "a non-negative integer");
283
284 auto thresh_p = section->query("thresh");
285 if(thresh_p.second && !thresh_p.first.parse(_dparam[0]))
286 throw ParseError(section_name + ".thresh", thresh_p.first, "a positive float");
287
288 auto filter_p = section->query("filter");
289 if(filter_p.second && !filter_p.first.parse(_dparam[1]))
290 throw ParseError(section_name + ".filter", filter_p.first, "a positive float");
291 }
292
293 virtual String name() const override
294 {
295 return "ParaSailsPrecond";
296 }
297
298 virtual void init_numeric() override
299 {
300 BaseClass::init_numeric();
301
302 // create ParaSails preconditioner
303 this->_solver = Hypre::create_parasails(this->_core, this->_iparam, this->_dparam);
304 }
305
306 virtual void done_numeric() override
307 {
308 if(_solver != nullptr)
309 {
310 Hypre::destroy_parasails(_solver);
311 _solver = nullptr;
312 }
313
314 BaseClass::done_numeric();
315 }
316
317 virtual Status apply(VectorType& vec_cor, const VectorType& vec_def) override
318 {
319 // upload defect vector and format correction
320 this->_upload_def(vec_def);
321 this->_format_cor();
322
323 // apply ParaSails preconditioner
324 Hypre::solve_parasails(this->_core, this->_solver);
325
326 // download correction
327 this->_download_cor(vec_cor);
328
329 // okay
330 return Status::success;
331 }
332 }; // class ParaSailsPrecond
