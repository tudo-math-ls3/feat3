170 static IntersectionResultType ray_segment_intersection(const Ray<Vector2D>& r, const Vector2D& a, const Vector2D& b)
171 {
172 const Vector2D ab = b - a;
173
174 // https://stackoverflow.com/a/565282
175 //
176 // Solve r.origin + t * r.direction = a + u * ab
177 // With x as the 2D cross product:
178 // => (r.origin + t * r.direction) x ab = (a + u* ab) x ab
179 // => t(r.direction x ab) = (a - r.origin) x ab
180 // => t = (a - r.origin) x ab / (r.direction x ab)
181 // Analoguous
182 // u = (r.origin - a) x r.direction / (ab x r.direction)
183 // with v x w = -w x v
184 // u = (a - r.origin) x r.direction / (r.direction x ab)
185
186 DataType denom = cross(r.direction, ab);
187 DataType t_num = cross(a - r.origin, ab);
188 DataType u_num = cross(a - r.origin, r.direction);
189
190 // Are directions of ray and segment parallel
191 bool dir_parallel = Math::abs(denom) < eps;
192
193 // Are ray and segment colinear
194 bool aligned = Math::abs(u_num) < eps;
195
196 if(dir_parallel && aligned)
197 {
198 // Solve for overlap between ray and segment
199
200 // Solve r.origin + t * r.direction = a
201 // => t * r.direction = a - r.origin
202 // => t = (a - r.origin) * r.direction / (r.direction * r.direction)
203
204 // Solve r.origin + t * r.direction = b
205 // => t = (a - r.origin) * r.direction / (r.direction * r.direction)
206
207 const DataType rr = Tiny::dot(r.direction, r.direction);
208 DataType t1 = Tiny::dot(a - r.origin, r.direction) / rr;
209 DataType t2 = Tiny::dot(b - r.origin, r.direction) / rr;
210
211 if(t1 > t2)
212 {
213 std::swap(t1, t2);
214 }
215
216 if(t2 < DataType(0.0))
217 {
218 // Exit point is behind ray start => no overlap
219 return std::nullopt;
220 }
221
222 // Discard overlap behind ray origin
223 t1 = Math::max(t1, DataType(0.0));
224 t2 = Math::max(t2, DataType(0.0));
225
226 return std::make_optional<IntersectionDataType>(t1, t2);
227 }
228 else if(dir_parallel && !aligned)
229 {
230 // Lines are parallel and non-intersecting
231 return std::nullopt;
232 }
233 else
234 {
235 const DataType t = t_num / denom;
236 const DataType u = u_num / denom;
237
238 if(t >= DT_(0.0) && DT_(0.0) <= u && u <= DT_(1.0))
239 {
240 // Intersection in front of ray
241 return std::make_optional<IntersectionDataType>(t);
242 }
243 else
244 {
245 // Either no intersection or intersection behind ray
246 return std::nullopt;
247 }
248 }
249 }

266 static IntersectionResultType
267 ray_triangle_intersection(const Ray<Vector3D>& r, const Vector3D& a, const Vector3D& b, const Vector3D& c)
268 {
269 // Solve system
270 // r.origin + t * r.direction = (1 - u - v)a + ub + uc
271 // with u >= 0, v >= 0, u + v <= 1.0
272 // Rearranging yields
273 // [-r.direction, b - a, c - a] [t, u, v]' = ray.origin - a
274 // The below code solves the system using Cramer's rule
275
276 const Vector3D e1 = b - a;
277 const Vector3D e2 = c - a;
278
279 Vector3D normal;
280 Tiny::cross(normal, r.direction, e2);
281
282 DataType det = Tiny::dot(e1, normal);
283
284 if(det > -eps && det < eps)
285 {
286 // Ray is parallel to triangle
287
288 if(Math::abs(Tiny::dot(normal, r.origin - a)) > eps)
289 {
290 // Ray is not coplanar => no intersection
291 return std::nullopt;
292 }
293
294 // Ray is coplanar. There might be an interval of intersection
295 // Reduce to 2D polygon intersection test
296
297 // Using a as origin and e1, e2 as axes for the 2D coordinate system
298 // The triangle is then always (0, 0), (1, 0), (0, 1)
299
300 // Project ray into this coordinate system
301
302 // https://math.stackexchange.com/a/1307635
303 const auto world_to_plane = [&](const Vector3D& v)
304 {
305 const DataType alpha = Tiny::dot(normal, cross(e2, v)) / Tiny::dot(normal, cross(e2, e1));
306 const DataType beta = Tiny::dot(normal, cross(e1, v)) / Tiny::dot(normal, cross(e1, e2));
307 return Vector2D{alpha, beta};
308 };
309
310 const auto plane_to_world = [&](const Vector2D& v) { return a + v[0] * e1 + v[1] * e2; };
311
312 Vector2D origin = world_to_plane(r.origin - a);
313 Vector2D direction = world_to_plane(r.direction);
314
315 auto i1 = ray_segment_intersection(Ray<Vector2D>{origin, direction}, Vector2D{0.0, 0.0}, Vector2D{1.0, 0.0});
316 auto i2 = ray_segment_intersection(Ray<Vector2D>{origin, direction}, Vector2D{1.0, 0.0}, Vector2D{0.0, 1.0});
317 auto i3 = ray_segment_intersection(Ray<Vector2D>{origin, direction}, Vector2D{0.0, 1.0}, Vector2D{0.0, 0.0});
318
319 DataType t1(std::numeric_limits<DataType>::max());
320 DataType t2(0.0);
321
322 const DataType ray_length = r.direction.norm_euclid();
323
324 auto i = merge(merge(i1, i2), i3);
325
326 if(i)
327 {
328 const Vector2D p1 = origin + i.value().t_entry * direction;
329 const Vector2D p2 = origin + i.value().t_exit * direction;
330
331 t1 = Math::min(t1, (plane_to_world(p1) - r.origin).norm_euclid() / ray_length);
332 t2 = Math::max(t2, (plane_to_world(p2) - r.origin).norm_euclid() / ray_length);
333
334 return std::make_optional<IntersectionDataType>(t1, t2);
335 }
336 else
337 {
338 return std::nullopt;
339 }
340 }
341
342 DataType inv_det = DataType(1.0) / det;
343
344 Vector3D tvec = r.origin - a;
345
346 // Calculate u
347 const DataType u = Tiny::dot(tvec, normal) * inv_det;
348 if(u < DataType(0.0) || u > DataType(1.0))
349 {
350 return std::nullopt;
351 }
352
353 Vector3D qvec;
354 Tiny::cross(qvec, tvec, e1);
355
356 // Calculate v
357 const DataType v = Tiny::dot(r.direction, qvec) * inv_det;
358 if(v < DataType(0.0) || u + v > DataType(1.0))
359 {
360 return std::nullopt;
361 }
362
363 const DataType t = Tiny::dot(e2, qvec) * inv_det;
364
365 if(t > eps)
366 {
367 return std::make_optional<IntersectionDataType>(t);
368 }
369 else
370 {
371 return std::nullopt;
372 }
373 }
374
375 protected:
376 static DataType cross(const Vector2D& a, const Vector2D& b)
377 {
378 return (a[0] * b[1]) - (a[1] * b[0]);
379 }
380
381 static Vector3D cross(const Vector3D& a, const Vector3D& b)
382 {
383 Vector3D result;
384 Tiny::cross(result, a, b);
385 return result;
386 }
387 };

389 template<typename MeshType_>
390 std::optional<IntersectionData<typename MeshType_::CoordType>>
391 facet_raycast(const MeshType_& mesh, const Ray<typename MeshType_::VertexType>& r, Index facet)
392 {
393 using DataType = typename MeshType_::CoordType;
394 using IntersectionData = IntersectionData<DataType>;
395 using RIP = RayIntersectionPrimitives<DataType>;
396
397 using ShapeType = typename MeshType_::ShapeType;
398
399 using FacetShapeType = typename Shape::FaceTraits<ShapeType, ShapeType::dimension - 1>::ShapeType;
400 static constexpr int facet_dim = FacetShapeType::dimension;
401 static constexpr int world_dim = MeshType_::world_dim;
402
403 const auto& v_at_f = mesh.template get_index_set<facet_dim, 0>();
404 const auto& vtx = mesh.get_vertex_set();
405
406 if constexpr(std::is_same_v<FacetShapeType, Shape::Vertex> && world_dim == 1)
407 {
408 // Raycasting on 1d mesh
409 DataType origin = r.origin[0];
410 DataType direction = r.direction[0];
411
412 DataType facet_coord = mesh.get_vertex_set()[facet][0];
413
414 if(direction > 0 && facet >= origin)
415 {
416 return std::make_optional<IntersectionData>((facet_coord - origin) / direction);
417 }
418 else if(direction < 0 && facet <= origin)
419 {
420 return std::make_optional<IntersectionData>((origin - facet_coord) / direction);
421 }
422 else
423 {
424 return std::nullopt;
425 }
426 }
427
428 if constexpr(std::is_same_v<FacetShapeType, Shape::Hypercube<1>> && world_dim == 2)
429 {
430 return RIP::ray_segment_intersection(r, vtx[v_at_f(facet, 0)], vtx[v_at_f(facet, 1)]);
431 }
432
433 if constexpr(std::is_same_v<FacetShapeType, Shape::Simplex<1>> && world_dim == 2)
434 {
435 return RIP::ray_segment_intersection(r, vtx[v_at_f(facet, 0)], vtx[v_at_f(facet, 1)]);
436 }
437
438 if constexpr(std::is_same_v<FacetShapeType, Shape::Simplex<2>> && world_dim == 3)
439 {
440 return RIP::ray_triangle_intersection(r, vtx[v_at_f(facet, 0)], vtx[v_at_f(facet, 1)], vtx[v_at_f(facet, 2)]);
441 }
442
443 if constexpr(std::is_same_v<FacetShapeType, Shape::Hypercube<2>> && world_dim == 3)
444 {
445 // Split face into two triangles
446 auto i1 = RIP::ray_triangle_intersection(r, vtx[v_at_f(facet, 0)], vtx[v_at_f(facet, 1)], vtx[v_at_f(facet, 2)]);
447 auto i2 = RIP::ray_triangle_intersection(r, vtx[v_at_f(facet, 2)], vtx[v_at_f(facet, 1)], vtx[v_at_f(facet, 3)]);
448
449 return merge(i1, i2);
450 }
451
452 XABORTM("Unsupported raycast case");
453 }

458 template<typename MeshType_>
459 class VertexRaycaster
460 {
461 public:
463 using MeshType = MeshType_;
464
466 using DataType = typename MeshType::CoordType;
467
469 using IntersectionDataType = IntersectionData<DataType>;
470
472 using IntersectionResultType = std::optional<IntersectionDataType>;
473
475 using Vector = typename MeshType::VertexType;
476
478 static constexpr int shape_dim = MeshType::shape_dim;
479
481 static constexpr int facet_dim = shape_dim - 1;
482
483 protected:
485 const MeshType& _mesh;
486
488 Adjacency::Graph _c_at_v;
489
490 public:
492 explicit VertexRaycaster(const MeshType& mesh) :
493 _mesh(mesh),
494 _c_at_v(Adjacency::RenderType::transpose, _mesh.template get_index_set<shape_dim, 0>())
495 {
496 }
497
513 IntersectionResultType cast(const Index vertex, const Vector& direction) const
514 {
515 const DataType eps = DataType(1e-8);
516 const Ray<Vector> r{_mesh.get_vertex_set()[vertex], direction};
517
518 const auto& f_at_c = _mesh.template get_index_set<shape_dim, facet_dim>();
519 const Adjacency::CompositeAdjactor adj(_c_at_v, f_at_c);
520
521 IntersectionResultType result(std::nullopt);
522
523 for(auto face = adj.image_begin(vertex); face != adj.image_end(vertex); ++face)
524 {
525 const IntersectionResultType i = facet_raycast(_mesh, r, *face);
526
527 if(!i)
528 {
529 continue;
530 }
531
532 const IntersectionDataType& hit = i.value();
533
534 if(!result && hit.t_exit > eps)
535 {
536 result = i;
537 }
538 else if(result)
539 {
540 const IntersectionDataType& res = result.value();
541
542 // Update the result if we either found an earlier exit point,
543 // or if we found an earlier entry point with the same exit distance.
544 if(
545 hit.t_exit > eps &&
546 (hit.t_exit < res.t_exit || (Math::abs(hit.t_exit - res.t_exit) < eps && hit.t_entry < res.t_entry)))
547 {
548 result = hit;
549 }
550 }
551 }
552
553 return result;
554 }
