128 void prepare(const TrafoEvaluator& trafo_eval)
129 {
130 // domain point
131 DomainPointType dom_point;
132 ImagePointType img_point;
133
134 // create sub-index mapping
135 typedef Geometry::Intern::SubIndexMapping<Shape::Simplex<2>, 1, 0> SimType;
136
137 // compute edge orientations
138 SimType sim(
139 trafo_eval.get_trafo().get_mesh().template get_index_set<2,0>()[trafo_eval.get_cell_index()],
140 trafo_eval.get_trafo().get_mesh().template get_index_set<2,1>()[trafo_eval.get_cell_index()],
141 trafo_eval.get_trafo().get_mesh().template get_index_set<1,0>());
142 DataType edge_ori[3] =
143 {
144 DataType(1) - DataType(2*sim.map(0, 0)),
145 DataType(1) - DataType(2*sim.map(1, 0)),
146 DataType(1) - DataType(2*sim.map(2, 0))
147 };
148
149 // initialize edge vectors
150 ImagePointType edge_vec[3];
151 edge_vec[0].format();
152 edge_vec[1].format();
153 edge_vec[2].format();
154
155 // compute barycentre
156 dom_point[0] = dom_point[1] = DataType(1) / DataType(3);
157 trafo_eval.map_point(_barycentre, dom_point);
158
159 // nodal matrix
160 Tiny::Matrix<DataType, 21, 21> node_mat;
161 node_mat.format();
162
163 // loop over all vertices
164 for(int vi(0); vi < 3; ++vi)
165 {
166 // compute vertex coordinate
167 dom_point[0] = (vi == 1) ? DataType(1) : DataType(0);
168 dom_point[1] = (vi == 2) ? DataType(1) : DataType(0);
169 trafo_eval.map_point(img_point, dom_point);
170 img_point -= _barycentre;
171
172 // update edge vectors
173 edge_vec[(vi+1) % 3] += img_point;
174 edge_vec[(vi+2) % 3] -= img_point;
175
176 // initialize monomial powers
177 Tiny::Vector<DataType, 6> vx, vy;
178 vx(0) = vy(0) = DataType(1);
179 for(int l(0); l < 5; ++l)
180 {
181 vx(l+1) = vx(l) * img_point[0];
182 vy(l+1) = vy(l) * img_point[1];
183 }
184
185 // set monomial values
186 int k(0);
187 for(int i(0); i < 6; ++i)
188 {
189 for(int j(0); i+j < 6; ++j, ++k)
190 {
191 // monomial value
192 node_mat(k, 6*vi+0) = vx(i) * vy(j);
193 // dx-derivative
194 if(i > 0)
195 node_mat(k, 6*vi+1) = DataType(i) * vx(i-1) * vy(j);
196 // dy-derivative
197 if(j > 0)
198 node_mat(k, 6*vi+2) = DataType(j) * vy(j-1) * vx(i);
199 // dxx-derivative
200 if(i > 1)
201 node_mat(k, 6*vi+3) = DataType(i) * DataType(i-1) * vx(i-2) * vy(j);
202 // dyy-derivative
203 if(j > 1)
204 node_mat(k, 6*vi+4) = DataType(j) * DataType(j-1) * vy(j-2) * vx(i);
205 // dxy-derivative
206 if(i*j > 0)
207 node_mat(k, 6*vi+5) = DataType(i) * vx(i-1) * DataType(j) * vy(j-1);
208 }
209 }
210 }
211
212 for(int ei(0); ei < 3; ++ei)
213 {
214 // compute edge midpoint
215 dom_point[0] = (ei != int(1) ? DataType(0.5) : DataType(0));
216 dom_point[1] = (ei != int(2) ? DataType(0.5) : DataType(0));
217 trafo_eval.map_point(img_point, dom_point);
218 img_point -= _barycentre;
219
220 // compute edge normal and apply orientation
221 DataType dnorm = edge_vec[ei].norm_euclid();
222 DataType dnx = +(edge_vec[ei](1) * edge_ori[ei]) / dnorm;
223 DataType dny = -(edge_vec[ei](0) * edge_ori[ei]) / dnorm;
224
225 // initialize monomial powers
226 Tiny::Vector<DataType, 6> vx, vy;
227 vx(0) = vy(0) = DataType(1);
228 for(int l(0); l < 5; ++l)
229 {
230 vx(l+1) = vx(l) * img_point[0];
231 vy(l+1) = vy(l) * img_point[1];
232 }
233
234 // set monomial values
235 int k(0);
236 for(int i(0); i < 6; ++i)
237 {
238 for(int j(0); i+j < 6; ++j, ++k)
239 {
240 // edge normal derivative
241 if(i > 0)
242 node_mat(k, 18+ei) += DataType(i) * dnx * vx(i-1) * vy(j);
243 if(j > 0)
244 node_mat(k, 18+ei) += DataType(j) * dny * vy(j-1) * vx(i);
245 }
246 }
247 }
248
249 // invert nodal matrix to get the coefficient matrix
250 _coeff.set_inverse(node_mat);
251 }

253 template<SpaceTags space_cfg_, TrafoTags trafo_cfg_>
254 void eval_values(
255 EvalData<SpaceEvalTraits, space_cfg_>& phi,
256 const Trafo::EvalData<TrafoEvalTraits, trafo_cfg_>& tau) const
257 {
258 // initialize monomial powers
259 Tiny::Vector<DataType, 6> vx, vy;
260 vx(0) = vy(0) = DataType(1);
261 for(int l(0); l < 5; ++l)
262 {
263 vx(l+1) = vx(l) * (tau.img_point[0] - _barycentre[0]);
264 vy(l+1) = vy(l) * (tau.img_point[1] - _barycentre[1]);
265 }
266
267 // compute basis function values
268 for(int l(0); l < 21; ++l)
269 {
270 int k(0);
271 DataType v(DataType(0));
272 for(int i(0); i < 6; ++i)
273 for(int j(0); i+j < 6; ++j, ++k)
274 v += _coeff(l,k) * vx(i) * vy(j);
275 phi.phi[l].value = v;
276 }
277 }

279 template<SpaceTags space_cfg_, TrafoTags trafo_cfg_>
280 void eval_gradients(
281 EvalData<SpaceEvalTraits, space_cfg_>& phi,
282 const Trafo::EvalData<TrafoEvalTraits, trafo_cfg_>& tau) const
283 {
284 // initialize monomial powers
285 Tiny::Vector<DataType, 6> vx, vy;
286 vx(0) = vy(0) = DataType(1);
287 for(int l(0); l < 5; ++l)
288 {
289 vx(l+1) = vx(l) * (tau.img_point[0] - _barycentre[0]);
290 vy(l+1) = vy(l) * (tau.img_point[1] - _barycentre[1]);
291 }
292
293 // compute basis function gradients
294 for(int l(0); l < 21; ++l)
295 {
296 int k(0);
297 DataType dx(DataType(0)), dy(DataType(0));
298 for(int i(0); i < 6; ++i)
299 {
300 for(int j(0); i+j < 6; ++j, ++k)
301 {
302 if(i > 0)
303 dx += _coeff(l,k) * DataType(i) * vx(i-1) * vy(j);
304 if(j > 0)
305 dy += _coeff(l,k) * DataType(j) * vy(j-1) * vx(i);
306 }
307 }
308 phi.phi[l].grad[0] = dx;
309 phi.phi[l].grad[1] = dy;
310 }
311 }

313 template<SpaceTags space_cfg_, TrafoTags trafo_cfg_>
314 void eval_hessians(
315 EvalData<SpaceEvalTraits, space_cfg_>& phi,
316 const Trafo::EvalData<TrafoEvalTraits, trafo_cfg_>& tau) const
317 {
318 // initialize monomial powers
319 Tiny::Vector<DataType, 6> vx, vy;
320 vx(0) = vy(0) = DataType(1);
321 for(int l(0); l < 5; ++l)
322 {
323 vx(l+1) = vx(l) * (tau.img_point[0] - _barycentre[0]);
324 vy(l+1) = vy(l) * (tau.img_point[1] - _barycentre[1]);
325 }
326
327 // compute basis function hessians
328 for(int l(0); l < 21; ++l)
329 {
330 int k(0);
331 DataType dxx(DataType(0)), dyy(DataType(0)), dxy(DataType(0));
332 for(int i(0); i < 6; ++i)
333 {
334 for(int j(0); i+j < 6; ++j, ++k)
335 {
336 if(i > 1)
337 dxx += _coeff(l,k) * DataType(i) * DataType(i-1) * vx(i-2) * vy(j);
338 if(j > 1)
339 dyy += _coeff(l,k) * DataType(j) * DataType(j-1) * vy(j-2) * vx(i);
340 if(i*j > 0)
341 dxy += _coeff(l,k) * DataType(i) * vx(i-1) * DataType(j) * vy(j-1);
342 }
343 }
344 phi.phi[l].hess(0,0) = dxx;
345 phi.phi[l].hess(1,1) = dyy;
346 phi.phi[l].hess(0,1) = phi.phi[l].hess(1,0) = dxy;
347 }
348 }
