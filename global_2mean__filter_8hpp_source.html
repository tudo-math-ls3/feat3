218 void filter_rhs(VectorType& vector) const
219 {
220 if(empty())
221 return;
222
223 // compute dual integral
224 DataType integ = DataType(0);
225 if(_vec_freq.empty() || (_comm == nullptr))
226 integ = vector.dot(_vec_prim);
227 else
228 {
229 integ = _vec_freq.triple_dot(vector, _vec_prim);
230 _comm->allreduce(&integ, &integ, std::size_t(1), Dist::op_sum);
231 }
232 // subtract mean
233 vector.axpy(_vec_dual, vector, -integ / _volume);
234 }
235
242 void filter_sol(VectorType& vector) const
243 {
244 if(empty())
245 return;
246 // compute primal integral
247 DataType integ = DataType(0);
248 if(_vec_freq.empty() || (_comm == nullptr))
249 integ = vector.dot(_vec_dual);
250 else
251 {
252 integ = _vec_freq.triple_dot(vector, _vec_dual);
253 _comm->allreduce(&integ, &integ, std::size_t(1), Dist::op_sum);
254 }
255 // subtract mean
256 vector.axpy(_vec_prim, vector, -integ / _volume);
257 }
