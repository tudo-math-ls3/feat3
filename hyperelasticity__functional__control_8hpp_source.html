165 template<typename... Args_>
166 explicit HyperelasticityFunctionalControl(
167 DomainControl_& dom_ctrl,
168 const int meshopt_lvl_,
169 const std::deque<String>& dirichlet_list,
170 const std::deque<String>& slip_list,
171 const String& solver_name_,
172 PropertyMap& solver_config_,
173 Args_&&... args):
174 BaseClass(dom_ctrl, dirichlet_list, slip_list),
175 _system_levels(),
176 solver_config(solver_config_),
177 solver_name(solver_name_),
178 precond(nullptr),
179 meshopt_lvl(meshopt_lvl_),
180 meshopt_lvl_pos(~size_t(1))
181 {
182 XASSERT(meshopt_lvl >= -1);
183
184 // If the input level was set to -1, take the max level of the domain control
185 if(meshopt_lvl == -1)
186 {
187 meshopt_lvl = dom_ctrl.max_level_index();
188 }
189
190 XASSERT(meshopt_lvl <= dom_ctrl.max_level_index());
191
192 // Now find the position of the mesh optimization level in the domain levels
193 for(size_t i(0); i < dom_ctrl.size_physical(); ++i)
194 {
195 if(dom_ctrl.at(i)->get_level_index() == meshopt_lvl)
196 {
197 meshopt_lvl_pos = i;
198 }
199 }
200
201 XASSERT(meshopt_lvl_pos < dom_ctrl.size_physical());
202
203 for(size_t i(0); i < dom_ctrl.size_physical(); ++i)
204 {
205 _system_levels.push_back( new SystemLevelType(
206 dom_ctrl.at(i)->get_level_index(),
207 dom_ctrl.at(i)->get_mesh_node(),
208 dom_ctrl.at(i)->trafo,
209 dirichlet_list, slip_list,
210 std::forward<Args_>(args)...));
211 }
212
213 for(Index i(0); i < get_num_levels(); ++i)
214 {
215 _system_levels.at(i)->assemble_gates(dom_ctrl.at(i).layer());
216 // Call the operator's init() on the current level. This needs the gates, so it cannot be called
217 // earlier
218 _system_levels.at(i)->global_functional.init();
219 }
220
221 for(Index i(0); (i+1) < get_num_levels(); ++i)
222 {
223 _system_levels.at(i)->assemble_system_transfer(*_system_levels.at(i+1));
224 }
225
226 auto* solver_section = solver_config.query_section(solver_name);
227 if(solver_section == nullptr)
228 {
229 XABORTM("Could not find section for solver "+solver_name);
230 }
231
232 // create our preconditioner
233 precond = MeshoptPrecondFactory::create_nlopt_precond(*this, dom_ctrl, solver_section);
234
235 // create the actual nonlinear optimizer
236 _create_nonlinear_optimizer();
237 }

262 virtual CoordType compute_cell_size_defect(CoordType& lambda_min, CoordType& lambda_max,
263 CoordType& vol_min, CoordType& vol_max, CoordType& vol) const override
264 {
265
266 _system_levels.front()->global_functional.local().compute_cell_size_defect_pre_sync(vol_min, vol_max, vol);
267
268 vol_min = _system_levels.front()->gate_sys.min(vol_min);
269 vol_max = _system_levels.front()->gate_sys.max(vol_max);
270 vol = _system_levels.front()->gate_sys.sum(vol);
271
272 CoordType cell_size_defect = _system_levels.front()->global_functional.local().compute_cell_size_defect_post_sync(
273 lambda_min, lambda_max, vol_min, vol_max, vol);
274
275 lambda_min = _system_levels.front()->gate_sys.min(lambda_min);
276 lambda_max = _system_levels.front()->gate_sys.max(lambda_max);
277 cell_size_defect = _system_levels.front()->gate_sys.sum(cell_size_defect);
278
279 return cell_size_defect;
280 }

350 virtual void buffer_to_mesh() override
351 {
352 // Write finest level
353 _system_levels.front()->global_functional.local().buffer_to_mesh();
354
355 // Get the coords buffer on the finest level
356 const auto& coords_buffer_loc = _system_levels.front()->coords_buffer.local();
357
358 // Transfer fine coords buffer to coarser levels and perform buffer_to_mesh
359 for(size_t level(0); level < get_num_levels(); ++level )
360 {
361 Index ndofs(_system_levels.at(level)->local_functional.trafo_space.get_num_dofs());
362
363 // At this point, what we really need is a primal restriction operator that restricts the FE function
364 // representing the coordinate distribution to the coarser level. This is very simple for continuous
365 // Lagrange elements (just discard the additional information from the fine level), but not clear in
366 // the generic case. So we use an evil hack here:
367 // Because of the underlying two level ordering, we just need to copy the first ndofs entries from
368 // the fine level vector.
369 typename SystemLevelType::LocalCoordsBuffer
370 vec_level(coords_buffer_loc, ndofs, Index(0));
371
372 _system_levels.at(level)->global_functional.local().get_coords().copy(vec_level);
373 _system_levels.at(level)->global_functional.local().buffer_to_mesh();
374 }

404 virtual void add_to_vtk_exporter(
405 Geometry::ExportVTK<MeshType>& exporter, const int lvl_index) const override
406 {
407
408 for(size_t pos(0); pos < get_num_levels(); ++pos)
409 {
410 if(this->_system_levels.at(pos)->get_level_index() == lvl_index)
411 {
412
413 const auto& sys_lvl = this->_system_levels.at(pos);
414 sys_lvl->global_functional.local().add_to_vtk_exporter(exporter);
415
416 DataType fval(0);
417 auto grad = sys_lvl->global_functional.create_vector_r();
418
419 sys_lvl->global_functional.eval_fval_grad(fval, grad);
420 sys_lvl->filter_sys.filter_def(grad);
421 exporter.add_vertex_vector("grad", grad.local());
422
423 for(auto& it:sys_lvl->filter_sys.local().template at<0>())
424 {
425 const String field_name("nu_"+it.first);
426 // WARNING: This explicitly assumes that the filter vector belong to a P1/Q1 space and thus "lives"
427 // in the mesh's vertices
428 exporter.add_vertex_vector(field_name, it.second.get_filter_vector());
429 }

445 {
446 --level;

468 virtual void optimize() override
469 {
470 // fetch our finest levels
471 SystemLevelType& the_system_level = *_system_levels.at(meshopt_lvl_pos);
472
473 // create our RHS and SOL vectors
474 typename SystemLevelType::GlobalSystemVectorR vec_rhs(the_system_level.assemble_rhs_vector());
475 typename SystemLevelType::GlobalSystemVectorL vec_sol(the_system_level.assemble_sol_vector());
476
477 // solve
478 //Solver::solve(*solver, vec_sol, vec_rhs, the_system_level.global_functional, the_system_level.filter_sys);
479
480 // Let it be knownst to Statistics that it was Us who called the solver
481 FEAT::Statistics::expression_target = name();
482
483 the_system_level.global_functional.reset_num_evals();
484 solver->correct(vec_sol, vec_rhs);
485
486 //If the mesh was not optimized on the finest domain level, we now need to prolongate the solution by:
487 // - refine the coarse vertex set using the StandardRefinery
488 // - copy the results to the CoordsBuffer
489 // - convert the CoordsBuffer to a vector type that we can filter with the system's Dirichlet filter
490 // - copy the filtered vector's contents back to the buffer and write the buffer to the mesh
491 // - apply the nonlinear filter representing unilateral BCs of place by calling adapt() for the
492 // corresponding meshparts
493 for(size_t pos(meshopt_lvl_pos); pos > size_t(0);)
494 {
495 --pos;
496
497 auto& coarse_mesh = this->_dom_ctrl.at(pos+1)->get_mesh();
498 auto& fine_mesh = this->_dom_ctrl.at(pos)->get_mesh();
499 auto& fine_vtx = fine_mesh.get_vertex_set();
500
501 // Refine coarse vertex set and write the result to the CoordsBuffer
502 Geometry::StandardRefinery<MeshType> refinery(coarse_mesh);
503 refinery.fill_vertex_set(fine_vtx);
504 _system_levels.at(pos)->global_functional.local().mesh_to_buffer();
505 // Convert the buffer to a filterable vector
506 typename SystemLevelType::GlobalSystemVectorL::LocalVectorType vec_sol_lvl;
507 vec_sol_lvl.convert(_system_levels.at(pos)->coords_buffer.local());
508 // Filter this vector, copy back the contents and write the changes to the mesh
509 auto& dirichlet_filters_lvl = (_system_levels.at(pos)->filter_sys).local().template at<1>();
510 dirichlet_filters_lvl.filter_sol(vec_sol_lvl);
511 _system_levels.at(pos)->coords_buffer.local().copy(vec_sol_lvl);
512 _system_levels.at(pos)->global_functional.local().buffer_to_mesh();
513 // Now call adapt() on the slip boundaries
514 auto* fine_mesh_node = this->_dom_ctrl.at(pos)->get_mesh_node();
515 for(const auto& it:this->get_slip_boundaries())
516 {
517 fine_mesh_node->adapt_by_name(it);
518 }
519 }
520
521 // Now we need to update all coarser levels
522 for(size_t pos(meshopt_lvl_pos+1); pos < get_num_levels(); ++pos)
523 {
524 Index ndofs(_system_levels.at(pos)->local_functional.trafo_space.get_num_dofs());
525
526 // At this point, what we really need is a primal restriction operator that restricts the FE function
527 // representing the coordinate distribution to the coarser level. This is very simple for continuous
528 // Lagrange elements (just discard the additional information from the fine level), but not clear in
529 // the generic case. So we use an evil hack here:
530 // Because of the underlying two level ordering, we just need to copy the first ndofs entries from
531 // the fine level vector.
532 typename SystemLevelType::GlobalCoordsBuffer
533 global_sol_level( &(_system_levels.at(pos)->gate_sys), vec_sol.local(), ndofs, Index(0));
534
535 _system_levels.at(pos)->global_functional.prepare(
536 global_sol_level, _system_levels.at(pos)->filter_sys);
537
538 _system_levels.at(pos)->global_functional.local().get_coords().copy(global_sol_level.local());
539 _system_levels.at(pos)->global_functional.local().buffer_to_mesh();
540 }

588 void _create_nonlinear_optimizer()
589 {
590 // query our solver section
591 PropertyMap* solver_section = solver_config.query_section(solver_name);
592 String solver_type = "none";
593 String nlcg_section_name = solver_name;
594
595 // Get the String identifying the solver type
596 auto solver_p = solver_section->query("type");
597 if (!solver_p.second)
598 {
599 XABORTM("no type key found in property map: " + solver_name + "!");
600 }
601 solver_type = solver_p.first;
602
603 // there are only 2 allowed solver types: 'QPenalty' or 'NLCG'
604 // QPenalty might be used as an outer solver around NLCG, so let's check for this
605
606 const bool is_qpenalty = (solver_type == "QPenalty");
607 PropertyMap* qpenalty_section = nullptr;
608 if(is_qpenalty)
609 {
610 // ok, use QPenalty on the outside, so get its inner solver
611 qpenalty_section = solver_section;
612
613 // Create inner solver first
614 auto inner_solver_p = solver_section->query("inner_solver");
615 // Safety catches for recursions
616 XASSERTM(inner_solver_p.second, "QPenalty solver section is missing mandatory inner_solver key.");
617 XASSERTM(inner_solver_p.first != "QPenalty", "QPenalty cannot be the inner solver for QPenalty.");
618
619 // now replace our solver_section pointer by the inner solver, which must be NLCG now
620 nlcg_section_name = inner_solver_p.first;
621 solver_section = solver_config.query_section(inner_solver_p.first);
622 auto solver_type_p = solver_section->query("type");
623 if (!solver_type_p.second)
624 {
625 XABORTM("no type key found in property map: " + inner_solver_p.first + "!");
626 }
627 solver_type = solver_type_p.first;
628 }
629
630 // now the solver type must be 'NLCG'
631 if(solver_type != "NLCG")
632 {
633 XABORTM("Solver type key "+stringify(solver_type)+" unknown.");
634 }
635
636 // okay, let's create an NLCG
637 String linesearch_name("");
638 auto linesearch_p = solver_section->query("linesearch");
639 if(linesearch_p.second)
640 {
641 linesearch_name = linesearch_p.first;
642 }
643 else
644 {
645 XABORTM("NLCG config section is missing the mandatory linesearch key!");
646 }
647
648 // create the linesearch
649 _create_linesearch(linesearch_name);
650
651 // create the actual NLCG solver
652 solver = Solver::new_nlcg(nlcg_section_name, solver_section,
653 _system_levels.at(meshopt_lvl_pos)->global_functional,
654 _system_levels.at(meshopt_lvl_pos)->filter_sys,
655 linesearch, precond);
656
657 // if we wanted a QPenalty in the first place, create it around the NLCG now
658 if(is_qpenalty)
659 {
660 solver = Solver::new_qpenalty(
661 solver_name, qpenalty_section, _system_levels.at(meshopt_lvl_pos)->global_functional, solver);
662 }
663
664 // finally, initialize it
665 solver->init();
666 solver->set_plot_name(solver->name() + " (meshopt-Hyper)");
667 }
