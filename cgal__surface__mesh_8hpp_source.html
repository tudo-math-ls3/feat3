169 void project_meshpart(Mesh_& mesh, const MeshPart<Mesh_>& meshpart) const
170 {
171 Index num_facets(meshpart.get_num_entities(shape_dim));
172
173 // There is nothing to do if the meshpart does not have any cells (which refer to facets of the original
174 // mesh)
175 if(num_facets == Index(0))
176 return;
177
178 // The number of vertices that need to be projected
179 const Index num_verts(meshpart.get_num_entities(0));
180 // Mapping of vertices from the meshpart to the real mesh
181 const auto& ts_verts(meshpart.template get_target_set<0>());
182
183 //get vertex set of mesh
184 auto& vtx(mesh.get_vertex_set());
185
186 //we could simply go through the vertices of the meshpart, project and hope for the best...
187 // but we should and can at least check, if the new facet is at about the same distance away
188 for(Index i = 0; i < num_verts; ++i)
189 {
190 project_point(vtx[ts_verts[i]]);
191 }
192 //already done
193 }

197 CoordType compute_dist(const WorldPoint& point) const
198 {
199 return Math::sqrt(cgal.squared_distance(point[0], point[1], point[2]));
200 }
201
204 CoordType compute_dist(const WorldPoint& point, WorldPoint& grad_distance) const
205 {
206 // WorldPoint normal;
207 WorldPoint projected_point(point);
208 project_point(projected_point);
209 // project_point(projected_point, normal);
210
211 grad_distance = (projected_point - point);
212 CoordType dist = grad_distance.norm_euclid();
213
214 // If the distance is too small, we set the gradient vector to zero
215 if(grad_distance.norm_euclid() < Math::eps<CoordType>())
216 {
217 grad_distance.format(CoordType(0));
218 }
219 else
220 {
221 grad_distance.normalize();
222 }
223
224 return dist;
225 }

228 CoordType compute_signed_dist(const WorldPoint& point) const
229 {
