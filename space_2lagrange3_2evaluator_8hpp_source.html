33 // p0, p1, p2 and p3 are the 1D basis functions on the reference interval [-1,+1].
34 // These are used for the tensor-product approach in the Hypercube evaluators.
35
36 // basis function for left vertex
37 template<typename T_>
38 inline T_ p0(T_ x)
39 {
40 return T_(0.0625) * (-T_(1) + x*( T_(1) + T_(9)*x*(T_(1) - x)));
41 }
42
43 // basis function for right vertex
44 template<typename T_>
45 inline T_ p1(T_ x)
46 {
47 return T_(0.0625) * (-T_(1) + x*(-T_(1) + T_(9)*x*(T_(1) + x)));
48 }
49
50 // basis function for first edge point
51 template<typename T_>
52 inline T_ p2(T_ x)
53 {
54 return T_(0.5625) * (T_(1) + x*(-T_(3) + x*(-T_(1) + T_(3)*x)));
55 }
56
57 // basis function for second edge point
58 template<typename T_>
59 inline T_ p3(T_ x)
60 {
61 return T_(0.5625) * (T_(1) + x*( T_(3) + x*(-T_(1) - T_(3)*x)));
62 }
63
64 // first order derivatives
65
66 template<typename T_>
67 inline T_ d1p0(T_ x)
68 {
69 return T_(0.0625) * ( T_(1) + x*(T_(18) - T_(27)*x));
70 }
71
72 template<typename T_>
73 inline T_ d1p1(T_ x)
74 {
75 return T_(0.0625) * (-T_(1) + x*(T_(18) + T_(27)*x));
76 }
77
78 template<typename T_>
79 inline T_ d1p2(T_ x)
80 {
81 return T_(0.0625) * (-T_(27) + x*(-T_(18) + T_(81)*x));
82 }
83
84 template<typename T_>
85 inline T_ d1p3(T_ x)
86 {
87 return T_(0.0625) * ( T_(27) + x*(-T_(18) - T_(81)*x));
88 }
89
90 // second order derivatives
91
92 template<typename T_>
93 inline T_ d2p0(T_ x)
94 {
95 return T_(1.125) * (T_(1) - T_(3)*x);
96 }
97
98 template<typename T_>
99 inline T_ d2p1(T_ x)
100 {
101 return T_(1.125) * (T_(1) + T_(3)*x);
102 }
103
104 template<typename T_>
105 inline T_ d2p2(T_ x)
106 {
107 return T_(1.125) * (-T_(1) + T_(9)*x);
108 }
109
110 template<typename T_>
111 inline T_ d2p3(T_ x)
112 {
113 return T_(1.125) * (-T_(1) - T_(9)*x);
114 }
