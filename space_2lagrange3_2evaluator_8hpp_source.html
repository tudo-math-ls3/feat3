32 // p0, p1, p2 and p3 are the 1D basis functions on the reference interval [-1,+1].
33 // These are used for the tensor-product approach in the Hypercube evaluators.
34
35 // basis function for left vertex
36 template<typename T_>
37 inline T_ p0(T_ x)
38 {
39 return T_(0.0625) * (-T_(1) + x*( T_(1) + T_(9)*x*(T_(1) - x)));
40 }
41
42 // basis function for right vertex
43 template<typename T_>
44 inline T_ p1(T_ x)
45 {
46 return T_(0.0625) * (-T_(1) + x*(-T_(1) + T_(9)*x*(T_(1) + x)));
47 }
48
49 // basis function for first edge point
50 template<typename T_>
51 inline T_ p2(T_ x)
52 {
53 return T_(0.5625) * (T_(1) + x*(-T_(3) + x*(-T_(1) + T_(3)*x)));
54 }
55
56 // basis function for second edge point
57 template<typename T_>
58 inline T_ p3(T_ x)
59 {
60 return T_(0.5625) * (T_(1) + x*( T_(3) + x*(-T_(1) - T_(3)*x)));
61 }
62
63 // first order derivatives
64
65 template<typename T_>
66 inline T_ d1p0(T_ x)
67 {
68 return T_(0.0625) * ( T_(1) + x*(T_(18) - T_(27)*x));
69 }
70
71 template<typename T_>
72 inline T_ d1p1(T_ x)
73 {
74 return T_(0.0625) * (-T_(1) + x*(T_(18) + T_(27)*x));
75 }
76
77 template<typename T_>
78 inline T_ d1p2(T_ x)
79 {
80 return T_(0.0625) * (-T_(27) + x*(-T_(18) + T_(81)*x));
81 }
82
83 template<typename T_>
84 inline T_ d1p3(T_ x)
85 {
86 return T_(0.0625) * ( T_(27) + x*(-T_(18) - T_(81)*x));
87 }
88
89 // second order derivatives
90
91 template<typename T_>
92 inline T_ d2p0(T_ x)
93 {
94 return T_(1.125) * (T_(1) - T_(3)*x);
95 }
96
97 template<typename T_>
98 inline T_ d2p1(T_ x)
99 {
100 return T_(1.125) * (T_(1) + T_(3)*x);
101 }
102
103 template<typename T_>
104 inline T_ d2p2(T_ x)
105 {
106 return T_(1.125) * (-T_(1) + T_(9)*x);
107 }
108
109 template<typename T_>
110 inline T_ d2p3(T_ x)
111 {
112 return T_(1.125) * (-T_(1) - T_(9)*x);
113 }
