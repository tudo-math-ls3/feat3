156 template<typename LambdaSet_, int dim_, typename DT_, typename VT_, typename GT_>
157 class LambdaGradHelper<LambdaSet_, dim_, DT_, VT_, GT_, false>
158 {
159 public:
161 LambdaSet_& lambda_set;
163 const DT_ initial_h;
165 const std::size_t max_steps;
166
167 explicit LambdaGradHelper(LambdaSet_& lambda_set_, DT_ initial_h_, int max_steps_) :
168 lambda_set(lambda_set_),
169 initial_h(initial_h_),
170 max_steps(std::size_t(max_steps_))
171 {
172 }
173
175 void eval_grad(GT_& grad, const Tiny::Vector<DT_, dim_>& point)
176 {
177 // first, create a mutable copy of our point
178 Tiny::Vector<DT_, dim_> v(point);
179
181 std::vector<GT_> gradex(max_steps);
182
183 // next, choose the initial h
184 DT_ h(initial_h);
185
186 // evaluate gradient
187 eval_grad_diff_quot(gradex[0], v, h);
188
189 // Now comes the Richardson extrapolation loop
190 const std::size_t n(gradex.size()-1);
191 DT_ def(Math::huge<DT_>());
192 for(std::size_t i(0); i < n; ++i)
193 {
194 // evaluate next gradient
195 eval_grad_diff_quot(gradex[i+1], v, h *= DT_(0.5));
196
197 // initialize scaling factor
198 DT_ q = DT_(1);
199
200 // perform extrapolation steps except for the last one
201 for(std::size_t k(i); k > std::size_t(0); --k)
202 {
203 q *= DT_(4);
204 (gradex[k] -= q*gradex[k+1]) *= DT_(1) / (DT_(1) - q);
205 }
206
207 // compute and check our defect
208 DT_ d = def_norm_sqr(gradex[1], gradex[0]);
209 if(def <= d)
210 {
211 // The defect has increased, so we return the previous extrapolation result
212 break;
213 }
214
215 // remember current defect
216 def = d;
217
218 // perform last extrapolation step
219 q *= DT_(4);
220 (gradex[0] -= q*gradex[1]) *= DT_(1) / (DT_(1) - q);
221 }
222
223 // return our extrapolated gradient
224 grad = gradex.front();
225 }

229 static DT_ def_norm_sqr(const Tiny::Vector<DT_, n_, s_>& x, const Tiny::Vector<DT_, n_, s_>& y)
230 {
231 return (x - y).norm_euclid_sqr();

235 template<int m_, int n_, int sm_, int sn_>
236 static DT_ def_norm_sqr(
237 const Tiny::Matrix<DT_, m_, n_, sm_, sn_>& x,
238 const Tiny::Matrix<DT_, m_, n_, sm_, sn_>& y)

243 // compute scalar difference quotient
244 static void set_diff_quot(Tiny::Vector<DT_, dim_>& x, int i, const VT_& vr, const VT_& vl, const DT_ denom)
245 {
246 x[i] = denom * (vr - vl);
247 }
248
249 // compute vector difference quotient
250 static void set_diff_quot(Tiny::Matrix<DT_, dim_, dim_>& x, int i, const VT_& vr, const VT_& vl, const DT_ denom)
251 {
252 for(int k(0); k < dim_; ++k)
253 x[k][i] = denom * (vr[k] - vl[k]);
254 }
255
257 void eval_grad_diff_quot(GT_& x, Tiny::Vector<DT_, dim_>& v, const DT_ h)
258 {
259 // difference quotient denominator
260 const DT_ denom = DT_(1) / (DT_(2) * h);
261 VT_ vr, vl;
262
263 // loop over all dimensions
264 for(int i(0); i < dim_; ++i)
265 {
266 // backup coordinate i
267 const DT_ vi(v[i]);
268
269 // evaluate f(v + h*e_i)
270 v[i] = vi + h;
271 vr = lambda_set.eval_value(v);
272
273 // evaluate f(v - h*e_i)
274 v[i] = vi - h;
275 vl = lambda_set.eval_value(v);
276
277 // compute difference quotient
278 set_diff_quot(x, i, vr, vl, denom);
279
280 // restore coord
281 v[i] = vi;
282 }
283 }

467 static DT_ def_norm_sqr(

475 template<int l_, int m_, int n_, int sl_, int sm_, int sn_>
476 static DT_ def_norm_sqr(
477 const Tiny::Tensor3<DT_, l_, m_, n_, sl_, sm_, sn_>& x,
478 const Tiny::Tensor3<DT_, l_, m_, n_, sl_, sm_, sn_>& y)
479 {
480 DT_ r(DT_(0));
481 for(int i(0); i < l_; ++i)
482 {
483 for(int j(0); j < m_; ++j)
484 {
485 for(int k(0); k < n_; ++k)
486 {
487 r += Math::sqr(x(i,j,k) - y(i,j,k));
488 }
489 }
490 }
491 return r;

494 // compute scalar difference quotient for i == j
495 static void set_diff_quot(Tiny::Matrix<DT_, dim_, dim_>& x, int i, const VT_& vr, const VT_& vl, const VT_& vc, const DT_ denom)
496 {
497 x[i][i] = denom * (vr + vl - DT_(2)*vc);
498 }
499
500 // compute scalar difference quotient for i != j
501 static void set_diff_quot(Tiny::Matrix<DT_, dim_, dim_>& x, int i, int j, const VT_& vne, const VT_& vsw, const VT_& vnw, const VT_& vse, const DT_ denom)
502 {
503 x[i][j] = x[j][i] = denom * ((vne + vsw) - (vnw + vse));
504 }
505
506 // compute vector difference quotient for i == j
507 static void set_diff_quot(Tiny::Tensor3<DT_, dim_, dim_, dim_>& x, int i, const VT_& vr, const VT_& vl, const VT_& vc, const DT_ denom)
508 {
509 for(int k(0); k < dim_; ++k)
510 x[k][i][i] = denom * (vr[k] + vl[k] - DT_(2)*vc[k]);
511 }
512
513 // compute vector difference quotient for i != j
514 static void set_diff_quot(Tiny::Tensor3<DT_, dim_, dim_, dim_>& x, int i, int j, const VT_& vne, const VT_& vsw, const VT_& vnw, const VT_& vse, const DT_ denom)
515 {
516 for(int k(0); k < dim_; ++k)
517 x[k][i][j] = x[k][j][i] = denom * ((vne[k] + vsw[k]) - (vnw[k] + vse[k]));
518 }
519
521 void eval_hess_diff_quot(HT_& x, Tiny::Vector<DT_, dim_>& v, const DT_ h)
522 {
523 // difference quotient denominators
524 const DT_ denom1 = DT_(1) / (h * h);
525 const DT_ denom2 = DT_(1) / (DT_(4) * h * h);
526 VT_ vc, vr, vl, vne, vnw, vse, vsw;
527
528 // evaluate at point
529 vc = lambda_set.eval_value(v);
530
531 // loop over all dimensions
532 for(int i(0); i < dim_; ++i)
533 {
534 // backup coord
535 const DT_ vi(v[i]);
536
537 // eval f(x + h*e_i)
538 v[i] = vi + h;
539 vr = lambda_set.eval_value(v);
540
541 // eval f(x-h)
542 v[i] = vi - h;
543 vl = lambda_set.eval_value(v);
544
545 // compute difference quotient
546 set_diff_quot(x, i, vr, vl, vc, denom1);
547
548 // now the mixed derivatives
549 for(int j(i+1); j < dim_; ++j)
550 {
551 // backup coord
552 const DT_ vj(v[j]);
553
554 // we need four points here:
555 // north-east: f(v + h*e_i + h*e_j)
556 v[i] = vi + h;
557 v[j] = vj + h;
558 vne = lambda_set.eval_value(v);
559
560 // north-west: f(v - h*e_i + h*e_j)
561 v[i] = vi - h;
562 v[j] = vj + h;
563 vnw = lambda_set.eval_value(v);
564
565 // south-east: f(v + h*e_i - h*e_j)
566 v[i] = vi + h;
567 v[j] = vj - h;
568 vse = lambda_set.eval_value(v);
569
570 // south-west: f(v - h*e_i - h*e_j)
571 v[i] = vi - h;
572 v[j] = vj - h;
573 vsw = lambda_set.eval_value(v);
574
575 // combine into difference quotient
576 set_diff_quot(x, i, j, vne, vsw, vnw, vse, denom2);
577
578 // restore coord
579 v[j] = vj;
580 }
581
582 // restore coord
583 v[i] = vi;
584 }
585 }
