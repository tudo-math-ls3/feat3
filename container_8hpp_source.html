84 explicit SerialConfig(bool zlib_compression, bool zfp_compression, FEAT::Real tol = FEAT::Real(-1.))
85 {
86 set_elements_compression(CompressionModes::elements_off);
87 set_indices_compression(CompressionModes::indices_off);
88 if(zlib_compression)
89 {
90 set_elements_compression(CompressionModes::elements_zlib);
91 set_indices_compression(CompressionModes::indices_zlib);
92 }
93 if(zfp_compression)
94 {
95 set_tolerance(tol);
96 set_elements_compression(CompressionModes::elements_zfp);
97 }

106 void set_elements_compression(CompressionModes comp)
107 {
108 CompressionModes temp = comp & CompressionModes::elements_mask;
109 switch(temp)
110 {
111 case CompressionModes::elements_off:
112 break;
113 case CompressionModes::elements_zlib:
114#ifdef FEAT_HAVE_ZLIB
115 break;
116#else
117 XABORTM("Zlib compression was enabled for elements, but zlib bibliography was not loaded. Please configure FEAT with -zlib flag!");
118#endif
119 case CompressionModes::elements_zfp:
120#ifdef FEAT_HAVE_ZFP
121 XASSERTM(tolerance > FEAT::Real(0.), "tolerance ist smaller or equall 0 while zfp is enabled!");
122 break;
123#else
124 XABORTM("Zfp compression was enabled for elements, but zfp bibliography was not loaded. Please configure FEAT with -zfp flag!");
125#endif
126 default:
127 XABORTM("No legal CompressionModes was given");
128 }
129 elements_compression = temp;
130 }
137 void set_indices_compression(CompressionModes comp)
138 {
139 CompressionModes temp = comp & CompressionModes::indices_mask;
140 switch(temp)
141 {
142 case CompressionModes::indices_off:
143 break;
144 case CompressionModes::indices_zlib:
145#ifdef FEAT_HAVE_ZLIB
146 break;
147#else
148 XABORTM("Zlib compression was enabled for indices, but zlib bibliography was not loaded. Please configure FEAT with -zlib flag!");
149#endif
150 default:
151 XABORTM("No legal CompressionModes was given");
152 }
153 indices_compression = temp;
154 }
155

164 if(elements_compression == CompressionModes::elements_zfp)
165 {
166 XASSERTM(tol > FEAT::Real(0.), "Zfp enabled, but tolerance is smaller or equal to 0!");

190 }

220 template <typename DT_, typename IT_>
221 class Container
222 {
223 template <typename DT2_, typename IT2_>
224 friend class Container;
225
226 protected:
228 std::vector<DT_*> _elements;
230 std::vector<IT_*> _indices;
232 std::vector<Index> _elements_size;
234 std::vector<Index> _indices_size;
236 std::vector<Index> _scalar_index;
238 std::vector<DT_> _scalar_dt;
240 bool _foreign_memory;
241
242 void _copy_content(const Container & other, bool full)
243 {
244 // avoid self-copy
245 if(this == &other)
246 return;
247
248 XASSERTM(_elements.size() == other.get_elements().size(), "Container size mismatch!");
249 XASSERTM(_indices.size() == other.get_indices().size(), "Container size mismatch!");
250 XASSERTM(_scalar_index.size() == other.get_scalar_index().size(), "Container size mismatch!");
251 XASSERTM(_scalar_dt.size() == other.get_scalar_dt().size(), "Container size mismatch!");
252
253 if (full)
254 {
255 this->_scalar_index.assign(other._scalar_index.begin(), other._scalar_index.end());
256 this->_scalar_dt.assign(other._scalar_dt.begin(), other._scalar_dt.end());
257
258 for (Index i(0) ; i < _indices.size() ; ++i)
259 {
260 XASSERTM(_indices_size.at(i) == other.get_indices_size().at(i), "Container size mismatch!");
261 MemoryPool::copy(_indices.at(i), other.get_indices().at(i), _indices_size.at(i));
262 }
263 }
264
265 for (Index i(0) ; i < _elements.size() ; ++i)
266 {
267 XASSERTM(_elements_size.at(i) == other.get_elements_size().at(i), "Container size mismatch!");
268 MemoryPool::copy(_elements.at(i), other.get_elements().at(i), _elements_size.at(i));
269 }
270
271 }

281 template <typename DT2_, typename IT2_>
282 void assign(const Container<DT2_, IT2_> & other)
283 {
284 for (Index i(0) ; i < this->_elements.size() ; ++i)
285 MemoryPool::release_memory(this->_elements.at(i));
286 for (Index i(0) ; i < this->_indices.size() ; ++i)
287 MemoryPool::release_memory(this->_indices.at(i));
288
289 this->_elements.clear();
290 this->_indices.clear();
291 this->_elements_size.clear();
292 this->_indices_size.clear();
293 this->_scalar_index.clear();
294 this->_scalar_dt.clear();
295
296 this->_elements_size.assign(other.get_elements_size().begin(), other.get_elements_size().end());
297 this->_indices_size.assign(other.get_indices_size().begin(), other.get_indices_size().end());
298 this->_scalar_index.assign(other.get_scalar_index().begin(), other.get_scalar_index().end());
299 if constexpr (std::is_same<DT_, DT2_>::value)
300 {
301 this->_scalar_dt.assign(other.get_scalar_dt().begin(), other.get_scalar_dt().end());
302
303 }
304 else
305 {
306 for(auto t : other.get_scalar_dt())
307 {
308 this->_scalar_dt.push_back(DT_(t));
309 }
310 }
311
312 if constexpr (std::is_same<DT_, DT2_>::value)
313 {
314 this->_elements.assign(other.get_elements().begin(), other.get_elements().end());
315
316 for (Index i(0) ; i < this->_elements.size() ; ++i)
317 MemoryPool::increase_memory(this->_elements.at(i));
318 }
319 else
320 {
321 for (Index i(0) ; i < this->_elements_size.size() ; ++i)
322 {
323 const Index tsize(this->_elements_size.at(i));
324 this->_elements.push_back(MemoryPool::template allocate_memory<DT_>(tsize));
325 MemoryPool::convert(this->_elements.at(i), other.get_elements().at(i), tsize);
326 }
327 }
328
329 if constexpr (std::is_same<IT_, IT2_>::value)
330 {
331 this->_indices.assign(other.get_indices().begin(), other.get_indices().end());
332
333 for (Index i(0) ; i < this->_indices.size() ; ++i)
334 MemoryPool::increase_memory(this->_indices.at(i));
335 }
336 else
337 {
338 for (Index i(0) ; i < this->_indices_size.size() ; ++i)
339 {
340 const Index tsize(this->_indices_size.at(i));
341 this->_indices.push_back(MemoryPool::template allocate_memory<IT_>(tsize));
342 MemoryPool::convert(this->_indices.at(i), other.get_indices().at(i), tsize);
343 }
344 }

356 template <typename DT2_ = DT_, typename IT2_ = IT_>
357 std::uint64_t _serialized_size(const LAFEM::SerialConfig& config = LAFEM::SerialConfig()) const
358 {
359 Container<DT2_, IT2_> tc(0);
360 tc.assign(*this);
361
362 std::uint64_t gsize(4 * sizeof(std::uint64_t)); //raw array size + magic number + type_index DT_ + type_index IT_
363 gsize += 7 * sizeof(std::uint64_t); // size of all seven stl containers
364 gsize += 2 * tc._elements_size.size() * sizeof(std::uint64_t); // _elements_size contents + _elements_arrays bytesize
365 gsize += 2 * tc._indices_size.size() * sizeof(std::uint64_t); // _indices_size contents + _indizes_arrays bytesize
366 gsize += tc._scalar_index.size() * sizeof(std::uint64_t); // _scalar_index contents
367 gsize += tc._scalar_dt.size() * sizeof(DT2_); // _scalar_dt contents
368
369 CompressionModes compress = config.get_elements_compression() | config.get_indices_compression();
370 Pack::Type compression_type_elements = Pack::deduct_type<DT2_>();
371 Pack::Type compression_type_index = Pack::deduct_type<IT2_>();
372 if((compress & CompressionModes::elements_mask) == CompressionModes::elements_zfp) // If zfp is used for elements
373 {
374 compression_type_elements = compression_type_elements | Pack::Type::Mask_P;
375 }
376 else if((compress & CompressionModes::elements_mask) == CompressionModes::elements_zlib) //If zlib is used and zfp is not used elements
377 {
378 compression_type_elements = compression_type_elements | Pack::Type::Mask_Z;
379 }
380 if((compress & CompressionModes::indices_mask) == CompressionModes::indices_zlib) // If zlib is used for indices
381 compression_type_index = compression_type_index | Pack::Type::Mask_Z;
382
383
384 FEAT::Real tolerance = config.get_tolerance();
385
386 for (Index i(0) ; i < tc._elements_size.size() ; ++i)
387 {
388 gsize += Pack::estimate_size(tc._elements_size.at(i), compression_type_elements, (double)tolerance); // upper_bound for (compressed) elements
389 }
390
391 for (Index i(0) ; i < tc._indices_size.size() ; ++i)
392 {
393 gsize += Pack::estimate_size(tc._indices_size.at(i), compression_type_index); // upper_bound for (compressed) indizes
394 }
395 gsize += 16; //padding for datatype alignment mismatch
396
397 return gsize;
398 }

439 template <typename DT2_ = DT_, typename IT2_ = IT_>
440 std::vector<char> _serialize(FileMode mode, const SerialConfig& config = SerialConfig()) const
441 {
442 std::uint64_t raw_size = 0u;
443 FEAT::Real tolerance = config.get_tolerance();
444
445 CompressionModes compress = config.get_elements_compression() | config.get_indices_compression();
446 Pack::Type compression_type_elements = Pack::deduct_type<DT2_>();
447 Pack::Type compression_type_index = Pack::deduct_type<IT2_>();
448 if((compress & CompressionModes::elements_mask) == CompressionModes::elements_zfp) // If zfp is used for elements
449 {
450 compression_type_elements = compression_type_elements | Pack::Type::Mask_P;
451 }
452 else if((compress & CompressionModes::elements_mask) == CompressionModes::elements_zlib) //If zlib is used and zfp is not used elements
453 {
454 compression_type_elements = compression_type_elements | Pack::Type::Mask_Z;
455 }
456 if((compress & CompressionModes::indices_mask) == CompressionModes::indices_zlib) // If zlib is used for indices
457 compression_type_index = compression_type_index | Pack::Type::Mask_Z;
458
459 Container<DT2_, IT2_> tc(0);
460 tc.assign(*this);
461
462 std::uint64_t gsize = this->template _serialized_size<DT2_, IT2_>(config);
463
464 std::vector<char> result((size_t(gsize)));
465 char * array(result.data());
466 std::uint64_t * uiarray(reinterpret_cast<std::uint64_t *>(array));
467 DT2_ * dtarray(reinterpret_cast<DT2_ *>(array));
468 IT2_ * itarray(reinterpret_cast<IT2_ *>(array));
469
471 #if defined(FEAT_COMPILER_CLANG)
472 std::uint64_t magic = (std::uint64_t)static_cast<__underlying_type(FileMode)>(mode);
473 #else
474 std::uint64_t magic = (std::uint64_t)static_cast<typename std::underlying_type<FileMode>::type>(mode);
475 #endif
476 uiarray[0] = gsize; //we will overwrite this at the end with the finalized data...
477 uiarray[1] = magic;
478 uiarray[2] = Type::Traits<DT_>::feature_hash();
479 uiarray[3] = Type::Traits<IT_>::feature_hash();
480 uiarray[4] = tc._elements.size();
481 uiarray[5] = tc._indices.size();
482 uiarray[6] = tc._elements_size.size();
483 uiarray[7] = tc._indices_size.size();
484 uiarray[8] = tc._scalar_index.size();
485 uiarray[9] = tc._scalar_dt.size();
486 uiarray[10] = (std::uint64_t)compress;
487 raw_size += 11 * (std::uint64_t)sizeof(std::uint64_t);
488
489 Index global_i(11); // count how many elements of std::uint64_t have been inserted so far
490 Index local_elements(0); // count where elements array bytesize starts
491 Index local_index(0); // same for index...
492
493 for (Index i(0) ; i < tc._elements_size.size() ; ++i)
494 {
495 uiarray[i + global_i] = tc._elements_size.at(i);
496 }
497 global_i += Index(tc._elements_size.size());
498 local_elements = global_i;
499 for (Index i(0) ; i < tc._elements_size.size() ; ++i)
500 {
501 uiarray[i + global_i] = (std::uint64_t)0u; //placeholder, until we calculate the real data...
502 }
503 global_i += Index(tc._elements_size.size());
504 raw_size += 2 * std::uint64_t(tc._elements_size.size()) * std::uint64_t(sizeof(std::uint64_t));
505
506 for (Index i(0) ; i < tc._indices_size.size() ; ++i)
507 {
508 uiarray[i + global_i] = tc._indices_size.at(i);
509 }
510 global_i += Index(tc._indices_size.size());
511 local_index = global_i;
512 for (Index i(0) ; i < tc._indices_size.size() ; ++i)
513 {
514 uiarray[i + global_i] = 0;
515 }
516 global_i += Index(tc._indices_size.size());
517 raw_size += 2 * std::uint64_t(tc._indices_size.size()) * std::uint64_t(sizeof(std::uint64_t));
518
519 for (Index i(0) ; i < tc._scalar_index.size() ; ++i)
520 {
521 uiarray[i + global_i] = tc._scalar_index.at(i);
522 }
523 global_i += Index(tc._scalar_index.size());
524 raw_size += std::uint64_t(tc._scalar_index.size()) * std::uint64_t(sizeof(std::uint64_t));
525
526 global_i = Index((global_i * sizeof(std::uint64_t) + sizeof(DT2_) - 1u) / sizeof(DT2_)); // now counting how many DT2 have been inserted so far
527
528 for (Index i(0) ; i < tc._scalar_dt.size() ; ++i)
529 {
530 dtarray[i + global_i] = tc._scalar_dt.at(i);
531 }
532 global_i += Index(tc._scalar_dt.size());
533 raw_size += std::uint64_t(tc._scalar_dt.size()) * std::uint64_t(sizeof(DT2_));
534
535 if((compress & CompressionModes::elements_mask) != CompressionModes::elements_off)
536 {
537 global_i = Index((global_i * sizeof(DT2_) + sizeof(char) - 1u) / sizeof(char)); // now counting how many bytes/chars have been inserted so far
538 for (Index i(0) ; i < tc._elements.size() ; ++i)
539 {
540 char* dest = &result[global_i];
541 std::size_t est_buff_size = Pack::estimate_size(tc._elements_size.at(i), compression_type_elements, (double)tolerance);
542 std::size_t real_size = Pack::encode<DT2_>(dest, tc._elements.at(i), est_buff_size, tc._elements_size.at(i), compression_type_elements, false, (double)tolerance);
543 XASSERTM(real_size != 0u, "could not encode elements");
544 uiarray[i + local_elements] = (std::uint64_t)real_size; //save used memory for compressed array
545 global_i += (Index)real_size; //go forward the number of bytes(chars)
546 raw_size += (std::uint64_t)real_size * (std::uint64_t)sizeof(char);
547 }
548 global_i = Index((global_i * sizeof(char) + sizeof(DT2_) - 1u) / sizeof(DT2_)); // go back to DT2_ indexing
549 }
550 else
551 {
552 for (Index i(0) ; i < tc._elements.size() ; ++i)
553 {
554 std::memcpy(&dtarray[global_i], tc._elements.at(i), tc._elements_size.at(i) * sizeof(DT2_));
555 uiarray[i + local_elements] = (std::uint64_t) tc._elements_size.at(i) * sizeof(DT2_);
556 global_i += tc._elements_size.at(i);
557 raw_size += (std::uint64_t) tc._elements_size.at(i) * sizeof(DT2_);
558 }
559 }
560 if((compress & CompressionModes::indices_mask) != CompressionModes::indices_off)
561 {
562 global_i = Index((global_i * sizeof(DT2_) + sizeof(char) - 1u) / sizeof(char)); // now counting how many bytes/chars have been inserted so far
563 for (Index i(0) ; i < tc._indices.size() ; ++i)
564 {
565 char* dest = &result[global_i];
566 std::size_t est_buff_size = Pack::estimate_size(tc._indices_size.at(i), compression_type_index);
567 std::size_t real_size = Pack::encode<IT2_>(dest, tc._indices.at(i), est_buff_size, tc._indices_size.at(i), compression_type_index, false);
568 XASSERTM(real_size != 0u, "could not encode indices");
569 uiarray[i + local_index] = (std::uint64_t) real_size;
570 global_i += Index(real_size);
571 raw_size += std::uint64_t(real_size) * std::uint64_t(sizeof(char));
572 }
573 }
574 else
575 {
576 global_i = Index((global_i * sizeof(DT2_) + sizeof(IT2_) - 1u) / sizeof(IT2_)); // now counting IT2_ elements
577 for (Index i(0) ; i < tc._indices.size() ; ++i)
578 {
579 std::memcpy(&itarray[global_i], tc._indices.at(i), tc._indices_size.at(i) * sizeof(IT2_));
580 uiarray[i + local_index] = (std::uint64_t) tc._indices_size.at(i) * sizeof(IT2_);
581 global_i += tc._indices_size.at(i);
582 raw_size += (std::uint64_t) tc._indices_size.at(i) * sizeof(IT2_);
583 }
584 }
585 uiarray[0] = raw_size + 16u; //padding
586 //std::cout << "Compressed size is: " << uiarray[0] << std::endl;
587
588 result.resize(std::size_t(uiarray[0]));
589 return result;
590 }

621 template <typename DT2_ = DT_, typename IT2_ = IT_>
622 void _deserialize(FileMode mode, std::vector<char> & input)
623 {
624 this->clear();
625 Container<DT2_, IT2_> tc(0);
626 tc.clear();
627
628 char * array(input.data());
629 std::uint64_t * uiarray(reinterpret_cast<std::uint64_t *>(array));
630 DT2_ * dtarray(reinterpret_cast<DT2_ *>(array));
631 IT2_ * itarray(reinterpret_cast<IT2_ *>(array));
632
634#if defined(FEAT_COMPILER_CLANG)
635 std::uint64_t magic = (std::uint64_t)static_cast<__underlying_type(FileMode)>(mode);
636#else
637 std::uint64_t magic = (std::uint64_t)static_cast<typename std::underlying_type<FileMode>::type>(mode);
638#endif
639 XASSERTM(magic == uiarray[1], "_deserialize: given FileMode incompatible with given array!");
640
641 //ensure that we have the same integral/floating type configuration, that was used when storing the serialized data
642 XASSERT(Type::Traits<DT_>::is_int == Type::Helper::extract_intness(uiarray[2]));
643 XASSERT(Type::Traits<DT_>::is_float == Type::Helper::extract_floatness(uiarray[2]));
644 XASSERT(Type::Traits<DT_>::is_signed == Type::Helper::extract_signedness(uiarray[2]));
645 XASSERT(Type::Traits<IT_>::is_int == Type::Helper::extract_intness(uiarray[3]));
646 XASSERT(Type::Traits<IT_>::is_float == Type::Helper::extract_floatness(uiarray[3]));
647 XASSERT(Type::Traits<IT_>::is_signed == Type::Helper::extract_signedness(uiarray[3]));
648
649 if (sizeof(DT_) > Type::Helper::extract_type_size(uiarray[2]))
650 std::cerr<<"Warning: You are reading a container floating point in higher precision then it was saved before!"<<std::endl;
651
652 if (sizeof(IT_) > Type::Helper::extract_type_size(uiarray[3]))
653 std::cerr<<"Warning: You are reading a container integral type in higher precision then it was saved before!"<<std::endl;
654
655 CompressionModes compress = (CompressionModes)uiarray[10];
656 //test whether system has right third_party software loaded:
657#ifndef FEAT_HAVE_ZLIB
658 XASSERTM((compress & CompressionModes::elements_mask) != CompressionModes::elements_zlib,
659 "Data was compressed with ZLIB! To read in data, please configure FEAT with zlib enabled.\n For more information see FEAT documentation.");
660 XASSERTM((compress & CompressionModes::indices_mask) != CompressionModes::indices_zlib,
661 "Data was compressed with ZLIB! To read in data, please configure FEAT with zlib enabled.\n For more information see FEAT documentation.");
662#endif
663#ifndef FEAT_HAVE_ZFP
664 XASSERTM((compress & CompressionModes::elements_mask) != CompressionModes::elements_zfp,
665 "Data was compressed with zfp! To read in data, please configure FEAT with zfp enabled.\n For more information see FEAT documentation.");
666#endif
667 std::vector<std::uint64_t> elements_bytes(uiarray[6]); //temp arrays to keep track of bytessize of compressed arrays
668 std::vector<std::uint64_t> indices_bytes(uiarray[7]);
669 Index global_i(11);
670 for (std::uint64_t i(0) ; i < uiarray[6] ; ++i)
671 {
672 tc._elements_size.push_back(Index(uiarray[i + global_i]));
673 }
674 global_i += Index(uiarray[6]);
675 for (std::uint64_t i(0) ; i < uiarray[6] ; ++i)
676 {
677 elements_bytes[i] = uiarray[i + global_i];
678 }
679 global_i += Index(uiarray[6]);
680
681 for (std::uint64_t i(0) ; i < uiarray[7] ; ++i)
682 {
683 tc._indices_size.push_back(Index(uiarray[i + global_i]));
684 }
685 global_i += Index(uiarray[7]);
686 for (std::uint64_t i(0) ; i < uiarray[7] ; ++i)
687 {
688 indices_bytes[i] = uiarray[i + global_i];
689 }
690 global_i += Index(uiarray[7]);
691
692 for (std::uint64_t i(0) ; i < uiarray[8] ; ++i)
693 {
694 tc._scalar_index.push_back(Index(uiarray[i + global_i]));
695 }
696 global_i += Index(uiarray[8]);
697
698 global_i = Index((global_i * sizeof(std::uint64_t) + sizeof(DT2_) - 1u) / sizeof(DT2_));
699 for (std::uint64_t i(0) ; i < uiarray[9] ; ++i)
700 {
701 tc._scalar_dt.push_back(dtarray[i + global_i]);
702 }
703 global_i += Index(uiarray[9]);
704
705 Pack::Type compression_type_elements = Pack::deduct_type<DT2_>();
706 Pack::Type compression_type_index = Pack::deduct_type<IT2_>();
707 if((compress & CompressionModes::elements_mask) == CompressionModes::elements_zfp)// If zfp is used
708 {
709 compression_type_elements = compression_type_elements | Pack::Type::Mask_P;
710 }
711 else if((compress & CompressionModes::elements_mask) == CompressionModes::elements_zlib) //If zlib is loaded and zfp is not used
712 {
713 compression_type_elements = compression_type_elements | Pack::Type::Mask_Z;
714 }
715 if((compress & CompressionModes::indices_mask) == CompressionModes::indices_zlib)
716 {
717 compression_type_index = compression_type_index | Pack::Type::Mask_Z;
718 }
719
720 if((compress & CompressionModes::elements_mask) != CompressionModes::elements_off)
721 {
722 global_i = Index((global_i * sizeof(DT2_) + sizeof(char) - 1u) / sizeof(char));
723 for(Index i(0); i < Index(uiarray[4]); ++i)
724 {
725 tc._elements.push_back(MemoryPool::template allocate_memory<DT2_>(tc._elements_size.at(i)));
726 if(0u == Pack::decode(tc._elements.at(i), &array[global_i], (std::size_t) tc._elements_size.at(i), (std::size_t) elements_bytes[i], compression_type_elements, false))
727 XABORTM("Cannot decode compressed elements array");
728 global_i += (Index)elements_bytes[i];
729 }
730 global_i = Index((global_i * sizeof(char) + sizeof(DT2_) - 1u) / sizeof(DT2_));
731 }
732 else
733 {
734 for (Index i(0) ; i < Index(uiarray[4]) ; ++i)
735 {
736 tc._elements.push_back(MemoryPool::template allocate_memory<DT2_>(tc._elements_size.at(i)));
737 MemoryPool::template copy<DT2_>(tc._elements.at(i), &dtarray[global_i], tc._elements_size.at(i));
738 global_i += tc._elements_size.at(i);
739 }
740 }
741
742 if((compress & CompressionModes::indices_mask) != CompressionModes::indices_off)
743 {
744 global_i = Index((global_i * sizeof(DT2_) + sizeof(char) - 1u) / sizeof(char));
745 for (Index i(0) ; i < Index(uiarray[5]) ; ++i)
746 {
747 tc._indices.push_back(MemoryPool::template allocate_memory<IT2_>(tc._indices_size.at(i)));
748 if(0u == Pack::decode(tc._indices.at(i), &array[global_i], (std::size_t) tc._indices_size.at(i), (std::size_t) indices_bytes[i], compression_type_index, false))
749 XABORTM("Cannot decode compressed indice array");
750 global_i += (Index)indices_bytes[i];
751 }
752 }
753 else
754 {
755 global_i = Index((global_i * sizeof(DT2_) + sizeof(IT2_) - 1u) / sizeof(IT2_));
756 for (Index i(0) ; i < Index(uiarray[5]) ; ++i)
757 {
758 tc._indices.push_back(MemoryPool::template allocate_memory<IT2_>(tc._indices_size.at(i)));
759 MemoryPool::template copy<IT2_>(tc._indices.at(i), &itarray[global_i], tc._indices_size.at(i));
760 global_i += tc._indices_size.at(i);
761 }
762 }
763
764 this->assign(tc);
765 }

901 void clone(const Container & other, CloneMode clone_mode = CloneMode::Weak)
902 {
903 if (this == &other)
904 {
905 XABORTM("Trying to self-clone a lafem container!");
906 }
907
908 XASSERTM(other._foreign_memory == false || clone_mode == CloneMode::Deep, "Must use deep cloning with ranged based source containers");
909
910 this->clear();
911
912 this->_scalar_index.assign(other._scalar_index.begin(), other._scalar_index.end());
913 this->_scalar_dt.assign(other._scalar_dt.begin(), other._scalar_dt.end());
914 this->_elements_size.assign(other._elements_size.begin(), other._elements_size.end());
915 this->_indices_size.assign(other._indices_size.begin(), other._indices_size.end());
916
917
918 if (clone_mode == CloneMode::Deep || clone_mode == CloneMode::Allocate)
919 {
920 for (Index i(0) ; i < other._indices.size() ; ++i)
921 {
922 this->_indices.push_back(MemoryPool::template allocate_memory<IT_>(this->_indices_size.at(i)));
923 if (clone_mode == CloneMode::Deep)
924 MemoryPool::template copy<IT_>(this->_indices.at(i), other._indices.at(i), this->_indices_size.at(i));
925 }
926
927 for (Index i(0) ; i < other._elements.size() ; ++i)
928 {
929 this->_elements.push_back(MemoryPool::template allocate_memory<DT_>(this->_elements_size.at(i)));
930 if (clone_mode == CloneMode::Deep)
931 MemoryPool::template copy<DT_>(this->_elements.at(i), other._elements.at(i), this->_elements_size.at(i));
932 }
933
934 return;
935 }
936 else
937 {
938 this->_indices.assign(other._indices.begin(), other._indices.end());
939 for (Index i(0) ; i < this->_indices.size() ; ++i)
940 MemoryPool::increase_memory(this->_indices.at(i));
941 }
942
943 if(clone_mode == CloneMode::Shallow)
944 {
945 this->_elements.assign(other._elements.begin(), other._elements.end());
946 for (Index i(0) ; i < this->_elements.size() ; ++i)
947 MemoryPool::increase_memory(this->_elements.at(i));
948
949 return;
950 }
951 else if(clone_mode == CloneMode::Layout)
952 {
953 for (Index i(0) ; i < other._elements.size() ; ++i)
954 {
955 this->_elements.push_back(MemoryPool::template allocate_memory<DT_>(this->_elements_size.at(i)));
956 }
957
958 return;
959 }
960 else if(clone_mode == CloneMode::Weak)
961 {
962 for (Index i(0) ; i < other._elements.size() ; ++i)
963 {
964 this->_elements.push_back(MemoryPool::template allocate_memory<DT_>(this->_elements_size.at(i)));
965 MemoryPool::template copy<DT_>(this->_elements.at(i), other._elements.at(i), this->_elements_size.at(i));
966 }
967
968 return;
969 }
