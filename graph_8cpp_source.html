95 // "Permutation" copy CTOR
96 Graph::Graph(const Graph& other, const Permutation& domain_perm, const Permutation& image_perm) :
97 _num_nodes_image(other.get_num_nodes_image()),
98 _domain_ptr(other.get_num_nodes_domain()+1),
99 _image_idx(other.get_num_indices())
100 {
101 // get domain permutation
102 const Index* domain_perm_pos = domain_perm.get_perm_pos();
103
104 // calculate new domain vector
105 _domain_ptr[0] = other._domain_ptr[0];
106 for(Index i(0); i < other.get_num_nodes_domain(); ++i)
107 {
108 _domain_ptr[i+1] = other._domain_ptr[domain_perm_pos[i]+1]
109 - other._domain_ptr[domain_perm_pos[i]]
110 + _domain_ptr[i];
111 }
112
113 // get image permutation
114 const Index* image_perm_pos = image_perm.get_perm_pos();
115
116 // calculate new image vector
117 Index count = 0;
118 for(Index i(0); i < other.get_num_nodes_domain(); ++i)
119 {
120 Index _offset = other._domain_ptr[domain_perm_pos[i]];
121 Index _row_end = other._domain_ptr[domain_perm_pos[i]+1];
122 for(Index j(_offset); j < _row_end; ++j)
123 {
124 _image_idx[count] = image_perm_pos[other._image_idx[j]];
125 ++count;
126 }
127 }

130 Graph::Graph(const std::vector<char>& buffer) :
131 _num_nodes_image(0),
132 _domain_ptr(),
133 _image_idx()
134 {
135
136 // 40 bytes required by header only
137 XASSERTM(buffer.size() >= std::size_t(40u), "invalid buffer size");
138
139 typedef std::uint64_t u64;
140 const u64* v = reinterpret_cast<const u64*>(buffer.data());
141
142 // check magic
143 XASSERTM(v[0] == Graph::magic, "invalid magic number");
144
145 // check buffer size
146 XASSERTM(v[1] == u64(buffer.size()), "invalid buffer size");
147
148 Index domain_ptr_size = Index(v[2]);
149 this->_num_nodes_image = Index(v[3]);
150 Index num_indices_image = Index(v[4]);
151
152 const u64* x = &v[5];
153
154 // allocate and copy domain pointer vector
155 if(domain_ptr_size > Index(0))
156 {
157 this->_domain_ptr = IndexVector(domain_ptr_size+1u);
158 for(Index i(0); i <= this->get_num_nodes_domain(); ++i)
159 this->_domain_ptr[i] = Index(x[i]);
160 x += std::streamoff(this->get_num_nodes_domain() +1u);
161 }
162
163 // allocate and copy image index vector
164 if(num_indices_image > Index(0))
165 {
166 this->_image_idx = IndexVector(num_indices_image);
167 for(Index i(0); i < num_indices_image; ++i)
168 this->_image_idx[i] = Index(x[i]);
169 }
170 }

193 void Graph::sort_indices()
194 {
195 XASSERTM(!_domain_ptr.empty(), "domain pointer vector is missing");
196 XASSERTM(!_image_idx.empty(), "image index vector is missing");
197
198 // loop over all domain nodes
199 for(Index i(0); i+1 < _domain_ptr.size(); ++i)
200 {
201 // let the STL do the sorting work
202 std::sort(_image_idx.begin() + IndexVector::difference_type(_domain_ptr[i]),
203 _image_idx.begin() + IndexVector::difference_type(_domain_ptr[i+1]));
204 }

210 XASSERT(Index(_image_idx.size()) == inv_perm.size());

217 std::vector<char> Graph::serialize() const
218 {
219 // compute buffer size
220 typedef std::uint64_t u64;
221 u64 s = 5u + u64(_domain_ptr.size()) + u64(_image_idx.size());
222 // allocate buffer
223 std::vector<char> buffer(s * 8u);
224 u64* v = reinterpret_cast<u64*>(buffer.data());
225
226 // create header
227 v[0] = Graph::magic;
228 v[1] = s * u64(8); // buffer size
229 v[2] = u64(this->_domain_ptr.size()-1);
230 v[3] = u64(this->_num_nodes_image);
231 v[4] = u64(this->_image_idx.size());
232
233
234 // empty graph?
235 if(this->_domain_ptr.size() <= Index(1))
236 return buffer;
237
238 XASSERT(!(this->_domain_ptr.empty()));
239
240 // copy domain pointer
241 u64* x = &v[5];
242 for(Index i(0); i < this->_domain_ptr.size(); ++i)
243 x[i] = u64(this->_domain_ptr[i]);
244
245 // copy image indices
246 x += std::streamoff(this->_domain_ptr.size());
247 for(Index i(0); i < this->_image_idx.size(); ++i)
248 x[i] = u64(this->_image_idx[i]);
249
250 // okay
251 return buffer;
252 }
