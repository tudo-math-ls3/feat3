147 virtual Status correct(VectorType& vec_sol, const VectorType& vec_rhs) override
148 {
149 // compute defect
150 this->_system_matrix.apply(this->_vec_r, vec_sol, vec_rhs, -DataType(1));
151 this->_system_filter.filter_def(this->_vec_r);
152
153 // apply solver
154 this->_status = _apply_intern(vec_sol);
155
156 // plot summary
157 this->plot_summary();
158

160 return this->_status;

163 protected:
164 virtual Status _apply_intern(VectorType& vec_sol)
165 {
166 Statistics::add_solver_expression(std::make_shared<ExpressionStartSolve>(this->name()));
167
168 const MatrixType& matrix(this->_system_matrix);
169 const FilterType& filter(this->_system_filter);
170 VectorType& vec_r(this->_vec_r);
171 VectorType& vec_q(this->_vec_q);
172 VectorType& vec_z(this->_vec_z);
173
174 // set initial defect:
175 // r[0] := b - A*x[0]
176 Status status = this->_set_initial_defect(vec_r, vec_sol);
177 if(status != Status::progress)
178 {
179 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), status, this->get_num_iter()));
180 return status;
181 }
182
183 // start iterating
184 while(status == Status::progress)
185 {
186 IterationStats stat(*this);
187
188 // apply preconditioner to defect vector
189 // z[k] := M^{-1} * r[k]
190 if(!this->_apply_precond(vec_z, vec_r, filter))
191 {
192 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
193 return Status::aborted;
194 }
195
196 // q[k] := A*z[k]
197 matrix.apply(vec_q, vec_z);
198 filter.filter_def(vec_q);
199
200 // alpha[k] := < z[k], r[k] > / < z[k], q[k] >
201 DataType alpha = vec_z.dot(vec_r) / vec_z.dot(vec_q);
202
203 // update solution vector:
204 // x[k+1] := x[k] + alpha[k] * z[k]
205 vec_sol.axpy(vec_z, vec_sol, alpha);
206
207 // update defect vector:
208 // r[k+1] := r[k] - alpha[k] * q[k]
209 vec_r.axpy(vec_q, vec_r, -alpha);
210
211 // compute defect norm
212 status = this->_set_new_defect(vec_r, vec_sol);
213 if(status != Status::progress)
214 {
215 stat.destroy();
216 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), status, this->get_num_iter()));
217 return status;
218 }
219 }
220
221 // we should never reach this point...
222 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::undefined, this->get_num_iter()));
223 return Status::undefined;
224 }
