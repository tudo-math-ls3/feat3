117 template<typename LocalMatrix_, typename RowMapping_, typename ColMapping_>
118 void operator()(const LocalMatrix_& loc_mat, const RowMapping_& row_map,
119 const ColMapping_& col_map, DT_ alpha = DT_(1))
120 {
121 // loop over all local row entries
122 for(int i(0); i < row_map.get_num_local_dofs(); ++i)
123 {
124 // fetch row index
125 const Index ix = row_map.get_index(i);
126
127 // build column pointer for this row entry contribution
128 for(IT_ k(_row_ptr[ix]); k < _row_ptr[ix + 1]; ++k)
129 {
130 _col_ptr[_col_idx[k]] = k;
131 }
132
133 // loop over all local column entries
134 for(int j(0); j < col_map.get_num_local_dofs(); ++j)
135 {
136 // fetch column index
137 const Index jx = col_map.get_index(j);
138
139 // ensure that the column pointer is valid for this index
140 ASSERTM(_col_ptr[jx] != _deadcode, "invalid column index");
141
142 // incorporate data into global matrix
143 _data[_col_ptr[jx]] += alpha * loc_mat[i][j];
144
145 // continue with next column entry
146 }
147
148#ifdef DEBUG
149 // reformat column-pointer array
150 for(IT_ k(_row_ptr[ix]); k < _row_ptr[ix + 1]; ++k)
151 {
152 _col_ptr[_col_idx[k]] = _deadcode;
153 }
154#endif
155 // continue with next row entry
156 }
157 }

408 explicit SparseMatrixCSR(const Adjacency::Graph & graph) :
409 Container<DT_, IT_>(0)
410 {
411 Index num_rows = graph.get_num_nodes_domain();
412 Index num_cols = graph.get_num_nodes_image();
413 Index num_nnze = graph.get_num_indices();
414
415 if (num_nnze == 0)
416 {
417 this->assign(SparseMatrixCSR(num_rows, num_cols));
418 return;
419 }
420
421 // create temporary vectors
422 LAFEM::DenseVector<IT_, IT_> vrow_ptr(num_rows+1);
423 LAFEM::DenseVector<IT_, IT_> vcol_idx(num_nnze);
424 LAFEM::DenseVector<DT_, IT_> vdata(num_nnze, DT_(0));
425
426 const Index * dom_ptr(graph.get_domain_ptr());
427 const Index * img_idx(graph.get_image_idx());
428 IT_ * prow_ptr(vrow_ptr.elements());
429 IT_ * pcol_idx(vcol_idx.elements());
430
431 // build row-end
432 prow_ptr[0] = IT_(dom_ptr[0]);
433 for(Index i(0); i < num_rows; ++i)
434 prow_ptr[i+1] = IT_(dom_ptr[i+1]);
435
436 // build col-idx
437 for(Index i(0); i < num_nnze; ++i)
438 pcol_idx[i] = IT_(img_idx[i]);
439
440 // build the matrix
441 this->assign(SparseMatrixCSR<DT_, IT_>(num_rows, num_cols, vcol_idx, vdata, vrow_ptr));
442 }

599 template <typename DT2_, typename IT2_>
600 void convert(const SparseMatrixBanded<DT2_, IT2_> & other)
601 {
602 this->clear();
603
604 this->_scalar_index.push_back(other.size());
605 this->_scalar_index.push_back(other.rows());
606 this->_scalar_index.push_back(other.columns());
607 this->_scalar_index.push_back(other.used_elements());
608
609 if (other.used_elements() == 0)
610 return;
611
612 this->_elements.push_back(MemoryPool::template allocate_memory<DT_>(_used_elements()));
613 this->_elements_size.push_back(_used_elements());
614 this->_indices.push_back(MemoryPool::template allocate_memory<IT_>(_used_elements()));
615 this->_indices_size.push_back(_used_elements());
616 this->_indices.push_back(MemoryPool::template allocate_memory<IT_>(_rows() + 1));
617 this->_indices_size.push_back(_rows() + 1);
618
619 DT_ * tval(nullptr);
620 IT_ * tcol_ind(nullptr);
621 IT_ * trow_ptr(nullptr);
622 tval = this->_elements.at(0);
623 tcol_ind = this->_indices.at(0);
624 trow_ptr = this->_indices.at(1);
625
626 trow_ptr[0] = 0;
627
628 const DT_ * cval(other.val());
629 const IT_ * coffsets(other.offsets());
630 const Index cnum_of_offsets(other.num_of_offsets());
631 const Index crows(other.rows());
632
633 // Search first offset of the upper triangular matrix
634 Index k(0);
635 while (k < cnum_of_offsets && coffsets[k] + 1 < crows)
636 {
637 ++k;
638 }
639
640 IT_ ue(0);
641 // iteration over all offsets of the lower triangular matrix
642 for (Index i(k + 1); i > 0;)
643 {
644 --i;
645
646 // iteration over all offsets of the upper triangular matrix
647 for (Index j(cnum_of_offsets + 1); j > 0;)
648 {
649 --j;
650
651 // iteration over all rows which contain the offsets between offset i and offset j
652 const Index start(Math::max(other.start_offset(i),
653 other.end_offset(j) + 1));
654 const Index end (Math::min(other.start_offset(i-1),
655 other.end_offset(j-1) + 1));
656 for (Index l(start); l < end; ++l)
657 {
658 for (Index a(i); a < j; ++a)
659 {
660 tval[ue] = cval[a * crows + l];
661 tcol_ind[ue] = IT_(l + coffsets[a] + 1 - crows);
662 ++ue;
663 }
664 trow_ptr[l + 1] = ue;
665 }
666 }
667 }

677 template <typename DT2_, typename IT2_, int BlockHeight_, int BlockWidth_>
678 void convert(const SparseMatrixBCSR<DT2_, IT2_, BlockHeight_, BlockWidth_> & other)
679 {
680 this->clear();
681
682 this->_scalar_index.push_back(other.template rows<Perspective::pod>() * other.template columns<Perspective::pod>());
683 this->_scalar_index.push_back(other.template rows<Perspective::pod>());
684 this->_scalar_index.push_back(other.template columns<Perspective::pod>());
685 this->_scalar_index.push_back(other.template used_elements<Perspective::pod>());
686
687 if (other.template used_elements<Perspective::pod>() == 0)
688 return;
689
690 this->_elements.push_back(MemoryPool::template allocate_memory<DT_>(_used_elements()));
691 this->_elements_size.push_back(_used_elements());
692 this->_indices.push_back(MemoryPool::template allocate_memory<IT_>(_used_elements()));
693 this->_indices_size.push_back(_used_elements());
694 this->_indices.push_back(MemoryPool::template allocate_memory<IT_>(_rows() + 1));
695 this->_indices_size.push_back(_rows() + 1);
696
697 DT_ * tval(nullptr);
698 IT_ * tcol_ind(nullptr);
699 IT_ * trow_ptr(nullptr);
700 tval = this->_elements.at(0);
701 tcol_ind = this->_indices.at(0);
702 trow_ptr = this->_indices.at(1);
703
704 Index ait(0);
705 trow_ptr[0] = IT_(0);
706 for (Index orow(0) ; orow < other.rows() ; ++orow)
707 {
708 for (int row(0) ; row < BlockHeight_ ; ++row)
709 {
710 for(Index ocol(other.row_ptr()[orow]) ; ocol < other.row_ptr()[orow + 1] ; ++ocol)
711 {
712 Tiny::Matrix<DT_, BlockHeight_, BlockWidth_> tbm(other.val()[ocol]);
713 for (int col(0) ; col < BlockWidth_ ; ++col)
714 {
715 tval[ait] = tbm(row,col);
716 tcol_ind[ait] = other.col_ind()[ocol] * (IT_)BlockWidth_ + (IT_)col;
717 ++ait;
718 }
719 }
720 trow_ptr[orow * Index(BlockHeight_) + Index(row) + 1] = (IT_)ait;
721 }
722 }

735 XASSERT(a.template used_elements<Perspective::pod>() > 0);
736
737 typename MT_::template ContainerType<DT_, IT_> ta;
738 ta.convert(a);
739
740 const Index arows(ta.template rows<Perspective::pod>());
741 const Index acolumns(ta.template columns<Perspective::pod>());
742 const Index aused_elements(ta.template used_elements<Perspective::pod>());
743
744 DenseVector<DT_, IT_> tval(aused_elements);
745 DenseVector<IT_, IT_> tcol_ind(aused_elements);
746 DenseVector<IT_, IT_> trow_ptr(arows + 1);
747
748 DT_ * pval(tval.elements());

750 IT_ * prow_ptr(trow_ptr.elements());
751
752 for (Index i(0); i < arows; ++i)

754 prow_ptr[i + 1] = IT_(a.get_length_of_line(i));

758
759 for (Index i(1); i < arows + 1; ++i)

761 prow_ptr[i] += prow_ptr[i - 1];
762 }
763
764 for (Index i(0); i < arows; ++i)
765 {
766 ta.set_line(i, pval + prow_ptr[i], pcol_ind + prow_ptr[i], 0);
767 }
768
769 SparseMatrixCSR<DT_, IT_> a_csr(arows, acolumns, tcol_ind, tval, trow_ptr);
770
771 this->assign(a_csr);

898 void read_from(FileMode mode, std::istream& file)
899 {
900 switch(mode)
901 {
902 case FileMode::fm_mtx:
903 {
904 this->clear();
905 this->_scalar_index.push_back(0);
906 this->_scalar_index.push_back(0);
907 this->_scalar_index.push_back(0);
908 this->_scalar_index.push_back(0);
909
910 std::map<IT_, std::map<IT_, DT_> > entries; // map<row, map<column, value> >
911
912 Index ue(0);
913 String line;
914 std::getline(file, line);
915 const bool general((line.find("%%MatrixMarket matrix coordinate real general") != String::npos) ? true : false);
916 const bool symmetric((line.find("%%MatrixMarket matrix coordinate real symmetric") != String::npos) ? true : false);
917
918 if (symmetric == false && general == false)
919 {
920 XABORTM("Input-file is not a compatible mtx-file");
921 }
922
923 while(!file.eof())
924 {
925 std::getline(file,line);
926 if (file.eof())
927 XABORTM("Input-file is empty");
928
929 String::size_type begin(line.find_first_not_of(" "));
930 if (line.at(begin) != '%')
931 break;
932 }
933 {

2345 {

2347 return;
2348
2349 XASSERTM(perm_row.size() == this->rows(), "Container rows does not match permutation size");
2350 XASSERTM(perm_col.size() == this->columns(), "Container columns does not match permutation size");

2352 // http://de.mathworks.com/help/matlab/math/sparse-matrix-operations.html#f6-13070
2353 IT_ * temp_row_ptr = new IT_[rows() + 1];
2354 IT_ * temp_col_ind = new IT_[used_elements()];

2358 perm_pos = perm_row.get_perm_pos();
2359
2360 //permute rows from source to temp_*
2361 Index new_start(0);
2362 temp_row_ptr[0] = 0;
2363 for (Index row(0) ; row < this->rows() ; ++row)
2364 {
2365 Index row_size(this->row_ptr()[perm_pos[row] + 1] - this->row_ptr()[perm_pos[row]]);

2368 for (Index i(new_start), j(this->row_ptr()[perm_pos[row]]) ; i < new_start + row_size ; ++i, ++j)
2369 {
2370 temp_col_ind[i] = this->col_ind()[j];
2371 temp_val[i] = this->val()[j];
2372 }
2373
2374 new_start += row_size;
2375 temp_row_ptr[row+1] = (IT_)new_start;
