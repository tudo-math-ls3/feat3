253 const Partition* find_partition(int size, const std::deque<String>& names, int prio = 0) const
254 {
255 const Partition* part(nullptr);
256
257 // loop over all partitions
258 for(auto it = _parts.begin(); it != _parts.end(); ++it)
259 {
260 // check size
261 if(int(it->size()) != size)
262 continue;
263
264 // check name
265 if(!names.empty())
266 {
267 bool match = false;
268 for(const auto& name : names)
269 {
270 match = match || (name == it->get_name());
271 }
272 if(!match)
273 continue;
274 }
275
276 // check priority
277 if((it->get_priority() <= 0) || (it->get_priority() < prio))
278 continue;
279
280 // okay, that's a candidate
281 // does it have a greater or equal priority to the previously found one?
282 if((part != nullptr) && (part->get_priority() > it->get_priority()))
283 continue;
284
285 // we have found a matching partition
286 part = &(*it);
287 }
288
289 return part;
290 }
