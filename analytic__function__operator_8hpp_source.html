269 void compute_approx_hess(HessianType& mat_out)

273 for(int i(1); i < mat_out.m; ++i)
274 {
275 for(int j(0); j < i; ++j)
276 {
277 mat_out(i,j) = DT_(0);
278 mat_out(j,i) = DT_(0);
279 }
280 }
