22 // p0, p1 and p2 are the 1D basis functions on the reference interval [-1,+1].
23 // These are used for the tensor-product approach in the Hypercube evaluators.
24
25 // basis function for left vertex
26 template<typename T_>
27 inline T_ p0(T_ x)
28 {
29 return T_(0.5) * x * (x - T_(1));
30 }
31
32 // basis function for right vertex
33 template<typename T_>
34 inline T_ p1(T_ x)
35 {
36 return T_(0.5) * x * (x + T_(1));
37 }
38
39 // basis function for edge midpoint
40 template<typename T_>
41 inline T_ p2(T_ x)
42 {
43 return (T_(1) - x) * (T_(1) + x);
44 }
45
46 // first order derivatives
47
48 template<typename T_>
49 inline T_ d1p0(T_ x)
50 {
51 return x - T_(0.5);
52 }
53
54 template<typename T_>
55 inline T_ d1p1(T_ x)
56 {
57 return x + T_(0.5);
58 }
59
60 template<typename T_>
61 inline T_ d1p2(T_ x)
62 {
63 return -T_(2) * x;
64 }
65
66 // second order derivatives
67
68 template<typename T_>
69 inline T_ d2p0(T_)
70 {
71 return T_(1);
72 }
73
74 template<typename T_>
75 inline T_ d2p1(T_)
76 {
77 return T_(1);
78 }
79
80 template<typename T_>
81 inline T_ d2p2(T_)
82 {
83 return -T_(2);
84 }
85 } // namespace Intern

276 void prepare(Index cell_index)
277 {
278 // prepare base-class
279 BaseClass::prepare(cell_index);
280
281 // fetch the mesh from the trafo
282 const MeshType& mesh = this->_trafo.get_mesh();
283
284 // fetch the vertex set from the mesh
285 typedef typename MeshType::VertexSetType VertexSetType;
286 const VertexSetType& vertex_set = mesh.get_vertex_set();
287
288 // fetch the index set
289 typedef typename MeshType::template IndexSet<domain_dim, 0>::Type IndexSetType;
290 const IndexSetType& index_set = mesh.template get_index_set<domain_dim, 0>();
291

294 _coeffs[1] = vertex_set[index_set(cell_index, 1)];
295
296 // get edge midpoint
297 _coeffs[2] = _coeffs[0] + _coeffs[1];
298
299 // normalize all coefficients
300 for(int i(0); i < 3; ++i)
301 _coeffs[i].normalize();
302 }
303
313 void map_point(ImagePointType& img_point, const DomainPointType& dom_point) const
314 {
315 const DataType x = dom_point[0];
316
317 for(int i(0); i < image_dim; ++i)
318 {
319 img_point[i] =
320 _coeffs[0][i] * DataType(0.5) * x * (x - DataType(1)) +
321 _coeffs[1][i] * DataType(0.5) * x * (x + DataType(1)) +
322 _coeffs[2][i] * (DataType(1) - x) * (DataType(1) - x);
323 }
324 }
325
335 void calc_jac_mat(JacobianMatrixType& jac_mat, const DomainPointType& dom_point) const
336 {
337 const DataType x = dom_point[0];
338
339 for(int i(0); i < image_dim; ++i)
340 {
341 jac_mat[i][0] =
342 _coeffs[0][i] * (x - DataType(0.5)) +
343 _coeffs[1][i] * (x + DataType(0.5)) -
344 _coeffs[2][i] * DataType(2) * x;
345 }
346 }
347
357 void calc_hess_ten(HessianTensorType& hess_ten, const DomainPointType& DOXY(dom_point)) const
358 {
359 for(int i(0); i < image_dim; ++i)
360 {
361 hess_ten[i][0] = _coeffs[0][i] + _coeffs[1][i] - DataType(2) *_coeffs[2][i];
362 }
363 }
364 }; // class Evaluator<Simplex<1>,...>

407 typedef typename EvalPolicy::HessianInverseType HessianInverseType;
408
410 static constexpr int domain_dim = EvalPolicy::domain_dim;

413
414 static_assert(domain_dim < image_dim, "this trafo is for surfaces only");
415
418 static constexpr TrafoTags eval_caps =
419 TrafoTags::dom_point | TrafoTags::img_point |
420 TrafoTags::jac_mat | TrafoTags::jac_det | TrafoTags::hess_ten;
421
422 protected:
424 Tiny::Matrix<DataType, 6, image_dim> _coeffs;
425
426 public:
433 explicit Evaluator(const TrafoType& trafo) :
434 BaseClass(trafo)
435 {
436 }
437
444 void prepare(Index cell_index)
445 {
446 // prepare base-class
447 BaseClass::prepare(cell_index);
448
449 // fetch the mesh from the trafo
450 const MeshType& mesh = this->_trafo.get_mesh();
451
452 // fetch the vertex set from the mesh
453 typedef typename MeshType::VertexSetType VertexSetType;
454 const VertexSetType& vertex_set = mesh.get_vertex_set();
455
456 // fetch the index set
457 typedef typename MeshType::template IndexSet<domain_dim, 0>::Type IndexSetType;
458 const IndexSetType& index_set = mesh.template get_index_set<domain_dim, 0>();
459
460 // get vertex coords
461 _coeffs[0] = vertex_set[index_set(cell_index, 0)];
462 _coeffs[1] = vertex_set[index_set(cell_index, 1)];
463 _coeffs[2] = vertex_set[index_set(cell_index, 2)];
464
465 // get edge midpoints
466 _coeffs[3] = _coeffs[1] + _coeffs[2];
467 _coeffs[4] = _coeffs[0] + _coeffs[2];
468 _coeffs[5] = _coeffs[0] + _coeffs[1];
469
470 // normalize all coefficients
471 for(int i(0); i < 6; ++i)
472 _coeffs[i].normalize();
473 }
474
484 void map_point(ImagePointType& img_point, const DomainPointType& dom_point) const
485 {
486 const DataType x = dom_point[0];
487 const DataType y = dom_point[1];
488
489 for(int i(0); i < image_dim; ++i)
490 {
491 img_point[i] =
492 _coeffs[0][i] * (DataType(2) * (x + y - DataType(0.5)) * (x + y - DataType(1))) +
493 _coeffs[1][i] * (DataType(2) * x * (x - DataType(0.5))) +
494 _coeffs[2][i] * (DataType(2) * y * (y - DataType(0.5))) +
495 _coeffs[3][i] * (DataType(4) * x * y) +
496 _coeffs[4][i] * (DataType(4) * y * (DataType(1) - x - y)) +
497 _coeffs[5][i] * (DataType(4) * x * (DataType(1) - x - y));
498 }
499 }
500
510 void calc_jac_mat(JacobianMatrixType& jac_mat, const DomainPointType& dom_point) const
511 {
512 const DataType x = dom_point[0];
513 const DataType y = dom_point[1];
514
515 for(int i(0); i < image_dim; ++i)
516 {
517 jac_mat[i][0] =
518 _coeffs[0][i] * (DataType(4) * (x + y) - DataType(3)) +
519 _coeffs[1][i] * (DataType(4) * x - DataType(1)) +
520 //_coeffs[2][i] * (DataType(0)) +
521 _coeffs[3][i] * (DataType(4) * y) -
522 _coeffs[4][i] * (DataType(4) * y) -
523 _coeffs[5][i] * (DataType(4) * (DataType(2)*x + y - DataType(1)));
524 jac_mat[i][1] =
525 _coeffs[0][i] * (DataType(4) * (x + y) - DataType(3)) +
526 //_coeffs[1][i] * (DataType(0)) +
527 _coeffs[2][i] * (DataType(4) * y - DataType(1)) +
528 _coeffs[3][i] * (DataType(4) * x) -
529 _coeffs[4][i] * (DataType(4) * (DataType(2)*y + x - DataType(1))) -
530 _coeffs[5][i] * (DataType(4) * x);
531 }
532 }
533
543 void calc_hess_ten(HessianTensorType& hess_ten, const DomainPointType& DOXY(dom_point)) const
544 {
545 for(int i(0); i < image_dim; ++i)
546 {
547 hess_ten[i][0][0] = DataType(4) * (_coeffs[0][i] + _coeffs[1][i] - DataType(2) * _coeffs[5][i]);
548 hess_ten[i][1][1] = DataType(4) * (_coeffs[0][i] + _coeffs[2][i] - DataType(2) * _coeffs[4][i]);
549 hess_ten[i][0][1] =
550 hess_ten[i][1][0] = DataType(4) * (_coeffs[0][i] + _coeffs[3][i] + _coeffs[4][i] + _coeffs[5][i]);
551 }
552 }
553 }; // class Evaluator<Simplex<2>,...>

872 void calc_jac_mat(JacobianMatrixType& jac_mat, const DomainPointType& dom_point) const
873 {
874 for(int i(0); i < image_dim; ++i)
875 {
876 jac_mat[i][0] =
877 _coeffs[0][i] * Intern::d1p0(dom_point[0]) * Intern::p0(dom_point[1]) +
878 _coeffs[1][i] * Intern::d1p1(dom_point[0]) * Intern::p0(dom_point[1]) +
879 _coeffs[2][i] * Intern::d1p0(dom_point[0]) * Intern::p1(dom_point[1]) +
880 _coeffs[3][i] * Intern::d1p1(dom_point[0]) * Intern::p1(dom_point[1]) +
881 _coeffs[4][i] * Intern::d1p2(dom_point[0]) * Intern::p0(dom_point[1]) +
882 _coeffs[5][i] * Intern::d1p2(dom_point[0]) * Intern::p1(dom_point[1]) +
883 _coeffs[6][i] * Intern::d1p0(dom_point[0]) * Intern::p2(dom_point[1]) +
884 _coeffs[7][i] * Intern::d1p1(dom_point[0]) * Intern::p2(dom_point[1]) +
885 _coeffs[8][i] * Intern::d1p2(dom_point[0]) * Intern::p2(dom_point[1]);
886 jac_mat[i][1] =
887 _coeffs[0][i] * Intern::p0(dom_point[0]) * Intern::d1p0(dom_point[1]) +
888 _coeffs[1][i] * Intern::p1(dom_point[0]) * Intern::d1p0(dom_point[1]) +
889 _coeffs[2][i] * Intern::p0(dom_point[0]) * Intern::d1p1(dom_point[1]) +
890 _coeffs[3][i] * Intern::p1(dom_point[0]) * Intern::d1p1(dom_point[1]) +
