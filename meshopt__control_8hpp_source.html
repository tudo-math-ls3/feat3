276 void compute_mesh_quality(CoordType& edge_angle, CoordType& qi_min, CoordType& qi_mean,
277 CoordType* edge_angle_cellwise = nullptr, CoordType* qi_cellwise = nullptr,
278 int lvl_index = -1) const
279 {
280 // max_level_index cannot be called for the default argument, so we do it here
281 if(lvl_index == -1)
282 {
283 lvl_index = _dom_ctrl.max_level_index();
284 }
285 XASSERT(lvl_index >= _dom_ctrl.min_level_index());
286 XASSERT(lvl_index <= _dom_ctrl.max_level_index());
287
288 const Dist::Comm& comm = _dom_ctrl.comm();
289
290 CoordType qi_sum(0);
291
292 for(std::size_t i(0u); i < _dom_ctrl.size_physical(); ++i)
293 {
294 const auto& dom_lvl = *_dom_ctrl.at(i);
295 if(dom_lvl.get_level_index() == lvl_index)
296 {
297 const auto& my_mesh = dom_lvl.get_mesh();
298
299 Index ncells(my_mesh.get_num_entities(MeshType::shape_dim));
300 comm.allreduce(&ncells, &ncells, std::size_t(1), Dist::op_sum);
301
302 Geometry::MeshQualityHeuristic<typename MeshType::ShapeType>::compute(qi_min, qi_sum,
303 my_mesh.template get_index_set<MeshType::shape_dim, 0>(), my_mesh.get_vertex_set(), qi_cellwise);
304
305 comm.allreduce(&qi_min, &qi_min, std::size_t(1), Dist::op_min);
306 comm.allreduce(&qi_sum, &qi_sum, std::size_t(1), Dist::op_sum);
307
308 edge_angle = Geometry::MeshQualityHeuristic<typename MeshType::ShapeType>::angle(
309 my_mesh.template get_index_set<MeshType::shape_dim, 0>(), my_mesh.get_vertex_set(), edge_angle_cellwise);
310 comm.allreduce(&edge_angle, &edge_angle, std::size_t(1), Dist::op_min);
311
312 qi_mean = qi_sum/CoordType(ncells);
313
314 return;
315 }
316 }
317
318 // We should never get to this point
319 XABORTM("Could not find level with index "+stringify(lvl_index)+"!\n");
320
321 }

538 void sync_system_filter()
539 {
540 // Sync the filter vectors in the SlipFilters
541 auto& slip_filters = filter_sys.local().template at<0>();
542 {
543 for(auto& it : slip_filters)
544 {
545 // Get the filter vector
546 auto& slip_filter_vector = it.second.get_filter_vector();
547
548 if(slip_filter_vector.used_elements() > 0)
549 {
550 // Temporary DenseVector for syncing
551 LocalSystemVectorL tmp(slip_filter_vector.size());
552 auto* tmp_elements = tmp.template elements<LAFEM::Perspective::native>();
553 auto* sfv_elements = slip_filter_vector.template elements<LAFEM::Perspective::native>();
554
555 // Copy sparse filter vector contents to DenseVector
556 for(Index isparse(0); isparse < slip_filter_vector.used_elements(); ++isparse)
557 {
558 Index idense(slip_filter_vector.indices()[isparse]);
559 tmp_elements[idense] = sfv_elements[isparse];
560 }
561
562 gate_sys.sync_0(tmp);
563
564 // Copy sparse filter vector contents to DenseVector
565 for(Index isparse(0); isparse < slip_filter_vector.used_elements(); ++isparse)
566 {
567 Index idense(slip_filter_vector.indices()[isparse]);
568 sfv_elements[isparse] = tmp_elements[idense];
569 }
570 }
571 else
572 {
573 // Temporary DenseVector for syncing
574 LocalSystemVectorL tmp(slip_filter_vector.size());
575 gate_sys.sync_0(tmp);
576 }
577 }
578
579 } // evil slip filter sync

629 template<typename DomainLayer_>
630 void assemble_gates(const DomainLayer_& dom_layer)
631 {
632 // set the gate comm
633 gate_sys.set_comm(dom_layer.comm_ptr());
634 gate_scalar.set_comm(dom_layer.comm_ptr());
635
636 // Loop over all ranks
637 for(Index i(0); i < dom_layer.neighbor_count(); ++i)
638 {
639 int rank(dom_layer.neighbor_rank(i));
640
641 // try to find our halo
642 auto* halo = local_functional.get_mesh_node()->get_halo(rank);
643 XASSERTM(halo != nullptr, "Halo not found.");
644
645 // assemble the system mirror
646 SystemMirror sys_mirror;
647 Assembly::MirrorAssembler::assemble_mirror(sys_mirror, local_functional.trafo_space, *halo);
648
649 // push mirror into gates
650 gate_sys.push(rank, std::move(sys_mirror));
651
652 // assemble the scalar mirror
653 ScalarMirror scalar_mirror;
654 Assembly::MirrorAssembler::assemble_mirror(scalar_mirror, local_functional.trafo_space, *halo);
655
656 // push mirror into gates
657 gate_scalar.push(rank, std::move(scalar_mirror));
658 }
659
660 // create local template vectors
661 LocalSystemVectorR tmp_sys(local_functional.trafo_space.get_num_dofs());
662 LocalScalarVector tmp_scalar(local_functional.trafo_space.get_num_dofs());
663
664 // compile gates
665 gate_sys.compile(std::move(tmp_sys));
666 gate_scalar.compile(std::move(tmp_scalar));
667 }

676 void assemble_system_transfer(const MeshoptSystemLevel& level_coarse)
677 {
678 // get global transfer operator
679 GlobalSystemTransfer& glob_trans = this->transfer_sys;
680
681 // get local transfer operator
682 LocalSystemTransfer& loc_trans = glob_trans.local();
683
684 // get local transfer matrices
685 typename LocalSystemTransferMatrix::BaseClass& loc_prol = loc_trans.get_mat_prol();
686 typename LocalSystemTransferMatrix::BaseClass& loc_rest = loc_trans.get_mat_rest();
687
688 // assemble structure?
689 if (loc_prol.empty())
690 {
691 Assembly::SymbolicAssembler::assemble_matrix_2lvl(
692 loc_prol, local_functional.trafo_space, level_coarse.local_functional.trafo_space);
693 }
694
695 // Create a global scalar weight vector
696 GlobalScalarVector glob_vec_weight( &this->gate_scalar, loc_prol.create_vector_l());
697 // Get local scalar weight vector
698 auto& loc_vec_weight = glob_vec_weight.local();
699
700 // Assemble the underlying scalar prolongation matrix
701 {
702 Cubature::DynamicFactory cubature_factory("auto-degree:3");
703
704 loc_prol.format();
705 loc_vec_weight.format();
706
707 // Assemble prolongation matrix
708 Assembly::GridTransfer::assemble_prolongation(loc_prol, loc_vec_weight,
709 local_functional.trafo_space, level_coarse.local_functional.trafo_space, cubature_factory);
710
711 // Synchronize weight vector
712 glob_vec_weight.sync_0();
713
714 // Invert components
715 loc_vec_weight.component_invert(loc_vec_weight);
716
717 // Scale prolongation matrix
718 loc_prol.scale_rows(loc_prol, loc_vec_weight);
719
720 // Copy and transpose
721 loc_rest = loc_prol.transpose();
722 }
723 }
