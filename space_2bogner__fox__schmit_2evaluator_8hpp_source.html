29 // p1, p2, q1 and q2 are the 1D basis functions on the reference interval [-1,+1].
30 // These are used for the tensor-product approach in the evaluators.
31
32 template<typename T_>
33 inline T_ p1(T_ x)
34 {
35 return T_(0.5) + T_(0.25) * x * (x*x - T_(3));
36 }
37
38 template<typename T_>
39 inline T_ p2(T_ x)
40 {
41 return T_(0.5) - T_(0.25) * x * (x*x - T_(3));
42 }
43
44 template<typename T_>
45 inline T_ q1(T_ x)
46 {
47 return T_(0.25) * (x + T_(1)) * Math::sqr(x - T_(1));
48 }
49
50 template<typename T_>
51 inline T_ q2(T_ x)
52 {
53 return T_(0.25) * (x - T_(1)) * Math::sqr(x + T_(1));
54 }
55
56 // first order derivatives follow
57 template<typename T_>
58 inline T_ dp1(T_ x)
59 {
60 return +T_(0.75) * (x*x - T_(1));
61 }
62
63 template<typename T_>
64 inline T_ dp2(T_ x)
65 {
66 return -T_(0.75) * (x*x - T_(1));
67 }
68
69 template<typename T_>
70 inline T_ dq1(T_ x)
71 {
72 return T_(0.25) * (T_(3)*x + T_(1)) * (x - T_(1));
73 }
74
75 template<typename T_>
76 inline T_ dq2(T_ x)
77 {
78 return T_(0.25) * (T_(3)*x - T_(1)) * (x + T_(1));
79 }
80
81 // second order derivatives follow
82 template<typename T_>
83 inline T_ ddp1(T_ x)
84 {
85 return +T_(1.5) * x;
86 }
87
88 template<typename T_>
89 inline T_ ddp2(T_ x)
90 {
91 return -T_(1.5) * x;
92 }
93
94 template<typename T_>
95 inline T_ ddq1(T_ x)
96 {
97 return T_(0.5) * (T_(3)*x - T_(1));
98 }
99
100 template<typename T_>
101 inline T_ ddq2(T_ x)
102 {
103 return T_(0.5) * (T_(3)*x + T_(1));
104 }
105 } // namespace Intern

217 template<typename EvalData_>
218 void eval_ref_values(
219 EvalData_& data,
220 const DomainPointType& point) const
221 {
222 // P1(x) = 1/2 + 1/4 * x * (x^2 - 3)
223 data.phi[0].ref_value = Intern::p1(DataType(point[0]));
224 // Q1(x) = c/4 * (x + 1) * (x - 1)^2
225 data.phi[1].ref_value = _coeff * Intern::q1(DataType(point[0]));
226 // P2(x) = 1/2 - 1/4 * x * (x^2 - 3)
227 data.phi[2].ref_value = Intern::p2(DataType(point[0]));
228 // Q2(x) = c/4 * (x - 1) * (x + 1)^2
229 data.phi[3].ref_value = _coeff * Intern::q2(DataType(point[0]));
230 }
231
241 template<typename EvalData_>
242 void eval_ref_gradients(
243 EvalData_& data,
244 const DomainPointType& point) const
245 {
246 // dx P1(x) = +3/4 * (x^2 - 1)
247 data.phi[0].ref_grad[0] = Intern::dp1(DataType(point[0]));
248 // dx Q1(x) = c/4 * (3*x + 1) * (x - 1)
249 data.phi[1].ref_grad[0] = _coeff * Intern::dq1(DataType(point[0]));
250 // dx P2(x) = -3/4 * (x^2 - 1)
251 data.phi[2].ref_grad[0] = Intern::dp2(DataType(point[0]));
252 // dx Q2(x) = c/4 * (3*x - 1) * (x + 1)
253 data.phi[3].ref_grad[0] = _coeff * Intern::dq2(DataType(point[0]));
254 }
255
265 template<typename EvalData_>
266 void eval_ref_hessians(
267 EvalData_& data,
268 const DomainPointType& point) const
269 {
270 // dxx P1(x) = +3/2 * x
271 data.phi[0].ref_hess[0][0] = Intern::ddp1(DataType(point[0]));
272 // dxx Q1(x) = 1/2 * (3*x - 1)
273 data.phi[1].ref_hess[0][0] = _coeff * Intern::ddq1(DataType(point[0]));
274 // dxx P2(x) = -3/2 * x
275 data.phi[2].ref_hess[0][0] = Intern::ddp2(DataType(point[0]));
276 // dxx Q2(x) = 1/2 * (3*x + 1)
277 data.phi[3].ref_hess[0][0] = _coeff * Intern::ddq2(DataType(point[0]));
278 }

339 DataType _trans_dx(Index i, DataType rx, DataType ry) const
340 {
341 return _coeff_fod[i](0,0) * rx + _coeff_fod[i](0,1) * ry;
342 }

356 DataType _trans_dy(Index i, DataType rx, DataType ry) const

358 return _coeff_fod[i](1,0) * rx + _coeff_fod[i](1,1) * ry;

389 void prepare(const TrafoEvaluator& trafo_eval)
390 {
391 // domain point
392 DomainPointType dom_point(DataType(0));
393
394 // evaluate trafo in interval midpoint
395 typedef typename TrafoEvaluator::template ConfigTraits<TrafoTags::jac_mat>::EvalDataType CoeffEvalData;
396 CoeffEvalData coeff_data;
397
398 // loop over all four vertices of the quad
399 for(int i(0); i < 4; ++i)
400 {
401 // compute reference vertex coordinates
402 dom_point[0] = DataType(-1 + ((i << 1) & 2));
403 dom_point[1] = DataType(-1 + ((i ) & 2));
404
405 // evaluate trafo data
406 trafo_eval(coeff_data, dom_point);
407
408 // store jacobian matrix
409 _coeff_fod[i] = coeff_data.jac_mat;
410 }
411 }

422 template<typename EvalData_>
423 void eval_ref_values(
424 EvalData_& data,
425 const DomainPointType& point) const
426 {
427 const DataType x(point[0]), y(point[1]);
428 DataType d0, d1;
429
430 // Vertex 1: (-1,-1)
431 data.phi[ 0].ref_value = Intern::p1(x) * Intern::p1(y);
432 d0 = Intern::q1(x) * Intern::p1(y);
433 d1 = Intern::p1(x) * Intern::q1(y);
434 data.phi[ 1].ref_value = _trans_dx(0, d0, d1);
435 data.phi[ 2].ref_value = _trans_dy(0, d0, d1);
436 data.phi[ 3].ref_value = Intern::q1(x) * Intern::q1(y);
437
438 // Vertex 2: (+1,-1)
439 data.phi[ 4].ref_value = Intern::p2(x) * Intern::p1(y);
440 d0 = Intern::q2(x) * Intern::p1(y);
441 d1 = Intern::p2(x) * Intern::q1(y);
442 data.phi[ 5].ref_value = _trans_dx(1, d0, d1);
443 data.phi[ 6].ref_value = _trans_dy(1, d0, d1);
444 data.phi[ 7].ref_value = Intern::q2(x) * Intern::q1(y);
445
446 // Vertex 3: (-1,+1)
447 data.phi[ 8].ref_value = Intern::p1(x) * Intern::p2(y);
448 d0 = Intern::q1(x) * Intern::p2(y);
449 d1 = Intern::p1(x) * Intern::q2(y);
450 data.phi[ 9].ref_value = _trans_dx(2, d0, d1);
451 data.phi[10].ref_value = _trans_dy(2, d0, d1);
452 data.phi[11].ref_value = Intern::q1(x) * Intern::q2(y);
453
454 // Vertex 4: (+1,+1)
455 data.phi[12].ref_value = Intern::p2(x) * Intern::p2(y);
456 d0 = Intern::q2(x) * Intern::p2(y);
457 d1 = Intern::p2(x) * Intern::q2(y);
458 data.phi[13].ref_value = _trans_dx(3, d0, d1);
459 data.phi[14].ref_value = _trans_dy(3, d0, d1);
460 data.phi[15].ref_value = Intern::q2(x) * Intern::q2(y);
461 }

472 template<typename EvalData_>
473 void eval_ref_gradients(
474 EvalData_& data,
475 const DomainPointType& point) const
476 {
477 const DataType x(point[0]), y(point[1]);
478 DataType d0x, d0y, d1x, d1y;
479
480 // Vertex 1: (-1,-1)
481 data.phi[ 0].ref_grad[0] = Intern::dp1(x) * Intern::p1(y);
482 data.phi[ 0].ref_grad[1] = Intern::p1(x) * Intern::dp1(y);
483 d0x = Intern::dq1(x) * Intern::p1(y);
484 d0y = Intern::q1(x) * Intern::dp1(y);
485 d1x = Intern::dp1(x) * Intern::q1(y);
486 d1y = Intern::p1(x) * Intern::dq1(y);
487 data.phi[ 1].ref_grad[0] = _trans_dx(0, d0x, d1x);
488 data.phi[ 1].ref_grad[1] = _trans_dx(0, d0y, d1y);
489 data.phi[ 2].ref_grad[0] = _trans_dy(0, d0x, d1x);
490 data.phi[ 2].ref_grad[1] = _trans_dy(0, d0y, d1y);
491 data.phi[ 3].ref_grad[0] = Intern::dq1(x) * Intern::q1(y);
492 data.phi[ 3].ref_grad[1] = Intern::q1(x) * Intern::dq1(y);
493
494 // Vertex 2: (+1,-1)
495 data.phi[ 4].ref_grad[0] = Intern::dp2(x) * Intern::p1(y);
496 data.phi[ 4].ref_grad[1] = Intern::p2(x) * Intern::dp1(y);
497 d0x = Intern::dq2(x) * Intern::p1(y);
498 d0y = Intern::q2(x) * Intern::dp1(y);
499 d1x = Intern::dp2(x) * Intern::q1(y);
500 d1y = Intern::p2(x) * Intern::dq1(y);
501 data.phi[ 5].ref_grad[0] = _trans_dx(1, d0x, d1x);
502 data.phi[ 5].ref_grad[1] = _trans_dx(1, d0y, d1y);
503 data.phi[ 6].ref_grad[0] = _trans_dy(1, d0x, d1x);
504 data.phi[ 6].ref_grad[1] = _trans_dy(1, d0y, d1y);
505 data.phi[ 7].ref_grad[0] = Intern::dq2(x) * Intern::q1(y);
506 data.phi[ 7].ref_grad[1] = Intern::q2(x) * Intern::dq1(y);
507
508 // Vertex 3: (-1,+1)
509 data.phi[ 8].ref_grad[0] = Intern::dp1(x) * Intern::p2(y);
510 data.phi[ 8].ref_grad[1] = Intern::p1(x) * Intern::dp2(y);
511 d0x = Intern::dq1(x) * Intern::p2(y);
512 d0y = Intern::q1(x) * Intern::dp2(y);
513 d1x = Intern::dp1(x) * Intern::q2(y);
514 d1y = Intern::p1(x) * Intern::dq2(y);
515 data.phi[ 9].ref_grad[0] = _trans_dx(2, d0x, d1x);
516 data.phi[ 9].ref_grad[1] = _trans_dx(2, d0y, d1y);
517 data.phi[10].ref_grad[0] = _trans_dy(2, d0x, d1x);
518 data.phi[10].ref_grad[1] = _trans_dy(2, d0y, d1y);
519 data.phi[11].ref_grad[0] = Intern::dq1(x) * Intern::q2(y);
520 data.phi[11].ref_grad[1] = Intern::q1(x) * Intern::dq2(y);
521
522 // Vertex 4: (+1,+1)
523 data.phi[12].ref_grad[0] = Intern::dp2(x) * Intern::p2(y);
524 data.phi[12].ref_grad[1] = Intern::p2(x) * Intern::dp2(y);
525 d0x = Intern::dq2(x) * Intern::p2(y);
526 d0y = Intern::q2(x) * Intern::dp2(y);
527 d1x = Intern::dp2(x) * Intern::q2(y);
528 d1y = Intern::p2(x) * Intern::dq2(y);
529 data.phi[13].ref_grad[0] = _trans_dx(3, d0x, d1x);
530 data.phi[13].ref_grad[1] = _trans_dx(3, d0y, d1y);
531 data.phi[14].ref_grad[0] = _trans_dy(3, d0x, d1x);
532 data.phi[14].ref_grad[1] = _trans_dy(3, d0y, d1y);
533 data.phi[15].ref_grad[0] = Intern::dq2(x) * Intern::q2(y);
534 data.phi[15].ref_grad[1] = Intern::q2(x) * Intern::dq2(y);
