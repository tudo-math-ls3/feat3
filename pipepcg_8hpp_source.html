181 protected:
182 virtual Status _apply_intern(VectorType& vec_sol)
183 {
184 IterationStats pre_iter(*this);
185 Statistics::add_solver_expression(std::make_shared<ExpressionStartSolve>(this->name()));
186
187 const MatrixType& matrix(this->_system_matrix);
188 const FilterType& filter(this->_system_filter);
189 VectorType& vec_r(this->_vec_r);
190 VectorType& vec_u(this->_vec_u);
191 VectorType& vec_w(this->_vec_w);
192 VectorType& vec_z(this->_vec_z);
193 VectorType& vec_q(this->_vec_q);
194 VectorType& vec_s(this->_vec_s);
195 VectorType& vec_p(this->_vec_p);
196 VectorType& vec_m(this->_vec_m);
197 VectorType& vec_n(this->_vec_n);
198
199 DataType gamma, gamma_old, delta, beta, alpha;
200 gamma = DataType(0);
201 gamma_old = DataType(0);
202 delta = DataType(0);
203 beta = DataType(0);
204 alpha = DataType(0);
205
206 Status status = this->_set_initial_defect(vec_r, vec_sol);
207 if(status != Status::progress)
208 {
209 pre_iter.destroy();
210 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), status, this->get_num_iter()));
211 return status;
212 }
213
214 if(!this->_apply_precond(vec_u, vec_r, filter))
215 {
216 pre_iter.destroy();
217 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
218 return Status::aborted;
219 }
220
221 matrix.apply(vec_w, vec_u);
222 filter.filter_def(vec_w);
223
224 pre_iter.destroy();
225
226 // start iterating
227 while(status == Status::progress)
228 {
229 IterationStats stat(*this);
230
231 auto norm_def_cur = vec_r.norm2_async();
232 auto dot_gamma = vec_r.dot_async(vec_u);
233 auto dot_delta = vec_w.dot_async(vec_u);
234
235 if(!this->_apply_precond(vec_m, vec_w, filter))
236 {
237 stat.destroy();
238 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
239 return Status::aborted;
240 }
241
242 matrix.apply(vec_n, vec_m);
243 filter.filter_def(vec_n);
244
245 gamma = dot_gamma.wait();
246 delta = dot_delta.wait();
247
248 status = this->_update_defect(norm_def_cur.wait());
249 if(status != Status::progress)
250 {
251 stat.destroy();
252 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), status, this->get_num_iter()));
253 return status;
254 }
255
256 if (this->_num_iter == 1) // num_iter has already been increased to 1 by previous _update_defect call
257 {
258 alpha = gamma / delta;
259 vec_z.copy(vec_n);
260 vec_q.copy(vec_m);
261 vec_p.copy(vec_u);
262 vec_s.copy(vec_w);
263 }
264 else
265 {
266 beta = gamma / gamma_old;
267 alpha = gamma / (delta - beta / alpha * gamma);
268 vec_z.axpy(vec_z, vec_n, beta);
269 vec_q.axpy(vec_q, vec_m, beta);
270 vec_p.axpy(vec_p, vec_u, beta);
271 vec_s.axpy(vec_s, vec_w, beta);
272 }
273
274 vec_sol.axpy(vec_p, vec_sol, alpha);
275
276 /* Stabilization, recalculate real values
277 if(this->_num_iter > 40 && this->_num_iter % 50 == 0)
278 {
279 matrix.apply(vec_r, vec_sol, DataType(-1));
280 filter.filter_def(vec_r);
281 if(!this->_apply_precond(vec_u, vec_r, filter))
282 return Status::aborted;
283 matrix.apply(vec_w, vec_u);
284 filter.filter_def(vec_w);
285 }
286 else */
287 {
288 vec_u.axpy(vec_q, vec_u, -alpha);
289 vec_w.axpy(vec_z, vec_w, -alpha);
290 vec_r.axpy(vec_s, vec_r, -alpha);
291 }
292
293 gamma_old = gamma;
294 }
295
296 // we should never reach this point...
297 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::undefined, this->get_num_iter()));
298 return Status::undefined;
299 }
