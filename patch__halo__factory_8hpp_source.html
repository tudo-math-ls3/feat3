23 template<typename Shape_, int codim_>
24 class PatchHaloBuild
25 {
26 static constexpr int face_dim = Shape_::dimension - codim_;
27
28 public:
29 const TargetSet& _target_face;
30 Adjacency::Graph _elem_at_face;
31 std::vector<Index> _indices;
32
33 public:
34 explicit PatchHaloBuild(const TargetSetHolder<Shape_>& tsh, const IndexSetHolder<Shape_>& ish) :
35 _target_face(tsh.template get_target_set<face_dim>()),
36 _elem_at_face(Adjacency::RenderType::transpose, ish.template get_index_set<Shape_::dimension, face_dim>())
37 {
38 }
39
40 void build(const Index halo_rank, const Adjacency::Graph& ranks_at_elem)
41 {
42 _indices.clear();
43
44 // loop over all faces in the patch mesh part target set
45 for(Index face(0); face < _target_face.get_num_entities(); ++face)
46 {
47 // get index of base-mesh face
48 const Index base_face = _target_face[face];
49
50 // check whether one of them has our desired rank
51 if(_has_face_rank(base_face, halo_rank, ranks_at_elem))
52 {
53 _indices.push_back(face);
54 }
55 }
56 }
57
58 Index size() const
59 {
60 return Index(_indices.size());
61 }
62
63 void fill(TargetSet& target_set) const
64 {
65 XASSERT(target_set.get_num_entities() == size());
66 for(Index i(0); i < size(); ++i)
67 {
68 target_set[i] = _indices[i];
69 }
70 }
71
72 private:
73 bool _has_face_rank(const Index base_face, const Index halo_rank, const Adjacency::Graph& ranks_at_elem)
74 {
75 // loop over all elements adjacent to that face and
76 auto ib = _elem_at_face.image_begin(base_face);
77 auto ie = _elem_at_face.image_end(base_face);
78 for(auto ii(ib); ii != ie; ++ii)
79 {
80 // now loop over all ranks of this element
81 auto jb = ranks_at_elem.image_begin(*ii);
82 auto je = ranks_at_elem.image_end(*ii);
83 for(auto jj(jb); jj != je; ++jj)
84 {
85 // check rank
86 if(*jj == halo_rank)
87 return true;
88 }
89 }
90 // nope
91 return false;
92 }
93 };

148 template<typename Shape_, int face_dim_ = Shape_::dimension>
149 class PatchHaloBuildWrapper :
150 public PatchHaloBuildWrapper<Shape_, face_dim_ - 1>
151 {
152 typedef PatchHaloBuildWrapper<Shape_, face_dim_ - 1> BaseClass;
153
154 static constexpr int face_codim = Shape_::dimension - face_dim_;
155
156 PatchHaloBuild<Shape_, face_codim> _hbuild;
157
158 public:
159 explicit PatchHaloBuildWrapper(const TargetSetHolder<Shape_>& tsh, const IndexSetHolder<Shape_>& ish) :
160 BaseClass(tsh, ish),
161 _hbuild(tsh, ish)
162 {
163 }
164
165 void build(const Index halo_rank, const Adjacency::Graph& ranks_at_elem)
166 {
167 BaseClass::build(halo_rank, ranks_at_elem);
168 _hbuild.build(halo_rank, ranks_at_elem);
169 }
170
171 Index get_num_entities(int dim) const
172 {
173 XASSERT(dim <= face_dim_);
174 if(dim == face_dim_)
175 return Index(_hbuild.size());
176 else
177 return BaseClass::get_num_entities(dim);
178 }
179
180 void fill(TargetSetHolder<Shape_>& tsh) const
181 {
182 BaseClass::fill(tsh);
183 _hbuild.fill(tsh.template get_target_set<face_dim_>());
184 }
