373 template<typename X_, typename T_>
374 static std::size_t xencode(void* buf, const T_* src, const std::size_t count, bool swap_bytes)
375 {
376 X_* x = reinterpret_cast<X_*>(buf);
377 if(swap_bytes)
378 {
379 for(std::size_t i(0); i < count; ++i)
380 {
381 x[i] = xswap(static_cast<X_>(src[i]));
382 }
383 }
384 else
385 {
386 for(std::size_t i(0); i < count; ++i)
387 {
388 x[i] = static_cast<X_>(src[i]);
389 }
390 }
391 return count * sizeof(X_);
392 }

735 static std::size_t lossy_estimate_size(const std::size_t count, const Pack::Type type, const double tolerance)
736 {
737 if(count <= std::size_t(0))
738 return std::size_t(0);
739#ifdef FEAT_HAVE_ZFP
740 //get type T_, int would be handleable... but why lossy save an int array?:
741 zfp_type t = zfp_type::zfp_type_none;
742 switch(type)
743 {
744 case Pack::Type::PF32:
745 t = zfp_type::zfp_type_float;
746 break;
747 case Pack::Type::PF64:
748 t = zfp_type::zfp_type_double;
749 break;
750 default:
751 XABORTM("cannot encode compressed type; data typ not available");
752 }
753 XASSERT(t != zfp_type::zfp_type_none);
754
755 //declare needed zfp internal variables:
756 zfp_field* field;
757 zfp_stream* zfp;
758 std::size_t bytes;
759 std::size_t x_size, y_size;
760 void* temp_arr = nullptr;
761
762 //calculate needed representation size, e.g. 4 times (size/4 +1):
763 x_size = 4u;
764 y_size = count/x_size + 1;
765
766 //check if we lose significant bits through type conversion...
767 if((sizeof(std::size_t) > sizeof(uint)) && (count > std::numeric_limits<uint>::max()))
768 XABORTM("cannot encode compressed type; array size to big for internal data");
769
770 //initialize zfp field as 2D variant, that src is not right type does not matter at this point...( memory has to be freed in the end):
771 field = zfp_field_2d(temp_arr, t, (uint)x_size, (uint)y_size);
772 //open stream(has to be freed)
773 zfp = zfp_stream_open(NULL);
774 //set error tolerance of stream:
775 zfp_stream_set_accuracy(zfp, tolerance);
776 //get a conservativ estimate for the buffer size(header size included)
777 bytes = zfp_stream_maximum_size(zfp, field);
778
779 //free allocated data:
780 zfp_field_free(field);
781 zfp_stream_close(zfp);
782
783 //return compression size
784 return bytes;
785
786#else // no FEAT_HAVE_ZFP
787 (void) type;
788 (void) tolerance;
789 XABORTM("cannot encode compressed type; zfp not available");
790 return std::size_t(0);
791#endif // FEAT_HAVE_ZFP*/
792 }

806 inline std::size_t estimate_size(const std::size_t count, const Pack::Type type, const double tolerance = 1e-16)
807 {
808 if((type & Pack::Type::Mask_P) != Pack::Type::None)
809 return lossy_estimate_size(count, type, tolerance);
810 // compute raw element size
811 std::size_t raw_size = count * Pack::element_size(type & Pack::Type::Mask_T);
812
813#ifdef FEAT_HAVE_ZLIB
814 if((type & Pack::Type::Mask_Z) != Pack::Type::None)
815 {
816 return std::size_t(::compressBound(uLong(raw_size)));
817 }
818#else // no FEAT_HAVE_ZLIB
819 XASSERTM((type & Pack::Type::Mask_Z) == Pack::Type::None, "cannot estimate compressed size; zlib not available");
820#endif // FEAT_HAVE_ZLIB
821
822 // return raw size
823 return raw_size;
824 }

1110 static std::size_t lossy_encode(void* buf, T_* src, const std::size_t buf_size, const std::size_t count,
1111 const Pack::Type pack_type, bool swap_bytes, const double tolerance)

1113 // get raw type
1114 const Pack::Type raw_type = pack_type & Pack::Type::Mask_T;
1115 //Test if zfp is loaded:
1116#ifdef FEAT_HAVE_ZFP
1117 //get type T_ :

1119 switch(pack_type)
1120 {
1121 case Pack::Type::PF32:
1122 t = zfp_type::zfp_type_float;
1123 break;
1124 case Pack::Type::PF64:
1125 t = zfp_type::zfp_type_double;
1126 break;
1127 default:
1128 XABORTM("cannot encode compressed type; data typ not available");
1129 }
1130 XASSERT(t != zfp_type::zfp_type_none);

1132 //limitation is only for the header, not for the underlying zfp_array, so there could be a workaround if needed...
1133 XASSERT(count <= u64(2e14));
1134
1135 //cast src to the needed type:
1136 // ensure that the temporary buffer is large enough
1137 std::size_t raw_size = element_size(raw_type) * count + 3 * element_size(raw_type);
1138 std::vector<char> tmp(raw_size);

1140 // encode into temporary buffer
1141 encode_raw(tmp.data(), src, count, raw_type, swap_bytes);
1142

1144 zfp_field* field;
1145 zfp_stream* zfp;
1146 std::size_t real_bytes;
1147 bitstream* stream;
1148 std::size_t x_size, y_size;
1149
1150 //check if we lose significant bits through type conversion...
1151 if((sizeof(std::size_t) > sizeof(uint)) && (count > std::numeric_limits<uint>::max()))
1152 XABORTM("cannot encode compressed type; array size to big for internal data structure");
1153 //calculate needed representation size and rest:
1154 x_size = 4u;
1155 if(count%4u == 0)
1156 y_size = count/x_size;
1157 else
1158 y_size = count/x_size + 1;

1160 //initialize zfp structures:
1161 field = zfp_field_2d(tmp.data(), t, (uint)x_size, (uint)y_size);
1162 zfp = zfp_stream_open(NULL);
1163 zfp_stream_set_accuracy(zfp, tolerance);

1165 zfp_stream_set_bit_stream(zfp, stream);
1166 stream_rewind(stream);

1168 zfp_write_header(zfp, field, zfp_header_mask);
1169 //compress
1170 real_bytes = zfp_compress(zfp, field);
1171
1172 //free allocated data:
1173 zfp_field_free(field);
1174 zfp_stream_close(zfp);
1175 stream_close(stream);
1176
1177 return real_bytes;
1178#else
1179 (void)buf_size;
1180 (void) buf;
1181 (void) src;
1182 (void) raw_type;
1183 (void) swap_bytes;
1184 (void) pack_type;
1185 (void) count;
1186 (void) tolerance;
1187 XABORTM("cannot encode compressed type; zfp not available");
1188 return std::size_t(0);
1189#endif //FEAT_HAVE_ZFP

1221 static std::size_t lossy_decode(T_* dst, void* buf, const std::size_t count, const std::size_t buf_size,
1222 const Pack::Type pack_type, bool swap_bytes)

1225 const Pack::Type raw_type = pack_type & Pack::Type::Mask_T;
