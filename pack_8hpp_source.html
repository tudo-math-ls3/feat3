103 enum class Type : u16
104 {

111 Type_I = 0x0010, //< signed integer type

295 static void swap(void* x)
296 {
297 u32& v = *reinterpret_cast<u32*>(x);
298 v = ((v & 0x000000FF) << 24)
299 | ((v & 0x0000FF00) << 8)
300 | ((v & 0x00FF0000) >> 8)
301 | ((v & 0xFF000000) >> 24);

306 struct SwapHelper<8>
307 {
308 static void swap(void* x)
309 {
310 u64& v = *reinterpret_cast<u64*>(x);
311 v = ((v & 0x00000000000000FFull) << 56)
312 | ((v & 0x000000000000FF00ull) << 40)
313 | ((v & 0x0000000000FF0000ull) << 24)
314 | ((v & 0x00000000FF000000ull) << 8)
315 | ((v & 0x000000FF00000000ull) >> 8)
316 | ((v & 0x0000FF0000000000ull) >> 24)
317 | ((v & 0x00FF000000000000ull) >> 40)
318 | ((v & 0xFF00000000000000ull) >> 56);
319 }

323 struct SwapHelper<16>

375 template<typename X_, typename T_>
376 static std::size_t xencode(void* buf, const T_* src, const std::size_t count, bool swap_bytes)
377 {
378 X_* x = reinterpret_cast<X_*>(buf);
379 if(swap_bytes)
380 {
381 for(std::size_t i(0); i < count; ++i)
382 {
383 x[i] = xswap(static_cast<X_>(src[i]));
384 }
385 }
386 else
387 {
388 for(std::size_t i(0); i < count; ++i)
389 {
390 x[i] = static_cast<X_>(src[i]);
391 }
392 }
393 return count * sizeof(X_);
394 }

433 }
434 else
435 {
436 for(std::size_t i(0); i < count; ++i)
437 {
438 dest[i] = static_cast<T_>(x[i]);

737 static std::size_t lossy_estimate_size(const std::size_t count, const Pack::Type type, const double tolerance)
738 {
739 if(count <= std::size_t(0))
740 return std::size_t(0);
741#ifdef FEAT_HAVE_ZFP
742 //get type T_, int would be handleable... but why lossy save an int array?:
743 zfp_type t = zfp_type::zfp_type_none;
744 switch(type)
745 {
746 case Pack::Type::PF32:
747 t = zfp_type::zfp_type_float;
748 break;
749 case Pack::Type::PF64:
750 t = zfp_type::zfp_type_double;
751 break;
752 default:
753 XABORTM("cannot encode compressed type; data typ not available");
754 }
755 XASSERT(t != zfp_type::zfp_type_none);
756
757 //declare needed zfp internal variables:
758 zfp_field* field;
759 zfp_stream* zfp;
760 std::size_t bytes;
761 std::size_t x_size, y_size;
762 void* temp_arr = nullptr;
763
764 //calculate needed representation size, e.g. 4 times (size/4 +1):
765 x_size = 4u;
766 y_size = count/x_size + 1;
767
768 //check if we lose significant bits through type conversion...
769 if((sizeof(std::size_t) > sizeof(uint)) && (count > std::numeric_limits<uint>::max()))
770 XABORTM("cannot encode compressed type; array size to big for internal data");
771
772 //initialize zfp field as 2D variant, that src is not right type does not matter at this point...( memory has to be freed in the end):
773 field = zfp_field_2d(temp_arr, t, (uint)x_size, (uint)y_size);
774 //open stream(has to be freed)
775 zfp = zfp_stream_open(NULL);
776 //set error tolerance of stream:
777 zfp_stream_set_accuracy(zfp, tolerance);
778 //get a conservativ estimate for the buffer size(header size included)
779 bytes = zfp_stream_maximum_size(zfp, field);
780
781 //free allocated data:
782 zfp_field_free(field);
783 zfp_stream_close(zfp);
784
785 //return compression size
786 return bytes;
787
788#else // no FEAT_HAVE_ZFP
789 (void) type;
790 (void) tolerance;
791 XABORTM("cannot encode compressed type; zfp not available");
792 return std::size_t(0);
793#endif // FEAT_HAVE_ZFP*/
794 }

808 inline std::size_t estimate_size(const std::size_t count, const Pack::Type type, const double tolerance = 1e-16)
809 {
810 if((type & Pack::Type::Mask_P) != Pack::Type::None)
811 return lossy_estimate_size(count, type, tolerance);
812 // compute raw element size
813 std::size_t raw_size = count * Pack::element_size(type & Pack::Type::Mask_T);
814
815#ifdef FEAT_HAVE_ZLIB
816 if((type & Pack::Type::Mask_Z) != Pack::Type::None)
817 {
818 return std::size_t(::compressBound(uLong(raw_size)));
819 }
820#else // no FEAT_HAVE_ZLIB
821 XASSERTM((type & Pack::Type::Mask_Z) == Pack::Type::None, "cannot estimate compressed size; zlib not available");
822#endif // FEAT_HAVE_ZLIB
823
824 // return raw size
825 return raw_size;
826 }

1112 static std::size_t lossy_encode(void* buf, T_* src, const std::size_t buf_size, const std::size_t count,
1113 const Pack::Type pack_type, bool swap_bytes, const double tolerance)

1126 case Pack::Type::PF64:
1127 t = zfp_type::zfp_type_double;

1135 XASSERT(count <= u64(2e14));
1136
1137 //cast src to the needed type:
1138 // ensure that the temporary buffer is large enough
1139 std::size_t raw_size = element_size(raw_type) * count + 3 * element_size(raw_type);
1140 std::vector<char> tmp(raw_size);

1142 // encode into temporary buffer
1143 encode_raw(tmp.data(), src, count, raw_type, swap_bytes);
1144

1146 zfp_field* field;
1147 zfp_stream* zfp;
1148 std::size_t real_bytes;
1149 bitstream* stream;
1150 std::size_t x_size, y_size;
1151
1152 //check if we lose significant bits through type conversion...
1153 if((sizeof(std::size_t) > sizeof(uint)) && (count > std::numeric_limits<uint>::max()))
1154 XABORTM("cannot encode compressed type; array size to big for internal data structure");
1155 //calculate needed representation size and rest:
1156 x_size = 4u;
1157 if(count%4u == 0)
1158 y_size = count/x_size;
1159 else
1160 y_size = count/x_size + 1;

1162 //initialize zfp structures:
1163 field = zfp_field_2d(tmp.data(), t, (uint)x_size, (uint)y_size);

1165 zfp_stream_set_accuracy(zfp, tolerance);

1167 zfp_stream_set_bit_stream(zfp, stream);
1168 stream_rewind(stream);

1170 zfp_write_header(zfp, field, zfp_header_mask);
1171 //compress
1172 real_bytes = zfp_compress(zfp, field);
1173
1174 //free allocated data:
1175 zfp_field_free(field);
1176 zfp_stream_close(zfp);
1177 stream_close(stream);
1178
1179 return real_bytes;
1180#else
1181 (void)buf_size;
1182 (void) buf;
1183 (void) src;
1184 (void) raw_type;
1185 (void) swap_bytes;
1186 (void) pack_type;
1187 (void) count;
1188 (void) tolerance;
1189 XABORTM("cannot encode compressed type; zfp not available");
1190 return std::size_t(0);
