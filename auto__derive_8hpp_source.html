177 GradientType extrapol_grad(const PointType& point)
178 {
179 // first, create a mutable copy of our point
180 PointType v(point);
181
182 // next, choose the initial h
183 DataType h(_init_grad_h);
184
185 // Note: the '_grad' vector was already allocated to
186 // the correct size by our constructor
187
188 // evaluate gradient
189 this->_eval_grad_quot(_grad[0], v, h);
190
191 // Now comes the Richardson extrapolation loop
192 const std::size_t n(_grad.size()-1);
193 DataType def(Math::huge<DataType>());
194 for(std::size_t i(0); i < n; ++i)
195 {
196 // evaluate next gradient
197 this->_eval_grad_quot(_grad[i+1], v, h *= DataType(0.5));
198
199 // initialize scaling fator
200 DataType q = DataType(1);
201
202 // perform extrapolation steps except for the last one
203 for(std::size_t k(i); k > std::size_t(0); --k)
204 {
205 q *= DataType(4);
206 (_grad[k] -= q*_grad[k+1]) *= DataType(1) / (DataType(1) - q);
207 }
208
209 // compute and check our defect
210 DataType d = def_norm_sqr(_grad[1], _grad[0]);
211 if(def <= d)
212 {
213 // The defect has increased, so we return the previous extrapolation result
214 return _grad.front();
215 }
216
217 // remember current defect
218 def = d;
219
220 // perform last extrapolation step
221 q *= DataType(4);
222 (_grad[0] -= q*_grad[1]) *= DataType(1) / (DataType(1) - q);
223 }
224
225 // return our extrapolated gradient
226 return _grad.front();
227 }

300 template<int l_, int m_, int n_, int sl_, int sm_, int sn_>
301 static DataType def_norm_sqr(
302 const Tiny::Tensor3<DataType, l_, m_, n_, sl_, sm_, sn_>& x,
303 const Tiny::Tensor3<DataType, l_, m_, n_, sl_, sm_, sn_>& y)
304 {
305 DataType r(DataType(0));
306 for(int i(0); i < l_; ++i)
307 {
308 for(int j(0); j < m_; ++j)
309 {
310 for(int k(0); k < n_; ++k)
311 {
312 r += Math::sqr(x(i,j,k) - y(i,j,k));
313 }
314 }
315 }

335 void _eval_grad_quot(GradientType& x, PointType& v, const DataType h)
336 {
337 // difference quotient denominator
338 const DataType denom = DataType(1) / (DataType(2) * h);
339 ValueType vr, vl;
340
341 // loop over all dimensions
342 for(int i(0); i < domain_dim; ++i)
343 {
344 // backup coordinate i
345 const DataType vi(v[i]);
346
347 // evaluate f(v + h*e_i)
348 v[i] = vi + h;
349 vr = _func_eval.value(v);
350
351 // evaluate f(v - h*e_i)
352 v[i] = vi - h;
353 vl = _func_eval.value(v);
354
355 // compute difference quotient with appropriate overload (scalar/vector)
356 _set_grad_quot(x, i, vr, vl, denom);
357
358 // restore coord
359 v[i] = vi;
360 }
361 }

394 void _eval_hess_quot(HessianType& x, PointType& v, const DataType h)
395 {
396 // difference quotient denominators
397 const DataType denom1 = DataType(1) / (h * h);
398 const DataType denom2 = DataType(1) / (DataType(4) * h * h);
399 ValueType vc, vr, vl, vne, vnw, vse, vsw;
400
401 // evaluate at point
402 vc = _func_eval.value(v);
403
404 // loop over all dimensions
405 for(int i(0); i < domain_dim; ++i)
406 {
407 // backup coord
408 const DataType vi(v[i]);
409
410 // eval f(x + h*e_i)
411 v[i] = vi + h;
412 vr = _func_eval.value(v);
413
414 // eval f(x-h)
415 v[i] = vi - h;
416 vl = _func_eval.value(v);
417
418 // compute difference quotient
419 _set_hess_quot(x, i, vr, vl, vc, denom1);
420
421 // now the mixed derivatives
422 for(int j(i+1); j < domain_dim; ++j)
423 {
424 // backup coord
425 const DataType vj(v[j]);
426
427 // we need four points here:
428 // north-east: f(v + h*e_i + h*e_j)
429 v[i] = vi + h;
430 v[j] = vj + h;
431 vne = _func_eval.value(v);
432
433 // north-west: f(v - h*e_i + h*e_j)
434 v[i] = vi - h;
435 v[j] = vj + h;
436 vnw = _func_eval.value(v);
437
438 // south-east: f(v + h*e_i - h*e_j)
439 v[i] = vi + h;
440 v[j] = vj - h;
441 vse = _func_eval.value(v);
442
443 // south-west: f(v - h*e_i - h*e_j)
444 v[i] = vi - h;
445 v[j] = vj - h;
446 vsw = _func_eval.value(v);
447
448 // combine into difference quotient
449 _set_hess_quot(x, i, j, vne, vsw, vnw, vse, denom2);
450
451 // restore coord
452 v[j] = vj;
453 }
454
455 // restore coord
456 v[i] = vi;
457 }
