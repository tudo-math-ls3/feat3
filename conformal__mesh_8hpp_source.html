320 void create_permutation(PermutationStrategy strategy)
321 {
322 // make sure that we don't already have a permutation
323 XASSERTM(this->_permutation.empty(), "mesh is already permuted!");
324
325 // create the permutation
326 this->_permutation.create(strategy, this->_index_set_holder, this->_vertex_set);
327
328 // permute vertex set
329 this->_vertex_set.permute(this->_permutation.get_perm(0));
330
331 // permute index sets
332 this->_index_set_holder.permute(this->_permutation.get_perms(), this->_permutation.get_inv_perms());
333
334 // rebuild neighbor index set
335 fill_neighbors();
336 }

354 void set_permutation(MeshPermutationType&& mesh_perm)
355 {
356 // make sure that we don't already have a permutation
357 XASSERTM(this->_permutation.empty(), "mesh is already permuted!");
358
359 // check the dimensions
360 XASSERTM(mesh_perm.validate_sizes(this->_num_entities) == 0, "mesh permutation has invalid size!");

363 this->_permutation = std::forward<MeshPermutationType>(mesh_perm);

367
368 // permute index sets
369 this->_index_set_holder.permute(this->_permutation.get_perms(), this->_permutation.get_inv_perms());
370
371 // rebuild neighbor index set
372 fill_neighbors();
373 }

383 bool validate_element_coloring() const
384 {
385 // no coloring?
386 const std::vector<Index>& coloring = this->get_mesh_permutation().get_element_coloring();
387 if(coloring.empty())
388 return true;
389
390 // get vertices-at-element index set
391 const auto& verts_at_elem = this->template get_index_set<shape_dim, 0>();
392
393 // render transpose
394 Adjacency::Graph elems_at_vert(Adjacency::RenderType::transpose, verts_at_elem);
395
396 // loop over all color blocks
397 for(std::size_t icol(0); icol+1u < coloring.size(); ++icol)
398 {
399 // get the bounds of our current color block
400 const Index iel_beg = coloring[icol];
401 const Index iel_end = coloring[icol+1u];
402
403 // loop over all elements in the current color block
404 for(Index iel(iel_beg); iel < iel_end; ++iel)
405 {
406 // loop over all vertices adjacent to this element
407 for(int ivt(0); ivt < verts_at_elem.num_indices; ++ivt)
408 {
409 // loop over all elements adjacent to this vertex
410 const Index ivtx = verts_at_elem(iel, ivt);
411 for(auto it = elems_at_vert.image_begin(ivtx); it != elems_at_vert.image_end(ivtx); ++it)
412 {
413 // two adjacent element must not be in the same color block
414 if((iel_beg <= *it) && (*it < iel_end) && (*it != iel))
415 return false; // invalid coloring
416 }
417 }
418 }
419 } // next color block
420
421 // ok, coloring is valid
422 return true;
423 }

433 bool validate_element_layering() const
434 {
435 // no layering?
436 const std::vector<Index>& layering = this->get_mesh_permutation().get_element_layering();
437 if(layering.empty())
438 return true;
439
440 // get vertices-at-element index set
441 const auto& verts_at_elem = this->template get_index_set<shape_dim, 0>();
442
443 // render transpose
444 Adjacency::Graph elems_at_vert(Adjacency::RenderType::transpose, verts_at_elem);
445
446 // loop over all layers
447 for(std::size_t ilay(0); ilay+1u < layering.size(); ++ilay)
448 {
449 // get the bounds of our current layer
450 const Index iel_beg = layering[ilay];
451 const Index iel_end = layering[ilay+1u];
452
453 // get the lower bound for valid neighbors of our current layer = beginning of previous layer
454 const Index iel_lower = layering[Math::max(ilay, std::size_t(1)) - 1u];
455
456 // get the upper bound for valid neighbors of our current layer = end of next layer
457 const Index iel_upper = layering[Math::min(ilay+2u, layering.size()-1u)];
458
459 // loop over all elements in the current layer
460 for(Index iel(iel_beg); iel < iel_end; ++iel)
461 {
462 // loop over all vertices adjacent to this element
463 for(int ivt(0); ivt < verts_at_elem.num_indices; ++ivt)
464 {
465 // loop over all elements adjacent to this vertex
466 const Index ivtx = verts_at_elem(iel, ivt);
467 for(auto it = elems_at_vert.image_begin(ivtx); it != elems_at_vert.image_end(ivtx); ++it)
468 {
469 // adjacent element outside of adjacent layers?
470 if(!((iel_lower <= *it) && (*it < iel_upper)))
471 return false; // invalid layer
472 }
473 }
474 }
475 } // next color block
476
477 // ok, layering is valid
478 return true;
479 }

482 void fill_neighbors()
483 {
484 // Facet at cell index set
485 auto& facet_idx = get_index_set<shape_dim, shape_dim -1>();
486
487 XASSERTM(get_num_entities(shape_dim-1) == facet_idx.get_index_bound(), "mesh num_entities / index_set num_entities mismatch");
488
489 if(_neighbors.get_num_indices() == Index(0))
490 _neighbors = std::move(typename IndexSet<shape_dim, shape_dim-1>::Type(get_num_entities(shape_dim)));
491
492 Intern::FacetNeighbors::compute(_neighbors, facet_idx);
493 }
