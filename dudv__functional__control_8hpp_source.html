155 explicit DuDvFunctionalControl(
156 DomainControl_& dom_ctrl,
157 const int meshopt_lvl_,
158 const std::deque<String>& dirichlet_list, const std::deque<String>& slip_list,
159 const String& solver_name_, PropertyMap& solver_config_,
160 bool fixed_reference_domain_):
161 BaseClass(dom_ctrl, dirichlet_list, slip_list),
162 _system_levels(),
163 solver_config(solver_config_),
164 solver_name(solver_name_),
165 solver(nullptr),
166 fixed_reference_domain(fixed_reference_domain_),
167 meshopt_lvl(meshopt_lvl_),
168 meshopt_lvl_pos(~size_t(0))
169 {
170 XASSERT(meshopt_lvl >= -1);
171
172 // If the input level was set to -1, take the max level of the domain control
173 if(meshopt_lvl == -1)
174 {
175 meshopt_lvl = dom_ctrl.max_level_index();
176 }
177
178 XASSERT(meshopt_lvl<= dom_ctrl.max_level_index());
179
180 // Now find the position of the coarsest mesh optimization level in the domain levels
181 for(size_t i(0); i < dom_ctrl.size_physical(); ++i)
182 {
183 if(dom_ctrl.at(i)->get_level_index() == meshopt_lvl)
184 {
185 meshopt_lvl_pos = i;
186 break;
187 }
188 }
189
190 XASSERT(meshopt_lvl_pos < dom_ctrl.size_physical());
191
192 for(size_t i(0); i < dom_ctrl.size_physical(); ++i)
193 {
194 _system_levels.push_back( new SystemLevelType(
195 dom_ctrl.at(i)->get_level_index(),
196 dom_ctrl.at(i)->get_mesh_node(),
197 dom_ctrl.at(i)->trafo,
198 dirichlet_list, slip_list));
199
200 // This assembles the system matrix numerically
201 _system_levels.at(i)->local_functional.init();
202 }
203
204 // Now that _system_levels has the correct size, we can use get_num_levels()
205 for(Index i(0); i < get_num_levels(); ++i)
206 {
207 _system_levels.at(i)->assemble_gates(dom_ctrl.at(i).layer());
208 }
209
210 // Assemble the transfer matrices on all levels except for the coarsest
211 for(Index i(0); i+1 < get_num_levels(); ++i)
212 {
213 _system_levels.at(i)->assemble_system_transfer(*_system_levels.at(i+1));
214 }
215
216 // Now we need to assemble the filters on all levels before any shallow copies are made. This is done
217 // by calling prepare()
218 typename SystemLevelType::GlobalSystemVectorR vec_buf;
219 vec_buf.local().convert(_system_levels.front()->coords_buffer.local());
220 prepare(vec_buf);
221
222 // create the solver
223 this->_create_solver();
224 }

297 virtual CoordType compute_cell_size_defect(CoordType& lambda_min, CoordType& lambda_max,
298 CoordType& vol_min, CoordType& vol_max, CoordType& vol) const override
299 {
300
301 _system_levels.front()->local_functional.compute_cell_size_defect_pre_sync(vol_min, vol_max, vol);
302
303 vol_min = _system_levels.front()->gate_sys.min(vol_min);
304 vol_max = _system_levels.front()->gate_sys.max(vol_max);
305 vol = _system_levels.front()->gate_sys.sum(vol);
306
307 CoordType cell_size_defect =
308 _system_levels.front()->local_functional.compute_cell_size_defect_post_sync(lambda_min, lambda_max, vol_min, vol_max, vol);
309
310 lambda_min = _system_levels.front()->gate_sys.min(lambda_min);
311 lambda_max = _system_levels.front()->gate_sys.max(lambda_max);
312 cell_size_defect = _system_levels.front()->gate_sys.sum(cell_size_defect);
313
314 return cell_size_defect;
315 }

330 virtual void buffer_to_mesh() override
331 {
332 // Write finest level
333 _system_levels.front()->local_functional.buffer_to_mesh();
334
335 // Get the coords buffer on the finest level
336 const auto& coords_buffer_loc = _system_levels.front()->local_functional.get_coords();
337
338 // Transfer fine coords buffer to coarser levels and perform buffer_to_mesh
339 for(size_t level(0); level < get_num_levels(); ++level)
340 {
341 Index ndofs(_system_levels.at(level)->local_functional.trafo_space.get_num_dofs());
342
343 // At this point, what we really need is a primal restriction operator that restricts the FE function
344 // representing the coordinate distribution to the coarser level. This is very simple for continuous
345 // Lagrange elements (just discard the additional information from the fine level), but not clear in
346 // the generic case. So we use an evil hack here:
347 // Because of the underlying two level ordering, we just need to copy the first ndofs entries from
348 // the fine level vector.
349 typename SystemLevelType::LocalCoordsBuffer
350 vec_level(coords_buffer_loc, ndofs, Index(0));
351
352 _system_levels.at(level)->local_functional.get_coords().copy(vec_level);
353 _system_levels.at(level)->local_functional.buffer_to_mesh();
354 }

467 virtual Solver::Status apply(GlobalSystemVectorR& vec_sol, const GlobalSystemVectorL& vec_rhs)
468 {
469 // Get our global system matrix and filter
470 GlobalSystemMatrix& mat_sys = _system_levels.at(meshopt_lvl_pos)->matrix_sys;
471 GlobalSystemFilter& filter_sys = _system_levels.at(meshopt_lvl_pos)->filter_sys;
472
473 // Update the containers in the MatrixStock
474 _multigrid_hierarchy->done_numeric();
475 _multigrid_hierarchy->init_numeric();
476
477 return Solver::solve(*solver, vec_sol, vec_rhs, mat_sys, filter_sys);
478 }

481 virtual void optimize() override
482 {
483 // Reassemble the system matrix
484 init_numeric();
485
486 // fetch our finest levels
487 SystemLevelType& the_system_level = *_system_levels.at(meshopt_lvl_pos);
488
489 // Create our RHS and SOL vectors
490 GlobalSystemVectorR vec_rhs = the_system_level.assemble_rhs_vector();
491 GlobalSystemVectorL vec_sol = the_system_level.assemble_sol_vector();
492
493 // Let it be known to Statistics that it was Us who called the solver
494 FEAT::Statistics::expression_target = name();
495
496 // solve
497 this->apply(vec_sol, vec_rhs);
498
499 // Write the solution to the control object's buffer and the buffer to mesh
500 typename SystemLevelType::LocalCoordsBuffer vec_buf;
501 vec_buf.convert(vec_sol.local());
502 the_system_level.coords_buffer.local().copy(vec_buf);
503
504 the_system_level.local_functional.buffer_to_mesh();
505
506 //If the mesh was not optimized on the finest domain level, we now need to prolongate the solution by:
507 // - refine the coarse vertex set using the StandardRefinery
508 // - copy the results to the CoordsBuffer
509 // - convert the CoordsBuffer to a vector type that we can filter with the system's Dirichlet filter
510 // - copy the filtered vector's contents back to the buffer and write the buffer to the mesh
511 // - apply the nonlinear filter representing unilateral BCs of place by calling adapt() for the
512 // corresponding meshparts
513 for(size_t pos(meshopt_lvl_pos); pos > size_t(0);)
514 {
515 --pos;
516
517 auto& coarse_mesh = this->_dom_ctrl.at(pos+1)->get_mesh();
518 auto& fine_mesh = this->_dom_ctrl.at(pos)->get_mesh();
519 auto& fine_vtx = fine_mesh.get_vertex_set();
520
521 // Refine coarse vertex set and write the result to the CoordsBuffer
522 Geometry::StandardRefinery<MeshType> refinery(coarse_mesh);
523 refinery.fill_vertex_set(fine_vtx);
524 _system_levels.at(pos)->local_functional.mesh_to_buffer();
525 // Convert the buffer to a filterable vector
526 typename GlobalSystemVectorL::LocalVectorType vec_sol_lvl;
527 vec_sol_lvl.convert(_system_levels.at(pos)->coords_buffer.local());
528 // Filter this vector, copy back the contents and write the changes to the mesh
529 auto& dirichlet_filters_lvl = _system_levels.at(pos)->filter_sys.local().template at<1>();
530 dirichlet_filters_lvl.filter_sol(vec_sol_lvl);
531 _system_levels.at(pos)->coords_buffer.local().copy(vec_sol_lvl);
532 _system_levels.at(pos)->local_functional.buffer_to_mesh();
533 // Now call adapt() on the slip boundaries
534 auto* fine_mesh_node = this->_dom_ctrl.at(pos)->get_mesh_node();
535 for(const auto& it:this->get_slip_boundaries())
536 {
537 fine_mesh_node->adapt_by_name(it);
538 }
539 }
540
541 // Now we need to update all coarser levels
542 for(size_t pos(meshopt_lvl_pos+1); pos < get_num_levels(); ++pos)
543 {
544 Index ndofs(_system_levels.at(pos)->local_functional.trafo_space.get_num_dofs());
545
546 // At this point, what we really need is a primal restriction operator that restricts the FE function
547 // representing the coordinate distribution to the coarser level. This is very simple for continuous
548 // Lagrange elements (just discard the additional information from the fine level), but not clear in
549 // the generic case. So we use an evil hack here:
550 // Because of the underlying two level ordering, we just need to copy the first ndofs entries from
551 // the fine level vector.
552 typename SystemLevelType::GlobalCoordsBuffer
553 global_sol_level( &(_system_levels.at(pos)->gate_sys), vec_sol.local(), ndofs, Index(0));
554
555 // prepare() needs to be called of the local object, because there is a Global::Matrix around it
556 // which knows nothing of prepare()
557 _system_levels.at(pos)->local_functional.prepare(
558 global_sol_level.local(), _system_levels.at(pos)->filter_sys.local());
559
560 _system_levels.at(pos)->local_functional.get_coords().copy(global_sol_level.local());
561 _system_levels.at(pos)->local_functional.buffer_to_mesh();
562 }
563
564 } // void optimize()

566 protected:
567 void _create_solver()
568 {
569 // first of all, let's fetch all the subsections for the DuDv solver
570 auto* sec_linsol = solver_config.query_section(this->solver_name);
571 auto* sec_mg_coarse = solver_config.query_section("DuDvMGCoarseSolver");
572 auto* sec_mg_smooth = solver_config.query_section("DuDvMGSmoother");
573 XASSERTM(sec_linsol != nullptr, "mandatory DuDv solver config section is missing");
574 XASSERTM(sec_mg_coarse != nullptr, "mandatory solver section [DuDvMGCoarseSolver] is missing");
575 XASSERTM(sec_mg_smooth != nullptr, "mandatory solver section [DuDvMGSmoother] is missing");
576
577 // our multigrid smoother is a simple Jacobi-smoother, so parse the damping parameter
578 // as well as the number of smoothing steps
579 Index smooth_steps = 4;
580 auto smooth_steps_p = sec_mg_smooth->get_entry("steps");
581 XASSERTM(smooth_steps_p.second, "DuDvMGSmoother.steps parameter is missing");
582 smooth_steps_p.first.parse(smooth_steps);
583
584 DT_ smooth_omega = DT_(0.5);
585 auto smooth_damp_p = sec_mg_smooth->get_entry("omega");
586 XASSERTM(smooth_damp_p.second, "DuDvMGSmoother.omega parameter is missing");
587 smooth_damp_p.first.parse(smooth_omega);
588
589 // now let's build the multigrid hierarchy
590 _multigrid_hierarchy = std::make_shared<Solver::MultiGridHierarchy<GlobalSystemMatrix,
591 GlobalSystemFilter, typename SystemLevelType::GlobalSystemTransfer>>(this->_dom_ctrl.size_virtual());
592
593 // create smoothers for all levels except the coarsest one
594 for(Index i(0); (i+1) < get_num_levels(); ++i)
595 {
596 const auto& lvl = *(_system_levels.at(i));
597 auto jacobi = Solver::new_jacobi_precond(lvl.matrix_sys, lvl.filter_sys);
598 auto smooth = Solver::new_richardson(lvl.matrix_sys, lvl.filter_sys, smooth_omega, jacobi);
599 smooth->set_min_iter(smooth_steps);
600 smooth->set_max_iter(smooth_steps);
601 _multigrid_hierarchy->push_level(lvl.matrix_sys, lvl.filter_sys, lvl.transfer_sys, smooth, smooth, smooth);
602 }
603
604 // now let's create the coarse grid solver; this is always a PCG-Jacobi, which is configured
605 // by the [DuDvMGCoarseSolver] section
606 {
607 const auto& lvl = *_system_levels.back();
608 auto jacobi = Solver::new_jacobi_precond(lvl.matrix_sys, lvl.filter_sys);
609 auto cg_pcg = Solver::new_pcg("DuDvMGCoarseSolver", sec_mg_coarse, lvl.matrix_sys, lvl.filter_sys, jacobi);
610 _multigrid_hierarchy->push_level(lvl.matrix_sys, lvl.filter_sys, cg_pcg);
611 }
612
613 // now, let's create the actual multigrid solver object
614 auto multigrid = Solver::new_multigrid(_multigrid_hierarchy, Solver::MultiGridCycle::V);
615
616 // get the finest level
617 const auto& lvl = *_system_levels.front();
618
619 // finally, create the outer solver, which is always a PCG solver
620 solver = Solver::new_pcg("DuDvLinearSolver", sec_linsol, lvl.matrix_sys, lvl.filter_sys, multigrid);
621
622 // initialize the hierarchy and the solver
623 _multigrid_hierarchy->init();
624 solver->init();
625 }
