33 template<int dim>
34 struct IndexSetFiller
35 {
36 template<typename IndexSetHolderType, typename TargetSetHolderType, typename ParentIndexSetHolderType>
37 static void fill_ish(IndexSetHolderType& ish, const TargetSetHolderType& tsh, const ParentIndexSetHolderType& parent_ish)
38 {
39 // Recurse down
40 IndexSetFiller<dim-1>::fill_ish(ish, tsh, parent_ish);
41
42 // The MeshPart's vertex@shape set
43 auto& index_set(ish.template get_index_set<dim, 0>());
44 // The Mesh's vertex@shape set
45 auto& index_set_parent(parent_ish.template get_index_set<dim, 0>());
46
47 auto& target_set_vertex(tsh.template get_target_set<0>());
48 auto& target_set_dim(tsh.template get_target_set<dim>());
49
50 // For every vertex in the parent, this will contain its index in the MeshPart
51 std::vector<Index> inverse_target_map(index_set_parent.get_index_bound());
52
53 // Set every index to something out of range to catch errors
54 for(Index i(0); i < index_set_parent.get_index_bound(); ++i)
55 inverse_target_map[i] = target_set_vertex.get_num_entities() + Index(1);
56
57 for(Index i(0); i < target_set_vertex.get_num_entities(); ++i)
58 inverse_target_map[target_set_vertex[i]] = i;
59
60 // Now we can just iterate over the shapes of the MeshPart
61 for(Index cell(0); cell < target_set_dim.get_num_entities(); ++cell)
62 {
63 // For the shape cell, get its index in the parent. Then get the local vertex' index in the parent and
64 // map that back to the MeshPart with the inverse_target_map. Ez!
65 for(int i(0); i < index_set_parent.get_num_indices(); ++i)
66 index_set[cell][i] = inverse_target_map[index_set_parent[target_set_dim[cell]][i]];
67 }
68 }
69 }; // struct IndexSetFiller<int>
