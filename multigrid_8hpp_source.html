1270 virtual Status apply(VectorType& vec_cor, const VectorType& vec_def) override
1271 {
1272 Statistics::add_solver_expression(std::make_shared<ExpressionStartSolve>(this->name()));
1273
1274 // reset statistics counters
1275 for(std::size_t i(0); i < std::size_t(_hierarchy->size_virtual()); ++i)
1276 {
1277 _toes.at(i) = 0.0;
1278 _mpi_execs_reduction.at(i) = 0.0;
1279 _mpi_execs_blas2.at(i) = 0.0;
1280 _mpi_execs_blas3.at(i) = 0.0;
1281 _mpi_execs_collective.at(i) = 0.0;
1282 _mpi_waits_reduction.at(i) = 0.0;
1283 _mpi_waits_blas2.at(i) = 0.0;
1284 _mpi_waits_blas3.at(i) = 0.0;
1285 _mpi_waits_collective.at(i) = 0.0;
1286 }
1287
1288 // get a reference to the finest level
1289 LevelInfo& lvl_top = _hierarchy->_get_level_info(_top_level);
1290
1291 // copy defect to RHS vector
1292 lvl_top.vec_rhs.copy(vec_def);
1293
1294 // apply the corresponding cycle
1295 Status status(Status::undefined);
1296 switch(_cycle)
1297 {
1298 case MultiGridCycle::V:
1299 status = this->_apply_cycle_v();
1300 break;
1301
1302 case MultiGridCycle::F:
1303 status = this->_apply_cycle_f();
1304 break;
1305
1306 case MultiGridCycle::W:
1307 status = this->_apply_cycle_w();
1308 break;
1309
1310 default:
1311 // whoops...
1312 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, 1));
1313 status = Status::aborted;
1314 break;
1315 }
1316
1317 // copy solution to correction vector
1318 vec_cor.copy(lvl_top.vec_sol);
1319
1320 // propagate solver statistics
1321 for(std::size_t i(0); i < std::size_t(_hierarchy->size_virtual()); ++i)
1322 {
1323 Statistics::add_solver_expression(std::make_shared<ExpressionLevelTimings>(this->name(), Index(i),
1324 _toes.at(i), _mpi_execs_reduction.at(i), _mpi_execs_blas2.at(i), _mpi_execs_blas3.at(i), _mpi_execs_collective.at(i), _mpi_waits_reduction.at(i), _mpi_waits_blas2.at(i),
1325 _mpi_waits_blas3.at(i), _mpi_waits_collective.at(i)));
1326 }
1327
1328 // okay
1329 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::success, 1));
1330 return status;
1331 }

1333 protected:
1335 Status _apply_cycle_v()
1336 {
1337 // The V-cycle is pretty simple:
1338 // 1) Restrict from the top level to the coarse level
1339 // 2) Solve coarse level system
1340 // 3) Prolongate from the coarse level to the top level
1341
1342 // restrict (apply top-level pre-smoother if it exists)
1343 this->_apply_rest(_top_level, true);
1344
1345 // solve coarse level
1346 this->_apply_coarse();
1347

1356 Status _apply_cycle_f()
1357 {
1358 // The F-cycle is slightly more complex than the V-cycle:
1359 // 1) Restrict from the top level to the coarse level
1360 //
1361 // 2) For each intermediate level (that's everything between the
1362 // coarse and top level) in ascending order:
1363 // 2.1) solve the coarse system
1364 // 2.2) prolongate to the intermediate level without applying
1365 // the post-smoother on the intermediate level
1366 // 2.3) apply peak-smoother
1367 // 2.4) restrict to the coarse level without applying
1368 // the pre-smoother on the intermediate level
1369 //
1370 // 3) Solve the coarse level system
1371 //
1372 // 4) Prolongate from the coarse level to the top level
1373
1374 // restrict (apply top-level pre-smoother if it exists)
1375 this->_apply_rest(_top_level, true);
1376
1377 // get the coarsest level on this process; that's either the
1378 // coarse level or the last level for this process:
1379 const Index last_level = Math::min(_crs_level, _hierarchy->size_physical());
1380
1381 // ensure that we do not apply this if there is just one level
1382 if(last_level > Index(0))
1383 {
1384 // F-cycle intermediate peak-level loop
1385 for(Index peak_lvl(last_level-1); peak_lvl > _top_level; --peak_lvl)
1386 {
1387 // solve coarse level
1388 this->_apply_coarse();
1389
1390 // prolongate to current peak-level without post-smoothing
1391 this->_apply_prol(peak_lvl, false);
1392
1393 // apply peak-smoother
1394 this->_apply_smooth_peak(peak_lvl);
1395
1396 // restrict from current peak-level without pre-smoothing
1397 this->_apply_rest(peak_lvl, false);
1398 }
1399 }
1400
1401 // solve coarse level
1402 this->_apply_coarse();
1403
1404 // prolongate (apply top-level post-smoother if it exists)
1405 this->_apply_prol(_top_level, true);
1406
1407 // okay
1408 return Status::success;
1409 }

1412 Status _apply_cycle_w()
1413 {
1414 // The W-cycle is a tough nut to implement/understand:
1415 // As always, we start by restricting from the top level to the
1416 // coarse level and solving the coarse level system afterwards.
1417 //
1418 // The next part is the tricky bit:
1419 // We need to implement the "inner W-cycle part". In theory,
1420 // this is usually defined as a "recursive 2x V-cycle application",
1421 // but we do not want to use recursion here explicitly.
1422 // We know that the W-cycle performs 2^(top_level - crs_level)
1423 // coarse grid solves, so we use that for our for-loop.
1424 // Moreover, we manage a counter for each level above the coarse
1425 // level and count how often we have applied the peak-smoother.
1426 // Each iteration we choose the lowest level that has a peak-smooth
1427 // count equal to 0 as the next intermediate level, increment
1428 // its counter by 1 and reset all level counters below this level
1429 // back to 0. This may sound perfectly confusing, but it actually
1430 // does the job :) See the description of the multigrid cycles
1431 // in the documentation for further explanations and a fancy image
1432 // of the W-cycle level counters in action.
1433 //
1434 // Oh yes, and finally we also need to prolongate from the
1435 // coarse level to the top level, of course.
1436
1437 // restrict (apply top-level pre-smoother if it exists)
1438 this->_apply_rest(_top_level, true);
1439
1440 // solve coarse level
1441 this->_apply_coarse();
1442
1443 // get the coarsest level on this process; that's either the
1444 // coarse level or the last level for this process:
1445 const Index last_level = Math::min(_crs_level, _hierarchy->size_physical());
1446
1447 // reset all level counters to 0
1448 for(std::size_t i(_top_level); i <= std::size_t(last_level); ++i)
1449 {
1450 _counters[i] = 0;
1451 }
1452
1453 // compute the total number of coarse-grid solves to perform:
1454 // that's = 2^(top_level - crs_level)
1455 const int num_cgs = 1 << (last_level - _top_level);
1456
1457 // W-cycle loop: count the number of coarse-grid solves
1458 for(int cgs(1); cgs < num_cgs; ++cgs)
1459 {
1460 // find the next peak level; that's the lowest level
1461 // above the coarse level whose counter is 0
1462 std::size_t peak_lvl = std::size_t(last_level);
1463 while(peak_lvl > std::size_t(_top_level))
1464 {
1465 // if the counter is 0, we have our next peak level
1466 if(_counters[--peak_lvl] == 0)
1467 break;
1468 }
1469
1470 // >>>>> DEBUG >>>>>
1471 //std::cout << stringify(cgs).pad_front(3) << " ";
1472 //std::cout << stringify(peak_lvl).pad_front(2) << ":";
1473 //for(std::size_t i(0); i <= std::size_t(top_lvl); ++i)
1474 // std::cout << " " << _counters[i];
1475 // <<<<< DEBUG <<<<<
1476
1477 // sanity check: do not go beyond our top level
1478 XASSERTM(peak_lvl >= _top_level, "W-cycle sanity check failed: invalid peak level");
1479
1480 // reset all counters below our peak level to 0
1481 for(std::size_t i = std::size_t(last_level-1); i > peak_lvl; --i)
1482 _counters[i] = 0;
1483
1484 // increment the counter of our peak level by 1,
1485 // indicating that we are going to perform an
1486 // peak-smoother step on this level
1487 ++_counters[peak_lvl];
1488
1489 // >>>>> DEBUG >>>>>
1490 //std::cout << " >";
1491 //for(std::size_t i(0); i <= std::size_t(top_lvl); ++i)
1492 // std::cout << " " << _counters[i];
1493 //std::cout << std::endl;
1494 // <<<<< DEBUG <<<<<
1495
1496 // prolongate to current peak-level without post-smoothing
1497 this->_apply_prol(Index(peak_lvl), false);
1498
1499 // apply peak-smoother
1500 this->_apply_smooth_peak(Index(peak_lvl));
1501
1502 // restrict from current peak-level without pre-smoothing
1503 this->_apply_rest(Index(peak_lvl), false);
1504
1505 // solve coarse level
1506 this->_apply_coarse();
1507 }
1508
1509 // sanity check: all level counters above the coarse level must be 1
1510 for(std::size_t i(last_level); i > std::size_t(_top_level); )
1511 {
1512 --i;
1513 XASSERTM(this->_counters[i] == 1, "W-cycle sanity check failed: invalid counters");
1514 }
1515
1516 // prolongate (apply top-level post-smoother if it exists)
1517 this->_apply_prol(_top_level, true);
1518
1519 // okay
1520 return Status::success;
1521 }

1876 lvl_f.time_transfer += stamp_prol.elapsed_now();
1877 }
1878 else
1879 {
1880 // get coarse level
1881 LevelInfo& lvl_c = _hierarchy->_get_level_info(i+1);
1882
1883 // prolongate
1884 TimeStamp stamp_prol;
1885 transfer_operator->prol(lvl_f.vec_cor, lvl_c.vec_sol);
1886 lvl_f.time_transfer += stamp_prol.elapsed_now();
1887 }
1888
1889 // get system matrix and filters
1890 const MatrixType& system_matrix = lvl_f.level->get_system_matrix();
1891 const FilterType& system_filter_f = lvl_f.level->get_system_filter();
1892
1893 // apply correction filter
1894 system_filter_f.filter_cor(lvl_f.vec_cor);
1895
1896 // initialize omega for (adaptive) coarse grid correction
1897 DataType omega_cgc = DataType(1);
1898
1899 // use some sort of adaptive cgc?
1900 if(_adapt_cgc != MultiGridAdaptCGC::Fixed)
1901 {
1902 // multiply coarse grid correction by system matrix
1903 TimeStamp stamp_defect;
1904 system_matrix.apply(lvl_f.vec_tmp, lvl_f.vec_cor);
1905 lvl_f.time_defect += stamp_defect.elapsed_now();
1906
1907 // apply filter
1908 system_filter_f.filter_def(lvl_f.vec_tmp);
1909
1910 // compute adaptive omega
1911 switch(_adapt_cgc)
1912 {
1913 case MultiGridAdaptCGC::MinEnergy:
1914 omega_cgc = lvl_f.vec_def.dot(lvl_f.vec_cor)
1915 / lvl_f.vec_tmp.dot(lvl_f.vec_cor);
1916 break;
1917
1918 case MultiGridAdaptCGC::MinDefect:
1919 omega_cgc = lvl_f.vec_def.dot(lvl_f.vec_tmp)
1920 / lvl_f.vec_tmp.dot(lvl_f.vec_tmp);
1921 break;
1922
1923 default:
1924 // This default block is required to make the GCC happy...
1925 break;
1926 }
1927 }
1928
1929 // update our solution vector
1930 lvl_f.vec_sol.axpy(lvl_f.vec_cor, lvl_f.vec_sol, omega_cgc);
1931
1932 // get our post-smoother
1933 std::shared_ptr<SolverType> smoother = lvl_f.level->get_smoother_post();
1934
1935 // apply post-smoother if we have one
1936 if(smoother && (cur_smooth || (i > cur_lvl)))
1937 {
1938 // compute new defect
1939 if(_adapt_cgc != MultiGridAdaptCGC::Fixed)
1940 {
1941 // We have used some sort of adaptive coarse grid correction.
1942 // In this case, the current solution vector is
1943 // sol_new = sol_old + omega * cor
1944 //
1945 // Moreover, we have the following vectors at hand:
1946 // 1) tmp = A*cor
1947 // 2) def_old = rhs - A*sol_old
1948 //
1949 // So we can apply the same "trick" as in the PCG method to update our
1950 // defect vector instead of recomputing it by a matrix-vector mult, i.e.:
1951 // def_new = rhs - A * sol_new
1952 // = rhs - A * (sol_old + omega * cor)
1953 // = rhs - A * sol_old - omega * A * cor
1954 // = def_old - omega * tmp
1955 lvl_f.vec_def.axpy(lvl_f.vec_tmp, lvl_f.vec_def, -omega_cgc);
1956 }
1957 else
1958 {
1959 // we have to recompute the defect by (rhs - A*sol_new)
1960 TimeStamp stamp_defect;
1961 system_matrix.apply(lvl_f.vec_def, lvl_f.vec_sol, lvl_f.vec_rhs, -DataType(1));
1962 lvl_f.time_defect += stamp_defect.elapsed_now();
1963
1964 // apply defect filter
1965 system_filter_f.filter_def(lvl_f.vec_def);
1966 }
1967
1968 // apply post-smoother
1969 Statistics::add_solver_expression(std::make_shared<ExpressionCallSmoother>(this->name(), smoother->name()));
1970 TimeStamp stamp_smooth;
1971 smoother->apply(lvl_f.vec_cor, lvl_f.vec_def);
1972 //if(!status_success(smoother->apply(lvl_f.vec_cor, lvl_f.vec_def)))
1973 //return Status::aborted;
1974 lvl_f.time_smooth += stamp_smooth.elapsed_now();
1975
1976 // update solution vector
1977 lvl_f.vec_sol.axpy(lvl_f.vec_cor, lvl_f.vec_sol);
1978 }
1979
1980 _toes.at((size_t)i) += at.elapsed_now();
1981 double mpi_exec_reduction_stop(Statistics::get_time_mpi_execute_reduction());
1982 double mpi_exec_blas2_stop(Statistics::get_time_mpi_execute_blas2());
1983 double mpi_exec_blas3_stop(Statistics::get_time_mpi_execute_blas3());
1984 double mpi_exec_collective_stop(Statistics::get_time_mpi_execute_collective());
1985 double mpi_wait_stop_reduction(Statistics::get_time_mpi_wait_reduction());
1986 double mpi_wait_stop_blas2(Statistics::get_time_mpi_wait_blas2());
1987 double mpi_wait_stop_blas3(Statistics::get_time_mpi_wait_blas3());
1988 double mpi_wait_stop_collective(Statistics::get_time_mpi_wait_collective());
1989 _mpi_execs_reduction.at((size_t)i) += mpi_exec_reduction_stop - mpi_exec_reduction_start;
1990 _mpi_execs_blas2.at((size_t)i) += mpi_exec_blas2_stop - mpi_exec_blas2_start;
1991 _mpi_execs_blas3.at((size_t)i) += mpi_exec_blas3_stop - mpi_exec_blas3_start;
1992 _mpi_execs_collective.at((size_t)i) += mpi_exec_collective_stop - mpi_exec_collective_start;
1993 _mpi_waits_reduction.at((size_t)i) += mpi_wait_stop_reduction - mpi_wait_start_reduction;
1994 _mpi_waits_blas2.at((size_t)i) += mpi_wait_stop_blas2 - mpi_wait_start_blas2;
1995 _mpi_waits_blas3.at((size_t)i) += mpi_wait_stop_blas3 - mpi_wait_start_blas3;
1996 _mpi_waits_collective.at((size_t)i) += mpi_wait_stop_collective - mpi_wait_start_collective;
1997
1998 // ascend to next level
1999 }
2000
2001 return Status::success;
2002 }
