192 enum class DSSBackend : int
193 {
194 none = 0x0000,
195 cudss = 0x0001,
196 mkldss = 0x0002,
197 superlu = 0x0004,
198 umfpack = 0x0008,
199 all = 0x000F
200 };
201
203 static constexpr inline DSSBackend operator|(DSSBackend a, DSSBackend b)
204 {
205 return static_cast<DSSBackend>(static_cast<int>(a) | static_cast<int>(b));
206 }
207
209 static constexpr inline DSSBackend operator&(DSSBackend a, DSSBackend b)
210 {
211 return static_cast<DSSBackend>(static_cast<int>(a) & static_cast<int>(b));
212 }
213
215 static constexpr inline bool operator*(DSSBackend a)
216 {
217 // just check the interesting bits
218 return (static_cast<int>(a) & static_cast<int>(DSSBackend::all)) != 0;
219 }
220
222 static std::ostream& operator<<(std::ostream& os, DSSBackend a)
223 {
224 int n = 0;
225 if(*(a & DSSBackend::cudss))
226 {
227 if(++n > 1)
228 os << '|';
229 os << "cudss";
230 }
231 if(*(a & DSSBackend::mkldss))
232 {
233 if(++n > 1)
234 os << '|';
235 os << "mkldss";
236 }
237 if(*(a & DSSBackend::superlu))
238 {
239 if(++n > 1)
240 os << '|';
241 os << "superlu";
242 }
243 if(*(a & DSSBackend::umfpack))
244 {
245 if(++n > 1)
246 os << '|';
247 os << "umfpack";
248 }
249 if(n <= 0)
250 os << "none";
251 return os;
252 }

501 static bool is_available(const MatrixType& matrix, const FilterType& DOXY(filter))
502 {
503 // For purely local matrices, everything except for SuperLU_DIST is applicable
504 if constexpr (MatrixType::is_local)
505 return have_backend_local;
506
507 if constexpr (MatrixType::is_global)
508 {
509 // potentially parallel case: check if we have a comm with more than 1 process
510 const Dist::Comm* comm = matrix.get_comm();
511 if((comm == nullptr) || (comm->size() <= 1))
512 return have_backend_local;
513
514 // MPI-parallel case with more than 1 process
515 return have_backend_global;
516 }
517
518 // we should never come out here
519 XABORTM("matrix type is neither local nor global!");
520 return false;
521 }

561 void push_allowed_backend(const String& backend)
562 {
563 DSSBackend b = DSSBackend::none;
564 std::deque<String> sbacks = backend.split_by_charset("|");
565 for(const String& s : sbacks)
566 {
567 if(s.compare_no_case("none") == 0)
568 b = b | DSSBackend::none;
569 else if(s.compare_no_case("all") == 0)
570 b = b | DSSBackend::all;
571 else if(s.compare_no_case("cudss") == 0)
572 b = b | DSSBackend::cudss;
573 else if(s.compare_no_case("mkldss") == 0)
574 b = b | DSSBackend::mkldss;
575 else if(s.compare_no_case("mkl-dss") == 0)
576 b = b | DSSBackend::mkldss;
577 else if(s.compare_no_case("superlu") == 0)
578 b = b | DSSBackend::superlu;
579 else if(s.compare_no_case("umfpack") == 0)
580 b = b | DSSBackend::umfpack;
581 else
582 throw DirectSparseSolverException("DirectSparseSolver: unknown DSS backend: '" + backend + "'");
583 }
584 push_allowed_backend(b);
585 }

640 virtual void init_symbolic() override
641 {
642 BaseClass::init_symbolic();
643
644 // First of all, let's see if we're in a single-process scenario here
645 const Dist::Comm* comm = this->_get_comm();
646 const bool single_process = ((comm == nullptr) || (comm->size() <= 1));
647
648 // if the user didn't specify any allowed backends, then we'll try a default setting
649 if(_allowed_backends.empty())
650 {
651 // First, let's see if the user wanted a cuda-based solver
652 if(_preferred_backend == PreferredBackend::cuda)
653 {
654 // Ok, try to create a cuDSS core then
655 if(!_create_core_cudss())
656 {
657 // Nope, apparently, cuDSS is not available
658 throw DSSBackendNotFoundException(DSSBackend::cudss);
659 }
660 }
661
662 // Next, let's see if the user wanted a MKL-based solver
663 if(_preferred_backend == PreferredBackend::mkl)
664 {
665 // Ok, try to create a MKL-DSS core then
666 if(!_create_core_mkldss())
667 {
668 // Nope, apparently, MKL-DSS is not available
669 throw DSSBackendNotFoundException(DSSBackend::mkldss);
670 }
671 }
672
673 // Now we're left with the generic case, in which we may choose any backend we see fit
674
675 // Let's try cuDSS first
676 if(_create_core_cudss())
677 return;
678
679 // Let's try MKL-DSS next
680 if(_create_core_mkldss())
681 return;
682
683 // Let's try UMFPACK if we're in a single-process case next
684 if(single_process && _create_core_umfpack())
685 return;
686
687 // Let's try SuperLU last
688 if(_create_core_superlu())
689 return;
690
691 // If we come out here, then we have failed to find a suitable backend...
692 throw DSSBackendNotFoundException();
693 }
694
695 // Let's loop over all allowed backends
696 for(auto ab : _allowed_backends)
697 {
698 // Let's try cuDSS first (if allowed)
699 if(*(ab & DSSBackend::cudss) && _create_core_cudss())
700 return;
701
702 // Let's try MKL-DSS next (if allowed)
703 if(*(ab & DSSBackend::mkldss) && _create_core_mkldss())
704 return;
705
706 // Let's try UMFPACK if we're in a single-process case next (if allowed)
707 if(*(ab & DSSBackend::umfpack) && single_process && _create_core_umfpack())
708 return;
709
710 // Let's try SuperLU last (if allowed)
711 if(*(ab & DSSBackend::superlu) && _create_core_superlu())
712 return;
713 }
714
715 // If we come out here, then we have failed to find a suitable backend...
716 throw DSSBackendNotFoundException();
717 }

839 virtual Status apply(VectorType& vec_cor, const VectorType& vec_def) override
840 {
841 // silence compiler warnings about unused variables if no backend is available
842 (void)vec_cor;
843 (void)vec_def;
844
845#ifdef FEAT_HAVE_CUDSS
846 if(this->_core_cudss)
847 {
848 // upload defect vector
849 this->_upload_vector(DSS::get_cudss_rhs_val(this->_core_cudss), vec_def);
850
851 // solve system via MKL DSS
852 try
853 {
854 DSS::solve_cudss(this->_core_cudss);
855 }
856 catch(const DirectSparseSolverException&)
857 {
858 return Status::aborted;
859 }
860 catch(...)
861 {
862 throw;
863 }
864
865 // download correction vector
866 this->_download_vector(vec_cor, DSS::get_cudss_sol_val(this->_core_cudss));
867
868 return Status::success;
869 }
870#endif // FEAT_HAVE_CUDSS
871#ifdef FEAT_HAVE_MKL
872 if(this->_core_mkldss)
873 {
874 // upload defect vector
875 this->_upload_vector(DSS::get_mkldss_rhs_val(this->_core_mkldss), vec_def);
876
877 // solve system via MKL DSS
878 try
879 {
880 DSS::solve_mkldss(this->_core_mkldss);
881 }
882 catch(const DirectSparseSolverException&)
883 {
884 return Status::aborted;
885 }
886 catch(...)
887 {
888 throw;
889 }
890
891 // download correction vector
892 this->_download_vector(vec_cor, DSS::get_mkldss_sol_val(this->_core_mkldss));
893
894 return Status::success;
895 }
896#endif // FEAT_HAVE_MKL
897#ifdef FEAT_HAVE_SUPERLU_DIST
898 if(this->_core_superlu)
899 {
900 // upload defect vector
901 this->_upload_vector(DSS::get_superlu_vector(this->_core_superlu), vec_def);

903 // solve system via SuperLU

914 throw;
915 }

919
920 return Status::success;
921 }
922#endif // FEAT_HAVE_SUPERLU_DIST
923#ifdef FEAT_HAVE_UMFPACK
