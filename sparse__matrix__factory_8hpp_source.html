123 SparseMatrixCSR<DT_, IT_> make_csr() const
124 {
125 //Creates CSR matrix with dimensions and number of NNZ of map.
126 FEAT::LAFEM::SparseMatrixCSR<DT_, IT_ > matrix(_num_row, _num_col, Index( memory.size()));
127 // pointer on the empty arrays of the CSR matrix structure
128 IT_* row_ptr = matrix.row_ptr();
129 IT_* col_ind = matrix.col_ind();
130 DT_* val = matrix.val();
131
132 row_ptr[0] = IT_(0);
133 // numberNotZero elements
134 IT_ _nnz = IT_(0);
135 // current row count
136 IT_ _row_cur = IT_(0);
137
138 //Iterates over the map to extract the data and paste in the corresponding arrays.
139 //typename std::map< IT_, DT_>::const_iterator it = memory.begin();
140 for(auto it = memory.begin(); it != memory.end();++it)
141 {
142 // extract column and row from the map key
143 IT_ col = it->first % max_size;
144 IT_ row = (it->first - col) / max_size;
145 // make sure the row pointer is valid even if there exist preceeding empty rows in the matrix
146 while (_row_cur <row )
147 {
148 row_ptr[++_row_cur] = _nnz;
149 }
150 val[_nnz] = it->second;
151 col_ind[_nnz] = col;
152 ++_nnz;
153 }
154 // make sure that the row pointer is valid even if there are empty rows at the end of the matrix
155 while (_row_cur < _num_row)
156 {
157 row_ptr[++_row_cur] = _nnz;
158 }
