345 template<
346 typename Vector_,
347 typename Matrix_,
348 typename Filter_>
349 inline Status solve(
350 SolverBase<Vector_>& solver,
351 Vector_& vec_sol,
352 const Vector_& vec_rhs,
353 const Matrix_& matrix,
354 const Filter_& filter)
355 {
356 typedef typename Vector_::DataType DataType;
357
358 // create two temporary vectors
359 Vector_ vec_def(vec_rhs.clone(LAFEM::CloneMode::Layout));
360 Vector_ vec_cor(vec_sol.clone(LAFEM::CloneMode::Layout));
361
362 // compute defect
363 matrix.apply(vec_def, vec_sol, vec_rhs, -DataType(1));
364
365 // apply defect filter
366 filter.filter_def(vec_def);
367
368 // apply solver
369 Status status = solver.apply(vec_cor, vec_def);
370 if(status_success(status))
371 {
372 // apply correction filter
373 filter.filter_cor(vec_cor);
374
375 // update solution
376 vec_sol.axpy(vec_sol, vec_cor);
377 }
378
379 // return status
380 return status;
381 }
