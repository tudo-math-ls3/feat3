97 template<typename DataType_, typename IndexType_, typename Space_>
98 void assemble(LAFEM::UnitFilter<DataType_, IndexType_>& filter, const Space_& space) const
99 {
100 // build index set
101 std::set<Index> idx_set;
102 Intern::UnitAsmWrapper<shape_dim>::assemble(idx_set, space, _cells);
103
104 // allocate filter if necessary
105 if(filter.size() == Index(0))
106 {
107 filter = LAFEM::UnitFilter<DataType_, IndexType_>(space.get_num_dofs());
108 }
109
110 // loop over all dof-indices
111 typename std::set<Index>::const_iterator it(idx_set.begin());
112 typename std::set<Index>::const_iterator jt(idx_set.end());
113 for(Index i(0); it != jt; ++it, ++i)
114 {
115 // store the dof-index
116 filter.add(IndexType_(*it), DataType_(0));
117 }
118 }

264 void assemble(
265 LAFEM::UnitFilterBlocked<DataType_, IndexType_, BlockSize_>& filter,
266 const Space_& space,
267 const Function_& function) const
268 {
269 // ensure that the function is a vector field of correct dimension
270 typedef typename Function_::ImageType FuncImageType;
271 static_assert(FuncImageType::is_vector, "only vector fields are supported");
272 static_assert(FuncImageType::image_dim == BlockSize_, "invalid filter block size");
273
274 // get the value type of the function

276
277 // build index-value map

279 Intern::UnitAsmWrapper<shape_dim>::assemble(idx_map, space, _cells, function);
280
281 // allocate filter if necessary

283 {
284 filter = LAFEM::UnitFilterBlocked<DataType_, IndexType_, BlockSize_>(space.get_num_dofs());
285 }
286
287 // loop over all dof-indices
288 auto it(idx_map.begin());
289 auto jt(idx_map.end());
290 for(Index i(0); it != jt; ++it, ++i)

344 struct UnitAsmHelper
345 {
346 template<typename MeshPart_>
347 static void merge(std::set<Index>& idx, const MeshPart_& mesh_part)
348 {
349 // fetch the target set for this dimension
350 const typename MeshPart_::template TargetSet<shape_dim_>::Type&
351 target_set(mesh_part.template get_target_set<shape_dim_>());
352
353 // merge
354 const Index num_entities = target_set.get_num_entities();
355 for(Index i(0); i < num_entities; ++i)
356 {
357 idx.insert(target_set[i]);
358 }
359 }

362 template<typename Space_>
363 static void assemble(std::set<Index>& idx, const Space_& space, const std::set<Index>& cells)
364 {
365 // create a dof-assignment object
366 typename Space_::template DofAssignment<shape_dim_>::Type dof_assign(space);
367
368 // loop over all target indices
369 std::set<Index>::const_iterator it(cells.begin());
370 std::set<Index>::const_iterator jt(cells.end());
371 for(; it != jt; ++it)
372 {
373 dof_assign.prepare(*it);
374 const int num_assign(dof_assign.get_num_assigned_dofs());
375 for(int j(0); j < num_assign; ++j)
376 {
377 idx.insert(dof_assign.get_index(j));
378 }
379 dof_assign.finish();
380 }
381 }

384 template<
385 typename DataType_,
386 typename Space_,
387 typename Function_>
388 static void assemble(
389 std::map<Index, DataType_>& idx,
390 const Space_& space,
391 const std::set<Index>& cells,
392 const Function_& function)
393 {
394 // create a node-functional object
395 typedef typename Space_::template NodeFunctional<shape_dim_, DataType_>::Type NodeFunc;
396
397 // check for empty node functional set
398 static constexpr Index max_dofs = NodeFunc::max_assigned_dofs;
399 if(max_dofs <= 0)
400 return;
401
402 // create node functional
403 NodeFunc node_func(space);
404
405 // create a dof-assignment object
406 typename Space_::template DofAssignment<shape_dim_, DataType_>::Type dof_assign(space);
407
408 // create node data; avoid zero-length vectors
409 Tiny::Vector<DataType_, max_dofs+1> node_data;
410
411 // loop over all target indices
412 std::set<Index>::const_iterator it(cells.begin());
413 std::set<Index>::const_iterator jt(cells.end());
414 for(; it != jt; ++it)
415 {
416 node_func.prepare(*it);
417 node_func(node_data, function);
418 node_func.finish();
419
420 dof_assign.prepare(*it);
421
422 const int num_assign(dof_assign.get_num_assigned_dofs());
423 for(int j(0); j < num_assign; ++j)
424 {
425 idx.insert(std::make_pair(dof_assign.get_index(j), node_data[j]));
426 }
427 dof_assign.finish();
428 }
429 }
