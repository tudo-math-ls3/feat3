239 void init_bounding_boxes(DataType bbox_tol)
240 {
241 // get the mesh
242 const auto& mesh = _trafo.get_mesh();
243 // get the vertex set
244 const auto& vtx_set = mesh.get_vertex_set();
245 // get the vertices-at-cell index set
246 const auto& vert_idx = mesh.template get_index_set<shape_dim, 0>();
247
248 // resize bounding boxes array
249 _bboxes.resize(std::size_t(mesh.get_num_elements()));
250
251 // loop over all cells
252 for(Index cell(0); cell < vert_idx.get_num_entities(); ++cell)
253 {
254 // get this cell's vertex indices
255 const auto& vidx = vert_idx[cell];
256
257 // get this cell's bounding box
258 BBox& bbox = _bboxes.at(std::size_t(cell));
259
260 // build cell bounding box
261 bbox[0] = bbox[1] = vtx_set[vidx[0]];
262 for(int i(1); i < vert_idx.get_num_indices(); ++i)
263 {
264 // get vertex and update bounding box
265 const auto& vtx = vtx_set[vidx[i]];
266 for(int j(0); j < world_dim; ++j)
267 {
268 Math::minimax(DataType(vtx[j]), bbox[0][j], bbox[1][j]);
269 }
270 }
271
272 // finally, incorporate tolerances
273 for(int j(0); j < world_dim; ++j)
274 {
275 // compute bbox dimension
276 DataType bb_extra = bbox_tol * (bbox[1][j] - bbox[0][j]);
277 bbox[0][j] -= bb_extra;
278 bbox[1][j] += bb_extra;
279 }
280 }
281 }

319 InvMapDataType unmap_point(const ImagePointType& img_point, bool ignore_failures = false) const
320 {
321 // store image point
322 InvMapDataType inv_data;
323 inv_data.img_point = img_point;
324
325 // find candidate cells
326 std::vector<Index> cells;
327 if(!this->find_candidate_cells(cells, img_point))
328 return inv_data;
329
330 // loop over all candidate cells
331 for(Index cell : cells)
332 {
333 // try to unmap the point by newton
334 DomainPointType dom_point;
335 if(!this->unmap_point_by_newton(dom_point, img_point, cell))
336 {
337 // point unmapping failed!
338 if(ignore_failures)
339 continue;
340
341 // throw an inverse mapping error
342 throw InverseMappingError(dom_point, cell);
343 }
344
345 // check if the domain point is on the reference element
346 if(!this->test_domain_point(dom_point))
347 {
348 // nope ==> skip point
349 continue;
350 }
351
352 // okay
353 inv_data.cells.push_back(cell);
354 inv_data.dom_points.push_back(dom_point);
355 }
356
357 // found at least one cell?
358 return inv_data;
359 }

464 bool find_candidate_cells(std::vector<Index>& cells, const ImagePointType& img_point) const
465 {
466 // loop over all cells/bounding boxes
467 for(std::size_t cell(0); cell < _bboxes.size(); ++cell)
468 {
469 const BBox& bbox = _bboxes.at(cell);
470
471 // check whether the point is in the bounding box
472 bool is_in = true;
473 for(int j(0); j < shape_dim; ++j)
474 {
475 if((img_point[j] < bbox[0][j]) || (img_point[j] > bbox[1][j]))
476 {
477 is_in = false;
478 break;
479 }
480 }
481
482 // point inside bounding box?
483 if(is_in)
484 cells.push_back(Index(cell));
485 }
486
487 return !cells.empty();
488 }

506 bool unmap_point_by_newton(DomainPointType& dom_point, const ImagePointType& img_point, const Index cell) const
507 {
508 // define an evaluator
509 typedef typename TrafoType::template Evaluator<ShapeType, DataType>::Type TrafoEvaluator;
510 TrafoEvaluator trafo_eval(_trafo);
511
512 // for Newton iteration we need image points and inverse jacobians
513 static constexpr TrafoTags trafo_tags = TrafoTags::img_point|TrafoTags::jac_inv;
514
515 // define the trafo eval data
516 typename TrafoEvaluator::template ConfigTraits<trafo_tags>::EvalDataType trafo_data;
517
518 // initialize output domain point to cell centre
519 for(int i(0); i < shape_dim; ++i)
520 dom_point[i] = Shape::ReferenceCell<ShapeType>::template centre<DataType>(i);
521
522 // compute squared tolerance
523 const DataType tol_sqr = Math::sqr(_newton_tol);
524
525 // no convergence yet
526 bool converged = false;
527
528 // prepare trafo evaluator
529 trafo_eval.prepare(cell);
530
531 // newton iteration
532 for(Index iter(0); iter < _newton_max_iter; ++iter)
533 {
534 // evaluate trafo in current point
535 trafo_eval(trafo_data, dom_point);
536
537 // compute image point distance vector
538 ImagePointType def = trafo_data.img_point - img_point;
539
540 // check defect for tolerance
541 if(def.norm_euclid_sqr() < tol_sqr)
542 {
543 converged = true;
544 break;
545 }
546
547 // update reference point by newton
548 dom_point -= trafo_data.jac_inv * def;
549 }
550
551 // finish trafo evaluator
552 trafo_eval.finish();
553
554 // Newton converged?
555 return converged;
556 }

576 template<int dim_>
577 struct InverseMappingHelper<Shape::Simplex<dim_>>
578 {
579 // checks whether p is on the reference cell
580 template<typename PT_, typename DT_>
581 static bool is_on_ref(const PT_& p, const DT_ tol)
582 {
583 // reference simplex: all coords are in range [0,1]
584 // and the sum of all coords is <= 1
585 DT_ s = DT_(0);
586 for(int i(0); i < dim_; ++i)
587 {
588 // coord outside range [-tol, 1+tol] ?
589 if((p[i] < -tol) || (p[i] > DT_(1)+tol))
590 return false;
591 s += p[i];
592 }
593 // coord sum <= 1+tol ?
594 return (s <= DT_(1) + tol);
595 }
596 };
597
598 template<int dim_>
599 struct InverseMappingHelper<Shape::Hypercube<dim_>>
600 {
601 // checks whether p is on the reference cell
602 template<typename PT_, typename DT_>
603 static bool is_on_ref(const PT_& p, const DT_ tol)
604 {
605 // reference hypercube: all coordinates are in range [-1,+1]
606 for(int i(0); i < dim_; ++i)
607 {
608 // coord outside range [-1-tol, +1+tol] ?
609 if((p[i] < -DT_(1)-tol) || (p[i] > DT_(1)+tol))
610 return false;
611 }
612 return true;
613 }
