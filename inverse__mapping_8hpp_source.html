239 void init_bounding_boxes(DataType bbox_tol)
240 {
241 // get the mesh
242 const auto& mesh = _trafo.get_mesh();
243 // get the vertex set
244 const auto& vtx_set = mesh.get_vertex_set();
245 // get the vertices-at-cell index set
246 const auto& vert_idx = mesh.template get_index_set<shape_dim, 0>();
247
248 // resize bounding boxes array
249 _bboxes.resize(std::size_t(mesh.get_num_elements()));
250
251 // loop over all cells
252 for(Index cell(0); cell < vert_idx.get_num_entities(); ++cell)
253 {
254 // get this cell's vertex indices
255 const auto& vidx = vert_idx[cell];
256
257 // get this cell's bounding box
258 BBox& bbox = _bboxes.at(std::size_t(cell));
259
260 // build cell bounding box
261 bbox[0] = bbox[1] = vtx_set[vidx[0]];
262 for(int i(1); i < vert_idx.get_num_indices(); ++i)
263 {
264 // get vertex and update bounding box
265 const auto& vtx = vtx_set[vidx[i]];
266 for(int j(0); j < world_dim; ++j)
267 {
268 Math::minimax(DataType(vtx[j]), bbox[0][j], bbox[1][j]);
269 }
270 }
271
272 // finally, incorporate tolerances
273 for(int j(0); j < world_dim; ++j)
274 {
275 // compute bbox dimension
276 DataType bb_extra = bbox_tol * (bbox[1][j] - bbox[0][j]);
277 bbox[0][j] -= bb_extra;
278 bbox[1][j] += bb_extra;
279 }
280 }
281 }

319 InvMapDataType unmap_point(const ImagePointType& img_point, bool ignore_failures = false) const
320 {
321 // store image point
322 InvMapDataType inv_data;
323 inv_data.img_point = img_point;
324
325 // find candidate cells
326 std::vector<Index> cells;
327 if(!this->find_candidate_cells(cells, img_point))
328 return inv_data;
329
330 // loop over all candidate cells
331 for(Index cell : cells)
332 {
333 // try to unmap the point by newton
334 DomainPointType dom_point;
335 if(!this->unmap_point_by_newton(dom_point, img_point, cell))
336 {
337 // point unmapping failed!
338 if(ignore_failures)
339 continue;
340
341 // throw an inverse mapping error
342 throw InverseMappingError(dom_point, cell);
343 }
344
345 // check if the domain point is on the reference element
346 if(!this->test_domain_point(dom_point))
347 {
348 // nope ==> skip point
349 continue;
350 }
351
352 // okay
353 inv_data.cells.push_back(cell);
354 inv_data.dom_points.push_back(dom_point);
355 }
356
357 // found at least one cell?
358 return inv_data;
359 }

465 bool find_candidate_cells(std::vector<Index>& cells, const ImagePointType& img_point) const
466 {
467 // loop over all cells/bounding boxes
468 for(std::size_t cell(0); cell < _bboxes.size(); ++cell)
469 {
470 const BBox& bbox = _bboxes.at(cell);
471
472 // check whether the point is in the bounding box
473 bool is_in = true;
474 for(int j(0); j < shape_dim; ++j)
475 {
476 if((img_point[j] < bbox[0][j]) || (img_point[j] > bbox[1][j]))
477 {
478 is_in = false;
479 break;
480 }
481 }
482
483 // point inside bounding box?
484 if(is_in)
485 cells.push_back(Index(cell));
486 }
487
488 return !cells.empty();
489 }

507 bool unmap_point_by_newton(DomainPointType& dom_point, const ImagePointType& img_point, const Index cell) const
508 {
509 // define an evaluator
510 typedef typename TrafoType::template Evaluator<ShapeType, DataType>::Type TrafoEvaluator;
511 TrafoEvaluator trafo_eval(_trafo);
512
513 // for Newton iteration we need image points and inverse jacobians
514 static constexpr TrafoTags trafo_tags = TrafoTags::img_point|TrafoTags::jac_inv;
515
516 // define the trafo eval data
517 typename TrafoEvaluator::template ConfigTraits<trafo_tags>::EvalDataType trafo_data;
518
519 // initialize output domain point to cell centre
520 for(int i(0); i < shape_dim; ++i)
521 dom_point[i] = Shape::ReferenceCell<ShapeType>::template centre<DataType>(i);
522
523 // compute squared tolerance
524 const DataType tol_sqr = Math::sqr(_newton_tol);
525
526 // no convergence yet
527 bool converged = false;
528
529 // prepare trafo evaluator
530 trafo_eval.prepare(cell);
531
532 // newton iteration
533 for(Index iter(0); iter < _newton_max_iter; ++iter)
534 {
535 // evaluate trafo in current point
536 trafo_eval(trafo_data, dom_point);
537
538 // compute image point distance vector
539 ImagePointType def = trafo_data.img_point - img_point;
540
541 // check defect for tolerance
542 if(def.norm_euclid_sqr() < tol_sqr)
543 {
544 converged = true;
545 break;
546 }
547
548 // update reference point by newton
549 dom_point -= trafo_data.jac_inv * def;
550 }
551
552 // finish trafo evaluator
553 trafo_eval.finish();
554
555 // Newton converged?
556 return converged;
557 }

577 template<int dim_>
578 struct InverseMappingHelper<Shape::Simplex<dim_>>
579 {
580 // checks whether p is on the reference cell
581 template<typename PT_, typename DT_>
582 static bool is_on_ref(const PT_& p, const DT_ tol)
583 {
584 // reference simplex: all coords are in range [0,1]
585 // and the sum of all coords is <= 1
586 DT_ s = DT_(0);
587 for(int i(0); i < dim_; ++i)
588 {
589 // coord outside range [-tol, 1+tol] ?
590 if((p[i] < -tol) || (p[i] > DT_(1)+tol))
591 return false;
592 s += p[i];
593 }
594 // coord sum <= 1+tol ?
595 return (s <= DT_(1) + tol);
596 }
597 };
598
599 template<int dim_>
600 struct InverseMappingHelper<Shape::Hypercube<dim_>>
601 {
602 // checks whether p is on the reference cell
603 template<typename PT_, typename DT_>
604 static bool is_on_ref(const PT_& p, const DT_ tol)
605 {
606 // reference hypercube: all coordinates are in range [-1,+1]
607 for(int i(0); i < dim_; ++i)
608 {
609 // coord outside range [-1-tol, +1+tol] ?
610 if((p[i] < -DT_(1)-tol) || (p[i] > DT_(1)+tol))
611 return false;
612 }
613 return true;
614 }
