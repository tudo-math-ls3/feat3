280 template<typename Space_>
281 static Adjacency::Graph assemble_graph_diag(const Space_& space)
282 {
283 // get number of DOFs
284 const Index n = space.get_num_dofs();
285
286 // create a graph
287 Adjacency::Graph graph(n, n, n);
288
289 // get and fill the arrays
290 Index* dom_ptr = graph.get_domain_ptr();
291 Index* img_idx = graph.get_image_idx();
292 for(Index i(0); i < n; ++i)
293 dom_ptr[i] = img_idx[i] = i;
294 dom_ptr[n] = n;
295
296 // return the graph
297 return graph;
298 }

317 template<typename TestSpace_, typename TrialSpace_, typename TrialToTestAdjator_>
318 static Adjacency::Graph assemble_graph_intermesh(
319 const TestSpace_& test_space,
320 const TrialSpace_& trial_space,
321 const TrialToTestAdjator_& trial2test_adjactor)
322 {
323 // create test- and trial-dof-mappers
324 Adjacency::Graph test_dof_mapping(Space::DofMappingRenderer::render(test_space));
325 Adjacency::Graph trial_dof_mapping(Space::DofMappingRenderer::render(trial_space));
326
327 // get trial and test mesh permutations
328 const Adjacency::Permutation& trial_perm = trial_space.get_trafo().get_mesh().get_mesh_permutation().get_perm();
329 const Adjacency::Permutation& test_inv_perm = test_space.get_trafo().get_mesh().get_mesh_permutation().get_inv_perm();
330 if(!trial_perm.empty() || !test_inv_perm.empty())
331 {
332 // render trial2test adjactor
333 Adjacency::Graph trial2test_graph(Adjacency::RenderType::as_is, trial2test_adjactor);
334
335 // permute graph indices
336 Adjacency::Graph permuted_graph;
337 if(trial_perm.empty())
338 {
339 // no trial permutation, only test mesh permuted
340 permuted_graph = std::move(trial2test_graph);
341 permuted_graph.permute_indices(test_inv_perm);
342 }
343 else if(test_inv_perm.empty())
344 {
345 // no test permutation, only trial mesh permuted
346 // create a dummy identity permutation for the test mesh
347 Adjacency::Permutation id_perm(trial2test_graph.get_num_nodes_image(), Adjacency::Permutation::type_identity);
348 permuted_graph = Adjacency::Graph(trial2test_graph, trial_perm, id_perm);
349 }
350 else
351 {
352 // both trial and test permutations exist
353 permuted_graph = Adjacency::Graph(trial2test_graph, trial_perm, test_inv_perm);
354 }
355
356 // render transposed test-dof-mapping
357 Adjacency::Graph test_dof_support(Adjacency::RenderType::injectify_transpose, permuted_graph, test_dof_mapping);
358
359 // render composite test-dof-mapping/trial-dof-support graph
360 Adjacency::Graph dof_adjactor(Adjacency::RenderType::injectify_sorted, test_dof_support, trial_dof_mapping);
361
362 // return the graph
363 return dof_adjactor;
364 }
365 else // no permutation
366 {
367 // render transposed test-dof-mapping
368 Adjacency::Graph test_dof_support(Adjacency::RenderType::injectify_transpose, trial2test_adjactor, test_dof_mapping);
369
370 // render composite test-dof-mapping/trial-dof-support graph
371 Adjacency::Graph dof_adjactor(Adjacency::RenderType::injectify_sorted, test_dof_support, trial_dof_mapping);
372
373 // return the graph
374 return dof_adjactor;
375 }
