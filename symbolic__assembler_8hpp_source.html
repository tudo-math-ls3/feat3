280 template<typename Space_>
281 static Adjacency::Graph assemble_graph_diag(const Space_& space)
282 {
283 // get number of DOFs
284 const Index n = space.get_num_dofs();
285
286 // create a graph
287 Adjacency::Graph graph(n, n, n);
288
289 // get and fill the arrays
290 Index* dom_ptr = graph.get_domain_ptr();
291 Index* img_idx = graph.get_image_idx();
292 for(Index i(0); i < n; ++i)
293 dom_ptr[i] = img_idx[i] = i;
294 dom_ptr[n] = n;
295
296 // return the graph
297 return graph;
298 }

317 template<typename FineTestSpace_, typename CoarseTrialSpace_>
318 static Adjacency::Graph assemble_graph_2lvl(
319 const FineTestSpace_& fine_space,
320 const CoarseTrialSpace_& coarse_space)
321 {
322 // create test- and trial-dof-mappers
323 Adjacency::Graph test_dof_mapping(Space::DofMappingRenderer::render(fine_space));
324 Adjacency::Graph trial_dof_mapping(Space::DofMappingRenderer::render(coarse_space));
325
326 // create an refinement adjactor
327 Geometry::Intern::CoarseFineCellMapping<typename FineTestSpace_::MeshType, typename CoarseTrialSpace_::MeshType>
328 refine_adjactor(fine_space.get_trafo().get_mesh(), coarse_space.get_trafo().get_mesh());
329
330 // get coarse and fine mesh permutations
331 const Adjacency::Permutation& coarse_perm =
332 coarse_space.get_trafo().get_mesh().get_mesh_permutation().get_perm();
333 const Adjacency::Permutation& fine_inv_perm =
334 fine_space.get_trafo().get_mesh().get_mesh_permutation().get_inv_perm();
335 if(!coarse_perm.empty() || !fine_inv_perm.empty())
336 {
337 // render refinement adjactor
338 Adjacency::Graph refine_graph(Adjacency::RenderType::as_is, refine_adjactor);
339
340 // permute refinement graph indices
341 Adjacency::Graph permuted_graph;
342 if(coarse_perm.empty())
343 {
344 // no coarse permutation, only fine mesh permuted
345 permuted_graph = std::move(refine_graph);
346 permuted_graph.permute_indices(fine_inv_perm);
347 }
348 else if(fine_inv_perm.empty())
349 {
350 // no fine permutation, only coarse mesh permuted
351 // create a dummy identity permutation for the fine mesh
352 Adjacency::Permutation id_perm(refine_graph.get_num_nodes_image(), Adjacency::Permutation::type_identity);
353 permuted_graph = Adjacency::Graph(refine_graph, coarse_perm, id_perm);
354 }
355 else
356 {
357 // both coarse and fine permutations exist
358 permuted_graph = Adjacency::Graph(refine_graph, coarse_perm, fine_inv_perm);
359 }
360
361 // render transposed test-dof-mapping
362 Adjacency::Graph test_dof_support(Adjacency::RenderType::injectify_transpose, permuted_graph, test_dof_mapping);
363
364 // render composite test-dof-mapping/trial-dof-support graph
365 Adjacency::Graph dof_adjactor(Adjacency::RenderType::injectify_sorted, test_dof_support, trial_dof_mapping);
366
367 // return the graph
368 return dof_adjactor;
369 }
370 else // no permutation
371 {
372 // render transposed test-dof-mapping
373 Adjacency::Graph test_dof_support(Adjacency::RenderType::injectify_transpose, refine_adjactor, test_dof_mapping);
374
375 // render composite test-dof-mapping/trial-dof-support graph
376 Adjacency::Graph dof_adjactor(Adjacency::RenderType::injectify_sorted, test_dof_support, trial_dof_mapping);
377
378 // return the graph
379 return dof_adjactor;
380 }
