199 template<typename Tx_>
200 void apply(Tx_* x, bool invert = false) const
201 {
202 XASSERT(x != nullptr);
203
204 if(!invert)
205 {
206 // apply in-situ swapping
207 for(Index i(0); i+1 < this->size(); ++i)
208 {
209 Index j(_swap_pos[i]);
210 ASSERTM((j >= i) && (j < this->size()), "invalid swap position");
211 if(j > i)
212 {
213 Tx_ t(x[i]);
214 x[i] = x[j];
215 x[j] = t;
216 }
217 }
218 }
219 else
220 {
221 // apply inverse in-situ swapping
222 for(Index i(this->size() -1); i > 0; --i)
223 {
224 Index j(_swap_pos[i-1]);
225 ASSERTM((j >= i-1) && (j < this->size()), "invalid swap position");
226 if(j > i-1)
227 {
228 Tx_ t(x[i-1]);
229 x[i-1] = x[j];
230 x[j] = t;
231 }
232 }
233 }
234 }

269 y[_perm_pos[i]] = Ty_(x[i]);
