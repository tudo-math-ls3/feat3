35 static inline size_t extract_type_size(uint64_t feature_hash)
36 {
37 return (size_t) feature_hash&0xFFFFFFFF; //take only the lower 32 bits
38 }
39
41 static inline bool extract_intness(uint64_t feature_hash)
42 {
43 return (feature_hash&(uint64_t(1)<<32)) != uint64_t(0);
44 }
45
47 static inline bool extract_floatness(uint64_t feature_hash)
48 {
49 return (feature_hash&(uint64_t(1)<<33)) != uint64_t(0);
50 }
51
53 static inline bool extract_signedness(uint64_t feature_hash)
54 {

56 }
