360 CUDA_HOST_DEVICE static inline DataType width_directed(const ImagePointType& ray, const Tiny::Matrix<DataType, image_dim, num_verts>& coeffs)
361 {
362 // This one is a little tricky:
363 // We follow a similar approach as on quadrilaterals here, i.e. we first map the ray
364 // vector onto the reference cells by multiplying it to the inverse jacobian matrix.
365 // However, our reference cell is the "right triangle" spanned by the three points
366 // (0,0) (1,0) (0,1) and therefore its edges have different lengths, which would
367 // result in a "skew" width. To circumvent this, we will afterwards map the ray
368 // vector onto a "equilateral unit triangle" (i.e. the triangle where all edges
369 // have length = 1) and we compute the norm of that mapped vector then.
370 JacobianMatrixType jac_mat;
371 JacobianInverseType jac_inv;
372 DomainPointType ref_ray;
373
374 // compute jacobian matrix
375 for(int i(0); i < image_dim; ++i)
376 {
377 jac_mat(i,0) = coeffs[i][1];
378 jac_mat(i,1) = coeffs[i][2];
379 }
380
381 // invert jacobian matrix and multiply by ray vector
382 jac_inv.set_inverse(jac_mat);
383 ref_ray.set_mat_vec_mult(jac_inv, ray);
384
385 // We have mapped the ray onto our reference triangle. Now let's map that one
386 // onto a equilateral unit triangle, which can be performed by multiplying
387 // the following matrix by the reference ray vector:
388 //
389 // eut_ray := [ -1 +1 ] * ref_ray
390 // [ sqrt(3) sqrt(3) ]
391 //
392 // Finally, we have to compute the norm of that vector, which is explicitly
393 // written down in the denominator of the following expression:
394 #ifndef __CUDA_ARCH__
395 return DataType(2) / Math::sqrt(Math::sqr(ref_ray[1]-ref_ray[0]) + DataType(3) * Math::sqr(ref_ray[0]+ref_ray[1]));
396 #else
397 return DataType(2) * CudaMath::cuda_rsqrt(Math::sqr(ref_ray[1]-ref_ray[0]) + DataType(3) * Math::sqr(ref_ray[0]+ref_ray[1]));
398 #endif
399 }
400 }; // EvalHelper<Simplex<2>,...>

560 CUDA_HOST_DEVICE static inline DataType width_directed(const ImagePointType& ray, const Tiny::Matrix<DataType, image_dim, num_verts>& coeffs)
561 {
562 // We follow the same approach as for triangles here:
563 // First, map the ray onto the reference element and then
564 // map it to a regular tetrahedron to compute its norm.
565 JacobianMatrixType jac_mat;
566 JacobianInverseType jac_inv;
567 DomainPointType ref_ray;
568
569 // compute jacobian matrix
570 for(int i(0); i < image_dim; ++i)
571 {
572 jac_mat(i,0) = coeffs[i][1];
573 jac_mat(i,1) = coeffs[i][2];
574 jac_mat(i,2) = coeffs[i][3];
575 }
576
577 // invert jacobian matrix and multiply by ray vector
578 jac_inv.set_inverse(jac_mat);
579 ref_ray.set_mat_vec_mult(jac_inv, ray);
580
581 // The transformation from the reference tetrahedron to the regular unit tetrahedron
582 // can be performed by applying the following matrix-vector product:
583 //
584 // [ 2 1 1 ]
585 // eut_ray := [ 0 1 -1 ] * ref_ray
586 // [ 0 sqrt(2) sqrt(2) ]
587 //
588 // Finally, we have to compute the norm of that vector, which is explicitly
589 // written down in the denominator of the following expression:
590 #ifndef __CUDA_ARCH__
591 return DataType(2) / Math::sqrt(Math::sqr(DataType(2) * ref_ray[0] + ref_ray[1] + ref_ray[2]) +
592 Math::sqr(ref_ray[1] - ref_ray[2]) + DataType(2) * Math::sqr(ref_ray[1] + ref_ray[2]));
593 #else
594 return DataType(2) * CudaMath::cuda_rsqrt(CudaMath::sqr(DataType(2) * ref_ray[0] + ref_ray[1] + ref_ray[2]) +
595 CudaMath::cuda_sqr(ref_ray[1] - ref_ray[2]) + DataType(2) * CudaMath::cuda_sqr(ref_ray[1] + ref_ray[2]));
596 #endif
597 }

1000 CUDA_HOST_DEVICE static inline DataType volume(const Tiny::Matrix<DataType, image_dim, num_verts>& coeffs)
1001 {
1002 // According to Varignon's theorem, the area/volume of a quadrilateral is
1003 // equal to twice the area of the dual parallelogram of the quadrilateral,
1004 // which is spanned by the four edge midpoints of the quadrilateral.
1005 // The Jacobian matrix of this transformation evaluated at the barycentre
1006 // of the reference element spans a parallelogram, which intersects with
1007 // our original quadrilateral in the edge midpoints and therefore (again
1008 // using Varignon's theorem) has the same area as the original quadrilateral.
1009 // Now the area of the "Jacobian parallelogram" is equal to four times
1010 // the determinant of its Jacobian determinant, which finally gives us a
1011 // formula for our quadrilateral area: 4*det(Jacobian(0,0)).
1012 // Note that this is not a lousy approximation, but a true identity :)
1013
1014 // compute jacobian matrix at barycentre
1015 JacobianMatrixType jac_mat;
1016 for(int i(0); i < image_dim; ++i)
1017 {
1018 jac_mat(i,0) = coeffs[i][1];
1019 jac_mat(i,1) = coeffs[i][2];
1020 }
1021
1022 // return scaled volume
1023 return DataType(4) * jac_mat.vol();
1024 }

1312 {

1314 // in the barycentre of the cell to compute the cell's volume, as this will give
1315 // very inaccurate results for non-parallelepiped cells. Instead, we have to use
1316 // the 2x2x2 Gauss-Legendre cubature rule to integrate the volume of the cell,
1317 // which is hard-coded in the code below.

1320 const DataType cx = DataType(FEAT_F128C(0.57735026918962576450914878050195745564760175127));

1323 DomainPointType cub_pt;
1324 DataType vol = DataType(0);
1325
1326 // loop over all 8 cubature points
1327 for(int i(0); i < 8; ++i)
1328 {
1329 // set cubature point coords by magic bitshifts
1330 for(int j(0); j < 3; ++j)
1331 cub_pt[j] = DataType((((i >> j) & 1) << 1) - 1) * cx;
1332
1333 // compute jacobian matrix and add its volume
1334 calc_jac_mat(jac_mat, cub_pt, coeffs);
1335 vol += jac_mat.vol();
1336 }
