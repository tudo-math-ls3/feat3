172 void read_from(FileMode mode, String filename)
173 {
174 String directory;
175 auto found = filename.rfind("/");
176 if (found != std::string::npos)
177 {
178 directory = filename.substr(0, found + 1);
179 }
180
181 std::ifstream file(filename.c_str(), std::ifstream::in);
182 if (! file.is_open())
183 XABORTM("Unable to open Matrix file " + filename);
184
185 String line;
186 std::getline(file, line);
187 if (line.find("%%MatrixMarket saddlepointmatrix coordinate real general") == String::npos)
188 XABORTM("Input-file is not a complatible file");
189
190 do {
191 if (file.eof())
192 XABORTM("Wrong Input-file");
193 std::getline(file, line);
194 line.trim_me();
195 } while (line.find("%%") == 0 || line == "");
196
197 MatrixA_ tmp_a(mode, directory + line);
198 _matrix_a = std::move(tmp_a);
199
200 std::getline(file, line);
201 line.trim_me();
202 MatrixB_ tmp_b(mode, directory + line);
203 _matrix_b = std::move(tmp_b);
204
205 std::getline(file, line);
206 line.trim_me();
207 MatrixD_ tmp_d(mode, directory + line);
208 _matrix_d = std::move(tmp_d);
209
210 file.close();
211 }

586 void restore_from_checkpoint_data(std::vector<char> & data)
587 {
588 std::uint64_t isize = *(std::uint64_t*) data.data(); //get size of checkpointed block a
589 std::vector<char>::iterator start = std::begin(data) + sizeof(std::uint64_t); //get iterator at the beginning of block a
590 std::vector<char>::iterator end = std::begin(data) + sizeof(std::uint64_t) + (int) isize; //get iterator at the beginning of block a
591 std::vector<char> buffer_a(start, end); //copy the data of block a to fresh vector
592 this->block_a().restore_from_checkpoint_data(buffer_a);
593
594 data.erase(std::begin(data), end);
595 isize = *(std::uint64_t*) data.data();
596 start = std::begin(data) + sizeof(std::uint64_t);
597 end = std::begin(data) + sizeof(std::uint64_t) + (int) isize;
598 std::vector<char> buffer_b(start, end);
599 this->block_b().restore_from_checkpoint_data(buffer_b);
600
601 data.erase(std::begin(data), end);
602 this->block_d().restore_from_checkpoint_data(data);
603 }

606 std::uint64_t set_checkpoint_data(std::vector<char>& data, SerialConfig& config)
607 {
608 std::size_t old_size = data.size();
609 data.insert(std::end(data), sizeof(std::uint64_t),0); //add placeholder
610 std::uint64_t ireal_size = this->block_a().set_checkpoint_data(data, config); //add data of block a to the overall checkpoint
611 std::uint64_t ret_size = ireal_size;
612 char* csize = reinterpret_cast<char*>(&ireal_size);
613 for(std::size_t i(0) ; i < sizeof(std::uint64_t) ; ++i) //overwrite the guessed datalength
614 {
615 data[old_size + i] = csize[i];
616 }
617
618 old_size = data.size();
619 data.insert(std::end(data), sizeof(std::uint64_t), 0); //add placeholder
620 ireal_size = this->block_b().set_checkpoint_data(data, config); //add data of block b to the overall checkpoint
621 ret_size += ireal_size;
622 csize = reinterpret_cast<char*>(&ireal_size);
623 for(std::size_t i(0) ; i < sizeof(std::uint64_t) ; ++i) //overwrite the guessed datalength
624 {
625 data[old_size + i] = csize[i];
626 }
627
628 return 2*sizeof(std::uint64_t) + ret_size + this->block_d().set_checkpoint_data(data, config); //add data of block d to the overall checkpoint
629 }
