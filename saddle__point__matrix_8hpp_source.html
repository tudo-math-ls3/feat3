170 void read_from(FileMode mode, const String& filename)
171 {
172 String directory;
173 auto found = filename.rfind("/");
174 if (found != std::string::npos)
175 {
176 directory = filename.substr(0, found + 1);
177 }
178
179 std::ifstream file(filename.c_str(), std::ifstream::in);
180 if (! file.is_open())
181 XABORTM("Unable to open Matrix file " + filename);
182
183 String line;
184 std::getline(file, line);
185 if (line.find("%%MatrixMarket saddlepointmatrix coordinate real general") == String::npos)
186 XABORTM("Input-file is not a complatible file");
187
188 do {
189 if (file.eof())
190 XABORTM("Wrong Input-file");
191 std::getline(file, line);
192 line.trim_me();
193 } while (line.find("%%") == 0 || line == "");
194
195 MatrixA_ tmp_a(mode, directory + line);
196 _matrix_a = std::move(tmp_a);
197
198 std::getline(file, line);
199 line.trim_me();
200 MatrixB_ tmp_b(mode, directory + line);
201 _matrix_b = std::move(tmp_b);
202
203 std::getline(file, line);
204 line.trim_me();
205 MatrixD_ tmp_d(mode, directory + line);
206 _matrix_d = std::move(tmp_d);
207
208 file.close();
209 }

664 void restore_from_checkpoint_data(std::vector<char> & data)
665 {
666 std::uint64_t isize = *(std::uint64_t*) data.data(); //get size of checkpointed block a
667 std::vector<char>::iterator start = std::begin(data) + sizeof(std::uint64_t); //get iterator at the beginning of block a
668 std::vector<char>::iterator end = std::begin(data) + sizeof(std::uint64_t) + (int) isize; //get iterator at the beginning of block a
669 std::vector<char> buffer_a(start, end); //copy the data of block a to fresh vector
670 this->block_a().restore_from_checkpoint_data(buffer_a);
671
672 data.erase(std::begin(data), end);
673 isize = *(std::uint64_t*) data.data();
674 start = std::begin(data) + sizeof(std::uint64_t);
675 end = std::begin(data) + sizeof(std::uint64_t) + (int) isize;
676 std::vector<char> buffer_b(start, end);
677 this->block_b().restore_from_checkpoint_data(buffer_b);
678
679 data.erase(std::begin(data), end);
680 this->block_d().restore_from_checkpoint_data(data);
681 }

684 std::uint64_t set_checkpoint_data(std::vector<char>& data, SerialConfig& config)
685 {
686 std::size_t old_size = data.size();
687 data.insert(std::end(data), sizeof(std::uint64_t),0); //add placeholder
688 std::uint64_t ireal_size = this->block_a().set_checkpoint_data(data, config); //add data of block a to the overall checkpoint
689 std::uint64_t ret_size = ireal_size;
690 char* csize = reinterpret_cast<char*>(&ireal_size);
691 for(std::size_t i(0) ; i < sizeof(std::uint64_t) ; ++i) //overwrite the guessed datalength
692 {
693 data[old_size + i] = csize[i];
694 }
695
696 old_size = data.size();
697 data.insert(std::end(data), sizeof(std::uint64_t), 0); //add placeholder
698 ireal_size = this->block_b().set_checkpoint_data(data, config); //add data of block b to the overall checkpoint
699 ret_size += ireal_size;
700 csize = reinterpret_cast<char*>(&ireal_size);
701 for(std::size_t i(0) ; i < sizeof(std::uint64_t) ; ++i) //overwrite the guessed datalength
702 {
703 data[old_size + i] = csize[i];
704 }
705
706 return 2*sizeof(std::uint64_t) + ret_size + this->block_d().set_checkpoint_data(data, config); //add data of block d to the overall checkpoint
707 }
