102 double elapsed(const TimeStamp& before) const
103 {
104#if defined(FEAT_HAVE_GETTIMEOFDAY)
105 return double(_time.tv_sec - before._time.tv_sec) + 1E-6 * double(_time.tv_usec - before._time.tv_usec);
106#elif defined(_WIN32)
107 long long freq = Windows::query_performance_frequency();
108 return (freq == 0ll) ? 0.0 : (double(_counter - before._counter) / double(freq));
109#else
110 return double(_clock - before._clock) / double(CLOCKS_PER_SEC);
111#endif
112 }

137 long long elapsed_micros(const TimeStamp& before) const
138 {
139#if defined(FEAT_HAVE_GETTIMEOFDAY)
140 return 1000000ll * (long long)(_time.tv_sec - before._time.tv_sec)
141 + (long long)(_time.tv_usec - before._time.tv_usec);
142#elif defined(_WIN32)
143 long long freq = Windows::query_performance_frequency();
144 return (freq == 0ll) ? 0ll : (1000000ll * (_counter - before._counter)) / freq;
145#else
146 return 1000000ll * (long long)(_clock - before._clock) / (long long)CLOCKS_PER_SEC;
147#endif
148 }

189 static String format_micros(long long micros, TimeFormat format = TimeFormat::s_m)
190 {
191 std::ostringstream oss;
192 oss << std::setfill('0');
193
194 // check whether the time is negative
195 if(micros < 0)
196 {
197 oss << '-';
198 micros = -micros;
199 }
200
201 switch (format)
202 {
203 case TimeFormat::h_m_s_m:
204 oss << (micros / 3600000000ll);
205 oss << ":" << std::setw(2) << ((micros / 60000000ll) % 60ll);
206 oss << ":" << std::setw(2) << ((micros / 1000000ll) % 60ll);
207 oss << "." << std::setw(3) << ((micros / 1000ll) % 1000ll);
208 break;
209
210 case TimeFormat::h_m_s:
211 oss << (micros / 3600000000ll);
212 oss << ":" << std::setw(2) << ((micros / 60000000ll) % 60ll);
213 oss << ":" << std::setw(2) << ((micros / 1000000ll) % 60ll);
214 break;
215
216 case TimeFormat::m_s_m:
217 oss << (micros / 60000000ll);
218 oss << ":" << std::setw(2) << ((micros / 1000000ll) % 60ll);
219 oss << "." << std::setw(3) << ((micros / 1000ll) % 1000ll);
220 break;
221
222 case TimeFormat::m_s:
223 oss << (micros / 60000000ll);
224 oss << ":" << std::setw(2) << ((micros / 1000000ll) % 60ll);
225 break;
226
227 case TimeFormat::s_m:
228 oss << (micros / 1000000ll);
229 oss << "." << std::setw(3) << ((micros / 1000ll) % 1000ll);
230 break;
231
232 default:
233 XABORTM("TimeFormat not supported!");
234 }
235
236 // return formatted string
237 return oss.str();
238 }

289 bool operator< (const TimeStamp & other) const
290 {
291#if defined(FEAT_HAVE_GETTIMEOFDAY)
292 return (_time.tv_sec < other._time.tv_sec) ||
293 ((_time.tv_sec == other._time.tv_sec) && (_time.tv_usec < other._time.tv_usec));
294#elif defined(_WIN32)
295 return (_counter < other._counter);
296#else
297 return (_clock < other._clock);
