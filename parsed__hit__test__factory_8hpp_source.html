79 void parse(const String& formula)
80 {
81 // add variables to our parser
82 String vars("x");
83 if (world_dim > 1) vars += ",y";
84 if (world_dim > 2) vars += ",z";
85
86 // try to parse the function
87 const int ret = _parser.Parse(formula.c_str(), vars.c_str());
88 if (ret >= 0)
89 {
90 String msg(_parser.ErrorMsg());
91 msg.append("\n>>> '");
92 msg.append(formula);
93 msg.append("'");
94 if (ret < int(formula.size()))
95 {
96 // ret contains the index of the first invalid character in the input string
97 // append an additional line to mark the faulty character
98 msg.append("\n>>>");
99 msg.append(String(std::size_t(ret + 2), '-'));
100 msg.append("^");
101 }
102
103 throw ParseError(msg);
104 } //end : can't parse function
105
106 // optimize the parsed function
107 _parser.Optimize();
108 }
109
110 bool operator()(const PointType& point) const
111 {
112 //convert DataType to double
113 const Tiny::Vector<double, world_dim> vars(point);
114
115 // evaluate the parser
116 const double val= _parser.Eval(vars.v);
117
118 // check for errors
119 switch (_parser.EvalError())
120 {
121 case 0: // no error
122 break;
123
124 case 1: // division by zero
125 throw ParseError("Error in ParsedScalarFunction evaluation: division by zero");
126
127 case 2: // sqrt error
128 throw ParseError("Error in ParsedScalarFunction evaluation: sqrt of negative value");
129
130 case 3: // log error
131 throw ParseError("Error in ParsedScalarFunction evaluation:logarithm of a negative value");
132
133 case 4: // trigonometric error
134 throw ParseError("Error in ParsedScalarFunction evaluation: illegal input value");
135
136 case 5: // recursion error
137 throw ParseError("Error in ParsedScalarFunction evaluation: maximum recursion depth reached");
138
139 default: // ???
140 throw ParseError("Error in ParsedScalarFunction evaluation: unknown error");
141 }
142 //Returns true, if point is in the "inner" part of the Mesh
143 //Otherwise returns false
144 return (val>=0.0);
145 }
146 };
