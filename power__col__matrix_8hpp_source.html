550
552 void restore_from_checkpoint_data(std::vector<char> & data)
553 {
554 std::uint64_t isize = *(std::uint64_t*) data.data(); //get size of checkpointed _first
555 std::vector<char>::iterator start = std::begin(data) + sizeof(std::uint64_t);
556 std::vector<char>::iterator last_of_first = std::begin(data) + sizeof(std::uint64_t) + (int) isize;
557 std::vector<char> buffer_first(start, last_of_first);
558 _first.restore_from_checkpoint_data(buffer_first);
559
560 data.erase(std::begin(data), last_of_first);
561 _rest.restore_from_checkpoint_data(data);
562 }
563
565 std::uint64_t set_checkpoint_data(std::vector<char>& data, SerialConfig& config)
566 {
567 std::size_t old_size = data.size();
568 data.insert(std::end(data), sizeof(std::uint64_t), 0); //add placeholder
569 std::uint64_t ireal_size = _first.set_checkpoint_data(data, config); //add data of _first to the overall checkpoint and save its size
570 char* csize = reinterpret_cast<char*>(&ireal_size);
571 for(std::size_t i(0) ; i < sizeof(std::uint64_t) ; ++i) //overwrite the guessed datalength
572 {
573 data[old_size + i] = csize[i];
574 }
575
576 return sizeof(std::uint64_t) + ireal_size + _rest.set_checkpoint_data(data, config); //generate and add checkpoint data for the _rest
577 }
