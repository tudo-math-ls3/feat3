106 explicit MatrixMirrorBuffer(const Adjacency::Graph & graph, Index entries_per_nonzero_in) :
107 Container<DT_, IT_>(0)
108 {
109 Index num_rows = graph.get_num_nodes_domain();
110 Index num_cols = graph.get_num_nodes_image();
111 Index num_nnze = graph.get_num_indices();
112
113 // create temporary vectors
114 LAFEM::DenseVector<IT_, IT_> vrow_ptr(num_rows+1);
115 LAFEM::DenseVector<IT_, IT_> vcol_idx(num_nnze);
116 LAFEM::DenseVector<DT_, IT_> vdata(num_nnze * entries_per_nonzero_in, DT_(0));
117
118 const Index * dom_ptr(graph.get_domain_ptr());
119 const Index * img_idx(graph.get_image_idx());
120 IT_ * prow_ptr(vrow_ptr.elements());
121 IT_ * pcol_idx(vcol_idx.elements());
122
123 // build row-end
124 prow_ptr[0] = IT_(dom_ptr[0]);
125 for(Index i(0); i < num_rows; ++i)
126 prow_ptr[i+1] = IT_(dom_ptr[i+1]);
127
128 // build col-idx
129 for(Index i(0); i < num_nnze; ++i)
130 pcol_idx[i] = IT_(img_idx[i]);
131
132 // build the matrix
133 this->assign(MatrixMirrorBuffer<DT_, IT_>(num_rows, num_cols, entries_per_nonzero_in, vcol_idx, vdata, vrow_ptr));
134 }

149 explicit MatrixMirrorBuffer(const Index rows_in, const Index columns_in, Index entries_per_nonzero_in,
150 DenseVector<IT_, IT_> & col_ind_in, DenseVector<DT_, IT_> & val_in, DenseVector<IT_, IT_> & row_ptr_in) :
151 Container<DT_, IT_>(rows_in * columns_in)
152 {
153 XASSERT(val_in.size() % entries_per_nonzero_in == 0);
154 XASSERT(val_in.size() == col_ind_in.size() * entries_per_nonzero_in);
155
156 this->_scalar_index.push_back(rows_in);
157 this->_scalar_index.push_back(columns_in);
158 this->_scalar_index.push_back(col_ind_in.size());
159 this->_scalar_index.push_back(entries_per_nonzero_in);
160
161 this->_elements.push_back(val_in.elements());
162 this->_elements_size.push_back(val_in.size());
163 this->_indices.push_back(col_ind_in.elements());
164 this->_indices_size.push_back(col_ind_in.size());
165 this->_indices.push_back(row_ptr_in.elements());
166 this->_indices_size.push_back(row_ptr_in.size());
167
168 for (Index i(0) ; i < this->_elements.size() ; ++i)
169 MemoryPool::increase_memory(this->_elements.at(i));
170 for (Index i(0) ; i < this->_indices.size() ; ++i)
171 MemoryPool::increase_memory(this->_indices.at(i));
172 }
