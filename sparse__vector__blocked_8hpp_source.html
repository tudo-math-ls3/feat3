71 template <typename T1_, typename T2_>
72 static void _insertion_sort(T1_ * key, T2_ * val1, Index size)
73 {
74 T1_ swap_key;
75 T2_ swap1;
76 for (Index i(1), j ; i < size ; ++i)
77 {
78 swap_key = key[i];
79 swap1 = val1[i];
80 j = i;
81 while (j > 0 && key[j-1] > swap_key)
82 {
83 key[j] = key[j-1];
84 val1[j] = val1[j-1];
85 --j;
86 }
87 key[j] = swap_key;
88 val1[j] = swap1;
89 }
90 }

375 const ValueType operator()(Index index) const
376 {
377 ASSERTM(index < this->_scalar_index.at(0), "index exceeds sparse vector size");
378
379 MemoryPool::synchronize();
380
381 if (this->_elements.empty())
382 return ValueType(0.);
383
384 if (sorted() == 0)
385 const_cast<SparseVectorBlocked *>(this)->sort();
386
387 Index i(0);
388 while (i < used_elements())
389 {
390 if (indices()[i] >= index)
391 break;
392 ++i;
393 }
394
395 if (i < used_elements() && indices()[i] == index)
396 {
397 return this->elements()[i];
398 }
399 else
400 return ValueType(0.);
401 }

410 void operator()(Index index, const ValueType& val)
411 {
412 ASSERTM(index < this->_scalar_index.at(0), "index exceeds sparse vector size");
413
414 // flag container as not sorted anymore
415 // CAUTION: do not use any method triggering resorting until we are finished
416 _sorted() = 0;
417
418 // vector is empty, no arrays allocated
419 if (this->_elements.size() == 0)
420 {
421 this->_elements.push_back(MemoryPool::template allocate_memory<DT_>(alloc_increment() * Index(BlockSize_)));
422 this->_elements_size.push_back(alloc_increment() * Index(BlockSize_));
423 MemoryPool::template set_memory<DT_>(this->_elements.back(), DT_(4711), alloc_increment() * Index(BlockSize_));
424 this->_indices.push_back(MemoryPool::template allocate_memory<IT_>(alloc_increment()));
425 this->_indices_size.push_back(alloc_increment());
426 MemoryPool::template set_memory<IT_>(this->_indices.back(), IT_(4711), alloc_increment());
427 _allocated_elements() = alloc_increment();
428 MemoryPool::copy(this->_elements.at(0), val.v, Index(BlockSize_));
429 MemoryPool::set_memory(this->_indices.at(0), IT_(index));
430 _used_elements() = 1;
431 }
432
433 // append element in already allocated arrays
434 else if(_used_elements() < allocated_elements())
435 {
436 MemoryPool::copy(this->_elements.at(0) + _used_elements() * Index(BlockSize_), val.v,
437 Index(BlockSize_));
438 MemoryPool::set_memory(this->_indices.at(0) + _used_elements(), IT_(index));
439 ++_used_elements();
440 }
441
442 // reallocate arrays, append element
443 else
444 {
445 _allocated_elements() += alloc_increment();
446
447 DT_ * elements_new(MemoryPool::template allocate_memory<DT_>(
448 allocated_elements() * Index(BlockSize_)));
449 MemoryPool::template set_memory<DT_>(elements_new, DT_(4711),
450 allocated_elements() * Index(BlockSize_));
451 IT_ * indices_new(MemoryPool::template allocate_memory<IT_>(allocated_elements()));
452 MemoryPool::template set_memory<IT_>(indices_new, IT_(4711), allocated_elements());
453
454 MemoryPool::copy(elements_new, this->_elements.at(0), _used_elements() * Index(BlockSize_));
455 MemoryPool::copy(indices_new, this->_indices.at(0), _used_elements());
456
457 MemoryPool::release_memory(this->_elements.at(0));
458 MemoryPool::release_memory(this->_indices.at(0));
459
460 this->_elements.at(0) = elements_new;
461 this->_indices.at(0) = indices_new;
462
463 MemoryPool::copy(this->_elements.at(0) + used_elements() * Index(BlockSize_), val.v,
464 Index(BlockSize_));
465 MemoryPool::set_memory(this->_indices.at(0) + _used_elements(), IT_(index));
466
467 ++_used_elements();
468 this->_elements_size.at(0) = allocated_elements() * Index(BlockSize_);
469 this->_indices_size.at(0) = allocated_elements();
470 }
471 }

474 void sort()
475 {
476 if (sorted() == 0)
477 {
478 //first of all, mark vector as sorted, because otherwise we would call ourselves inifite times
479 // CAUTION: do not use any method triggering resorting until we are finished
480 _sorted() = 1;
481
482 // check if there is anything to be sorted
483 if(_used_elements() == Index(0))
484 return;
485
486 IT_ * pindices;
487 ValueType * pelements;
488 pindices = this->_indices.at(0);
489 pelements = this->elements();
490
491 _insertion_sort(pindices, pelements, _used_elements());
492
493 // find and mark duplicate entries
494 for (Index i(1) ; i < _used_elements() ; ++i)
495 {
496 if (pindices[i-1] == pindices[i])
497 {
498 pindices[i-1] = std::numeric_limits<IT_>::max();
499 }
500 }
501
502 // sort out marked duplicated elements
503 _insertion_sort(pindices, pelements, _used_elements());
504 Index junk(0);
505 while (pindices[_used_elements() - 1 - junk] == std::numeric_limits<IT_>::max() && junk < _used_elements())
506 ++junk;
507 _used_elements() -= junk;
508 }
509 }

692 void permute(Adjacency::Permutation & perm)
693 {
694 if (perm.size() == 0)
695 return;
696
697 XASSERTM(perm.size() == this->size(), "Container size does not match permutation size");
698
699 SparseVectorBlocked<DT_, IT_, BlockSize_> target(this->size());
700
701 auto inv = perm.inverse();
702 const Index * const inv_pos(inv.get_perm_pos());
703 for (Index i(0) ; i < this->used_elements() ; ++i)
704 {
705 const Index col = this->indices()[i];
706 target(inv_pos[col], (*this)(col));
707 }
708
709 target.sort();
710 this->assign(target);
711 }

782 friend bool operator== (const SparseVectorBlocked & a, const SparseVectorBlocked & b)
783 {
784 if (a.size() != b.size())
785 return false;
786 if (a.get_elements().size() != b.get_elements().size())
787 return false;
788 if (a.get_indices().size() != b.get_indices().size())
789 return false;
790
791 if(a.size() == 0 && b.size() == 0 && a.get_elements().size() == 0 && b.get_elements().size() == 0)
792 return true;
793
794 for (Index i(0) ; i < a.size() ; ++i)
795 {
796 auto ta = a(i);
797 auto tb = b(i);
798 for (int j(0) ; j < BlockSize_ ; ++j)
799 if (ta[j] != tb[j])
800 return false;
801 }
802
803 return true;
804 }

815 bool compare_layout(const SparseVectorBlocked& other) const
816 {
817 if(this->size() != other.size())
818 return false;
819 if(this->used_elements() != other.used_elements())
820 return false;
821
822 Index n = this->used_elements();
823 if(n == Index(0))
824 return true; // both vectors are empty
825
826 const IT_* a = this->indices();
827 const IT_* b = this->indices();
828
829 // shallow copy? (aka same array)
830 if(a == b)
831 return true;
832
833 // check all array entries
834 for(Index i(0); i < n; ++i)
835 {
836 if(a[i] != b[i])
837 return false;
838 }
839
840 // okay, arrays are identical
841 return true;
842 }
