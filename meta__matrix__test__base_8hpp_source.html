138 static void gen_system(Index m, SystemDiagMatrix& mat_sys, SystemVector& vec_sol, SystemVector& vec_rhs)
139 {
141 PointstarFactoryFD<DataType_, IndexType_> ps_fd(m, Index(2));
142 PointstarFactoryFE<DataType_, IndexType_> ps_fe(m);
143
145 const SparseMatrixCSR<DataType_, IndexType_> mat_fd(ps_fd.matrix_csr());
146 const SparseMatrixCSR<DataType_, IndexType_> mat_fe(ps_fe.matrix_csr());
147
148 // generate Q2-bubble and eigenvector
149 const DenseVector<DataType_, IndexType_> vec_eigen1(ps_fd.eigenvector_min());
150 const DenseVector<DataType_, IndexType_> vec_eigen2(ps_fd.eigenvector_min());
151 const DenseVector<DataType_, IndexType_> vec_bubble(ps_fd.vector_q2_bubble());
152
153 // set system matrix
154 mat_sys.template at<0,0>().template at<0,0>().convert(mat_fe);
155 mat_sys.template at<0,0>().template at<1,1>().convert(mat_fe);
156 mat_sys.template at<0,1>().template at<0,0>().convert(mat_fd);
157 mat_sys.template at<0,1>().template at<1,0>().convert(mat_fd);
158 mat_sys.template at<1,0>().template at<0,0>().convert(mat_fd);
159 mat_sys.template at<1,0>().template at<0,1>().convert(mat_fd);
160
161 // set solution vector
162 vec_sol.template at<0>().template at<0>().convert(vec_bubble); // u1
163 vec_sol.template at<0>().template at<1>().convert(vec_eigen1); // u2
164 vec_sol.template at<1>().convert(vec_eigen2); // p
165
166 // create vectors for rhs computation
167 DenseVector<DataType_, IndexType_> vec_rhs1(vec_bubble.size());
168 DenseVector<DataType_, IndexType_> vec_rhs2(vec_bubble.size());
169 DenseVector<DataType_, IndexType_> vec_rhs3(vec_bubble.size());
170
171 // compute rhs vector (by exploiting the eigenvector property)
172 mat_fe.apply(vec_rhs1, vec_bubble); // A11*u1
173 vec_rhs1.axpy(vec_eigen1, ps_fd.lambda_min()); // B1*p
174 vec_rhs2.scale(vec_eigen1, ps_fe.lambda_min() + ps_fd.lambda_min()); // A22*u2 + B2*p
175 mat_fd.apply(vec_rhs3, vec_bubble); // D1*u1
176 vec_rhs3.axpy(vec_eigen1, ps_fd.lambda_min()); // D2*u2
177
178 // set rhs vector
179 vec_rhs.template at<0>().template at<0>().convert(vec_rhs1);
180 vec_rhs.template at<0>().template at<1>().convert(vec_rhs2);
181 vec_rhs.template at<1>().convert(vec_rhs3);
182 }
