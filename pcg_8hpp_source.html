188 virtual Status _apply_intern(VectorType& vec_sol)
189 {
190 IterationStats pre_iter(*this);
191 Statistics::add_solver_expression(std::make_shared<ExpressionStartSolve>(this->name()));
192
193 const MatrixType& matrix(this->_system_matrix);
194 const FilterType& filter(this->_system_filter);
195 VectorType& vec_r(this->_vec_r);
196 VectorType& vec_p(this->_vec_p);
197 // Note: q and z are temporary vectors whose usage does
198 // not overlap, so we use the same vector for them
199 VectorType& vec_q(this->_vec_t);
200 VectorType& vec_z(this->_vec_t);
201
202 // Note:
203 // In the algorithm below, the following relations hold:
204 // q[k] = A * p[k]
205 // z[k] = M^{-1} * r[k]
206
207 // set initial defect:
208 // r[0] := b - A*x[0]
209 Status status = this->_set_initial_defect(vec_r, vec_sol);
210 if(status != Status::progress)
211 {
212 pre_iter.destroy();
213 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), status, this->get_num_iter()));
214 return status;
215 }
216
217 // apply preconditioner to defect vector
218 // p[0] := M^{-1} * r[0]
219 if(!this->_apply_precond(vec_p, vec_r, filter))
220 {
221 pre_iter.destroy();
222 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
223 return Status::aborted;
224 }
225
226 // compute initial gamma:
227 // gamma[0] := < r[0], p[0] >
228 DataType gamma = vec_r.dot(vec_p);
229
230 pre_iter.destroy();
231
232 // start iterating
233 while(status == Status::progress)
234 {
235 IterationStats stat(*this);
236
237 // q[k] := A*p[k]
238 matrix.apply(vec_q, vec_p);
239 filter.filter_def(vec_q);
240
241 // compute alpha
242 // alpha[k] := gamma[k] / < q[k], p[k] >
243 DataType alpha = gamma / vec_q.dot(vec_p);
244
245 // update solution vector:
246 // x[k+1] := x[k] + alpha[k] * p[k]
247 vec_sol.axpy(vec_p, vec_sol, alpha);
248
249 // update defect vector:
250 // r[k+1] := r[k] - alpha[k] * q[k]
251 vec_r.axpy(vec_q, vec_r, -alpha);
252
253 // compute defect norm
254 status = this->_set_new_defect(vec_r, vec_sol);
255 if(status != Status::progress)
256 {
257 stat.destroy();
258 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), status, this->get_num_iter()));
259 return status;
260 }
261
262 // apply preconditioner
263 // z[k+1] := M^{-1} * r[k+1]
264 if(!this->_apply_precond(vec_z, vec_r, filter))
265 {
266 stat.destroy();
267 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
268 return Status::aborted;
269 }
270
271 // compute new gamma:
272 // gamma[k+1] := < r[k+1] , z[k+1] >
273 DataType gamma2 = gamma;
274 gamma = vec_r.dot(vec_z);
275
276 // compute beta:
277 // beta[k] := gamma[k+1] / gamma[k]
278 DataType beta = gamma / gamma2;
279
280 // update direction vector:
281 // p[k+1] := z[k+1] + beta[k] * p[k]
282 vec_p.axpy(vec_p, vec_z, beta);
283 }
284
285 // we should never reach this point...
286 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::undefined, this->get_num_iter()));
287 return Status::undefined;
288 }
