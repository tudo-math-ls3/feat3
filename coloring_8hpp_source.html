372 void fill_color(const std::vector<int>& coloring, int hint = -1)
373 {
374 int num_colors = hint;
375 if(hint < 0)
376 {
377 num_colors = *std::max_element(coloring.begin(), coloring.end()) + 1;
378 }
379 // fill tmp vector with coloring
380 std::vector<std::vector<int>> tmp_vector;
381 tmp_vector.resize(Index(num_colors));
382 for(std::size_t i = 0; i < coloring.size(); ++i)
383 {
384 tmp_vector.at(std::size_t(coloring.at(i))).push_back(int(i));
385 }
386 _coloring_maps.resize(std::size_t(num_colors));
387 _coloring_map_sizes.resize(std::size_t(num_colors));
388 for(Index i = 0; i < Index(num_colors); ++i)
389 {
390 _coloring_maps.at(i) = MemoryPool::allocate_memory<int>(tmp_vector.at(i).size());
391 MemoryPool::copy(_coloring_maps.at(i), tmp_vector.at(i).data(), tmp_vector.at(i).size());
392 _coloring_map_sizes.at(i) = tmp_vector.at(i).size();
393 }

404 void fill_color(const Coloring& coloring, int hint = -1)
405 {
406 int num_colors = int(coloring.get_num_colors());
407 if(hint >= 0)
408 {
409 ASSERTM(num_colors == hint, "Hint and number of colors do not fit!");
410 }
411
412 // fill tmp vector with coloring
413 std::vector<std::vector<int>> tmp_vector;
414 tmp_vector.resize(Index(num_colors));
415 for(std::size_t i = 0; i < coloring.size(); ++i)
416 {
417 tmp_vector.at(std::size_t(coloring[i])).push_back(int(i));
418 }
419 _coloring_maps.resize(std::size_t(num_colors));
420 _coloring_map_sizes.resize(std::size_t(num_colors));
421 for(Index i = 0; i < Index(num_colors); ++i)
422 {
423 _coloring_maps.at(i) = MemoryPool::allocate_memory<int>(tmp_vector.at(i).size());
424 MemoryPool::copy(_coloring_maps.at(i), tmp_vector.at(i).data(), tmp_vector.at(i).size());
425 _coloring_map_sizes.at(i) = Index(tmp_vector.at(i).size());
426 }
427 }

429 void release_color()

431 for(Index i = 0; i < _coloring_maps.size(); ++i)
432 {
433 MemoryPool::release_memory(_coloring_maps.at(i));

435 _coloring_maps.clear();
436 _coloring_map_sizes.clear();
