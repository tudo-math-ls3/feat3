351 virtual void init_symbolic() override
352 {
353 BaseClass::init_symbolic();
354
355 // get sub-matrices
356 const MatrixTypeA& matrix_a = _system_matrix.block_a();
357 const MatrixTypeB& matrix_b = _system_matrix.block_b();
358 const MatrixTypeD& matrix_d = _system_matrix.block_d();
359
360 // get velocity/pressure space dimensions
361 const Index n_v = matrix_b.rows();
362 const Index n_p = matrix_b.columns();
363
364 // verify matrix dimensions
365 XASSERT(n_v == matrix_a.rows());
366 XASSERT(n_v == matrix_a.columns());
367 XASSERT(n_p == matrix_d.rows());
368 XASSERT(n_v == matrix_d.columns());
369 XASSERTM(n_p == _weight_vector.size(), "invalid weight vector/mean filter size");
370
371 // get the number of non-zeroes in sub-matrices
372 const Index nnze_a = matrix_a.used_elements();
373 const Index nnze_b = matrix_b.used_elements();
374 const Index nnze_d = matrix_d.used_elements();
375
376 // allocate our solver matrix
377 _solver_matrix = UmfMatrixType(dim_*n_v+n_p+1, dim_*n_v+n_p+1, dim_*dim_*nnze_a + dim_*(nnze_b+nnze_d) + 2*n_p);
378
379 // get our input matrix arrays
380 const IT_* irow_ptr_a = matrix_a.row_ptr();
381 const IT_* icol_idx_a = matrix_a.col_ind();
382 const IT_* irow_ptr_b = matrix_b.row_ptr();
383 const IT_* icol_idx_b = matrix_b.col_ind();
384 const IT_* irow_ptr_d = matrix_d.row_ptr();
385 const IT_* icol_idx_d = matrix_d.col_ind();
386
387 // get our output matrix arrays
388 Index* orow_ptr = _solver_matrix.row_ptr();
389 Index* ocol_idx = _solver_matrix.col_ind();
390
391 // assemble the solver matrix structure
392 orow_ptr[0] = Index(0);
393 // loop over all rows of A and B
394 for(Index i(0); i < n_v; ++i)
395 {
396 Index id = i*Index(dim_);
397 // loop over all block dimensions
398 for(Index j(0); j < Index(dim_); ++j, ++id)
399 {
400 Index op = orow_ptr[id];
401 // copy input matrix A row pattern
402 for(Index ip(irow_ptr_a[i]); ip < irow_ptr_a[i+1]; ++ip)
403 {
404 // convert block to scalar indices
405 for(Index k(0); k < Index(dim_); ++k, ++op)
406 ocol_idx[op] = Index(icol_idx_a[ip])*Index(dim_) + k;
407 }
408 // copy input matrix B row pattern
409 for(Index ip(irow_ptr_b[i]); ip < irow_ptr_b[i+1]; ++ip, ++op)
410 ocol_idx[op] = n_v*Index(dim_) + Index(icol_idx_b[ip]);
411 // set next row pointer
412 orow_ptr[id+1] = op;
413 }
414 }
415 // loop over all rows of D
416 for(Index i(0); i < n_p; ++i)
417 {
418 Index op = orow_ptr[Index(dim_)*n_v + i];
419 // copy input matrix D row pattern
420 for(Index ip(irow_ptr_d[i]); ip < irow_ptr_d[i+1]; ++ip)
421 {
422 // convert block to scalar value
423 for(Index k(0); k < Index(dim_); ++k, ++op)
424 ocol_idx[op] = Index(icol_idx_d[ip])*Index(dim_) + k;
425 }
426 // append a single entry in the last column
427 ocol_idx[op] = Index(dim_)*n_v + n_p;
428 // set next row pointer
429 orow_ptr[Index(dim_)*n_v+i+1] = ++op;
430 }
431
432 // append an (almost) dense row for pressure
433 Index op(orow_ptr[Index(dim_)*n_v + n_p]);
434 for(Index j(0); j < n_p; ++j, ++op)
435 {
436 ocol_idx[op] = Index(dim_)*n_v + j;
437 }
438 // set last row pointer
439 orow_ptr[Index(dim_)*n_v+n_p+1] = op;
440
441 // Okay, solver matrix structure assembly complete
442
443 // create temporary vectors
444 _vec_x = _solver_matrix.create_vector_r();
445 _vec_b = _solver_matrix.create_vector_r();
446
447 // initialize umfpack
448 _umfpack.init_symbolic();
449 }

451 virtual void done_symbolic() override

453 _umfpack.done_symbolic();
454 _vec_b.clear();
455 _vec_x.clear();
456 _solver_matrix.clear();
457
458 BaseClass::done_symbolic();
459 }

461 virtual void init_numeric() override
462 {
463 BaseClass::init_numeric();
464
465 // get sub-matrices
466 const MatrixTypeA& matrix_a = _system_matrix.block_a();
467 const MatrixTypeB& matrix_b = _system_matrix.block_b();
468 const MatrixTypeD& matrix_d = _system_matrix.block_d();
469
470 // get velocity/pressure space dimensions
471 const Index n_v = matrix_b.rows();
472 const Index n_p = matrix_b.columns();
473
474 // get our input matrix arrays
475 const IT_* irow_ptr_a = matrix_a.row_ptr();
476 const auto* idata_a = matrix_a.val();
477 const IT_* irow_ptr_b = matrix_b.row_ptr();
478 const auto* idata_b = matrix_b.val();
479 const IT_* irow_ptr_d = matrix_d.row_ptr();
480 const auto* idata_d = matrix_d.val();
481
482 // get input vector array
483 const DT_* weight = _weight_vector.elements();
484
485 // get output matrix arrays
486 const Index* orow_ptr = _solver_matrix.row_ptr();
487 double* odata = _solver_matrix.val();
488
489 // loop over all rows of A and B
490 for(Index i(0); i < n_v; ++i)
491 {
492 Index id = i*Index(dim_);
493 // loop over all block dimensions
494 for(int j(0); j < dim_; ++j, ++id)
495 {
496 Index op = orow_ptr[id];
497 // copy input matrix A row data
498 for(Index ip(irow_ptr_a[i]); ip < irow_ptr_a[i+1]; ++ip)
499 {
500 for(int k(0); k < dim_; ++k, ++op)
501 odata[op] = double(idata_a[ip][j][k]);
502 }
503 // copy input matrix B row data
504 for(Index ip(irow_ptr_b[i]); ip < irow_ptr_b[i+1]; ++ip, ++op)
505 odata[op] = double(idata_b[ip][j][0]);
506 }
507 }
508 // loop over all rows of D
509 for(Index i(0); i < n_p; ++i)
510 {
511 Index op = orow_ptr[Index(dim_)*n_v + i];
512 // copy input matrix D row data
513 for(Index ip(irow_ptr_d[i]); ip < irow_ptr_d[i+1]; ++ip)
514 {
515 // convert block to scalar value
516 for(int k(0); k < dim_; ++k, ++op)
517 odata[op] = double(idata_d[ip][0][k]);
518 }
519 // copy weight vector entry as last column entry
520 odata[op] = double(weight[i]);
521 }
522
523 // copy weight vector into last row
524 Index op(orow_ptr[Index(dim_)*n_v + n_p]);
525 for(Index j(0); j < n_p; ++j, ++op)
526 {
527 odata[op] = double(weight[j]);
528 }
529
530 // okay, solver matrix data assembly completed
531
532 // initialize umfpack
533 _umfpack.init_numeric();
534 }

537 {
538 _umfpack.done_numeric();
539
540 BaseClass::done_numeric();
541 }

553 virtual Status apply(VectorType& vec_sol, const VectorType& vec_rhs) override
554 {
555 const Index n_v = vec_sol.template at<0>().size() * Index(dim_);
556 const Index n_p = vec_sol.template at<1>().size();
557
558 // copy RHS vector
559 const DT_* vr_v = vec_rhs.template at<0>().template elements<LAFEM::Perspective::pod>();
560 const DT_* vr_p = vec_rhs.template at<1>().elements();
561 double* vb = _vec_b.elements();
562
563 // copy velocity RHS
564 for(Index i(0); i < n_v; ++i, ++vb)
565 *vb = double(vr_v[i]);
566 // copy pressure RHS
567 for(Index i(0); i < n_p; ++i, ++vb)
568 *vb = double(vr_p[i]);
569 // append Lagrange multiplier RHS
570 *vb = 0.0;
571
572 // solve system
573 Status status = _umfpack.apply(_vec_x, _vec_b);
574
575 // copy sol vector
576 const double* vx = _vec_x.elements();
577 DT_* vs_v = vec_sol.template at<0>().template elements<LAFEM::Perspective::pod>();
578 DT_* vs_p = vec_sol.template at<1>().elements();
579 // copy velocity solution
580 for(Index i(0); i < n_v; ++i, ++vx)
581 vs_v[i] = DT_(*vx);
582 // copy pressure solution
583 for(Index i(0); i < n_p; ++i, ++vx)
584 vs_p[i] = DT_(*vx);
585
586 // okay
587 return status;
588 }

648 template<typename Matrix_>
649 class GenericUmfpack :
650 public SolverBase<typename Matrix_::VectorTypeL>
651 {
652 public:
654 typedef SolverBase<typename Matrix_::VectorTypeL> BaseClass;
656 typedef Matrix_ MatrixType;
658 typedef typename MatrixType::VectorTypeL VectorType;
659
660 protected:
662 const MatrixType& _matrix;
664 typename Umfpack::MatrixType _umf_matrix;
666 typename Umfpack::VectorType _umf_vsol, _umf_vrhs;
668 LAFEM::DenseVector<typename VectorType::DataType, typename VectorType::IndexType> _tmp_vsol, _tmp_vrhs;
670 Umfpack _umfpack;
671
672 public:
673 explicit GenericUmfpack(const MatrixType& matrix) :
674 _matrix(matrix),
675 _umfpack(_umf_matrix)
676 {
677 }
678
680 virtual ~GenericUmfpack()
681 {
682 }
683
685 virtual String name() const override
686 {
687 return "GenericUmfpack";
688 }
689
690 virtual void init_symbolic() override
691 {
692 BaseClass::init_symbolic();
693
694 // convert our system matrix to <double, Index> (if necessary)
695 typename MatrixType::template ContainerType<double, Index> mat_double;
696 mat_double.convert(_matrix);
697
698 // convert matrix to obtain the structure
699 _umf_matrix.convert(mat_double);
700
701 // create umfpack vectors
702 _umf_vsol = _umf_matrix.create_vector_l();
703 _umf_vrhs = _umf_matrix.create_vector_l();
704
705 // convert to temporary vectors; note that these share their data arrays
706 // with the umfpack vectors if their datatype is also 'double'
707 _tmp_vsol.convert(_umf_vsol);
708 _tmp_vrhs.convert(_umf_vrhs);
709
710 // factorize symbolic
711 _umfpack.init_symbolic();
712 }
713
714 virtual void done_symbolic() override
715 {
716 _umfpack.done_symbolic();
717
718 _tmp_vrhs.clear();
719 _tmp_vsol.clear();
720 _umf_vrhs.clear();
721 _umf_vsol.clear();
722 _umf_matrix.clear();
723
724 BaseClass::done_symbolic();
725 }
726
727 virtual void init_numeric() override
728 {
729 BaseClass::init_numeric();
730
731 // convert our system matrix to <double, Index> (if necessary)
732 typename MatrixType::template ContainerType<double, Index> mat_double;
733 mat_double.convert(_matrix);
734
735 // get the array of our CSR matrix
736 const Index* row_ptr = _umf_matrix.row_ptr();
737 /*const*/ Index* col_idx = _umf_matrix.col_ind();
738 double* val = _umf_matrix.val();
739
740 // copy entries into our CSR matrix
741 for(Index i(0); i < _umf_matrix.rows(); ++i)
742 mat_double.set_line(i, val + row_ptr[i], col_idx + row_ptr[i], 0);
743
744 // factorize
745 _umfpack.init_numeric();
746 }
747
748 virtual void done_numeric() override
749 {
750 _umfpack.done_numeric();
751 BaseClass::done_numeric();
752 }
753
754 virtual Status apply(VectorType& vec_sol, const VectorType& vec_rhs) override
755 {
756 // convert RHS vector
757 _tmp_vrhs.copy(vec_rhs);
758 _umf_vrhs.convert(_tmp_vrhs);
759
760 // solve
761 Status status = _umfpack.apply(_umf_vsol, _umf_vrhs);
762
763 // convert sol vector
764 _tmp_vsol.convert(_umf_vsol);
765 _tmp_vsol.copy_inv(vec_sol);
766
767 return status;
768 }
769 }; // class GenericUmfpack<...>
