192 void sync(const Geometry::RootMeshNode<MeshType>& mesh_node, const Dist::Comm& comm)
193 {
194 XASSERTM(!_compiled, "assembler has already been compiled!");
195
196 // get halo map
197 const auto& halo_map = mesh_node.get_halo_map();
198 const std::size_t num_halos = halo_map.size();
199
200 static constexpr std::size_t dim1(shape_dim + 1);
201
202 // create buffers for all halos [halo][idx]
203 std::vector<std::array<std::vector<int>, dim1>> send_bufs, recv_bufs;
204 send_bufs.resize(num_halos);
205 recv_bufs.resize(num_halos);
206
207 // create request vectors
208 Dist::RequestVector send_reqs, recv_reqs;
209 send_reqs.reserve(num_halos*dim1);
210 recv_reqs.reserve(num_halos*dim1);
211
212 // loop over all halos
213 auto it = halo_map.begin();
214 for(Index h(0); h < num_halos; ++h, ++it)
215 {
216 // loop over all dimensions
217 for(std::size_t d(0); d < dim1; ++d)
218 {
219 // get halo target set and its size
220 const Geometry::TargetSet& halo_trg_set = _get_target_set(it->second->get_target_set_holder(), int(d));
221 const Index n = halo_trg_set.get_num_entities();
222
223 // nothing to do?
224 if(n == Index(0))
225 continue;
226
227 // resize buffers
228 send_bufs[h][d].resize(n);
229 recv_bufs[h][d].resize(n);
230 int* sbuf = send_bufs[h][d].data();
231
232 // post receive request
233 recv_reqs.push_back(comm.irecv(recv_bufs[h][d].data(), n, it->first));
234
235 // gather send data via mirror
236 for(Index i(0); i < n; ++i)
237 sbuf[i] = _fbm_masks[d][halo_trg_set[i]];
238
239 // send
240 send_reqs.push_back(comm.isend(sbuf, n, it->first));
241 }
242 }
243
244 // wait for all receives to finish
245 recv_reqs.wait_all();
246
247 // loop over all halos
248 it = halo_map.begin();
249 for(Index h(0); h < num_halos; ++h, ++it)
250 {
251 // loop over all dimensions
252 for(std::size_t d(0); d < dim1; ++d)
253 {
254 // get halo target set and its size
255 const Geometry::TargetSet& halo_trg_set = _get_target_set(it->second->get_target_set_holder(), int(d));
256 const Index n = halo_trg_set.get_num_entities();
257
258 // nothing to do?
259 if(n == Index(0))
260 continue;
261
262 int* rbuf = recv_bufs[h][d].data();
263
264 // scatter receive data via mirror
265 for(Index i(0); i < n; ++i)
266 _fbm_masks[d][halo_trg_set[i]] |= rbuf[i];
267 }
268 }
269
270 // wait for all sends to finish
271 send_reqs.wait_all();
272 }

277 void compile()
278 {
279 XASSERTM(!_compiled, "assembler has already been compiled!");
280
281 // process the mask
282 _process_shapes(_fbm_masks, _mesh.get_index_set_holder());
283
284 // count masked entities
285 Index counts_i[shape_dim+1], counts_o[shape_dim+1];
286 for(int d(0); d <= shape_dim; ++d)
287 {
288 Index c0(0u), c1(0u);
289 for(int x : _fbm_masks.at(std::size_t(d)))
290 {
291 //c0 += Index( x & 1); // count number of bit 0 set to 1
292 //c1 += Index((x >> 1) & 1); // count number of bit 1 set to 1
293 c0 += Index(x == 1);
294 c1 += Index(x == 3);
295 }
296 counts_i[d] = c0;
297 counts_o[d] = c1;
298 }
299
300 // allocate mesh-part
301 this->_fbm_meshpart_interface.reset(new Geometry::MeshPart<MeshType>(counts_i, false));
302 this->_fbm_meshpart_inside.reset(new Geometry::MeshPart<MeshType>(counts_o, false));
303
304 // compute target sets
305 for(int d(0); d <= shape_dim; ++d)
306 {
307 _build_tbm_target_set(_get_target_set(_fbm_meshpart_interface->get_target_set_holder(), d), d, 1);
308 _build_tbm_target_set(_get_target_set(_fbm_meshpart_inside->get_target_set_holder(), d), d, 3);
309 }
310
311 // set up unit filter assemblers
312 this->_unit_asm_interface.reset(new Assembly::UnitFilterAssembler<MeshType>());
313 this->_unit_asm_inside.reset(new Assembly::UnitFilterAssembler<MeshType>());
314 _unit_asm_interface->add_mesh_part(*this->_fbm_meshpart_interface);
315 _unit_asm_inside->add_mesh_part(*this->_fbm_meshpart_inside);
316
317 // done
318 _compiled = true;
319 }

367 template<typename DT_, typename IT_, typename Space_>
368 void assemble_characteristic_vector(LAFEM::DenseVector<DT_, IT_>& vector, const Space_& space,
369 const DT_ val_outside = DT_(0), const DT_ val_inside = DT_(1), const DT_ val_intface = DT_(1)) const
370 {
371 XASSERTM(_compiled, "FBM assembler must be compiled first!");
372 if(vector.empty())
373 vector = LAFEM::DenseVector<DT_, IT_>(space.get_num_dofs());
374
375 LAFEM::UnitFilter<DT_, IT_> filter_i(space.get_num_dofs());
376 LAFEM::UnitFilter<DT_, IT_> filter_o(space.get_num_dofs());
377 _unit_asm_interface->assemble(filter_i, space);
378 _unit_asm_inside->assemble(filter_o, space);
379
380 filter_i.get_filter_vector().format(val_intface);
381 filter_o.get_filter_vector().format(val_inside);
382
383 vector.format(val_outside);
384 filter_i.filter_sol(vector);
385 filter_o.filter_sol(vector);
386 }

632 template<int m_>
633 static void _process_mask(std::vector<int>& shape_mask, const std::vector<int>& face_mask, const Geometry::IndexSet<m_>& faces_at_shape)
634 {
635 XASSERT(Index(shape_mask.size()) == faces_at_shape.get_num_entities());
636 XASSERT(Index(face_mask.size()) == faces_at_shape.get_index_bound());
637
638 const Index n = faces_at_shape.get_num_entities();
639 const int m = faces_at_shape.get_num_indices();
640 for(Index i(0); i < n; ++i)
641 {
642 // bit 1 of shape mask is only 1 if bit 0 of all faces (of all dimensions) are 1
643 int bit1 = (shape_mask[i] >> 1) & 1;
644 for(int j(0); j < m; ++j)
645 bit1 &= (face_mask[faces_at_shape(i,j)] & 1);
646 shape_mask[i] = (shape_mask[i] & 1) | (bit1 << 1);
647 }
648 }
