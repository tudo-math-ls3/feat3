24 template<typename Shape_, int dim_ = Shape_::dimension>
25 class TopoWriteHelper
26 {
27 public:
28 static void write_topology(std::ostream& os, const IndexSetHolder<Shape_>& ish, const String& sindent, bool skip_empty)
29 {
30 // recurse down
31 TopoWriteHelper<Shape_, dim_-1>::write_topology(os, ish, sindent, skip_empty);
32
33 String sind(sindent);
34 if(!sind.empty())
35 sind.append(" ");
36
37 const auto& index_set = ish.template get_index_set<dim_, 0>();
38
39 // empty topology?
40 // Note: this case is valid for MeshPart topologies
41 if(skip_empty && (index_set.get_num_entities() == Index(0)))
42 return;
43
44 os << sindent << "<Topology dim=\"" << dim_ << "\">" << std::endl;
45 for(Index i(0); i < index_set.get_num_entities(); ++i)
46 {
47 const auto& idx = index_set[i];
48 os << sind << idx[0];
49 for(int j(1); j < index_set.num_indices; ++j)
50 os << ' ' << idx[j];
51 os << std::endl;
52 }
53 os << sindent << "</Topology>" << std::endl;
54 }
55 };

244 template<typename Shape_, int num_coords_, typename Coord_>
245 void write_mesh(const ConformalMesh<Shape_, num_coords_, Coord_>& mesh)
246 {
247 typedef ConformalMesh<Shape_, num_coords_, Coord_> MeshType;
248 _os << _sindent << "<Mesh type=\"" << aux_meshtype_string(mesh) << "\"";
249 _os << " size=\"" << mesh.get_num_entities(0);
250 for(int i(1); i <= MeshType::shape_dim; ++i)
251 _os << " " << mesh.get_num_entities(i);
252 _os << "\">" << std::endl;
253 _push_indent();
254
255 _write_vertex_set(mesh.get_vertex_set());
256 Intern::TopoWriteHelper<Shape_>::write_topology(_os, mesh.get_index_set_holder(), _sindent, false);
257
258 _pop_indent();
259 _os << _sindent << "</Mesh>" << std::endl;
260 }

277 template<typename Mesh_>
278 void write_meshpart(const MeshPart<Mesh_>& meshpart, const String& parent_name, const String& part_name, const String& chart_name)
279 {
280 typedef typename Mesh_::ShapeType ShapeType;
281
282 _os << _sindent << "<MeshPart";
283 _os << " name=\"" << part_name << "\"";
284 _os << " parent=\"" << parent_name << "\"";
285 if(!chart_name.empty())
286 _os << " chart=\"" << chart_name << "\"";
287 _os << " topology=\"" << (meshpart.has_topology() ? "full" : "none") << "\"";
288 _os << " size=\"" << meshpart.get_num_entities(0);
289 for(int i(1); i <= ShapeType::dimension; ++i)
290 _os << " " << meshpart.get_num_entities(i);
291 _os << "\">" << std::endl;
292 _push_indent();
293
294 // write mapping
295 Intern::MappWriteHelper<ShapeType>::write_mapping(_os, meshpart.get_target_set_holder(), _sindent, true);
296
297 // write topology
298 if(meshpart.has_topology())
299 Intern::TopoWriteHelper<ShapeType>::write_topology(_os, *meshpart.get_topology(), _sindent, true);
300
301 // write attributes
302 const auto& attrs = meshpart.get_mesh_attributes();
303 for(auto it = attrs.begin(); it != attrs.end(); ++it)
304 _write_attribute(*(it->second), it->first);
305
306 _pop_indent();
307 _os << _sindent << "</MeshPart>" << std::endl;
308 }

316 void write_partition(const Partition& partition)
317 {
318 _os << _sindent << "<Partition";
319 if(!partition.get_name().empty())
320 _os << " name=\"" << partition.get_name() << "\"";
321 _os << " priority=\"" << partition.get_priority() << "\"";
322 _os << " level=\"" << partition.get_level() << "\"";
323 _os << " size=\"" << partition.get_num_patches() << ' ' << partition.get_num_elements() << "\"";
324 _os << ">" << std::endl;
325 _push_indent();
326
327 // loop over all patches
328 const Adjacency::Graph& graph = partition.get_patches();
329 for(Index i(0); i < graph.get_num_nodes_domain(); ++i)
330 {
331 _os << _sindent << "<Patch rank=\"" << i << "\" size=\"" << graph.degree(i) << "\">" << std::endl;
332 _push_indent();
333 for(auto it = graph.image_begin(i); it != graph.image_end(i); ++it)
334 _os << _sindent << (*it) << std::endl;
335 _pop_indent();
336 _os << _sindent << "</Patch>" << std::endl;
337 }
338
339 _pop_indent();
340 _os << _sindent << "</Partition>" << std::endl;
341 }

374 template<typename RootMesh_>
375 void write(
376 const RootMeshNode<RootMesh_>* mesh_node,
377 const MeshAtlas<RootMesh_>* mesh_atlas = nullptr,
378 const PartitionSet* part_set = nullptr,
379 bool skip_internal_meshparts = true)
380 {
381 const RootMesh_* root_mesh(nullptr);
382 if(mesh_node != nullptr)
383 root_mesh = mesh_node->get_mesh();
384
385 // print root markup
386 _os << "<FeatMeshFile version=\"1\"";
387 if(root_mesh != nullptr)
388 _os << " mesh=\"" << aux_meshtype_string(*root_mesh) << "\"";
389 _os << ">" << std::endl;
390
391 // increase indent
392 _push_indent();
393
394 // write mesh atlas
395 if(mesh_atlas != nullptr)
396 {
397 // write atlas
398 write_atlas(*mesh_atlas);
399 }
400
401 // write mesh node
402 if(mesh_node != nullptr)
403 {
404 // write mesh
405 if(root_mesh != nullptr)
406 write_mesh(*root_mesh);
407
408 // write meshparts
409 std::deque<String> part_names = mesh_node->get_mesh_part_names();
410 for(auto it = part_names.begin(); it != part_names.end(); ++it)
411 {
412 // skip mesh parts starting with an underscore;
413 // these are reserved for internal use and should not be exported
414 // unless explicitly specified by the caller
415 if(skip_internal_meshparts && it->starts_with('_'))
416 continue;
417
418 const MeshPart<RootMesh_>* meshpart = mesh_node->find_mesh_part(*it);
419 String chart_name = mesh_node->find_mesh_part_chart_name(*it);
420 write_meshpart(*meshpart, "root", *it, chart_name);
421 }
422 }
423
424 // write partitions
425 if(part_set != nullptr)
426 {
427 write_partition_set(*part_set);
428 }
429
430 _pop_indent();
431
432 // write terminator
433 _os << "</FeatMeshFile>" << std::endl;
434 }

468 template<typename Data_>
469 void _write_attribute(const AttributeSet<Data_>& attr, const String& name)
470 {
471 _os << _sindent << "<Attribute";
472 _os << " name=\"" << name << "\"";
473 _os << " dim=\"" << attr.get_dimension() << "\"";
474 _os << ">" << std::endl;
475 _push_indent();
476 for(Index i(0); i < attr.get_num_values(); ++i)
477 {
478 _os << _sindent << attr(i, 0);
479 for(int j(1); j < attr.get_dimension(); ++j)
480 _os << ' ' << attr(i, j);
481 _os << std::endl;
482 }
483 _pop_indent();
484 _os << _sindent << "</Attribute>" << std::endl;
