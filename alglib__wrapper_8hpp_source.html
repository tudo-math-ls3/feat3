344 protected:
358 virtual Status _apply_intern(VectorType& vec_sol)
359 {
360 Statistics::add_solver_expression(std::make_shared<ExpressionStartSolve>(this->name()));
361
362 // Write initial guess to iterates if desired
363 if(iterates != nullptr)
364 {
365 auto tmp = vec_sol.clone();
366 iterates->push_back(std::move(tmp));
367 }
368
369 // compute initial defect
370 Status status = this->_set_initial_defect(this->_vec_def, vec_sol);
371
372 if(status != Status::progress)
373 {
374 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), status, this->get_num_iter()));
375 return status;
376 }
377
378 // Copy initial guess to optimization state variable
379 auto vec_sol_elements = Intern::derefer<VectorType>(vec_sol, nullptr).template elements<LAFEM::Perspective::pod>();
380 for(alglib::ae_int_t i(0); i < _functionalt_var.length(); ++i)
381 {
382 _functionalt_var[i] = vec_sol_elements[i];
383 }
384
385 alglib::minlbfgsrestartfrom(_state, _functionalt_var);
386
387 IterationStats stat(*this);
388 alglib::minlbfgsoptimize(_state, _func_grad, _log, this);
389 alglib::minlbfgsresults(_state, _functionalt_var, _report);
390
391 for(alglib::ae_int_t i(0); i < _functionalt_var.length(); ++i)
392 {
393 vec_sol_elements[i] = DataType(_functionalt_var[i]);
394 }
395
396 switch(_report.terminationtype)
397 {
398 case(-8):
399 //std::cout << "ALGLIB: Got inf or NaN in function/gradient evaluation." << std::endl;
400 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
401 return Status::aborted;
402 case(-7):
403 //std::cout << "ALGLIB: Gradient verification failed." << std::endl;
404 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
405 return Status::aborted;
406 case(1):
407 //std::cout << "ALGLIB: Function value improvement criterion fulfilled." << std::endl;
408 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::success, this->get_num_iter()));
409 return Status::success;
410 case(2):
411 //std::cout << "ALGLIB: Update step size stagnated." << std::endl;
412 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::success, this->get_num_iter()));
413 return Status::success;
414 case(4):
415 //std::cout << "ALGLIB: Gradient norm criterion fulfilled." << std::endl;
416 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::success, this->get_num_iter()));
417 return Status::success;
418 case(5):
419 //std::cout << "ALGLIB: Maximum number of iterations" << std::endl;
420 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::max_iter, this->get_num_iter()));
421 return Status::max_iter;
422 case(7):
423 //std::cout << "ALGLIB: Stopping criteria too stringent, further improvement impossible." << std::endl;
424 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::stagnated, this->get_num_iter()));
425 return Status::stagnated;
426 case(8):
427 //std::cout << "ALGLIB: Stopped by user" << std::endl;
428 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::success, this->get_num_iter()));
429 return Status::success;
430 default:
431 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::undefined, this->get_num_iter()));
432 return Status::undefined;
433 }

449 static void _log(const alglib::real_1d_array& DOXY(x), double DOXY(func), void* ptr)
450 {
451 ALGLIBMinLBFGS<FunctionalType, FilterType>* me =
452 reinterpret_cast<ALGLIBMinLBFGS<FunctionalType, FilterType>*>(ptr);
453
454 // Because of how ALGLIB counts its iterations, we have to make sure we do not call this before the first
455 // functional evaluation (repnfev is the reported number of functional evaluations)
456 if(me->_state.c_ptr()->repnfev> 0)
457 {
458 me->_steplength = DataType(me->_state.c_ptr()->stp);
459 me->_ls_its = Index(me->_state.c_ptr()->nfev);
460 Status st = me->_set_new_defect(me->_vec_def, me->_vec_tmp);
461 // Because ALGLIB knows no relative tolerance, we have to do it here
462 if( st != Status::progress )
463 {
464 alglib::minlbfgsrequesttermination(me->_state);
465 }
466 }
467
468 }
469
486 static void _func_grad(const alglib::real_1d_array& x, double& fval, alglib::real_1d_array& grad, void* ptr)
487 {
488 // Downcast because we know what we are doing, right?
489 ALGLIBMinLBFGS<FunctionalType, FilterType>* me = reinterpret_cast<ALGLIBMinLBFGS<FunctionalType, FilterType>*>
490 (ptr);
491
492 auto vec_tmp_elements = Intern::derefer<VectorType>(me->_vec_tmp, nullptr).template elements<LAFEM::Perspective::pod>();
493
494 // Copy back ALGLIB's state variable to our solver
495 for(alglib::ae_int_t i(0); i < x.length(); ++i)
496 vec_tmp_elements[i] = DataType(x[i]);
497
498 me->_fval_prev = me->_fval;
499 // Prepare the functional
500 me->_functional.prepare(me->_vec_tmp, me->_filter);
501 // Compute functional value and gradient
502 me->_functional.eval_fval_grad(me->_fval, me->_vec_def);
503 me->_filter.filter_def(me->_vec_def);
504
505 fval = double(me->_fval);
506
507 // Copy the functional's gradient to ALGLIB's grad variable
508 auto vec_def_elements = Intern::derefer<VectorType>(me->_vec_def, nullptr).template elements<LAFEM::Perspective::pod>();
509 for(alglib::ae_int_t i(0); i < grad.length(); ++i)
510 {
511 grad[i] = double(vec_def_elements[i]);
512 }
513
514 }

1012 static void _func_grad(const alglib::real_1d_array& x, double& fval, alglib::real_1d_array& grad, void* ptr)
1013 {
1014 // Downcast because we know what we are doing, right?
1015 ALGLIBMinCG<FunctionalType, FilterType>* me = reinterpret_cast<ALGLIBMinCG<FunctionalType, FilterType>*>(ptr);
1016
1017 auto vec_tmp_elements = Intern::derefer<VectorType>(me->_vec_tmp, nullptr).template
1018 elements<LAFEM::Perspective::pod>();
1019
1020 // Copy back ALGLIB's state variable to our solver
1021 for(alglib::ae_int_t i(0); i < x.length(); ++i)
1022 {
1023 vec_tmp_elements[i] = DataType(x[i]);
1024 }
1025
1026 me->_fval_prev = me->_fval;
1027 // Prepare the functional
1028 me->_functional.prepare(me->_vec_tmp, me->_filter);
1029 // Compute functional value and gradient
1030 me->_functional.eval_fval_grad(me->_fval, me->_vec_def);
1031 me->_filter.filter_def(me->_vec_def);
1032
1033 fval = double(me->_fval);
1034
1035 // Copy the functional's gradient to ALGLIB's grad variable
1036 auto vec_def_elements = Intern::derefer<VectorType>(me->_vec_def, nullptr).template
1037 elements<LAFEM::Perspective::pod>();
1038
1039 for(alglib::ae_int_t i(0); i < grad.length(); ++i)
1040 {
1041 grad[i] = double(vec_def_elements[i]);
1042 }
1043
1044 }
