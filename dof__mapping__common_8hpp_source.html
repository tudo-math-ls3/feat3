82 typename Space_,
83 int codim_,
84 int dofs_per_cell_ = 1>
85 class DofMappingSingleEntity :
86 public DofMappingBase<Space_>
87 {
88 public:
90 typedef Space_ SpaceType;
92 typedef typename SpaceType::ShapeType ShapeType;
93
95 static constexpr int shape_dim = ShapeType::dimension;
97 static constexpr int dof_dim = shape_dim - codim_;
98
99 // make sure the dof-dimension is less than the shape dimension
100 static_assert((codim_ >= 0) && (codim_ <= shape_dim), "invalid co-dimension");
101
102 protected:
104 typedef typename SpaceType::TrafoType TrafoType;
106 typedef typename TrafoType::MeshType MeshType;
108 typedef typename MeshType::template IndexSet<shape_dim, dof_dim>::Type IndexSetType;
109
110 protected:
112 const IndexSetType& _index_set;
113
114 public:
116 explicit DofMappingSingleEntity(const Space_& space) :
117 DofMappingBase<Space_>(space),
118 _index_set(space.get_trafo().get_mesh().template get_index_set<shape_dim, dof_dim>())
119 {
120 }
121
123 int get_num_local_dofs() const
124 {
125 return IndexSetType::num_indices * dofs_per_cell_;
126 }
127
129 Index get_num_global_dofs() const
130 {
131 return this->_space.get_trafo().get_mesh().get_num_entities(dof_dim) * Index(dofs_per_cell_);
132 }
133
135 Index get_index(int local_dof_idx) const
136 {
137 int ldi_q = local_dof_idx / dofs_per_cell_;
138 int ldi_r = local_dof_idx % dofs_per_cell_;
139 return Index(dofs_per_cell_) * _index_set(this->_cell_index, ldi_q) + Index(ldi_r);
140 }
141 };

189 template<
190 typename Space_,
191 template<typename Tag_, int dim_> class DofTraits_,
192 typename DofTag_>
193 class DofMappingUniform
194 : public DofMappingBase<Space_>
195 {
196 public:
198 typedef DofMappingBase<Space_> BaseClass;
200 typedef Space_ SpaceType;
202 typedef typename SpaceType::ShapeType ShapeType;
203
204 // total number of local dofs
205 static constexpr int dof_count = Intern::UniformDofMappingHelper<ShapeType, DofTraits_, DofTag_>::dof_count;
206
207 private:
209 Index dof_idx[dof_count];
210
211 public:
212 explicit DofMappingUniform(const SpaceType& space) :
213 BaseClass(space)
214 {
215 }
216
218 void prepare(Index cell_index)
219 {
220 BaseClass::prepare(cell_index);
221 Index count(0);
222 Intern::UniformDofMappingHelper<ShapeType, DofTraits_, DofTag_>
223 ::assemble(dof_idx, this->_space.get_mesh(), cell_index, count);
224 XASSERTM(count == Index(dof_count), "dof-count mismatch");
225 }
226
228 int get_num_local_dofs() const
229 {
230 return dof_count;
231 }
232
234 Index get_num_global_dofs() const
235 {
236 return Intern::UniformDofMappingHelper<ShapeType, DofTraits_, DofTag_>::global_count(this->_space.get_mesh());
237 }
238
240 Index get_index(int local_dof_idx) const
241 {
242 ASSERTM((local_dof_idx >= 0) && (local_dof_idx < dof_count), "local dof-index out-of-range");
243 return dof_idx[local_dof_idx];
244 }
245 };

250 template<typename Shape_, template<typename, int> class Traits_, typename Tag_, int cell_dim_, int shape_dim_>
251 struct UniformDofMappingHelper
252 {
253 static_assert(cell_dim_ < shape_dim_, "invalid cell dimension");
254
255 static constexpr int cell_count = Shape::FaceTraits<Shape_, cell_dim_>::count;
256 // number of dofs per cell
257 static constexpr int dofs_per_cell = Traits_<Tag_, cell_dim_>::count;
258 // number of dofs for all cells of this dimension
259 static constexpr int dof_cell_count = cell_count * dofs_per_cell;
260 // total dof count
261 static constexpr int dof_count = UniformDofMappingHelper<Shape_, Traits_, Tag_, cell_dim_-1>::dof_count + dof_cell_count;
262
263 template<typename MeshType_>
264 static Index assemble(Index dof_idx[], const MeshType_& mesh, Index cell, Index& k)
265 {
266 Index offs(UniformDofMappingHelper<Shape_, Traits_, Tag_, cell_dim_-1>::assemble(dof_idx, mesh, cell, k));
267
268 const auto& index_set(mesh.template get_index_set<shape_dim_, cell_dim_>());
269 for(int i(0); i < cell_count; ++i)
270 {
271 for(int j(0); j < dofs_per_cell; ++j, ++k)
272 {
273 dof_idx[k] = offs + index_set(cell,i) * Index(dofs_per_cell) + Index(j);
274 }
275 }
276 return offs + Index(dofs_per_cell) * mesh.get_num_entities(cell_dim_);
277 }
278
279 template<typename MeshType_>
280 static Index global_count(const MeshType_& mesh)
281 {
282 return UniformDofMappingHelper<Shape_, Traits_, Tag_, cell_dim_-1>::global_count(mesh) +
283 Index(Traits_<Tag_, cell_dim_>::count) * mesh.get_num_entities(cell_dim_);
284 }
285 };
