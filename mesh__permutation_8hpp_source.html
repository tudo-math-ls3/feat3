58 template<typename Coord_, int dim_>
59 class LexiPoint
60 {
61 public:
63 Index idx;
65 Tiny::Vector<Coord_, dim_> v;
66
67 // tolerance for coordinate equality check
68 static constexpr Coord_ tol_ = Coord_(1e-6);
69
70 LexiPoint()
71 {
72 }
73
74 explicit LexiPoint(Index _idx) :
75 idx(_idx),
76 v(Coord_(0))
77 {
78 }
79
80 template<int s_>
81 explicit LexiPoint(Index _idx, const Tiny::Vector<Coord_, dim_, s_>& _vtx) :
82 idx(_idx),
83 v(_vtx)
84 {
85 }
86
87 void format()
88 {
89 v.format();
90 }
91
92 template<int s_>
93 void add(const Tiny::Vector<Coord_, dim_, s_>& vtx)
94 {
95 v += vtx;
96 }
97
98 bool operator<(const LexiPoint& other) const
99 {
100 // loop over the higher dimension coordinates
101 for(int i(dim_-1); i > 0; --i)
102 {
103 if(v[i] + tol_ < other.v[i])
104 return true;
105 if(other.v[i] + tol_ <= v[i])
106 return false;
107 }
108 // all coordinates of dimension > 1 are equal, so check the X-coordinate
109 return v[0] < other.v[0];
110 }
111 }; // class LexiPoint

114 template<typename Shape_, int shape_dim_ = Shape_::dimension>
115 class LexiPermuter
116 {
117 public:
118 template<int nc_, typename Coord_>
119 static void compute(
120 std::array<Adjacency::Permutation, Shape_::dimension + 1>& perms,
121 const IndexSetHolder<Shape_>& ish, const VertexSet<nc_, Coord_>& vtx)
122 {
123 // recurse down
124 LexiPermuter<Shape_, shape_dim_-1>::compute(perms, ish, vtx);
125
126 // get the vertices-at-shape index set
127 const auto& idx = ish.template get_index_set<shape_dim_, 0>();
128 const Index n = idx.get_num_entities();
129 const int nidx = idx.get_num_indices();
130
131 std::vector<LexiPoint<Coord_, nc_>> vset(n);
132
133 // loop over all entities and accumulate all vertices adjacent to that entity
134 for(Index i(0); i < n; ++i)
135 {
136 LexiPoint<Coord_, nc_> v(i);
137 for(int j(0); j < nidx; ++j)
138 v.add(vtx[idx(i,j)]);
139 vset.at(i) = v;
140 }
141
142 // sort lexicographically
143 std::sort(vset.begin(), vset.end());
144
145 // create permutation
146 Adjacency::Permutation p(n);
147 Index* vp = p.get_perm_pos();
148 for(auto it = vset.begin(); it != vset.end(); ++it, ++vp)
149 *vp = it->idx;
150 p.calc_swap_from_perm();
151 perms.at(shape_dim_) = std::move(p);
152 }
153 }; // class LexiPermuter
154
156 template<typename Shape_>
157 class LexiPermuter<Shape_, 0>
158 {
159 public:
160 template<int nc_, typename Coord_>
161 static void compute(
162 std::array<Adjacency::Permutation, Shape_::dimension + 1>& perms,
163 const IndexSetHolder<Shape_>&, const VertexSet<nc_, Coord_>& vtx)
164 {
165 const Index n = vtx.get_num_vertices();
166 std::vector<LexiPoint<Coord_, nc_>> vset(n);
167
168 // loop over all entities and accumulate all vertices adjacent to that entity
169 for(Index i(0); i < n; ++i)
170 {
171 vset.at(i) = LexiPoint<Coord_, nc_>(i, vtx[i]);
172 }
173
174 // sort lexicographically
175 std::sort(vset.begin(), vset.end());
176
177 // create permutation
178 Adjacency::Permutation p(n);
179 Index* vp = p.get_perm_pos();
180 for(auto it = vset.begin(); it != vset.end(); ++it, ++vp)
181 *vp = it->idx;
182 p.calc_swap_from_perm();
183 perms.front() = std::move(p);
184 }
185 }; // class LexiPermuter<...,0>

586 template<int num_coords_, typename Coord_>
587 void create(PermutationStrategy strategy, const IndexSetHolder<Shape_>& ish, const VertexSet<num_coords_, Coord_>& vtx)
588 {
589 switch(strategy)
590 {
591 case PermutationStrategy::none:
592 break;
593
594 case PermutationStrategy::other:
595 XABORTM("use the MeshPermutation::create_other() function");
596 break;
597
598 case PermutationStrategy::random:
599 create_random(ish);
600 break;
601
602 case PermutationStrategy::lexicographic:
603 create_lexicographic(ish, vtx);
604 break;
605
606 case PermutationStrategy::colored:
607 create_colored(ish);
608 break;
609
610 case PermutationStrategy::cuthill_mckee:
611 create_cmk(ish, false);
612 break;
613
614 case PermutationStrategy::cuthill_mckee_reversed:
615 create_cmk(ish, true);
616 break;
617
618 case PermutationStrategy::geometric_cuthill_mckee:
619 create_gcmk(ish, vtx, false);
620 break;
621
622 case PermutationStrategy::geometric_cuthill_mckee_reversed:
623 create_gcmk(ish, vtx, true);
624 break;
625 }
626 }

679 Index num_entities[shape_dim+1];

701 template<int num_coords_, typename Coord_>
702 void create_lexicographic(const IndexSetHolder<Shape_>& ish, const VertexSet<num_coords_, Coord_>& vtx)
703 {
704 XASSERTM(this->_strategy == PermutationStrategy::none, "permutation already created!");
705
706 // call the actual helper class
707 Intern::LexiPermuter<ShapeType>::compute(this->_perms, ish, vtx);
708
709 // compute inverse permutations
710 for(std::size_t dim(0); dim <= std::size_t(shape_dim); ++dim)
711 {
712 _inv_perms.at(dim) = _perms.at(dim).inverse();
713 }
714
715 // save strategy
716 this->_strategy = PermutationStrategy::lexicographic;
717 }
718
732 void create_colored(const IndexSetHolder<Shape_>& ish)
733 {
734 XASSERTM(this->_strategy == PermutationStrategy::none, "permutation already created!");
735
736 const auto& verts_at_elem = ish.template get_index_set<shape_dim, 0>();
737 Adjacency::Graph elems_at_vert(Adjacency::RenderType::transpose, verts_at_elem);
738 Adjacency::Graph elems_at_elem(Adjacency::RenderType::injectify, verts_at_elem, elems_at_vert);
739
740 // create coloring
741 Adjacency::Coloring col(elems_at_elem);
742
743 // create coloring partition graph
744 Adjacency::Graph cparti = col.create_partition_graph();
745
746 // get the total number of colors
747 const Index num_elems = elems_at_elem.get_num_nodes_domain();
748 const Index num_colors = cparti.get_num_nodes_domain();
749
750 XASSERT(num_elems == cparti.get_num_indices());
751
752 // create element permutation
753 _perms.back() = Adjacency::Permutation(num_elems, Adjacency::Permutation::type_perm, cparti.get_image_idx());
754 _inv_perms.back() = _perms.back().inverse();
755
756 // store element coloring
757 const Index* dom_ptr = cparti.get_domain_ptr();
758 this->_element_coloring.assign(dom_ptr, &dom_ptr[num_colors]);
759
760 // save strategy
761 this->_strategy = PermutationStrategy::colored;
762 }

780 void create_cmk(const IndexSetHolder<Shape_>& ish, bool reverse)
781 {
782 XASSERTM(this->_strategy == PermutationStrategy::none, "permutation already created!");
783
784 const auto& verts_at_elem = ish.template get_index_set<shape_dim, 0>();
785 Adjacency::Graph elems_at_vert(Adjacency::RenderType::transpose, verts_at_elem);
786 Adjacency::Graph elems_at_elem(Adjacency::RenderType::injectify, verts_at_elem, elems_at_vert);
787
788 // create Cuthill-McKee permutation
789 _perms.back() = Adjacency::CuthillMcKee::compute(this->_element_layering, elems_at_elem, reverse,
790 Adjacency::CuthillMcKee::root_minimum_degree, Adjacency::CuthillMcKee::sort_asc);
791 _inv_perms.back() = _perms.back().inverse();
792
793 // save strategy
794 this->_strategy = (reverse ? PermutationStrategy::cuthill_mckee_reversed : PermutationStrategy::cuthill_mckee);
795 }

820 template<int num_coords_, typename Coord_>
821 void create_gcmk(const IndexSetHolder<Shape_>& ish, const VertexSet<num_coords_, Coord_>& DOXY(vtx), bool reverse)
822 {
823 // render all <k-faces>-at-element index sets into graph array for convenience
824 std::array<Adjacency::Graph, shape_dim> idx_set;
825 Intern::ISWRenderer<Shape_>::render(idx_set, ish.template get_index_set_wrapper<shape_dim>());
826
827 // transpose vertices-at-element graph
828 Adjacency::Graph& verts_at_elem = idx_set.front();
829 Adjacency::Graph elems_at_vert(Adjacency::RenderType::transpose, verts_at_elem);
830
831 // get the number of entities in the mesh
832 Index num_entities[shape_dim+1];
833 NumEntitiesExtractor<shape_dim>::set_num_entities_with_numverts(ish, num_entities);
834
835 // get number of vertices and elements
836 const Index num_verts = num_entities[0];
837 const Index num_elems = num_entities[shape_dim];
838
839 // allocate temporary permutation vectors
840 std::array<std::vector<Index>, shape_dim+1> perms;
841 for(std::size_t i(0); i <= std::size_t(shape_dim); ++i)
842 perms.at(i).reserve(num_entities[i]);
843
844 // create mask vectors and initialize them to 0
845 // a mask value != 0 indicates that the entity has already been processed
846 std::array<std::vector<int>, shape_dim+1> masks;
847 for(std::size_t i(0); i <= std::size_t(shape_dim); ++i)
848 masks.at(i).resize(num_entities[i]);
849
850 // initialize layer vectors
851 std::vector<Index> layer_cur, layer_vert, layer_next;
852 layer_cur.reserve(num_elems);
853 layer_next.reserve(num_elems);
854 layer_vert.reserve(num_verts);
855
856 // initialize element layering
857 this->_element_layering.clear();
858 this->_element_layering.push_back(Index(0));
859
860 // loop until all elements are processed; this loop performs 1 iteration per domain
861 // connectivity region (so usually 1), but it may perform several iterations in case the
862 // elements of this domain are not connected -- this may happen e.g. in the MPI parallel
863 // case where our local domain is actually just a partition of the whole (connected) domain
864 while(Index(perms.back().size()) < num_elems)
865 {
866 layer_cur.clear();
867
868 // pick a root element of minimal degree (least amount of neighbors)
869 // this chosen root element will usually be a boundary or even a corner element
870 Index idx = _pick_gcmk_root(idx_set.front(), elems_at_vert, masks.back());
871 layer_cur.push_back(idx);
872 masks.back()[idx] = 1;
873
874 // loop through the current element layer
875 while(!layer_cur.empty())
876 {
877 // loop over all elements in current layer
878 layer_next.clear();
879 for(auto it = layer_cur.begin(); it != layer_cur.end(); ++it)
880 {
881 // get and add current element
882 const Index cur_elem = *it;
883 perms.back().push_back(cur_elem);
884
885 // add all vertices adjacent to the current element to the next vertex layer
886 layer_vert.clear();
887 for(auto kt = verts_at_elem.image_begin(cur_elem); kt != verts_at_elem.image_end(cur_elem); ++kt)
888 {
889 const Index cur_vert = *kt;
890 // already processed?
891 if(masks.front()[cur_vert] != 0)
892 continue;
893 // add layer vertex
894 layer_vert.push_back(cur_vert);
895 perms.front().push_back(cur_vert);
896 masks.front()[cur_vert] = 1;
897 }
898
899 // add all k-faces (edges, faces, ...) adjacent to the current element to the next k-face layer
900 for(std::size_t k(1); k < std::size_t(shape_dim); ++k)
901 {
902 for(auto kt = idx_set.at(k).image_begin(cur_elem); kt != idx_set.at(k).image_end(cur_elem); ++kt)
903 {
904 const Index cur_face = *kt;
905 if(masks.at(k)[cur_face] != 0)
906 continue;
907 perms.at(k).push_back(cur_face);
908 masks.at(k)[cur_face] = 1;
909 }
910 }
911
912 // add all elements adjacent via a vertex to the next element layer
913 for(auto jt = layer_vert.begin(); jt != layer_vert.end(); ++jt)
914 {
915 const Index cur_vert = *jt;
916 for(auto kt = elems_at_vert.image_begin(cur_vert); kt != elems_at_vert.image_end(cur_vert); ++kt)
917 {
918 const Index next_elem = *kt;
919 if(masks.back()[next_elem] != 0)
920 continue;
921 layer_next.push_back(next_elem);
922 masks.back()[next_elem] = 1;
923 }
924 }
925 }
926
927 // continue with next element layer
928 layer_cur.swap(layer_next);
929
930 // store element layer size
931 this->_element_layering.push_back(Index(perms.back().size()));
932
933 } // while(!layer_cur.empty())
934 } // while(perms.back().size() < num_elems)
935
936 // revert if desired
937 if(reverse)
938 {
939 // reverse permutations
940 for(std::size_t dim(0); dim <= std::size_t(shape_dim); ++dim)
941 {
942 for(std::size_t i(0), j(perms[dim].size()-1); i < j; ++i, --j)
943 std::swap(perms[dim][i], perms[dim][j]);
944 }
945
946 // reverse layers
947 const Index lay_back = Index(this->_element_layering.size()) - 1u;
948 for(Index k(lay_back); k > 0u; --k)
949 {
950 // compute layer sizes from offsets
951 this->_element_layering.at(k) -= this->_element_layering.at(k-1u);
952 }
953 for(Index k(1u), l(lay_back); k < l; ++k, --l)
954 {
955 // reverse layer sizes
956 std::swap(this->_element_layering.at(k), this->_element_layering.at(l));
957 }
958 for(Index k(1u); k <= lay_back; ++k)
959 {
960 // compute layering offsets from sizes
961 this->_element_layering.at(k) += this->_element_layering.at(k-1u);
962 }
963 }
964
965 // create the actual permutation objects
966 for(std::size_t dim(0); dim <= std::size_t(shape_dim); ++dim)
967 {
968 // sanity check: all entities should have been processed
969 XASSERT(Index(perms.at(dim).size()) == num_entities[dim]);
970
971 // create actual permutation and its inverse
972 _perms.at(dim) = Adjacency::Permutation(Index(perms.at(dim).size()),
973 Adjacency::Permutation::type_perm, perms.at(dim).data());
974 _inv_perms.at(dim) = _perms.at(dim).inverse();
975 }
976
977 // save strategy
978 this->_strategy = (reverse ? PermutationStrategy::geometric_cuthill_mckee_reversed : PermutationStrategy::geometric_cuthill_mckee);
979 }
980
984 void create_inverse_permutations()
985 {
986 for(std::size_t dim(0); dim <= std::size_t(shape_dim); ++dim)
987 {
988 if(_perms.at(dim).empty())
989 _inv_perms.at(dim) = Adjacency::Permutation();
990 else
991 _inv_perms.at(dim) = _perms.at(dim).inverse();
992 }
993 }

1005 int validate_sizes(const Index* num_entities) const
1006 {
1007 XASSERT(num_entities != nullptr);
1008
1009 // check permutations
1010 for(std::size_t i(0); i <= std::size_t(shape_dim); ++i)
1011 {
1012 const Adjacency::Permutation& perm = this->_perms.at(i);
1013 const Adjacency::Permutation& iperm = this->_inv_perms.at(i);
1014
1015 if(!perm.empty() && (perm.size() != num_entities[i]))
1016 return 2*int(i)+1;
1017 if(!iperm.empty() && (iperm.size() != num_entities[i]))
1018 return 2*int(i)+2;
1019 }
1020
1021 // ok
1022 return 0;
1023 }

1025 protected:
1029 static Index _pick_gcmk_root(const Adjacency::Graph& v_at_e, const Adjacency::Graph& e_at_v,
1030 const std::vector<int>& mask)
1031 {
1032 std::set<Index> iset;
1033
1034 // pick a root node of minimal degree
1035 Index deg = v_at_e.get_num_nodes_domain() + e_at_v.get_num_nodes_domain();
1036 Index idx(0);
1037 for(Index i(0); i < v_at_e.get_num_nodes_domain(); ++i)
1038 {
1039 // skip masked elements
1040 if(mask[i] != 0)
1041 continue;
1042
1043 // compute degree
1044 iset.clear();
1045 for(auto it = v_at_e.image_begin(i); it != v_at_e.image_end(i); ++it)
1046 for(auto jt = e_at_v.image_begin(*it); jt != e_at_v.image_end(*it); ++jt)
1047 iset.insert(*jt);
1048
1049 // is this the new minimum?
1050 Index d = Index(iset.size());
1051 if(d < deg)
1052 {
1053 idx = i;
1054 deg = d;
1055 }
1056 }
1057
1058 return idx;
1059 }
