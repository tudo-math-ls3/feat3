238 template<typename SubChart_>
239 bool try_extrude_chart(std::unique_ptr<HexaChart>& hexa_chart, const QuadChart& quad_chart) const
240 {
241 // try down-cast to sub-chart
242 const SubChart_* sub_chart = dynamic_cast<const SubChart_*>(&quad_chart);
243 if(sub_chart == nullptr)
244 return false;
245
246 // success; create the extruded hexa chart
247 std::unique_ptr<Atlas::Extrude<HexaMesh, SubChart_>> the_chart(
248 new Atlas::Extrude<HexaMesh, SubChart_>(
249 std::unique_ptr<SubChart_>(new SubChart_(*sub_chart))));
250
251 // set our transformation
252 the_chart->set_origin(_origin_x, _origin_y);
253 the_chart->set_offset(_offset_x, _offset_y, _offset_z);
254 the_chart->set_angles(_angle_y, _angle_p, _angle_r);
255
256 hexa_chart = std::move(the_chart);
257 return true;
258 }

272 std::unique_ptr<HexaChart> extrude_chart(const QuadChart& quad_chart) const
273 {
274 // The following ugly piece of code is a trial-&-error sequence using
275 // dynamic-cast to figure out the actual type of the 2D chart.
276 // That's not an elegant solution, but I don't have any other ideas...
277
278 std::unique_ptr<HexaChart> hexa_chart;
279
280 // Is it a Circle chart?
281 if(this->template try_extrude_chart<Atlas::Circle<QuadMesh>>(hexa_chart, quad_chart))
282 return hexa_chart;
283
284 // Is it a Bezier chart?
285 if(this->template try_extrude_chart<Atlas::Bezier<QuadMesh>>(hexa_chart, quad_chart))
286 return hexa_chart;
287
288 // If we come out here then we failed to extrude the 2D chart for some reason...
289 XABORTM("Could not extrude 2D chart to 3D");
290 }

332 void extrude_vertex_set(HexaVertexSet& hexa_vtx, const QuadVertexSet& quad_vtx) const
333 {
334 // get number of source vertices
335 const Index quad_num_verts = quad_vtx.get_num_vertices();
336
337 // validate vertex count in hexa vertex set
338 XASSERT(hexa_vtx.get_num_vertices() == (quad_num_verts * (_slices+1)));
339
340 // compute rigid rotation matrix
341 Tiny::Matrix<CoordType, 3, 3> rotation;
342 rotation.set_rotation_3d(_angle_y, _angle_p, _angle_r);
343 Tiny::Vector<CoordType, 3> tmp1, tmp2;
344
345 // loop over all slices
346 for(Index j(0); j <= _slices; ++j)
347 {
348 // compute slice index offset
349 const Index vo = j * quad_num_verts;
350
351 // compute slice z-coord
352 //const CoordType z = _z_min + (CoordType(j) / CoordType(_slices))*(_z_max - _z_min);
353 const CoordType z = _z_list.at(j);
354
355 // loop over all vertices
356 for(Index i(0); i < quad_num_verts; ++i)
357 {
358 // apply our rigid transformation
359 tmp1[0] = quad_vtx[i][0] - _origin_x;
360 tmp1[1] = quad_vtx[i][1] - _origin_y;
361 tmp1[2] = z;
362 tmp2.set_mat_vec_mult(rotation, tmp1);
363 hexa_vtx[vo+i][0] = tmp2[0] + _offset_x;
364 hexa_vtx[vo+i][1] = tmp2[1] + _offset_y;
365 hexa_vtx[vo+i][2] = tmp2[2] + _offset_z;
366 }
367 }

379 void extrude_topology(HexaTopology& hexa_topo, const QuadTopology& quad_topo) const
380 {
381 // fetch index sets
382 const auto& quad_edge = quad_topo.template get_index_set<1,0>();
383 const auto& quad_quad = quad_topo.template get_index_set<2,0>();
384 auto& hexa_edge = hexa_topo.template get_index_set<1,0>();
385 auto& hexa_quad = hexa_topo.template get_index_set<2,0>();
386 auto& hexa_hexa = hexa_topo.template get_index_set<3,0>();
387
388 const Index quad_num_entities[] =
389 {
390 quad_edge.get_index_bound(),
391 quad_edge.get_num_entities(),
392 quad_quad.get_num_entities()
393 };
394
395 // validate dimensions
396 XASSERT(hexa_edge.get_index_bound() == (quad_num_entities[0] * (_slices+1)));
397 XASSERT(hexa_edge.get_num_entities() == (quad_num_entities[0] * _slices + quad_num_entities[1] * (_slices+1)));
398 XASSERT(hexa_quad.get_num_entities() == (quad_num_entities[1] * _slices + quad_num_entities[2] * (_slices+1)));
399 XASSERT(hexa_hexa.get_num_entities() == (quad_num_entities[2] * _slices));
400
401 // loop over all slices (edges in x/y-direction)
402 for(Index j(0); j <= _slices; ++j)
403 {
404 // compute slice vertex index offset
405 const Index vo = j * quad_num_entities[0];
406
407 // compute slice edge index offset
408 const Index eo = j * quad_num_entities[1];
409
410 // loop over all edges
411 for(Index i(0); i < quad_num_entities[1]; ++i)
412 {
413 hexa_edge[eo+i][0] = vo + quad_edge[i][0];
414 hexa_edge[eo+i][1] = vo + quad_edge[i][1];
415 }
416 }
417
418 // loop over all slice intervals (edges in z-direction)
419 for(Index j(0); j < _slices; ++j)
420 {
421 // compute slice vertex index offset
422 const Index vo = j * quad_num_entities[0];
423
424 // compute slice edge index offset
425 const Index eo = j * quad_num_entities[0] + (_slices+1) * quad_num_entities[1];
426
427 // loop over all vertices
428 for(Index i(0); i < quad_num_entities[0]; ++i)
429 {
430 hexa_edge[eo+i][0] = vo + i;
431 hexa_edge[eo+i][1] = vo + i + quad_num_entities[0];
432 }
433 }
434
435 // loop over all slices (quads in x/y-direction)
436 for(Index j(0); j <= _slices; ++j)
437 {
438 // compute slice vertex index offset
439 const Index vo = j * quad_num_entities[0];
440
441 // compute slice quad index offset
442 const Index qo = j * quad_num_entities[2];
443
444 // loop over all quads
445 for(Index i(0); i < quad_num_entities[2]; ++i)
446 {
447 hexa_quad[qo+i][0] = vo + quad_quad[i][0];
448 hexa_quad[qo+i][1] = vo + quad_quad[i][1];
449 hexa_quad[qo+i][2] = vo + quad_quad[i][2];
450 hexa_quad[qo+i][3] = vo + quad_quad[i][3];
451 }
452 }
453
454 // loop over all slice intervals (quads in z-direction)
455 for(Index j(0); j < _slices; ++j)
456 {
457 // compute slice vertex index offset
458 const Index vo = j * quad_num_entities[0];
459
460 // compute slice quad index offset
461 const Index qo = j * quad_num_entities[1] + (_slices+1) * quad_num_entities[2];
462
463 // loop over all edges
464 for(Index i(0); i < quad_num_entities[1]; ++i)
465 {
466 hexa_quad[qo+i][0] = vo + quad_edge[i][0];
467 hexa_quad[qo+i][1] = vo + quad_edge[i][1];
468 hexa_quad[qo+i][2] = vo + quad_edge[i][0] + quad_num_entities[0];
469 hexa_quad[qo+i][3] = vo + quad_edge[i][1] + quad_num_entities[0];
470 }
471 }
472
473 // loop over all slices (quads in x/y-direction)
474 for(Index j(0); j < _slices; ++j)
475 {
476 // compute slice vertex index offset
477 const Index vo = j * quad_num_entities[0];
478
479 // compute slice quad index offset
480 const Index ho = j * quad_num_entities[2];
481
482 // loop over all quads
483 for(Index i(0); i < quad_num_entities[2]; ++i)
484 {
485 hexa_hexa[ho+i][0] = vo + quad_quad[i][0];
486 hexa_hexa[ho+i][1] = vo + quad_quad[i][1];
487 hexa_hexa[ho+i][2] = vo + quad_quad[i][2];
488 hexa_hexa[ho+i][3] = vo + quad_quad[i][3];
489 hexa_hexa[ho+i][4] = vo + quad_quad[i][0] + quad_num_entities[0];
490 hexa_hexa[ho+i][5] = vo + quad_quad[i][1] + quad_num_entities[0];
491 hexa_hexa[ho+i][6] = vo + quad_quad[i][2] + quad_num_entities[0];
492 hexa_hexa[ho+i][7] = vo + quad_quad[i][3] + quad_num_entities[0];
493 }
494 }
495
496 // build redundant index sets
497 RedundantIndexSetBuilder<Shape::Hexahedron>::compute(hexa_topo);
498 }

512 void extrude_mapping(HexaTrgSetHolder& hexa_mapp, const QuadTrgSetHolder& quad_mapp, const QuadMesh& quad_parent) const
513 {
514 // fetch target sets
515 const auto& qv = quad_mapp.template get_target_set<0>();
516 const auto& qe = quad_mapp.template get_target_set<1>();
517 const auto& qq = quad_mapp.template get_target_set<2>();
518 auto& hv = hexa_mapp.template get_target_set<0>();
519 auto& he = hexa_mapp.template get_target_set<1>();
520 auto& hq = hexa_mapp.template get_target_set<2>();
521 auto& hh = hexa_mapp.template get_target_set<3>();
522
523 const Index quad_num_entities[] =
524 {
525 quad_parent.get_num_entities(0),
526 quad_parent.get_num_entities(1),
527 quad_parent.get_num_entities(2)
528 };
529
530 const Index quad_part_num_entities[] =
531 {
532 qv.get_num_entities(),
533 qe.get_num_entities(),
534 qq.get_num_entities()
535 };
536
537 // compute vertices from vertices
538 for(Index j(0); j <= _slices; ++j)
539 {
540 const Index vom = j * quad_num_entities[0];
541 const Index vop = j * quad_part_num_entities[0];
542
543 for(Index i(0); i < quad_part_num_entities[0]; ++i)
544 hv[vop+i] = vom + qv[i];
545 }
546
547 // compute edges from edges
548 for(Index j(0); j <= _slices; ++j)
549 {
550 const Index eom = j * quad_num_entities[1];
551 const Index eop = j * quad_part_num_entities[1];
552
553 for(Index i(0); i < quad_part_num_entities[1]; ++i)
554 he[eop+i] = eom + qe[i];
555 }
556
557 // compute quads from quads
558 for(Index j(0); j <= _slices; ++j)
559 {
560 const Index qom = j * quad_num_entities[2];
561 const Index qop = j * quad_part_num_entities[2];
562
563 for(Index i(0); i < quad_part_num_entities[2]; ++i)
564 hq[qop+i] = qom + qq[i];
565 }
566
567 // compute edges from extruded vertices
568 for(Index j(0); j < _slices; ++j)
569 {
570 const Index eom = j * quad_num_entities[0] + (_slices+1)*quad_num_entities[1];
571 const Index eop = j * quad_part_num_entities[0] + (_slices+1)*quad_part_num_entities[1];
572
573 for(Index i(0); i < quad_part_num_entities[0]; ++i)
574 he[eop+i] = eom + qv[i];
575 }
576
577 // compute quads from extruded edges
578 for(Index j(0); j < _slices; ++j)
579 {
580 const Index qom = j * quad_num_entities[1] + (_slices+1)*quad_num_entities[2];
581 const Index qop = j * quad_part_num_entities[1] + (_slices+1)*quad_part_num_entities[2];
582
583 for(Index i(0); i < quad_part_num_entities[1]; ++i)
584 hq[qop+i] = qom + qe[i];
585 }
586
587 // compute hexas from extruded quads
588 for(Index j(0); j < _slices; ++j)
589 {
590 const Index hom = j * quad_num_entities[2];
591 const Index hop = j * quad_part_num_entities[2];
592
593 for(Index i(0); i < quad_part_num_entities[2]; ++i)
594 hh[hop+i] = hom + qq[i];
595 }
596 }

598 void extrude_slice_mapping(HexaTrgSetHolder& hexa_mapp, const QuadMesh& quad_parent, const Index slice) const
599 {
600 XASSERT(slice <= _slices);
601
602 const Index nv = quad_parent.get_num_entities(0);
603 const Index ne = quad_parent.get_num_entities(1);
604 const Index nq = quad_parent.get_num_entities(2);
605
606 // compute offsets
607 const Index ov = slice * nv;
608 const Index oe = slice * ne;
609 const Index oq = slice * nq;
610
611 // fill vertex mapping
612 auto& hv = hexa_mapp.template get_target_set<0>();
613 for(Index i(0); i < nv; ++i)
614 hv[i] = ov + i;
615
616 // fill vertex mapping
617 auto& he = hexa_mapp.template get_target_set<1>();
618 for(Index i(0); i < ne; ++i)
619 he[i] = oe + i;
620
621 // fill vertex mapping
622 auto& hq = hexa_mapp.template get_target_set<2>();
623 for(Index i(0); i < nq; ++i)
624 hq[i] = oq + i;
625 }

636 void extrude_attribute(std::unique_ptr<HexaAttrib>& hexa_attrib, const QuadAttrib& quad_attrib) const
637 {
638 const Index quad_num_values = quad_attrib.get_num_values();
639 const int attrib_dim = quad_attrib.get_dimension();
640
641 // create attribute
642 XASSERT(hexa_attrib.get() == nullptr);
643 hexa_attrib.reset(new HexaAttrib(quad_num_values * (_slices+1), attrib_dim+1));
644
645 // loop over all slices
646 for(Index j(0); j <= _slices; ++j)
647 {
648 // compute slice index offset
649 const Index vo = j * quad_num_values;
650
651 // compute slice z-coord
652 //const CoordType z = _z_min + (CoordType(j) / CoordType(_slices))*(_z_max - _z_min);
653 const CoordType z = _z_list.at(j);
654
655 // loop over all vertices
656 for(Index i(0); i < quad_num_values; ++i)
657 {
658 for(int k(0); k < attrib_dim; ++k)
659 {
660 hexa_attrib->operator()(vo+i,k) = quad_attrib(i,k);
661 }
662 hexa_attrib->operator()(vo+i, attrib_dim) = z;
663 }
664 }

676 void extrude_partition(Partition& hexa_parti, const Partition& quad_parti) const
677 {
678 const Adjacency::Graph& quad_graph = quad_parti.get_patches();
679
680 const Index num_patches = quad_parti.get_num_patches();
681 const Index quad_num_elems = quad_parti.get_num_elements();
682
683 // allocate adjacency graph
684 Adjacency::Graph hexa_graph(quad_graph.get_num_nodes_domain(),
685 _slices * quad_graph.get_num_nodes_image(), _slices * quad_graph.get_num_indices());
686
687 // get arrays
688 const Index* q_ptr = quad_graph.get_domain_ptr();
689 const Index* q_idx = quad_graph.get_image_idx();
690 Index* h_ptr = hexa_graph.get_domain_ptr();
691 Index* h_idx = hexa_graph.get_image_idx();
692
693 // loop over all patches
694 h_ptr[0] = Index(0);
695 for(Index i(0); i < num_patches; ++i)
696 {
697 Index hp = h_ptr[i];
698 // loop over all elements in patch
699 for(Index j(q_ptr[i]); j < q_ptr[i+1]; ++j)
700 {
701 // add the element for each slice
702 for(Index k(0); k < _slices; ++k, ++hp)
703 h_idx[hp] = (k*quad_num_elems) + q_idx[j];
704 }
705 h_ptr[i+1] = hp;
706 }
707
708 // create the partition
709 hexa_parti = Partition(std::move(hexa_graph), quad_parti.get_name(),
710 quad_parti.get_priority(), quad_parti.get_level());
711 }

757 void extrude_root_node(HexaRootNode& hexa_root_node, const QuadRootNode& quad_root_node, const HexaAtlas* hexa_atlas) const
758 {
759 // get root mesh
760 const QuadMesh* quad_mesh = quad_root_node.get_mesh();
761 XASSERT(quad_mesh != nullptr);
762
763 // extrude root mesh
764 {
765 MeshExtruderFactory<QuadMesh> factory(*this, *quad_mesh);
766 hexa_root_node.set_mesh(factory.make_unique());
767 }
768
769 // extrude mesh parts
770 auto part_names = quad_root_node.get_mesh_part_names();
771 for(const auto& name : part_names)
772 {
773 // get the quad part
774 const QuadPart* quad_part = quad_root_node.find_mesh_part(name);
775
776 if(quad_part == nullptr)
777 continue;
778
779 // get the chart name
780 String chart_name = quad_root_node.find_mesh_part_chart_name(name);
781 const HexaChart* hexa_chart(nullptr);
782
783 // try to find the chart
784 if(!chart_name.empty() && (hexa_atlas != nullptr))
