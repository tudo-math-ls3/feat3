528 DataType calc_convergence_rate() const
529 {
530 // no iterations performed?
531 if(_num_iter <= Index(0))
532 return DataType(0);
533
534 // initial defect zero?
535 if(_def_init < Math::eps<DataType>())
536 return DataType(0);
537
538 // compute convergence rate: (def_final / def_initial) ^ (1 / #iter)
539 return Math::pow(_def_cur / _def_init, DataType(1) / DataType(_num_iter));
540 }
541
543 DataType calc_defect_reduction() const
544 {
545 // avoid division by zero
546 if(this->_def_init <= Math::abs(this->_def_cur * Math::eps<DataType>()))
547 return DataType(0);
548
549 // compute defect reduction (def_final / def_inital)
550 return this->_def_cur / this->_def_init;
551 }

748 virtual Status _set_initial_defect(const VectorType& vec_def, const VectorType& vec_sol)
749 {
750 // store new defect
751 this->_def_init = this->_def_cur = this->_def_prev = this->_calc_def_norm(vec_def, vec_sol);
752 this->_num_iter = Index(0);
753 this->_num_stag_iter = Index(0);
754 Statistics::add_solver_expression(std::make_shared<ExpressionDefect>(this->name(), this->_def_init, this->get_num_iter()));
755
756 // plot iteration line?
757 if(this->_plot_iter())
758 this->_plot_iter_line(Index(0), this->_def_init, this->_def_init);
759
760 // ensure that the defect is neither NaN nor infinity
761 if(!Math::isfinite(this->_def_init))
762 return Status::aborted;
763
764 // check if the initial defect is zero; we test against eps^2 here
765 if(this->_def_init <= Math::sqr(Math::eps<DataType>()))
766 return Status::success;
767
768 // continue iterating
769 return Status::progress;
770 }

795 virtual Status _analyse_defect(Index num_iter, DataType def_cur, DataType def_prev, bool check_stag)
796 {
797 // ensure that the defect is neither NaN nor infinity
798 if(!Math::isfinite(def_cur))
799 return Status::aborted;
800
801 // is diverged?
802 if(this->is_diverged(def_cur))
803 return Status::diverged;
804
805 // minimum number of iterations performed?
806 if(num_iter < this->_min_iter)
807 return Status::progress;
808
809 // is converged?
810 if(this->is_converged(def_cur))
811 return Status::success;
812
813 // maximum number of iterations performed?
814 if(num_iter >= this->_max_iter)
815 return Status::max_iter;
816
817 // check for stagnation?
818 if(check_stag && (this->_min_stag_iter > Index(0)))
819 {
820 // did this iteration stagnate?
821 if(def_cur >= this->_stag_rate * def_prev)
822 {
823 // increment stagnation count
824 if(++this->_num_stag_iter >= this->_min_stag_iter)
825 return Status::stagnated;
826 }
827 else
828 {
829 // this iteration did not stagnate
830 this->_num_stag_iter = Index(0);
831 }
832 }
833
834 // continue iterating
835 return Status::progress;
836 }

853 virtual Status _set_new_defect(const VectorType& vec_def, const VectorType& vec_sol)
854 {
855 // increase iteration count
856 ++this->_num_iter;
857
858 // store previous defect
859 this->_def_prev = this->_def_cur;
860
861 // first, let's see if we have to compute the defect at all
862 bool calc_def = !_skip_def_calc;
863 calc_def = calc_def || (this->_min_iter < this->_max_iter);
864 calc_def = calc_def || (this->_plot_iter());
865 calc_def = calc_def || (this->_min_stag_iter > Index(0));
866
867 // compute new defect
868 if(calc_def)
869 {
870 this->_def_cur = this->_calc_def_norm(vec_def, vec_sol);
871 Statistics::add_solver_expression(std::make_shared<ExpressionDefect>(this->name(), this->_def_cur, this->get_num_iter()));
872 }
873
874 // analyse defect
875 Status status = this->_analyse_defect(this->_num_iter, this->_def_cur, this->_def_prev, true);
876
877 // plot defect?
878 if(this->_plot_iter(status))
879 this->_plot_iter_line(this->_num_iter, this->_def_cur, this->_def_prev);
880
881 // return status
882 return status;
883 }

899 virtual Status _update_defect(const DataType def_cur_norm)
