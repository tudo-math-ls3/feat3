222 class PatchPartMap
223 {
224 private:
226 std::map<Index,Index> _idx_map;
228 std::map<Index,Index> _io_map;
230 std::vector<Index> _indices;
231
232 public:
239 explicit PatchPartMap(const TargetSet& target_set) :
240 _idx_map(),
241 _io_map(),
242 _indices()
243 {
244 // loop over all indices
245 for(Index i(0); i < target_set.get_num_entities(); ++i)
246 {
247 // This maps the BaseMesh entity j = target_set[i] to the patch MeshPart entity i
248 XASSERTM(_idx_map.emplace(target_set[i], i).second, "internal error");
249 }
250 }
251
261 bool build(const TargetSet& target_in)
262 {
263 // Very important: Clear old information
264 _indices.clear();
265 _io_map.clear();
266
267 // We need information for every entity in the BaseMesh MeshPart
268 for(Index i(0); i < target_in.get_num_entities(); ++i)
269 {
270 // Check if the BaseMesh entity referenced by the BaseMesh MeshPart is in the PatchMeshPart
271 auto it = _idx_map.find(target_in[i]);
272 if(it != _idx_map.end())
273 {
274 // We create new entries in the PatchMeshPart MeshPart here, so the entity i in the BaseMesh MeshPart
275 // refers to the newly created entity in the PatchMeshPart MeshPart
276 _io_map.emplace(i, Index(_indices.size()));
277 // Add the new entity
278 _indices.push_back(it->second);
279 }
280 }
281 return !_indices.empty();
282 }
283
289 Index size() const
290 {
291 return Index(_indices.size());
292 }

306 template<typename Attribute_>
307 void fill_attribute_set(
308 std::map<String, std::unique_ptr<Attribute_>>& attribute_container_out,
309 const std::map<String, std::unique_ptr<Attribute_>>& attribute_container_in) const
310 {
311 for(auto it(attribute_container_in.begin()); it != attribute_container_in.end(); ++it)
312 {
313 // Create new attribute for the PatchMeshPart MeshPart
314 std::unique_ptr<Attribute_> new_attribute(new Attribute_(
315 Index(_indices.size()), it->second->get_dimension()));
316
317 const int dim = it->second->get_dimension();
318
319 // Emplace all elements referred to by the _io_map into new_attribute
320 for(auto jt: _io_map)
321 {
322 for(int k(0); k < dim; ++k)
323 new_attribute->operator()(jt.second, k) = it->second->operator()(jt.first, k);
324 }
325
326 // Push the new attribute to set_out
327 XASSERTM(attribute_container_out.insert(std::make_pair(it->first, std::move(new_attribute))).second, "Error inserting new AttributeSet");
328 }
329 }
330
337 void fill_target_set(TargetSet& target_out) const
338 {
339 XASSERT(target_out.get_num_entities() == size());
340 for(Index i(0); i < target_out.get_num_entities(); ++i)
341 {
342 target_out[i] = _indices[i];
343 }
344 }

373 template<typename IndexSetType_>
374 void fill_index_set(IndexSetType_& index_set_out, const IndexSetType_& index_set_in,
375 const std::map<Index, Index>& vertex_map) const
376 {
377 // We need to check all shape entities in the MeshPart referring to the BaseMesh
378 for(Index k(0); k < index_set_in.get_num_entities(); ++k)
379 {
380 // Check if the BaseMesh entity referred to by the BaseMesh MeshPart is also present in the PatchMeshPart
381 auto kt = _io_map.find(k);
382 if(kt != _io_map.end())
383 {
384 // For all vertices at that entity, we need to find them in the PatchMeshPart's MeshPart
385 for(int j(0); j < index_set_in.get_num_indices(); ++j)
386 {
387 // Fetch the BaseMesh MeshPart's vertex number
388 Index i_in(index_set_in(k,j));
389 // That vertex is definitely in the PatchMeshPart's MeshPart, but we need to catch any errors here
390 auto jt = vertex_map.find(i_in);
391 if(jt != vertex_map.end())
392 // This is the correct number
393 index_set_out(kt->second, j) = jt->second;
394 else
395 XABORTM("Vertex "+stringify(i_in)+" missing in MeshPart topology!");
396 }
397 }
398 }
399 }

407 const std::map<Index, Index>& get_io_map() const
408 {
409 return _io_map;
410 }

464 bool build(const TargetSetHolder<Shape_>& tsh, const IndexSetHolder<Shape_>* ish)
465 {
