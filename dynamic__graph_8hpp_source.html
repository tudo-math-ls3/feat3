336 template<typename Adjactor_>
337 void compose(const Adjactor_& adjactor)
338 {
339 XASSERTM(_num_nodes_image != adjactor.get_num_nodes_domain(), "Adjactor dimension mismatch!");
340
341 typedef typename Adjactor_::ImageIterator AImIt;
342
343 // image index buffer vector
344 std::vector<Index> img_idx;
345
346 // loop over all domain nodes
347 for(Index i(0); i < _num_nodes_domain; ++i)
348 {
349 // clear image index buffer
350 img_idx.clear();
351
352 // loop over all image indices of this adjactor and make a backup of them
353 {
354 ImageIterator it(_indices[i].begin());
355 ImageIterator jt(_indices[i].end());
356 for(; it != jt; ++it)
357 {
358 img_idx.push_back(*it);
359 }
360 }
361
362 // clear the current domain node
363 _indices[i].clear();
364
365 // loop over all entries of the index buffer vector
366 for(Index j(0); j < img_idx.size(); ++j)
367 {
368 // iterator over all adjacent nodes
369 AImIt it(adjactor.image_begin(img_idx[j]));
370 AImIt jt(adjactor.image_end(img_idx[j]));
371 for(; it != jt; ++it)
372 {
373 _indices[i].insert(*it);
374 }
375 }
376 }
377
378 // store new image node count
379 _num_nodes_image = adjactor.get_num_nodes_image();
380 }

391 typedef typename Adjactor_::ImageIterator AImIt;

394 _num_nodes_domain = adj.get_num_nodes_domain();
395 _num_nodes_image = adj.get_num_nodes_image();

400 {
401 AImIt cur(adj.image_begin(i));
402 AImIt end(adj.image_end(i));
403 for(; cur != end; ++cur)

405 insert(i, *cur);
406 }
407 }

442 XASSERTM(adj1.get_num_nodes_image() != adj2.get_num_nodes_domain(), "Adjactor dimension mismatch!");
443
444 typedef typename Adjactor1_::ImageIterator AImIt1;
445 typedef typename Adjactor2_::ImageIterator AImIt2;
446
447 // set counts
448 _num_nodes_domain = adj1.get_num_nodes_domain();
449 _num_nodes_image = adj2.get_num_nodes_image();
450 _indices.resize(_num_nodes_domain);
451
452 // loop over all domain nodes
453 for(Index i(0); i < _num_nodes_domain; ++i)
454 {
455 AImIt1 cur1(adj1.image_begin(i));
456 AImIt1 end1(adj1.image_end(i));
457 for(; cur1 != end1; ++cur1)
458 {
459 AImIt2 cur2(adj2.image_begin(*cur1));
460 AImIt2 end2(adj2.image_end(*cur1));
461 for(; cur2 != end2; ++cur2)
462 {
463 insert(i, *cur2);
464 }
465 }
