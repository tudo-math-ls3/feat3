240 protected:
241 virtual Status _apply_intern(VectorType& vec_sol, const VectorType& vec_rhs)
242 {
243 IterationStats pre_iter(*this);
244 Statistics::add_solver_expression(std::make_shared<ExpressionStartSolve>(this->name()));
245 const MatrixType& matrix(this->_system_matrix);
246 const FilterType& filter(this->_system_filter);
247
248 // compute initial defect
249 Status status = this->_set_initial_defect(this->_vec_v.at(0), vec_sol);
250
251 pre_iter.destroy();
252
253 // outer GMRES loop
254 while(status == Status::progress)
255 {
256 IterationStats stat(*this);
257
258 _q.clear();
259 _s.clear();
260 _c.clear();
261 _q.push_back(this->_def_cur);
262
263 // normalize v[0]
264 this->_vec_v.at(0).scale(this->_vec_v.at(0), DataType(1) / _q.back());
265
266 // inner GMRES loop
267 Index i(0);
268 while(i < this->_krylov_dim)
269 {
270 // apply preconditioner
271 if(!this->_apply_precond(this->_vec_z.at(i), this->_vec_v.at(i), filter))
272 {
273 stat.destroy();
274 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
275 return Status::aborted;
276 }
277 //filter.filter_cor(this->_vec_z.at(i));
278
279 // v[i+1] := A*z[i]
280 matrix.apply(this->_vec_v.at(i+1), this->_vec_z.at(i));
281 filter.filter_def(this->_vec_v.at(i+1));
282
283 // Gram-Schmidt process
284 for(Index k(0); k <= i; ++k)
285 {
286 this->_h.at(i).at(k) = this->_vec_v.at(i+1).dot(this->_vec_v.at(k));
287 this->_vec_v.at(i+1).axpy(this->_vec_v.at(k), this->_vec_v.at(i+1), -this->_h.at(i).at(k));
288 }
289
290 // normalize v[i+1]
291 DataType alpha = this->_vec_v.at(i+1).norm2();
292 this->_vec_v.at(i+1).scale(this->_vec_v.at(i+1), DataType(1) / alpha);
293
294 // apply Givens rotations
295 for(Index k(0); k < i; ++k)
296 {
297 DataType t(this->_h.at(i).at(k));
298 this->_h.at(i).at(k ) = this->_c.at(k) * t + this->_s.at(k) * this->_h.at(i).at(k+1);
299 this->_h.at(i).at(k+1) = this->_s.at(k) * t - this->_c.at(k) * this->_h.at(i).at(k+1);
300 }
301
302 // compute beta
303 DataType beta = Math::sqrt(Math::sqr(this->_h.at(i).at(i)) + Math::sqr(alpha));
304
305 // compute next plane rotation
306 _s.push_back(alpha / beta);
307 _c.push_back(this->_h.at(i).at(i) / beta);
308
309 this->_h.at(i).at(i) = beta;
310 this->_q.push_back(this->_s.back() * this->_q.at(i));
311 this->_q.at(i) *= this->_c.back();
312
313 // push our new defect
314 if(++i < this->_krylov_dim)
315 {
316 // get the absolute defect
317 DataType def_cur = Math::abs(this->_q.back());
318
319 // did we diverge?
320 if((def_cur > this->_div_abs) || (def_cur > (this->_div_rel * this->_def_init)))
321 break;
322
323 // minimum number of iterations performed?
324 if(!(this->_num_iter < this->_min_iter))
325 {
326 // did we converge?
327 if((def_cur <= _inner_res_scale * this->_tol_abs) &&
328 ((def_cur <= _inner_res_scale * (this->_tol_rel * this->_def_init)) || (def_cur <= _inner_res_scale * this->_tol_abs_low) ))
329 break;
330
331 // maximum number of iterations performed?
332 if(this->_num_iter >= this->_max_iter)
333 break;
334 }
335
336 // set our pseudo defect
337 // increase iteration count
338 ++this->_num_iter;
339
340 // compute new defect
341 this->_def_cur = def_cur;
342
343 // plot?
344 if(this->_plot_iter())
345 {
346 String msg = this->_plot_name
347 + "* " + stringify(this->_num_iter).pad_front(this->_iter_digits)
348 + " : " + stringify_fp_sci(this->_def_cur);
349 this->_print_line(msg);
350 }
351 }
352 }
353
354 Index n = Math::min(i, this->_krylov_dim);
355
356 // solve H*q = q
357 for(Index k(n); k > 0;)
358 {
359 --k;
360 this->_q.at(k) /= this->_h.at(k).at(k);
361 for(Index j(k); j > 0;)
362 {
363 --j;
364 this->_q.at(j) -= this->_h.at(k).at(j) * this->_q.at(k);
365 }
366 }
367
368 // update solution
369 for(Index k(0); k < n; ++k)
370 vec_sol.axpy(this->_vec_z.at(k), vec_sol, this->_q.at(k));
371
372 // compute "real" residual
373 matrix.apply(this->_vec_v.at(0), vec_sol, vec_rhs, -DataType(1));
374 filter.filter_def(this->_vec_v.at(0));
375
376 // set the current defect
377 status = this->_set_new_defect(this->_vec_v.at(0), vec_sol);
378 }
379
380 // finished
381 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), status, this->get_num_iter()));
382 return status;
383 }
