258 protected:
259 virtual Status _apply_intern(VectorType& vec_sol, const VectorType& vec_rhs)
260 {
261 IterationStats pre_iter(*this);
262 Statistics::add_solver_expression(std::make_shared<ExpressionStartSolve>(this->name()));
263 const MatrixType& matrix(this->_system_matrix);
264 const FilterType& filter(this->_system_filter);
265
266 // compute initial defect
267 Status status = this->_set_initial_defect(this->_vec_v.at(0), vec_sol);
268
269 pre_iter.destroy();
270
271 // outer GMRES loop
272 while(status == Status::progress)
273 {
274 IterationStats stat(*this);
275
276 _q.clear();
277 _s.clear();
278 _c.clear();
279 _q.push_back(this->_def_cur);
280
281 // normalize v[0]
282 this->_vec_v.at(0).scale(this->_vec_v.at(0), DataType(1) / _q.back());
283
284 // inner GMRES loop
285 Index i(0);
286 while(i < this->_krylov_dim)
287 {
288 // apply preconditioner
289 if(!this->_apply_precond(this->_vec_z.at(i), this->_vec_v.at(i), filter))
290 {
291 stat.destroy();
292 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
293 return Status::aborted;
294 }
295 //filter.filter_cor(this->_vec_z.at(i));
296
297 // v[i+1] := A*z[i]
298 matrix.apply(this->_vec_v.at(i+1), this->_vec_z.at(i));
299 filter.filter_def(this->_vec_v.at(i+1));
300
301 // Gram-Schmidt process
302 for(Index k(0); k <= i; ++k)
303 {
304 this->_h.at(i).at(k) = this->_vec_v.at(i+1).dot(this->_vec_v.at(k));
305 this->_vec_v.at(i+1).axpy(this->_vec_v.at(k), -this->_h.at(i).at(k));
306 }
307
308 // normalize v[i+1]
309 DataType alpha = this->_vec_v.at(i+1).norm2();
310 this->_vec_v.at(i+1).scale(this->_vec_v.at(i+1), DataType(1) / alpha);
311
312 // apply Givens rotations
313 for(Index k(0); k < i; ++k)
314 {
315 DataType t(this->_h.at(i).at(k));
316 this->_h.at(i).at(k ) = this->_c.at(k) * t + this->_s.at(k) * this->_h.at(i).at(k+1);
317 this->_h.at(i).at(k+1) = this->_s.at(k) * t - this->_c.at(k) * this->_h.at(i).at(k+1);
318 }
319
320 // compute beta
321 DataType beta = Math::sqrt(Math::sqr(this->_h.at(i).at(i)) + Math::sqr(alpha));
322
323 // compute next plane rotation
324 _s.push_back(alpha / beta);
325 _c.push_back(this->_h.at(i).at(i) / beta);
326
327 this->_h.at(i).at(i) = beta;
328 this->_q.push_back(this->_s.back() * this->_q.at(i));
329 this->_q.at(i) *= this->_c.back();
330
331 // push our new defect
332 if(++i < this->_krylov_dim)
333 {
334 // get the absolute defect
335 DataType def_cur = Math::abs(this->_q.back());
336
337 // did we diverge?
338 if((def_cur > this->_div_abs) || (def_cur > (this->_div_rel * this->_def_init)))
339 break;
340
341 // minimum number of iterations performed?
342 if(!(this->_num_iter < this->_min_iter))
343 {
344 // did we converge?
345 if((def_cur <= _inner_res_scale * this->_tol_abs) &&
346 ((def_cur <= _inner_res_scale * (this->_tol_rel * this->_def_init)) || (def_cur <= _inner_res_scale * this->_tol_abs_low) ))
347 break;
348
349 // maximum number of iterations performed?
350 if(this->_num_iter >= this->_max_iter)
351 break;
352 }
353
354 // set our pseudo defect
355 // increase iteration count
356 ++this->_num_iter;
357
358 // compute new defect
359 this->_def_cur = def_cur;
360
361 // plot?
362 if(this->_plot_iter())
363 {
364 String msg = this->_plot_name
365 + "* " + stringify(this->_num_iter).pad_front(this->_iter_digits)
366 + " : " + stringify_fp_sci(this->_def_cur);
367 this->_print_line(msg);
368 }
369 }
370 }
371
372 Index n = Math::min(i, this->_krylov_dim);
373
374 // solve H*q = q
375 for(Index k(n); k > 0;)
376 {
377 --k;
378 this->_q.at(k) /= this->_h.at(k).at(k);
379 for(Index j(k); j > 0;)
380 {
381 --j;
382 this->_q.at(j) -= this->_h.at(k).at(j) * this->_q.at(k);
383 }
384 }
385
386 // update solution
387 for(Index k(0); k < n; ++k)
388 vec_sol.axpy(this->_vec_z.at(k), this->_q.at(k));
389
390 // compute "real" residual
391 matrix.apply(this->_vec_v.at(0), vec_sol, vec_rhs, -DataType(1));
392 filter.filter_def(this->_vec_v.at(0));
393
394 // set the current defect
395 status = this->_set_new_defect(this->_vec_v.at(0), vec_sol);
396 }
397
398 // finished
399 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), status, this->get_num_iter()));
400 return status;
401 }
