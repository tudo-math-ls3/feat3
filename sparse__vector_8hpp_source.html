51 template <typename T1_, typename T2_>
52 static void _insertion_sort(T1_ * key, T2_ * val1, Index size)
53 {
54 T1_ swap_key;
55 T2_ swap1;
56 for (Index i(1), j ; i < size ; ++i)
57 {
58 swap_key = key[i];
59 swap1 = val1[i];
60 j = i;
61 while (j > 0 && key[j-1] > swap_key)
62 {
63 key[j] = key[j-1];
64 val1[j] = val1[j-1];
65 --j;
66 }
67 key[j] = swap_key;
68 val1[j] = swap1;
69 }
70 }

328 const DT_ operator()(Index index) const
329 {
330 ASSERTM(index < this->_scalar_index.at(0), "index exceeds sparse vector size");
331
332 MemoryPool::synchronize();
333
334 if (this->_elements.empty())
335 return DT_(0.);
336
337 if (sorted() == 0)
338 const_cast<SparseVector *>(this)->sort();
339
340 Index i(0);
341 while (i < used_elements())
342 {
343 if (indices()[i] >= index)
344 break;
345 ++i;
346 }
347
348 if (i < used_elements() && indices()[i] == index)
349 return elements()[i];
350 else
351 return DT_(0.);
352 }

354
361 void operator()(Index index, DT_ val)
362 {
363 ASSERTM(index < this->_scalar_index.at(0), "index exceeds sparse vector size");
364
365 // flag container as not sorted anymore
366 // CAUTION: do not use any method triggering resorting until we are finished
367 _sorted() = 0;
368
369 // vector is empty, no arrays allocated
370 if (this->_elements.empty())
371 {
372 this->_elements.push_back(MemoryPool::template allocate_memory<DT_>(alloc_increment()));
373 this->_elements_size.push_back(alloc_increment());
374 MemoryPool::template set_memory<DT_>(this->_elements.back(), DT_(4711), alloc_increment());
375 this->_indices.push_back(MemoryPool::template allocate_memory<IT_>(alloc_increment()));
376 this->_indices_size.push_back(alloc_increment());
377 MemoryPool::template set_memory<IT_>(this->_indices.back(), IT_(4711), alloc_increment());
378 this->_allocated_elements() = alloc_increment();
379 this->_elements.at(0)[0] = val;
380 this->_indices.at(0)[0] = IT_(index);
381 this->_used_elements() = 1;
382 }
383
384 // append element in already allocated arrays
385 else if(_used_elements() < allocated_elements())
386 {
387 this->_elements.at(0)[this->_used_elements()] = val;
388 this->_indices.at(0)[this->_used_elements()] = IT_(index);
389 ++this->_used_elements();
390 }
391
392 // reallocate arrays, append element
393 else
394 {
395 _allocated_elements() += alloc_increment();
396
397 DT_ * elements_new(MemoryPool::template allocate_memory<DT_>(allocated_elements()));
398 MemoryPool::template set_memory<DT_>(elements_new, DT_(4711), allocated_elements());
399 IT_ * indices_new(MemoryPool::template allocate_memory<IT_>(allocated_elements()));
400 MemoryPool::template set_memory<IT_>(indices_new, IT_(4711), allocated_elements());
401
402 MemoryPool::copy(elements_new, this->_elements.at(0), _used_elements());
403 MemoryPool::copy(indices_new, this->_indices.at(0), _used_elements());
404
405 MemoryPool::release_memory(this->_elements.at(0));
406 MemoryPool::release_memory(this->_indices.at(0));
407
408 this->_elements.at(0) = elements_new;
409 this->_indices.at(0) = indices_new;
410
411 this->_elements.at(0)[this->_used_elements()] = val;
412 this->_indices.at(0)[this->_used_elements()] = IT_(index);
413
414 ++this->_used_elements();
415 this->_elements_size.at(0) = allocated_elements();
416 this->_indices_size.at(0) = allocated_elements();
417 }

420 void sort()
421 {
422 if (sorted() == 0)
423 {
424 //first of all, mark vector as sorted, because otherwise we would call ourselves inifite times
425 // CAUTION: do not use any method triggering resorting until we are finished
426 _sorted() = 1;
427
428 // check if there is anything to be sorted
429 if(_used_elements() == Index(0))
430 return;
431
432 IT_ * pindices = this->_indices.at(0);
433 DT_ * pelements = this->_elements.at(0);
434
435 _insertion_sort(pindices, pelements, _used_elements());
436
437 // find and mark duplicate entries
438 for (Index i(1) ; i < _used_elements() ; ++i)
439 {
440 if (pindices[i-1] == pindices[i])
441 {
442 pindices[i-1] = std::numeric_limits<IT_>::max();
443 }
444 }
445
446 // sort out marked duplicated elements
447 _insertion_sort(pindices, pelements, _used_elements());
448 Index junk(0);
449 while (pindices[_used_elements() - 1 - junk] == std::numeric_limits<IT_>::max() && junk < _used_elements())
450 ++junk;
451 _used_elements() -= junk;
452 }
453 }

583 void read_from(FileMode mode, std::istream& file)
584 {
585 switch(mode)
586 {
587 case FileMode::fm_binary:
588 case FileMode::fm_sv:
589 this->template _deserialize<double, std::uint64_t>(FileMode::fm_sv, file);
590 break;
591 case FileMode::fm_mtx:
592 {
593 this->clear();
594 this->_scalar_index.push_back(0);
595 this->_scalar_index.push_back(0);
596 this->_scalar_index.push_back(Math::min<Index>(0, 1000));
597 this->_scalar_index.push_back(1);
598
599 Index rows;
600 Index nnz;
601 String line;
602 std::getline(file, line);
603 if (line.find("%%MatrixMarket matrix coordinate real general") == String::npos)
604 {
605 XABORTM("Input-file is not a compatible mtx-file");
606 }
607 while(!file.eof())
608 {
609 std::getline(file,line);
610 if (file.eof())
611 XABORTM("Input-file is empty");
612
613 String::size_type begin(line.find_first_not_of(" "));
614 if (line.at(begin) != '%')
615 break;
616 }
617 {
618 String::size_type begin(line.find_first_not_of(" "));
619 line.erase(0, begin);
620 String::size_type end(line.find_first_of(" "));
621 String srows(line, 0, end);
622 rows = (Index)atol(srows.c_str());
623 line.erase(0, end);
624
625 begin = line.find_first_not_of(" ");
626 line.erase(0, begin);
627 end = line.find_first_of(" ");
628 String scols(line, 0, end);
629 Index cols((Index)atol(scols.c_str()));
630 line.erase(0, end);
631 if (cols != 1)
632 XABORTM("Input-file is no sparse-vector-file");
633
634 begin = line.find_first_not_of(" ");
635 line.erase(0, begin);
636 end = line.find_first_of(" ");
637 String snnz(line, 0, end);
638 nnz = (Index)atol(snnz.c_str());
639 line.erase(0, end);
640 }
641
642 DenseVector<IT_, IT_> ind(nnz);
643 DenseVector<DT_, IT_> val(nnz);
644
645 IT_ * pind(ind.elements());
646 DT_ * pval(val.elements());
647
648 while(!file.eof())
649 {
650 std::getline(file, line);
651 if (file.eof())
652 break;
653
654 String::size_type begin(line.find_first_not_of(" "));
655 line.erase(0, begin);
656 String::size_type end(line.find_first_of(" "));
657 String srow(line, 0, end);
658 IT_ row((IT_)atol(srow.c_str()));
659 --row;
660 line.erase(0, end);
661
662 begin = line.find_first_not_of(" ");
663 line.erase(0, begin);
664 end = line.find_first_of(" ");
665 line.erase(0, end);
666
667 begin = line.find_first_not_of(" ");
668 line.erase(0, begin);
669 end = line.find_first_of(" ");
670 String sval(line, 0, end);
671 DT_ tval((DT_)atof(sval.c_str()));
672
673 *pval = tval;
674 *pind = row;
675 ++pval;
676 ++pind;
677 }
678 SparseVector<DT_, IT_> tmp(rows, val, ind, false);
679 this->assign(tmp);
680 break;
681 }
682 default:
683 XABORTM("Filemode not supported!");
684 }
685 }

788 void permute(Adjacency::Permutation & perm)
789 {
790 if (perm.size() == 0)
791 return;
792
793 XASSERTM(perm.size() == this->size(), "Container size does not match permutation size");
794
795 SparseVector<DT_, IT_> target(this->size());
796
797 auto inv = perm.inverse();
798 const Index * const inv_pos(inv.get_perm_pos());
799 for (Index i(0) ; i < this->used_elements() ; ++i)
800 {
801 const Index col = this->indices()[i];
802 target(inv_pos[col], (*this)(col));
803 }
804
805 target.sort();
806 this->assign(target);
807 }

829 bool compare_layout(const SparseVector& other) const
830 {
831 if(this->size() != other.size())
832 return false;
833 if(this->used_elements() != other.used_elements())
834 return false;
835
836 Index n = this->used_elements();
837 if(n == Index(0))
838 return true; // both vectors are empty
839
840 const IT_* a = this->indices();
841 const IT_* b = this->indices();
842
843 // shallow copy? (aka same array)
844 if(a == b)
845 return true;
846
847 // check all array entries
848 for(Index i(0); i < n; ++i)
849 {
850 if(a[i] != b[i])
851 return false;
852 }
853
854 // okay, arrays are identical
855 return true;
856 }
