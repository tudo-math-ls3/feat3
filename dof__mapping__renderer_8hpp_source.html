35 static Adjacency::Graph render(const Space_& space)
36 {
37 // create a dof-mapping
38 typename Space_::DofMappingType dof_map(space);
39
40 // fetch the number of cells and dofs
41 const Index num_cells(space.get_mesh().get_num_entities(Space_::shape_dim)); //num_cells
42 const Index num_dofs(space.get_num_dofs()); //num_dofs
43
44 // loop over all cells and build the domain pointer array
45
46 Index num_indices_image(0);
47 for(Index i(0); i < num_cells; ++i)
48 {
49 dof_map.prepare(i);
50 num_indices_image += Index(dof_map.get_num_local_dofs());
51 dof_map.finish();
52 }
53 // create an adjacency graph
54 Adjacency::Graph graph (num_cells, num_dofs, num_indices_image) ;
55 Index* dom_ptr = graph.get_domain_ptr();
56 Index* img_idx = graph.get_image_idx();
57
58 //build the domain pointer array
59 //loop over all cells and build the image index array
60 dom_ptr[0] = Index(0);
61
62 for(Index i(0); i < num_cells; ++i)
63 {
64 Index l(dom_ptr[i]);
65 dof_map.prepare(i);
66 dom_ptr[i + 1] = l + Index(dof_map.get_num_local_dofs());
67 for(int j(0); j < dof_map.get_num_local_dofs(); ++j, ++l)
68 {
69 img_idx[l] = dof_map.get_index(j);
70 }
71 dof_map.finish();
72 }
73
74 // create an adjacency graph
75 return graph;
76 }
