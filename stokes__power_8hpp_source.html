170 void compile_system_transfer()
171 {
172 // clone content into our global transfer matrix
173 transfer_sys.get_mat_prol().template at<0,0>().clone(transfer_velo.get_mat_prol(), LAFEM::CloneMode::Shallow);
174 transfer_sys.get_mat_rest().template at<0,0>().clone(transfer_velo.get_mat_rest(), LAFEM::CloneMode::Shallow);
175 transfer_sys.get_mat_prol().template at<1,1>().clone(transfer_pres.get_mat_prol(), LAFEM::CloneMode::Shallow);
176 transfer_sys.get_mat_rest().template at<1,1>().clone(transfer_pres.get_mat_rest(), LAFEM::CloneMode::Shallow);
177 transfer_sys.get_mat_trunc().template at<0,0>().clone(transfer_velo.get_mat_trunc(), LAFEM::CloneMode::Shallow);
178 transfer_sys.get_mat_trunc().template at<1,1>().clone(transfer_pres.get_mat_trunc(), LAFEM::CloneMode::Shallow);
179 transfer_sys.compile();
180 }

182 void compile_system_matrix()
183 {
184 matrix_sys.local().block_a() = matrix_a.local().clone(LAFEM::CloneMode::Shallow);
185 matrix_sys.local().block_b() = matrix_b.local().clone(LAFEM::CloneMode::Shallow);
186 matrix_sys.local().block_d() = matrix_d.local().clone(LAFEM::CloneMode::Shallow);

189 template<typename D_, typename I_, typename SM_, typename TM_>
190 void convert(const StokesPowerSystemLevel<dim_, D_, I_, SM_, TM_> & other)
191 {
192 gate_velo.convert(other.gate_velo);
193 gate_pres.convert(other.gate_pres);
194 gate_sys.convert(other.gate_sys);
195
196 coarse_muxer_velo.convert(other.coarse_muxer_velo);
197 coarse_muxer_pres.convert(other.coarse_muxer_pres);
198 coarse_muxer_sys.convert(other.coarse_muxer_sys);
199
200 matrix_a.convert(&gate_velo, &gate_velo, other.matrix_a);
201 matrix_b.convert(&gate_velo, &gate_pres, other.matrix_b);
202 matrix_d.convert(&gate_pres, &gate_velo, other.matrix_d);
203 matrix_s.convert(&gate_pres, &gate_pres, other.matrix_s);
204
205 compile_system_matrix();
206
207 compile_system_transfer();
208 }

211 template<typename DomainLevel_>
212 void assemble_gates(const Domain::VirtualLevel<DomainLevel_>& virt_dom_lvl)
213 {
214 const auto& dom_level = virt_dom_lvl.level();
215 const auto& dom_layer = virt_dom_lvl.layer();
216 const auto& space_velo = dom_level.space_velo;
217 const auto& space_pres = dom_level.space_pres;
218
219 // set the gate comm
220 this->gate_velo.set_comm(dom_layer.comm_ptr());
221 this->gate_pres.set_comm(dom_layer.comm_ptr());
222 this->gate_sys.set_comm(dom_layer.comm_ptr());
223
224 // loop over all ranks
225 for(Index i(0); i < dom_layer.neighbor_count(); ++i)
226 {
227 int rank = dom_layer.neighbor_rank(i);
228
229 // try to find our halo
230 const auto* halo = dom_level.find_halo_part(rank);
231 XASSERT(halo != nullptr);
232
233 // assemble the velocity components mirror
234 ScalarMirror mirror_velo_comp;
235 Assembly::MirrorAssembler::assemble_mirror(mirror_velo_comp, space_velo, *halo);
236
237 // create a velocity mirror
238 VeloMirror mirror_velo(mirror_velo_comp.clone());
239
240 // create (empty) pressure mirror
241 PresMirror mirror_pres;
242 Assembly::MirrorAssembler::assemble_mirror(mirror_pres, space_pres, *halo);
243
244 // create a system mirror
245 SystemMirror mirror_sys(mirror_velo.clone(), mirror_pres.clone());
246
247 // push mirrors into gates
248 if(!mirror_velo.empty())
249 this->gate_velo.push(rank, std::move(mirror_velo));
250 if(!mirror_pres.empty())
251 this->gate_pres.push(rank, std::move(mirror_pres));
252 if(!mirror_sys.empty())
253 this->gate_sys.push(rank, std::move(mirror_sys));
254 }
255
256 // create local template vectors
257 LocalVeloVector tmpl_v(space_velo.get_num_dofs());
258 LocalPresVector tmpl_p(space_pres.get_num_dofs());
259 LocalSystemVector tmpl_s(tmpl_v.clone(), tmpl_p.clone());
260
261 // compile gates
262 this->gate_velo.compile(std::move(tmpl_v));
263 this->gate_pres.compile(std::move(tmpl_p));
264 this->gate_sys.compile(std::move(tmpl_s));
265 }

267 template<typename DomainLevel_>
268 void assemble_coarse_muxers(const Domain::VirtualLevel<DomainLevel_>& virt_lvl_coarse)
269 {
270 // assemble muxer parent
271 if(virt_lvl_coarse.is_parent())
272 {
273 XASSERT(virt_lvl_coarse.is_child());
274
275 const auto& layer_c = virt_lvl_coarse.layer_c();
276 const DomainLevel_& level_p = virt_lvl_coarse.level_p();
277
278 // loop over all children
279 for(Index i(0); i < layer_c.child_count(); ++i)
280 {
281 const auto* child = level_p.find_patch_part(int(i));
282 XASSERT(child != nullptr);
283 SystemMirror child_mirror_sys;
284 VeloMirror& child_mirror_v = child_mirror_sys.template at<0>();
285 PresMirror& child_mirror_p = child_mirror_sys.template at<1>();
286 Assembly::MirrorAssembler::assemble_mirror(child_mirror_v._sub_mirror, level_p.space_velo, *child);
287 Assembly::MirrorAssembler::assemble_mirror(child_mirror_p, level_p.space_pres, *child);
288 this->coarse_muxer_velo.push_child(child_mirror_v.clone(LAFEM::CloneMode::Shallow));
289 this->coarse_muxer_pres.push_child(child_mirror_p.clone(LAFEM::CloneMode::Shallow));
290 this->coarse_muxer_sys.push_child(std::move(child_mirror_sys));
291 }
292 }
293
294 // assemble muxer child
295 if(virt_lvl_coarse.is_child())
296 {
297 const auto& layer_c = virt_lvl_coarse.layer_c();
298 const DomainLevel_& level_c = virt_lvl_coarse.level_c();
299
300 SystemMirror parent_mirror_sys;
301 VeloMirror& parent_mirror_v = parent_mirror_sys.template at<0>();
302 PresMirror& parent_mirror_p = parent_mirror_sys.template at<1>();
303
304 // manually set up an identity gather/scatter matrix
305 {
306 Index n = level_c.space_velo.get_num_dofs();
307 parent_mirror_v._sub_mirror = ScalarMirror(n, n);
308 auto* idx = parent_mirror_v._sub_mirror.indices();
309 for(Index i(0); i < n; ++i)
310 idx[i] = i;
311 }
312 {
313 Index n = level_c.space_pres.get_num_dofs();
314 parent_mirror_p = ScalarMirror(n, n);
315 auto* idx = parent_mirror_p.indices();
316 for(Index i(0); i < n; ++i)
317 idx[i] = i;
318 }
319
320 // set parent and sibling comms
321 this->coarse_muxer_velo.set_parent(
322 layer_c.sibling_comm_ptr(),
323 layer_c.get_parent_rank(),
324 parent_mirror_v.clone(LAFEM::CloneMode::Shallow)
325 );
326 this->coarse_muxer_pres.set_parent(
327 layer_c.sibling_comm_ptr(),
328 layer_c.get_parent_rank(),
329 parent_mirror_p.clone(LAFEM::CloneMode::Shallow)
330 );
331 this->coarse_muxer_sys.set_parent(
332 layer_c.sibling_comm_ptr(),
333 layer_c.get_parent_rank(),
334 std::move(parent_mirror_sys)
335 );
336
337 // compile muxer
338 LocalVeloVector tmpl_v(level_c.space_velo.get_num_dofs());
339 LocalPresVector tmpl_p(level_c.space_pres.get_num_dofs());
340 LocalSystemVector tmpl_s(tmpl_v.clone(), tmpl_p.clone());
341 this->coarse_muxer_velo.compile(tmpl_v);
342 this->coarse_muxer_pres.compile(tmpl_p);
343 this->coarse_muxer_sys.compile(tmpl_s);
344 }

347 template<typename DomainLevel_, typename Cubature_>
348 void assemble_velocity_transfer(
349 const Domain::VirtualLevel<DomainLevel_>& virt_lvl_fine,
350 const Domain::VirtualLevel<DomainLevel_>& virt_lvl_coarse,
351 const Cubature_& cubature)
352 {
353 // get fine and coarse domain levels
354 const DomainLevel_& level_f = *virt_lvl_fine;
355 const DomainLevel_& level_c = virt_lvl_coarse.is_child() ? virt_lvl_coarse.level_c() : *virt_lvl_coarse;
356
357 const auto& space_f = level_f.space_velo;
358 const auto& space_c = level_c.space_velo;
359
360 // get local transfer operator
361 LocalVeloTransfer& loc_trans = this->transfer_velo.local();
362
363 // get local transfer matrices
364 LocalVeloTransferMatrix& loc_prol_v = loc_trans.get_mat_prol();
365 LocalVeloTransferMatrix& loc_rest_v = loc_trans.get_mat_rest();
366
367 // get the matrix blocks
368 LocalScalarTransferMatrix& loc_prol_vx = loc_prol_v.get(0,0);
369 LocalScalarTransferMatrix& loc_rest_vx = loc_rest_v.get(0,0);
370
371 // assemble structure?
372 if(loc_prol_vx.empty())
373 {
374 Assembly::SymbolicAssembler::assemble_matrix_2lvl(loc_prol_vx, space_f, space_c);
375
376 for(int i(1); i < loc_prol_v.num_row_blocks; ++i)
377 loc_prol_v.get(i,i) = loc_prol_vx.clone(LAFEM::CloneMode::Layout);
378 }
379
380 // get local velocity weight vector
381 LocalVeloVector loc_vec_weight = loc_prol_v.create_vector_l();
382
383 // get local weight vector components
384 LocalScalarVector& loc_vec_wx = loc_vec_weight.get(0);
385
386 // assemble prolongation matrix
387 {
388 loc_prol_vx.format();
389 loc_vec_weight.format();
390
391 // assemble prolongation matrix
392 Assembly::GridTransfer::assemble_prolongation(loc_prol_vx, loc_vec_wx, space_f, space_c, cubature);
393
394 // synchronize weight vector
395 for(int i(1); i < loc_vec_weight.num_blocks; ++i)
396 loc_vec_weight.get(i).copy(loc_vec_wx);
397
398 this->gate_velo.sync_0(loc_vec_weight);
399
400 // invert components
401 loc_vec_weight.component_invert(loc_vec_weight);
402
403 // scale prolongation matrix
404 loc_prol_vx.scale_rows(loc_prol_vx, loc_vec_wx);
405
406 // copy and transpose
407 loc_rest_vx = loc_prol_vx.transpose();
408 for(int i(1); i < loc_prol_v.num_row_blocks; ++i)
409 {
410 loc_prol_v.get(i,i) = loc_prol_vx.clone(LAFEM::CloneMode::Shallow);
411 loc_rest_v.get(i,i) = loc_rest_vx.clone(LAFEM::CloneMode::Shallow);
412 }
413 }
414
415 // compile velocity transfer
416 this->transfer_velo.compile();
417 }

419 template<typename DomainLevel_, typename Cubature_>
420 void assemble_pressure_transfer(
421 const Domain::VirtualLevel<DomainLevel_>& virt_lvl_fine,
422 const Domain::VirtualLevel<DomainLevel_>& virt_lvl_coarse,
423 const Cubature_& cubature)
424 {
425 // get fine and coarse domain levels
426 const DomainLevel_& level_f = *virt_lvl_fine;
427 const DomainLevel_& level_c = virt_lvl_coarse.is_child() ? virt_lvl_coarse.level_c() : *virt_lvl_coarse;
428
429 const auto& space_f = level_f.space_pres;
430 const auto& space_c = level_c.space_pres;
431
432 // get local transfer operator
433 LocalPresTransfer& loc_trans = this->transfer_pres.local();
434
435 // get local transfer matrices
436 LocalPresTransferMatrix& loc_prol_p = loc_trans.get_mat_prol();
437 LocalPresTransferMatrix& loc_rest_p = loc_trans.get_mat_rest();
438
439 // assemble structure?
440 if(loc_prol_p.empty())
441 {
442 Assembly::SymbolicAssembler::assemble_matrix_2lvl(loc_prol_p, space_f, space_c);
443 }
444
445 LocalPresVector loc_vec_weight = loc_prol_p.create_vector_l();
446
447 // assemble prolongation matrix
448 {
449 loc_prol_p.format();
450 loc_vec_weight.format();
451
452 // assemble prolongation matrix
453 Assembly::GridTransfer::assemble_prolongation(loc_prol_p, loc_vec_weight, space_f, space_c, cubature);
454
455 // synchronize weight vector
456 this->gate_pres.sync_0(loc_vec_weight);
457
458 // invert components
459 loc_vec_weight.component_invert(loc_vec_weight);
460
461 // scale prolongation matrix
462 loc_prol_p.scale_rows(loc_prol_p, loc_vec_weight);
463
464 // copy and transpose
465 loc_rest_p = loc_prol_p.transpose();
466 }
467
468 // compile pressure transfer
469 this->transfer_pres.compile();
470 }

483 template<typename DomainLevel_, typename Cubature_>
484 void assemble_velocity_truncation(
485 const Domain::VirtualLevel<DomainLevel_>& virt_lvl_fine,
486 const Domain::VirtualLevel<DomainLevel_>& virt_lvl_coarse,
487 const Cubature_& cubature,
488 const StokesPowerSystemLevel* sys_lvl_coarse = nullptr)
489 {
490 // if the coarse level is a parent, then we need the coarse system level
491 XASSERT((sys_lvl_coarse != nullptr) || !virt_lvl_coarse.is_parent());
492
493 // get fine and coarse domain levels
494 const DomainLevel_& level_f = *virt_lvl_fine;
495 const DomainLevel_& level_c = virt_lvl_coarse.is_child() ? virt_lvl_coarse.level_c() : *virt_lvl_coarse;
496
497 const auto& space_f = level_f.space_velo;
498 const auto& space_c = level_c.space_velo;
499
500 // get local transfer operator
501 LocalVeloTransfer& loc_trans = this->transfer_velo.local();
502
503 // get local transfer matrices
504 const LocalVeloTransferMatrix& loc_rest_v = loc_trans.get_mat_rest();
505 LocalVeloTransferMatrix& loc_trunc_v = loc_trans.get_mat_trunc();
506
507 // get the matrix blocks
508 const LocalScalarTransferMatrix& loc_rest_vx = loc_rest_v.get(0,0);
509 LocalScalarTransferMatrix& loc_trunc_vx = loc_trunc_v.get(0,0);
510
511 // restriction matrix must be already assembled
512 XASSERTM(loc_rest_vx.size() > Index(0), "you need to call 'assemble_velocity_transfer' first");
513
514 // clone sparsity pattern of restriction matrix
515 loc_trunc_vx = loc_rest_vx.clone(LAFEM::CloneMode::Layout);
516
517 // create local weight vector
518 LocalVeloVector loc_vec_weight = loc_rest_v.create_vector_l();
519
520 // get local weight vector component
521 LocalScalarVector& loc_vec_wx = loc_vec_weight.get(0);
522
523 // format
524 loc_trunc_vx.format();
525 loc_vec_wx.format();
526
527 // assemble truncation matrix
528 Assembly::GridTransfer::assemble_truncation(loc_trunc_vx, loc_vec_wx, space_f, space_c, cubature);
529
530 // expand weight vector
531 for(int i(1); i < loc_vec_weight.num_blocks; ++i)
532 loc_vec_weight.get(i).copy(loc_vec_wx);
533
534 // We now need to synchronize the weight vector in analogy to the prolongation matrix assembly.
535 // Note that the weight vector is now a coarse-level vector, so we need to synchronize over
536 // the coarse-level gate. This may be a bit more complicated if the coarse level is a ghost
537 // level, as in this case we have to join/split around the synch operation.
538
539 // synchronize weight vector using the muxer/gate
540 if(!virt_lvl_coarse.is_child())
541 {
542 // The coarse level is a simple (non-child) level that exists on all processes,
543 // so simply synch over the coarse-level gate:
544 sys_lvl_coarse->gate_velo.sync_0(loc_vec_weight);
545 }
546 else if(virt_lvl_coarse.is_parent())
547 {
548 // The coarse level is a child level and this is one of the parent processes which contain
549 // the coarse-level gate. So we first need to join the weight vector onto the parent processes
550 // first, then synch that joined vector over the parent gate and finally split the result
551 // to all child processes -- this "emulates" a synch over the (non-existent) coarse-level
552 // child gate, which is what we actually require here...
553
554 // create temporary vector on parent partitioning
555 LocalVeloVector loc_tmp = sys_lvl_coarse->gate_velo._freqs.clone(LAFEM::CloneMode::Allocate);
556
557 // join child weights over muxer
558 this->coarse_muxer_velo.join(loc_vec_weight, loc_tmp);
559
560 // sync over coarse gate
561 sys_lvl_coarse->gate_velo.sync_0(loc_tmp);
562
563 // split over muxer
564 this->coarse_muxer_velo.split(loc_vec_weight, loc_tmp);
565 }
566 else // ghost
567 {
568 // The coarse level is a ghost level, i.e. a child but not a parent. In this case, we
569 // only have to participate in the join/send operations of the muxer, which are part
570 // of the operations executed on the parents handled by the else-if case above.
571
572 this->coarse_muxer_velo.join_send(loc_vec_weight);
573
574 // parent performs sync over its gate here (see above else-if)
575
576 this->coarse_muxer_velo.split_recv(loc_vec_weight);
577 }
578
579 // invert components
580 loc_vec_weight.component_invert(loc_vec_weight);
581

584

590 template<typename DomainLevel_, typename Cubature_>
591 void assemble_pressure_truncation(
592 const Domain::VirtualLevel<DomainLevel_>& virt_lvl_fine,
593 const Domain::VirtualLevel<DomainLevel_>& virt_lvl_coarse,
594 const Cubature_& cubature,
595 const StokesPowerSystemLevel* sys_lvl_coarse = nullptr)
596 {
597 // if the coarse level is a parent, then we need the coarse system level
598 XASSERT((sys_lvl_coarse != nullptr) || !virt_lvl_coarse.is_parent());
599
600 // get fine and coarse domain levels
601 const DomainLevel_& level_f = *virt_lvl_fine;
602 const DomainLevel_& level_c = virt_lvl_coarse.is_child() ? virt_lvl_coarse.level_c() : *virt_lvl_coarse;
603
604 const auto& space_f = level_f.space_pres;
605 const auto& space_c = level_c.space_pres;
606
607 // get local transfer operator
608 LocalPresTransfer& loc_trans = this->transfer_pres.local();
609
610 // get local transfer matrices
611 const LocalPresTransferMatrix& loc_rest = loc_trans.get_mat_rest();
612 LocalPresTransferMatrix& loc_trunc = loc_trans.get_mat_trunc();
613
614 // restriction matrix must be already assembled
615 XASSERTM(loc_rest.size() > Index(0), "you need to call 'assemble_prescity_transfer' first");
616
617 // clone sparsity pattern of restriction matrix
618 loc_trunc = loc_rest.clone(LAFEM::CloneMode::Layout);
619
620 // create local weight vector
621 LocalPresVector loc_vec_weight = loc_trunc.create_vector_l();
622
623 // format
624 loc_trunc.format();
625 loc_vec_weight.format();
626
627 // assemble truncation matrix
628 Assembly::GridTransfer::assemble_truncation(loc_trunc, loc_vec_weight, space_f, space_c, cubature);
629
630 // We now need to synchronize the weight vector in analogy to the prolongation matrix assembly.
631 // Note that the weight vector is now a coarse-level vector, so we need to synchronize over
632 // the coarse-level gate. This may be a bit more complicated if the coarse level is a ghost
633 // level, as in this case we have to join/split around the synch operation.
634
635 // synchronize weight vector using the muxer/gate
636 if(!virt_lvl_coarse.is_child())
637 {
638 // The coarse level is a simple (non-child) level that exists on all processes,
639 // so simply synch over the coarse-level gate:
640 sys_lvl_coarse->gate_pres.sync_0(loc_vec_weight);
641 }
642 else if(virt_lvl_coarse.is_parent())
643 {
644 // The coarse level is a child level and this is one of the parent processes which contain
645 // the coarse-level gate. So we first need to join the weight vector onto the parent processes
646 // first, then synch that joined vector over the parent gate and finally split the result
647 // to all child processes -- this "emulates" a synch over the (non-existent) coarse-level
648 // child gate, which is what we actually require here...
649
650 // create temporary vector on parent partitioning
651 LocalPresVector loc_tmp = sys_lvl_coarse->gate_pres._freqs.clone(LAFEM::CloneMode::Allocate);
652
653 // join child weights over muxer
654 this->coarse_muxer_pres.join(loc_vec_weight, loc_tmp);
655
656 // sync over coarse gate
657 sys_lvl_coarse->gate_pres.sync_0(loc_tmp);
658
659 // split over muxer
660 this->coarse_muxer_pres.split(loc_vec_weight, loc_tmp);
661 }
662 else // ghost
663 {
664 // The coarse level is a ghost level, i.e. a child but not a parent. In this case, we
665 // only have to participate in the join/send operations of the muxer, which are part
666 // of the operations executed on the parents handled by the else-if case above.
667
668 this->coarse_muxer_pres.join_send(loc_vec_weight);
669
670 // parent performs sync over its gate here (see above else-if)
671
672 this->coarse_muxer_pres.split_recv(loc_vec_weight);
673 }
674
675 // invert components
676 loc_vec_weight.component_invert(loc_vec_weight);
677
678 // scale reduction matrix
679 loc_trunc.scale_rows(loc_trunc, loc_vec_weight);
680 }

729 const SpaceVelo_& space_velo, const SpacePres_& space_pres, const Cubature_& cubature)
