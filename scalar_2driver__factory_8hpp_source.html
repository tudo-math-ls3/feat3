212 template<typename Weight_, typename Coord_>
213 static bool create(Rule<Weight_, Coord_>& rule, const String& name_in)
214 {
215 // map alias names
216 Intern::DriverFactoryAliasMapper<DriverType> mapper(name_in);
217 DriverType::alias(mapper);
218 String mapped_name(mapper.name());
219
220 // try to find a colon within the string
221 String::size_type k = mapped_name.find_first_of(':');
222 if(k == mapped_name.npos)
223 return false;
224
225 // extract substrings until the colon
226 String head(mapped_name.substr(0, k));
227 String tail(mapped_name.substr(k + 1));
228
229 // check head - this is the name of the formula
230 if(head.trim().compare_no_case(DriverType::name()) != 0)
231 return false;
232
233 // check substring
234 int num_points = 0;
235 if(!tail.trim().parse(num_points))
236 return false;
237
238 // try to create the rule
239 return create(rule, num_points);
240 }
