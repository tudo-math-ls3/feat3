31 template<typename VertexSet_>
32 struct StandardVertexRefiner<Shape::Vertex, VertexSet_>
33 {
34 typedef VertexSet_ VertexSetType;
35
36 static Index refine(
37 Index offset,
38 VertexSetType& vertex_set_out,
39 const VertexSetType& vertex_set_in)
40 {
41 // get the number of coarse mesh vertices
42 const Index num_verts = vertex_set_in.get_num_vertices();
43 XASSERT(vertex_set_out.get_num_vertices() >= offset+num_verts);
44
45 // loop over all vertices
46 for(Index i(0); i < num_verts; ++i)
47 {
48 // copy source vertex
49 vertex_set_out[offset + i] = vertex_set_in[i];
50 }
51
52 // return number of created vertices
53 return num_verts;
54 }
55 }; // class StandardVertexRefiner<Vertex,...>

62 template<
63 int cell_dim_,
64 typename VertexSet_>
65 struct StandardVertexRefiner<Shape::Hypercube<cell_dim_>, VertexSet_>
66 {
67 typedef Shape::Hypercube<cell_dim_> ShapeType;
68 typedef IndexSet<Shape::FaceTraits<ShapeType, 0>::count> IndexSetType;
69 typedef VertexSet_ VertexSetType;
70
71 static Index refine(
72 Index offset,
73 VertexSetType& vertex_set_out,
74 const VertexSetType& vertex_set_in,
75 const IndexSetType& index_set_in)
76 {
77 typedef typename IndexSetType::IndexTupleType IndexTupleType;
78 typedef typename VertexSetType::VertexType VertexType;
79 typedef typename VertexSetType::CoordType CoordType;
80
81 // scaling factor
82 static const CoordType scale = CoordType(1) / CoordType(IndexSetType::num_indices);
83
84 // get number of cells
85 Index num_cells = index_set_in.get_num_entities();
86 XASSERT(vertex_set_out.get_num_vertices() >= offset+num_cells);
87
88 // loop over all cells
89 for(Index i(0); i < num_cells; ++i)
90 {
91 // get input index vector
92 const IndexTupleType& idx_in = index_set_in[i];
93
94 // get output vertex
95 VertexType& vtx_out = vertex_set_out[offset + i];
96
97 // clear output vertex
98 vtx_out.format();
99
100 // add all other vertices onto it
101 for(int k(0); k < IndexSetType::num_indices; ++k)
102 {
103 vtx_out.axpy(scale, vertex_set_in[idx_in[k]]);
104 }
105 }
106
107 // return number of created vertices
108 return num_cells;
109 }

196 template<
197 typename Shape_,
198 typename VertexSet_>
199 struct StandardVertexRefineWrapper
200 {
201 typedef VertexSet_ VertexSetType;
202 typedef IndexSetHolder<Shape_> IndexSetHolderType;
203 typedef IndexSet<Shape::FaceTraits<Shape_, 0>::count> IndexSetType;
204
205 static Index refine(
206 VertexSetType& vertex_set_out,
207 const VertexSetType& vertex_set_in,
208 const IndexSetHolderType& index_set_holder_in)
209 {
210 typedef typename Shape::FaceTraits<Shape_, Shape_::dimension-1>::ShapeType FacetType;
211
212 // recursive call of VertexRefineWrapper
213 Index offset = StandardVertexRefineWrapper<FacetType, VertexSet_>
214 ::refine(vertex_set_out, vertex_set_in, index_set_holder_in);
215
216 // get index set of current shape
217 const IndexSetType& index_set_in = index_set_holder_in.
218 template get_index_set_wrapper<Shape_::dimension>().template get_index_set<0>();
219
220 // call VertexRefiner
221 Index num_verts = StandardVertexRefiner<Shape_, VertexSet_>
222 ::refine(offset, vertex_set_out, vertex_set_in, index_set_in);
223
224 // validate number of created vertices
225 XASSERTM(num_verts == (StandardRefinementTraits<Shape_, 0>::count * index_set_in.get_num_entities()),
226 "VertexRefiner output does not match StdRefTraits prediction");
227
228 // return new offset
229 return offset + num_verts;
230 }
231 }; // struct StandardVertexRefineWrapper<...>
