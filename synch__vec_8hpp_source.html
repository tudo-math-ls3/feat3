85 SynchVectorTicket(VT_ & target, const Dist::Comm& comm, const std::vector<int>& ranks, const std::vector<VMT_> & mirrors) :
86 _finished(false),
87 _target(&target),
88 _comm(&comm),
89 _mirrors(&mirrors)
90 {
91 TimeStamp ts_start;
92 const std::size_t n = ranks.size();
93
94 XASSERTM(_mirrors->size() == n, "invalid vector mirror count");
95
96 // post receives
97 _recv_reqs.reserve(n);
98 _recv_bufs.resize(n);
99 for(std::size_t i(0); i < n; ++i)
100 {
101 // create buffer vector in main memory
102 _recv_bufs.at(i) = BufferType(_mirrors->at(i).buffer_size(*_target));
103
104 // post receive
105 _recv_reqs.push_back(_comm->irecv(_recv_bufs.at(i).elements(), _recv_bufs.at(i).size(), ranks.at(i)));
106 }
107
108 // post sends
109 _send_reqs.reserve(n);
110 _send_bufs.resize(n);
111 for(std::size_t i(0); i < n; ++i)
112 {
113 // create buffer in device memory
114 _send_bufs.at(i) = BufferType(_mirrors->at(i).buffer_size(*_target));
115
116 // gather from mirror
117 _mirrors->at(i).gather(_send_bufs.at(i), *_target);
118
119 // post send
120 _send_reqs.push_back(_comm->isend(_send_bufs.at(i).elements(), _send_bufs.at(i).size(), ranks.at(i)));
121 }
122
123 Statistics::add_time_mpi_execute_blas2(ts_start.elapsed_now());
124 }

195 void wait()
196 {
197 XASSERTM(!_finished, "ticket was already completed by a wait call");
198
199#ifdef FEAT_HAVE_MPI
200 TimeStamp ts_start;
201
202 // process all pending receives
203 for(std::size_t idx(0u); _recv_reqs.wait_any(idx); )
204 {
205 // scatter the receive buffer
206 _mirrors->at(idx).scatter_axpy(*_target, _recv_bufs.at(idx));
207 }
208
209 // wait for all sends to finish
210 _send_reqs.wait_all();
211
212 Statistics::add_time_mpi_wait_blas2(ts_start.elapsed_now());
213#endif // FEAT_HAVE_MPI
214
215 _finished = true;
216 }
