208 virtual Status apply(VectorType& vec_cor, const VectorType& vec_def) override
209 {
210 // Clear solution vector
211 vec_cor.format();
212
213 this->_functional.prepare(vec_cor, this->_filter);
214 this->_functional.eval_fval_grad(this->_fval, this->_vec_r);
215
216 // Copy back given defect
217 this->_vec_r.copy(vec_def);
218
219 if(this->_precond != nullptr)
220 {
221 this->_precond->prepare(vec_cor, this->_filter);
222 }
223
224 // apply
225 this->_status = _apply_intern(vec_cor);
226 this->plot_summary();
227 return this->_status;
228 }

231 virtual Status correct(VectorType& vec_sol, const VectorType& DOXY(vec_rhs)) override
232 {
233 this->_functional.prepare(vec_sol, this->_filter);
234 // Compute functional value and gradient
235 this->_functional.eval_fval_grad(this->_fval, this->_vec_r);
236 this->_vec_r.scale(this->_vec_r,DataType(-1));
237 this->_filter.filter_def(this->_vec_r);
238
239 if(this->_precond != nullptr)
240 {
241 this->_precond->prepare(vec_sol, this->_filter);
242 }
243
244 // apply
245 this->_status = _apply_intern(vec_sol);
246 this->plot_summary();
247 return this->_status;
248 }

250 protected:
264 virtual Status _apply_intern(VectorType& vec_sol)
265 {
266 Statistics::add_solver_expression(std::make_shared<ExpressionStartSolve>(this->name()));
267
268 // Reset member variables in the LineSearch
269 _linesearch->reset();
270
271 if(iterates != nullptr)
272 {
273 iterates->push_back(std::move(vec_sol.clone()));
274 }
275
276 // compute initial defect
277 Status status = this->_set_initial_defect(this->_vec_r, vec_sol);
278 if(status != Status::progress)
279 {
280 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), status, this->get_num_iter()));
281 return status;
282 }
283
284 // The first direction has to be the steepest descent direction
285 this->_vec_p.copy(this->_vec_r);
286
287 // apply preconditioner to defect vector
288 if(!this->_apply_precond(this->_vec_p, this->_vec_r, this->_filter))
289 {
290 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
291 return Status::aborted;
292 }
293
294 // Compute initial eta = <d, r>
295 DataType eta(this->_vec_p.dot(this->_vec_r));
296
297 // If the preconditioned search direction is not a descent direction, reset it to steepest descent
298 // TODO: Correct the output of the preconditioner if it turns out to not have been positive definite
299 if(eta <= DataType(0))
300 {
301 this->_vec_p.copy(this->_vec_r);
302 }
303
304 // start iterating
305 while(status == Status::progress)
306 {
307 IterationStats stat(*this);
308
309 this->_fval_prev = this->_fval;
310
311 // Copy information to the linesearch
312 _linesearch->set_initial_fval(this->_fval);
313 _linesearch->set_grad_from_defect(this->_vec_r);
314 // If we are using a preconditioner, use the additional information about the preconditioned step length in the line search
315 if(this->_precond != nullptr)
316 {
317 _linesearch->set_dir_scaling(true);
318 }
319
320 // Call the linesearch to update vec_sol
321 status = _linesearch->correct(vec_sol, this->_vec_p);
322
323 // Copy back information from the linesearch
324 this->_fval = _linesearch->get_final_fval();
325 this->_ls_its = _linesearch->get_num_iter();
326 this->_steplength = _linesearch->get_rel_update();
327 _linesearch->get_defect_from_grad(this->_vec_r);
328
329 // Log iterates if necessary
330 if(iterates != nullptr)
331 {
332 iterates->push_back(vec_sol.clone());
333 }
334
335 // Compute defect norm. This also performs the convergence/divergence checks.
336 status = this->_set_new_defect(this->_vec_r, vec_sol);
337
338 if(status != Status::progress)
339 {
340 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), status, this->get_num_iter()));
341 return status;
342 }
343
344 // Re-assemble preconditioner if necessary
345 if(this->_precond != nullptr)
346 {
347 this->_precond->prepare(vec_sol, this->_filter);
348 }
349
350 // apply preconditioner
351 if(!this->_apply_precond(_vec_p, _vec_r, this->_filter))
352 {
353 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::aborted, this->get_num_iter()));
354 return Status::aborted;
355 }
356
357 // Compute new eta
358 eta = this->_vec_p.dot(this->_vec_r);
359
360 // If the preconditioned search direction is not a descent direction, reset it to steepest descent
361 // TODO: Correct the output of the preconditioner if it turns out to not have been positive definite
362 if(eta <= DataType(0))
363 {
364 this->_vec_p.copy(this->_vec_r);
365 }
366 }
367
368 // We should never come to this point
369 Statistics::add_solver_expression(std::make_shared<ExpressionEndSolve>(this->name(), Status::undefined, this->get_num_iter()));
370 return Status::undefined;
371 }
