338 void _alloc_max_degrees()
339 {
340 _max_degree_dofs.resize(this->_macro_dofs.size());
341 _max_degree_macros.resize(this->_macro_dofs.size());
342 for(std::size_t i = 0; i < _max_degree_dofs.size(); ++i)
343 {
344 if constexpr(macro_type_ == FEAT::Intern::VankaMacroPolicy::uniformMacros)
345 _max_degree_dofs[i] = this->_macro_dofs[i].degree(Index(0));
346 else
347 _max_degree_dofs[i] = this->_macro_dofs[i].degree();
348 _max_degree_macros[i] = this->_dof_macros[i].degree();
349 }
350 }
351
353 void _alloc_device()
354 {
355 XASSERTM(_max_degree_dofs.size() == this->_macro_dofs.size(), "call _alloc_max_degrees beforehand");
356 if(!_allocate_device)
357 return;
358 #ifdef FEAT_HAVE_CUDA
359 _d_macro_dofs.resize(this->_macro_dofs.size());
360 _d_dof_macros.resize(this->_dof_macros.size());
361 for(int i = 0; i < int(this->_macro_dofs.size()); ++i)
362 {
363 Index malloc_size;
364 if constexpr(macro_type_ == FEAT::Intern::VankaMacroPolicy::uniformMacros)
365 malloc_size = this->_macro_dofs[i].get_num_nodes_domain() * _max_degree_dofs[i] * sizeof(Index);
366 else
367 malloc_size = this->_macro_dofs[i].get_num_nodes_domain() * (_max_degree_dofs[i]+1) * sizeof(Index);
368 _d_macro_dofs[i] = (Index*)Util::cuda_malloc(malloc_size);
369 // prepare tmp array
370 {
371 Index* tmp_pinned = (Index*)Util::cuda_malloc_host(malloc_size);
372 const Index* dom_ptr = this->_macro_dofs[i].get_domain_ptr();
373 const Index* img_ptr = this->_macro_dofs[i].get_image_idx();
374 for(int k = 0; k < int(this->_macro_dofs[i].get_num_nodes_domain()); ++k)
375 {
376 if constexpr(macro_type_ == FEAT::Intern::VankaMacroPolicy::uniformMacros)
377 {
378 std::memcpy(tmp_pinned + k*_max_degree_dofs[i], img_ptr + dom_ptr[k], _max_degree_dofs[i]*sizeof(Index));
379 }
380 else
381 {
382 const Index loc_size = dom_ptr[k+1] - dom_ptr[k];
383 tmp_pinned[k*(_max_degree_dofs[i]+1)] = loc_size;
384 std::memcpy(tmp_pinned + k*(_max_degree_dofs[i]+1) + 1, img_ptr + dom_ptr[k], loc_size*sizeof(Index));
385 std::memset(tmp_pinned + k*(_max_degree_dofs[i]+1) + loc_size + 1, ~int(0), (_max_degree_dofs[i] - loc_size)*sizeof(Index));
386 }
387 }
388 Util::cuda_copy_host_to_device((void*)(_d_macro_dofs[i]), (void*)tmp_pinned, malloc_size);
389 Util::cuda_free_host((void*)tmp_pinned);
390 }
391 malloc_size = this->_dof_macros[i].get_num_nodes_domain()*(_max_degree_macros[i]+1)*sizeof(Index);
392 _d_dof_macros[i] = (Index*)Util::cuda_malloc(malloc_size);
393 {
394 Index* tmp_pinned = (Index*)Util::cuda_malloc_host(malloc_size);
395 const Index* dom_ptr = this->_dof_macros[i].get_domain_ptr();
396 const Index* img_ptr = this->_dof_macros[i].get_image_idx();
397 for(int k = 0; k < int(this->_dof_macros[i].get_num_nodes_domain()); ++k)
398 {
399 const Index loc_size = dom_ptr[k+1] - dom_ptr[k];
400 tmp_pinned[k*(_max_degree_macros[i]+1)] = loc_size;
401 std::memcpy(tmp_pinned + k*(_max_degree_macros[i]+1) + 1, img_ptr + dom_ptr[k], loc_size*sizeof(Index));
402 std::memset(tmp_pinned + k*(_max_degree_macros[i]+1) + loc_size + 1, ~int(0), (_max_degree_macros[i] - loc_size)*sizeof(Index));
403 }
404 Util::cuda_copy_host_to_device((void*)(_d_dof_macros[i]), (void*)tmp_pinned, malloc_size);
405 Util::cuda_free_host((void*)tmp_pinned);
406 }
407 }
408 {
409 if(this->_skip_singular)
410 {
411 _d_macro_mask = (int*)Util::cuda_malloc(this->_macro_mask.size()*sizeof(int));
412 Util::cuda_set_memory(_d_macro_mask, 0, this->_macro_mask.size());
413 }
414 }
415 // initialize coloroing
416 _coloring_data.init_device();
417 #endif
418 }
