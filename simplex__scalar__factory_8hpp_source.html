25 template<typename Weight_, typename Coord_, typename Point_>
26 static void create(
27 Rule<Shape::Simplex<1>, Weight_, Coord_, Point_>& rule,
28 const Scalar::Rule<Weight_, Coord_>& scalar_rule)
29 {
30 int num_points = scalar_rule.get_num_points();
31#ifdef FEAT_CUBATURE_SCALAR_PREFIX
32 rule = Rule<Shape::Simplex<1>, Weight_, Coord_, Point_>(num_points, "scalar:" + scalar_rule.get_name());
33#else
34 rule = Rule<Shape::Simplex<1>, Weight_, Coord_, Point_>(num_points, scalar_rule.get_name());
35#endif // FEAT_CUBATURE_SCALAR_PREFIX
36 for(int i(0); i < num_points; ++i)
37 {
38 rule.get_weight(i) = scalar_rule.get_weight(i) * Weight_(0.5);
39 rule.get_coord(i, 0) = (scalar_rule.get_coord(i) + Coord_(1)) * Coord_(0.5);
40 }
41 }

43 template<typename Weight_, typename Coord_, typename Point_>
44 static bool create(Rule<Shape::Simplex<1>, Weight_, Coord_, Point_>& rule, const String& name)
45 {
46 typedef Scalar::Rule<Weight_, Coord_> ScalarRuleType;
47
48#ifdef FEAT_CUBATURE_SCALAR_PREFIX
49 // try to find a colon within the string
50 String::size_type k = name.find_first_of(':');
51 if(k == name.npos)
52 return false;
53
54 // extract substrings until the colon
55 String head(name.substr(0, k));
56 String tail(name.substr(k + 1));
57
58 // check head - this is the name of the formula
59 if(head.trim().compare_no_case("scalar") != 0)
60 return false;
61
62 // call scalar factory to create the scalar rule
63 ScalarRuleType scalar_rule;
64 if(!ScalarFactoryType::create(scalar_rule, tail.trim()))
65 return false;
66#else
67 // call scalar factory to create the scalar rule
68 ScalarRuleType scalar_rule;
69 if(!ScalarFactoryType::create(scalar_rule, name))
70 return false;
71#endif // FEAT_CUBATURE_SCALAR_PREFIX
72
73 // convert scalar rule
74 create(rule, scalar_rule);
75 return true;
76 }
