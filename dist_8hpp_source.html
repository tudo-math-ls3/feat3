860 std::size_t compress()
861 {
862 std::size_t i(0), j(0), n = _reqs.size();
863 while(i+j < n)
864 {
865 if(_reqs[i+j].is_null())
866 ++j;
867 else
868 {
869 _reqs[i] = std::move(_reqs[i+j]);
870 _stats[i] = std::move(_stats[i+j]);
871 ++i;
872 }
873 }
874 _reqs.resize(n-j);
875 _stats.resize(n-j);
876 return n-j;
877 }
